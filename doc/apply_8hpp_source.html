148  bool apw_only__,
149  bool add_o1__,
150  int N__,
151  int n__,
152  wave_functions& phi__,
153  wave_functions& ophi__) const
154 {
155  PROFILE("sirius::Band::apply_fv_o");
156 
157  if (!apw_only__) {
158  /* zero the local-orbital part */
159  ophi__.mt_coeffs().zero<memory_t::host>(N__, n__);
160  }
161 
162  /* interstitial part */
163  local_op_->apply_o(kp__->gkvec().partition(), N__, n__, phi__, ophi__);
164 
165  matrix<double_complex> alm(kp__->num_gkvec_loc(), unit_cell_.max_mt_aw_basis_size());
166  matrix<double_complex> oalm(kp__->num_gkvec_loc(), unit_cell_.max_mt_aw_basis_size());
167  matrix<double_complex> tmp(unit_cell_.max_mt_aw_basis_size(), n__);
168 
169  #ifdef __GPU
170  if (ctx_.processing_unit() == GPU) {
171  if (!apw_only__) {
172  phi__.mt_coeffs().copy_to_host(N__, n__);
173  }
174  alm.allocate(memory_t::device);
175  tmp.allocate(memory_t::device);
176  }
177  #endif
178 
179  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
180  auto& atom = unit_cell_.atom(ia);
181  auto& type = atom.type();
182  /* number of AW for this atom */
183  int naw = atom.mt_aw_basis_size();
184  /* number of lo for this atom */
185  int nlo = atom.mt_lo_basis_size();
186  kp__->alm_coeffs_loc().generate(ia, alm);
187 
188  #ifdef __GPU
189  if (ctx_.processing_unit() == GPU) {
190  alm.copy_to_device();
191  }
192  #endif
193 
194  sddk::timer t1("sirius::Band::apply_fv_o|apw-apw");
195 
196  if (ctx_.processing_unit() == CPU) {
197  /* tmp(lm, i) = A(G, lm)^{T} * C(G, i) */
198  linalg<CPU>::gemm(1, 0, naw, n__, kp__->num_gkvec_loc(),
199  alm.at<CPU>(), alm.ld(),
200  phi__.pw_coeffs().prime().at<CPU>(0, N__), phi__.pw_coeffs().prime().ld(),
201  tmp.at<CPU>(), tmp.ld());
202  }
203  #ifdef __GPU
204  if (ctx_.processing_unit() == GPU) {
205  /* tmp(lm, i) = A(G, lm)^{T} * C(G, i) */
206  linalg<GPU>::gemm(1, 0, naw, n__, kp__->num_gkvec_loc(),
207  alm.at<GPU>(), alm.ld(),
208  phi__.pw_coeffs().prime().at<GPU>(0, N__), phi__.pw_coeffs().prime().ld(),
209  tmp.at<GPU>(), tmp.ld());
210  tmp.copy_to_host(naw * n__);
211  }
212  #endif
213 
214  kp__->comm().allreduce(tmp.at<CPU>(), static_cast<int>(tmp.size()));
215 
216  #ifdef __GPU
217  if (ctx_.processing_unit() == GPU) {
218  tmp.copy_to_device(naw * n__);
219  }
220  #endif
221 
222  for (int xi = 0; xi < naw; xi++) {
223  for (int ig = 0; ig < kp__->num_gkvec_loc(); ig++) {
224  alm(ig, xi) = std::conj(alm(ig, xi));
225  }
226  }
227  #ifdef __GPU
228  if (ctx_.processing_unit() == GPU) {
229  alm.copy_to_device();
230  }
231  #endif
232 
233  if (ctx_.processing_unit() == CPU) {
234  /* APW-APW contribution to overlap */
235  linalg<CPU>::gemm(0, 0, kp__->num_gkvec_loc(), n__, naw,
236  linalg_const<double_complex>::one(),
237  alm.at<CPU>(), alm.ld(),
238  tmp.at<CPU>(), tmp.ld(),
239  linalg_const<double_complex>::one(),
240  ophi__.pw_coeffs().prime().at<CPU>(0, N__),
241  ophi__.pw_coeffs().prime().ld());
242  }
243  #ifdef __GPU
244  if (ctx_.processing_unit() == GPU) {
245  /* APW-APW contribution to overlap */
246  linalg<GPU>::gemm(0, 0, kp__->num_gkvec_loc(), n__, naw,
247  &linalg_const<double_complex>::one(),
248  alm.at<GPU>(), alm.ld(),
249  tmp.at<GPU>(), tmp.ld(),
250  &linalg_const<double_complex>::one(),
251  ophi__.pw_coeffs().prime().at<GPU>(0, N__),
252  ophi__.pw_coeffs().prime().ld());
253  }
254  #endif
255  t1.stop();
256 
257  if (!nlo || apw_only__) {
258  continue;
259  }
260 
261  /* local orbital coefficients of atom ia for all states */
262  matrix<double_complex> phi_lo_ia(nlo, n__);
263  auto ia_location = phi__.spl_num_atoms().location(ia);
264  if (ia_location.rank == kp__->comm().rank()) {
265  for (int i = 0; i < n__; i++) {
266  std::memcpy(&phi_lo_ia(0, i),
267  phi__.mt_coeffs().prime().at<CPU>(phi__.offset_mt_coeffs(ia_location.local_index), N__ + i),
268  nlo * sizeof(double_complex));
269  }
270  }
271  kp__->comm().bcast(phi_lo_ia.at<CPU>(), static_cast<int>(phi_lo_ia.size()), ia_location.rank);
272 
273  sddk::timer t2("sirius::Band::apply_fv_o|apw-lo");
274  oalm.zero();
275  #pragma omp parallel for
276  for (int ilo = 0; ilo < nlo; ilo++) {
277  int xi_lo = type.mt_aw_basis_size() + ilo;
278  /* local orbital indices */
279  int l_lo = type.indexb(xi_lo).l;
280  int lm_lo = type.indexb(xi_lo).lm;
281  int idxrf_lo = type.indexb(xi_lo).idxrf;
282  int order_lo = type.indexb(xi_lo).order;
283  /* use oalm as temporary buffer to compute alm*o */
284  for (int order_aw = 0; order_aw < (int)type.aw_descriptor(l_lo).size(); order_aw++) {
285  for (int igloc = 0; igloc < kp__->num_gkvec_loc(); igloc++) {
286  oalm(igloc, ilo) += alm(igloc, type.indexb_by_lm_order(lm_lo, order_aw)) *
287  atom.symmetry_class().o_radial_integral(l_lo, order_aw, order_lo);
288  }
289  }
290  }
291  linalg<CPU>::gemm(0, 0, kp__->num_gkvec_loc(), n__, nlo,
292  linalg_const<double_complex>::one(),
293  oalm.at<CPU>(), oalm.ld(),
294  phi_lo_ia.at<CPU>(), phi_lo_ia.ld(),
295  linalg_const<double_complex>::one(),
296  ophi__.pw_coeffs().prime().at<CPU>(0, N__),
297  ophi__.pw_coeffs().prime().ld());
298  t2.stop();
299 
300  sddk::timer t3("sirius::Band::apply_fv_o|lo-apw");
301  /* lo-APW contribution */
302  for (int i = 0; i < n__; i++) {
303  for (int ilo = 0; ilo < type.mt_lo_basis_size(); ilo++) {
304  int xi_lo = type.mt_aw_basis_size() + ilo;
305  /* local orbital indices */
306  int l_lo = type.indexb(xi_lo).l;
307  int lm_lo = type.indexb(xi_lo).lm;
308  int order_lo = type.indexb(xi_lo).order;
309  int idxrf_lo = type.indexb(xi_lo).idxrf;
310 
311  if (ia_location.rank == kp__->comm().rank()) {
312  /* lo-lo contribution */
313  for (int jlo = 0; jlo < type.mt_lo_basis_size(); jlo++) {
314  int xi_lo1 = type.mt_aw_basis_size() + jlo;
315  int lm1 = type.indexb(xi_lo1).lm;
316  int order1 = type.indexb(xi_lo1).order;
317  int idxrf1 = type.indexb(xi_lo1).idxrf;
318  if (lm_lo == lm1) {
319  ophi__.mt_coeffs().prime(ophi__.offset_mt_coeffs(ia_location.local_index) + ilo, N__ + i) +=
320  phi_lo_ia(jlo, i) * atom.symmetry_class().o_radial_integral(l_lo, order_lo, order1);
321  }
322  }
323 
324  for (int order_aw = 0; order_aw < (int)type.aw_descriptor(l_lo).size(); order_aw++) {
325  /* lo-APW contribution */
326  ophi__.mt_coeffs().prime(ophi__.offset_mt_coeffs(ia_location.local_index) + ilo, N__ + i) +=
327  atom.symmetry_class().o_radial_integral(l_lo, order_lo, order_aw) *
328  tmp(type.indexb_by_lm_order(lm_lo, order_aw), i);
329  }
330  }
331  }
332  }
333  t3.stop();
334  }
335 
336  #ifdef __GPU
337  if (ctx_.processing_unit() == GPU && !apw_only__) {
338  ophi__.mt_coeffs().copy_to_device(N__, n__);
339  }
340  #endif
341 }

693  std::vector<wave_functions>& hpsi__) const

698 
699  //#ifdef __GPU
700  //if (ctx_.processing_unit() == GPU) {
701  // fv_states__.allocate_on_device();

709  mdarray<double_complex, 3> zm(unit_cell_.max_mt_basis_size(), unit_cell_.max_mt_basis_size(), ctx_.num_mag_dims());
710 
711  for (int ialoc = 0; ialoc < fv_states__.spl_num_atoms().local_size(); ialoc++) {
712  int ia = fv_states__.spl_num_atoms()[ialoc];
713  auto& atom = unit_cell_.atom(ia);
714  int offset = fv_states__.offset_mt_coeffs(ialoc);
715  int mt_basis_size = atom.type().mt_basis_size();
716 
717  zm.zero();
718 
719  /* only upper triangular part of zm is computed because it is a hermitian matrix */
720  #pragma omp parallel for default(shared)
721  for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
722  int lm2 = atom.type().indexb(xi2).lm;
723  int idxrf2 = atom.type().indexb(xi2).idxrf;
724 
725  for (int i = 0; i < ctx_.num_mag_dims(); i++) {
726  for (int xi1 = 0; xi1 <= xi2; xi1++) {

728  int idxrf1 = atom.type().indexb(xi1).idxrf;
729

731  }
732  }
733  }
734  /* compute bwf = B_z*|wf_j> */
735  linalg<CPU>::hemm(0, 0, mt_basis_size, ctx_.num_fv_states(),
736  linalg_const<double_complex>::one(),
737  zm.at<CPU>(), zm.ld(),
738  fv_states__.mt_coeffs().prime().at<CPU>(offset, 0), fv_states__.mt_coeffs().prime().ld(),
739  linalg_const<double_complex>::zero(),
740  hpsi__[0].mt_coeffs().prime().at<CPU>(offset, 0), hpsi__[0].mt_coeffs().prime().ld());

743  if (hpsi__.size() == 3) {
744  /* reuse first (z) component of zm matrix to store (B_x - iB_y) */
745  for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
746  for (int xi1 = 0; xi1 <= xi2; xi1++) {
747  zm(xi1, xi2, 0) = zm(xi1, xi2, 1) - double_complex(0, 1) * zm(xi1, xi2, 2);
748  }
749

751  * only the upper triangular part */
752  for (int xi1 = xi2 + 1; xi1 < mt_basis_size; xi1++) {
753  zm(xi1, xi2, 0) = std::conj(zm(xi2, xi1, 1)) - double_complex(0, 1) * std::conj(zm(xi2, xi1, 2));
754  }
755  }

758  zm.at<CPU>(), zm.ld(),
759  fv_states__.mt_coeffs().prime().at<CPU>(offset, 0), fv_states__.mt_coeffs().prime().ld(),
760  hpsi__[2].mt_coeffs().prime().at<CPU>(offset, 0), hpsi__[2].mt_coeffs().prime().ld());
761  }
762  }
763 
764  /* copy Bz|\psi> to -Bz|\psi> */
765  for (int i = 0; i < ctx_.num_fv_states(); i++) {
766  for (int j = 0; j < fv_states__.pw_coeffs().num_rows_loc(); j++) {
767  hpsi__[1].pw_coeffs().prime(j, i) = -hpsi__[0].pw_coeffs().prime(j, i);

771  }
