36 /// Representation of the periodical function on the muffin-tin geometry.
37 /** Inside each muffin-tin the spherical expansion is used:
38  * \f[
39  * f({\bf r}) = \sum_{\ell m} f_{\ell m}(r) Y_{\ell m}(\hat {\bf r})
40  * \f]
41  * or
42  * \f[
43  * f({\bf r}) = \sum_{\ell m} f_{\ell m}(r) R_{\ell m}(\hat {\bf r})
44  * \f]
45  * In the interstitial region function is stored on the real-space grid or as a Fourier series:
46  * \f[
47  * f({\bf r}) = \sum_{{\bf G}} f({\bf G}) e^{i{\bf G}{\bf r}}
48  * \f]
49  */

217  T integrate(std::vector<T>& mt_val, T& it_val)
218  {
219  PROFILE("sirius::Periodic_function::integrate");
220 
221  it_val = 0;
222 
223  if (!ctx_.full_potential()) {
224  #pragma omp parallel
225  {
226  T it_val_t = 0;
227 
228  #pragma omp for schedule(static)
229  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
230  it_val_t += this->f_rg_(irloc);
231  }
232 
233  #pragma omp critical
234  it_val += it_val_t;
235  }
236  } else {
237  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
238  it_val += this->f_rg_(irloc) * step_function_.theta_r(irloc);
239  }
240  }
241  it_val *= (unit_cell_.omega() / this->fft_->size());
242  this->fft_->comm().allreduce(&it_val, 1);
243  T total = it_val;
244 
245  if (ctx_.full_potential()) {
246  mt_val = std::vector<T>(unit_cell_.num_atoms(), 0);
247 
248  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
249  int ia = unit_cell_.spl_num_atoms(ialoc);
250  mt_val[ia] = f_mt_local_(ialoc).component(0).integrate(2) * fourpi * y00;
251  }
252 
253  comm_.allreduce(&mt_val[0], unit_cell_.num_atoms());
254  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
255  total += mt_val[ia];
256  }
257  }
258 
259  return total;
260  }

400  double value(vector3d<double>& vc)
401  {
402  int ja{-1}, jr{-1};
403  double dr{0}, tp[2];
404 
405  if (unit_cell_.is_point_in_mt(vc, ja, jr, dr, tp)) {
406  int lmax = Utils::lmax_by_lmmax(angular_domain_size_);
407  std::vector<double> rlm(angular_domain_size_);
408  SHT::spherical_harmonics(lmax, tp[0], tp[1], &rlm[0]);
409  double p{0};
410  for (int lm = 0; lm < angular_domain_size_; lm++) {
411  double d = (f_mt_(lm, jr + 1, ja) - f_mt_(lm, jr, ja)) / unit_cell_.atom(ja).type().radial_grid().dx(jr);
412 
413  p += rlm[lm] * (f_mt_(lm, jr, ja) + d * dr);
414  }
415  return p;
416  } else {
417  double p{0};
418  for (int ig = 0; ig < gvec_.num_gvec(); ig++) {
419  vector3d<double> vgc = gvec_.gvec_cart(ig);
420  p += std::real(f_pw_(ig) * std::exp(double_complex(0.0, vc * vgc)));
421  }
422  return p;
423  }
424  }

463  /// Compute inner product <f|g>
464  T inner(Periodic_function<T> const* g__) const
465  {
466  PROFILE("sirius::Periodic_function::inner");
467 
468  assert(this->fft_ == g__->fft_);
469  assert(&step_function_ == &g__->step_function_);
470  assert(&unit_cell_ == &g__->unit_cell_);
471  assert(&comm_ == &g__->comm_);
472 
473  T result_rg{0};
474 
475  if (!ctx_.full_potential()) {
476  #pragma omp parallel
477  {
478  T rt{0};
479 
480  #pragma omp for schedule(static)
481  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
482  rt += type_wrapper<T>::conjugate(this->f_rg(irloc)) * g__->f_rg(irloc);
483  }
484 
485  #pragma omp critical
486  result_rg += rt;
487  }
488  } else {
489  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
490  result_rg += type_wrapper<T>::conjugate(this->f_rg(irloc)) * g__->f_rg(irloc) *
491  this->step_function_.theta_r(irloc);
492  }
493  }
494 
495  result_rg *= (unit_cell_.omega() / this->fft_->size());
496  this->fft_->comm().allreduce(&result_rg, 1);
497 
498  T result_mt{0};
499  if (ctx_.full_potential()) {
500  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
501  auto r = sirius::inner(f_mt(ialoc), g__->f_mt(ialoc));
502  result_mt += r;
503  }
504  comm_.allreduce(&result_mt, 1);
505  }
506 
507  return result_mt + result_rg;
508  }
