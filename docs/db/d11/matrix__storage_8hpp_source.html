55/// Specialization of matrix storage class for slab data distribution.
56/** \tparam T data type */
57template <typename T>
58class matrix_storage<T, matrix_storage_t::slab>
59{
60 private:
61
62 /// G-vector partitioning.
63 Gvec_partition const* gvp_{nullptr};
64
65 /// Local number of rows.
66 int num_rows_loc_{0};
67
68 /// Total number of columns.
69 int num_cols_;
70
71 /// Primary storage of matrix.
72 mdarray<T, 2> prime_;
73
74 /// Auxiliary matrix storage.
75 /** This distribution is used by the FFT driver */
76 mdarray<T, 2> extra_;
77
78 /// Raw buffer for the extra storage.
79 mdarray<T, 1> extra_buf_;
80
81 /// Raw send-receive buffer.

83

85 splindex<splindex_t::block> spl_num_col_;
86
87 public:
88 /// Constructor.
89 matrix_storage(Gvec_partition const& gvp__, int num_cols__)
90 : gvp_(&gvp__)
91 , num_rows_loc_(gvp__.gvec().count())
92 , num_cols_(num_cols__)
93 {
94 /* primary storage of PW wave functions: slabs */
95 prime_ = mdarray<T, 2>(num_rows_loc_, num_cols_, memory_t::host, "matrix_storage.prime_");
96 }
97
98 /// Constructor.
99 matrix_storage(int num_rows_loc__, int num_cols__)
100 : num_rows_loc_(num_rows_loc__)
101 , num_cols_(num_cols__)
102 {
103 /* primary storage of PW wave functions: slabs */
104 prime_ = mdarray<T, 2>(num_rows_loc_, num_cols_, memory_t::host, "matrix_storage.prime_");
105 }
106
107 /// Constructor.
108 /** Memory for prime storage is allocated from the memory pool */
109 matrix_storage(memory_pool& mp__, Gvec_partition const& gvp__, int num_cols__)
110 : gvp_(&gvp__)
111 , num_rows_loc_(gvp__.gvec().count())
112 , num_cols_(num_cols__)
113 {
114 /* primary storage of PW wave functions: slabs */
115 prime_ = mdarray<T, 2>(num_rows_loc_, num_cols_, mp__, "matrix_storage.prime_");
116 }
117
118 /// Check if data needs to be remapped.
119 /** Data is not remapped when communicator that is orthogonal to FFT communicator is trivial.
120 * In this case the FFT communicator coincides with the entire communicator used to distribute wave functions
121 * and the data is ready for FFT transformations. */
122 inline bool is_remapped() const
123 {
124 return (gvp_->comm_ortho_fft().size() > 1);
125 }
126
127 /// Set the dimensions of the extra matrix storage.
128 /** \param [in] n Number of matrix columns to distribute.
129 * \param [in] idx0 Starting column of the matrix.
130 * \param [in] mp Memory pool that can be used for fast reallocation.
131 *
132 * \image html matrix_storage.png "Redistribution of wave-functions between MPI ranks"
133 *
134 * The extra storage is always created in the CPU memory. If the data distribution doesn't
135 * change (no swapping between comm_col ranks is performed) – then the extra storage will mirror
136 * the prime storage (both on CPU and GPU) irrespective of the target processing unit. If
137 * data remapping is necessary extra storage is allocated only in the host memory because MPI is done using the
138 * host pointers.
139 */
140 void set_num_extra(int n__, int idx0__, memory_pool* mp__);
141
142 /// Remap data from prime to extra storage.
143 /** \param [in] n Number of matrix columns to distribute.
144 * \param [in] idx0 Starting column of the matrix.
145 *
146 * Prime storage is expected on the CPU (for the MPI a2a communication). */
147 void remap_forward(int n__, int idx0__, memory_pool* mp__);
148
149 /// Remap data from extra to prime storage.
150 /** \param [in] n Number of matrix columns to collect.
151 * \param [in] idx0 Starting column of the matrix.
152 *
153 * Extra storage is expected on the CPU (for the MPI a2a communication). If the prime storage is allocated on GPU
154 * remapped data will be copied to GPU. */
155 void remap_backward(int n__, int idx0__);
156
