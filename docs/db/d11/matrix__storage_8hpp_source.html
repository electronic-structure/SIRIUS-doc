122  /// Set the dimensions of the extra matrix storage.
123  /** \param [in] pu Target processing unit.
124  * \param [in] num_rows Local number of rows in the extra matrix storage.
125  * \param [in] n Number of matrix columns to distribute.
126  * \param [in] idx0 Starting column of the matrix.
127  *
128  * \image html matrix_storage.png "Redistribution of wave-functions between MPI ranks"
129  *
130  * The extra storage is always created in the CPU memory. If the data distribution doesn't
131  * change (no swapping between comm_col ranks is performed) – then the extra storage will mirror
132  * the prime storage (both on CPU and GPU) irrespective of the target processing unit. If
133  * data remapping is necessary and target processing unit is GPU then the extra storage will be
134  * allocated on the GPU as well.
135  */
136  inline void set_num_extra(device_t pu__,
137  int n__,
138  int idx0__ = 0)
139  {
140  auto& comm_col = gvp_->comm_ortho_fft();
141 
142  /* this is how n columns of the matrix will be distributed between columns of the MPI grid */
143  spl_num_col_ = splindex<block>(n__, comm_col.size(), comm_col.rank());
144 
145  T* ptr{nullptr};
146  T* ptr_d{nullptr};
147  int ncol{0};
148 
149  /* trivial case */
150  if (!is_remapped()) {
151  assert(num_rows_loc_ == gvp_->gvec_count_fft());
152  ncol = n__;
153  ptr = prime_.template at<CPU>(0, idx0__);
154  if (prime_.on_device()) {
155  ptr_d = prime_.template at<GPU>(0, idx0__);
156  }
157  } else {
158  /* maximum local number of matrix columns */
159  ncol = splindex_base<int>::block_size(n__, comm_col.size());
160  /* upper limit for the size of swapped extra matrix */
161  size_t sz = gvp_->gvec_count_fft() * ncol;
162  /* reallocate buffers if necessary */
163  if (extra_buf_.size() < sz) {
164  send_recv_buf_ = mdarray<T, 1>(sz, memory_t::host, "matrix_storage.send_recv_buf_");
165  memory_t mem_type = memory_t::none;
166  switch (pu__) {
167  case CPU: {
168  mem_type = memory_t:: host;
169  break;
170  }
171  case GPU: {
172  mem_type = memory_t:: host | memory_t::device;
173  break;
174  }
175  }
176  extra_buf_ = mdarray<T, 1>(sz, mem_type);
177  }
178  ptr = extra_buf_.template at<CPU>();
179  if (extra_buf_.on_device()) {
180  ptr_d = extra_buf_.template at<GPU>();
181  }
182  }
183  extra_ = mdarray<T, 2>(ptr, ptr_d, gvp_->gvec_count_fft(), ncol, "matrix_storage.extra_");
184  }

186  /// Remap data from prime to extra storage.
187  /** \param [in] pu Target processing unit.
188  * \param [in] row_distr Distribution of rows of prime matrix in the extra matrix storage.
189  * \param [in] n Number of matrix columns to distribute.
190  * \param [in] idx0 Starting column of the matrix.
191  *
192  * Prime storage is expected on the CPU (for the MPI a2a communication). If the target processing unit is GPU
193  * extra storage will be copied to the device memory. */
194  inline void remap_forward(device_t pu__,
195  int n__,
196  int idx0__ = 0)
197  {
198  PROFILE("sddk::matrix_storage::remap_forward");
199 
200  set_num_extra(pu__, n__, idx0__);
201 
202  /* trivial case when extra storage mirrors the prime storage */
203  if (!is_remapped()) {
204  return;
205  }
206 
207  auto& row_distr = gvp_->gvec_fft_slab();
208 
209  auto& comm_col = gvp_->comm_ortho_fft();
210 
211  /* local number of columns */
212  int n_loc = spl_num_col_.local_size();
213 
214  /* send and recieve dimensions */
215  block_data_descriptor sd(comm_col.size()), rd(comm_col.size());
216  for (int j = 0; j < comm_col.size(); j++) {
217  sd.counts[j] = spl_num_col_.local_size(j) * row_distr.counts[comm_col.rank()];
218  rd.counts[j] = spl_num_col_.local_size(comm_col.rank()) * row_distr.counts[j];
219  }
220  sd.calc_offsets();
221  rd.calc_offsets();
222 
223  T* send_buf = (num_rows_loc_ == 0) ? nullptr : prime_.template at<CPU>(0, idx0__);
224 
225  comm_col.alltoall(send_buf, sd.counts.data(), sd.offsets.data(), send_recv_buf_.template at<CPU>(),
226  rd.counts.data(), rd.offsets.data());
227 
228  /* reorder recieved blocks */
229  #pragma omp parallel for
230  for (int i = 0; i < n_loc; i++) {
231  for (int j = 0; j < comm_col.size(); j++) {
232  int offset = row_distr.offsets[j];
233  int count = row_distr.counts[j];
234  if (count) {
235  std::memcpy(&extra_(offset, i), &send_recv_buf_[offset * n_loc + count * i], count * sizeof(T));
236  }
237  }
238  }
239  /* copy extra storage to the device if needed */
240  if (pu__ == GPU) {
241  extra_.template copy<memory_t::host, memory_t::device>();
242  }
243  }
244 
245  /// Remap data from extra to prime storage.
246  /** \param [in] pu Target processing unit.
247  * \param [in] row_distr Distribution of rows of prime matrix in the extra matrix storage.
248  * \param [in] n Number of matrix columns to collect.
249  * \param [in] idx0 Starting column of the matrix.
250  *
251  * Extra storage is expected on the CPU (for the MPI a2a communication). If the target processing unit is GPU
252  * prime storage will be copied to the device memory. */
253  inline void remap_backward(device_t pu__,
254  int n__,
255  int idx0__ = 0)
256  {
257  PROFILE("sddk::matrix_storage::remap_backward");
258 
259  /* trivial case when extra storage mirrors the prime storage */
260  if (!is_remapped()) {
261  return;
262  }
263 
264  auto& comm_col = gvp_->comm_ortho_fft();
265 
266  auto& row_distr = gvp_->gvec_fft_slab();
267 
268  assert(n__ == spl_num_col_.global_index_size());
269 
270  /* local number of columns */
271  int n_loc = spl_num_col_.local_size();
272 
273  /* reorder sending blocks */
274  #pragma omp parallel for
275  for (int i = 0; i < n_loc; i++) {
276  for (int j = 0; j < comm_col.size(); j++) {
277  int offset = row_distr.offsets[j];
278  int count = row_distr.counts[j];
279  if (count) {
280  std::memcpy(&send_recv_buf_[offset * n_loc + count * i], &extra_(offset, i), count * sizeof(T));
281  }
282  }
283  }
284 
285  /* send and recieve dimensions */
286  block_data_descriptor sd(comm_col.size()), rd(comm_col.size());
287  for (int j = 0; j < comm_col.size(); j++) {
288  sd.counts[j] = spl_num_col_.local_size(comm_col.rank()) * row_distr.counts[j];
289  rd.counts[j] = spl_num_col_.local_size(j) * row_distr.counts[comm_col.rank()];
290  }
291  sd.calc_offsets();
292  rd.calc_offsets();
293 
294  T* recv_buf = (num_rows_loc_ == 0) ? nullptr : prime_.template at<CPU>(0, idx0__);
295 
296  comm_col.alltoall(send_recv_buf_.template at<CPU>(), sd.counts.data(), sd.offsets.data(), recv_buf,
297  rd.counts.data(), rd.offsets.data());
298 
299  /* move data back to device */
300  if (pu__ == GPU && prime_.on_device()) {
301  prime_.template copy<memory_t::host, memory_t::device>(idx0__ * num_rows_loc(), n__ * num_rows_loc());
302  }
303  }

305  inline void remap_from(dmatrix<T> const& mtrx__, int irow0__)
306  {
307  PROFILE("sddk::matrix_storage::remap_from");
308 
309  auto& comm = mtrx__.blacs_grid().comm();
310 
311  /* cache cartesian ranks */
312  mdarray<int, 2> cart_rank(mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().num_ranks_col());
313  for (int i = 0; i < mtrx__.blacs_grid().num_ranks_col(); i++) {
314  for (int j = 0; j < mtrx__.blacs_grid().num_ranks_row(); j++) {
315  cart_rank(j, i) = mtrx__.blacs_grid().cart_rank(j, i);
316  }
317  }
318 
319  if (send_recv_buf_.size() < prime_.size()) {
320  send_recv_buf_ = mdarray<T, 1>(prime_.size(), memory_t::host, "matrix_storage::send_recv_buf_");
321  }
322 
323  block_data_descriptor rd(comm.size());
324  rd.counts[comm.rank()] = num_rows_loc_;
325  comm.allgather(rd.counts.data(), comm.rank(), 1);
326  rd.calc_offsets();
327 
328  block_data_descriptor sd(comm.size());
329 
330  /* global index of column */
331  int j0 = 0;
332  /* actual number of columns in the submatrix */
333  int ncol = num_cols_;
334 
335  splindex<block_cyclic> spl_col_begin(j0, mtrx__.num_ranks_col(), mtrx__.rank_col(), mtrx__.bs_col());
336  splindex<block_cyclic> spl_col_end(j0 + ncol, mtrx__.num_ranks_col(), mtrx__.rank_col(), mtrx__.bs_col());
337 
338  int local_size_col = spl_col_end.local_size() - spl_col_begin.local_size();
339 
340  for (int rank_row = 0; rank_row < comm.size(); rank_row++) {
341  if (!rd.counts[rank_row]) {
342  continue;
343  }
344  /* global index of column */
345  int i0 = rd.offsets[rank_row];
346  /* actual number of rows in the submatrix */
347  int nrow = rd.counts[rank_row];
348 
349  assert(nrow != 0);
350 
351  splindex<block_cyclic> spl_row_begin(irow0__ + i0, mtrx__.num_ranks_row(), mtrx__.rank_row(),
352  mtrx__.bs_row());
353  splindex<block_cyclic> spl_row_end(irow0__ + i0 + nrow, mtrx__.num_ranks_row(), mtrx__.rank_row(),
354  mtrx__.bs_row());
355 
356  int local_size_row = spl_row_end.local_size() - spl_row_begin.local_size();
357 
358  mdarray<T, 1> buf(local_size_row * local_size_col);
359 
360  /* fetch elements of sub-matrix matrix */
361  if (local_size_row) {
362  for (int j = 0; j < local_size_col; j++) {
363  std::memcpy(&buf[local_size_row * j],
364  &mtrx__(spl_row_begin.local_size(), spl_col_begin.local_size() + j),
365  local_size_row * sizeof(T));
366  }
367  }
368 
369  sd.counts[comm.rank()] = local_size_row * local_size_col;
370  comm.allgather(sd.counts.data(), comm.rank(), 1);
371  sd.calc_offsets();
372 
373  /* collect buffers submatrix */
374  T* send_buf = (buf.size() == 0) ? nullptr : &buf[0];
375  T* recv_buf = (send_recv_buf_.size() == 0) ? nullptr : &send_recv_buf_[0];
376  comm.gather(send_buf, recv_buf, sd.counts.data(), sd.offsets.data(), rank_row);
377 
378  if (comm.rank() == rank_row) {
379  /* unpack data */
380  std::vector<int> counts(comm.size(), 0);
381  for (int jcol = 0; jcol < ncol; jcol++) {
382  auto pos_jcol = mtrx__.spl_col().location(j0 + jcol);
383  for (int irow = 0; irow < nrow; irow++) {
384  auto pos_irow = mtrx__.spl_row().location(irow0__ + i0 + irow);
385  int rank = cart_rank(pos_irow.rank, pos_jcol.rank);
386 
387  prime_(irow, jcol) = send_recv_buf_[sd.offsets[rank] + counts[rank]];
388  counts[rank]++;
389  }
390  }
391  for (int rank = 0; rank < comm.size(); rank++) {
392  assert(sd.counts[rank] == counts[rank]);
393  }
394  }
395  }
396  }

446  template <memory_t mem_type>
447  inline void scale(int i0__, int n__, double beta__)
448  {
449  switch (mem_type) {
450  case memory_t::host:
451  case memory_t::host_pinned: {
452  for (int i = 0; i < n__; i++) {
453  for (int j = 0; j < num_rows_loc(); j++) {
454  prime(j, i0__ + i) *= beta__;
455  }
456  }
457  break;
458  }
459  case memory_t::device: {
460  #ifdef __GPU
461  scale_matrix_elements_gpu(reinterpret_cast<cuDoubleComplex*>(prime().template at<GPU>(0, i0__)),
462  prime().ld(), num_rows_loc(), n__, beta__);
463  #endif
464  break;
465  }
466  }
467  }

500  inline double_complex checksum(device_t pu__,
501  int i0__,
502  int n__)
503  {
504  double_complex cs(0, 0);
505 
506  switch (pu__) {
507  case CPU: {
508  for (int i = 0; i < n__; i++) {
509  for (int j = 0; j < num_rows_loc(); j++) {
510  cs += prime(j, i0__ + i);
511  }
512  }
513  break;
514  }
515  case GPU: {
516  mdarray<double_complex, 1> cs1(n__, memory_t::host | memory_t::device, "checksum");
517  cs1.zero<memory_t::device>();
518  #ifdef __GPU
519  add_checksum_gpu(prime().template at<GPU>(0, i0__), num_rows_loc(), n__, cs1.at<GPU>());
520  cs1.copy<memory_t::device, memory_t::host>();
521  cs = cs1.checksum();
522  #endif
523  break;
524  }
525  }
526  return cs;
527  }
