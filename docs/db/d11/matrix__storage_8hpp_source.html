113  /// Check if data needs to be remapped.
114  /** Data is not remapped when communicator that is orthogonal to FFT communicator is trivial.
115  * In this case the FFT communicator coincides with the entire communicator used to distribute wave functions
116  * and the data is ready for FFT transformations. */
117  inline bool is_remapped() const
118  {
119  return (gvp_->comm_ortho_fft().size() > 1);
120  }

122  /// Set the dimensions of the extra matrix storage.
123  /** \param [in] n Number of matrix columns to distribute.
124  * \param [in] idx0 Starting column of the matrix.
125  * \param [in] mp Memory pool that can be used for fast reallocation.
126  *
127  * \image html matrix_storage.png "Redistribution of wave-functions between MPI ranks"
128  *
129  * The extra storage is always created in the CPU memory. If the data distribution doesn't
130  * change (no swapping between comm_col ranks is performed) – then the extra storage will mirror
131  * the prime storage (both on CPU and GPU) irrespective of the target processing unit. If
132  * data remapping is necessary extra storage is allocated only in the host memory because MPI is done using the
133  * host pointers.
134  */
135  inline void set_num_extra(int n__, int idx0__, memory_pool* mp__)
136  {
137  PROFILE("sddk::matrix_storage::set_num_extra");
138 
139  auto& comm_col = gvp_->comm_ortho_fft();
140 
141  /* this is how n columns of the matrix will be distributed between columns of the MPI grid */
142  spl_num_col_ = splindex<block>(n__, comm_col.size(), comm_col.rank());
143 
144  T* ptr{nullptr};
145  T* ptr_d{nullptr};
146  int ncol{0};
147 
148  /* trivial case */
149  if (!is_remapped()) {
150  assert(num_rows_loc_ == gvp_->gvec_count_fft());
151  ncol = n__;
152  ptr = prime_.at(memory_t::host, 0, idx0__);
153  if (prime_.on_device()) {
154  ptr_d = prime_.at(memory_t::device, 0, idx0__);
155  }
156  } else {
157  /* maximum local number of matrix columns */
158  ncol = splindex_base<int>::block_size(n__, comm_col.size());
159  /* upper limit for the size of swapped extra matrix */
160  size_t sz = gvp_->gvec_count_fft() * ncol;
161  /* reallocate buffers if necessary */
162  if (extra_buf_.size() < sz) {
163  utils::timer t1("sddk::matrix_storage::set_num_extra|alloc");
164  if (mp__) {
165  send_recv_buf_ = mdarray<T, 1>(*mp__, sz, "matrix_storage.send_recv_buf_");
166  extra_buf_ = mdarray<T, 1>(*mp__, sz, "matrix_storage.extra_buf_");
167  } else {
168  send_recv_buf_ = mdarray<T, 1>(sz, memory_t::host, "matrix_storage.send_recv_buf_");
169  extra_buf_ = mdarray<T, 1>(sz, memory_t::host, "matrix_storage.extra_buf_");
170  }
171  }
172  ptr = extra_buf_.at(memory_t::host);
173  }
174  /* create the extra storage */
175  extra_ = mdarray<T, 2>(ptr, ptr_d, gvp_->gvec_count_fft(), ncol, "matrix_storage.extra_");
176  }
177 
178  /// Remap data from prime to extra storage.
179  /** \param [in] n Number of matrix columns to distribute.
180  * \param [in] idx0 Starting column of the matrix.
181  *
182  * Prime storage is expected on the CPU (for the MPI a2a communication). */
183  inline void remap_forward(int n__, int idx0__, memory_pool* mp__)
184  {
185  PROFILE("sddk::matrix_storage::remap_forward");
186 
187  set_num_extra(n__, idx0__, mp__);
188 
189  /* trivial case when extra storage mirrors the prime storage */
190  if (!is_remapped()) {
191  return;
192  }
193 
194  auto& row_distr = gvp_->gvec_fft_slab();
195 
196  auto& comm_col = gvp_->comm_ortho_fft();
197 
198  /* local number of columns */
199  int n_loc = spl_num_col_.local_size();
200 
201  /* send and recieve dimensions */
202  block_data_descriptor sd(comm_col.size()), rd(comm_col.size());
203  for (int j = 0; j < comm_col.size(); j++) {
204  sd.counts[j] = spl_num_col_.local_size(j) * row_distr.counts[comm_col.rank()];
205  rd.counts[j] = spl_num_col_.local_size(comm_col.rank()) * row_distr.counts[j];
206  }
207  sd.calc_offsets();
208  rd.calc_offsets();
209 
210  T* send_buf = (num_rows_loc_ == 0) ? nullptr : prime_.at(memory_t::host, 0, idx0__);
211 
212  utils::timer t1("sddk::matrix_storage::remap_forward|mpi");
213  comm_col.alltoall(send_buf, sd.counts.data(), sd.offsets.data(), send_recv_buf_.at(memory_t::host),
214  rd.counts.data(), rd.offsets.data());
215  t1.stop();
216 
217  /* reorder recieved blocks */
218  #pragma omp parallel for
219  for (int i = 0; i < n_loc; i++) {
220  for (int j = 0; j < comm_col.size(); j++) {
221  int offset = row_distr.offsets[j];
222  int count = row_distr.counts[j];
223  if (count) {
224  std::memcpy(&extra_(offset, i), &send_recv_buf_[offset * n_loc + count * i], count * sizeof(T));
225  }
226  }
227  }
228  }
229 
230  /// Remap data from extra to prime storage.
231  /** \param [in] n Number of matrix columns to collect.
232  * \param [in] idx0 Starting column of the matrix.
233  *
234  * Extra storage is expected on the CPU (for the MPI a2a communication). If the prime storage is allocated on GPU
235  * remapped data will be copied to GPU. */
236  inline void remap_backward(int n__, int idx0__)
237  {
238  PROFILE("sddk::matrix_storage::remap_backward");
239 
240  /* trivial case when extra storage mirrors the prime storage */
241  if (!is_remapped()) {
242  return;
243  }
244 
245  auto& comm_col = gvp_->comm_ortho_fft();
246 
247  auto& row_distr = gvp_->gvec_fft_slab();
248 
249  assert(n__ == spl_num_col_.global_index_size());
250 
251  /* local number of columns */
252  int n_loc = spl_num_col_.local_size();
253 
254  /* reorder sending blocks */
255  #pragma omp parallel for
256  for (int i = 0; i < n_loc; i++) {
257  for (int j = 0; j < comm_col.size(); j++) {
258  int offset = row_distr.offsets[j];
259  int count = row_distr.counts[j];
260  if (count) {
261  std::memcpy(&send_recv_buf_[offset * n_loc + count * i], &extra_(offset, i), count * sizeof(T));
262  }
263  }
264  }
265 
266  /* send and recieve dimensions */
267  block_data_descriptor sd(comm_col.size()), rd(comm_col.size());
268  for (int j = 0; j < comm_col.size(); j++) {
269  sd.counts[j] = spl_num_col_.local_size(comm_col.rank()) * row_distr.counts[j];
270  rd.counts[j] = spl_num_col_.local_size(j) * row_distr.counts[comm_col.rank()];
271  }
272  sd.calc_offsets();
273  rd.calc_offsets();
274 
275  T* recv_buf = (num_rows_loc_ == 0) ? nullptr : prime_.at(memory_t::host, 0, idx0__);
276 
277  utils::timer t1("sddk::matrix_storage::remap_backward|mpi");
278  comm_col.alltoall(send_recv_buf_.at(memory_t::host), sd.counts.data(), sd.offsets.data(), recv_buf,
279  rd.counts.data(), rd.offsets.data());
280  t1.stop();
281 
282  /* move data back to device */
283  if (prime_.on_device()) {
284  prime_.copy_to(memory_t::device, idx0__ * num_rows_loc(), n__ * num_rows_loc());
285  }
286  }

288  inline void remap_from(dmatrix<T> const& mtrx__, int irow0__)
289  {
290  PROFILE("sddk::matrix_storage::remap_from");
291 
292  auto& comm = mtrx__.blacs_grid().comm();
293 
294  /* cache cartesian ranks */
295  mdarray<int, 2> cart_rank(mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().num_ranks_col());
296  for (int i = 0; i < mtrx__.blacs_grid().num_ranks_col(); i++) {
297  for (int j = 0; j < mtrx__.blacs_grid().num_ranks_row(); j++) {
298  cart_rank(j, i) = mtrx__.blacs_grid().cart_rank(j, i);
299  }
300  }
301 
302  if (send_recv_buf_.size() < prime_.size()) {
303  send_recv_buf_ = mdarray<T, 1>(prime_.size(), memory_t::host, "matrix_storage::send_recv_buf_");
304  }
305 
306  block_data_descriptor rd(comm.size());
307  rd.counts[comm.rank()] = num_rows_loc();
308  comm.allgather(rd.counts.data(), comm.rank(), 1);
309  rd.calc_offsets();
310 
311  block_data_descriptor sd(comm.size());
312 
313  /* global index of column */
314  int j0 = 0;
315  /* actual number of columns in the submatrix */
316  int ncol = num_cols_;
317 
318  splindex<block_cyclic> spl_col_begin(j0, mtrx__.num_ranks_col(), mtrx__.rank_col(), mtrx__.bs_col());
319  splindex<block_cyclic> spl_col_end(j0 + ncol, mtrx__.num_ranks_col(), mtrx__.rank_col(), mtrx__.bs_col());
320 
321  int local_size_col = spl_col_end.local_size() - spl_col_begin.local_size();
322 
323  for (int rank_row = 0; rank_row < comm.size(); rank_row++) {
324  if (!rd.counts[rank_row]) {
325  continue;
326  }
327  /* global index of column */
328  int i0 = rd.offsets[rank_row];
329  /* actual number of rows in the submatrix */
330  int nrow = rd.counts[rank_row];
331 
332  assert(nrow != 0);
333 
334  splindex<block_cyclic> spl_row_begin(irow0__ + i0, mtrx__.num_ranks_row(), mtrx__.rank_row(),
335  mtrx__.bs_row());
336  splindex<block_cyclic> spl_row_end(irow0__ + i0 + nrow, mtrx__.num_ranks_row(), mtrx__.rank_row(),
337  mtrx__.bs_row());
338 
339  int local_size_row = spl_row_end.local_size() - spl_row_begin.local_size();
340 
341  mdarray<T, 1> buf(local_size_row * local_size_col);
342 
343  /* fetch elements of sub-matrix matrix */
344  if (local_size_row) {
345  for (int j = 0; j < local_size_col; j++) {
346  std::memcpy(&buf[local_size_row * j],
347  &mtrx__(spl_row_begin.local_size(), spl_col_begin.local_size() + j),
348  local_size_row * sizeof(T));
349  }
350  }
351 
352  sd.counts[comm.rank()] = local_size_row * local_size_col;
353  comm.allgather(sd.counts.data(), comm.rank(), 1);
354  sd.calc_offsets();
355 
356  /* collect buffers submatrix */
357  T* send_buf = (buf.size() == 0) ? nullptr : &buf[0];
358  T* recv_buf = (send_recv_buf_.size() == 0) ? nullptr : &send_recv_buf_[0];
359  comm.gather(send_buf, recv_buf, sd.counts.data(), sd.offsets.data(), rank_row);
360 
361  if (comm.rank() == rank_row) {
362  /* unpack data */
363  std::vector<int> counts(comm.size(), 0);
364  for (int jcol = 0; jcol < ncol; jcol++) {
365  auto pos_jcol = mtrx__.spl_col().location(j0 + jcol);
366  for (int irow = 0; irow < nrow; irow++) {
367  auto pos_irow = mtrx__.spl_row().location(irow0__ + i0 + irow);
368  int rank = cart_rank(pos_irow.rank, pos_jcol.rank);
369 
370  prime_(irow, jcol) = send_recv_buf_[sd.offsets[rank] + counts[rank]];
371  counts[rank]++;
372  }
373  }
374  for (int rank = 0; rank < comm.size(); rank++) {
375  assert(sd.counts[rank] == counts[rank]);
376  }
377  }
378  }
379  }

469  inline double_complex checksum(device_t pu__, int i0__, int n__)
470  {
471  double_complex cs(0, 0);
472 
473  switch (pu__) {
474  case device_t::CPU: {
475  for (int i = 0; i < n__; i++) {
476  for (int j = 0; j < num_rows_loc(); j++) {
477  cs += prime(j, i0__ + i);
478  }
479  }
480  break;
481  }
482  case device_t::GPU: {
483  mdarray<double_complex, 1> cs1(n__, memory_t::host, "checksum");
484  cs1.allocate(memory_t::device).zero(memory_t::device);
485 #if defined(__GPU)
486  add_checksum_gpu(prime().at(memory_t::device, 0, i0__), num_rows_loc(), n__, cs1.at(memory_t::device));
487  cs1.copy_to(memory_t::host);
488  cs = cs1.checksum();
489 #endif
490  break;
491  }
492  }
493  return cs;
494  }
