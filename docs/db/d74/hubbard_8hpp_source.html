138  if (ctx_.num_mag_dims() != 3) {
139  calculate_hubbard_potential_and_energy_colinear_case();

141  calculate_hubbard_potential_and_energy_non_colinear_case();

144  // The potential should be hermitian from the calculations but
145  // by security I make it hermitian again
146 
147  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
148  auto& atom = unit_cell_.atom(ia);
149  if (atom.type().hubbard_correction()) {
150  // diagonal up up down down blocks
151  for (int is = 0; is < ctx_.num_spins(); is++) {
152  for (int m1 = 0; m1 < 2 * atom.type().hubbard_l() + 1; ++m1) {
153  for (int m2 = m1 + 1; m2 < 2 * atom.type().hubbard_l() + 1; ++m2) {

161  for (int m2 = 0; m2 < 2 * atom.type().hubbard_l() + 1; ++m2) {
162  this->U(m1, m2, 3, ia) = std::conj(this->U(m2, m1, 2, ia));

257  inline void calculate_wavefunction_with_U_offset()
258  {
259  offset.clear();
260  offset.resize(ctx_.unit_cell().num_atoms(), -1);
261 
262  int counter = 0;
263  for (auto ia = 0; ia < unit_cell_.num_atoms(); ia++) {
264  auto& atom = unit_cell_.atom(ia);
265 
266  if (atom.type().hubbard_correction()) {
267  // search for the orbital of given l corresponding to the
268  // hubbard l, with strickly positive occupation
269  for (int wfc = 0; wfc < atom.type().num_ps_atomic_wf(); wfc++) {
270  const int l = std::abs(atom.type().ps_atomic_wf(wfc).first);
271  const double occ = atom.type().ps_atomic_wf_occ()[wfc];
272  if ((occ >= 0.0) && (l == atom.type().hubbard_l())) {
273  // a wave function is hubbard if and only if the occupation
274  // number is positive and l corresponds to hubbard_lmax;
275  bool hubbard_wfc = (occ > 0);
276  if (hubbard_wfc && (offset[ia] < 0)) {
277  offset[ia] = counter;
278  }
279 
280  // the atom has spin orbit coupling so we have
281  // two wave functions with same l but different
282  // j
283  if (atom.type().spin_orbit_coupling() && hubbard_wfc) {
284  counter += (2 * l + 1);
285  } else {
286  if (hubbard_wfc && (ctx_.num_mag_dims() == 3)) {
287  // the pseudo potential does not include
288  // spin orbit coupling but we do
289  // calculation with non colinear
290  // magnetism so we still have full
291  // hubbard spinors
292  counter += 2 * (2 * l + 1);
293  }
294  if (hubbard_wfc && (ctx_.num_mag_dims() != 3)) {
295  // colinear or conventional LDA
296  counter += (2 * l + 1);
297  }
298  }
299  }
300  }
301  }
302  }
303  // compute the number of orbitals
304  this->number_of_hubbard_orbitals_ = counter;
305  }
