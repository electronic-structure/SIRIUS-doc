101  inline void calculate_wavefunction_with_U_offset()
102  {
103  offset.clear();
104  offset.resize(ctx_.unit_cell().num_atoms(), -1);
105 
106  int counter = 0;
107 
108  // we loop over atoms to check which atom has hubbard orbitals
109  // and then compute the number of hubbard orbitals associated to
110  // it.
111  for (auto ia = 0; ia < unit_cell_.num_atoms(); ia++) {
112  auto& atom = unit_cell_.atom(ia);
113  if (atom.type().hubbard_correction()) {
114  offset[ia] = counter;
115  if (ctx_.num_mag_dims() == 3) {
116  for (auto&& orb : atom.type().hubbard_orbital()) {
117  if (atom.type().spin_orbit_coupling()) {
118  counter += (2 * orb.l() + 1);
119  } else {
120  counter += 2 * (2 * orb.l() + 1);
121  }
122  }
123  } else {
124  for (auto&& orb : atom.type().hubbard_orbital()) {
125  counter += (2 * orb.l() + 1);
126  }
127  }
128  }
129  }
130 
131  this->number_of_hubbard_orbitals_ = counter;
132  }

242  void calculate_hubbard_potential_and_energy()
243  {
244  this->hubbard_energy_ = 0.0;
245  this->hubbard_energy_u_ = 0.0;
246  this->hubbard_energy_dc_contribution_ = 0.0;
247  this->hubbard_energy_noflip_ = 0.0;
248  this->hubbard_energy_flip_ = 0.0;
249  // the hubbard potential has the same structure than the occupation
250  // numbers
251  this->hubbard_potential_.zero();
252 
253  if (ctx_.num_mag_dims() != 3) {
254  calculate_hubbard_potential_and_energy_colinear_case();
255  } else {
256  calculate_hubbard_potential_and_energy_non_colinear_case();
257  }
258  }

270  Hubbard(Simulation_context& ctx__)
271  : ctx_(ctx__)
272  , unit_cell_(ctx__.unit_cell())
273  {
274  if (!ctx_.hubbard_correction()) {
275  return;
276  }
277  orthogonalize_hubbard_orbitals_ = ctx_.Hubbard().orthogonalize_hubbard_orbitals_;
278  normalize_orbitals_only_ = ctx_.Hubbard().normalize_hubbard_orbitals_;
279  projection_method_ = ctx_.Hubbard().projection_method_;
280 
281  // if the projectors are defined externaly then we need the file
282  // that contains them. All the other methods do not depend on
283  // that parameter
284  if (this->projection_method_ == 1) {
285  this->wave_function_file_ = ctx_.Hubbard().wave_function_file_;
286  }
287 
288  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
289  auto& atom_type = ctx_.unit_cell().atom(ia).type();
290  if (ctx__.unit_cell().atom(ia).type().hubbard_correction()) {
291  for (int channel = 0; channel < atom_type.number_of_hubbard_channels(); channel++) {
292  this->lmax_ = std::max(this->lmax_, atom_type.hubbard_orbital(channel).l());
293  }
294  }
295  }
296 
297  /// if spin orbit coupling or non colinear magnetisms are
298  /// activated, then we consider the full spherical hubbard
299  /// correction
300  if ((ctx_.so_correction()) || (ctx_.num_mag_dims() == 3)) {
301  approximation_ = false;
302  }
303 
304  // prepare things for the multi channel case. The last index
305  // indicates which channel we consider. By default we only have
306  // one channel per atomic type
307  occupancy_number_ = mdarray<double_complex, 5>(2 * lmax_ + 1, 2 * lmax_ + 1, 4, ctx_.unit_cell().num_atoms(), 1);
308  hubbard_potential_ = mdarray<double_complex, 5>(2 * lmax_ + 1, 2 * lmax_ + 1, 4, ctx_.unit_cell().num_atoms(), 1);
309 
310  calculate_wavefunction_with_U_offset();
311  calculate_initial_occupation_numbers();
312  calculate_hubbard_potential_and_energy();
313  }
