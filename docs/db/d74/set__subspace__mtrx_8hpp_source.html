32 {
33  PROFILE("sirius::Band::set_subspace_mtrx");
34 
35  assert(n__ != 0);
36  if (mtrx_old__.size()) {
37  assert(&mtrx__.blacs_grid() == &mtrx_old__.blacs_grid());
38  }
39 
40  /* copy old N x N distributed matrix */
41  if (N__ > 0) {
42  splindex<block_cyclic> spl_row(N__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(),
43  mtrx__.bs_row());

45  mtrx__.bs_col());
46 
47  #pragma omp parallel for schedule(static)
48  for (int i = 0; i < spl_col.local_size(); i++) {
49  std::copy(&mtrx_old__(0, i), &mtrx_old__(0, i) + spl_row.local_size(), &mtrx__(0, i));
50  }
51

54  for (int i = 0; i < spl_col.local_size(); i++) {
55  for (int j = 0; j < spl_row.local_size(); j++) {
56  cs += mtrx__(j, i);
57  }
58  }
59  mtrx__.blacs_grid().comm().allreduce(&cs, 1);

63  }

68  mtrx__, 0, N__);

90  mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
91  splindex<block_cyclic> spl_col(N__ + n__, mtrx__.blacs_grid().num_ranks_col(),
92  mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());

95  for (int j = 0; j < spl_row.local_size(); j++) {
96  cs += mtrx__(j, i);
97  }
98  }

101  utils::print_checksum("subspace_mtrx", cs);
102  }
103  }
104 
105  /* kill any numerical noise */
106  mtrx__.make_real_diag(N__ + n__);
107 
108  /* save new matrix */
109  if (mtrx_old__.size()) {
110  splindex<block_cyclic> spl_row(N__ + n__, mtrx__.blacs_grid().num_ranks_row(),
111  mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
112  splindex<block_cyclic> spl_col(N__ + n__, mtrx__.blacs_grid().num_ranks_col(),
113  mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
114 
115  #pragma omp parallel for schedule(static)
116  for (int i = 0; i < spl_col.local_size(); i++) {
117  std::copy(&mtrx__(0, i), &mtrx__(0, i) + spl_row.local_size(), &mtrx_old__(0, i));
118  }
119  }
120 }
