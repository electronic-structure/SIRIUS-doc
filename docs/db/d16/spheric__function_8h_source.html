236 /// Inner product of two spherical functions.
237 template <function_domain_t domain_t, typename T>
238 T inner(Spheric_function<domain_t, T> const& f1, Spheric_function<domain_t, T> const& f2)
239 {
240  /* check radial grid */
241  //if (f1.radial_grid().hash() != f2.radial_grid().hash()) {
242  // TERMINATE("radial grids don't match");
243  //}
244 
245  Spline<T> s(f1.radial_grid());
246 
247  if (domain_t == spectral) {
248  int lmmax = std::min(f1.angular_domain_size(), f2.angular_domain_size());
249  for (int ir = 0; ir < s.num_points(); ir++) {
250  for (int lm = 0; lm < lmmax; lm++) {
251  s(ir) += type_wrapper<T>::bypass(std::conj(f1(lm, ir))) * f2(lm, ir);
252  }
253  }
254  } else {
255  TERMINATE_NOT_IMPLEMENTED
256  }
257  return s.interpolate().integrate(2);
258 }

260 /// Compute Laplacian of the spheric function.
261 /** Laplacian in spherical coordinates has the following expression:
262  * \f[
263  * \Delta = \frac{1}{r^2}\frac{\partial}{\partial r}\Big( r^2 \frac{\partial}{\partial r} \Big) + \frac{1}{r^2}\Delta_{\theta, \phi}
264  * \f]
265  */
266 template <typename T>
267 Spheric_function<spectral, T> laplacian(Spheric_function<spectral, T> const& f__)
268 {
269  Spheric_function<spectral, T> g;
270  auto& rgrid = f__.radial_grid();
271  int lmmax = f__.angular_domain_size();
272  int lmax = utils::lmax(lmmax);
273  g = Spheric_function<spectral, T>(lmmax, rgrid);
274 
275  Spline<T> s1(rgrid);
276  for (int l = 0; l <= lmax; l++) {
277  int ll = l * (l + 1);
278  for (int m = -l; m <= l; m++) {
279  int lm = utils::lm(l, m);
280  /* get lm component */
281  auto s = f__.component(lm);
282  /* compute 1st derivative */
283  for (int ir = 0; ir < s.num_points(); ir++) {
284  s1(ir) = s.deriv(1, ir);
285  }
286  s1.interpolate();
287 
288  for (int ir = 0; ir < s.num_points(); ir++) {
289  g(lm, ir) = 2 * s1(ir) * rgrid.x_inv(ir) + s1.deriv(1, ir) - s(ir) * ll / std::pow(rgrid[ir], 2);
290  }
291  }
292  }
293 
294  return std::move(g);
295 }

297 /// Convert from Ylm to Rlm representation.
298 inline Spheric_function<spectral, double> convert(Spheric_function<spectral, double_complex> const& f__)
299 {
300  int lmax = utils::lmax(f__.angular_domain_size());
301 
302  /* cache transformation arrays */
303  std::vector<double_complex> tpp(f__.angular_domain_size());
304  std::vector<double_complex> tpm(f__.angular_domain_size());
305  for (int l = 0; l <= lmax; l++) {
306  for (int m = -l; m <= l; m++) {
307  int lm = utils::lm(l, m);
308  tpp[lm] = SHT::rlm_dot_ylm(l, m, m);
309  tpm[lm] = SHT::rlm_dot_ylm(l, m, -m);
310  }
311  }
312 
313  Spheric_function<spectral, double> g(f__.angular_domain_size(), f__.radial_grid());
314 
315  for (int ir = 0; ir < f__.radial_grid().num_points(); ir++) {
316  int lm = 0;
317  for (int l = 0; l <= lmax; l++) {
318  for (int m = -l; m <= l; m++) {
319  if (m == 0) {
320  g(lm, ir) = std::real(f__(lm, ir));
321  } else {
322  int lm1 = utils::lm(l, -m);
323  g(lm, ir) = std::real(tpp[lm] * f__(lm, ir) + tpm[lm] * f__(lm1, ir));
324  }
325  lm++;
326  }
327  }
328  }
329 
330  return std::move(g);
331 }

333 /// Convert from Rlm to Ylm representation.
334 inline Spheric_function<spectral, double_complex> convert(Spheric_function<spectral, double> const& f__)
335 {
336  int lmax = utils::lmax(f__.angular_domain_size());
337 
338  /* cache transformation arrays */
339  std::vector<double_complex> tpp(f__.angular_domain_size());
340  std::vector<double_complex> tpm(f__.angular_domain_size());
341  for (int l = 0; l <= lmax; l++) {
342  for (int m = -l; m <= l; m++) {
343  int lm = utils::lm(l, m);
344  tpp[lm] = SHT::ylm_dot_rlm(l, m, m);
345  tpm[lm] = SHT::ylm_dot_rlm(l, m, -m);
346  }
347  }
348 
349  Spheric_function<spectral, double_complex> g(f__.angular_domain_size(), f__.radial_grid());
350 
351  for (int ir = 0; ir < f__.radial_grid().num_points(); ir++) {
352  int lm = 0;
353  for (int l = 0; l <= lmax; l++) {
354  for (int m = -l; m <= l; m++) {
355  if (m == 0) {
356  g(lm, ir) = f__(lm, ir);
357  } else {
358  int lm1 = utils::lm(l, -m);
359  g(lm, ir) = tpp[lm] * f__(lm, ir) + tpm[lm] * f__(lm1, ir);
360  }
361  lm++;
362  }
363  }
364  }
365 
366  return std::move(g);
367 }

435 /// Gradient of the function in complex spherical harmonics.
436 inline Spheric_function_gradient<spectral, double_complex> gradient(Spheric_function<spectral, double_complex>& f)
437 {
438  Spheric_function_gradient<spectral, double_complex> g(f.angular_domain_size(), f.radial_grid());
439  for (int i = 0; i < 3; i++) {
440  g[i] = Spheric_function<spectral, double_complex>(f.angular_domain_size(), f.radial_grid());
441  g[i].zero();
442  }
443 
444  int lmax = utils::lmax(f.angular_domain_size());
445 
446  //Spline<double_complex> s(f.radial_grid());
447 
448  for (int l = 0; l <= lmax; l++) {
449  double d1 = sqrt(double(l + 1) / double(2 * l + 3));
450  double d2 = sqrt(double(l) / double(2 * l - 1));
451 
452  for (int m = -l; m <= l; m++) {
453  int lm = utils::lm(l, m);
454  auto s = f.component(lm);
455 
456  for (int mu = -1; mu <= 1; mu++) {
457  int j = (mu + 2) % 3; // map -1,0,1 to 1,2,0
458 
459  if ((l + 1) <= lmax && abs(m + mu) <= l + 1) {
460  int lm1 = utils::lm(l + 1, m + mu);
461  double d = d1 * SHT::clebsch_gordan(l, 1, l + 1, m, mu, m + mu);
462  for (int ir = 0; ir < f.radial_grid().num_points(); ir++)
463  g[j](lm1, ir) += (s.deriv(1, ir) - f(lm, ir) * f.radial_grid().x_inv(ir) * double(l)) * d;
464  }
465  if ((l - 1) >= 0 && abs(m + mu) <= l - 1) {
466  int lm1 = utils::lm(l - 1, m + mu);
467  double d = d2 * SHT::clebsch_gordan(l, 1, l - 1, m, mu, m + mu);
468  for (int ir = 0; ir < f.radial_grid().num_points(); ir++)
469  g[j](lm1, ir) -= (s.deriv(1, ir) + f(lm, ir) * f.radial_grid().x_inv(ir) * double(l + 1)) * d;
470  }
471  }
472  }
473  }
474 
475  double_complex d1(1.0 / sqrt(2.0), 0);
476  double_complex d2(0, 1.0 / sqrt(2.0));
477 
478  for (int ir = 0; ir < f.radial_grid().num_points(); ir++) {
479  for (int lm = 0; lm < f.angular_domain_size(); lm++) {
480  double_complex g_p = g[0](lm, ir);
481  double_complex g_m = g[1](lm, ir);
482  g[0](lm, ir) = d1 * (g_m - g_p);
483  g[1](lm, ir) = d2 * (g_m + g_p);
484  }
485  }
486 
487  return g;
488 }

490 /// Gradient of the function in real spherical harmonics.
491 inline Spheric_function_gradient<spectral, double> gradient(Spheric_function<spectral, double> const& f)
492 {
493  int lmax = utils::lmax(f.angular_domain_size());
494  SHT sht(lmax);
495  auto zf = convert(f);
496  auto zg = gradient(zf);
497  Spheric_function_gradient<spectral, double> g(f.angular_domain_size(), f.radial_grid());
498  for (int x: {0, 1, 2}) {
499  g[x] = convert(zg[x]);
500  }
501  return g;
502 }

504 /// Dot product of two gradiensts.
505 inline Spheric_function<spatial, double> operator*(Spheric_function_gradient<spatial, double> const& f,
506  Spheric_function_gradient<spatial, double> const& g)
507 {
508  for (int x: {0, 1, 2}) {
509  if (f[x].angular_domain_size() != g[x].angular_domain_size()) {
510  TERMINATE("wrong number of angular points");
511  }
512  }
513 
514  Spheric_function<spatial, double> result(f.angular_domain_size(), f.radial_grid());
515  result.zero();
516 
517  for (int x: {0, 1, 2}) {
518  for (int ir = 0; ir < f[x].radial_grid().num_points(); ir++) {
519  for (int tp = 0; tp < f[x].angular_domain_size(); tp++) {
520  result(tp, ir) += f[x](tp, ir) * g[x](tp, ir);
521  }
522  }
523  }
524 
525  return result;
526 }
