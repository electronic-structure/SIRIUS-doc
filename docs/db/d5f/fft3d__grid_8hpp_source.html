33/// Handling of FFT grids.
34class FFT3D_grid : public std::array<int, 3>
35{
36 private:
37 /// Reciprocal space range.
38 std::array<std::pair<int, int>, 3> grid_limits_;
39
40 /// Find smallest optimal grid size starting from n.
41 int find_grid_size(int n)
42 {
43 while (true) {
44 int m = n;
45 for (int k = 2; k <= 5; k++) {
46 while (m % k == 0) {
47 m /= k;
48 }
49 }
50 if (m == 1) {
51 return n;
52 } else {
53 n++;
54 }
55 }
56 }
57
58 /// Find grid sizes and limits for all three dimensions.
59 void find_grid_size(std::array<int, 3> initial_dims__)
60 {
61 for (int i = 0; i < 3; i++) {
62 (*this)[i] = find_grid_size(initial_dims__[i]);
63
64 grid_limits_[i].second = (*this)[i] / 2;
65 grid_limits_[i].first = grid_limits_[i].second - (*this)[i] + 1;
66 }
67
68 for (int x = 0; x < (*this)[0]; x++) {
69 if (coord_by_freq<0>(freq_by_coord<0>(x)) != x) {
70 throw std::runtime_error("FFT3D_grid::find_grid_size(): wrong mapping of x-coordinates");
71 }
72 }
73 for (int x = 0; x < (*this)[1]; x++) {
74 if (coord_by_freq<1>(freq_by_coord<1>(x)) != x) {
75 throw std::runtime_error("FFT3D_grid::find_grid_size(): wrong mapping of y-coordinates");
76 }
77 }
78 for (int x = 0; x < (*this)[2]; x++) {
79 if (coord_by_freq<2>(freq_by_coord<2>(x)) != x) {
80 throw std::runtime_error("FFT3D_grid::find_grid_size(): wrong mapping of z-coordinates");
81 }
82 }
83 }
84
85 public:
86
87 /// Default constructor.
88 FFT3D_grid()
89 {
90 }
91
92 /// Create FFT grid with initial dimensions.
93 FFT3D_grid(std::array<int, 3> initial_dims__)
94 {
95 find_grid_size(initial_dims__);
96 }
97
98 /// Limits of a given dimension.
99 inline const std::pair<int, int>& limits(int idim__) const
100 {
101 assert(idim__ >= 0 && idim__ < 3);
102 return grid_limits_[idim__];
103 }
104
105 /// Total size of the FFT grid.
106 inline int num_points() const
107 {
108 return (*this)[0] * (*this)[1] * (*this)[2];
109 }
110
111 /// Get coordinate in range [0, N_d) by the frequency index.
112 template <int d>
113 inline int coord_by_freq(int i__) const
114 {
115 if (i__ < 0) {
116 i__ += (*this)[d];
117 }
118 return i__;
119 }
120
121 /// Return {x, y, z} coordinates by frequency indices.
122 inline std::array<int, 3> coord_by_freq(int i0__, int i1__, int i2__) const
123 {
124 return {coord_by_freq<0>(i0__), coord_by_freq<1>(i1__), coord_by_freq<2>(i2__)};
125 }
126
127 /// Get frequency by coordinate.
128 template <int d>
129 inline int freq_by_coord(int x__) const
130 {
131 if (x__ > grid_limits_[d].second) {
132 x__ -= (*this)[d];
133 }
134 return x__;
135 }
136
137 /// Return 3d vector of frequencies corresponding to {x, y, z} position in the FFT buffer.
138 inline std::array<int, 3> freq_by_coord(int x__, int y__, int z__) const
139 {
140 return {freq_by_coord<0>(x__), freq_by_coord<1>(y__), freq_by_coord<2>(z__)};
141 }
142
143 /// Linear index inside FFT buffer by grid coordinates.
144 inline int index_by_coord(int x__, int y__, int z__) const
145 {
146 return (x__ + (*this)[0] * (y__ + z__ * (*this)[1]));
147 }
148
149 /// Return linear index of a plane-wave harmonic with fractional coordinates (i0, i1, i2) inside FFT buffer.
150 inline int index_by_freq(int i0__, int i1__, int i2__) const
151 {
152 auto coord = coord_by_freq(i0__, i1__, i2__);
153 return index_by_coord(coord[0], coord[1], coord[2]);
154 }
155};
