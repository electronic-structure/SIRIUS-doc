30 /// Handling of FFT grids.
31 class FFT3D_grid
32 {
33  private:
34  /// Size of each dimension.
35  int grid_size_[3];
36 
37  /// Reciprocal space range.
38  std::pair<int, int> grid_limits_[3];
39 
40  /// Find smallest optimal grid size starting from n.
41  int find_grid_size(int n)
42  {
43  while (true) {
44  int m = n;
45  for (int k = 2; k <= 5; k++) {
46  while (m % k == 0) {
47  m /= k;
48  }
49  }
50  if (m == 1) {
51  return n;
52  } else {
53  n++;
54  }
55  }
56  }
57 
58  void find_grid_size(std::array<int, 3> initial_dims__)
59  {
60  for (int i = 0; i < 3; i++) {
61  grid_size_[i] = find_grid_size(initial_dims__[i]);
62 
63  grid_limits_[i].second = grid_size_[i] / 2;
64  grid_limits_[i].first = grid_limits_[i].second - grid_size_[i] + 1;
65  }
66 
67  for (int i = 0; i < 3; i++) {
68  for (int x = 0; x < size(i); x++) {
69  if (coord_by_gvec(gvec_by_coord(x, i), i) != x) {
70  throw std::runtime_error("find_grid_size: wrong mapping of coordinates");
71  }
72  }
73  }
74  }
75 
76  public:
77  FFT3D_grid()
78  {
79  }

81  FFT3D_grid(std::array<int, 3> initial_dims__)
82  {
83  find_grid_size(initial_dims__);
84  }

86  /// Limits of a given dimension.
87  inline const std::pair<int, int>& limits(int idim__) const
88  {
89  assert(idim__ >= 0 && idim__ < 3);
90  return grid_limits_[idim__];
91  }
92 
93  /// Size of a given dimension.
94  inline int size(int idim__) const
95  {
96  assert(idim__ >= 0 && idim__ < 3);
97  return grid_size_[idim__];
98  }
99 
100  /// Total size of the FFT grid.
101  inline int size() const
102  {
103  return grid_size_[0] * grid_size_[1] * grid_size_[2];
104  }
105 
106  inline std::array<int, 3> coord_by_gvec(int i0__, int i1__, int i2__) const
107  {
108  if (i0__ < 0) {
109  i0__ += grid_size_[0];
110  }
111  if (i1__ < 0) {
112  i1__ += grid_size_[1];
113  }
114  if (i2__ < 0) {
115  i2__ += grid_size_[2];
116  }
117 
118  return {i0__, i1__, i2__};
119  }
120 
121  inline int coord_by_gvec(int i__, int idim__) const
122  {
123  if (i__ < 0) {
124  i__ += grid_size_[idim__];
125  }
126  return i__;
127  }
128 
129  inline std::array<int, 3> gvec_by_coord(int x__, int y__, int z__) const
130  {
131  if (x__ > grid_limits_[0].second) {
132  x__ -= grid_size_[0];
133  }
134  if (y__ > grid_limits_[1].second) {
135  y__ -= grid_size_[1];
136  }
137  if (z__ > grid_limits_[2].second) {
138  z__ -= grid_size_[2];
139  }
140 
141  return {x__, y__, z__};
142  }
143 
144  inline int gvec_by_coord(int x__, int idim__) const
145  {
146  if (x__ > limits(idim__).second) {
147  x__ -= size(idim__);
148  }
149  return x__;
150  }
151 
152  /// Linear index inside FFT buffer by grid coordinates.
153  inline int index_by_coord(int x__, int y__, int z__) const
154  {
155  return (x__ + y__ * grid_size_[0] + z__ * grid_size_[0] * grid_size_[1]);
156  }
157 
158  /// Return linear index of a plane-wave harmonic with fractional coordinates (i0, i1, i2) inside FFT buffer.
159  inline int index_by_gvec(int i0__, int i1__, int i2__) const
160  {
161  auto coord = coord_by_gvec(i0__, i1__, i2__);
162  return index_by_coord(coord[0], coord[1], coord[2]);
163  }
164 };
