93  /// Split beta-projectors into chunks.
94  void split_in_chunks()
95  {
96  auto& uc = ctx_.unit_cell();
97  /* initial chunk size */
98  int chunk_size = std::min(uc.num_atoms(), 256);
99  /* maximum number of chunks */
100  int num_chunks = uc.num_atoms() / chunk_size + std::min(1, uc.num_atoms() % chunk_size);
101  /* final maximum chunk size */
102  chunk_size = uc.num_atoms() / num_chunks + std::min(1, uc.num_atoms() % num_chunks);
103 
104  int offset_in_beta_gk{0};
105  beta_chunks_ = std::vector<beta_chunk_t>(num_chunks);
106 
107  for (int ib = 0; ib < num_chunks; ib++) {
108  /* number of atoms in this chunk */
109  int na = std::min(uc.num_atoms(), (ib + 1) * chunk_size) - ib * chunk_size;
110  beta_chunks_[ib].num_atoms_ = na;
111  beta_chunks_[ib].desc_ = mdarray<int, 2>(4, na);
112  beta_chunks_[ib].atom_pos_ = mdarray<double, 2>(3, na);
113 
114  int num_beta{0};
115  for (int i = 0; i < na; i++) {
116  /* global index of atom by local index and chunk */
117  int ia = ib * chunk_size + i;
118  auto pos = uc.atom(ia).position();
119  auto& type = uc.atom(ia).type();
120  /* atom fractional coordinates */
121  for (int x: {0, 1, 2}) {
122  beta_chunks_[ib].atom_pos_(x, i) = pos[x];
123  }
124  /* number of beta functions for atom */
125  beta_chunks_[ib].desc_(beta_desc_idx::nbf, i) = type.mt_basis_size();
126  /* offset in beta_gk*/
127  beta_chunks_[ib].desc_(beta_desc_idx::offset, i) = num_beta;
128  /* offset in beta_gk_t */
129  beta_chunks_[ib].desc_(beta_desc_idx::offset_t, i) = type.offset_lo();
130  /* global index of atom */
131  beta_chunks_[ib].desc_(beta_desc_idx::ia, i) = ia;
132 
133  num_beta += type.mt_basis_size();
134  }
135  /* number of beta-projectors in this chunk */
136  beta_chunks_[ib].num_beta_ = num_beta;
137  beta_chunks_[ib].offset_ = offset_in_beta_gk;
138  offset_in_beta_gk += num_beta;
139 
140  if (ctx_.processing_unit() == GPU) {
141  beta_chunks_[ib].desc_.allocate(memory_t::device);
142  beta_chunks_[ib].desc_.copy<memory_t::host, memory_t::device>();
143 
144  beta_chunks_[ib].atom_pos_.allocate(memory_t::device);
145  beta_chunks_[ib].atom_pos_.copy<memory_t::host, memory_t::device>();
146  }
147  }
148 
149  max_num_beta_ = 0;
150  for (auto& e: beta_chunks_) {
151  max_num_beta_ = std::max(max_num_beta_, e.num_beta_);
152  }
153 
154  num_beta_t_ = 0;
155  for (int iat = 0; iat < uc.num_atom_types(); iat++) {
156  num_beta_t_ += uc.atom_type(iat).mt_lo_basis_size();
157  }
158  }

241  /// Calculate inner product between beta-projectors and wave-functions.
242  /** The following is computed: <beta|phi> */
243  template <typename T>
244  inline matrix<T> inner(int chunk__,
245  Wave_functions& phi__,
246  int ispn__,
247  int idx0__,
248  int n__)
249  {
250  PROFILE("sirius::Beta_projectors_base::inner");
251 
252  assert(num_gkvec_loc() == phi__.pw_coeffs(ispn__).num_rows_loc());
253 
254  int nbeta = chunk(chunk__).num_beta_;
255 
256  static_assert(std::is_same<T, double_complex>::value || std::is_same<T, double>::value, "wrong type");
257 
258  int tsz = std::is_same<T, double_complex>::value ? 2 : 1;
259 
260  auto& buf = beta_phi_shared(tsz * nbeta * n__, ctx_.dual_memory_t());
261 
262  matrix<T> beta_phi;
263 
264  switch (ctx_.processing_unit()) {
265  case CPU: {
266  beta_phi = matrix<T>(reinterpret_cast<T*>(buf.template at<CPU>()), nbeta, n__);
267  break;
268  }
269  case GPU: {
270  beta_phi = matrix<T>(reinterpret_cast<T*>(buf.template at<CPU>()), reinterpret_cast<T*>(buf.template at<GPU>()), nbeta, n__);
271  break;
272  }
273  }
274 
275  if (std::is_same<T, double_complex>::value) {
276  switch (ctx_.processing_unit()) {
277  case CPU: {
278  /* compute <beta|phi> */
279  linalg<CPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc(),
280  pw_coeffs_a().template at<CPU>(), num_gkvec_loc(),
281  phi__.pw_coeffs(ispn__).prime().at<CPU>(0, idx0__), phi__.pw_coeffs(ispn__).prime().ld(),
282  reinterpret_cast<double_complex*>(beta_phi.template at<CPU>()), nbeta);
283  break;
284  }
285  case GPU: {
286  #ifdef __GPU
287  linalg<GPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc(),
288  pw_coeffs_a().template at<GPU>(), num_gkvec_loc(),
289  phi__.pw_coeffs(ispn__).prime().at<GPU>(0, idx0__), phi__.pw_coeffs(ispn__).prime().ld(),
290  reinterpret_cast<double_complex*>(beta_phi.template at<GPU>()), nbeta);
291  beta_phi.template copy<memory_t::device, memory_t::host>();
292  #else
293  TERMINATE_NO_GPU
294  #endif
295  break;
296  }
297  }
298  }
299  if (std::is_same<T, double>::value) {
300  double a{2};
301  double a1{-1};
302  double b{0};
303 
304  switch (ctx_.processing_unit()) {
305  case CPU: {
306  /* compute <beta|phi> */
307  linalg<CPU>::gemm(2, 0, nbeta, n__, 2 * num_gkvec_loc(),
308  a,
309  reinterpret_cast<double*>(pw_coeffs_a().template at<CPU>()), 2 * num_gkvec_loc(),
310  reinterpret_cast<double*>(phi__.pw_coeffs(ispn__).prime().at<CPU>(0, idx0__)),
311  2 * phi__.pw_coeffs(ispn__).prime().ld(),
312  b,
313  reinterpret_cast<double*>(beta_phi.template at<CPU>()), nbeta);
314 
315  if (gkvec_.comm().rank() == 0) {
316  /* subtract one extra G=0 contribution */
317  linalg<CPU>::ger(nbeta, n__, a1,
318  reinterpret_cast<double*>(pw_coeffs_a().template at<CPU>()), 2 * num_gkvec_loc(),
319  reinterpret_cast<double*>(phi__.pw_coeffs(ispn__).prime().at<CPU>(0, idx0__)),
320  2 * phi__.pw_coeffs(ispn__).prime().ld(),
321  reinterpret_cast<double*>(beta_phi.template at<CPU>()), nbeta);
322  }
323  break;
324  }
325  case GPU: {
326  #ifdef __GPU
327  linalg<GPU>::gemm(2, 0, nbeta, n__, 2 * num_gkvec_loc(),
328  &a,
329  reinterpret_cast<double*>(pw_coeffs_a().template at<GPU>()), 2 * num_gkvec_loc(),
330  reinterpret_cast<double*>(phi__.pw_coeffs(ispn__).prime().at<GPU>(0, idx0__)),
331  2 * phi__.pw_coeffs(ispn__).prime().ld(),
332  &b,
333  reinterpret_cast<double*>(beta_phi.template at<GPU>()), nbeta);
334 
335  if (gkvec_.comm().rank() == 0) {
336  /* subtract one extra G=0 contribution */
337  linalg<GPU>::ger(nbeta, n__, &a1,
338  reinterpret_cast<double*>(pw_coeffs_a().template at<GPU>()), 2 * num_gkvec_loc(),
339  reinterpret_cast<double*>(phi__.pw_coeffs(ispn__).prime().template at<GPU>(0, idx0__)),
340  2 * phi__.pw_coeffs(ispn__).prime().ld(),
341  reinterpret_cast<double*>(beta_phi.template at<GPU>()), nbeta);
342  }
343  beta_phi.template copy<memory_t::device, memory_t::host>();
344  #else
345  TERMINATE_NO_GPU
346  #endif
347  break;
348  }
349  }
350  }
351 
352  gkvec_.comm().allreduce(beta_phi.template at<CPU>(), static_cast<int>(beta_phi.size()));
353 
354  if (ctx_.processing_unit() == GPU) {
355  beta_phi.template copy<memory_t::host, memory_t::device>();
356  }
357 
358  return std::move(beta_phi);
359  }

361  /// Generate beta-projectors for a chunk of atoms.
362  void generate(int ichunk__, int j__)
363  {
364  PROFILE("sirius::Beta_projectors_base::generate");
365 
366  auto& pw_coeffs = pw_coeffs_a();
367 
368  switch (ctx_.processing_unit()) {
369  case CPU: {
370  #pragma omp for
371  for (int i = 0; i < chunk(ichunk__).num_atoms_; i++) {
372  int ia = chunk(ichunk__).desc_(beta_desc_idx::ia, i);
373 
374  double phase = twopi * dot(gkvec_.vk(), ctx_.unit_cell().atom(ia).position());
375  double_complex phase_k = std::exp(double_complex(0.0, phase));
376 
377  std::vector<double_complex> phase_gk(num_gkvec_loc());
378  for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
379  auto G = gkvec_.gvec(igk_[igk_loc]);
380  /* total phase e^{i(G+k)r_{\alpha}} */
381  phase_gk[igk_loc] = std::conj(ctx_.gvec_phase_factor(G, ia) * phase_k);
382  }
383  for (int xi = 0; xi < chunk(ichunk__).desc_(beta_desc_idx::nbf, i); xi++) {
384  for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
385  pw_coeffs(igk_loc, chunk(ichunk__).desc_(beta_desc_idx::offset, i) + xi) =
386  pw_coeffs_t_[j__](igk_loc, chunk(ichunk__).desc_(beta_desc_idx::offset_t, i) + xi) * phase_gk[igk_loc];
387  }
388  }
389  }
390  break;
391  }
392  case GPU: {
393  #ifdef __GPU
394  auto& desc = chunk(ichunk__).desc_;
395  create_beta_gk_gpu(chunk(ichunk__).num_atoms_,
396  num_gkvec_loc(),
397  desc.template at<GPU>(),
398  pw_coeffs_t_[j__].template at<GPU>(),
399  gkvec_coord_.template at<GPU>(),
400  chunk(ichunk__).atom_pos_.template at<GPU>(),
401  pw_coeffs.template at<GPU>());
402  #endif
403  break;
404  }
405  }
406  }
