158  /// Calculate inner product between beta-projectors and wave-functions.
159  /** The following is computed: <beta|phi> */
160  template <typename T>
161  inline matrix<T> inner(int chunk__,
162  wave_functions& phi__,
163  int idx0__,
164  int n__)
165  {
166  PROFILE("sirius::Beta_projectors_base::inner");
167 
168  assert(num_gkvec_loc_ == phi__.pw_coeffs().num_rows_loc());
169 
170  auto& bp_chunks = ctx_.beta_projector_chunks();
171 
172  int nbeta = bp_chunks(chunk__).num_beta_;
173 
174  static_assert(std::is_same<T, double_complex>::value || std::is_same<T, double>::value, "wrong type");
175 
176  int tsz = std::is_same<T, double_complex>::value ? 2 : 1;
177 
178  auto& buf = beta_phi_shared(tsz * nbeta * n__, ctx_.dual_memory_t());
179 
180  matrix<T> beta_phi;
181 
182  switch (ctx_.processing_unit()) {
183  case CPU: {
184  beta_phi = matrix<T>(reinterpret_cast<T*>(buf.template at<CPU>()), nbeta, n__);
185  break;
186  }
187  case GPU: {
188  beta_phi = matrix<T>(reinterpret_cast<T*>(buf.template at<CPU>()), reinterpret_cast<T*>(buf.template at<GPU>()), nbeta, n__);
189  break;
190  }
191  }
192 
193  if (std::is_same<T, double_complex>::value) {
194  switch (ctx_.processing_unit()) {
195  case CPU: {
196  /* compute <beta|phi> */
197  linalg<CPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc_,
198  pw_coeffs_a().template at<CPU>(), num_gkvec_loc_,
199  phi__.pw_coeffs().prime().at<CPU>(0, idx0__), phi__.pw_coeffs().prime().ld(),
200  reinterpret_cast<double_complex*>(beta_phi.template at<CPU>()), nbeta);
201  break;
202  }
203  case GPU: {
204  #ifdef __GPU
205  linalg<GPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc_,
206  pw_coeffs_a().template at<GPU>(), num_gkvec_loc_,
207  phi__.pw_coeffs().prime().at<GPU>(0, idx0__), phi__.pw_coeffs().prime().ld(),
208  reinterpret_cast<double_complex*>(beta_phi.template at<GPU>()), nbeta);
209  beta_phi.template copy<memory_t::device, memory_t::host>();
210  #else
211  TERMINATE_NO_GPU
212  #endif
213  break;
214  }
215  }
216  }
217  if (std::is_same<T, double>::value) {
218  double a{2};
219  double a1{-1};
220  double b{0};
221 
222  switch (ctx_.processing_unit()) {
223  case CPU: {
224  /* compute <beta|phi> */
225  linalg<CPU>::gemm(2, 0, nbeta, n__, 2 * num_gkvec_loc_,
226  a,
227  reinterpret_cast<double*>(pw_coeffs_a().template at<CPU>()), 2 * num_gkvec_loc_,
228  reinterpret_cast<double*>(phi__.pw_coeffs().prime().at<CPU>(0, idx0__)), 2 * phi__.pw_coeffs().prime().ld(),
229  b,
230  reinterpret_cast<double*>(beta_phi.template at<CPU>()), nbeta);
231 
232  if (gkvec_.comm().rank() == 0) {
233  /* subtract one extra G=0 contribution */
234  linalg<CPU>::ger(nbeta, n__, a1,
235  reinterpret_cast<double*>(pw_coeffs_a().template at<CPU>()), 2 * num_gkvec_loc_,
236  reinterpret_cast<double*>(phi__.pw_coeffs().prime().at<CPU>(0, idx0__)), 2 * phi__.pw_coeffs().prime().ld(),
237  reinterpret_cast<double*>(beta_phi.template at<CPU>()), nbeta);
238  }
239  break;
240  }
241  case GPU: {
242  #ifdef __GPU
243  linalg<GPU>::gemm(2, 0, nbeta, n__, 2 * num_gkvec_loc_,
244  &a,
245  reinterpret_cast<double*>(pw_coeffs_a().template at<GPU>()), 2 * num_gkvec_loc_,
246  reinterpret_cast<double*>(phi__.pw_coeffs().prime().at<GPU>(0, idx0__)), 2 * phi__.pw_coeffs().prime().ld(),
247  &b,
248  reinterpret_cast<double*>(beta_phi.template at<GPU>()), nbeta);
249 
250  if (gkvec_.comm().rank() == 0) {
251  /* subtract one extra G=0 contribution */
252  linalg<GPU>::ger(nbeta, n__, &a1,
253  reinterpret_cast<double*>(pw_coeffs_a().template at<GPU>()), 2 * num_gkvec_loc_,
254  reinterpret_cast<double*>(phi__.pw_coeffs().prime().at<GPU>(0, idx0__)), 2 * phi__.pw_coeffs().prime().ld(),
255  reinterpret_cast<double*>(beta_phi.template at<GPU>()), nbeta);
256  }
257  beta_phi.template copy<memory_t::device, memory_t::host>();
258  #else
259  TERMINATE_NO_GPU
260  #endif
261  break;
262  }
263  }
264  }
265 
266  gkvec_.comm().allreduce(beta_phi.template at<CPU>(), static_cast<int>(beta_phi.size()));
267 
268  if (ctx_.processing_unit() == GPU) {
269  beta_phi.template copy<memory_t::host, memory_t::device>();
270  }
271 
272  return std::move(beta_phi);
273  }

275  /// Generate beta-projectors for a chunk of atoms.
276  void generate(int ichunk__, int j__)
277  {
278  PROFILE("sirius::Beta_projectors_base::generate");
279 
280  auto& bchunk = ctx_.beta_projector_chunks();
281 
282  auto& pw_coeffs = pw_coeffs_a();
283 
284  switch (ctx_.processing_unit()) {
285  case CPU: {
286  #pragma omp for
287  for (int i = 0; i < bchunk(ichunk__).num_atoms_; i++) {
288  int ia = bchunk(ichunk__).desc_(beta_desc_idx::ia, i);
289 
290  double phase = twopi * dot(gkvec_.vk(), ctx_.unit_cell().atom(ia).position());
291  double_complex phase_k = std::exp(double_complex(0.0, phase));
292 
293  std::vector<double_complex> phase_gk(num_gkvec_loc());
294  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
295  int igk = gkvec_.offset() + igk_loc;
296  auto G = gkvec_.gvec(igk);
297  /* total phase e^{i(G+k)r_{\alpha}} */
298  phase_gk[igk_loc] = std::conj(ctx_.gvec_phase_factor(G, ia) * phase_k);
299  }
300  for (int xi = 0; xi < bchunk(ichunk__).desc_(beta_desc_idx::nbf, i); xi++) {
301  for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
302  pw_coeffs(igk_loc, bchunk(ichunk__).desc_(beta_desc_idx::offset, i) + xi) =
303  pw_coeffs_t_[j__](igk_loc, bchunk(ichunk__).desc_(beta_desc_idx::offset_t, i) + xi) * phase_gk[igk_loc];
304  }
305  }
306  }
307  break;
308  }
309  case GPU: {
310  #ifdef __GPU
311  auto& desc = bchunk(ichunk__).desc_;
312  create_beta_gk_gpu(bchunk(ichunk__).num_atoms_,
313  num_gkvec_loc_,
314  desc.template at<GPU>(),
315  pw_coeffs_t_[j__].template at<GPU>(),
316  gkvec_coord_.template at<GPU>(),
317  bchunk(ichunk__).atom_pos_.template at<GPU>(),
318  pw_coeffs.template at<GPU>());
319  #endif
320  break;
321  }
322  }
323  }
