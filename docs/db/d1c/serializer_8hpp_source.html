32 /// Serialize and deserialize objects.
33 class serializer
34 {
35  private:
36  /// Position in the stream. This is used during unpacking.
37  size_t pos_{0};
38  /// Data stream is represendted as a sequence of characters.
39  std::vector<uint8_t> stream_;
40  public:
41 
42  /// Copy n bytes into a serialization stream.
43  void copyin(uint8_t const* ptr__, size_t nbytes__)
44  {
45  /* resize the array */
46  stream_.resize(stream_.size() + nbytes__);
47  /* copy from pointer */
48  std::memcpy(&stream_[stream_.size() - nbytes__], ptr__, nbytes__);
49  }
50 
51  /// Copy n bytes from the serialization stream.
52  /** When data is copied out, the position inside a stream is shifted to n bytes forward. */
53  void copyout(uint8_t* ptr__, size_t nbytes__)
54  {
55  std::memcpy(ptr__, &stream_[pos_], nbytes__);
56  pos_ += nbytes__;
57  }

59  void send_recv(Communicator const& comm__, int source__, int dest__)
60  {
61  if (source__ == dest__) {
62  return;
63  }
64 
65  size_t sz;
66 
67  Request r1, r2;
68 
69  int tag = Communicator::get_tag(source__, dest__);
70 
71  if (comm__.rank() == source__) {
72  sz = stream_.size();
73  assert(sz < std::numeric_limits<int>::max());
74  r1 = comm__.isend(&sz, 1, dest__, tag++);
75  r2 = comm__.isend(&stream_[0], (int)sz, dest__, tag++);
76  }
77 
78  if (comm__.rank() == dest__) {
79  comm__.recv(&sz, 1, source__, tag++);
80  stream_.resize(sz);
81  comm__.recv(&stream_[0], (int)sz, source__, tag++);
82  }
83 
84  if (comm__.rank() == source__) {
85  r1.wait();
86  r2.wait();
87  }
88  }
89 
90  std::vector<uint8_t> const& stream() const
91  {
92  return stream_;
93  }

128 /// Serialize multidimentional array.
129 template <typename T, int N>
130 void serialize(serializer& s__, mdarray<T, N> const& array__)
131 {
132  serialize(s__, array__.size());
133  if (array__.size() == 0) {
134  return;
135  }
136  for (int i = 0; i < N; i++) {
137  serialize(s__, array__.dim(i).begin());
138  serialize(s__, array__.dim(i).end());
139  }
140  s__.copyin(reinterpret_cast<uint8_t const*>(&array__[0]), sizeof(T) * array__.size());
141 }

143 /// Deserialize multidimentional array.
144 template <typename T, int N>
145 void deserialize(serializer& s__, mdarray<T, N>& array__)
146 {
147  size_t sz;
148  deserialize(s__, sz);
149  if (sz == 0) {
150  array__ = mdarray<T, N>();
151  return;
152  }
153  std::array<mdarray_index_descriptor, N> dims;
154  for (int i = 0; i < N; i++) {
155  mdarray_index_descriptor::index_type begin, end;
156  deserialize(s__, begin);
157  deserialize(s__, end);
158  dims[i] = mdarray_index_descriptor(begin, end);
159  }
160  array__ = mdarray<T, N>(dims);
161  s__.copyout(reinterpret_cast<uint8_t*>(&array__[0]), sizeof(T) * array__.size());
162 }
