33/// Serialize and deserialize objects.
34class serializer
35{
36 private:
37 /// Position in the stream. This is used during unpacking.
38 size_t pos_{0};
39 /// Data stream is represendted as a sequence of characters.
40 std::vector<uint8_t> stream_;
41 public:
42
43 /// Copy n bytes into a serialization stream.
44 void copyin(uint8_t const* ptr__, size_t nbytes__)
45 {
46 /* resize the array */
47 stream_.resize(stream_.size() + nbytes__);
48 /* copy from pointer */
49 std::memcpy(&stream_[stream_.size() - nbytes__], ptr__, nbytes__);
50 }
51
52 /// Copy n bytes from the serialization stream.
53 /** When data is copied out, the position inside a stream is shifted to n bytes forward. */
54 void copyout(uint8_t* ptr__, size_t nbytes__)
55 {
56 std::memcpy(ptr__, &stream_[pos_], nbytes__);
57 pos_ += nbytes__;
58 }
59
60 void send_recv(Communicator const& comm__, int source__, int dest__)
61 {
62 if (source__ == dest__) {
63 return;
64 }
65
66 size_t sz;
67
68 Request r1, r2;
69
70 int tag = Communicator::get_tag(source__, dest__);
71
72 if (comm__.rank() == source__) {
73 sz = stream_.size();
74 assert(sz < static_cast<size_t>(std::numeric_limits<int>::max()));
75 r1 = comm__.isend(&sz, 1, dest__, tag++);
76 r2 = comm__.isend(&stream_[0], (int)sz, dest__, tag++);
77 }
78
79 if (comm__.rank() == dest__) {
80 comm__.recv(&sz, 1, source__, tag++);
81 stream_.resize(sz);
82 comm__.recv(&stream_[0], (int)sz, source__, tag++);
83 }
84
85 if (comm__.rank() == source__) {
86 r1.wait();
87 r2.wait();
88 }
89 }
90
91 std::vector<uint8_t> const& stream() const
92 {
93 return stream_;
94 }
95};

97/// Serialize a single element.
98template <typename T>
99inline void serialize(serializer& s__, T var__)
100{

104/// Deserialize a single element.

111/// Serialize a vector.
112template <typename T>
113inline void serialize(serializer& s__, std::vector<T> const& vec__)
114{
115 serialize(s__, vec__.size());
116 s__.copyin(reinterpret_cast<uint8_t const*>(&vec__[0]), sizeof(T) * vec__.size());
117}

119/// Deserialize a vector.
120template <typename T>
121inline void deserialize(serializer& s__, std::vector<T>& vec__)
122{
123 size_t sz;
124 deserialize(s__, sz);

129/// Serialize multidimentional array.
130template <typename T, int N>
131void serialize(serializer& s__, mdarray<T, N> const& array__)
132{
133 serialize(s__, array__.size());
134 if (array__.size() == 0) {
135 return;
136 }
137 for (int i = 0; i < N; i++) {
138 serialize(s__, array__.dim(i).begin());
139 serialize(s__, array__.dim(i).end());
140 }
141 s__.copyin(reinterpret_cast<uint8_t const*>(&array__[0]), sizeof(T) * array__.size());
142}
143
144/// Deserialize multidimentional array.
145template <typename T, int N>
146void deserialize(serializer& s__, mdarray<T, N>& array__)
147{
148 size_t sz;
149 deserialize(s__, sz);
150 if (sz == 0) {
151 array__ = mdarray<T, N>();
152 return;
153 }
154 std::array<mdarray_index_descriptor, N> dims;
155 for (int i = 0; i < N; i++) {
156 mdarray_index_descriptor::index_type begin, end;
157 deserialize(s__, begin);
158 deserialize(s__, end);
159 dims[i] = mdarray_index_descriptor(begin, end);
160 }
161 array__ = mdarray<T, N>(dims);
162 s__.copyout(reinterpret_cast<uint8_t*>(&array__[0]), sizeof(T) * array__.size());
163}
