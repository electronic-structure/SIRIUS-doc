32 __global__ void create_beta_gk_gpu_kernel
33 (
34  int num_gkvec__,
35  int const* beta_desc__,
36  acc_complex_double_t const* beta_gk_t,
37  double const* gkvec,
38  double const* atom_pos,
39  acc_complex_double_t* beta_gk
40 )
41 {
42  int ia = blockIdx.y;
43  int igk = blockDim.x * blockIdx.x + threadIdx.x;
44 
45  int nbf = beta_desc__[array2D_offset(0, ia, 4)];
46  int offset_beta_gk = beta_desc__[array2D_offset(1, ia, 4)];
47  int offset_beta_gk_t = beta_desc__[array2D_offset(2, ia, 4)];
48 
49  if (igk < num_gkvec__) {
50  double p = 0;
51  for (int x = 0; x < 3; x++) {
52  p += atom_pos[array2D_offset(x, ia, 3)] * gkvec[array2D_offset(x, igk, 3)];
53  }
54  p *= twopi;
55 
56  double sinp = sin(p);
57  double cosp = cos(p);
58 
59  for (int xi = 0; xi < nbf; xi++) {
60  beta_gk[array2D_offset(igk, offset_beta_gk + xi, num_gkvec__)] =
61  accCmul(beta_gk_t[array2D_offset(igk, offset_beta_gk_t + xi, num_gkvec__)],
62  make_accDoubleComplex(cosp, -sinp));
63  }
64  }
65 }
