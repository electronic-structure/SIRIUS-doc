63  inline auto hubbard_F_coefficients() const
64  {
65  std::vector<double> F(4);
66  F[0] = Hubbard_U();
67 
68  switch (l) {
69  case 0: {
70  F[1] = Hubbard_J();
71  break;
72  }
73  case 1: {
74  F[1] = 5.0 * Hubbard_J();
75  break;
76  }
77  case 2: {
78  F[1] = 5.0 * Hubbard_J() + 31.5 * Hubbard_B();
79  F[2] = 9.0 * Hubbard_J() - 31.5 * Hubbard_B();
80  break;
81  }
82  case 3: {
83  F[1] = (225.0 / 54.0) * Hubbard_J() + (32175.0 / 42.0) * Hubbard_E2() + (2475.0 / 42.0) * Hubbard_E3();
84  F[2] = 11.0 * Hubbard_J() - (141570.0 / 77.0) * Hubbard_E2() + (4356.0 / 77.0) * Hubbard_E3();
85  F[3] = (7361.640 / 594.0) * Hubbard_J() + (36808.20 / 66.0) * Hubbard_E2() - 111.54 * Hubbard_E3();
86  break;
87  }
88  default: {
89  std::stringstream s;
90  s << "Hubbard correction not implemented for l > 3\n"
91  << " current l: " << l;
92  RTE_THROW(s);
93  break;
94  }
95  }
96  return F;
97  }

100  inline void calculate_ak_coefficients(sddk::mdarray<double, 5>& ak)
101  {
102  // compute the ak coefficients appearing in the general treatment of
103  // hubbard corrections. expression taken from Liechtenstein {\it et
104  // al}, PRB 52, R5467 (1995)
105 
106  // Note that for consistency, the ak are calculated with complex
107  // harmonics in the gaunt coefficients <R_lm|Y_l'm'|R_l''m''>.
108  // we need to keep it that way because of the hubbard potential.
109  // With a spherical one it does not really matter-
110  ak.zero();
111 
112  for (int m1 = -l; m1 <= l; m1++) {
113  for (int m2 = -l; m2 <= l; m2++) {
114  for (int m3 = -l; m3 <= l; m3++) {
115  for (int m4 = -l; m4 <= l; m4++) {
116  for (int k = 0; k < 2 * l; k += 2) {
117  double sum = 0.0;
118  for (int q = -k; q <= k; q++) {
119  sum += SHT::gaunt_rlm_ylm_rlm(l, k, l, m1, q, m2) *
120  SHT::gaunt_rlm_ylm_rlm(l, k, l, m3, q, m4);
121  }
122  /* according to PRB 52, R5467 it is 4 \pi/(2 k + 1) -> 4 \pi / (4 * k + 1) because
123  only a_{k=0} a_{k=2}, a_{k=4} are considered */
124  ak(k / 2, m1 + l, m2 + l, m3 + l, m4 + l) =
125  4.0 * sum * pi / static_cast<double>(2 * k + 1);
126  }
127  }
128  }
129  }
130  }
131  }

133  /// this function computes the matrix elements of the orbital part of
134  /// the electron-electron interactions. we effectively compute
135 
136  /// \f[ u(m,m'',m',m''') = \left<m,m''|V_{e-e}|m',m'''\right> \sum_k
137  /// a_k(m,m',m'',m''') F_k \f] where the F_k are calculated for real
138  /// spherical harmonics
139 
140  inline void compute_hubbard_matrix()
141  {
142  this->hubbard_matrix_ = sddk::mdarray<double, 4>(2 * l + 1, 2 * l + 1,
143  2 * l + 1, 2 * l + 1);
144  sddk::mdarray<double, 5> ak(l, 2 * l + 1, 2 * l + 1, 2 * l + 1, 2 * l + 1);
145  auto F = hubbard_F_coefficients();
146  calculate_ak_coefficients(ak);
147 
148  // the indices are rotated around
149 
150  // <m, m |vee| m'', m'''> = hubbard_matrix(m, m'', m', m''')
151  this->hubbard_matrix_.zero();
152  for (int m1 = 0; m1 < 2 * l + 1; m1++) {
153  for (int m2 = 0; m2 < 2 * l + 1; m2++) {
154  for (int m3 = 0; m3 < 2 * l + 1; m3++) {
155  for (int m4 = 0; m4 < 2 * l + 1; m4++) {
156  for (int k = 0; k < l; k++) {
157  this->hubbard_matrix(m1, m2, m3, m4) += ak(k, m1, m3, m2, m4) * F[k];
158  }
159  }
160  }
161  }
162  }
163  }

165  void initialize_hubbard_matrix()
166  {
167  sddk::mdarray<double, 5> ak(l, 2 * l + 1, 2 * l + 1, 2 * l + 1, 2 * l + 1);
168  auto F = hubbard_F_coefficients();
169  calculate_ak_coefficients(ak);
170 
171  this->hubbard_matrix_ = sddk::mdarray<double, 4>(2 * l + 1, 2 * l + 1, 2 * l + 1, 2 * l + 1);
172  // the indices are rotated around
173 
174  // <m, m |vee| m'', m'''> = hubbard_matrix(m, m'', m', m''')
175  this->hubbard_matrix_.zero();
176  for (int m1 = 0; m1 < 2 * l + 1; m1++) {
177  for (int m2 = 0; m2 < 2 * l + 1; m2++) {
178  for (int m3 = 0; m3 < 2 * l + 1; m3++) {
179  for (int m4 = 0; m4 < 2 * l + 1; m4++) {
180  for (int k = 0; k < l; k++) {
181  this->hubbard_matrix(m1, m2, m3, m4) += ak(k, m1, m3, m2, m4) * F[k];
182  }
183  }
184  }
185  }
