34 /** The following matching conditions must be fulfilled:
35  * \f[
36  * \frac{\partial^j}{\partial r^j} \sum_{L \nu} A_{L \nu}^{\bf k}({\bf G})u_{\ell \nu}(r)
37  * Y_{L}(\hat {\bf r}) \bigg|_{R^{MT}} = \frac{\partial^j}{\partial r^j} \frac{4 \pi}{\sqrt \Omega}
38  * e^{i{\bf (G+k)\tau}} \sum_{L}i^{\ell} j_{\ell}(|{\bf G+k}|r) Y_{L}^{*}(\widehat {\bf G+k}) Y_{L}(\hat {\bf r})
39  * \bigg|_{R^{MT}}
40  * \f]
41  * where \f$ L = \{ \ell, m \} \f$. Dropping sum over L we arrive to the following system of linear equations:
42  * \f[ \sum_{\nu} \frac{\partial^j u_{\ell \nu}(r)}{\partial r^j} \bigg|_{R^{MT}} A_{L \nu}^{\bf k}({\bf G})
43  * = \frac{4 \pi}{\sqrt \Omega} e^{i{\bf (G+k)\tau}} i^{\ell} \frac{\partial^j j_{\ell}(|{\bf G+k}|r)}{\partial r^j}
44  * \bigg|_{R^{MT}} Y_{L}^{*}(\widehat {\bf G+k})
45  * \f]
46  * The matching coefficients are then equal to:
47  * \f[
48  * A_{L \nu}^{\bf k}({\bf G}) = \sum_{j} \bigg[ \frac{\partial^j u_{\ell \nu}(r)}{\partial r^j} \bigg|_{R^{MT}}
49  * \bigg]_{\nu j}^{-1} \frac{\partial^j j_{\ell}(|{\bf G+k}|r)}{\partial r^j} \bigg|_{R^{MT}}
50  * \frac{4 \pi}{\sqrt \Omega} i^{\ell} e^{i{\bf (G+k)\tau}} Y_{L}^{*}(\widehat {\bf G+k})
51  * \f]
52  */

79  /// Generate matching coefficients for a specific \f$ \ell \f$ and order.
80  /** \param [in] ngk Number of G+k vectors.
81  * \param [in] phase_factors Phase factors of G+k vectors.
82  * \param [in] iat Index of atom type.
83  * \param [in] l Orbital quantum nuber.
84  * \param [in] lm Composite l,m index.
85  * \param [in] nu Order of radial function \f$ u_{\ell \nu}(r) \f$ for which coefficients are generated.
86  * \param [in] A Inverse matrix of radial derivatives.
87  * \param [out] alm Pointer to alm coefficients.
88  */
89  template <int N, bool conjugate>
90  inline void generate(int ngk, std::vector<double_complex> const& phase_factors__, int iat, int l, int lm, int nu,
91  matrix3d<double> const& A, double_complex* alm) const
92  {
93  double_complex zt;
94 
95  for (int igk = 0; igk < ngk; igk++) {
96  switch (N) {
97  case 1: {
98  zt = alm_b_(0, igk, l, iat) * A(0, 0);
99  break;
100  }
101  case 2: {
102  zt = alm_b_(0, igk, l, iat) * A(nu, 0) + alm_b_(1, igk, l, iat) * A(nu, 1);
103  break;
104  }
105  case 3: {
106  zt = alm_b_(0, igk, l, iat) * A(nu, 0) + alm_b_(1, igk, l, iat) * A(nu, 1) +
107  alm_b_(2, igk, l, iat) * A(nu, 2);
108  break;
109  }
110  }
111  if (conjugate) {
112  alm[igk] = std::conj(phase_factors__[igk] * zt) * gkvec_ylm_(igk, lm);
113  } else {
114  alm[igk] = phase_factors__[igk] * zt * std::conj(gkvec_ylm_(igk, lm));
115  }
116  }
117  }

119  public:
120  /// Constructor
121  Matching_coefficients(Unit_cell const& unit_cell__, int lmax_apw__, int num_gkvec__, std::vector<int>& igk__,
122  Gvec const& gkvec__)
123  : unit_cell_(unit_cell__)
124  , num_gkvec_(num_gkvec__)
125  , igk_(igk__)
126  , gkvec_(gkvec__)
127  , lmax_apw_(lmax_apw__)
128  {
129  int lmmax_apw = utils::lmmax(lmax_apw__);
130 
131  gkvec_ylm_ = mdarray<double_complex, 2>(num_gkvec_, lmmax_apw);
132 
133  gkvec_len_.resize(num_gkvec_);
134 
135  /* get length and Ylm harmonics of G+k vectors */
136  #pragma omp parallel
137  {
138  std::vector<double_complex> ylm(lmmax_apw);
139 
140  #pragma omp for
141  for (int i = 0; i < num_gkvec_; i++) {
142  auto gkvec_cart = gkvec_.gkvec_cart<index_domain_t::global>(igk_[i]);
143  /* get r, theta, phi */
144  auto vs = SHT::spherical_coordinates(gkvec_cart);
145 
146  /* get spherical harmonics */
147  sf::spherical_harmonics(lmax_apw__, vs[1], vs[2], &ylm[0]);
148  gkvec_len_[i] = vs[0];
149 
150  for (int lm = 0; lm < lmmax_apw; lm++) {
151  gkvec_ylm_(i, lm) = ylm[lm];
152  }
153  }
154  }
155 
156  alm_b_ = mdarray<double_complex, 4>(3, num_gkvec_, lmax_apw__ + 1, unit_cell_.num_atom_types());
157  alm_b_.zero();
158 
159  #pragma omp parallel
160  {
161  /* value and first two derivatives of spherical Bessel functions */
162  mdarray<double, 2> sbessel_mt(lmax_apw__ + 2, 3);
163 
164  #pragma omp for
165  for (int igk = 0; igk < num_gkvec_; igk++) {
166  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
167  double R = unit_cell_.atom_type(iat).mt_radius();
168 
169  double RGk = R * gkvec_len_[igk];
170 
171  /* compute values and first and second derivatives of the spherical Bessel functions
172  at the MT boundary */
173  gsl_sf_bessel_jl_array(lmax_apw__ + 1, RGk, &sbessel_mt(0, 0));
174 
175  /* Bessel function derivative: f_{{n}}^{{\prime}}(z)=-f_{{n+1}}(z)+(n/z)f_{{n}}(z)
176  *
177  * In[]:= FullSimplify[D[SphericalBesselJ[n,a*x],{x,1}]]
178  * Out[]= (n SphericalBesselJ[n,a x])/x-a SphericalBesselJ[1+n,a x]
179  *
180  * In[]:= FullSimplify[D[SphericalBesselJ[n,a*x],{x,2}]]
181  * Out[]= (((-1+n) n-a^2 x^2) SphericalBesselJ[n,a x]+2 a x SphericalBesselJ[1+n,a x])/x^2
182  */
183  for (int l = 0; l <= lmax_apw__; l++) { // TODO: move to sbessel class
184  sbessel_mt(l, 1) = -sbessel_mt(l + 1, 0) * gkvec_len_[igk] + (l / R) * sbessel_mt(l, 0);
185  sbessel_mt(l, 2) = 2 * gkvec_len_[igk] * sbessel_mt(l + 1, 0) / R +
186  ((l - 1) * l - std::pow(RGk, 2)) * sbessel_mt(l, 0) / std::pow(R, 2);
187  }
188 
189  for (int l = 0; l <= lmax_apw__; l++) {
190  double_complex z = std::pow(double_complex(0, 1), l);
191  double f = fourpi / std::sqrt(unit_cell_.omega());
192  alm_b_(0, igk, l, iat) = z * f * sbessel_mt(l, 0);
193  alm_b_(1, igk, l, iat) = z * f * sbessel_mt(l, 1);
194  alm_b_(2, igk, l, iat) = z * f * sbessel_mt(l, 2);
195  }
196  }
197  }
198  }
199  }

201  /// Generate plane-wave matching coefficients for the radial solutions of a given atom.
202  /** \param [in] atom Atom, for which matching coefficients are generated.
203  \param [out] alm Array of matching coefficients with dimension indices \f$ ({\bf G+k}, \xi) \f$.
204  */
205  template <bool conjugate>
206  void generate(Atom const& atom__, sddk::mdarray<double_complex, 2>& alm__) const
207  {
208  auto& type = atom__.type();
209 
210  assert(type.max_aw_order() <= 3);
211 
212  int iat = type.id();
213 
214  std::vector<double_complex> phase_factors(num_gkvec_);
215  for (int i = 0; i < num_gkvec_; i++) {
216  double phase = twopi * dot(gkvec_.gkvec(igk_[i]), atom__.position());
217  phase_factors[i] = std::exp(double_complex(0, phase));
218  }
219 
220  const double eps{0.1};
221  std::vector<matrix3d<double>> Al(lmax_apw_ + 1);
222  /* create inverse matrix of radial derivatives for all values of \ell */
223  for (int l = 0; l <= lmax_apw_; l++) {
224  /* order of augmentation for a given orbital quantum number */
225  int num_aw = static_cast<int>(type.aw_descriptor(l).size());
226  matrix3d<double> A;
227  /* create matrix of radial derivatives */
228  for (int order = 0; order < num_aw; order++) {
229  for (int dm = 0; dm < num_aw; dm++) {
230  A(dm, order) = atom__.symmetry_class().aw_surface_deriv(l, order, dm);
231  }
232  }
233 
234  /* invert matrix of radial derivatives */
235  switch (num_aw) {
236  case 1: {
237  if (unit_cell_.parameters().cfg().control().verification() >= 1) {
238  if (std::abs(A(0, 0)) < eps * (1.0 / std::sqrt(unit_cell_.omega()))) {
239  std::stringstream s;
240  s << "Ill defined plane wave matching problem for atom type " << iat << ", l = " << l
241  << std::endl
242  << " radial function value at the MT boundary : " << A(0, 0);
243  WARNING(s.str());
244  }
245  }
246 
247  A(0, 0) = 1.0 / A(0, 0);
248  break;
249  }
250  case 2: {
251  double det = A(0, 0) * A(1, 1) - A(0, 1) * A(1, 0);
252 
253  if (unit_cell_.parameters().cfg().control().verification() >= 1) {
254  if (std::abs(det) < eps * (1.0 / std::sqrt(unit_cell_.omega()))) {
255  std::stringstream s;
256  s << "Ill defined plane wave matching problem for atom type " << iat << ", l = " << l
257  << std::endl
258  << " radial function value at the MT boundary : " << A(0, 0);
259  WARNING(s.str());
260  }
261  }
262 
263  std::swap(A(0, 0), A(1, 1));
264  A(0, 0) /= det;
265  A(1, 1) /= det;
266  A(0, 1) = -A(0, 1) / det;
267  A(1, 0) = -A(1, 0) / det;
268  break;
269  }
270  case 3: {
271  A = inverse(A);
272  break;
273  }
274  }
275  Al[l] = A;
276  }
277 
278  for (int xi = 0; xi < type.mt_aw_basis_size(); xi++) {
279  int l = type.indexb(xi).l;
280  int lm = type.indexb(xi).lm;
281  int nu = type.indexb(xi).order;
282 
283  /* order of augmentation for a given orbital quantum number */
284  int num_aw = static_cast<int>(type.aw_descriptor(l).size());
285 
286  switch (num_aw) {
287  /* APW */
288  case 1: {
289  generate<1, conjugate>(num_gkvec_, phase_factors, iat, l, lm, nu, Al[l], &alm__(0, xi));
290  break;
291  }
292  /* LAPW */
293  case 2: {
294  generate<2, conjugate>(num_gkvec_, phase_factors, iat, l, lm, nu, Al[l], &alm__(0, xi));
295  break;
296  }
297  /* Super LAPW */
298  case 3: {
299  generate<3, conjugate>(num_gkvec_, phase_factors, iat, l, lm, nu, Al[l], &alm__(0, xi));
300  break;
301  }
302  default: {
303  TERMINATE("wrong order of augmented wave");
304  }
305  }
306  }
307  }
308 };
