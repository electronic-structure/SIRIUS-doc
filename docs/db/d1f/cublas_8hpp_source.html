35 inline void error_message(cublasStatus_t status)
36 {
37  switch (status) {
38  case CUBLAS_STATUS_NOT_INITIALIZED: {
39  printf("the library was not initialized\n");
40  break;
41  }
42  case CUBLAS_STATUS_INVALID_VALUE: {
43  printf("the parameters m,n,k<0\n");
44  break;
45  }
46  case CUBLAS_STATUS_ARCH_MISMATCH: {
47  printf("the device does not support double-precision\n");
48  break;
49  }
50  case CUBLAS_STATUS_EXECUTION_FAILED: {
51  printf("the function failed to launch on the GPU\n");
52  break;
53  }
54  default: {
55  printf("cublas status unknown");
56  }
57  }
58 }

181 inline void create_stream_handles()
182 {
183  acc::set_device();
184  CALL_CUBLAS(cublasCreate, (&null_stream_handle()));
185 
186  stream_handles() = std::vector<cublasHandle_t>(acc::num_streams());
187  for (int i = 0; i < acc::num_streams(); i++) {
188  CALL_CUBLAS(cublasCreate, (&stream_handles()[i]));
189 
190  CALL_CUBLAS(cublasSetStream, (stream_handles()[i], acc::stream(stream_id(i))));
191  }
192 }
193 
194 inline void destroy_stream_handles()
195 {
196  acc::set_device();
197  CALL_CUBLAS(cublasDestroy, (null_stream_handle()));
198  for (int i = 0; i < acc::num_streams(); i++) {
199  CALL_CUBLAS(cublasDestroy, (stream_handles()[i]));
200  }
201 }
