254 /// Initialize descriptors of the augmented-wave radial functions.
255 inline void init_aw_descriptors(int lmax)
256 {
257 assert(lmax >= -1);
258
259 if (lmax >= 0 && aw_default_l_.size() == 0) {
260 TERMINATE("default AW descriptor is empty");
261 }
262
263 aw_descriptors_.clear();
264 for (int l = 0; l <= lmax; l++) {
265 aw_descriptors_.push_back(aw_default_l_);
266 for (size_t ord = 0; ord < aw_descriptors_[l].size(); ord++) {
267 aw_descriptors_[l][ord].n = l + 1;
268 aw_descriptors_[l][ord].l = l;
269 }
270 }
271
272 for (size_t i = 0; i < aw_specific_l_.size(); i++) {
273 int l = aw_specific_l_[i][0].l;
274 if (l < lmax) {
275 aw_descriptors_[l] = aw_specific_l_[i];
276 }
277 }

369 /// Add augmented-wave descriptor.
370 inline void add_aw_descriptor(int n, int l, double enu, int dme, int auto_enu)
371 {
372 if (static_cast<int>(aw_descriptors_.size()) < (l + 1)) {
373 aw_descriptors_.resize(l + 1, radial_solution_descriptor_set());
374 }
375
376 radial_solution_descriptor rsd;
377
378 rsd.n = n;
379 if (n == -1) {
380 /* default principal quantum number value for any l */
381 rsd.n = l + 1;
382 for (int ist = 0; ist < num_atomic_levels(); ist++) {
383 /* take next level after the core */
384 if (atomic_level(ist).core && atomic_level(ist).l == l) {
385 rsd.n = atomic_level(ist).n + 1;
386 }
387 }
388 }
389
390 rsd.l = l;
391 rsd.dme = dme;
392 rsd.enu = enu;
393 rsd.auto_enu = auto_enu;
394 aw_descriptors_[l].push_back(rsd);
395 }
396
397 /// Add local orbital descriptor
398 inline void add_lo_descriptor(int ilo, int n, int l, double enu, int dme, int auto_enu)
399 {
400 if ((int)lo_descriptors_.size() == ilo) {
401 lo_descriptors_.push_back(local_orbital_descriptor());
402 lo_descriptors_[ilo].l = l;
403 } else {
404 if (l != lo_descriptors_[ilo].l) {
405 std::stringstream s;
406 s << "wrong angular quantum number" << std::endl
407 << "atom type id: " << id() << " (" << symbol_ << ")" << std::endl
408 << "idxlo: " << ilo << std::endl
409 << "n: " << l << std::endl
410 << "l: " << n << std::endl
411 << "expected l: " << lo_descriptors_[ilo].l << std::endl;
412 TERMINATE(s);
413 }
414 }
415
416 radial_solution_descriptor rsd;
417
418 rsd.n = n;
419 if (n == -1) {
420 /* default value for any l */
421 rsd.n = l + 1;
422 for (int ist = 0; ist < num_atomic_levels(); ist++) {
423 if (atomic_level(ist).core && atomic_level(ist).l == l) {
424 /* take next level after the core */
425 rsd.n = atomic_level(ist).n + 1;
426 }
427 }
428 }
429
430 rsd.l = l;
431 rsd.dme = dme;
432 rsd.enu = enu;
433 rsd.auto_enu = auto_enu;
434 lo_descriptors_[ilo].rsd_set.push_back(rsd);
435 }

443 /// Add atomic radial function to the list.
444 inline void add_ps_atomic_wf(int n__, sirius::experimental::angular_momentum am__,
445 std::vector<double> f__, double occ__ = 0.0)
446 {
447 Spline<double> rwf(radial_grid_, f__);
448 auto d = std::sqrt(inner(rwf, rwf, 0, radial_grid_.num_points()));
449 if (d < 1e-4) {
450 std::stringstream s;
451 s << "small norm (" << d << ") of radial atomic pseudo wave-function for n=" << n__
452 << " and j=" << am__.j();
453 RTE_THROW(s);
454 }
455
456 ps_atomic_wfs_.emplace_back(n__, am__, occ__, std::move(rwf));
457 }

476 /// Add a radial function of beta-projector to a list of functions.
477 /** This is the only allowed way to add beta projectors. */
478 inline void add_beta_radial_function(int l__, std::vector<double> beta__)
479 {
480 if (augment_) {
481 TERMINATE("can't add more beta projectors");
482 }
483 Spline<double> s(radial_grid_, beta__);
484 beta_radial_functions_.push_back(std::make_pair(l__, std::move(s)));
485
486 local_orbital_descriptor lod;
487 lod.l = std::abs(l__);
488
489 /* for spin orbit coupling; we can always do that there is
490 no insidence on the rest when calculations exclude SO */
491 if (l__ < 0) {
492 lod.total_angular_momentum = lod.l - 0.5;
493 } else {
494 lod.total_angular_momentum = lod.l + 0.5;
495 }
496 /* add local orbital descriptor for the current beta-projector */
497 lo_descriptors_.push_back(lod);
498 }

506 /// Maximum orbital quantum number between all beta-projector radial functions.
507 inline int lmax_beta() const
508 {
509 int lmax{-1};
510
511 /* need to take |l| since the total angular momentum is encoded in the sign of l */
512 for (auto& e: beta_radial_functions_) {
513 lmax = std::max(lmax, std::abs(e.first));
514 }
515 return lmax;
516 }

525 /// Add radial function of the augmentation charge.
526 /** Radial functions of beta projectors must be added already. Their total number will be used to
527 deterimine the storage size for the radial functions of the augmented charge. */
528 inline void add_q_radial_function(int idxrf1__, int idxrf2__, int l__, std::vector<double> qrf__)
529 {
530 /* sanity check */
531 if (l__ > 2 * lmax_beta()) {
532 std::stringstream s;
533 s << "wrong l for Q radial functions of atom type " << label_ << std::endl
534 << "current l: " << l__ << std::endl
535 << "lmax_beta: " << lmax_beta() << std::endl
536 << "maximum allowed l: " << 2 * lmax_beta();
537
538 RTE_THROW(s);
539 }
540
541 if (!augment_) {
542 /* once we add a Q-radial function, we need to augment the charge */
543 augment_ = true;
544 /* number of radial beta-functions */
545 int nbrf = num_beta_radial_functions();
546 q_radial_functions_l_ = mdarray<Spline<double>, 2>(nbrf * (nbrf + 1) / 2, 2 * lmax_beta() + 1);
547
548 for (int l = 0; l <= 2 * lmax_beta(); l++) {
549 for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
550 q_radial_functions_l_(idx, l) = Spline<double>(radial_grid_);
551 }
552 }
553 }
554
555 int ijv = utils::packed_index(idxrf1__, idxrf2__);
556 q_radial_functions_l_(ijv, l__) = Spline<double>(radial_grid_, qrf__);
557 }
