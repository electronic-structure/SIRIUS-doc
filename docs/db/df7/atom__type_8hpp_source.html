262  /// Initialize descriptors of the augmented-wave radial functions.
263  inline void init_aw_descriptors(int lmax)
264  {
265  assert(lmax >= -1);
266 
267  if (lmax >= 0 && aw_default_l_.size() == 0) {
268  TERMINATE("default AW descriptor is empty");
269  }
270 
271  aw_descriptors_.clear();
272  for (int l = 0; l <= lmax; l++) {
273  aw_descriptors_.push_back(aw_default_l_);
274  for (size_t ord = 0; ord < aw_descriptors_[l].size(); ord++) {
275  aw_descriptors_[l][ord].n = l + 1;
276  aw_descriptors_[l][ord].l = l;
277  }
278  }
279 
280  for (size_t i = 0; i < aw_specific_l_.size(); i++) {
281  int l = aw_specific_l_[i][0].l;
282  if (l < lmax) {
283  aw_descriptors_[l] = aw_specific_l_[i];
284  }
285  }
286  }

377  /// Add augmented-wave descriptor.
378  inline void add_aw_descriptor(int n, int l, double enu, int dme, int auto_enu)
379  {
380  if (static_cast<int>(aw_descriptors_.size()) < (l + 1)) {
381  aw_descriptors_.resize(l + 1, radial_solution_descriptor_set());
382  }
383 
384  radial_solution_descriptor rsd;
385 
386  rsd.n = n;
387  if (n == -1) {
388  /* default principal quantum number value for any l */
389  rsd.n = l + 1;
390  for (int ist = 0; ist < num_atomic_levels(); ist++) {
391  /* take next level after the core */
392  if (atomic_level(ist).core && atomic_level(ist).l == l) {
393  rsd.n = atomic_level(ist).n + 1;
394  }
395  }
396  }
397 
398  rsd.l = l;
399  rsd.dme = dme;
400  rsd.enu = enu;
401  rsd.auto_enu = auto_enu;
402  aw_descriptors_[l].push_back(rsd);
403  }

405  /// Add local orbital descriptor
406  inline void add_lo_descriptor(int ilo, int n, int l, double enu, int dme, int auto_enu)
407  {
408  if ((int)lo_descriptors_.size() == ilo) {
409  lo_descriptors_.push_back(local_orbital_descriptor());
410  lo_descriptors_[ilo].l = l;
411  } else {
412  if (l != lo_descriptors_[ilo].l) {
413  std::stringstream s;
414  s << "wrong angular quantum number" << std::endl
415  << "atom type id: " << id() << " (" << symbol_ << ")" << std::endl
416  << "idxlo: " << ilo << std::endl
417  << "n: " << l << std::endl
418  << "l: " << n << std::endl
419  << "expected l: " << lo_descriptors_[ilo].l << std::endl;
420  TERMINATE(s);
421  }
422  }
423 
424  radial_solution_descriptor rsd;
425 
426  rsd.n = n;
427  if (n == -1) {
428  /* default value for any l */
429  rsd.n = l + 1;
430  for (int ist = 0; ist < num_atomic_levels(); ist++) {
431  if (atomic_level(ist).core && atomic_level(ist).l == l) {
432  /* take next level after the core */
433  rsd.n = atomic_level(ist).n + 1;
434  }
435  }
436  }
437 
438  rsd.l = l;
439  rsd.dme = dme;
440  rsd.enu = enu;
441  rsd.auto_enu = auto_enu;
442  lo_descriptors_[ilo].rsd_set.push_back(rsd);
443  }

475  /// Add a radial function of beta-projector to a list of functions.
476  /** This is the only allowed way to add beta projectors. */
477  inline void add_beta_radial_function(int l__, std::vector<double> beta__)
478  {
479  if (augment_) {
480  TERMINATE("can't add more beta projectors");
481  }
482  Spline<double> s(radial_grid_, beta__);
483  beta_radial_functions_.push_back(std::make_pair(l__, std::move(s)));
484 
485  local_orbital_descriptor lod;
486  lod.l = std::abs(l__);
487 
488  /* for spin orbit coupling; we can always do that there is
489  no insidence on the rest when calculations exclude SO */
490  if (l__ < 0) {
491  lod.total_angular_momentum = lod.l - 0.5;
492  } else {
493  lod.total_angular_momentum = lod.l + 0.5;
494  }
495  /* add local orbital descriptor for the current beta-projector */
496  lo_descriptors_.push_back(lod);
497  }

505  /// Maximum orbital quantum number between all beta-projector radial functions.
506  inline int lmax_beta() const

524  /// Add radial function of the augmentation charge.
525  /** Radial functions of beta projectors must be added already. Their total number will be used to
526  deterimine the storage size for the radial functions of the augmented charge. */
527  inline void add_q_radial_function(int idxrf1__, int idxrf2__, int l__, std::vector<double> qrf__)
528  {
529  /* sanity check */
530  if (l__ > 2 * lmax_beta()) {
531  std::stringstream s;
532  s << "wrong l for Q radial functions of atom type " << label_ << std::endl
533  << "current l: " << l__ << std::endl
534  << "lmax_beta: " << lmax_beta() << std::endl
535  << "maximum allowed l: " << 2 * lmax_beta();
536 
537  TERMINATE(s);
538  }
539 
540  if (!augment_) {
541  /* once we add a Q-radial function, we need to augment the charge */
542  augment_ = true;
543  /* number of radial beta-functions */
544  int nbrf = num_beta_radial_functions();
545  q_radial_functions_l_ = mdarray<Spline<double>, 2>(nbrf * (nbrf + 1) / 2, 2 * lmax_beta() + 1);
546 
547  for (int l = 0; l <= 2 * lmax_beta(); l++) {
548  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
549  q_radial_functions_l_(idx, l) = Spline<double>(radial_grid_);
550  }
551  }
552  }
553 
554  int ijv = utils::packed_index(idxrf1__, idxrf2__);
555  q_radial_functions_l_(ijv, l__) = Spline<double>(radial_grid_, qrf__);
556  }
