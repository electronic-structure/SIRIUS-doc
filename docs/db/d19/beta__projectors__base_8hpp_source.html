99  /// Split beta-projectors into chunks.
100  void split_in_chunks()
101  {
102  auto& uc = ctx_.unit_cell();
103  /* initial chunk size */
104  int chunk_size = std::min(uc.num_atoms(), ctx_.control().beta_chunk_size_);
105  /* maximum number of chunks */
106  int num_chunks = uc.num_atoms() / chunk_size + std::min(1, uc.num_atoms() % chunk_size);
107  /* final maximum chunk size */
108  chunk_size = uc.num_atoms() / num_chunks + std::min(1, uc.num_atoms() % num_chunks);
109 
110  int offset_in_beta_gk{0};
111  beta_chunks_ = std::vector<beta_chunk_t>(num_chunks);
112 
113  for (int ib = 0; ib < num_chunks; ib++) {
114  /* number of atoms in this chunk */
115  int na = std::min(uc.num_atoms(), (ib + 1) * chunk_size) - ib * chunk_size;
116  beta_chunks_[ib].num_atoms_ = na;
117  beta_chunks_[ib].desc_ = mdarray<int, 2>(4, na);
118  beta_chunks_[ib].atom_pos_ = mdarray<double, 2>(3, na);
119 
120  int num_beta{0};
121  for (int i = 0; i < na; i++) {
122  /* global index of atom by local index and chunk */
123  int ia = ib * chunk_size + i;
124  auto pos = uc.atom(ia).position();
125  auto& type = uc.atom(ia).type();
126  /* atom fractional coordinates */
127  for (int x: {0, 1, 2}) {
128  beta_chunks_[ib].atom_pos_(x, i) = pos[x];
129  }
130  /* number of beta functions for atom */
131  beta_chunks_[ib].desc_(beta_desc_idx::nbf, i) = type.mt_basis_size();
132  /* offset in beta_gk*/
133  beta_chunks_[ib].desc_(beta_desc_idx::offset, i) = num_beta;
134  /* offset in beta_gk_t */
135  beta_chunks_[ib].desc_(beta_desc_idx::offset_t, i) = type.offset_lo();
136  /* global index of atom */
137  beta_chunks_[ib].desc_(beta_desc_idx::ia, i) = ia;
138 
139  num_beta += type.mt_basis_size();
140  }
141  /* number of beta-projectors in this chunk */
142  beta_chunks_[ib].num_beta_ = num_beta;
143  beta_chunks_[ib].offset_ = offset_in_beta_gk;
144  offset_in_beta_gk += num_beta;
145 
146  if (ctx_.processing_unit() == device_t::GPU) {
147  beta_chunks_[ib].desc_.allocate(memory_t::device).copy_to(memory_t::device);
148  beta_chunks_[ib].atom_pos_.allocate(memory_t::device).copy_to(memory_t::device);
149  }
150  }
151 
152  max_num_beta_ = 0;
153  for (auto& e: beta_chunks_) {
154  max_num_beta_ = std::max(max_num_beta_, e.num_beta_);
155  }
156 
157  num_beta_t_ = 0;
158  for (int iat = 0; iat < uc.num_atom_types(); iat++) {
159  num_beta_t_ += uc.atom_type(iat).mt_lo_basis_size();
160  }
161  }

231  /// Calculate inner product between beta-projectors and wave-functions.
232  /** The following is computed: <beta|phi> */
233  template <typename T>
234  inline matrix<T> inner(int chunk__,
235  Wave_functions& phi__,
236  int ispn__,
237  int idx0__,
238  int n__)
239  {
240  PROFILE("sirius::Beta_projectors_base::inner");
241 
242  assert(num_gkvec_loc() == phi__.pw_coeffs(ispn__).num_rows_loc());
243 
244  int nbeta = chunk(chunk__).num_beta_;
245 
246  matrix<T> beta_phi(ctx_.mem_pool(ctx_.host_memory_t()), nbeta, n__);
247 
248  /* location of the beta-projectors is always on the memory of the processing unit being used */
249  T* pw_coeffs_a_ptr{nullptr};
250  switch (ctx_.processing_unit()) {
251  case device_t::CPU: {
252  pw_coeffs_a_ptr = reinterpret_cast<T*>(pw_coeffs_a().at(memory_t::host));
253  break;
254  }
255  case device_t::GPU: {
256  beta_phi.allocate(ctx_.mem_pool(memory_t::device));
257  pw_coeffs_a_ptr = reinterpret_cast<T*>(pw_coeffs_a().at(memory_t::device));
258  break;
259  }
260  }
261 
262  local_inner_aux<T>(pw_coeffs_a_ptr, nbeta, phi__, ispn__, idx0__, n__, beta_phi);
263 
264  /* copy to host in MPI sequential or parallel case */
265  if (is_device_memory(ctx_.preferred_memory_t())) {
266  beta_phi.copy_to(memory_t::host);
267  }
268 
269  /* in parallel case do a reduction */
270  if (gkvec_.comm().size() > 1) {
271  utils::timer t1("sirius::Beta_projectors_base::inner|comm");
272  /* MPI reduction on the host */
273  gkvec_.comm().allreduce(beta_phi.at(memory_t::host), static_cast<int>(beta_phi.size()));
274  }
275 
276  switch (ctx_.processing_unit()) {
277  case device_t::GPU: {
278  /* copy back to device */
279  if (gkvec_.comm().size() > 1 || is_host_memory(ctx_.preferred_memory_t())) {
280  beta_phi.copy_to(memory_t::device);
281  }
282  break;
283  }
284  case device_t::CPU: break;
285  }
286 
287  return std::move(beta_phi);
288  }

290  /// Generate beta-projectors for a chunk of atoms.
291  /** Beta-projectors are always generated and stored in the memory of a processing unit.
292  *
293  * \param [in] ichunk Index of a chunk of atoms for which beta-projectors are generated.
294  * \param [in] j index of the component (up to 9 components are used for the strain derivative)
295  */
296  void generate(int ichunk__, int j__)
297  {
298  PROFILE("sirius::Beta_projectors_base::generate");
299 
300  switch (ctx_.processing_unit()) {
301  case device_t::CPU: {
302  #pragma omp for
303  for (int i = 0; i < chunk(ichunk__).num_atoms_; i++) {
304  int ia = chunk(ichunk__).desc_(beta_desc_idx::ia, i);
305 
306  double phase = twopi * dot(gkvec_.vk(), ctx_.unit_cell().atom(ia).position());
307  double_complex phase_k = std::exp(double_complex(0.0, phase));
308 
309  std::vector<double_complex> phase_gk(num_gkvec_loc());
310  for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
311  auto G = gkvec_.gvec(igk_[igk_loc]);
312  /* total phase e^{i(G+k)r_{\alpha}} */
313  phase_gk[igk_loc] = std::conj(ctx_.gvec_phase_factor(G, ia) * phase_k);
314  }
315  for (int xi = 0; xi < chunk(ichunk__).desc_(beta_desc_idx::nbf, i); xi++) {
316  for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
317  pw_coeffs_a_(igk_loc, chunk(ichunk__).desc_(beta_desc_idx::offset, i) + xi) =
318  pw_coeffs_t_(igk_loc, chunk(ichunk__).desc_(beta_desc_idx::offset_t, i) + xi, j__) * phase_gk[igk_loc];
319  }
320  }
321  }
322  break;
323  }
324  case device_t::GPU: {
325 #ifdef __GPU
326  auto& desc = chunk(ichunk__).desc_;
327  create_beta_gk_gpu(chunk(ichunk__).num_atoms_,
328  num_gkvec_loc(),
329  desc.at(memory_t::device),
330  pw_coeffs_t_.at(memory_t::device, 0, 0, j__),
331  gkvec_coord_.at(memory_t::device),
332  chunk(ichunk__).atom_pos_.at(memory_t::device),
333  pw_coeffs_a().at(memory_t::device));
334 #endif
335  /* wave-functions are on CPU but the beta-projectors are on GPU */
336  if (gkvec_.comm().rank() == 0 && is_host_memory(ctx_.preferred_memory_t())) {
337  /* make beta-projectors for G=0 on the CPU */
338  #pragma omp for schedule(static)
339  for (int i = 0; i < chunk(ichunk__).num_atoms_; i++) {
340  for (int xi = 0; xi < chunk(ichunk__).desc_(beta_desc_idx::nbf, i); xi++) {
341  pw_coeffs_a_g0_(chunk(ichunk__).desc_(beta_desc_idx::offset, i) + xi) =
342  pw_coeffs_t_(0, chunk(ichunk__).desc_(beta_desc_idx::offset_t, i) + xi, j__);
343  }
344  }
345  }
346  break;
347  }
348  }
349  }

351  void prepare()

366  pw_coeffs_a_g0_.allocate(ctx_.mem_pool(memory_t::device));
367  break;
368  }
369  }
370 
371  if (ctx_.processing_unit() == device_t::GPU && reallocate_pw_coeffs_t_on_gpu_) {
372  pw_coeffs_t_.allocate(memory_t::device).copy_to(memory_t::device);
373  }
374  }

408 template<>
409 inline void Beta_projectors_base::local_inner_aux<double_complex>(double_complex* beta_pw_coeffs_a_ptr__, int nbeta__,
410  Wave_functions& phi__, int ispn__, int idx0__, int n__,
411  matrix<double_complex>& beta_phi__) const
412 {
413  utils::timer t1("sirius::Beta_projectors_base::local_inner_aux");
414  linalg2(ctx_.blas_linalg_t()).gemm('C', 'N', nbeta__, n__, num_gkvec_loc(),
415  &linalg_const<double_complex>::one(),
416  beta_pw_coeffs_a_ptr__,
417  num_gkvec_loc(),
418  phi__.pw_coeffs(ispn__).prime().at(phi__.preferred_memory_t(), 0, idx0__),
419  phi__.pw_coeffs(ispn__).prime().ld(),
420  &linalg_const<double_complex>::zero(),
421  beta_phi__.at(ctx_.preferred_memory_t()), beta_phi__.ld());
422 
423  auto pp = utils::get_env<int>("SIRIUS_PRINT_PERFORMANCE");
424  if (pp && gkvec_.comm().rank() == 0) {
425 #ifdef __GPU
426  if (ctx_.blas_linalg_t() == linalg_t::cublas) {
427  acc::sync_stream(stream_id(-1));
428  }
429 #endif
430  double t = t1.stop();
431  printf("Beta_projectors_base::local_inner performance: %12.6f GFlops [m,n,k=%i %i %i, time=%f (sec)]\n",
432  8e-9 * nbeta__ * n__ * num_gkvec_loc() / t, nbeta__, n__, num_gkvec_loc(), t);
433  }
434 }

436 template<>
437 inline void Beta_projectors_base::local_inner_aux<double>(double* beta_pw_coeffs_a_ptr__, int nbeta__,
438  Wave_functions& phi__, int ispn__, int idx0__, int n__,
439  matrix<double>& beta_phi__) const
440 {
441  linalg2(ctx_.blas_linalg_t()).gemm('C', 'N', nbeta__, n__, 2 * num_gkvec_loc(),
442  &linalg_const<double>::two(),
443  beta_pw_coeffs_a_ptr__,
444  2 * num_gkvec_loc(),
445  reinterpret_cast<double const*>(phi__.pw_coeffs(ispn__).prime().at(phi__.preferred_memory_t(), 0, idx0__)),
446  2 * phi__.pw_coeffs(ispn__).prime().ld(),
447  &linalg_const<double>::zero(),
448  beta_phi__.at(ctx_.preferred_memory_t()), beta_phi__.ld());
449 
450  /* rank 0 has to do some extra work for Gamma-point case */
451  if (gkvec_.comm().rank() == 0) {
452  int incx{2 * num_gkvec_loc()};
453  linalg_t la{linalg_t::none};
454  /* both wave-functions and beta-projectors are on GPU */
455  if (is_device_memory(ctx_.preferred_memory_t())) {
456  la = linalg_t::cublas;
457  } else { /* wave-functions are on CPU but the beta-projectors are in the memory of main device */
458  la = linalg_t::blas;
459  switch (ctx_.processing_unit()) {
460  case device_t::GPU: {
461  beta_pw_coeffs_a_ptr__ = reinterpret_cast<double*>(const_cast<double_complex*>(&pw_coeffs_a_g0_(0)));
462  incx = 1;
463  break;
464  }
465  case device_t::CPU: break;
466  }
467  }
468  linalg2(la).ger(nbeta__, n__,
469  &linalg_const<double>::m_one(),
470  beta_pw_coeffs_a_ptr__, incx,
471  reinterpret_cast<double*>(phi__.pw_coeffs(ispn__).prime().at(ctx_.preferred_memory_t(), 0, idx0__)),
472  2 * phi__.pw_coeffs(ispn__).prime().ld(),
473  beta_phi__.at(ctx_.preferred_memory_t()), beta_phi__.ld());
474  }
475 }
