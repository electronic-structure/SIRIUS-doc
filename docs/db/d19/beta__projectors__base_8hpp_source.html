51/// Named index of a descriptor of beta-projectors. The same order is used by the GPU kernel.
52enum class beta_desc_idx
53{
54 /// Number of beta-projector functions for this atom.
55 nbf = 0,
56 /// Offset of beta-projectors in this chunk.
57 offset = 1,
58 /// Offset of beta-projectors in the array for atom types.
59 offset_t = 2,
60 /// Global index of atom.
61 ia = 3
62};

118 template<typename F, std::enable_if_t<std::is_same<std::complex<T>, F>::value, bool> = true>
119 void local_inner_aux(F* beta_pw_coeffs_a_ptr__, int nbeta__, Wave_functions<T>& phi__,
120 int ispn__, int idx0__, int n__, matrix<F>& beta_phi__) const
121 {
122 auto pp = utils::get_env<int>("SIRIUS_PRINT_PERFORMANCE");
123 if (pp && gkvec_.comm().rank() == 0) {
124 PROFILE_START("sirius::Beta_projectors_base::local_inner_aux");
125 }
126
127 const auto t1 = std::chrono::high_resolution_clock::now();
128 linalg(ctx_.blas_linalg_t())
129 .gemm('C', 'N', nbeta__, n__, num_gkvec_loc(), &linalg_const<F>::one(), beta_pw_coeffs_a_ptr__,
130 num_gkvec_loc(), phi__.pw_coeffs(ispn__).prime().at(phi__.preferred_memory_t(), 0, idx0__),
131 phi__.pw_coeffs(ispn__).prime().ld(), &linalg_const<F>::zero(),
132 beta_phi__.at(ctx_.preferred_memory_t()), beta_phi__.ld());
133
134 if (pp && gkvec_.comm().rank() == 0) {
135#ifdef SIRIUS_GPU
136 if (ctx_.blas_linalg_t() == linalg_t::gpublas) {
137 acc::sync_stream(stream_id(-1));
138 }
139#endif
140 std::chrono::duration<double> t = std::chrono::high_resolution_clock::now() - t1;
141 PROFILE_STOP("sirius::Beta_projectors_base::local_inner_aux");
142 std::printf(
143 "Beta_projectors_base::local_inner performance: %12.6f GFlops [m,n,k=%i %i %i, time=%f (sec)]\n",
144 8e-9 * nbeta__ * n__ * num_gkvec_loc() / t.count(), nbeta__, n__, num_gkvec_loc(), t.count());
145 }
146 }

148 template<typename F, std::enable_if_t<std::is_same<T, F>::value, bool> = true>
149 void local_inner_aux(F* beta_pw_coeffs_a_ptr__, int nbeta__, Wave_functions<T>& phi__,
150 int ispn__, int idx0__, int n__, matrix<F>& beta_phi__) const
151 {
152 linalg(ctx_.blas_linalg_t())
153 .gemm('C', 'N', nbeta__, n__, 2 * num_gkvec_loc(), &linalg_const<F>::two(), beta_pw_coeffs_a_ptr__,
154 2 * num_gkvec_loc(),
155 reinterpret_cast<F const*>(phi__.pw_coeffs(ispn__).prime().at(phi__.preferred_memory_t(), 0, idx0__)),
156 2 * phi__.pw_coeffs(ispn__).prime().ld(), &linalg_const<F>::zero(),
157 beta_phi__.at(ctx_.preferred_memory_t()), beta_phi__.ld());
158
159 /* rank 0 has to do some extra work for Gamma-point case */
160 if (gkvec_.comm().rank() == 0) {
161 int incx{2 * num_gkvec_loc()};
162 linalg_t la{linalg_t::none};
163 /* both wave-functions and beta-projectors are on GPU */
164 if (is_device_memory(ctx_.preferred_memory_t())) {
165 la = linalg_t::gpublas;
166 } else { /* wave-functions are on CPU but the beta-projectors are in the memory of main device */
167 la = linalg_t::blas;
168 switch (ctx_.processing_unit()) {
169 case device_t::GPU: {
170 beta_pw_coeffs_a_ptr__ =
171 reinterpret_cast<F*>(const_cast<std::complex<T>*>(&pw_coeffs_a_g0_(0)));
172 incx = 2;
173 break;
174 }
175 case device_t::CPU:
176 break;
177 }
178 }
179 linalg(la).ger(
180 nbeta__, n__, &linalg_const<F>::m_one(), beta_pw_coeffs_a_ptr__, incx,
181 reinterpret_cast<F*>(phi__.pw_coeffs(ispn__).prime().at(phi__.preferred_memory_t(), 0, idx0__)),
182 2 * phi__.pw_coeffs(ispn__).prime().ld(), beta_phi__.at(ctx_.preferred_memory_t()), beta_phi__.ld());
183 }
184 }
