60 void
61 Hamiltonian0::apply_hmt_to_apw(Atom const& atom__, int ngv__, sddk::mdarray<double_complex, 2>& alm__,
62  sddk::mdarray<double_complex, 2>& halm__) const
63 {
64  auto& type = atom__.type();
65 
66  // TODO: this is k-independent and can in principle be precomputed together with radial integrals if memory is
67  // available
68  // TODO: for spin-collinear case hmt is Hermitian; compute upper triangular part and use zhemm
69  mdarray<double_complex, 2> hmt(type.mt_aw_basis_size(), type.mt_aw_basis_size());
70  /* compute the muffin-tin Hamiltonian */
71  for (int j2 = 0; j2 < type.mt_aw_basis_size(); j2++) {
72  int lm2 = type.indexb(j2).lm;
73  int idxrf2 = type.indexb(j2).idxrf;
74  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
75  int lm1 = type.indexb(j1).lm;
76  int idxrf1 = type.indexb(j1).idxrf;
77  hmt(j1, j2) = atom__.radial_integrals_sum_L3<sblock>(idxrf1, idxrf2,
78  gaunt_coefs_->gaunt_vector(lm1, lm2));
79  }
80  }
81  linalg(linalg_t::blas).gemm('N', 'T', ngv__, type.mt_aw_basis_size(), type.mt_aw_basis_size(),
82  &linalg_const<double_complex>::one(), alm__.at(memory_t::host), alm__.ld(),
83  hmt.at(memory_t::host), hmt.ld(), &linalg_const<double_complex>::zero(),
84  halm__.at(memory_t::host), halm__.ld());
85 }

87 void
88 Hamiltonian0::add_o1mt_to_apw(Atom const& atom__, int num_gkvec__, sddk::mdarray<double_complex, 2>& alm__) const
89 {
90  // TODO: optimize for the loop layout using blocks of G-vectors
91  auto& type = atom__.type();
92  std::vector<double_complex> alm(type.mt_aw_basis_size());
93  std::vector<double_complex> oalm(type.mt_aw_basis_size());
94  for (int ig = 0; ig < num_gkvec__; ig++) {
95  for (int j = 0; j < type.mt_aw_basis_size(); j++) {
96  alm[j] = oalm[j] = alm__(ig, j);
97  }
98  for (int j = 0; j < type.mt_aw_basis_size(); j++) {
99  int l = type.indexb(j).l;
100  int lm = type.indexb(j).lm;
101  int idxrf = type.indexb(j).idxrf;
102  for (int order = 0; order < type.aw_order(l); order++) {
103  int j1 = type.indexb().index_by_lm_order(lm, order);
104  int idxrf1 = type.indexr().index_by_l_order(l, order);
105  oalm[j] += atom__.symmetry_class().o1_radial_integral(idxrf, idxrf1) * alm[j1];
106  }
107  }
108  for (int j = 0; j < type.mt_aw_basis_size(); j++) {
109  alm__(ig, j) = oalm[j];
110  }
111  }

114 void
115 Hamiltonian0::apply_bmt(sddk::Wave_functions& psi__, std::vector<sddk::Wave_functions>& bpsi__) const
116 {
117  mdarray<double_complex, 3> zm(unit_cell_.max_mt_basis_size(), unit_cell_.max_mt_basis_size(), ctx_.num_mag_dims());
118 
119  for (int ialoc = 0; ialoc < psi__.spl_num_atoms().local_size(); ialoc++) {
120  int ia = psi__.spl_num_atoms()[ialoc];
121  auto& atom = unit_cell_.atom(ia);
122  int offset = psi__.offset_mt_coeffs(ialoc);
123  int mt_basis_size = atom.type().mt_basis_size();
124 
125  zm.zero();
126 
127  /* only upper triangular part of zm is computed because it is a hermitian matrix */
128  #pragma omp parallel for default(shared)
129  for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
130  int lm2 = atom.type().indexb(xi2).lm;
131  int idxrf2 = atom.type().indexb(xi2).idxrf;
132 
133  for (int i = 0; i < ctx_.num_mag_dims(); i++) {
134  for (int xi1 = 0; xi1 <= xi2; xi1++) {
135  int lm1 = atom.type().indexb(xi1).lm;
136  int idxrf1 = atom.type().indexb(xi1).idxrf;
137 
138  zm(xi1, xi2, i) = gaunt_coefs_->sum_L3_gaunt(lm1, lm2, atom.b_radial_integrals(idxrf1, idxrf2, i));
139  }
140  }
141  }
142  /* compute bwf = B_z*|wf_j> */
143  linalg(linalg_t::blas).hemm('L', 'U', mt_basis_size, ctx_.num_fv_states(),
144  &linalg_const<double_complex>::one(),
145  zm.at(memory_t::host), zm.ld(),
146  psi__.mt_coeffs(0).prime().at(memory_t::host, offset, 0),
147  psi__.mt_coeffs(0).prime().ld(),
148  &linalg_const<double_complex>::zero(),
149  bpsi__[0].mt_coeffs(0).prime().at(memory_t::host, offset, 0),
150  bpsi__[0].mt_coeffs(0).prime().ld());
151 
152  /* compute bwf = (B_x - iB_y)|wf_j> */
153  if (bpsi__.size() == 3) {
154  /* reuse first (z) component of zm matrix to store (B_x - iB_y) */
155  for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
156  for (int xi1 = 0; xi1 <= xi2; xi1++) {
157  zm(xi1, xi2, 0) = zm(xi1, xi2, 1) - double_complex(0, 1) * zm(xi1, xi2, 2);
158  }
159 
160  /* remember: zm for x,y,z, components of magnetic field is hermitian and we computed
161  * only the upper triangular part */
162  for (int xi1 = xi2 + 1; xi1 < mt_basis_size; xi1++) {
163  zm(xi1, xi2, 0) = std::conj(zm(xi2, xi1, 1)) - double_complex(0, 1) * std::conj(zm(xi2, xi1, 2));
164  }
165  }
166 
167  linalg(linalg_t::blas).gemm('N', 'N', mt_basis_size, ctx_.num_fv_states(), mt_basis_size,
168  &linalg_const<double_complex>::one(),
169  zm.at(memory_t::host), zm.ld(),
170  psi__.mt_coeffs(0).prime().at(memory_t::host, offset, 0), psi__.mt_coeffs(0).prime().ld(),
171  &linalg_const<double_complex>::zero(),
172  bpsi__[2].mt_coeffs(0).prime().at(memory_t::host, offset, 0), bpsi__[2].mt_coeffs(0).prime().ld());
173  }
174  }

178 Hamiltonian0::apply_so_correction(sddk::Wave_functions& psi__, std::vector<sddk::Wave_functions>& hpsi__) const
179 {
180  PROFILE("sirius::Hamiltonian0::apply_so_correction");
181 
182  for (int ialoc = 0; ialoc < psi__.spl_num_atoms().local_size(); ialoc++) {
183  int ia = psi__.spl_num_atoms()[ialoc];
184  auto& atom = unit_cell_.atom(ia);
185  int offset = psi__.offset_mt_coeffs(ialoc);
186 
187  for (int l = 0; l <= ctx_.lmax_apw(); l++) {
188  /* number of radial functions for this l */
189  int nrf = atom.type().indexr().num_rf(l);
190 
191  for (int order1 = 0; order1 < nrf; order1++) {
192  for (int order2 = 0; order2 < nrf; order2++) {
193  double sori = atom.symmetry_class().so_radial_integral(l, order1, order2);
194 
195  for (int m = -l; m <= l; m++) {
196  int idx1 = atom.type().indexb_by_l_m_order(l, m, order1);
197  int idx2 = atom.type().indexb_by_l_m_order(l, m, order2);
198  int idx3 = (m + l != 0) ? atom.type().indexb_by_l_m_order(l, m - 1, order2) : 0;
199  // int idx4 = (m - l != 0) ? atom.type().indexb_by_l_m_order(l, m + 1, order2) : 0;
200 
201  for (int ist = 0; ist < ctx_.num_fv_states(); ist++) {
202  double_complex z1 = psi__.mt_coeffs(0).prime(offset + idx2, ist) * double(m) * sori;
203  /* u-u part */
204  hpsi__[0].mt_coeffs(0).prime(offset + idx1, ist) += z1;
205  /* d-d part */
206  hpsi__[1].mt_coeffs(0).prime(offset + idx1, ist) -= z1;
207  /* apply L_{-} operator; u-d part */
208  if (m + l) {

210  psi__.mt_coeffs(0).prime(offset + idx3, ist) * sori *

212  }

217  // fv_states__.mt_coeffs(0).prime().at(memory_t::host, offset + idx4, ist) *
