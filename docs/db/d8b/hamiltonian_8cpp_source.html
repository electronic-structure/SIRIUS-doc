56template <typename T> template <spin_block_t sblock>
57void
58Hamiltonian0<T>::apply_hmt_to_apw(Atom const& atom__, int ngv__, sddk::mdarray<std::complex<T>, 2>& alm__,
59 sddk::mdarray<std::complex<T>, 2>& halm__) const
60{
61 auto& type = atom__.type();
62
63 // TODO: this is k-independent and can in principle be precomputed together with radial integrals if memory is
64 // available
65 // TODO: for spin-collinear case hmt is Hermitian; compute upper triangular part and use zhemm
66 mdarray<std::complex<T>, 2> hmt(type.mt_aw_basis_size(), type.mt_aw_basis_size());
67 /* compute the muffin-tin Hamiltonian */
68 for (int j2 = 0; j2 < type.mt_aw_basis_size(); j2++) {
69 int lm2 = type.indexb(j2).lm;
70 int idxrf2 = type.indexb(j2).idxrf;
71 for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
72 int lm1 = type.indexb(j1).lm;
73 int idxrf1 = type.indexb(j1).idxrf;
74 hmt(j1, j2) = atom__.radial_integrals_sum_L3<sblock>(idxrf1, idxrf2,
75 potential().gaunt_coefs().gaunt_vector(lm1, lm2));
76 }
77 }
78 linalg(linalg_t::blas)
79 .gemm('N', 'T', ngv__, type.mt_aw_basis_size(), type.mt_aw_basis_size(), &linalg_const<std::complex<T>>::one(),
80 alm__.at(memory_t::host), alm__.ld(), hmt.at(memory_t::host), hmt.ld(),
81 &linalg_const<std::complex<T>>::zero(), halm__.at(memory_t::host), halm__.ld());
82}

84template <typename T>
85void
86Hamiltonian0<T>::add_o1mt_to_apw(Atom const& atom__, int num_gkvec__, sddk::mdarray<std::complex<T>, 2>& alm__) const
87{
88 // TODO: optimize for the loop layout using blocks of G-vectors
89 auto& type = atom__.type();
90 std::vector<std::complex<T>> alm(type.mt_aw_basis_size());
91 std::vector<std::complex<T>> oalm(type.mt_aw_basis_size());
92 for (int ig = 0; ig < num_gkvec__; ig++) {
93 for (int j = 0; j < type.mt_aw_basis_size(); j++) {
94 alm[j] = oalm[j] = alm__(ig, j);
95 }
96 for (int j = 0; j < type.mt_aw_basis_size(); j++) {
97 int l = type.indexb(j).l;
98 int lm = type.indexb(j).lm;
99 int idxrf = type.indexb(j).idxrf;
100 for (int order = 0; order < type.aw_order(l); order++) {
101 int j1 = type.indexb().index_by_lm_order(lm, order);
102 int idxrf1 = type.indexr().index_by_l_order(l, order);
103 oalm[j] += static_cast<const T>(atom__.symmetry_class().o1_radial_integral(idxrf, idxrf1)) * alm[j1];
104 }
105 }
106 for (int j = 0; j < type.mt_aw_basis_size(); j++) {
107 alm__(ig, j) = oalm[j];
108 }
109 }
110}

112template <typename T>
113void
114Hamiltonian0<T>::apply_bmt(sddk::Wave_functions<T>& psi__, std::vector<sddk::Wave_functions<T>>& bpsi__) const
115{
116 mdarray<std::complex<T>, 3> zm(unit_cell_.max_mt_basis_size(), unit_cell_.max_mt_basis_size(), ctx_.num_mag_dims());
117
118 for (int ialoc = 0; ialoc < psi__.spl_num_atoms().local_size(); ialoc++) {
119 int ia = psi__.spl_num_atoms()[ialoc];
120 auto& atom = unit_cell_.atom(ia);
121 int offset = psi__.offset_mt_coeffs(ialoc);
122 int mt_basis_size = atom.type().mt_basis_size();
123
124 zm.zero();
125
126 /* only upper triangular part of zm is computed because it is a hermitian matrix */
127 #pragma omp parallel for default(shared)
128 for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
129 int lm2 = atom.type().indexb(xi2).lm;
130 int idxrf2 = atom.type().indexb(xi2).idxrf;
131
132 for (int i = 0; i < ctx_.num_mag_dims(); i++) {
133 for (int xi1 = 0; xi1 <= xi2; xi1++) {
134 int lm1 = atom.type().indexb(xi1).lm;
135 int idxrf1 = atom.type().indexb(xi1).idxrf;
136
137 zm(xi1, xi2, i) = this->potential().gaunt_coefs().sum_L3_gaunt(lm1, lm2, atom.b_radial_integrals(idxrf1, idxrf2, i));
138 }
139 }
140 }
141 /* compute bwf = B_z*|wf_j> */
142 linalg(linalg_t::blas).hemm(
143 'L', 'U', mt_basis_size, ctx_.num_fv_states(), &linalg_const<std::complex<T>>::one(),
144 zm.at(memory_t::host), zm.ld(), psi__.mt_coeffs(0).prime().at(memory_t::host, offset, 0),
145 psi__.mt_coeffs(0).prime().ld(), &linalg_const<std::complex<T>>::zero(),
146 bpsi__[0].mt_coeffs(0).prime().at(memory_t::host, offset, 0), bpsi__[0].mt_coeffs(0).prime().ld());
147
148 /* compute bwf = (B_x - iB_y)|wf_j> */
149 if (bpsi__.size() == 3) {
150 /* reuse first (z) component of zm matrix to store (B_x - iB_y) */
151 for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
152 for (int xi1 = 0; xi1 <= xi2; xi1++) {
153 zm(xi1, xi2, 0) = zm(xi1, xi2, 1) - std::complex<T>(0, 1) * zm(xi1, xi2, 2);
154 }
155
156 /* remember: zm for x,y,z, components of magnetic field is hermitian and we computed
157 * only the upper triangular part */
158 for (int xi1 = xi2 + 1; xi1 < mt_basis_size; xi1++) {
159 zm(xi1, xi2, 0) = std::conj(zm(xi2, xi1, 1)) - std::complex<T>(0, 1) * std::conj(zm(xi2, xi1, 2));
160 }
161 }
162
163 linalg(linalg_t::blas).gemm(
164 'N', 'N', mt_basis_size, ctx_.num_fv_states(), mt_basis_size, &linalg_const<std::complex<T>>::one(),
165 zm.at(memory_t::host), zm.ld(), psi__.mt_coeffs(0).prime().at(memory_t::host, offset, 0),
166 psi__.mt_coeffs(0).prime().ld(), &linalg_const<std::complex<T>>::zero(),
167 bpsi__[2].mt_coeffs(0).prime().at(memory_t::host, offset, 0), bpsi__[2].mt_coeffs(0).prime().ld());
168 }
169 }

172template <typename T>
173void
174Hamiltonian0<T>::apply_so_correction(sddk::Wave_functions<T>& psi__, std::vector<sddk::Wave_functions<T>>& hpsi__) const
175{
176 PROFILE("sirius::Hamiltonian0::apply_so_correction");
177
178 for (int ialoc = 0; ialoc < psi__.spl_num_atoms().local_size(); ialoc++) {
179 int ia = psi__.spl_num_atoms()[ialoc];
180 auto& atom = unit_cell_.atom(ia);
181 int offset = psi__.offset_mt_coeffs(ialoc);
182
183 for (int l = 0; l <= ctx_.lmax_apw(); l++) {
184 /* number of radial functions for this l */
185 int nrf = atom.type().indexr().num_rf(l);
186
187 for (int order1 = 0; order1 < nrf; order1++) {
188 for (int order2 = 0; order2 < nrf; order2++) {
189 T sori = atom.symmetry_class().so_radial_integral(l, order1, order2);
190
191 for (int m = -l; m <= l; m++) {
192 int idx1 = atom.type().indexb_by_l_m_order(l, m, order1);
193 int idx2 = atom.type().indexb_by_l_m_order(l, m, order2);
194 int idx3 = (m + l != 0) ? atom.type().indexb_by_l_m_order(l, m - 1, order2) : 0;
195 // int idx4 = (m - l != 0) ? atom.type().indexb_by_l_m_order(l, m + 1, order2) : 0;
196
197 for (int ist = 0; ist < ctx_.num_fv_states(); ist++) {
198 std::complex<T> z1 = psi__.mt_coeffs(0).prime(offset + idx2, ist) * T(m) * sori;
199 /* u-u part */
200 hpsi__[0].mt_coeffs(0).prime(offset + idx1, ist) += z1;
201 /* d-d part */
202 hpsi__[1].mt_coeffs(0).prime(offset + idx1, ist) -= z1;
203 /* apply L_{-} operator; u-d part */
204 if (m + l) {
205 hpsi__[2].mt_coeffs(0).prime(offset + idx1, ist) +=
206 psi__.mt_coeffs(0).prime(offset + idx3, ist) * sori *
207 std::sqrt(T(l * (l + 1) - m * (m - 1)));
208 }
209 /* for the d-u part */
210 ///* apply L_{+} operator */
211 // if (m - l) {
212 // hpsi[3].mt_coeffs(0).prime().at(memory_t::host, offset + idx1, ist) +=
213 // fv_states__.mt_coeffs(0).prime().at(memory_t::host, offset + idx4, ist) *
214 // sori * std::sqrt(double(l * (l + 1) - m * (m + 1)));
215 //}
216 }
217 }
218 }
219 }
220 }
