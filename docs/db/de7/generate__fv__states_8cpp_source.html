29 void K_point::generate_fv_states()
30 {
31  PROFILE("sirius::K_point::generate_fv_states");
32 
33  if (!ctx_.full_potential()) {
34  return;
35  }
36 
37  mdarray<double_complex, 2> alm(num_gkvec_loc(), unit_cell_.max_mt_aw_basis_size(), memory_t::host);
38  mdarray<double_complex, 2> tmp(unit_cell_.max_mt_aw_basis_size(), ctx_.num_fv_states());
39 
40  if (ctx_.processing_unit() == device_t::GPU) {
41  fv_eigen_vectors_slab().pw_coeffs(0).allocate(memory_t::device);
42  fv_eigen_vectors_slab().pw_coeffs(0).copy_to(memory_t::device, 0, ctx_.num_fv_states());
43  alm.allocate(memory_t::device);
44  tmp.allocate(memory_t::device);
45  }
46 
47  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
48  auto location = fv_eigen_vectors_slab().spl_num_atoms().location(ia);
49  /* number of alm coefficients for atom */
50  int mt_aw_size = unit_cell_.atom(ia).mt_aw_basis_size();
51  int mt_lo_size = unit_cell_.atom(ia).mt_lo_basis_size();
52  /* generate matching coefficients for all G-vectors */
53  alm_coeffs_loc_->generate<false>(unit_cell_.atom(ia), alm);
54 
55  double_complex* tmp_ptr_gpu{nullptr};
56 
57  auto la = linalg_t::none;
58  auto mt = memory_t::none;
59  switch (ctx_.processing_unit()) {
60  case device_t::CPU: {
61  la = linalg_t::blas;
62  mt = memory_t::host;
63  break;
64  }
65  case device_t::GPU: {
66  alm.copy_to(memory_t::device, 0, mt_aw_size * num_gkvec_loc());
67  la = linalg_t::gpublas;
68  mt = memory_t::device;
69  tmp_ptr_gpu = tmp.at(memory_t::device);
70  break;
71  }
72  }
73 
74  mdarray<double_complex, 2> tmp1(tmp.at(memory_t::host), tmp_ptr_gpu, mt_aw_size, ctx_.num_fv_states());
75 
76  /* compute F(lm, i) = A(lm, G)^{T} * evec(G, i) for a single atom */
77  linalg(la).gemm('T', 'N', mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(),
78  &linalg_const<double_complex>::one(), alm.at(mt), alm.ld(),
79  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(mt),
80  fv_eigen_vectors_slab().pw_coeffs(0).prime().ld(),
81  &linalg_const<double_complex>::zero(), tmp1.at(mt), tmp1.ld());
82 
83  switch (ctx_.processing_unit()) {
84  case device_t::CPU: {
85  break;
86  }
87  case device_t::GPU: {
88  tmp1.copy_to(memory_t::host);
89  break;
90  }
91  }
92 
93  comm_.reduce(tmp1.at(memory_t::host), static_cast<int>(tmp1.size()), location.rank);
94 
95 #ifdef __PRINT_OBJECT_CHECKSUM
96  auto z1 = tmp1.checksum();
97  DUMP("checksum(tmp1): %18.10f %18.10f", std::real(z1), std::imag(z1));
98 #endif
99 
100  if (location.rank == comm_.rank()) {
101  int offset1 = fv_states().offset_mt_coeffs(location.local_index);
102  int offset2 = fv_eigen_vectors_slab().offset_mt_coeffs(location.local_index);
103  for (int i = 0; i < ctx_.num_fv_states(); i++) {
104  /* aw block */
105  std::memcpy(fv_states().mt_coeffs(0).prime().at(memory_t::host, offset1, i),
106  tmp1.at(memory_t::host, 0, i), mt_aw_size * sizeof(double_complex));
107  /* lo block */
108  if (mt_lo_size) {
109  std::memcpy(fv_states().mt_coeffs(0).prime().at(memory_t::host, offset1 + mt_aw_size, i),
110  fv_eigen_vectors_slab().mt_coeffs(0).prime().at(memory_t::host, offset2, i),
111  mt_lo_size * sizeof(double_complex));
112  }
113  }
114  }
115  }
116 
117  #pragma omp parallel for
118  for (int i = 0; i < ctx_.num_fv_states(); i++) {
119  /* G+k block */
120  std::memcpy(fv_states().pw_coeffs(0).prime().at(memory_t::host, 0, i),
121  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::host, 0, i),
122  num_gkvec_loc() * sizeof(double_complex));
123  }
124 
125  if (ctx_.processing_unit() == device_t::GPU) {
126  fv_eigen_vectors_slab().pw_coeffs(0).deallocate(memory_t::device);
127  }
128 }
129 } // namespace sirius
