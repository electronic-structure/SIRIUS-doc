29 void K_point::generate_fv_states()
30 {
31  PROFILE("sirius::K_point::generate_fv_states");
32 
33  if (!ctx_.full_potential()) {
34  return;
35  }
36 
37  mdarray<double_complex, 2> alm(num_gkvec_loc(), unit_cell_.max_mt_aw_basis_size(), memory_t::host);
38  mdarray<double_complex, 2> tmp(unit_cell_.max_mt_aw_basis_size(), ctx_.num_fv_states());
39 
40  if (ctx_.processing_unit() == device_t::GPU) {
41  fv_eigen_vectors_slab().pw_coeffs(0).allocate(memory_t::device);
42  fv_eigen_vectors_slab().pw_coeffs(0).copy_to(memory_t::device, 0, ctx_.num_fv_states());
43  alm.allocate(memory_t::device);
44  tmp.allocate(memory_t::device);
45  }
46 
47  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
48  auto location = fv_eigen_vectors_slab().spl_num_atoms().location(ia);
49  /* number of alm coefficients for atom */
50  int mt_aw_size = unit_cell_.atom(ia).mt_aw_basis_size();
51  int mt_lo_size = unit_cell_.atom(ia).mt_lo_basis_size();
52  /* generate matching coefficients for all G-vectors */
53  alm_coeffs_loc_->generate<false>(unit_cell_.atom(ia), alm);
54 
55  double_complex* tmp_ptr_gpu = (ctx_.processing_unit() == device_t::GPU) ? tmp.at(memory_t::device) : nullptr;
56  mdarray<double_complex, 2> tmp1(tmp.at(memory_t::host), tmp_ptr_gpu, mt_aw_size, ctx_.num_fv_states());
57 
58  /* compute F(lm, i) = A(lm, G)^{T} * evec(G, i) for a single atom */
59  if (ctx_.processing_unit() == device_t::CPU) {
60  linalg<device_t::CPU>::gemm(1, 0, mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(), alm.at(memory_t::host),
61  alm.ld(), fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::host),
62  fv_eigen_vectors_slab().pw_coeffs(0).prime().ld(), tmp1.at(memory_t::host),
63  tmp1.ld());
64  }
65 #ifdef __GPU
66  if (ctx_.processing_unit() == device_t::GPU) {
67  alm.copy_to(memory_t::device, 0, mt_aw_size * num_gkvec_loc());
68  linalg<device_t::GPU>::gemm(
69  1, 0, mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(), alm.at(memory_t::device), alm.ld(),
70  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::device),
71  fv_eigen_vectors_slab().pw_coeffs(0).prime().ld(), tmp1.at(memory_t::device), tmp1.ld());
72  tmp1.copy_to(memory_t::host);
73  }
74 #endif
75 
76  comm_.reduce(tmp1.at(memory_t::host), static_cast<int>(tmp1.size()), location.rank);
77 
78 #ifdef __PRINT_OBJECT_CHECKSUM
79  auto z1 = tmp1.checksum();
80  DUMP("checksum(tmp1): %18.10f %18.10f", std::real(z1), std::imag(z1));
81 #endif
82 
83  if (location.rank == comm_.rank()) {
84  int offset1 = fv_states().offset_mt_coeffs(location.local_index);
85  int offset2 = fv_eigen_vectors_slab().offset_mt_coeffs(location.local_index);
86  for (int i = 0; i < ctx_.num_fv_states(); i++) {
87  /* aw block */
88  std::memcpy(fv_states().mt_coeffs(0).prime().at(memory_t::host, offset1, i),
89  tmp1.at(memory_t::host, 0, i), mt_aw_size * sizeof(double_complex));
90  /* lo block */
91  if (mt_lo_size) {
92  std::memcpy(fv_states().mt_coeffs(0).prime().at(memory_t::host, offset1 + mt_aw_size, i),
93  fv_eigen_vectors_slab().mt_coeffs(0).prime().at(memory_t::host, offset2, i),
94  mt_lo_size * sizeof(double_complex));
95  }
96  }
97  }
98  }
99 
100  #pragma omp parallel for
101  for (int i = 0; i < ctx_.num_fv_states(); i++) {
102  /* G+k block */
103  std::memcpy(fv_states().pw_coeffs(0).prime().at(memory_t::host, 0, i),
104  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::host, 0, i),
105  num_gkvec_loc() * sizeof(double_complex));
106  }
107 
108  if (ctx_.processing_unit() == device_t::GPU) {
109  fv_eigen_vectors_slab().pw_coeffs(0).deallocate(memory_t::device);
110  }
111 }
112 } // namespace sirius
