37  void split_in_chunks()

39  /* initial chunk size */
40  int chunk_size = std::min(unit_cell_.num_atoms(), 256);
41  /* maximum number of chunks */
42  int num_chunks = unit_cell_.num_atoms() / chunk_size + std::min(1, unit_cell_.num_atoms() % chunk_size);
43  /* final maximum chunk size */
44  chunk_size = unit_cell_.num_atoms() / num_chunks + std::min(1, unit_cell_.num_atoms() % num_chunks);
45 
46  int offset_in_beta_gk{0};
47  beta_chunks_ = std::vector<beta_chunk_t>(num_chunks);
48 
49  for (int ib = 0; ib < num_chunks; ib++) {
50  /* number of atoms in this chunk */
51  int na = std::min(unit_cell_.num_atoms(), (ib + 1) * chunk_size) - ib * chunk_size;
52  beta_chunks_[ib].num_atoms_ = na;
53  beta_chunks_[ib].desc_ = mdarray<int, 2>(4, na);
54  beta_chunks_[ib].atom_pos_ = mdarray<double, 2>(3, na);
55 
56  int num_beta{0};
57  for (int i = 0; i < na; i++) {
58  /* global index of atom by local index and chunk */
59  int ia = ib * chunk_size + i;
60  auto pos = unit_cell_.atom(ia).position();
61  auto& type = unit_cell_.atom(ia).type();
62  /* atom fractional coordinates */
63  for (int x: {0, 1, 2}) {
64  beta_chunks_[ib].atom_pos_(x, i) = pos[x];
65  }
66  /* number of beta functions for atom */
67  beta_chunks_[ib].desc_(beta_desc_idx::nbf, i) = type.mt_basis_size();
68  /* offset in beta_gk*/
69  beta_chunks_[ib].desc_(beta_desc_idx::offset, i) = num_beta;
70  /* offset in beta_gk_t */
71  beta_chunks_[ib].desc_(beta_desc_idx::offset_t, i) = type.offset_lo();
72  /* global index of atom */
73  beta_chunks_[ib].desc_(beta_desc_idx::ia, i) = ia;
74 
75  num_beta += type.mt_basis_size();
76  }
77  /* number of beta-projectors in this chunk */
78  beta_chunks_[ib].num_beta_ = num_beta;
79  beta_chunks_[ib].offset_ = offset_in_beta_gk;
80  offset_in_beta_gk += num_beta;
81 
82  if (unit_cell_.parameters().processing_unit() == GPU) {
83  beta_chunks_[ib].desc_.allocate(memory_t::device);
84  beta_chunks_[ib].desc_.copy<memory_t::host, memory_t::device>();
85 
86  beta_chunks_[ib].atom_pos_.allocate(memory_t::device);
87  beta_chunks_[ib].atom_pos_.copy<memory_t::host, memory_t::device>();
88  }
89  }
90

92  for (auto& e: beta_chunks_) {

94  }
95
