39/// Generate SU(2) rotation matrix from the axes and angle.
40inline auto
41rotation_matrix_su2(std::array<double, 3> u__, double theta__)
42{
43 sddk::mdarray<std::complex<double>, 2> rotm(2, 2);
44
45 auto cost = std::cos(theta__ / 2);
46 auto sint = std::sin(theta__ / 2);
47
48 rotm(0, 0) = std::complex<double>(cost, -u__[2] * sint);
49 rotm(1, 1) = std::complex<double>(cost, u__[2] * sint);
50 rotm(0, 1) = std::complex<double>(-u__[1] * sint, -u__[0] * sint);
51 rotm(1, 0) = std::complex<double>( u__[1] * sint, -u__[0] * sint);
52
53 return rotm;
54}

56/// Generate SU2(2) rotation matrix from a 3x3 rotation matrix in Cartesian coordinates.
57/** Create quaternion components from the 3x3 matrix. The components are just a w = Cos(\Omega/2)
58 * and {x,y,z} = unit rotation vector multiplied by Sin(\Omega/2)
59 *
60 * See https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation
61 * and https://en.wikipedia.org/wiki/Rotation_group_SO(3)#Quaternions_of_unit_norm */
62inline auto
63rotation_matrix_su2(matrix3d<double> R__)
64{
65 double det = R__.det() > 0 ? 1.0 : -1.0;
66
67 matrix3d<double> mat = R__ * det;
68
69 sddk::mdarray<std::complex<double>, 2> su2mat(2, 2);
70
71 su2mat.zero();
72
73 /* make quaternion components*/
74 double w = std::sqrt(std::max(0.0, 1.0 + mat(0, 0) + mat(1, 1) + mat(2, 2))) / 2.0;
75 double x = std::sqrt(std::max(0.0, 1.0 + mat(0, 0) - mat(1, 1) - mat(2, 2))) / 2.0;
76 double y = std::sqrt(std::max(0.0, 1.0 - mat(0, 0) + mat(1, 1) - mat(2, 2))) / 2.0;
77 double z = std::sqrt(std::max(0.0, 1.0 - mat(0, 0) - mat(1, 1) + mat(2, 2))) / 2.0;
78
79 x = std::copysign(x, mat(2, 1) - mat(1, 2));
80 y = std::copysign(y, mat(0, 2) - mat(2, 0));
81 z = std::copysign(z, mat(1, 0) - mat(0, 1));
82
83 su2mat(0, 0) = std::complex<double>(w, -z);
84 su2mat(1, 1) = std::complex<double>(w, z);
85 su2mat(0, 1) = std::complex<double>(-y, -x);
86 su2mat(1, 0) = std::complex<double>(y, -x);
87
88 return su2mat;
89}

91/// Get axis and angle from rotation matrix.
92inline std::pair<vector3d<double>, double>
93axis_angle(matrix3d<double> R__)
94{
95 vector3d<double> u;
96 /* make proper rotation */
97 R__ = R__ * R__.det();
98 u[0] = R__(2, 1) - R__(1, 2);
99 u[1] = R__(0, 2) - R__(2, 0);
100 u[2] = R__(1, 0) - R__(0, 1);
101
102 double sint = u.length() / 2.0;
103 double cost = (R__(0, 0) + R__(1, 1) + R__(2, 2) - 1) / 2.0;
104
105 double theta = utils::phi_by_sin_cos(sint, cost);
106
107 /* rotation angle is zero */
108 if (std::abs(theta) < 1e-12) {
109 u = {0, 0, 1};
110 } else if (std::abs(theta - pi) < 1e-12) { /* rotation angle is Pi */
111 /* rotation matrix for Pi angle has this form
112
113 [-1+2ux^2 | 2 ux uy | 2 ux uz]
114 [2 ux uy | -1+2uy^2 | 2 uy uz]
115 [2 ux uz | 2 uy uz | -1+2uz^2] */
116
117 if (R__(0, 0) >= R__(1, 1) && R__(0, 0) >= R__(2, 2)) { /* x-component is largest */
118 u[0] = std::sqrt(std::abs(R__(0, 0) + 1) / 2);
119 u[1] = (R__(0, 1) + R__(1, 0)) / 4 / u[0];
120 u[2] = (R__(0, 2) + R__(2, 0)) / 4 / u[0];
121 } else if (R__(1, 1) >= R__(0, 0) && R__(1, 1) >= R__(2, 2)) { /* y-component is largest */
122 u[1] = std::sqrt(std::abs(R__(1, 1) + 1) / 2);
123 u[0] = (R__(1, 0) + R__(0, 1)) / 4 / u[1];
124 u[2] = (R__(1, 2) + R__(2, 1)) / 4 / u[1];
125 } else {
126 u[2] = std::sqrt(std::abs(R__(2, 2) + 1) / 2);
127 u[0] = (R__(2, 0) + R__(0, 2)) / 4 / u[2];
128 u[1] = (R__(2, 1) + R__(1, 2)) / 4 / u[2];
129 }
130 } else {
131 u = u * (1.0 / u.length());
132 }
133
134 return std::pair<vector3d<double>, double>(u, theta);
135}

137/// Generate rotation matrix from three Euler angles
138/** Euler angles \f$ \alpha, \beta, \gamma \f$ define the general rotation as three consecutive rotations:
139 * - about \f$ \hat e_z \f$ through the angle \f$ \gamma \f$ (\f$ 0 \le \gamma < 2\pi \f$)
140 * - about \f$ \hat e_y \f$ through the angle \f$ \beta \f$ (\f$ 0 \le \beta \le \pi \f$)
141 * - about \f$ \hat e_z \f$ through the angle \f$ \alpha \f$ (\f$ 0 \le \gamma < 2\pi \f$)
142 *
143 * The total rotation matrix is defined as a product of three rotation matrices:
144 * \f[
145 * R(\alpha, \beta, \gamma) =
146 * \left( \begin{array}{ccc} \cos(\alpha) & -\sin(\alpha) & 0 \\
147 * \sin(\alpha) & \cos(\alpha) & 0 \\
148 * 0 & 0 & 1 \end{array} \right)
149 * \left( \begin{array}{ccc} \cos(\beta) & 0 & \sin(\beta) \\
150 * 0 & 1 & 0 \\
151 * -\sin(\beta) & 0 & \cos(\beta) \end{array} \right)
152 * \left( \begin{array}{ccc} \cos(\gamma) & -\sin(\gamma) & 0 \\
153 * \sin(\gamma) & \cos(\gamma) & 0 \\
154 * 0 & 0 & 1 \end{array} \right) =
155 * \left( \begin{array}{ccc} \cos(\alpha) \cos(\beta) \cos(\gamma) - \sin(\alpha) \sin(\gamma) &
156 * -\sin(\alpha) \cos(\gamma) - \cos(\alpha) \cos(\beta) \sin(\gamma) &
157 * \cos(\alpha) \sin(\beta) \\
158 * \sin(\alpha) \cos(\beta) \cos(\gamma) + \cos(\alpha) \sin(\gamma) &
159 * \cos(\alpha) \cos(\gamma) - \sin(\alpha) \cos(\beta) \sin(\gamma) &
160 * \sin(\alpha) \sin(\beta) \\
161 * -\sin(\beta) \cos(\gamma) &
162 * \sin(\beta) \sin(\gamma) &
163 * \cos(\beta) \end{array} \right)
164 * \f]
165 */
166inline matrix3d<double> rot_mtrx_cart(vector3d<double> euler_angles__)
167{
168 double alpha = euler_angles__[0];
169 double beta = euler_angles__[1];
170 double gamma = euler_angles__[2];
171
172 matrix3d<double> rm;
173 rm(0, 0) = std::cos(alpha) * std::cos(beta) * std::cos(gamma) - std::sin(alpha) * std::sin(gamma);
174 rm(0, 1) = -std::cos(gamma) * std::sin(alpha) - std::cos(alpha) * std::cos(beta) * std::sin(gamma);
175 rm(0, 2) = std::cos(alpha) * std::sin(beta);
176 rm(1, 0) = std::cos(beta) * std::cos(gamma) * std::sin(alpha) + std::cos(alpha) * std::sin(gamma);
177 rm(1, 1) = std::cos(alpha) * std::cos(gamma) - std::cos(beta) * std::sin(alpha) * std::sin(gamma);
178 rm(1, 2) = std::sin(alpha) * std::sin(beta);
179 rm(2, 0) = -std::cos(gamma) * std::sin(beta);
180 rm(2, 1) = std::sin(beta) * std::sin(gamma);
181 rm(2, 2) = std::cos(beta);
182
183 return rm;
184}

186/// Compute Euler angles corresponding to the proper rotation matrix.
187inline vector3d<double> euler_angles(matrix3d<double> const& rot__, double tolerance__)
188{
189 vector3d<double> angles(0, 0, 0);
190
191 if (std::abs(rot__.det() - 1) > 1e-10) {
192 std::stringstream s;
193 s << "determinant of rotation matrix is " << rot__.det();
194 RTE_THROW(s);
195 }
196
197 auto rit = inverse(transpose(rot__));
198 for (int i = 0; i < 3; i++) {
199 for (int j = 0; j < 3; j++) {
200 if (std::abs(rot__(i, j) - rit(i, j)) > tolerance__) {
201 std::stringstream s;
202 s << "rotation matrix is not unitary" << std::endl
203 << "initial symmetry matrix:" << std::endl
204 << rot__ << std::endl
205 << "inverse transpose matrix:" << std::endl
206 << rit;
207 RTE_THROW(s);
208 }
209 }
210 }
211
212 if (std::abs(rot__(2, 2) - 1.0) < 1e-10) { // cos(beta) == 1, beta = 0
213 angles[0] = utils::phi_by_sin_cos(rot__(1, 0), rot__(0, 0));
214 } else if (std::abs(rot__(2, 2) + 1.0) < 1e-10) { // cos(beta) == -1, beta = Pi
215 angles[0] = utils::phi_by_sin_cos(-rot__(0, 1), rot__(1, 1));
216 angles[1] = pi;
217 } else {
218 double beta = std::acos(rot__(2, 2));
219 angles[0] = utils::phi_by_sin_cos(rot__(1, 2) / std::sin(beta), rot__(0, 2) / std::sin(beta));
220 angles[1] = beta;
221 angles[2] = utils::phi_by_sin_cos(rot__(2, 1) / std::sin(beta), -rot__(2, 0) / std::sin(beta));
222 }
223
224 auto rm1 = rot_mtrx_cart(angles);
225
226 for (int i = 0; i < 3; i++) {
227 for (int j = 0; j < 3; j++) {
228 if (std::abs(rot__(i, j) - rm1(i, j)) > tolerance__) {
229 std::stringstream s;
230 s << "matrices don't match" << std::endl
231 << "initial symmetry matrix: " << std::endl
232 << rot__ << std::endl
233 << "euler angles : " << angles[0] / pi << " " << angles[1] / pi << " " << angles[2] / pi << std::endl
234 << "computed symmetry matrix : " << std::endl
235 << rm1;
236 RTE_THROW(s);
237 }
238 }
239 }
240
241 return angles;
242}
