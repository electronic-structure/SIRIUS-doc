38 /// Generate SU(2) rotation matrix from the axes and angle.
39 inline mdarray<std::complex<double>, 2> rotation_matrix_su2(std::array<double, 3> u__, double theta__)
40 {
41  mdarray<std::complex<double>, 2> rotm(2, 2);
42 
43  auto cost = std::cos(theta__ / 2);
44  auto sint = std::sin(theta__ / 2);
45 
46  rotm(0, 0) = std::complex<double>(cost, -u__[2] * sint);
47  rotm(1, 1) = std::complex<double>(cost, u__[2] * sint);
48  rotm(0, 1) = std::complex<double>(-u__[1] * sint, -u__[0] * sint);
49  rotm(1, 0) = std::complex<double>( u__[1] * sint, -u__[0] * sint);
50 
51  return rotm;
52 }

54 /// Generate SU2(2) rotation matrix from a 3x3 rotation matrix in Cartesian coordinates.
55 /** Create quaternion components from the 3x3 matrix. The components are just a w = Cos(\Omega/2)
56  * and {x,y,z} = unit rotation vector multiplied by Sin(\Omega/2)
57  *
58  * See https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation
59  * and https://en.wikipedia.org/wiki/Rotation_group_SO(3)#Quaternions_of_unit_norm */
60 inline mdarray<std::complex<double>, 2> rotation_matrix_su2(matrix3d<double> R__)
61 {
62  double det = R__.det() > 0 ? 1.0 : -1.0;
63 
64  matrix3d<double> mat = R__ * det;
65 
66  mdarray<std::complex<double>, 2> su2mat(2, 2);
67 
68  su2mat.zero();
69 
70  /* make quaternion components*/
71  double w = std::sqrt(std::max(0.0, 1.0 + mat(0, 0) + mat(1, 1) + mat(2, 2))) / 2.0;
72  double x = std::sqrt(std::max(0.0, 1.0 + mat(0, 0) - mat(1, 1) - mat(2, 2))) / 2.0;
73  double y = std::sqrt(std::max(0.0, 1.0 - mat(0, 0) + mat(1, 1) - mat(2, 2))) / 2.0;
74  double z = std::sqrt(std::max(0.0, 1.0 - mat(0, 0) - mat(1, 1) + mat(2, 2))) / 2.0;
75 
76  x = std::copysign(x, mat(2, 1) - mat(1, 2));
77  y = std::copysign(y, mat(0, 2) - mat(2, 0));
78  z = std::copysign(z, mat(1, 0) - mat(0, 1));
79 
80  su2mat(0, 0) = std::complex<double>(w, -z);
81  su2mat(1, 1) = std::complex<double>(w, z);
82  su2mat(0, 1) = std::complex<double>(-y, -x);
83  su2mat(1, 0) = std::complex<double>(y, -x);
84 
85  return su2mat;
86 }

88 /// Get axis and angle from rotation matrix.
89 inline std::pair<vector3d<double>, double> axis_angle(matrix3d<double> R__)
90 {
91  vector3d<double> u;
92  /* make proper rotation */
93  R__ = R__ * R__.det();
94  u[0] = R__(2, 1) - R__(1, 2);
95  u[1] = R__(0, 2) - R__(2, 0);
96  u[2] = R__(1, 0) - R__(0, 1);
97 
98  double sint = u.length() / 2.0;
99  double cost = (R__(0, 0) + R__(1, 1) + R__(2, 2) - 1) / 2.0;
100 
101  double theta = utils::phi_by_sin_cos(sint, cost);
102 
103  /* rotation angle is zero */
104  if (std::abs(theta) < 1e-12) {
105  u = {0, 0, 1};
106  } else if (std::abs(theta - pi) < 1e-12) { /* rotation angle is Pi */
107  /* rotation matrix for Pi angle has this form
108 
109  [-1+2ux^2 | 2 ux uy | 2 ux uz]
110  [2 ux uy | -1+2uy^2 | 2 uy uz]
111  [2 ux uz | 2 uy uz | -1+2uz^2] */
112 
113  if (R__(0, 0) >= R__(1, 1) && R__(0, 0) >= R__(2, 2)) { /* x-component is largest */
114  u[0] = std::sqrt(std::abs(R__(0, 0) + 1) / 2);
115  u[1] = (R__(0, 1) + R__(1, 0)) / 4 / u[0];
116  u[2] = (R__(0, 2) + R__(2, 0)) / 4 / u[0];
117  } else if (R__(1, 1) >= R__(0, 0) && R__(1, 1) >= R__(2, 2)) { /* y-component is largest */
118  u[1] = std::sqrt(std::abs(R__(1, 1) + 1) / 2);
119  u[0] = (R__(1, 0) + R__(0, 1)) / 4 / u[1];
120  u[2] = (R__(1, 2) + R__(2, 1)) / 4 / u[1];
121  } else {
122  u[2] = std::sqrt(std::abs(R__(2, 2) + 1) / 2);
123  u[0] = (R__(2, 0) + R__(0, 2)) / 4 / u[2];
124  u[1] = (R__(2, 1) + R__(1, 2)) / 4 / u[2];
125  }
126  } else {
127  u = u * (1.0 / u.length());
128  }
129 
130  return std::pair<vector3d<double>, double>(u, theta);
131 }

133 /// Generate rotation matrix from three Euler angles
134 /** Euler angles \f$ \alpha, \beta, \gamma \f$ define the general rotation as three consecutive rotations:
135  * - about \f$ \hat e_z \f$ through the angle \f$ \gamma \f$ (\f$ 0 \le \gamma < 2\pi \f$)
136  * - about \f$ \hat e_y \f$ through the angle \f$ \beta \f$ (\f$ 0 \le \beta \le \pi \f$)
137  * - about \f$ \hat e_z \f$ through the angle \f$ \alpha \f$ (\f$ 0 \le \gamma < 2\pi \f$)
138  *
139  * The total rotation matrix is defined as a product of three rotation matrices:
140  * \f[
141  * R(\alpha, \beta, \gamma) =
142  * \left( \begin{array}{ccc} \cos(\alpha) & -\sin(\alpha) & 0 \\
143  * \sin(\alpha) & \cos(\alpha) & 0 \\
144  * 0 & 0 & 1 \end{array} \right)
145  * \left( \begin{array}{ccc} \cos(\beta) & 0 & \sin(\beta) \\
146  * 0 & 1 & 0 \\
147  * -\sin(\beta) & 0 & \cos(\beta) \end{array} \right)
148  * \left( \begin{array}{ccc} \cos(\gamma) & -\sin(\gamma) & 0 \\
149  * \sin(\gamma) & \cos(\gamma) & 0 \\
150  * 0 & 0 & 1 \end{array} \right) =
151  * \left( \begin{array}{ccc} \cos(\alpha) \cos(\beta) \cos(\gamma) - \sin(\alpha) \sin(\gamma) &
152  * -\sin(\alpha) \cos(\gamma) - \cos(\alpha) \cos(\beta) \sin(\gamma) &
153  * \cos(\alpha) \sin(\beta) \\
154  * \sin(\alpha) \cos(\beta) \cos(\gamma) + \cos(\alpha) \sin(\gamma) &
155  * \cos(\alpha) \cos(\gamma) - \sin(\alpha) \cos(\beta) \sin(\gamma) &
156  * \sin(\alpha) \sin(\beta) \\
157  * -\sin(\beta) \cos(\gamma) &
158  * \sin(\beta) \sin(\gamma) &
159  * \cos(\beta) \end{array} \right)
160  * \f]
161  */
162 inline matrix3d<double> rot_mtrx_cart(vector3d<double> euler_angles__)
163 {
164  double alpha = euler_angles__[0];
165  double beta = euler_angles__[1];
166  double gamma = euler_angles__[2];
167 
168  matrix3d<double> rm;
169  rm(0, 0) = std::cos(alpha) * std::cos(beta) * std::cos(gamma) - std::sin(alpha) * std::sin(gamma);
170  rm(0, 1) = -std::cos(gamma) * std::sin(alpha) - std::cos(alpha) * std::cos(beta) * std::sin(gamma);
171  rm(0, 2) = std::cos(alpha) * std::sin(beta);
172  rm(1, 0) = std::cos(beta) * std::cos(gamma) * std::sin(alpha) + std::cos(alpha) * std::sin(gamma);
173  rm(1, 1) = std::cos(alpha) * std::cos(gamma) - std::cos(beta) * std::sin(alpha) * std::sin(gamma);
174  rm(1, 2) = std::sin(alpha) * std::sin(beta);
175  rm(2, 0) = -std::cos(gamma) * std::sin(beta);
176  rm(2, 1) = std::sin(beta) * std::sin(gamma);
177  rm(2, 2) = std::cos(beta);
178 
179  return rm;
180 }

182 /// Compute Euler angles corresponding to the proper rotation matrix.
183 inline vector3d<double> euler_angles(matrix3d<double> const& rot__)
184 {
185  vector3d<double> angles(0, 0, 0);
186 
187  if (std::abs(rot__.det() - 1) > 1e-10) {
188  std::stringstream s;
189  s << "determinant of rotation matrix is " << rot__.det();
190  TERMINATE(s);
191  }
192 
193  if (std::abs(rot__(2, 2) - 1.0) < 1e-10) { // cos(beta) == 1, beta = 0
194  angles[0] = utils::phi_by_sin_cos(rot__(1, 0), rot__(0, 0));
195  } else if (std::abs(rot__(2, 2) + 1.0) < 1e-10) { // cos(beta) == -1, beta = Pi
196  angles[0] = utils::phi_by_sin_cos(-rot__(0, 1), rot__(1, 1));
197  angles[1] = pi;
198  } else {
199  double beta = std::acos(rot__(2, 2));
200  angles[0] = utils::phi_by_sin_cos(rot__(1, 2) / std::sin(beta), rot__(0, 2) / std::sin(beta));
201  angles[1] = beta;
202  angles[2] = utils::phi_by_sin_cos(rot__(2, 1) / std::sin(beta), -rot__(2, 0) / std::sin(beta));
203  }
204 
205  auto rm1 = rot_mtrx_cart(angles);
206 
207  for (int i = 0; i < 3; i++) {
208  for (int j = 0; j < 3; j++) {
209  if (std::abs(rot__(i, j) - rm1(i, j)) > 1e-8) {
210  std::stringstream s;
211  s << "matrices don't match" << std::endl
212  << "initial symmetry matrix: " << std::endl
213  << rot__(0, 0) << " " << rot__(0, 1) << " " << rot__(0, 2) << std::endl
214  << rot__(1, 0) << " " << rot__(1, 1) << " " << rot__(1, 2) << std::endl
215  << rot__(2, 0) << " " << rot__(2, 1) << " " << rot__(2, 2) << std::endl
216  << "euler angles : " << angles[0] / pi << " " << angles[1] / pi << " " << angles[2] / pi << std::endl
217  << "computed symmetry matrix : " << std::endl
218  << rm1(0, 0) << " " << rm1(0, 1) << " " << rm1(0, 2) << std::endl
219  << rm1(1, 0) << " " << rm1(1, 1) << " " << rm1(1, 2) << std::endl
220  << rm1(2, 0) << " " << rm1(2, 1) << " " << rm1(2, 2) << std::endl;
221  TERMINATE(s);
222  }
223  }
224  }
225 
226  return angles;
227 }
