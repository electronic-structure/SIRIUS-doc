94 /// Initialize the library.
95 inline void initialize(bool call_mpi_init__ = true)
96 {
97  PROFILE_START("sirius");
98  PROFILE("sirius::initialize");
99  if (is_initialized()) {
100  TERMINATE("SIRIUS library is already initialized");
101  }
102 #if defined(SIRIUS_USE_POWER_COUNTER)
103  energy() = -utils::power::energy();
104  energy_acc() = -utils::power::device_energy();
105 #endif
106  if (call_mpi_init__) {
107  Communicator::initialize(MPI_THREAD_MULTIPLE);
108  }
109 #if defined(__APEX)
110  apex::init("sirius", Communicator::world().rank(), Communicator::world().size());
111 #endif
112 
113  if (Communicator::world().rank() == 0) {
114  std::printf("SIRIUS %i.%i.%i, git hash: %s\n", sirius::major_version(), sirius::minor_version(),
115  sirius::revision(), sirius::git_hash().c_str());
116 #if !defined(NDEBUG)
117  std::printf("Warning! Compiled in 'debug' mode with assert statements enabled!\n");
118 #endif
119  }
120  /* get number of ranks per node during the global call to sirius::initialize() */
121  sddk::num_ranks_per_node();
122  if (acc::num_devices() > 0) {
123  int devid = sddk::get_device_id(acc::num_devices());
124  acc::set_device_id(devid);
125  /* create extensive amount of streams */
126  /* some parts of the code rely on the number of streams not related to the
127  number of OMP threads */
128  acc::create_streams(std::max(omp_get_max_threads(), 6));
129 #if defined(SIRIUS_GPU)
130  accblas::create_stream_handles();
131 #endif
132 #if defined(SIRIUS_CUDA)
133  accblas::xt::create_handle();
134  cusolver::create_handle();
135 #endif
136  }
137  splablas::reset_handle();
138 
139 #if defined(SIRIUS_MAGMA)
140  magma::init();
141 #endif
142 #if defined(__PLASMA)
143  plasma_init(omp_get_max_threads());
144 #endif
145 #if defined(__LIBSCI_ACC)
146  libsci_acc_init();
147 #endif
148 #if defined(SIRIUS_ELPA)
149  if (elpa_init(20170403) != ELPA_OK) {
150  TERMINATE("ELPA API version not supported");
151  }
152 #endif
153  /* for the fortran interface to blas/lapack */
154  assert(sizeof(int) == 4);
155  assert(sizeof(double) == 8);
156 
157  is_initialized() = true;
158 }

160 /// Shut down the library.
161 inline void finalize(bool call_mpi_fin__ = true, bool reset_device__ = true, bool fftw_cleanup__ = true)
162 {
163  PROFILE_START("sirius::finalize");
164  if (!is_initialized()) {
165  TERMINATE("SIRIUS library was not initialized");
166  }
167 #if defined(SIRIUS_MAGMA)
168  magma::finalize();
169 #endif
170 #if defined(__LIBSCI_ACC)
171  libsci_acc_finalize();
172 #endif
173 
174  // must be called before device is reset
175  splablas::reset_handle();
176 
177  if (acc::num_devices()) {
178 #if defined(SIRIUS_GPU)
179  accblas::destroy_stream_handles();
180 #endif
181 #if defined(SIRIUS_CUDA)
182  cusolver::destroy_handle();
183  accblas::xt::destroy_handle();
184 #endif
185  acc::destroy_streams();
186  if (reset_device__) {
187  acc::reset();
188  }
189  }
190 
191  //utils::stop_global_timer();
192 #if defined(__APEX)
193  apex::finalize();
194 #endif
195 #if defined(SIRIUS_USE_POWER_COUNTER)
196  double e = energy() + utils::power::energy();
197  double e_acc = energy_acc() + utils::power::device_energy();
198  if (Communicator::world().rank() == 0) {
199  printf("=== Energy consumption (root MPI rank) ===\n");
200  printf("energy : %9.2f Joules\n", e);
201  printf("energy_acc : %9.2f Joules\n", e_acc);
202  }
203  Communicator::world().allreduce(&e, 1);
204  Communicator::world().allreduce(&e_acc, 1);
205  int nn = utils::power::num_nodes();
206  if (Communicator::world().rank() == 0 && nn > 0) {
207  printf("=== Energy consumption (all nodes) ===\n");
208  printf("energy : %9.2f Joules\n", e * nn / Communicator::world().size());
209  printf("energy_acc : %9.2f Joules\n", e_acc * nn / Communicator::world().size());
210  }
211 #endif
212  if (call_mpi_fin__) {
213  Communicator::finalize();
214  }
215 #if defined(SIRIUS_ELPA)
216  int ierr;
217  elpa_uninit(&ierr);
218 #endif
219 
220  is_initialized() = false;
221 
222  PROFILE_STOP("sirius::finalize");
223  PROFILE_STOP("sirius");
224 }

229 /** \mainpage Welcome to SIRIUS
230 
231  SIRIUS is a domain specific library for electronic structure calculations. It implements pseudopotential plane
232  wave (PP-PW) and full potential linearized augmented plane wave (FP-LAPW) methods and is designed for
233  GPU acceleration of popular community codes such as Exciting, Elk and Quantum ESPRESSO.
234  SIRIUS is written in C++11 with MPI, OpenMP and CUDA/ROCm programming models. SIRIUS is organised as a
235  collection of classes that abstract away the different building blocks of DFT self-consistency cycle.
236 
237  For a quick start please refer to the main development page at
238  <a href="https://github.com/electronic-structure/SIRIUS">GitHub</a>.
239 
240  The generated Fortran API is described here: generated.f90
241 
242  The frequent variable names are listed on the page \ref stdvarname.
243 
244  We use the following \ref coding.
245 
246  The library files and directories are organised in the following way:
247  - \b apps -
248  - \b atoms - utility program to generate FP-LAPW atomic species files
249  - \b bands - band plotting
250  - \b cif_input - CIF parser
251  - \b dft_loop - DFT miniapp
252  - \b hydrogen - solve hydrogen-like atom using Schrödinger equation
253  - \b tests - tests of various functionality
254  - \b timers - scripts to analyze timer outputs
255  - \b unit_tests - unit tests
256  - \b upf - scripts to parse and convert UPF files
257  - \b utils - utilities to work with unit cell
258  - \b ci - directory with Jenkins, Travis CI and GitHub action scripts
259  - \b cmake - directory with CMake scripts
260  - \b doc - this directory contains configuration file for Doxygen documentation and PNG images
261  - \b examples - examples of input files for pseudopotential and full-potential calculations
262  - \b python_module - Python interface module
263  - \b reframe - ReFrame regression tests description
264  - \b src - main directory with the source code
265  - \b verification - verification tests
266  - .clang-format - source code formatting rules
267  - CMakeLists.txt - CMake file of the project
268  - check_format.py, check_format.x - scripts to check source code formatting
269  - clang_format.x - script to apply Clang format to a file
270 
271 */

296 /**
297 \page coding Coding style
298 
299 Below are some basic style rules that we follow:
300  - Page width is approximately 120 characters. Screens are wide nowadays and 80 characters is an
301  obsolete restriction. Going slightly over 120 characters is allowed if it is requird for the line continuity.
302  - Indentation: 4 spaces (no tabs)
303 
304  Exception: class access modifiers are idented with 2 spaces
305  \code{.cpp}
306  class A
307  {
308  private:
309  int n_;
310  public:
311  A();
312  };
313  \endcode
314 
315  - Comments are inserted before the code with slash-star style starting with the lower case:
316  \code{.cpp}
317  // call a very important function
318  do_something();
319  \endcode
320  - Spaces between most operators:
321  \code{.cpp}
322  if (i < 5) {
323  j = 5;
324  }
325 
326  for (int k = 0; k < 3; k++)
327 
328  int lm = l * l + l + m;
329 
330  double d = std::abs(e);
331 
332  int k = idx[3];
333  \endcode
334  - Spaces between function arguments:
335  \code{.cpp}
336  double d = some_func(a, b, c);
337  \endcode
338  but not
339  \code{.cpp}
340  double d=some_func(a,b,c);
341  \endcode
342  and not
343  \code{.cpp}
344  double d = some_func( a, b, c );
345  \endcode
346  - Spaces between template arguments, but not between <,> brackets:
347  \code{.cpp}
348  std::vector<std::array<int, 2>> vec;
349  \endcode
350  but not
351  \code{.cpp}
352  std::vector< std::array< int, 2 > > vec;
353  \endcode
354  - Curly braces for classes and functions start form the new line:
355  \code{.cpp}
356  class A
357  {
358  ....
359  };
360 
361  inline int num_points()
362  {
363  return num_points_;
364  }
365  \endcode
366  - Curly braces for if-statements, for-loops, switch-case statements, etc. start at the end of the line:
367  \code{.cpp}
368  for (int i: {0, 1, 2}) {
369  some_func(i);
370  }
371 
372  if (a == 0) {
373  std::printf("a is zero");
374  } else {
375  std::printf("a is not zero");
376  }
377 
378  switch (i) {
379  case 1: {
380  do_something();
381  break;
382  case 2: {
383  do_something_else();
384  break;
385  }
386  }
387  \endcode
388  - Even single line 'if' statements and 'for' loops must have the curly brackes:
389  \code{.cpp}
390  if (i == 4) {
391  some_variable = 5;
392  }
393 
394  for (int k = 0; k < 10; k++) {
395  do_something(k);
396  }
397  \endcode
398  - Reference and pointer symbols are part of type:
399  \code{.cpp}
400  std::vector<double>& vec = make_vector();
401 
402  double* ptr = &vec[0];
403 
404  auto& atom = unit_cell().atom(ia);
405  \endcode
406  - Const modifier follows the type declaration:
407  \code{.cpp}
408  std::vector<int> const& idx() const
409  {
410  return idx_;
411  }
412  // or
413  auto const& atom = unit_cell().atom(ia);
414  \endcode
415  - Names of class members end with underscore:
416  \code{.cpp}
417  class A
418  {
419  private:
420  int lmax_;
421  };
422  \endcode
423  - Setter method starts from set_, getter method is a variable name itself:
424  \code{.cpp}
425  class A
426  {
427  private:
428  int lmax_;
429  public:
430  int lmax() const
431  {
432  return lmax_;
433  }
434  void set_lmax(int lmax__)
435  {
436  lmax_ = lmax__;
437  }
438  };
439  \endcode
440  However, the new style for setter methods is preferable:
441  \code{.cpp}
442  class A
443  {
444  private:
445  int lmax_;
446  public:
447  int lmax() const
448  {
449  return lmax_;
450  }
451  int lmax(int lmax__)
452  {
453  lmax_ = lmax__;
454  return lmax_;
455  }
456  };
457  \endcode
458  - Order of class members: private, protected, public
459  \code{.cpp}
460  class A
461  {
462  private:
463  int lmax_;
464  void bar();
465  protected:
466  void foo();
467  public:
468  int lmax() const
469  {
470  return lmax_;
471  }
472  };
473  \endcode
474  - Single-line functions should not be flattened:
475  \code{.cpp}
476  struct A
477  {
478  int lmax() const
479  {
480  return lmax_;
481  }
482  };
483  \endcode
484  but not
485  \code{.cpp}
486  struct A
487  {
488  int lmax() const { return lmax_; }
489  };
490  \endcode
491  - Header guards have a standard name: double underscore + file name in capital letters + double underscore
492  \code{.cpp}
493  #ifndef __HEADER_HPP__
494  #define __HEADER_HPP__
495  ...
496  #endif // __HEADER_HPP__
497  \endcode
498  - Variable names are all in lowercase and underscore-separated (aka 'snake_case'):
499  \code{.cpp}
500  int num_bands;
501  std::complex<double> beta_psi;
502  \endcode
503  but not
504  \code{.cpp}
505  int NumBands;
506  // or
507  std::complex<double> BetaPsi;
508  // or
509  std::complex<double> Another_BetaPsi;
510  \endcode
511 
512 We use clang-format utility to enforce the basic formatting style. Please have a look at .clang-format config file
513 in the source root folder for the definitions and use helper script 'clang_format.x'.
514 
515 <b>Class naming convention.</b>
516 
517 Problem: all 'standard' naming conventions are not satisfactory. For example, we have a class
518 which does a DFT ground state. Following the common naming conventions it could be named like this:
519 DFTGroundState, DftGroundState, dft_ground_state. Last two are bad, because DFT (and not Dft or dft)
520 is a well recognized abbreviation. First one is band because capital G adds to DFT and we automaticaly
521 read DFTG round state.
522 
523 Solution: we can propose the following: DFTgroundState or DFT_ground_state. The first variant still
524 doens't look very good because one of the words is captalized (State) and one (ground) - is not. So we pick
525 the second variant: DFT_ground_state (by the way, this is close to the Bjarne Stroustrup's naiming convention,
526 where he uses first capital letter and underscores, for example class Io_obj).
527 
528 Some other examples:
529  - class Ground_state (composed of two words)
530  - class FFT_interface (composed of an abbreviation and a word)
531  - class Interface_XC (composed of a word and abbreviation)
532  - class Spline (single word)
533 
534 Exceptions are allowed if it makes sense. For example, low level utility classes like 'mdarray' (multi-dimensional
535 array) or 'pstdout' (parallel standard output) are named with small letters.
536 */
