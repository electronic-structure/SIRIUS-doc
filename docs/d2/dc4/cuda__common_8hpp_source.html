92  minval = std::min(minval, it->second[i]);
93  maxval = std::max(maxval, it->second[i]);
94  }
95  double average = (count == 0) ? 0.0 : total / count;
96  if (count == 0) minval = 0.0;
97 
98  printf("%-60s : %5i %10.4f %10.4f %10.4f %10.4f\n", it->first.c_str(), count, total, minval, maxval, average);
99  }

119  void stop()
120  {
121  float time;
122  cudaEventRecord(e_stop_, 0);
123  cudaEventSynchronize(e_stop_);
124  cudaEventElapsedTime(&time, e_start_, e_stop_);
125  cudaEventDestroy(e_start_);
