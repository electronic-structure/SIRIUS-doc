39template <typename F>
40inline int call_test(std::string label__, F&& f__)
41{
42 int err{0};
43 std::string msg;
44 try {
45 err = f__();
46 }
47 catch (std::exception const& e) {
48 err = 1;
49 msg = e.what();
50 }
51 catch (...) {
52 err = 1;
53 msg = "unknown exception";
54 }
55 if (err) {
56 std::cout << label__ << " : Failed" << std::endl;
57 if (msg.size()) {
58 std::cout << "exception occured:" << std::endl;
59 std::cout << msg << std::endl;
60 }
61 } else {
62 std::cout << label__ << " : OK" << std::endl;
63 }
64 return err;
65}

80class Measurement: public std::vector<double>
81{
82 public:
83
84 double average() const
85 {
86 double d = 0;
87 for (size_t i = 0; i < this->size(); i++) {
88 d += (*this)[i];
89 }
90 d /= static_cast<double>(this->size());
91 return d;
92 }
93
94 double sigma() const
95 {
96 double avg = average();
97 double variance = 0;
98 for (size_t i = 0; i < this->size(); i++) {
99 variance += std::pow((*this)[i] - avg, 2);
100 }
101 variance /= static_cast<double>(this->size());
102 return std::sqrt(variance);
103 }
104};

106template <typename T>
107sddk::dmatrix<T> random_symmetric(int N__, int bs__, sddk::BLACS_grid const& blacs_grid__)
108{
109 PROFILE("random_symmetric");
110
111 sddk::dmatrix<T> A(N__, N__, blacs_grid__, bs__, bs__);
112 sddk::dmatrix<T> B(N__, N__, blacs_grid__, bs__, bs__);
113 for (int j = 0; j < A.num_cols_local(); j++) {
114 for (int i = 0; i < A.num_rows_local(); i++) {
115 A(i, j) = utils::random<T>();
116 }
117 }
118
119#ifdef SIRIUS_SCALAPACK
120 sddk::linalg(sddk::linalg_t::scalapack).tranc(N__, N__, A, 0, 0, B, 0, 0);
121#else
122 for (int i = 0; i < N__; i++) {
123 for (int j = 0; j < N__; j++) {
124 B(i, j) = utils::conj(A(j, i));
125 }
126 }
127#endif
128
129 for (int j = 0; j < A.num_cols_local(); j++) {
130 for (int i = 0; i < A.num_rows_local(); i++) {
131 A(i, j) = 0.5 * (A(i, j) + B(i, j));
132 }
133 }
134
135 for (int i = 0; i < N__; i++) {
136 A.set(i, i, 50.0);
137 }
138
139 return A;
140}
