374 bool Unit_cell::is_point_in_mt(vector3d<double> vc, int& ja, int& jr, double& dr, double tp[2]) const
375 {
376  /* reduce coordinates to the primitive unit cell */
377  auto vr = reduce_coordinates(get_fractional_coordinates(vc));
378 
379  for (int ia = 0; ia < num_atoms(); ia++) {
380  for (int i0 = -1; i0 <= 1; i0++) {
381  for (int i1 = -1; i1 <= 1; i1++) {
382  for (int i2 = -1; i2 <= 1; i2++) {
383  /* atom position */
384  vector3d<double> posf = vector3d<double>(i0, i1, i2) + atom(ia).position();
385 
386  /* vector connecting center of atom and reduced point */
387  vector3d<double> vf = vr.first - posf;
388 
389  /* convert to spherical coordinates */
390  auto vs = SHT::spherical_coordinates(get_cartesian_coordinates(vf));
391 
392  if (vs[0] < atom(ia).mt_radius()) {
393  ja = ia;
394  tp[0] = vs[1]; // theta
395  tp[1] = vs[2]; // phi
396 
397  if (vs[0] < atom(ia).type().radial_grid(0)) {
398  jr = 0;
399  dr = 0.0;
400  } else {
401  for (int ir = 0; ir < atom(ia).num_mt_points() - 1; ir++) {
402  if (vs[0] >= atom(ia).type().radial_grid(ir) &&
403  vs[0] < atom(ia).type().radial_grid(ir + 1)) {
404  jr = ir;
405  dr = vs[0] - atom(ia).type().radial_grid(ir);
406  break;
407  }
408  }
409  }
410 
411  return true;
412  }
413  }
414  }
415  }
416  }
417  ja = -1;
418  jr = -1;

600 void Unit_cell::get_symmetry()
601 {
602  PROFILE("sirius::Unit_cell::get_symmetry");
603 
604  if (num_atoms() == 0) {
605  return;
606  }
607 
608  if (atom_symmetry_classes_.size() != 0) {
609  atom_symmetry_classes_.clear();
610  for (int ia = 0; ia < num_atoms(); ia++) {
611  atom(ia).set_symmetry_class(nullptr);
612  }
613  }
614 
615  mdarray<double, 2> positions(3, num_atoms());
616  mdarray<double, 2> spins(3, num_atoms());
617  std::vector<int> types(num_atoms());
618  for (int ia = 0; ia < num_atoms(); ia++) {
619  auto vp = atom(ia).position();
620  auto vf = atom(ia).vector_field();
621  for (int x : {0, 1, 2}) {
622  positions(x, ia) = vp[x];
623  spins(x, ia) = vf[x];
624  }
625  types[ia] = atom(ia).type_id();
626  }
627 
628  symmetry_ = std::unique_ptr<Unit_cell_symmetry>(
629  new Unit_cell_symmetry(lattice_vectors_, num_atoms(), types, positions, spins, parameters_.so_correction(),
630  parameters_.spglib_tolerance(), parameters_.use_symmetry()));
631 
632  int atom_class_id{-1};
633  std::vector<int> asc(num_atoms(), -1);
634  for (int i = 0; i < num_atoms(); i++) {
635  /* if symmetry class is not assigned to this atom */
636  if (asc[i] == -1) {
637  /* take next id */
638  atom_class_id++;
639  atom_symmetry_classes_.push_back(std::shared_ptr<Atom_symmetry_class>(
640  new Atom_symmetry_class(atom_class_id, atom(i).type())));
641 
642  /* scan all atoms */
643  for (int j = 0; j < num_atoms(); j++) {
644  bool is_equal = (equivalent_atoms_.size())
645  ? (equivalent_atoms_[j] == equivalent_atoms_[i])
646  : (symmetry_->atom_symmetry_class(j) == symmetry_->atom_symmetry_class(i));
647  /* assign new class id for all equivalent atoms */
648  if (is_equal) {
649  asc[j] = atom_class_id;
650  atom_symmetry_classes_.back()->add_atom_id(j);
651  }
652  }
653  }
654  }
655 
656  for (auto e : atom_symmetry_classes_) {
657  for (int i = 0; i < e->num_atoms(); i++) {
658  int ia = e->atom_id(i);
659  atom(ia).set_symmetry_class(e);
660  }
661  }
662 
663  assert(num_atom_symmetry_classes() != 0);
664 }

666 void Unit_cell::import(config_t::unit_cell_t const &inp__)
667 {
668  auto lv = matrix3d<double>(inp__.lattice_vectors());
669  lv *= inp__.lattice_vectors_scale();
670  set_lattice_vectors(vector3d<double>(lv(0, 0), lv(0, 1), lv(0, 2)),
671  vector3d<double>(lv(1, 0), lv(1, 1), lv(1, 2)),
672  vector3d<double>(lv(2, 0), lv(2, 1), lv(2, 2)));
673 
674  auto ilv = inverse(lv);
675 
676  auto units = inp__.atom_coordinate_units();
677 
678  /* first, load all types */
679  for (auto label: inp__.atom_types()) {
680  auto fname = inp__.atom_files(label);
681  add_atom_type(label, fname);
682  }
683  for (auto label : inp__.atom_types()) {
684  for (auto v : inp__.atoms(label)) {
685  vector3d<double> p(v[0], v[1], v[2]);
686  vector3d<double> f;
687  if (v.size() == 6) {
688  f = vector3d<double>(v[3], v[4], v[5]);
689  }
690  /* convert to atomic units */
691  if (units == "A") {
692  for (int x : {0, 1, 2}) {
693  p[x] /= bohr_radius;
694  }
695  }
696  /* convert from Cartesian to lattice coordinates */
697  if (units == "au" || units == "A") {
698  p = dot(ilv, p);
699  auto rc = reduce_coordinates(p);
700  for (int x : {0, 1, 2}) {
701  p[x] = rc.first[x];
702  }
703  }
704  add_atom(label, p, f);
705  }
706  }

709 void Unit_cell::update()
710 {
711  PROFILE("sirius::Unit_cell::update");
712 
713  auto v0 = lattice_vector(0);
714  auto v1 = lattice_vector(1);
715  auto v2 = lattice_vector(2);
716 
717  double r = std::max(std::max(v0.length(), std::max(v1.length(), v2.length())),
718  parameters_.cfg().parameters().nn_radius());
719 
720  find_nearest_neighbours(r);
721 
722  if (parameters_.full_potential()) {
723  /* find new MT radii and initialize radial grid */
724  if (parameters_.auto_rmt()) {
725  auto rg = get_radial_grid_t(parameters_.cfg().settings().radial_grid());
726  std::vector<double> Rmt = find_mt_radii();
727  for (int iat = 0; iat < num_atom_types(); iat++) {
728  double r0 = atom_type(iat).radial_grid().first();
729 
730  atom_type(iat).set_radial_grid(rg.first, atom_type(iat).num_mt_points(), r0, Rmt[iat], rg.second);
731  }
732  }
733 
734  int ia, ja;
735  if (check_mt_overlap(ia, ja)) {
736  std::stringstream s;
737  s << "overlaping muffin-tin spheres for atoms " << ia << "(" << atom(ia).type().symbol() << ")"
738  << " and " << ja << "(" << atom(ja).type().symbol() << ")" << std::endl
739  << " radius of atom " << ia << " : " << atom(ia).mt_radius() << std::endl
740  << " radius of atom " << ja << " : " << atom(ja).mt_radius() << std::endl
741  << " distance : " << nearest_neighbours_[ia][1].distance << " " << nearest_neighbours_[ja][1].distance;
742  TERMINATE(s);
743  }
744 
745  min_mt_radius_ = 1e100;
746  max_mt_radius_ = 0;
747  for (int i = 0; i < num_atom_types(); i++) {
748  min_mt_radius_ = std::min(min_mt_radius_, atom_type(i).mt_radius());
749  max_mt_radius_ = std::max(max_mt_radius_, atom_type(i).mt_radius());
750  }
751  }
752 
753  get_symmetry();
754 
755  spl_num_atom_symmetry_classes_ = splindex<splindex_t::block>(num_atom_symmetry_classes(), comm_.size(), comm_.rank());
756 
757  volume_mt_ = 0.0;
758  if (parameters_.full_potential()) {
759  for (int ia = 0; ia < num_atoms(); ia++) {
760  volume_mt_ += fourpi * std::pow(atom(ia).mt_radius(), 3) / 3.0;
761  }
762  }
763 
764  volume_it_ = omega() - volume_mt_;
765 
766  for (int iat = 0; iat < num_atom_types(); iat++) {
767  int nat = atom_type(iat).num_atoms();
768  if (nat > 0) {
769  for (int i = 0; i < nat; i++) {
770  int ia = atom_type(iat).atom_id(i);
771  for (int x: {0, 1, 2}) {
772  atom_coord_[iat](i, x) = atom(ia).position()[x];
773  }
774  }
775  if (parameters_.processing_unit() == device_t::GPU) {
776  atom_coord_[iat].copy_to(memory_t::device);
777  }
778  }
779  }
