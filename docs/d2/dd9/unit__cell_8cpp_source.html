39 std::vector<double>
40 Unit_cell::find_mt_radii()
41 {
42  if (nearest_neighbours_.size() == 0) {
43  TERMINATE("array of nearest neighbours is empty");
44  }
45 
46  std::vector<double> Rmt(num_atom_types(), 1e10);
47 
48  if (parameters_.auto_rmt() == 1) {
49  for (int ia = 0; ia < num_atoms(); ia++) {
50  int id1 = atom(ia).type_id();
51  if (nearest_neighbours_[ia].size() > 1) {
52  int ja = nearest_neighbours_[ia][1].atom_id;
53  int id2 = atom(ja).type_id();
54  /* don't allow spheres to touch: take a smaller value than half a distance */
55  double R = std::min(parameters_.rmt_max(), 0.95 * nearest_neighbours_[ia][1].distance / 2);
56  /* take minimal R for the given atom type */
57  Rmt[id1] = std::min(R, Rmt[id1]);
58  Rmt[id2] = std::min(R, Rmt[id2]);
59  } else {
60  Rmt[id1] = parameters_.rmt_max();
61  }
62  }
63  }
64 
65  if (parameters_.auto_rmt() == 2) {
66  std::vector<double> scale(num_atom_types(), 1e10);
67 
68  for (int ia = 0; ia < num_atoms(); ia++) {
69  int id1 = atom(ia).type_id();
70  if (nearest_neighbours_[ia].size() > 1) {
71  int ja = nearest_neighbours_[ia][1].atom_id;
72  int id2 = atom(ja).type_id();
73 
74  double d = nearest_neighbours_[ia][1].distance;
75  double s = 0.95 * d / (atom_type(id1).mt_radius() + atom_type(id2).mt_radius());
76  scale[id1] = std::min(s, scale[id1]);
77  scale[id2] = std::min(s, scale[id2]);
78  } else {
79  scale[id1] = parameters_.rmt_max() / atom_type(id1).mt_radius();
80  }
81  }
82 
83  for (int iat = 0; iat < num_atom_types(); iat++) {
84  Rmt[iat] = std::min(parameters_.rmt_max(), atom_type(iat).mt_radius() * scale[iat]);
85  }
86  }
87 
88  /* Suppose we have 3 different atoms. First we determint Rmt between 1st and 2nd atom,
89  * then we determine Rmt between (let's say) 2nd and 3rd atom and at this point we reduce
90  * the Rmt of the 2nd atom. This means that the 1st atom gets a possibility to expand if
91  * it is far from the 3rd atom. */
92  bool inflate = true;
93 
94  if (inflate) {
95  std::vector<bool> scale_Rmt(num_atom_types(), true);
96  for (int ia = 0; ia < num_atoms(); ia++) {
97  int id1 = atom(ia).type_id();
98 
99  if (nearest_neighbours_[ia].size() > 1) {
100  int ja = nearest_neighbours_[ia][1].atom_id;
101  int id2 = atom(ja).type_id();
102  double dist = nearest_neighbours_[ia][1].distance;
103 
104  if (Rmt[id1] + Rmt[id2] > dist * 0.94) {
105  scale_Rmt[id1] = false;
106  scale_Rmt[id2] = false;
107  }
108  }
109  }
110 
111  for (int ia = 0; ia < num_atoms(); ia++) {
112  int id1 = atom(ia).type_id();
113  if (nearest_neighbours_[ia].size() > 1) {
114  int ja = nearest_neighbours_[ia][1].atom_id;
115  int id2 = atom(ja).type_id();
116  double dist = nearest_neighbours_[ia][1].distance;
117 
118  if (scale_Rmt[id1] && !scale_Rmt[id2]) {
119  Rmt[id1] = std::min(parameters_.rmt_max(), 0.95 * (dist - Rmt[id2]));
120  scale_Rmt[id1] = false;
121  }
122  }
123  }
124  }
125 
126  for (int i = 0; i < num_atom_types(); i++) {
127  if (Rmt[i] < 0.3) {
128  std::stringstream s;
129  s << "muffin-tin radius for atom type " << i << " (" << atom_type(i).label()
130  << ") is too small: " << Rmt[i];
131  TERMINATE(s);
132  }
133  }
134 
135  return Rmt;
136 }

138 bool
139 Unit_cell::check_mt_overlap(int& ia__, int& ja__)
140 {
141  if (num_atoms() != 0 && nearest_neighbours_.size() == 0) {
142  TERMINATE("array of nearest neighbours is empty");
143  }
144 
145  for (int ia = 0; ia < num_atoms(); ia++) {
146  /* first atom is always the central one itself */
147  if (nearest_neighbours_[ia].size() <= 1) {
148  continue;
149  }
150 
151  int ja = nearest_neighbours_[ia][1].atom_id;
152  double dist = nearest_neighbours_[ia][1].distance;
153 
154  if (atom(ia).mt_radius() + atom(ja).mt_radius() >= dist) {
155  ia__ = ia;
156  ja__ = ja;
157  return true;
158  }
159  }
160 
161  return false;

250 unit_cell_parameters_descriptor
251 Unit_cell::unit_cell_parameters()
252 {
253  unit_cell_parameters_descriptor d;
254 
255  vector3d<double> v0(lattice_vectors_(0, 0), lattice_vectors_(1, 0), lattice_vectors_(2, 0));
256  vector3d<double> v1(lattice_vectors_(0, 1), lattice_vectors_(1, 1), lattice_vectors_(2, 1));
257  vector3d<double> v2(lattice_vectors_(0, 2), lattice_vectors_(1, 2), lattice_vectors_(2, 2));
258 
259  d.a = v0.length();
260  d.b = v1.length();
261  d.c = v2.length();
262 
263  d.alpha = std::acos(dot(v1, v2) / d.b / d.c) * 180 / pi;
264  d.beta = std::acos(dot(v0, v2) / d.a / d.c) * 180 / pi;
265  d.gamma = std::acos(dot(v0, v1) / d.a / d.b) * 180 / pi;
266 
267  return d;
268 }

340 void
341 Unit_cell::find_nearest_neighbours(double cluster_radius)
342 {
343  PROFILE("sirius::Unit_cell::find_nearest_neighbours");
344 
345  auto max_frac_coord = find_translations(cluster_radius, lattice_vectors_);
346 
347  nearest_neighbours_.clear();
348  nearest_neighbours_.resize(num_atoms());
349 
350  #pragma omp parallel for default(shared)
351  for (int ia = 0; ia < num_atoms(); ia++) {
352 
353  std::vector<nearest_neighbour_descriptor> nn;
354 
355  std::vector<std::pair<double, int>> nn_sort;
356 
357  for (int i0 = -max_frac_coord[0]; i0 <= max_frac_coord[0]; i0++) {
358  for (int i1 = -max_frac_coord[1]; i1 <= max_frac_coord[1]; i1++) {
359  for (int i2 = -max_frac_coord[2]; i2 <= max_frac_coord[2]; i2++) {
360  nearest_neighbour_descriptor nnd;
361  nnd.translation[0] = i0;
362  nnd.translation[1] = i1;
363  nnd.translation[2] = i2;
364 
365  for (int ja = 0; ja < num_atoms(); ja++) {
366  auto v1 = atom(ja).position() + vector3d<int>(nnd.translation) - atom(ia).position();
367  auto rc = get_cartesian_coordinates(v1);
368 
369  nnd.atom_id = ja;
370  nnd.rc = rc;
371  nnd.distance = rc.length();
372 
373  if (nnd.distance <= cluster_radius) {
374  nn.push_back(nnd);
375 
376  nn_sort.push_back(std::pair<double, int>(nnd.distance, (int)nn.size() - 1));
377  }
378  }
379  }
380  }
381  }
382 
383  std::sort(nn_sort.begin(), nn_sort.end());
384  nearest_neighbours_[ia].resize(nn.size());
385  for (int i = 0; i < (int)nn.size(); i++) {
386  nearest_neighbours_[ia][i] = nn[nn_sort[i].second];
387  }
388  }
389 }

421 bool
422 Unit_cell::is_point_in_mt(vector3d<double> vc, int& ja, int& jr, double& dr, double tp[2]) const
423 {
424  /* reduce coordinates to the primitive unit cell */
425  auto vr = reduce_coordinates(get_fractional_coordinates(vc));
426 
427  for (int ia = 0; ia < num_atoms(); ia++) {
428  for (int i0 = -1; i0 <= 1; i0++) {
429  for (int i1 = -1; i1 <= 1; i1++) {
430  for (int i2 = -1; i2 <= 1; i2++) {
431  /* atom position */
432  vector3d<double> posf = vector3d<double>(i0, i1, i2) + atom(ia).position();
433 
434  /* vector connecting center of atom and reduced point */
435  vector3d<double> vf = vr.first - posf;
436 
437  /* convert to spherical coordinates */
438  auto vs = SHT::spherical_coordinates(get_cartesian_coordinates(vf));
439 
440  if (vs[0] < atom(ia).mt_radius()) {
441  ja = ia;
442  tp[0] = vs[1]; // theta
443  tp[1] = vs[2]; // phi
444 
445  if (vs[0] < atom(ia).type().radial_grid(0)) {
446  jr = 0;
447  dr = 0.0;
448  } else {
449  for (int ir = 0; ir < atom(ia).num_mt_points() - 1; ir++) {
450  if (vs[0] >= atom(ia).type().radial_grid(ir) &&
451  vs[0] < atom(ia).type().radial_grid(ir + 1)) {
452  jr = ir;
453  dr = vs[0] - atom(ia).type().radial_grid(ir);
454  break;
455  }
456  }
457  }
458 
459  return true;
460  }
461  }
462  }
463  }
464  }
465  ja = -1;
466  jr = -1;
467  return false;
468 }

655 void
656 Unit_cell::get_symmetry()
657 {
658  PROFILE("sirius::Unit_cell::get_symmetry");
659 
660  if (num_atoms() == 0) {
661  return;
662  }
663 
664  if (atom_symmetry_classes_.size() != 0) {
665  atom_symmetry_classes_.clear();
666  for (int ia = 0; ia < num_atoms(); ia++) {
667  atom(ia).set_symmetry_class(nullptr);
668  }
669  }
670 
671  mdarray<double, 2> positions(3, num_atoms());
672  mdarray<double, 2> spins(3, num_atoms());
673  std::vector<int> types(num_atoms());
674  for (int ia = 0; ia < num_atoms(); ia++) {
675  auto vp = atom(ia).position();
676  auto vf = atom(ia).vector_field();
677  for (int x : {0, 1, 2}) {
678  positions(x, ia) = vp[x];
679  spins(x, ia) = vf[x];
680  }
681  types[ia] = atom(ia).type_id();
682  }
683 
684  symmetry_ = std::unique_ptr<Crystal_symmetry>(
685  new Crystal_symmetry(lattice_vectors_, num_atoms(), num_atom_types(), types, positions, spins,
686  parameters_.so_correction(), parameters_.spglib_tolerance(), parameters_.use_symmetry()));
687 
688  int atom_class_id{-1};
689  std::vector<int> asc(num_atoms(), -1);
690  for (int i = 0; i < num_atoms(); i++) {
691  /* if symmetry class is not assigned to this atom */
692  if (asc[i] == -1) {
693  /* take next id */
694  atom_class_id++;
695  atom_symmetry_classes_.push_back(std::shared_ptr<Atom_symmetry_class>(
696  new Atom_symmetry_class(atom_class_id, atom(i).type())));
697 
698  /* scan all atoms */
699  for (int j = 0; j < num_atoms(); j++) {
700  bool is_equal = (equivalent_atoms_.size())
701  ? (equivalent_atoms_[j] == equivalent_atoms_[i])
702  : (symmetry_->atom_symmetry_class(j) == symmetry_->atom_symmetry_class(i));
703  /* assign new class id for all equivalent atoms */
704  if (is_equal) {
705  asc[j] = atom_class_id;
706  atom_symmetry_classes_.back()->add_atom_id(j);
707  }
708  }
709  }
710  }
711 
712  for (auto e : atom_symmetry_classes_) {
713  for (int i = 0; i < e->num_atoms(); i++) {
714  int ia = e->atom_id(i);
715  atom(ia).set_symmetry_class(e);
716  }
717  }
718 
719  assert(num_atom_symmetry_classes() != 0);
720 }

722 void
723 Unit_cell::import(config_t::unit_cell_t const &inp__)
724 {
725  auto lv = matrix3d<double>(inp__.lattice_vectors());
726  lv *= inp__.lattice_vectors_scale();
727  set_lattice_vectors(vector3d<double>(lv(0, 0), lv(0, 1), lv(0, 2)),
728  vector3d<double>(lv(1, 0), lv(1, 1), lv(1, 2)),
729  vector3d<double>(lv(2, 0), lv(2, 1), lv(2, 2)));
730  /* here lv are copied from the JSON dictionary as three row vectors; however
731  in the code the lattice vectors are stored as three column vectors, so
732  transposition is needed here */
733  auto ilvT = transpose(inverse(lv));
734 
735  auto units = inp__.atom_coordinate_units();
736 
737  /* first, load all types */
738  for (auto label: inp__.atom_types()) {
739  auto fname = inp__.atom_files(label);
740  add_atom_type(label, fname);
741  }
742  for (auto label : inp__.atom_types()) {
743  for (auto v : inp__.atoms(label)) {
744  vector3d<double> p(v[0], v[1], v[2]);
745  vector3d<double> f;
746  if (v.size() == 6) {
747  f = vector3d<double>(v[3], v[4], v[5]);
748  }
749  /* convert to atomic units */
750  if (units == "A") {
751  for (int x : {0, 1, 2}) {
752  p[x] /= bohr_radius;
753  }
754  }
755  /* convert from Cartesian to lattice coordinates */
756  if (units == "au" || units == "A") {
757  p = dot(ilvT, p);
758  auto rc = reduce_coordinates(p);
759  for (int x : {0, 1, 2}) {
760  p[x] = rc.first[x];
761  }
762  }
763  add_atom(label, p, f);
764  }
765  }

768 void
769 Unit_cell::update()
770 {
771  PROFILE("sirius::Unit_cell::update");
772 
773  auto v0 = lattice_vector(0);
774  auto v1 = lattice_vector(1);
775  auto v2 = lattice_vector(2);
776 
777  double r = std::max(std::max(v0.length(), std::max(v1.length(), v2.length())),
778  parameters_.cfg().parameters().nn_radius());
779 
780  find_nearest_neighbours(r);
781 
782  if (parameters_.full_potential()) {
783  /* find new MT radii and initialize radial grid */
784  if (parameters_.auto_rmt()) {
785  auto rg = get_radial_grid_t(parameters_.cfg().settings().radial_grid());
786  auto Rmt = find_mt_radii();
787  for (int iat = 0; iat < num_atom_types(); iat++) {
788  double r0 = atom_type(iat).radial_grid().first();
789 
790  atom_type(iat).set_radial_grid(rg.first, atom_type(iat).num_mt_points(), r0, Rmt[iat], rg.second);
791  }
792  }
793 
794  int ia, ja;
795  if (check_mt_overlap(ia, ja)) {
796  std::stringstream s;
797  s << "overlaping muffin-tin spheres for atoms " << ia << "(" << atom(ia).type().symbol() << ")"
798  << " and " << ja << "(" << atom(ja).type().symbol() << ")" << std::endl
799  << " radius of atom " << ia << " : " << atom(ia).mt_radius() << std::endl
800  << " radius of atom " << ja << " : " << atom(ja).mt_radius() << std::endl
801  << " distance : " << nearest_neighbours_[ia][1].distance << " " << nearest_neighbours_[ja][1].distance;
802  TERMINATE(s);
803  }
804 
805  min_mt_radius_ = 1e100;
806  max_mt_radius_ = 0;
807  for (int i = 0; i < num_atom_types(); i++) {
808  min_mt_radius_ = std::min(min_mt_radius_, atom_type(i).mt_radius());
809  max_mt_radius_ = std::max(max_mt_radius_, atom_type(i).mt_radius());
810  }
811  }
812 
813  get_symmetry();
814 
815  spl_num_atom_symmetry_classes_ = splindex<splindex_t::block>(num_atom_symmetry_classes(), comm_.size(), comm_.rank());
816 
817  volume_mt_ = 0.0;
818  if (parameters_.full_potential()) {
819  for (int ia = 0; ia < num_atoms(); ia++) {
820  volume_mt_ += fourpi * std::pow(atom(ia).mt_radius(), 3) / 3.0;
821  }
822  }
823 
824  volume_it_ = omega() - volume_mt_;
825 
826  for (int iat = 0; iat < num_atom_types(); iat++) {
827  int nat = atom_type(iat).num_atoms();
828  if (nat > 0) {
829  for (int i = 0; i < nat; i++) {
830  int ia = atom_type(iat).atom_id(i);
831  for (int x: {0, 1, 2}) {
832  atom_coord_[iat](i, x) = atom(ia).position()[x];
833  }
834  }
835  if (parameters_.processing_unit() == device_t::GPU) {
836  atom_coord_[iat].copy_to(memory_t::device);
837  }
838  }
839  }
