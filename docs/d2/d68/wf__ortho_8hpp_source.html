25 /// Orthogonalize n new wave-functions to the N old wave-functions
26 template <typename T, int idx_bra__, int idx_ket__>
27 inline void orthogonalize(device_t pu__,
28  int ispn__,
29  std::vector<Wave_functions*> wfs__,
30  int N__,
31  int n__,
32  dmatrix<T>& o__,
33  Wave_functions& tmp__)
34 {
35  PROFILE("sddk::Wave_functions::orthogonalize");
36 
37  const char* sddk_pp_raw = std::getenv("SDDK_PRINT_PERFORMANCE");
38  int sddk_pp = (sddk_pp_raw == NULL) ? 0 : std::atoi(sddk_pp_raw);
39 
40  auto& comm = wfs__[0]->comm();
41 
42 #ifdef __GPU
43  if (pu__ == GPU) {
44  acc::set_device();
45  }
46 #endif
47  int K{0};
48  if (sddk_pp) {
49  K = wfs__[0]->gkvec().num_gvec() + wfs__[0]->num_mt_coeffs();
50  if (std::is_same<T, double>::value) {
51  K *= 2;
52  }
53  }
54 
55  const char* sddk_debug_raw = std::getenv("SDDK_DEBUG");
56  int sddk_debug = (sddk_debug_raw == NULL) ? 0 : std::atoi(sddk_debug_raw);
57 
58  double ngop{0};
59  if (std::is_same<T, double>::value) {
60  ngop = 2e-9;
61  }
62  if (std::is_same<T, double_complex>::value) {
63  ngop = 8e-9;
64  }
65 
66  if (sddk_pp) {
67  comm.barrier();
68  }
69  //double time = -omp_get_wtime();
70 
71  double gflops{0};
72 
73  /* project out the old subspace:
74  * |\tilda phi_new> = |phi_new> - |phi_old><phi_old|phi_new> */
75  if (N__ > 0) {
76  inner(pu__, ispn__, *wfs__[idx_bra__], 0, N__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
77  transform(pu__, ispn__, -1.0, wfs__, 0, N__, o__, 0, 0, 1.0, wfs__, N__, n__);
78 
79  if (sddk_pp) {
80  gflops += static_cast<int>(1 + wfs__.size()) * ngop * N__ * n__ * K; // inner and transfrom have the same number of flops
81  }
82  }
83 
84  if (sddk_debug >= 2) {
85  if (o__.comm().rank() == 0) {
86  printf("check QR decomposition, matrix size : %i\n", n__);
87  }
88  inner(pu__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
89 
90  linalg<CPU>::geqrf(n__, n__, o__, 0, 0);
91  auto diag = o__.get_diag(n__);
92  if (o__.comm().rank() == 0) {
93  for (int i = 0; i < n__; i++) {
94  if (std::abs(diag[i]) < 1e-6) {
95  std::cout << "small norm: " << i << " " << diag[i] << std::endl;
96  }
97  }
98  }
99 
100  if (o__.comm().rank() == 0) {
101  printf("check eigen-values, matrix size : %i\n", n__);
102  }
103  inner(pu__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
104 
105  //if (sddk_debug >= 3) {
106  // save_to_hdf5("nxn_overlap.h5", o__, n__);
107  //}
108 
109  std::vector<double> eo(n__);
110  dmatrix<T> evec(o__.num_rows(), o__.num_cols(), o__.blacs_grid(), o__.bs_row(), o__.bs_col());
111 
112  auto solver = Eigensolver_factory<T>(ev_solver_t::scalapack);
113  solver->solve(n__, o__, eo.data(), evec);
114 
115  if (o__.comm().rank() == 0) {
116  for (int i = 0; i < n__; i++) {
117  if (eo[i] < 1e-6) {
118  std::cout << "small eigen-value " << i << " " << eo[i] << std::endl;
119  }
120  }
121  }
122  }
123 
124  /* orthogonalize new n__ x n__ block */
125  inner(pu__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
126 
127  if (sddk_debug >= 1) {
128  if (o__.comm().rank() == 0) {
129  printf("check diagonal\n");
130  }
131  auto diag = o__.get_diag(n__);
132  for (int i = 0; i < n__; i++) {
133  if (std::real(diag[i]) <= 0 || std::imag(diag[i]) > 1e-12) {
134  std::cout << "wrong diagonal: " << i << " " << diag[i] << std::endl;
135  }
136  }
137  if (o__.comm().rank() == 0) {
138  printf("check hermitian\n");
139  }
140  double d = check_hermitian(o__, n__);
141  if (d > 1e-12 && o__.comm().rank() == 0) {
142  std::stringstream s;
143  s << "matrix is not hermitian, max diff = " << d;
144  WARNING(s);
145  }
146  }
147 
148  if (sddk_pp) {
149  gflops += ngop * n__ * n__ * K;
150  }
151 
152  /* single MPI rank */
153  if (o__.comm().size() == 1) {
154  bool use_magma{false};
155  #if defined(__GPU) && defined(__MAGMA)
156  if (pu__ == GPU) {
157  use_magma = true;
158  }
159  #endif
160 
161  if (use_magma) {
162  #ifdef __GPU
163  /* Cholesky factorization */
164  if (int info = linalg<GPU>::potrf(n__, o__.template at<GPU>(), o__.ld())) {
165  std::stringstream s;
166  s << "error in GPU factorization, info = " << info;
167  TERMINATE(s);
168  }
169  /* inversion of triangular matrix */
170  if (linalg<GPU>::trtri(n__, o__.template at<GPU>(), o__.ld())) {
171  TERMINATE("error in inversion");
172  }
173  #endif
174  } else { /* CPU version */
175  /* Cholesky factorization */
176  if (int info = linalg<CPU>::potrf(n__, &o__(0, 0), o__.ld())) {
177  std::stringstream s;
178  s << "error in factorization, info = " << info << std::endl
179  << "number of existing states: " << N__ << std::endl
180  << "number of new states: " << n__ << std::endl
181  << "number of wave_functions: " << wfs__.size() << std::endl
182  << "idx_bra: " << idx_bra__ << " " << "idx_ket:" << idx_ket__;
183  TERMINATE(s);
184  }
185  /* inversion of triangular matrix */
186  if (linalg<CPU>::trtri(n__, &o__(0, 0), o__.ld())) {
187  TERMINATE("error in inversion");
188  }
189  if (pu__ == GPU) {
190  #ifdef __GPU
191  acc::copyin(o__.template at<GPU>(), o__.ld(), o__.template at<CPU>(), o__.ld(), n__, n__);
192  #endif
193  }
194  }
195 
196  int s0{0};
197  int s1{1};
198  if (ispn__ != 2) {
199  s0 = s1 = ispn__;
200  }
201  for (int s = s0; s <= s1; s++) {
202  /* CPU version */
203  if (pu__ == CPU) {
204  /* multiplication by triangular matrix */
205  for (auto& e: wfs__) {
206  /* wave functions are complex, transformation matrix is complex */
207  if (std::is_same<T, double_complex>::value) {
208  linalg<CPU>::trmm('R', 'U', 'N', e->pw_coeffs(s).num_rows_loc(), n__, double_complex(1, 0),
209  reinterpret_cast<double_complex*>(o__.template at<CPU>()), o__.ld(),
210  e->pw_coeffs(s).prime().at<CPU>(0, N__), e->pw_coeffs(s).prime().ld());
211 
212  if (e->has_mt()) {
213  linalg<CPU>::trmm('R', 'U', 'N', e->mt_coeffs(s).num_rows_loc(), n__, double_complex(1, 0),
214  reinterpret_cast<double_complex*>(o__.template at<CPU>()), o__.ld(),
215  e->mt_coeffs(s).prime().at<CPU>(0, N__), e->mt_coeffs(s).prime().ld());
216  }
217  }
218  /* wave functions are real (psi(G) = psi^{*}(-G)), transformation matrix is real */
219  if (std::is_same<T, double>::value) {
220  linalg<CPU>::trmm('R', 'U', 'N', 2 * e->pw_coeffs(s).num_rows_loc(), n__, 1.0,
221  reinterpret_cast<double*>(o__.template at<CPU>()), o__.ld(),
222  reinterpret_cast<double*>(e->pw_coeffs(s).prime().at<CPU>(0, N__)), 2 * e->pw_coeffs(s).prime().ld());
223 
224  if (e->has_mt()) {
225  linalg<CPU>::trmm('R', 'U', 'N', 2 * e->mt_coeffs(s).num_rows_loc(), n__, 1.0,
226  reinterpret_cast<double*>(o__.template at<CPU>()), o__.ld(),
227  reinterpret_cast<double*>(e->mt_coeffs(s).prime().at<CPU>(0, N__)), 2 * e->mt_coeffs(s).prime().ld());
228  }
229  }
230  }
231  }
232  #ifdef __GPU
233  if (pu__ == GPU) {
234  /* multiplication by triangular matrix */
235  for (auto& e: wfs__) {
236  if (std::is_same<T, double_complex>::value) {
237  double_complex alpha(1, 0);
238 
239  linalg<GPU>::trmm('R', 'U', 'N', e->pw_coeffs(s).num_rows_loc(), n__, &alpha,
240  reinterpret_cast<double_complex*>(o__.template at<GPU>()), o__.ld(),
241  e->pw_coeffs(s).prime().at<GPU>(0, N__), e->pw_coeffs(s).prime().ld());
242 
243  if (e->has_mt()) {
244  linalg<GPU>::trmm('R', 'U', 'N', e->mt_coeffs(s).num_rows_loc(), n__, &alpha,
245  reinterpret_cast<double_complex*>(o__.template at<GPU>()), o__.ld(),
246  e->mt_coeffs(s).prime().at<GPU>(0, N__), e->mt_coeffs(s).prime().ld());
247  }
248  /* alpha should not go out of the scope, so wait */
249  acc::sync_stream(-1);
250  }
251  if (std::is_same<T, double>::value) {
252  double alpha{1};
253 
254  linalg<GPU>::trmm('R', 'U', 'N', 2 * e->pw_coeffs(s).num_rows_loc(), n__, &alpha,
255  reinterpret_cast<double*>(o__.template at<GPU>()), o__.ld(),
256  reinterpret_cast<double*>(e->pw_coeffs(s).prime().at<GPU>(0, N__)), 2 * e->pw_coeffs(s).prime().ld());
257 
258  if (e->has_mt()) {
259  linalg<GPU>::trmm('R', 'U', 'N', 2 * e->mt_coeffs(s).num_rows_loc(), n__, &alpha,
260  reinterpret_cast<double*>(o__.template at<GPU>()), o__.ld(),
261  reinterpret_cast<double*>(e->mt_coeffs(s).prime().at<GPU>(0, N__)), 2 * e->mt_coeffs(s).prime().ld());
262  }
263  acc::sync_stream(-1);
264  }
265  }
266  acc::sync_stream(-1);
267  }
268  #endif
269  }
270  } else { /* parallel transformation */
271  utils::timer t1("sddk::Wave_functions::orthogonalize|potrf");
272  mdarray<T, 1> diag;
273  o__.make_real_diag(n__);
274  if (sddk_debug >= 1) {
275  diag = o__.get_diag(n__);
276  }
277  if (int info = linalg<CPU>::potrf(n__, o__)) {
278  std::stringstream s;
279  s << "error in Cholesky factorization, info = " << info << ", matrix size = " << n__;
280  if (sddk_debug >= 1) {
281  s << std::endl << " diag = " << diag[info - 1];
282  }
283  TERMINATE(s);
284  }
285  t1.stop();
286 
287  utils::timer t2("sddk::Wave_functions::orthogonalize|trtri");
288  if (linalg<CPU>::trtri(n__, o__)) {
289  TERMINATE("error in inversion");
290  }
291  t2.stop();
292 
293  /* o is upper triangular matrix */
294  for (int i = 0; i < n__; i++) {
295  for (int j = i + 1; j < n__; j++) {
296  o__.set(j, i, 0);
297  }
298  }
299 
300  /* phi is transformed into phi, so we can't use it as the output buffer; use tmp instead and then overwrite phi */
301  for (auto& e: wfs__) {
302  transform(pu__, ispn__, *e, N__, n__, o__, 0, 0, tmp__, 0, n__);
303  int s0{0};
304  int s1{1};
305  if (ispn__ != 2) {
306  s0 = s1 = ispn__;
307  }
308  for (int s = s0; s <= s1; s++) {
309  e->copy_from(pu__, n__, tmp__, s, 0, s, N__);
310  }
311  }
312  }
313 }
