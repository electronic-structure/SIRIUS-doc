25 /// Orthogonalize n new wave-functions to the N old wave-functions
26 template <typename T, int idx_bra__, int idx_ket__>
27 inline void orthogonalize(device_t pu__,
28  int ispn__,
29  std::vector<Wave_functions*> wfs__,
30  int N__,
31  int n__,
32  dmatrix<T>& o__,
33  Wave_functions& tmp__)
34 {
35  PROFILE("sddk::orthogonalize");
36 
37  const char* sddk_pp_raw = std::getenv("SDDK_PRINT_PERFORMANCE");
38  int sddk_pp = (sddk_pp_raw == NULL) ? 0 : std::atoi(sddk_pp_raw);
39 
40  auto& comm = wfs__[0]->comm();
41 
42 #ifdef __GPU
43  if (pu__ == GPU) {
44  acc::set_device();
45  }
46 #endif
47  int K{0};
48  if (sddk_pp) {
49  K = wfs__[0]->gkvec().num_gvec() + wfs__[0]->num_mt_coeffs();
50  if (std::is_same<T, double>::value) {
51  K *= 2;
52  }
53  }
54 
55  const char* sddk_debug_raw = std::getenv("SDDK_DEBUG");
56  int sddk_debug = (sddk_debug_raw == NULL) ? 0 : std::atoi(sddk_debug_raw);
57 
58  double ngop{0};
59  if (std::is_same<T, double>::value) {
60  ngop = 2e-9;
61  }
62  if (std::is_same<T, double_complex>::value) {
63  ngop = 8e-9;
64  }
65 
66  if (sddk_pp) {
67  comm.barrier();
68  }
69  //double time = -omp_get_wtime();
70 
71  double gflops{0};
72 
73  /* project out the old subspace:
74  * |\tilda phi_new> = |phi_new> - |phi_old><phi_old|phi_new> */
75  if (N__ > 0) {
76  inner(pu__, ispn__, *wfs__[idx_bra__], 0, N__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
77  transform(pu__, ispn__, -1.0, wfs__, 0, N__, o__, 0, 0, 1.0, wfs__, N__, n__);
78 
79  if (sddk_pp) {
80  gflops += static_cast<int>(1 + wfs__.size()) * ngop * N__ * n__ * K; // inner and transfrom have the same number of flops
81  }
82  }
83 
84  if (sddk_debug >= 2) {
85  if (o__.comm().rank() == 0) {
86  printf("check QR decomposition, matrix size : %i\n", n__);
87  }
88  inner(pu__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
89 
90  linalg<CPU>::geqrf(n__, n__, o__, 0, 0);
91  auto diag = o__.get_diag(n__);
92  if (o__.comm().rank() == 0) {
93  for (int i = 0; i < n__; i++) {
94  if (std::abs(diag[i]) < 1e-6) {
95  std::cout << "small norm: " << i << " " << diag[i] << std::endl;
96  }
97  }
98  }
99 
100  if (o__.comm().rank() == 0) {
101  printf("check eigen-values, matrix size : %i\n", n__);
102  }
103  inner(pu__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
104 
105  //if (sddk_debug >= 3) {
106  // save_to_hdf5("nxn_overlap.h5", o__, n__);
107  //}
108 
109  std::vector<double> eo(n__);
110  dmatrix<T> evec(o__.num_rows(), o__.num_cols(), o__.blacs_grid(), o__.bs_row(), o__.bs_col());
111 
112  auto solver = Eigensolver_factory<T>(ev_solver_t::scalapack);
113  solver->solve(n__, o__, eo.data(), evec);
114 
115  if (o__.comm().rank() == 0) {
116  for (int i = 0; i < n__; i++) {
117  if (eo[i] < 1e-6) {
118  std::cout << "small eigen-value " << i << " " << eo[i] << std::endl;
119  }
120  }
121  }
122  }
123 
124  /* orthogonalize new n__ x n__ block */
125  inner(pu__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
126 
127  if (sddk_debug >= 1) {
128  if (o__.comm().rank() == 0) {
129  printf("check diagonal\n");
130  }
131  auto diag = o__.get_diag(n__);
132  for (int i = 0; i < n__; i++) {
133  if (std::real(diag[i]) <= 0 || std::imag(diag[i]) > 1e-12) {
134  std::cout << "wrong diagonal: " << i << " " << diag[i] << std::endl;
135  }
136  }
137  if (o__.comm().rank() == 0) {
138  printf("check hermitian\n");
139  }
140  double d = check_hermitian(o__, n__);
141  if (d > 1e-12 && o__.comm().rank() == 0) {
142  std::stringstream s;
143  s << "matrix is not hermitian, max diff = " << d;
144  WARNING(s);
145  }
146  }
147 
148  if (sddk_pp) {
149  gflops += ngop * n__ * n__ * K;
150  }
151 
152  /* single MPI rank */
153  if (o__.comm().size() == 1) {
154  bool use_magma{false};
155 
156 // MAGMA performance for Cholesky and inversion is not good enough; use lapack for the moment
157 //#if defined(__GPU) && defined(__MAGMA)
158 // if (pu__ == GPU) {
159 // use_magma = true;
160 // }
161 //#endif
162 
163  utils::timer t1("sddk::orthogonalize|tmtrx");
164  if (use_magma) {
165 #ifdef __GPU
166  /* Cholesky factorization */
167  if (int info = linalg<GPU>::potrf(n__, o__.at(memory_t::device), o__.ld())) {
168  std::stringstream s;
169  s << "error in GPU factorization, info = " << info;
170  TERMINATE(s);
171  }
172  /* inversion of triangular matrix */
173  if (linalg<GPU>::trtri(n__, o__.at(memory_t::device), o__.ld())) {
174  TERMINATE("error in inversion");
175  }
176 #endif
177  } else { /* CPU version */
178  /* Cholesky factorization */
179  if (int info = linalg<CPU>::potrf(n__, &o__(0, 0), o__.ld())) {
180  std::stringstream s;
181  s << "error in factorization, info = " << info << std::endl
182  << "number of existing states: " << N__ << std::endl
183  << "number of new states: " << n__ << std::endl
184  << "number of wave_functions: " << wfs__.size() << std::endl
185  << "idx_bra: " << idx_bra__ << " " << "idx_ket:" << idx_ket__;
186  TERMINATE(s);
187  }
188  /* inversion of triangular matrix */
189  if (linalg<CPU>::trtri(n__, &o__(0, 0), o__.ld())) {
190  TERMINATE("error in inversion");
191  }
192  if (pu__ == GPU) {
193 #ifdef __GPU
194  acc::copyin(o__.at(memory_t::device), o__.ld(), o__.at(memory_t::host), o__.ld(), n__, n__);
195 #endif
196  }
197  }
198  t1.stop();
199 
200  utils::timer t2("sddk::orthogonalize|transform");
201  int s0{0};
202  int s1{1};
203  if (ispn__ != 2) {
204  s0 = s1 = ispn__;
205  }
206  for (int s = s0; s <= s1; s++) {
207  /* CPU version */
208  if (pu__ == CPU) {
209  /* multiplication by triangular matrix */
210  for (auto& e: wfs__) {
211  /* wave functions are complex, transformation matrix is complex */
212  if (std::is_same<T, double_complex>::value) {
213  linalg<CPU>::trmm('R', 'U', 'N', e->pw_coeffs(s).num_rows_loc(), n__, double_complex(1, 0),
214  reinterpret_cast<double_complex*>(o__.at(memory_t::host)), o__.ld(),
215  e->pw_coeffs(s).prime().at(memory_t::host, 0, N__), e->pw_coeffs(s).prime().ld());
216 
217  if (e->has_mt()) {
218  linalg<CPU>::trmm('R', 'U', 'N', e->mt_coeffs(s).num_rows_loc(), n__, double_complex(1, 0),
219  reinterpret_cast<double_complex*>(o__.at(memory_t::host)), o__.ld(),
220  e->mt_coeffs(s).prime().at(memory_t::host, 0, N__), e->mt_coeffs(s).prime().ld());
221  }
222  }
223  /* wave functions are real (psi(G) = psi^{*}(-G)), transformation matrix is real */
224  if (std::is_same<T, double>::value) {
225  linalg<CPU>::trmm('R', 'U', 'N', 2 * e->pw_coeffs(s).num_rows_loc(), n__, 1.0,
226  reinterpret_cast<double*>(o__.at(memory_t::host)), o__.ld(),
227  reinterpret_cast<double*>(e->pw_coeffs(s).prime().at(memory_t::host, 0, N__)), 2 * e->pw_coeffs(s).prime().ld());
228 
229  if (e->has_mt()) {
230  linalg<CPU>::trmm('R', 'U', 'N', 2 * e->mt_coeffs(s).num_rows_loc(), n__, 1.0,
231  reinterpret_cast<double*>(o__.at(memory_t::host)), o__.ld(),
232  reinterpret_cast<double*>(e->mt_coeffs(s).prime().at(memory_t::host, 0, N__)), 2 * e->mt_coeffs(s).prime().ld());
233  }
234  }
235  }
236  }
237 #ifdef __GPU
238  if (pu__ == GPU) {
239  /* multiplication by triangular matrix */
240  for (auto& e: wfs__) {
241  if (std::is_same<T, double_complex>::value) {
242  double_complex alpha(1, 0);
243 
244  linalg<GPU>::trmm('R', 'U', 'N', e->pw_coeffs(s).num_rows_loc(), n__, &alpha,
245  reinterpret_cast<double_complex*>(o__.at(memory_t::device)), o__.ld(),
246  e->pw_coeffs(s).prime().at(memory_t::device, 0, N__), e->pw_coeffs(s).prime().ld());
247 
248  if (e->has_mt()) {
249  linalg<GPU>::trmm('R', 'U', 'N', e->mt_coeffs(s).num_rows_loc(), n__, &alpha,
250  reinterpret_cast<double_complex*>(o__.at(memory_t::device)), o__.ld(),
251  e->mt_coeffs(s).prime().at(memory_t::device, 0, N__), e->mt_coeffs(s).prime().ld());
252  }
253  /* alpha should not go out of the scope, so wait */
254  acc::sync_stream(stream_id(-1));
255  }
256  if (std::is_same<T, double>::value) {
257  double alpha{1};
258 
259  linalg<GPU>::trmm('R', 'U', 'N', 2 * e->pw_coeffs(s).num_rows_loc(), n__, &alpha,
260  reinterpret_cast<double*>(o__.at(memory_t::device)), o__.ld(),
261  reinterpret_cast<double*>(e->pw_coeffs(s).prime().at(memory_t::device, 0, N__)), 2 * e->pw_coeffs(s).prime().ld());
262 
263  if (e->has_mt()) {
264  linalg<GPU>::trmm('R', 'U', 'N', 2 * e->mt_coeffs(s).num_rows_loc(), n__, &alpha,
265  reinterpret_cast<double*>(o__.at(memory_t::device)), o__.ld(),
266  reinterpret_cast<double*>(e->mt_coeffs(s).prime().at(memory_t::device, 0, N__)), 2 * e->mt_coeffs(s).prime().ld());
267  }
268  acc::sync_stream(stream_id(-1));
269  }
270  }
271  acc::sync_stream(stream_id(-1));
272  }
273 #endif
274  }
275  t2.stop();
276  } else { /* parallel transformation */
277  utils::timer t1("sddk::orthogonalize|potrf");
278  mdarray<T, 1> diag;
279  o__.make_real_diag(n__);
280  if (sddk_debug >= 1) {
281  diag = o__.get_diag(n__);
282  }
283  if (int info = linalg<CPU>::potrf(n__, o__)) {
284  std::stringstream s;
285  s << "error in Cholesky factorization, info = " << info << ", matrix size = " << n__;
286  if (sddk_debug >= 1) {
287  s << std::endl << " diag = " << diag[info - 1];
288  }
289  TERMINATE(s);
290  }
291  t1.stop();
292 
293  utils::timer t2("sddk::orthogonalize|trtri");
294  if (linalg<CPU>::trtri(n__, o__)) {
295  TERMINATE("error in inversion");
296  }
297  t2.stop();
298 
299  /* o is upper triangular matrix */
300  for (int i = 0; i < n__; i++) {
301  for (int j = i + 1; j < n__; j++) {
302  o__.set(j, i, 0);
303  }
304  }
305 
306  /* phi is transformed into phi, so we can't use it as the output buffer; use tmp instead and then overwrite phi */
307  for (auto& e: wfs__) {
308  transform(pu__, ispn__, *e, N__, n__, o__, 0, 0, tmp__, 0, n__);
309  int s0{0};
310  int s1{1};
311  if (ispn__ != 2) {
312  s0 = s1 = ispn__;
313  }
314  for (int s = s0; s <= s1; s++) {
315  e->copy_from(pu__, n__, tmp__, s, 0, s, N__);
316  }
317  }
318  }
319 }
