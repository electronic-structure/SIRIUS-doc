43  /// Generate plane-wave coefficients for beta-projectors of atom types.
44  void generate_pw_coefs_t()
45  {
46  PROFILE("sirius::Beta_projectors::generate_pw_coefs_t");
47  auto& bchunk = ctx_.beta_projector_chunks();
48  if (!bchunk.num_beta_t()) {
49  return;
50  }
51 
52  auto& comm = gkvec_.comm();
53 
54  auto& beta_radial_integrals = ctx_.beta_ri();
55 
56  /* compute <G+k|beta> */
57  #pragma omp parallel for
58  for (int igkloc = 0; igkloc < gkvec_.gvec_count(comm.rank()); igkloc++) {
59  int igk = gkvec_.gvec_offset(comm.rank()) + igkloc;
60  /* vs = {r, theta, phi} */
61  auto vs = SHT::spherical_coordinates(gkvec_.gkvec_cart(igk));
62  /* compute real spherical harmonics for G+k vector */
63  std::vector<double> gkvec_rlm(Utils::lmmax(lmax_beta_));
64  SHT::spherical_harmonics(lmax_beta_, vs[1], vs[2], &gkvec_rlm[0]);
65  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
66  auto& atom_type = ctx_.unit_cell().atom_type(iat);
67  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
68  int l = atom_type.indexb(xi).l;
69  int lm = atom_type.indexb(xi).lm;
70  int idxrf = atom_type.indexb(xi).idxrf;
71 
72  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(ctx_.unit_cell().omega());
73  pw_coeffs_t_[0](igkloc, atom_type.offset_lo() + xi) = z * gkvec_rlm[lm] * beta_radial_integrals.value(idxrf, iat, vs[0]);
74  }
75  }
76  }
77 
78  if (ctx_.control().print_checksum_) {
79  auto c1 = pw_coeffs_t_[0].checksum();
80  comm.allreduce(&c1, 1);
81  if (comm.rank() == 0) {
82  DUMP("checksum(beta_pw_coeffs_t) : %18.10f %18.10f", c1.real(), c1.imag())
83  }
84  }
85 
86  if (ctx_.processing_unit() == GPU) {
87  pw_coeffs_t_[0].copy<memory_t::host, memory_t::device>();
88  }
89  }
