26 inline void Band::set_fv_h_o<CPU, electronic_structure_method_t::full_potential_lapwlo>(K_point* kp__,
27  Potential const& potential__,
28  dmatrix<double_complex>& h__,
29  dmatrix<double_complex>& o__) const
30 {
31  PROFILE("sirius::Band::set_fv_h_o");
32 
33  h__.zero();
34  o__.zero();
35 
36  double_complex zone(1, 0);
37 
38  int num_atoms_in_block = 2 * omp_get_max_threads();
39  int nblk = unit_cell_.num_atoms() / num_atoms_in_block +
40  std::min(1, unit_cell_.num_atoms() % num_atoms_in_block);
41 
42  int max_mt_aw = num_atoms_in_block * unit_cell_.max_mt_aw_basis_size();
43 
44  if (kp__->comm().rank() == 0 && ctx_.control().verbosity_ >= 2) {
45  DUMP("nblk: %i", nblk);
46  DUMP("max_mt_aw: %i", max_mt_aw);
47  }
48 
49  mdarray<double_complex, 2> alm_row(kp__->num_gkvec_row(), max_mt_aw);
50  mdarray<double_complex, 2> alm_col(kp__->num_gkvec_col(), max_mt_aw);
51  mdarray<double_complex, 2> halm_col(kp__->num_gkvec_col(), max_mt_aw);
52  mdarray<double_complex, 2> oalm_col;
53  if (ctx_.valence_relativity() == relativity_t::iora) {
54  oalm_col = mdarray<double_complex, 2>(kp__->num_gkvec_col(), max_mt_aw);
55  } else {
56  oalm_col = mdarray<double_complex, 2>(alm_col.at<CPU>(), kp__->num_gkvec_col(), max_mt_aw);
57  }
58 
59  sddk::timer t1("sirius::Band::set_fv_h_o|zgemm");
60  /* loop over blocks of atoms */
61  for (int iblk = 0; iblk < nblk; iblk++) {
62  /* number of matching AW coefficients in the block */
63  int num_mt_aw{0};
64  /* offsets for matching coefficients of individual atoms in the AW block */
65  std::vector<int> offsets(num_atoms_in_block);
66  for (int ia = iblk * num_atoms_in_block; ia < std::min(unit_cell_.num_atoms(), (iblk + 1) * num_atoms_in_block); ia++) {
67  offsets[ia - iblk * num_atoms_in_block] = num_mt_aw;
68  num_mt_aw += unit_cell_.atom(ia).type().mt_aw_basis_size();
69  }
70 
71  if (ctx_.control().print_checksum_) {
72  alm_row.zero();
73  alm_col.zero();
74  halm_col.zero();
75  }
76 
77  #pragma omp parallel
78  {
79  int tid = omp_get_thread_num();
80  for (int ia = iblk * num_atoms_in_block; ia < std::min(unit_cell_.num_atoms(), (iblk + 1) * num_atoms_in_block); ia++) {
81  if (ia % omp_get_num_threads() == tid) {
82  int ialoc = ia - iblk * num_atoms_in_block;
83  auto& atom = unit_cell_.atom(ia);
84  auto& type = atom.type();
85  int naw = type.mt_aw_basis_size();
86 
87  mdarray<double_complex, 2> alm_row_tmp(alm_row.at<CPU>(0, offsets[ialoc]), kp__->num_gkvec_row(), naw);
88  mdarray<double_complex, 2> alm_col_tmp(alm_col.at<CPU>(0, offsets[ialoc]), kp__->num_gkvec_col(), naw);
89  mdarray<double_complex, 2> halm_col_tmp(halm_col.at<CPU>(0, offsets[ialoc]), kp__->num_gkvec_col(), naw);
90  mdarray<double_complex, 2> oalm_col_tmp;
91  if (ctx_.valence_relativity() == relativity_t::iora) {
92  oalm_col_tmp = mdarray<double_complex, 2>(oalm_col.at<CPU>(0, offsets[ialoc]), kp__->num_gkvec_col(), naw);
93  } else {
94  oalm_col_tmp = mdarray<double_complex, 2>(alm_col.at<CPU>(0, offsets[ialoc]), kp__->num_gkvec_col(), naw);
95  }
96 
97  kp__->alm_coeffs_row().generate(ia, alm_row_tmp);
98  for (int xi = 0; xi < naw; xi++) {
99  for (int igk = 0; igk < kp__->num_gkvec_row(); igk++) {
100  alm_row_tmp(igk, xi) = std::conj(alm_row_tmp(igk, xi));
101  }
102  }
103  kp__->alm_coeffs_col().generate(ia, alm_col_tmp);
104  apply_hmt_to_apw<spin_block_t::nm>(atom, kp__->num_gkvec_col(), alm_col_tmp, halm_col_tmp);
105 
106  if (ctx_.valence_relativity() == relativity_t::iora) {
107  alm_col_tmp >> oalm_col_tmp;
108  apply_o1mt_to_apw(atom, kp__->num_gkvec_col(), alm_col_tmp, oalm_col_tmp);
109  }
110 
111  /* setup apw-lo and lo-apw blocks */
112  set_fv_h_o_apw_lo(kp__, type, atom, ia, alm_row_tmp, alm_col_tmp, h__, o__);
113  }
114  }
115  }
116  if (ctx_.control().print_checksum_) {
117  double_complex z1 = alm_row.checksum();
118  double_complex z2 = alm_col.checksum();
119  double_complex z3 = halm_col.checksum();
120  DUMP("checksum(alm_row): %18.10f %18.10f", std::real(z1), std::imag(z1));
121  DUMP("checksum(alm_col): %18.10f %18.10f", std::real(z2), std::imag(z2));
122  DUMP("checksum(halm_col): %18.10f %18.10f", std::real(z3), std::imag(z3));
123  }
124  linalg<CPU>::gemm(0, 1, kp__->num_gkvec_row(), kp__->num_gkvec_col(), num_mt_aw,
125  zone,
126  alm_row.at<CPU>(), alm_row.ld(),
127  oalm_col.at<CPU>(), oalm_col.ld(),
128  zone,
129  o__.at<CPU>(), o__.ld());
130 
131  linalg<CPU>::gemm(0, 1, kp__->num_gkvec_row(), kp__->num_gkvec_col(), num_mt_aw,
132  zone,
133  alm_row.at<CPU>(), alm_row.ld(),
134  halm_col.at<CPU>(), halm_col.ld(),
135  zone,
136  h__.at<CPU>(), h__.ld());
137  }
138  double tval = t1.stop();
139  if (kp__->comm().rank() == 0 && ctx_.control().print_performance_) {
140  DUMP("effective zgemm performance: %12.6f GFlops",
141  2 * 8e-9 * kp__->num_gkvec() * kp__->num_gkvec() * unit_cell_.mt_aw_basis_size() / tval);
142  }
143 
144  /* add interstitial contributon */
145  set_fv_h_o_it(kp__, potential__, h__, o__);
146 
147  /* setup lo-lo block */
148  set_fv_h_o_lo_lo(kp__, h__, o__);
149 }

276 inline void Band::set_fv_h_o_apw_lo(K_point* kp,
277  Atom_type const& type,
278  Atom const& atom,
279  int ia,
280  mdarray<double_complex, 2>& alm_row, // alm_row comes conjugated
281  mdarray<double_complex, 2>& alm_col,
282  mdarray<double_complex, 2>& h,
283  mdarray<double_complex, 2>& o) const
284 {
285  /* apw-lo block */
286  for (int i = 0; i < kp->num_atom_lo_cols(ia); i++) {
287  int icol = kp->lo_col(ia, i);
288  /* local orbital indices */
289  int l = kp->lo_basis_descriptor_col(icol).l;
290  int lm = kp->lo_basis_descriptor_col(icol).lm;
291  int idxrf = kp->lo_basis_descriptor_col(icol).idxrf;
292  int order = kp->lo_basis_descriptor_col(icol).order;
293  /* loop over apw components */
294  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
295  int lm1 = type.indexb(j1).lm;
296  int idxrf1 = type.indexb(j1).idxrf;
297 
298  double_complex zsum = atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf, idxrf1, gaunt_coefs_->gaunt_vector(lm1, lm));
299 
300  if (std::abs(zsum) > 1e-14) {
301  for (int igkloc = 0; igkloc < kp->num_gkvec_row(); igkloc++) {
302  h(igkloc, kp->num_gkvec_col() + icol) += zsum * alm_row(igkloc, j1);
303  }
304  }
305  }
306 
307  for (int order1 = 0; order1 < type.aw_order(l); order1++) {
308  int xi1 = type.indexb().index_by_lm_order(lm, order1);
309  for (int igkloc = 0; igkloc < kp->num_gkvec_row(); igkloc++) {
310  o(igkloc, kp->num_gkvec_col() + icol) += atom.symmetry_class().o_radial_integral(l, order1, order) *
311  alm_row(igkloc, xi1);
312  }
313  if (ctx_.valence_relativity() == relativity_t::iora) {
314  int idxrf1 = type.indexr().index_by_l_order(l, order1);
315  for (int igkloc = 0; igkloc < kp->num_gkvec_row(); igkloc++) {
316  o(igkloc, kp->num_gkvec_col() + icol) += atom.symmetry_class().o1_radial_integral(idxrf1, idxrf) *
317  alm_row(igkloc, xi1);
318  }
319  }
320  }
321  }
322 
323  std::vector<double_complex> ztmp(kp->num_gkvec_col());
324  /* lo-apw block */
325  for (int i = 0; i < kp->num_atom_lo_rows(ia); i++) {
326  int irow = kp->lo_row(ia, i);
327  /* local orbital indices */
328  int l = kp->lo_basis_descriptor_row(irow).l;
329  int lm = kp->lo_basis_descriptor_row(irow).lm;
330  int idxrf = kp->lo_basis_descriptor_row(irow).idxrf;
331  int order = kp->lo_basis_descriptor_row(irow).order;
332 
333  std::memset(&ztmp[0], 0, kp->num_gkvec_col() * sizeof(double_complex));
334  /* loop over apw components */
335  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
336  int lm1 = type.indexb(j1).lm;
337  int idxrf1 = type.indexb(j1).idxrf;
338 
339  double_complex zsum = atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf1, idxrf, gaunt_coefs_->gaunt_vector(lm, lm1));
340 
341  if (std::abs(zsum) > 1e-14) {
342  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
343  ztmp[igkloc] += zsum * alm_col(igkloc, j1);
344  }
345  }
346  }
347 
348  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
349  h(irow + kp->num_gkvec_row(), igkloc) += ztmp[igkloc];
350  }
351 
352  for (int order1 = 0; order1 < type.aw_order(l); order1++) {
353  int xi1 = type.indexb().index_by_lm_order(lm, order1);
354  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
355  o(irow + kp->num_gkvec_row(), igkloc) += atom.symmetry_class().o_radial_integral(l, order, order1) *
356  alm_col(igkloc, xi1);
357  }
358  if (ctx_.valence_relativity() == relativity_t::iora) {
359  int idxrf1 = type.indexr().index_by_l_order(l, order1);
360  for (int igkloc = 0; igkloc < kp->num_gkvec_col(); igkloc++) {
361  o(irow + kp->num_gkvec_row(), igkloc) += atom.symmetry_class().o1_radial_integral(idxrf, idxrf1) *
362  alm_col(igkloc, xi1);
363  }
364  }
365  }
366  }
367 }

369 inline void Band::set_fv_h_o_it(K_point* kp,
370  Potential const& potential__,
371  mdarray<double_complex, 2>& h,
372  mdarray<double_complex, 2>& o) const
373 {
374  PROFILE("sirius::Band::set_fv_h_o_it");
375 
376  //#ifdef __PRINT_OBJECT_CHECKSUM
377  //double_complex z1 = mdarray<double_complex, 1>(&effective_potential->f_pw(0), ctx_.gvec().num_gvec()).checksum();
378  //DUMP("checksum(veff_pw): %18.10f %18.10f", std::real(z1), std::imag(z1));
379  //#endif
380 
381  double sq_alpha_half = 0.5 * std::pow(speed_of_light, -2);
382 
383  #pragma omp parallel for default(shared)
384  for (int igk_col = 0; igk_col < kp->num_gkvec_col(); igk_col++) {
385  int ig_col = kp->igk_col(igk_col);
386  auto gvec_col = kp->gkvec().gvec(ig_col);
387  auto gkvec_col_cart = kp->gkvec().gkvec_cart(ig_col);
388  for (int igk_row = 0; igk_row < kp->num_gkvec_row(); igk_row++) {
389  int ig_row = kp->igk_row(igk_row);
390  auto gvec_row = kp->gkvec().gvec(ig_row);
391  auto gkvec_row_cart = kp->gkvec().gkvec_cart(ig_row);
392  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
393  /* pw kinetic energy */
394  double t1 = 0.5 * (gkvec_row_cart * gkvec_col_cart);
395 
396  h(igk_row, igk_col) += potential__.veff_pw(ig12);
397  o(igk_row, igk_col) += ctx_.step_function().theta_pw(ig12);
398 
399  if (ctx_.valence_relativity() == relativity_t::none) {
400  h(igk_row, igk_col) += t1 * ctx_.step_function().theta_pw(ig12);
401  } else {
402  h(igk_row, igk_col) += t1 * potential__.rm_inv_pw(ig12);
403  }
404  if (ctx_.valence_relativity() == relativity_t::iora) {
405  o(igk_row, igk_col) += t1 * sq_alpha_half * potential__.rm2_inv_pw(ig12);
406  }
407  }
408  }
409 }

411 inline void Band::set_fv_h_o_lo_lo(K_point* kp,
412  mdarray<double_complex, 2>& h,
413  mdarray<double_complex, 2>& o) const
414 {
415  PROFILE("sirius::Band::set_fv_h_o_lo_lo");
416 
417  /* lo-lo block */
418  #pragma omp parallel for default(shared)
419  for (int icol = 0; icol < kp->num_lo_col(); icol++) {
420  int ia = kp->lo_basis_descriptor_col(icol).ia;
421  int lm2 = kp->lo_basis_descriptor_col(icol).lm;
422  int idxrf2 = kp->lo_basis_descriptor_col(icol).idxrf;
423 
424  for (int irow = 0; irow < kp->num_lo_row(); irow++) {
425  /* lo-lo block is diagonal in atom index */
426  if (ia == kp->lo_basis_descriptor_row(irow).ia) {
427  auto& atom = unit_cell_.atom(ia);
428  int lm1 = kp->lo_basis_descriptor_row(irow).lm;
429  int idxrf1 = kp->lo_basis_descriptor_row(irow).idxrf;
430 
431  h(kp->num_gkvec_row() + irow, kp->num_gkvec_col() + icol) +=
432  atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf1, idxrf2, gaunt_coefs_->gaunt_vector(lm1, lm2));
433 
434  if (lm1 == lm2) {
435  int l = kp->lo_basis_descriptor_row(irow).l;
436  int order1 = kp->lo_basis_descriptor_row(irow).order;
