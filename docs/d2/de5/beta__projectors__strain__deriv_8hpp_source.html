36  void generate_pw_coefs_t(std::vector<int> const& igk__)
37  {
38  PROFILE("sirius::Beta_projectors_strain_deriv::generate_pw_coefs_t");
39 
40  if (!num_beta_t()) {
41  return;
42  }
43 
44  auto& beta_ri0 = ctx_.beta_ri();
45  auto& beta_ri1 = ctx_.beta_ri_djl();
46 
47  int lmax = ctx_.unit_cell().lmax();
48  int lmmax = utils::lmmax(lmax);
49 
50  mdarray<double, 2> rlm_g(lmmax, num_gkvec_loc());
51  mdarray<double, 3> rlm_dg(lmmax, 3, num_gkvec_loc());
52 
53  /* array of real spherical harmonics and derivatives for each G-vector */
54  #pragma omp parallel for schedule(static)
55  for (int igkloc = 0; igkloc < num_gkvec_loc(); igkloc++) {
56  auto gvc = gkvec_.gkvec_cart<index_domain_t::global>(igk__[igkloc]);
57  auto rtp = SHT::spherical_coordinates(gvc);
58 
59  double theta = rtp[1];
60  double phi = rtp[2];
61 
62  SHT::spherical_harmonics(lmax, theta, phi, &rlm_g(0, igkloc));
63  mdarray<double, 2> rlm_dg_tmp(&rlm_dg(0, 0, igkloc), lmmax, 3);
64  SHT::dRlm_dr(lmax, gvc, rlm_dg_tmp);
65  }
66 
67  /* compute d <G+k|beta> / d epsilon_{mu, nu} */
68  #pragma omp parallel for schedule(static)
69  for (int igkloc = 0; igkloc < num_gkvec_loc(); igkloc++) {
70  auto gvc = gkvec_.gkvec_cart<index_domain_t::global>(igk__[igkloc]);
71  /* vs = {r, theta, phi} */
72  auto gvs = SHT::spherical_coordinates(gvc);
73 
74  /* |G+k|=0 case */
75  if (gvs[0] < 1e-10) {
76  for (int nu = 0; nu < 3; nu++) {
77  for (int mu = 0; mu < 3; mu++) {
78  double p = (mu == nu) ? 0.5 : 0;
79 
80  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
81  auto& atom_type = ctx_.unit_cell().atom_type(iat);
82  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
83  int l = atom_type.indexb(xi).l;
84  int idxrf = atom_type.indexb(xi).idxrf;
85 
86  if (l == 0) {
87  auto z = fourpi / std::sqrt(ctx_.unit_cell().omega());
88 
89  auto d1 = beta_ri0.value<int, int>(idxrf, iat, gvs[0]) * (-p * y00);
90 
91  pw_coeffs_t_(igkloc, atom_type.offset_lo() + xi, mu + nu * 3) = z * d1;
92  } else {
93  pw_coeffs_t_(igkloc, atom_type.offset_lo() + xi, mu + nu * 3) = 0;
94  }
95  }
96  }
97  }
98  }
99  continue;
100  }
101 
102  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
103  auto& atom_type = ctx_.unit_cell().atom_type(iat);
104 
105  auto ri0 = beta_ri0.values(iat, gvs[0]);
106  auto ri1 = beta_ri1.values(iat, gvs[0]);
107 
108  for (int nu = 0; nu < 3; nu++) {
109  for (int mu = 0; mu < 3; mu++) {
110  double p = (mu == nu) ? 0.5 : 0;
111 
112  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
113  int l = atom_type.indexb(xi).l;
114  int lm = atom_type.indexb(xi).lm;
115  int idxrf = atom_type.indexb(xi).idxrf;
116 
117  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(ctx_.unit_cell().omega());
118 
119  auto d1 = ri0(idxrf) * (-gvc[mu] * rlm_dg(lm, nu, igkloc) - p * rlm_g(lm, igkloc));
120 
121  auto d2 = ri1(idxrf) * rlm_g(lm, igkloc) * (-gvc[mu] * gvc[nu] / gvs[0]);
122 
123  pw_coeffs_t_(igkloc, atom_type.offset_lo() + xi, mu + nu * 3) = z * (d1 + d2);
124  }
125  }
126  }
127  }
128  }
129  }
