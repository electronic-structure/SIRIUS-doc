30 Hubbard_matrix::Hubbard_matrix(Simulation_context& ctx__)
31  : ctx_(ctx__)
32 {
33  if (!ctx_.full_potential() && ctx_.hubbard_correction()) {
34 
35  local_ = std::vector<sddk::mdarray<double_complex, 3>>(ctx_.unit_cell().num_atoms());
36 
37  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
38  if (ctx_.unit_cell().atom(ia).type().hubbard_correction()) {
39  int nb = ctx_.unit_cell().atom(ia).type().indexb_hub().size();
40  local_[ia] = sddk::mdarray<double_complex, 3>(nb, nb, 4);
41  local_[ia].zero();
42  }
43  }
44  nonlocal_ = std::vector<sddk::mdarray<double_complex, 3>>(ctx_.cfg().hubbard().nonlocal().size());
45  for (int i = 0; i < ctx_.cfg().hubbard().nonlocal().size(); i++) {
46  auto nl = ctx_.cfg().hubbard().nonlocal(i);
47  int ia = nl.atom_pair()[0];
48  int ja = nl.atom_pair()[1];
49  int il = nl.l()[0];
50  int jl = nl.l()[1];
51  int ib = ctx_.unit_cell().atom(ia).type().indexr_hub().subshell_size(il, 0);
52  int jb = ctx_.unit_cell().atom(ja).type().indexr_hub().subshell_size(jl, 0);
53  nonlocal_[i] = sddk::mdarray<double_complex, 3>(ib, jb, 4);
54  nonlocal_[i].zero();
55  }
56  }
57 }

59 void Hubbard_matrix::access(std::string const& what__, double_complex* occ__, int ld__) {
60  if (!(what__ == "get" || what__ == "set")) {
61  std::stringstream s;
62  s << "wrong access label: " << what__;
63  RTE_THROW(s);
64  }
65 
66  sddk::mdarray<double_complex, 4> occ_mtrx;
67  /* in non-collinear case the occupancy matrix is complex */
68  if (ctx_.num_mag_dims() == 3) {
69  occ_mtrx = sddk::mdarray<double_complex, 4>(occ__, ld__, ld__, 4, ctx_.unit_cell().num_atoms());
70  } else {
71  occ_mtrx = sddk::mdarray<double_complex, 4>(occ__, ld__, ld__, ctx_.num_spins(), ctx_.unit_cell().num_atoms());
72  }
73  if (what__ == "get") {
74  occ_mtrx.zero();
75  }
76 
77  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
78  auto& atom = ctx_.unit_cell().atom(ia);
79  if (atom.type().hubbard_correction()) {
80  const int l = ctx_.unit_cell().atom(ia).type().indexr_hub().am(0).l();
81  for (int m1 = -l; m1 <= l; m1++) {
82  for (int m2 = -l; m2 <= l; m2++) {
83  if (what__ == "get") {
84  for (int j = 0; j < ((ctx_.num_mag_dims() == 3) ? 4 : ctx_.num_spins()); j++) {
85  occ_mtrx(l + m1, l + m2, j, ia) = this->local(ia)(l + m1, l + m2, j);
86  }
87  } else {
88  for (int j = 0; j < ((ctx_.num_mag_dims() == 3) ? 4 : ctx_.num_spins()); j++) {
89  this->local(ia)(l + m1, l + m2, j) = occ_mtrx(l + m1, l + m2, j, ia);
90  }
91  }
92  }
93  }
94  }
95  }

98 void Hubbard_matrix::print_local(int ia__, std::ostream& out__) const
99 {
100  auto const& atom = ctx_.unit_cell().atom(ia__);
101  if (!atom.type().hubbard_correction()) {
102  return;
103  }
104  int const prec{5};
105  int const width{10};
106  auto draw_bar = [&](int w)
107  {
108  out__ << std::setfill('-') << std::setw(w) << '-' << std::setfill(' ') << std::endl;
109  };
110  auto print_number = [&](double x)
111  {
112  out__ << std::setw(width) << std::setprecision(prec) << std::fixed << x;
113  };
114 
115  out__ << "atom : " << ia__ << std::endl;
116  if (ctx_.num_mag_dims() != 3) {
117  int l = atom.type().indexr_hub().am(0).l();
118  int mmax = 2 *l + 1;
119  for (int is = 0; is < ctx_.num_spins(); is++) {
120  draw_bar(width * mmax);
121  bool has_imag{false};
122  for (int m = 0; m < mmax; m++) {
123  for (int mp = 0; mp < mmax; mp++) {
124  if (std::abs(std::imag(this->local(ia__)(m, mp, is))) > 1e-12) {
125  has_imag = true;
126  }
127  print_number(std::real(this->local(ia__)(m, mp, is)));
128  }
129  out__ << std::endl;
130  }
131  if (has_imag) {
132  out__ << "imaginary part:" << std::endl;
133  for (int m = 0; m < mmax; m++) {
134  for (int mp = 0; mp < mmax; mp++) {
135  print_number(std::imag(this->local(ia__)(m, mp, is)));
136  }
137  out__ << std::endl;
138  }
139  }
140  }
141  draw_bar(width * mmax);
142  } else {
143  int l = atom.type().indexr_hub().am(0).l();
144  int mmax = 2 *l + 1;
145  draw_bar(2 * width * mmax + 3);
146  for (int m = 0; m < mmax; m++) {
147  for (int mp = 0; mp < mmax; mp++) {
148  print_number(std::real(this->local(ia__)(m, mp, 0)));
149  }
150  out__ << " | ";
151  for (int mp = 0; mp < mmax; mp++) {
152  print_number(std::real(this->local(ia__)(m, mp, 2)));
153  }
154  out__ << std::endl;
155  }
156  draw_bar(2 * width * mmax + 3);
157  for (int m = 0; m < mmax; m++) {
158  for (int mp = 0; mp < mmax; mp++) {
159  print_number(std::real(this->local(ia__)(m, mp, 3)));
160  }
161  out__ << " | ";
162  for (int mp = 0; mp < mmax; mp++) {
163  print_number(std::real(this->local(ia__)(m, mp, 1)));
164  }
165  out__ << std::endl;
166  }
167  draw_bar(2 * width * mmax + 3);
168  }
169 }
