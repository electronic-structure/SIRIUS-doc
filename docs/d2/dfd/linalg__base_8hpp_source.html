61/// Type of linear algebra backend.
62enum class linalg_t
63{
64 /// None
65 none,
66 /// CPU BLAS
67 blas,
68 /// CPU LAPACK
69 lapack,
70 /// CPU ScaLAPACK
71 scalapack,
72 /// GPU BLAS (cuBlas or ROCblas)
73 gpublas,
74 /// cuBlasXt (cuBlas with CPU pointers and large matrices support)
75 cublasxt,
76 /// MAGMA with CPU pointers
77 magma,
78 /// SPLA library. Can take CPU and device pointers
79 spla
80};
81
82inline linalg_t get_linalg_t(std::string name__)
83{
84 std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
85
86 static const std::map<std::string, linalg_t> map_to_type = {
87 {"blas", linalg_t::blas},
88 {"lapack", linalg_t::lapack},
89 {"scalapack", linalg_t::scalapack},
90 {"cublas", linalg_t::gpublas},
91 {"gpublas", linalg_t::gpublas},
92 {"cublasxt", linalg_t::cublasxt},
93 {"magma", linalg_t::magma},
94 };
95
96 if (map_to_type.count(name__) == 0) {
97 std::stringstream s;
98 s << "wrong label of linear algebra type: " << name__;
99 throw std::runtime_error(s.str());
100 }
101
102 return map_to_type.at(name__);
103}

105inline std::string to_string(linalg_t la__)
106{
107 switch (la__) {
108 case linalg_t::none: {
109 return "none";
110 break;
111 }
112 case linalg_t::blas: {
113 return "blas";
114 break;
115 }
116 case linalg_t::lapack: {
117 return "lapack";
118 break;
119 }
120 case linalg_t::scalapack: {
121 return "scalapack";
122 break;
123 }
124 case linalg_t::gpublas: {
125 return "gpublas";
126 break;
127 }
128 case linalg_t::cublasxt: {
129 return "cublasxt";
130 break;
131 }
132 case linalg_t::magma: {
133 return "magma";
134 break;
135 }
136 case linalg_t::spla: {
137 return "spla";
138 break;
139 }
140 }
141 return ""; // make compiler happy
142}
