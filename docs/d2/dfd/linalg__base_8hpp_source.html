71 inline linalg_t get_linalg_t(std::string name__)
72 {
73  std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
74 
75  static const std::map<std::string, linalg_t> map_to_type = {
76  {"blas", linalg_t::blas},
77  {"lapack", linalg_t::lapack},
78  {"cublas", linalg_t::cublas},
79  {"cublasxt", linalg_t::cublasxt},
80  {"magma", linalg_t::magma},
81  };
82 
83  if (map_to_type.count(name__) == 0) {
84  std::stringstream s;
85  s << "wrong label of linear algebra type: " << name__;
86  throw std::runtime_error(s.str());
87  }
88 
89  return map_to_type.at(name__);
90 }
