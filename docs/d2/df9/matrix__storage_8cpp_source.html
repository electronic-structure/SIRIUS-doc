29 template <typename T>
30 void matrix_storage<T, matrix_storage_t::slab>::set_num_extra(int n__, int idx0__, memory_pool* mp__)
31 {
32  PROFILE("sddk::matrix_storage::set_num_extra");
33 
34  auto& comm_col = gvp_->comm_ortho_fft();
35 
36  /* this is how n columns of the matrix will be distributed between columns of the MPI grid */
37  spl_num_col_ = splindex<splindex_t::block>(n__, comm_col.size(), comm_col.rank());
38 
39  T* ptr{nullptr};
40  T* ptr_d{nullptr};
41  int ncol{0};
42 
43  /* trivial case */
44  if (!is_remapped()) {
45  assert(num_rows_loc_ == gvp_->gvec_count_fft());
46  ncol = n__;
47  ptr = prime_.at(memory_t::host, 0, idx0__);
48  if (prime_.on_device()) {
49  ptr_d = prime_.at(memory_t::device, 0, idx0__);
50  }
51  } else {
52  /* maximum local number of matrix columns */
53  ncol = splindex_base<int>::block_size(n__, comm_col.size());
54  /* upper limit for the size of swapped extra matrix */
55  size_t sz = gvp_->gvec_count_fft() * ncol;
56  /* reallocate buffers if necessary */
57  if (extra_buf_.size() < sz) {
58  utils::timer t1("sddk::matrix_storage::set_num_extra|alloc");
59  if (mp__) {
60  send_recv_buf_ = mdarray<T, 1>(sz, *mp__, "matrix_storage.send_recv_buf_");
61  extra_buf_ = mdarray<T, 1>(sz, *mp__, "matrix_storage.extra_buf_");
62  } else {
63  send_recv_buf_ = mdarray<T, 1>(sz, memory_t::host, "matrix_storage.send_recv_buf_");
64  extra_buf_ = mdarray<T, 1>(sz, memory_t::host, "matrix_storage.extra_buf_");
65  }
66  }
67  ptr = extra_buf_.at(memory_t::host);
68  }
69  /* create the extra storage */
70  extra_ = mdarray<T, 2>(ptr, ptr_d, gvp_->gvec_count_fft(), ncol, "matrix_storage.extra_");
71 }

73 template <typename T>
74 void matrix_storage<T, matrix_storage_t::slab>::remap_from(const dmatrix<T>& mtrx__, int irow0__)
75 {
76  PROFILE("sddk::matrix_storage::remap_from");
77 
78  auto& comm = mtrx__.blacs_grid().comm();
79 
80  /* cache cartesian ranks */
81  mdarray<int, 2> cart_rank(mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().num_ranks_col());
82  for (int i = 0; i < mtrx__.blacs_grid().num_ranks_col(); i++) {
83  for (int j = 0; j < mtrx__.blacs_grid().num_ranks_row(); j++) {
84  cart_rank(j, i) = mtrx__.blacs_grid().cart_rank(j, i);
85  }
86  }
87 
88  if (send_recv_buf_.size() < prime_.size()) {
89  send_recv_buf_ = mdarray<T, 1>(prime_.size(), memory_t::host, "matrix_storage::send_recv_buf_");
90  }
91 
92  block_data_descriptor rd(comm.size());
93  rd.counts[comm.rank()] = num_rows_loc();
94  comm.allgather(rd.counts.data(), comm.rank(), 1);
95  rd.calc_offsets();
96 
97  block_data_descriptor sd(comm.size());
98 
99  /* global index of column */
100  int j0 = 0;
101  /* actual number of columns in the submatrix */
102  int ncol = num_cols_;
103 
104  splindex<splindex_t::block_cyclic> spl_col_begin(j0, mtrx__.num_ranks_col(), mtrx__.rank_col(), mtrx__.bs_col());
105  splindex<splindex_t::block_cyclic> spl_col_end(j0 + ncol, mtrx__.num_ranks_col(), mtrx__.rank_col(),
106  mtrx__.bs_col());
107 
108  int local_size_col = spl_col_end.local_size() - spl_col_begin.local_size();
109 
110  for (int rank_row = 0; rank_row < comm.size(); rank_row++) {
111  if (!rd.counts[rank_row]) {
112  continue;
113  }
114  /* global index of column */
115  int i0 = rd.offsets[rank_row];
116  /* actual number of rows in the submatrix */
117  int nrow = rd.counts[rank_row];
118 
119  assert(nrow != 0);
120 
121  splindex<splindex_t::block_cyclic> spl_row_begin(irow0__ + i0, mtrx__.num_ranks_row(), mtrx__.rank_row(),
122  mtrx__.bs_row());
123  splindex<splindex_t::block_cyclic> spl_row_end(irow0__ + i0 + nrow, mtrx__.num_ranks_row(), mtrx__.rank_row(),
124  mtrx__.bs_row());
125 
126  int local_size_row = spl_row_end.local_size() - spl_row_begin.local_size();
127 
128  mdarray<T, 1> buf(local_size_row * local_size_col);
129 
130  /* fetch elements of sub-matrix matrix */
131  if (local_size_row) {
132  for (int j = 0; j < local_size_col; j++) {
133  std::memcpy(&buf[local_size_row * j],
134  &mtrx__(spl_row_begin.local_size(), spl_col_begin.local_size() + j),
135  local_size_row * sizeof(T));
136  }
137  }
138 
139  sd.counts[comm.rank()] = local_size_row * local_size_col;
140  comm.allgather(sd.counts.data(), comm.rank(), 1);
141  sd.calc_offsets();
142 
143  /* collect buffers submatrix */
144  T* send_buf = (buf.size() == 0) ? nullptr : &buf[0];
145  T* recv_buf = (send_recv_buf_.size() == 0) ? nullptr : &send_recv_buf_[0];
146  comm.gather(send_buf, recv_buf, sd.counts.data(), sd.offsets.data(), rank_row);
147 
148  if (comm.rank() == rank_row) {
149  /* unpack data */
150  std::vector<int> counts(comm.size(), 0);
151  for (int jcol = 0; jcol < ncol; jcol++) {
152  auto pos_jcol = mtrx__.spl_col().location(j0 + jcol);
153  for (int irow = 0; irow < nrow; irow++) {
154  auto pos_irow = mtrx__.spl_row().location(irow0__ + i0 + irow);
155  int rank = cart_rank(pos_irow.rank, pos_jcol.rank);
156 
157  prime_(irow, jcol) = send_recv_buf_[sd.offsets[rank] + counts[rank]];
158  counts[rank]++;
159  }
160  }
161  for (int rank = 0; rank < comm.size(); rank++) {
162  assert(sd.counts[rank] == counts[rank]);
163  }
164  }
165  }
166 }

168 template <typename T>
169 void matrix_storage<T, matrix_storage_t::slab>::remap_backward(int n__, int idx0__)
170 {
171  PROFILE("sddk::matrix_storage::remap_backward");
172 
173  /* trivial case when extra storage mirrors the prime storage */
174  if (!is_remapped()) {
175  return;
176  }
177 
178  auto& comm_col = gvp_->comm_ortho_fft();
179 
180  auto& row_distr = gvp_->gvec_fft_slab();
181 
182  assert(n__ == spl_num_col_.global_index_size());
183 
184  /* local number of columns */
185  int n_loc = spl_num_col_.local_size();
186 
187  /* reorder sending blocks */
188  #pragma omp parallel for
189  for (int i = 0; i < n_loc; i++) {
190  for (int j = 0; j < comm_col.size(); j++) {
191  int offset = row_distr.offsets[j];
192  int count = row_distr.counts[j];
193  if (count) {
194  std::memcpy(&send_recv_buf_[offset * n_loc + count * i], &extra_(offset, i), count * sizeof(T));
195  }
196  }
197  }
198 
199  /* send and recieve dimensions */
200  block_data_descriptor sd(comm_col.size()), rd(comm_col.size());
201  for (int j = 0; j < comm_col.size(); j++) {
202  sd.counts[j] = spl_num_col_.local_size(comm_col.rank()) * row_distr.counts[j];
203  rd.counts[j] = spl_num_col_.local_size(j) * row_distr.counts[comm_col.rank()];
204  }
205  sd.calc_offsets();
206  rd.calc_offsets();
207 
208  T* recv_buf = (num_rows_loc_ == 0) ? nullptr : prime_.at(memory_t::host, 0, idx0__);
209 
210  utils::timer t1("sddk::matrix_storage::remap_backward|mpi");
211  comm_col.alltoall(send_recv_buf_.at(memory_t::host), sd.counts.data(), sd.offsets.data(), recv_buf,
212  rd.counts.data(), rd.offsets.data());
213  t1.stop();
214 
215  /* move data back to device */
216  if (prime_.on_device()) {
217  prime_.copy_to(memory_t::device, idx0__ * num_rows_loc(), n__ * num_rows_loc());
218  }
219 }

221 template <typename T>
222 void matrix_storage<T, matrix_storage_t::slab>::remap_forward(int n__, int idx0__, memory_pool* mp__)
223 {
224  PROFILE("sddk::matrix_storage::remap_forward");
225 
226  set_num_extra(n__, idx0__, mp__);
227 
228  /* trivial case when extra storage mirrors the prime storage */
229  if (!is_remapped()) {
230  return;
231  }
232 
233  auto& row_distr = gvp_->gvec_fft_slab();
234 
235  auto& comm_col = gvp_->comm_ortho_fft();
236 
237  /* local number of columns */
238  int n_loc = spl_num_col_.local_size();
239 
240  /* send and recieve dimensions */
241  block_data_descriptor sd(comm_col.size()), rd(comm_col.size());
242  for (int j = 0; j < comm_col.size(); j++) {
243  sd.counts[j] = spl_num_col_.local_size(j) * row_distr.counts[comm_col.rank()];
244  rd.counts[j] = spl_num_col_.local_size(comm_col.rank()) * row_distr.counts[j];
245  }
246  sd.calc_offsets();
247  rd.calc_offsets();
248 
249  T* send_buf = (num_rows_loc_ == 0) ? nullptr : prime_.at(memory_t::host, 0, idx0__);
250 
251  utils::timer t1("sddk::matrix_storage::remap_forward|mpi");
252  comm_col.alltoall(send_buf, sd.counts.data(), sd.offsets.data(), send_recv_buf_.at(memory_t::host),
253  rd.counts.data(), rd.offsets.data());
254  t1.stop();
255 
256  /* reorder recieved blocks */
257  #pragma omp parallel for
258  for (int i = 0; i < n_loc; i++) {
259  for (int j = 0; j < comm_col.size(); j++) {
260  int offset = row_distr.offsets[j];
261  int count = row_distr.counts[j];
262  if (count) {
263  std::memcpy(&extra_(offset, i), &send_recv_buf_[offset * n_loc + count * i], count * sizeof(T));
264  }
265  }
266  }
267 }

286 template <>
287 double_complex matrix_storage<std::complex<double>, matrix_storage_t::slab>::checksum(device_t pu__, int i0__, int n__) const
288 {
289  double_complex cs(0, 0);
290 
291  switch (pu__) {
292  case device_t::CPU: {
293  for (int i = 0; i < n__; i++) {
294  for (int j = 0; j < num_rows_loc(); j++) {
295  cs += prime(j, i0__ + i);
296  }
297  }
298  break;
299  }
300  case device_t::GPU: {
301  mdarray<double_complex, 1> cs1(n__, memory_t::host, "checksum");
302  cs1.allocate(memory_t::device).zero(memory_t::device);
303 #if defined(__GPU)
304  add_checksum_gpu(prime().at(memory_t::device, 0, i0__), num_rows_loc(), n__, cs1.at(memory_t::device));
305  cs1.copy_to(memory_t::host);
306  cs = cs1.checksum();
307 #endif
308  break;
309  }
310  }
311  return cs;
312 }
