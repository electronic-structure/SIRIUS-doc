32  : ctx_(ctx__)
33 {
34  PROFILE("sirius::Non_local_operator");

36  pu_ = this->ctx_.processing_unit();

38  packed_mtrx_offset_ = sddk::mdarray<int, 1>(uc.num_atoms());

72 template <>
73 void Non_local_operator::apply<double_complex>(int chunk__, int ispn_block__, Wave_functions& op_phi__, int idx0__,
74  int n__, Beta_projectors_base& beta__,
75  matrix<double_complex>& beta_phi__)
76 {
77  PROFILE("sirius::Non_local_operator::apply");
78 
79  if (is_null_) {
80  return;
81  }
82 
83  auto& beta_gk = beta__.pw_coeffs_a();
84  int num_gkvec_loc = beta__.num_gkvec_loc();
85  int nbeta = beta__.chunk(chunk__).num_beta_;
86 
87  /* setup linear algebra parameters */
88  memory_t mem{memory_t::none};
89  linalg_t la{linalg_t::none};
90  switch (pu_) {
91  case device_t::CPU: {
92  mem = memory_t::host;
93  la = linalg_t::blas;
94  break;
95  }
96  case device_t::GPU: {
97  mem = memory_t::device;
98  la = linalg_t::gpublas;
99  break;
100  }
101  }
102 
103  auto work = mdarray<double_complex, 1>(nbeta * n__, ctx_.mem_pool(mem));
104 
105  /* compute O * <beta|phi> for atoms in a chunk */
106  #pragma omp parallel
107  {
108  acc::set_device_id(sddk::get_device_id(acc::num_devices())); // avoid cuda mth bugs
109 
110  #pragma omp for
111  for (int i = 0; i < beta__.chunk(chunk__).num_atoms_; i++) {
112  /* number of beta functions for a given atom */
113  int nbf = beta__.chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::nbf), i);
114  int offs = beta__.chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::offset), i);
115  int ia = beta__.chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::ia), i);
116 
117  if (nbf) {
118  linalg(la).gemm(
119  'N', 'N', nbf, n__, nbf, &linalg_const<double_complex>::one(),
120  reinterpret_cast<double_complex*>(op_.at(mem, 0, packed_mtrx_offset_(ia), ispn_block__)), nbf,
121  beta_phi__.at(mem, offs, 0), beta_phi__.ld(), &linalg_const<double_complex>::zero(),
122  work.at(mem, offs), nbeta, stream_id(omp_get_thread_num()));
123  }
124  }
125  }
126  switch (pu_) {
127  case device_t::GPU: {
128  /* wait for previous zgemms */
129  #pragma omp parallel
130  acc::sync_stream(stream_id(omp_get_thread_num()));
131  break;
132  }
133  case device_t::CPU: {
134  break;
135  }
136  }
137 
138  int jspn = ispn_block__ & 1;
139 
140  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
141  linalg(ctx_.blas_linalg_t())
142  .gemm('N', 'N', num_gkvec_loc, n__, nbeta, &linalg_const<double_complex>::one(), beta_gk.at(mem), num_gkvec_loc,
143  work.at(mem), nbeta, &linalg_const<double_complex>::one(),
144  op_phi__.pw_coeffs(jspn).prime().at(op_phi__.preferred_memory_t(), 0, idx0__),
145  op_phi__.pw_coeffs(jspn).prime().ld());
146 
147  switch (pu_) {
148  case device_t::GPU: {
149  acc::sync_stream(stream_id(-1));
150  break;
151  }
152  case device_t::CPU: {
153  break;
154  }
155  }
156 }

222 template <>
223 void Non_local_operator::apply<double>(int chunk__, int ispn_block__, Wave_functions& op_phi__, int idx0__, int n__,
224  Beta_projectors_base& beta__, matrix<double>& beta_phi__)
225 {
226  PROFILE("sirius::Non_local_operator::apply");
227 
228  if (is_null_) {
229  return;
230  }
231 
232  auto& beta_gk = beta__.pw_coeffs_a();
233  int num_gkvec_loc = beta__.num_gkvec_loc();
234  int nbeta = beta__.chunk(chunk__).num_beta_;
235 
236  /* setup linear algebra parameters */
237  memory_t mem{memory_t::none};
238  linalg_t la{linalg_t::none};
239  switch (pu_) {
240  case device_t::CPU: {
241  mem = memory_t::host;
242  la = linalg_t::blas;
243  break;
244  }
245  case device_t::GPU: {
246  mem = memory_t::device;
247  la = linalg_t::gpublas;
248  break;
249  }
250  }
251 
252  auto work = mdarray<double, 1>(nbeta * n__, ctx_.mem_pool(mem));
253 
254  /* compute O * <beta|phi> for atoms in a chunk */
255  #pragma omp parallel for
256  for (int i = 0; i < beta__.chunk(chunk__).num_atoms_; i++) {
257  /* number of beta functions for a given atom */
258  int nbf = beta__.chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::nbf), i);
259  int offs = beta__.chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::offset), i);
260  int ia = beta__.chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::ia), i);
261 
262  if (nbf == 0) {
263  continue;
264  }
265  linalg(la).gemm('N', 'N', nbf, n__, nbf, &linalg_const<double>::one(),
266  op_.at(mem, 0, packed_mtrx_offset_(ia), ispn_block__), nbf, beta_phi__.at(mem, offs, 0),
267  beta_phi__.ld(), &linalg_const<double>::zero(), work.at(mem, offs), nbeta,
268  stream_id(omp_get_thread_num()));
269  }
270  switch (pu_) {
271  case device_t::GPU: {
272  /* wait for previous zgemms */
273  #pragma omp parallel
274  acc::sync_stream(stream_id(omp_get_thread_num()));
275  break;
276  }
277  case device_t::CPU: {
278  break;
279  }
280  }
281 
282  int jspn = ispn_block__ & 1;
283 
284  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
285  linalg(ctx_.blas_linalg_t())
286  .gemm('N', 'N', 2 * num_gkvec_loc, n__, nbeta, &linalg_const<double>::one(),
287  reinterpret_cast<double*>(beta_gk.at(mem)), 2 * num_gkvec_loc, work.at(mem), nbeta,
288  &linalg_const<double>::one(),
289  reinterpret_cast<double*>(op_phi__.pw_coeffs(jspn).prime().at(op_phi__.preferred_memory_t(), 0, idx0__)),
290  2 * op_phi__.pw_coeffs(jspn).prime().ld());
291 
292  switch (pu_) {
293  case device_t::GPU: {
294  acc::sync_stream(stream_id(-1));
295  break;
296  }
297  case device_t::CPU: {
298  break;
299  }
300  }

315 void D_operator::initialize()
316 {
317  PROFILE("sirius::D_operator::initialize");
318 
319  auto& uc = this->ctx_.unit_cell();
320 
321  #pragma omp parallel for
322  for (int ia = 0; ia < uc.num_atoms(); ia++) {
323  auto& atom = uc.atom(ia);
324  int nbf = atom.mt_basis_size();
325  auto& dion = atom.type().d_mtrx_ion();
326 
327  /* in case of spin orbit coupling */
328  if (uc.atom(ia).type().spin_orbit_coupling()) {
329  mdarray<double_complex, 3> d_mtrx_so(nbf, nbf, 4);
330  d_mtrx_so.zero();
331 
332  /* transform the d_mtrx */
333  for (int xi2 = 0; xi2 < nbf; xi2++) {
334  for (int xi1 = 0; xi1 < nbf; xi1++) {
335 
336  /* first compute \f[A^_\alpha I^{I,\alpha}_{xi,xi}\f] cf Eq.19 in doi:10.1103/PhysRevB.71.115106 */
337 
338  /* note that the `I` integrals are already calculated and stored in atom.d_mtrx */
339  for (int sigma = 0; sigma < 2; sigma++) {
340  for (int sigmap = 0; sigmap < 2; sigmap++) {
341  double_complex result(0, 0);
342  for (auto xi2p = 0; xi2p < nbf; xi2p++) {
343  if (atom.type().compare_index_beta_functions(xi2, xi2p)) {
344  /* just sum over m2, all other indices are the same */
345  for (auto xi1p = 0; xi1p < nbf; xi1p++) {
346  if (atom.type().compare_index_beta_functions(xi1, xi1p)) {
347  /* just sum over m1, all other indices are the same */
348 
349  /* loop over the 0, z,x,y coordinates */
350  for (int alpha = 0; alpha < 4; alpha++) {
351  for (int sigma1 = 0; sigma1 < 2; sigma1++) {
352  for (int sigma2 = 0; sigma2 < 2; sigma2++) {
353  result += atom.d_mtrx(xi1p, xi2p, alpha) *
354  pauli_matrix[alpha][sigma1][sigma2] *
355  atom.type().f_coefficients(xi1, xi1p, sigma, sigma1) *
356  atom.type().f_coefficients(xi2p, xi2, sigma2, sigmap);
357  }
358  }
359  }
360  }
361  }
362  }
363  }
364  const int ind = (sigma == sigmap) * sigma + (1 + 2 * sigma + sigmap) * (sigma != sigmap);
365  d_mtrx_so(xi1, xi2, ind) = result;
366  }
367  }
368  }
369  }
370 
371  /* add ionic contribution */
372 
373  /* spin orbit coupling mixes terms */
374 
375  /* keep the order of the indices because it is crucial here;
376  permuting the indices makes things wrong */
377  for (int xi2 = 0; xi2 < nbf; xi2++) {
378  int l2 = atom.type().indexb(xi2).l;
379  double j2 = atom.type().indexb(xi2).j;
380  int idxrf2 = atom.type().indexb(xi2).idxrf;
381  for (int xi1 = 0; xi1 < nbf; xi1++) {
382  int l1 = atom.type().indexb(xi1).l;
383  double j1 = atom.type().indexb(xi1).j;
384  int idxrf1 = atom.type().indexb(xi1).idxrf;
385  if ((l1 == l2) && (std::abs(j1 - j2) < 1e-8)) {
386  /* up-up down-down */
387  d_mtrx_so(xi1, xi2, 0) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 0, 0);
388  d_mtrx_so(xi1, xi2, 1) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 1, 1);
389 
390  /* up-down down-up */
391  d_mtrx_so(xi1, xi2, 2) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 0, 1);
392  d_mtrx_so(xi1, xi2, 3) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 1, 0);
393  }
394  }
395  }
396 
397  /* the pseudo potential contains information about
398  spin orbit coupling so we use a different formula
399  Eq.19 doi:10.1103/PhysRevB.71.115106 for calculating the D matrix
400 
401  Note that the D matrices are stored and
402  calculated in the up-down basis already not the (Veff,Bx,By,Bz) one */
403  for (int xi2 = 0; xi2 < nbf; xi2++) {
404  for (int xi1 = 0; xi1 < nbf; xi1++) {
405  int idx = xi2 * nbf + xi1;
406  for (int s = 0; s < 4; s++) {
407  this->op_(0, this->packed_mtrx_offset_(ia) + idx, s) = d_mtrx_so(xi1, xi2, s).real();
408  this->op_(1, this->packed_mtrx_offset_(ia) + idx, s) = d_mtrx_so(xi1, xi2, s).imag();
409  }
410  }
411  }
412  } else {
413  /* No spin orbit coupling for this atom \f[D = D(V_{eff})
414  I + D(B_x) \sigma_x + D(B_y) sigma_y + D(B_z)
415  sigma_z\f] since the D matrices are calculated that way */
416  for (int xi2 = 0; xi2 < nbf; xi2++) {
417  int lm2 = atom.type().indexb(xi2).lm;
418  int idxrf2 = atom.type().indexb(xi2).idxrf;
419  for (int xi1 = 0; xi1 < nbf; xi1++) {
420  int lm1 = atom.type().indexb(xi1).lm;
421  int idxrf1 = atom.type().indexb(xi1).idxrf;
422 
423  int idx = xi2 * nbf + xi1;
424  switch (this->ctx_.num_mag_dims()) {
425  case 3: {
426  double bx = uc.atom(ia).d_mtrx(xi1, xi2, 2);
427  double by = uc.atom(ia).d_mtrx(xi1, xi2, 3);
428 
429  this->op_(0, this->packed_mtrx_offset_(ia) + idx, 2) = bx;
430  this->op_(1, this->packed_mtrx_offset_(ia) + idx, 2) = -by;
431 
432  this->op_(0, this->packed_mtrx_offset_(ia) + idx, 3) = bx;
433  this->op_(1, this->packed_mtrx_offset_(ia) + idx, 3) = by;
434  }
435  case 1: {
436  double v = uc.atom(ia).d_mtrx(xi1, xi2, 0);
437  double bz = uc.atom(ia).d_mtrx(xi1, xi2, 1);
438 
439  /* add ionic part */
440  if (lm1 == lm2) {
441  v += dion(idxrf1, idxrf2);
442  }
443 
444  this->op_(0, this->packed_mtrx_offset_(ia) + idx, 0) = v + bz;
445  this->op_(0, this->packed_mtrx_offset_(ia) + idx, 1) = v - bz;
446  break;
447  }
448  case 0: {
449  this->op_(0, this->packed_mtrx_offset_(ia) + idx, 0) = uc.atom(ia).d_mtrx(xi1, xi2, 0);
450  /* add ionic part */
451  if (lm1 == lm2) {
452  this->op_(0, this->packed_mtrx_offset_(ia) + idx, 0) += dion(idxrf1, idxrf2);
453  }
454  break;
455  }
456  default: {
457  TERMINATE("wrong number of magnetic dimensions");
458  }
459  }
460  }
461  }
462  }
463  }
464 
465  if (this->ctx_.control().print_checksum_ && this->ctx_.comm().rank() == 0) {
466  auto cs = this->op_.checksum();
467  utils::print_checksum("D_operator", cs);
468  }
469 
470  if (this->pu_ == device_t::GPU) {
471  this->op_.allocate(memory_t::device).copy_to(memory_t::device);
472  }
473 
474  /* D-operator is not diagonal in spin in case of non-collinear magnetism
475  (spin-orbit coupling falls into this case) */
476  if (ctx_.num_mag_dims() == 3) {
477  this->is_diag_ = false;
478  }
479 }

541  for (auto sj = 0; sj < 2; sj++) {
542 
543  double_complex result(0, 0);
544 
545  for (int xi2p = 0; xi2p < nbf; xi2p++) {
546  if (uc.atom(ia).type().compare_index_beta_functions(xi2, xi2p)) {
547  for (int xi1p = 0; xi1p < nbf; xi1p++) {
548  /* The F coefficients are already "block diagonal" so we do a full
549  summation. We actually rotate the q_matrices only */
550  if (uc.atom(ia).type().compare_index_beta_functions(xi1, xi1p)) {
551  result += this->ctx_.augmentation_op(iat)->q_mtrx(xi1p, xi2p) *
552  (uc.atom(ia).type().f_coefficients(xi1, xi1p, sj, 0) *
553  uc.atom(ia).type().f_coefficients(xi2p, xi2, 0, si) +
554  uc.atom(ia).type().f_coefficients(xi1, xi1p, sj, 1) *
555  uc.atom(ia).type().f_coefficients(xi2p, xi2, 1, si));
556  }
557  }
558  }
559  }
560 
561  /* the order of the index is important */
562  const int ind = (si == sj) ? si : sj + 2;
563  /* this gives
564  ind = 0 if si = up and sj = up
565  ind = 1 if si = sj = down
566  ind = 2 if si = down and sj = up
567  ind = 3 if si = up and sj = down */
568  this->op_(0, this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ind) = result.real();
569  this->op_(1, this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ind) = result.imag();
570  }
571  }
572  } else {
573  for (int ispn = 0; ispn < this->ctx_.num_spins(); ispn++) {
574  this->op_(0, this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ispn) =
575  this->ctx_.augmentation_op(iat)->q_mtrx(xi1, xi2);
576  }
577  }
578  }
579  }
580  }
581  if (this->ctx_.control().print_checksum_ && this->ctx_.comm().rank() == 0) {
582  auto cs = this->op_.checksum();
583  utils::print_checksum("Q_operator", cs);
584  }
585 
586  if (this->pu_ == device_t::GPU) {
587  this->op_.allocate(memory_t::device).copy_to(memory_t::device);
588  }
589 
590  this->is_null_ = true;
591  for (int iat = 0; iat < uc.num_atom_types(); iat++) {
592  if (uc.atom_type(iat).augment()) {
593  this->is_null_ = false;
594  }
595  /* Q-operator is not diagonal in spin only in the case of spin-orbit coupling */
596  if (uc.atom_type(iat).spin_orbit_coupling()) {
597  this->is_diag_ = false;
598  }
599  }
600 }

603 void
604 apply_non_local_d_q(spin_range spins__, int N__, int n__, Beta_projectors& beta__, Wave_functions& phi__,
605  D_operator* d_op__, Wave_functions* hphi__, Q_operator* q_op__, Wave_functions* sphi__)
606 {
607 
608  for (int i = 0; i < beta__.num_chunks(); i++) {
609  /* generate beta-projectors for a block of atoms */
610  beta__.generate(i);
611 
612  for (int ispn : spins__) {
613  auto beta_phi = beta__.inner<T>(i, phi__, ispn, N__, n__);
614 
615  if (hphi__ && d_op__) {
616  /* apply diagonal spin blocks */
617  d_op__->apply(i, ispn, *hphi__, N__, n__, beta__, beta_phi);
618  if (!d_op__->is_diag() && hphi__->num_sc() == 2) {
619  /* apply non-diagonal spin blocks */
620  /* xor 3 operator will map 0 to 3 and 1 to 2 */
621  d_op__->apply(i, ispn ^ 3, *hphi__, N__, n__, beta__, beta_phi);
622  }
623  }
624 
625  if (sphi__ && q_op__) {
626  /* apply Q operator (diagonal in spin) */
627  q_op__->apply(i, ispn, *sphi__, N__, n__, beta__, beta_phi);
628  if (!q_op__->is_diag() && sphi__->num_sc() == 2) {
629  q_op__->apply(i, ispn ^ 3, *sphi__, N__, n__, beta__, beta_phi);
630  }
631  }
632  }
633  }
