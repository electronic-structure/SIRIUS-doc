32template <typename T>
33Non_local_operator<T>::Non_local_operator(Simulation_context const& ctx__)
34 : ctx_(ctx__)
35{
36 PROFILE("sirius::Non_local_operator");
37
38 pu_ = this->ctx_.processing_unit();
39 auto& uc = this->ctx_.unit_cell();
40 packed_mtrx_offset_ = sddk::mdarray<int, 1>(uc.num_atoms());
41 packed_mtrx_size_ = 0;
42 for (int ia = 0; ia < uc.num_atoms(); ia++) {
43 int nbf = uc.atom(ia).mt_basis_size();
44 packed_mtrx_offset_(ia) = packed_mtrx_size_;
45 packed_mtrx_size_ += nbf * nbf;
46 }
47
48 switch (pu_) {
49 case device_t::GPU: {
50 packed_mtrx_offset_.allocate(memory_t::device).copy_to(memory_t::device);
51 break;
52 }
53 case device_t::CPU: {
54 break;
55 }
56 }
57}

59template <typename T>
60D_operator<T>::D_operator(Simulation_context const& ctx_)
61 : Non_local_operator<T>(ctx_)
62{
63 if (ctx_.gamma_point()) {
64 this->op_ = mdarray<T, 3>(1, this->packed_mtrx_size_, ctx_.num_mag_dims() + 1);
65 } else {
66 this->op_ = mdarray<T, 3>(2, this->packed_mtrx_size_, ctx_.num_mag_dims() + 1);
67 }
68 this->op_.zero();
69 initialize();
70}

72template <typename T>
73void D_operator<T>::initialize()
74{
75 PROFILE("sirius::D_operator::initialize");
76
77 auto& uc = this->ctx_.unit_cell();
78
79 #pragma omp parallel for
80 for (int ia = 0; ia < uc.num_atoms(); ia++) {
81 auto& atom = uc.atom(ia);
82 int nbf = atom.mt_basis_size();
83 auto& dion = atom.type().d_mtrx_ion();
84
85 /* in case of spin orbit coupling */
86 if (uc.atom(ia).type().spin_orbit_coupling()) {
87 mdarray<std::complex<T>, 3> d_mtrx_so(nbf, nbf, 4);
88 d_mtrx_so.zero();
89
90 /* transform the d_mtrx */
91 for (int xi2 = 0; xi2 < nbf; xi2++) {
92 for (int xi1 = 0; xi1 < nbf; xi1++) {
93
94 /* first compute \f[A^_\alpha I^{I,\alpha}_{xi,xi}\f] cf Eq.19 in doi:10.1103/PhysRevB.71.115106 */
95
96 /* note that the `I` integrals are already calculated and stored in atom.d_mtrx */
97 for (int sigma = 0; sigma < 2; sigma++) {
98 for (int sigmap = 0; sigmap < 2; sigmap++) {
99 std::complex<T> result(0, 0);
100 for (auto xi2p = 0; xi2p < nbf; xi2p++) {
101 if (atom.type().compare_index_beta_functions(xi2, xi2p)) {
102 /* just sum over m2, all other indices are the same */
103 for (auto xi1p = 0; xi1p < nbf; xi1p++) {
104 if (atom.type().compare_index_beta_functions(xi1, xi1p)) {
105 /* just sum over m1, all other indices are the same */
106
107 /* loop over the 0, z,x,y coordinates */
108 for (int alpha = 0; alpha < 4; alpha++) {
109 for (int sigma1 = 0; sigma1 < 2; sigma1++) {
110 for (int sigma2 = 0; sigma2 < 2; sigma2++) {
111 result += atom.d_mtrx(xi1p, xi2p, alpha) *
112 pauli_matrix[alpha][sigma1][sigma2] *
113 atom.type().f_coefficients(xi1, xi1p, sigma, sigma1) *
114 atom.type().f_coefficients(xi2p, xi2, sigma2, sigmap);
115 }
116 }
117 }
118 }
119 }
120 }
121 }
122 const int ind = (sigma == sigmap) * sigma + (1 + 2 * sigma + sigmap) * (sigma != sigmap);
123 d_mtrx_so(xi1, xi2, ind) = result;
124 }
125 }
126 }
127 }
128
129 /* add ionic contribution */
130
131 /* spin orbit coupling mixes terms */
132
133 /* keep the order of the indices because it is crucial here;
134 permuting the indices makes things wrong */
135 for (int xi2 = 0; xi2 < nbf; xi2++) {
136 int l2 = atom.type().indexb(xi2).l;
137 double j2 = atom.type().indexb(xi2).j;
138 int idxrf2 = atom.type().indexb(xi2).idxrf;
139 for (int xi1 = 0; xi1 < nbf; xi1++) {
140 int l1 = atom.type().indexb(xi1).l;
141 double j1 = atom.type().indexb(xi1).j;
142 int idxrf1 = atom.type().indexb(xi1).idxrf;
143 if ((l1 == l2) && (std::abs(j1 - j2) < 1e-8)) {
144 /* up-up down-down */
145 d_mtrx_so(xi1, xi2, 0) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 0, 0);
146 d_mtrx_so(xi1, xi2, 1) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 1, 1);
147
148 /* up-down down-up */
149 d_mtrx_so(xi1, xi2, 2) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 0, 1);
150 d_mtrx_so(xi1, xi2, 3) += dion(idxrf1, idxrf2) * atom.type().f_coefficients(xi1, xi2, 1, 0);
151 }
152 }
153 }
154
155 /* the pseudo potential contains information about
156 spin orbit coupling so we use a different formula
157 Eq.19 doi:10.1103/PhysRevB.71.115106 for calculating the D matrix
158
159 Note that the D matrices are stored and
160 calculated in the up-down basis already not the (Veff,Bx,By,Bz) one */
161 for (int xi2 = 0; xi2 < nbf; xi2++) {
162 for (int xi1 = 0; xi1 < nbf; xi1++) {
163 int idx = xi2 * nbf + xi1;
164 for (int s = 0; s < 4; s++) {
165 this->op_(0, this->packed_mtrx_offset_(ia) + idx, s) = d_mtrx_so(xi1, xi2, s).real();
166 this->op_(1, this->packed_mtrx_offset_(ia) + idx, s) = d_mtrx_so(xi1, xi2, s).imag();
167 }
168 }
169 }
170 } else {
171 /* No spin orbit coupling for this atom \f[D = D(V_{eff})
172 I + D(B_x) \sigma_x + D(B_y) sigma_y + D(B_z)
173 sigma_z\f] since the D matrices are calculated that way */
174 for (int xi2 = 0; xi2 < nbf; xi2++) {
175 int lm2 = atom.type().indexb(xi2).lm;
176 int idxrf2 = atom.type().indexb(xi2).idxrf;
177 for (int xi1 = 0; xi1 < nbf; xi1++) {
178 int lm1 = atom.type().indexb(xi1).lm;
179 int idxrf1 = atom.type().indexb(xi1).idxrf;
180
181 int idx = xi2 * nbf + xi1;
182 switch (this->ctx_.num_mag_dims()) {
183 case 3: {
184 T bx = uc.atom(ia).d_mtrx(xi1, xi2, 2);
185 T by = uc.atom(ia).d_mtrx(xi1, xi2, 3);
186
187 this->op_(0, this->packed_mtrx_offset_(ia) + idx, 2) = bx;
188 this->op_(1, this->packed_mtrx_offset_(ia) + idx, 2) = -by;
189
190 this->op_(0, this->packed_mtrx_offset_(ia) + idx, 3) = bx;
191 this->op_(1, this->packed_mtrx_offset_(ia) + idx, 3) = by;
192 }
193 case 1: {
194 T v = uc.atom(ia).d_mtrx(xi1, xi2, 0);
195 T bz = uc.atom(ia).d_mtrx(xi1, xi2, 1);
196
197 /* add ionic part */
198 if (lm1 == lm2) {
199 v += dion(idxrf1, idxrf2);
200 }
201
202 this->op_(0, this->packed_mtrx_offset_(ia) + idx, 0) = v + bz;
203 this->op_(0, this->packed_mtrx_offset_(ia) + idx, 1) = v - bz;
204 break;
205 }
206 case 0: {
207 this->op_(0, this->packed_mtrx_offset_(ia) + idx, 0) = uc.atom(ia).d_mtrx(xi1, xi2, 0);
208 /* add ionic part */
209 if (lm1 == lm2) {
210 this->op_(0, this->packed_mtrx_offset_(ia) + idx, 0) += dion(idxrf1, idxrf2);
211 }
212 break;
213 }
214 default: {
215 TERMINATE("wrong number of magnetic dimensions");
216 }
217 }
218 }
219 }
220 }
221 }
222
223 if (this->ctx_.print_checksum() && this->ctx_.comm().rank() == 0) {
224 auto cs = this->op_.checksum();
225 utils::print_checksum("D_operator", cs);
226 }
227
228 if (this->pu_ == device_t::GPU && uc.mt_lo_basis_size() != 0) {
229 this->op_.allocate(memory_t::device).copy_to(memory_t::device);
230 }
231
232 /* D-operator is not diagonal in spin in case of non-collinear magnetism
233 (spin-orbit coupling falls into this case) */
234 if (this->ctx_.num_mag_dims() == 3) {
235 this->is_diag_ = false;
236 }

240Q_operator<T>::Q_operator(Simulation_context const& ctx__)
241 : Non_local_operator<T>(ctx__)
242{
243 /* Q-operator is independent of spin if there is no spin-orbit; however, it simplifies the apply()

247 } else {
248 this->op_ = mdarray<T, 3>(2, this->packed_mtrx_size_, this->ctx_.num_mag_dims() + 1);
249 }
250 this->op_.zero();
251 initialize();

255void Q_operator<T>::initialize()
256{
257 PROFILE("sirius::Q_operator::initialize");
258
259 auto& uc = this->ctx_.unit_cell();
260 /* check eigen-values of Q_{xi,xi'} matrix for each atom;
261 not sure if it helps, so it's commented for now */
262 // if (this->ctx_.control().verification_ >= 1) {
263 // for (int ia = 0; ia < uc.num_atoms(); ia++) {
264 // int iat = uc.atom(ia).type().id();
265 // if (!uc.atom_type(iat).augment()) {
266 // continue;
267 // }
268 // int nbf = uc.atom(ia).mt_basis_size();
269 // Eigensolver_lapack evs;
270 // dmatrix<double> A(nbf, nbf);
271 // dmatrix<double> Z(nbf, nbf);
272 // std::vector<double> ev(nbf);
273 // for (int xi1 = 0; xi1 < nbf; xi1++) {
274 // for (int xi2 = 0; xi2 < nbf; xi2++) {
275 // A(xi1, xi2) = this->ctx_.augmentation_op(iat).q_mtrx(xi1, xi2);
276 // }
277 // }
278 // evs.solve(nbf, A, ev.data(), Z);
279 // if (this->ctx_.control().verbosity_ >= 0 && this->ctx_.comm().rank() == 0) {
280 // printf("eigen-values of the Q-matrix for atom %i\n", ia);
281 // for (int i = 0; i < nbf; i++) {
282 // printf("%18.12f\n", ev[i]);
283 // }
284 // }
285 // }
286 //}
287 #pragma omp parallel for
288 for (int ia = 0; ia < uc.num_atoms(); ia++) {
289 int iat = uc.atom(ia).type().id();
290 if (!uc.atom_type(iat).augment()) {
291 continue;
292 }
293 int nbf = uc.atom(ia).mt_basis_size();
294 for (int xi2 = 0; xi2 < nbf; xi2++) {
295 for (int xi1 = 0; xi1 < nbf; xi1++) {
296 /* The ultra soft pseudo potential has spin orbit coupling incorporated to it, so we
297 need to rotate the Q matrix */
298 if (uc.atom_type(iat).spin_orbit_coupling()) {
299 /* this is nothing else than Eq.18 of doi:10.1103/PhysRevB.71.115106 */
300 for (auto si = 0; si < 2; si++) {
301 for (auto sj = 0; sj < 2; sj++) {
302
303 std::complex<T> result(0, 0);
304
305 for (int xi2p = 0; xi2p < nbf; xi2p++) {
306 if (uc.atom(ia).type().compare_index_beta_functions(xi2, xi2p)) {
307 for (int xi1p = 0; xi1p < nbf; xi1p++) {
308 /* The F coefficients are already "block diagonal" so we do a full
309 summation. We actually rotate the q_matrices only */
310 if (uc.atom(ia).type().compare_index_beta_functions(xi1, xi1p)) {
311 result += this->ctx_.augmentation_op(iat)->q_mtrx(xi1p, xi2p) *
312 (uc.atom(ia).type().f_coefficients(xi1, xi1p, sj, 0) *
313 uc.atom(ia).type().f_coefficients(xi2p, xi2, 0, si) +
314 uc.atom(ia).type().f_coefficients(xi1, xi1p, sj, 1) *
315 uc.atom(ia).type().f_coefficients(xi2p, xi2, 1, si));
316 }
317 }
318 }
319 }
320
321 /* the order of the index is important */
322 const int ind = (si == sj) ? si : sj + 2;
323 /* this gives
324 ind = 0 if si = up and sj = up
325 ind = 1 if si = sj = down
326 ind = 2 if si = down and sj = up
327 ind = 3 if si = up and sj = down */
328 this->op_(0, this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ind) = result.real();
329 this->op_(1, this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ind) = result.imag();
330 }
331 }
332 } else {
333 for (int ispn = 0; ispn < this->ctx_.num_spins(); ispn++) {
334 this->op_(0, this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ispn) =
335 this->ctx_.augmentation_op(iat)->q_mtrx(xi1, xi2);
336 }
337 }
338 }
339 }
340 }
341 if (this->ctx_.print_checksum() && this->ctx_.comm().rank() == 0) {
342 auto cs = this->op_.checksum();
343 utils::print_checksum("Q_operator", cs);
344 }
345
346 if (this->pu_ == device_t::GPU && uc.mt_lo_basis_size() != 0) {
347 this->op_.allocate(memory_t::device).copy_to(memory_t::device);
348 }
349
350 this->is_null_ = true;
351 for (int iat = 0; iat < uc.num_atom_types(); iat++) {
352 if (uc.atom_type(iat).augment()) {
353 this->is_null_ = false;
354 }
355 /* Q-operator is not diagonal in spin only in the case of spin-orbit coupling */
356 if (uc.atom_type(iat).spin_orbit_coupling()) {
357 this->is_diag_ = false;
358 }
359 }
360}

362template <typename T>
363void
364apply_non_local_d_q(spin_range spins__, int N__, int n__, Beta_projectors<real_type<T>>& beta__, Wave_functions<real_type<T>>& phi__,
365 D_operator<real_type<T>>* d_op__, Wave_functions<real_type<T>>* hphi__, Q_operator<real_type<T>>* q_op__,
366 Wave_functions<real_type<T>>* sphi__)
367{
368
369 for (int i = 0; i < beta__.num_chunks(); i++) {
370 /* generate beta-projectors for a block of atoms */
371 beta__.generate(i);
372
373 for (int ispn : spins__) {
374 auto beta_phi = beta__.template inner<T>(i, phi__, ispn, N__, n__);
375
376 if (hphi__ && d_op__) {
377 /* apply diagonal spin blocks */
378 d_op__->apply(i, ispn, *hphi__, N__, n__, beta__, beta_phi);
379 if (!d_op__->is_diag() && hphi__->num_sc() == 2) {
380 /* apply non-diagonal spin blocks */
381 /* xor 3 operator will map 0 to 3 and 1 to 2 */
382 d_op__->apply(i, ispn ^ 3, *hphi__, N__, n__, beta__, beta_phi);
383 }
384 }
385
386 if (sphi__ && q_op__) {
387 /* apply Q operator (diagonal in spin) */
388 q_op__->apply(i, ispn, *sphi__, N__, n__, beta__, beta_phi);
389 if (!q_op__->is_diag() && sphi__->num_sc() == 2) {
390 q_op__->apply(i, ispn ^ 3, *sphi__, N__, n__, beta__, beta_phi);
391 }
392 }
393 }
394 }
395}

397/// Compute |sphi> = (1 + Q)|phi>
398template <typename T>
399void
400apply_S_operator(device_t pu__, spin_range spins__, int N__, int n__, Beta_projectors<real_type<T>>& beta__,
401 Wave_functions<real_type<T>>& phi__, Q_operator<real_type<T>>* q_op__, Wave_functions<real_type<T>>& sphi__)
402{
403 for (auto s: spins__) {
404 sphi__.copy_from(pu__, n__, phi__, s, N__, s, N__);
405 }
406
407 if (q_op__) {
408 apply_non_local_d_q<T>(spins__, N__, n__, beta__, phi__, nullptr, nullptr, q_op__, &sphi__);
409 }
410}

412template <typename T>
413void
414apply_U_operator(Simulation_context& ctx__, spin_range spins__, int N__, int n__, Wave_functions<T>& hub_wf__,
415 Wave_functions<T>& phi__, U_operator<T>& um__, Wave_functions<T>& hphi__)
416{
417 dmatrix<std::complex<T>> dm(hub_wf__.num_wf(), n__);
418 dm.zero();
419
420 if (ctx__.processing_unit() == device_t::GPU) {
421 dm.allocate(memory_t::device);
422 }
423
424 /* First calculate the local part of the projections
425 dm(i, n) = <phi_i| S |psi_{nk}> */
426 sddk::inner(ctx__.spla_context(), spins__, hub_wf__, 0, hub_wf__.num_wf(), phi__, N__, n__, dm, 0, 0);
427
428 dmatrix<std::complex<T>> Up(hub_wf__.num_wf(), n__);
429 Up.zero();
430
431 //auto nhwf = um__.nhwf();
432
433 #pragma omp parallel for schedule(static)
434 for (int ia = 0; ia < ctx__.unit_cell().num_atoms(); ia++) {
435 auto& atom = ctx__.unit_cell().atom(ia);
436 if (atom.type().hubbard_correction()) {
437 const int lmax_at = 2 * atom.type().indexr_hub().am(0).l() + 1;
438 // we apply the hubbard correction. For now I have no papers
439 // giving me the formula for the SO case so I rely on QE for it
440 // but I do not like it at all
441 if (ctx__.num_mag_dims() == 3) {
442 for (int s1 = 0; s1 < ctx__.num_spins(); s1++) {
443 for (int s2 = 0; s2 < ctx__.num_spins(); s2++) {
444 const int ind = (s1 == s2) * s1 + (1 + 2 * s2 + s1) * (s1 != s2);
445
446 // TODO: replace this with matrix matrix multiplication
447
448 for (int nbd = 0; nbd < n__; nbd++) {
449 for (int m1 = 0; m1 < lmax_at; m1++) {
450 for (int m2 = 0; m2 < lmax_at; m2++) {
451 Up(um__.nhwf() * s1 + um__.offset(ia) + m1, nbd) +=
452 um__(um__.offset(ia) + m2, um__.offset(ia) + m1, ind) *
453 dm(um__.nhwf() * s2 + um__.offset(ia) + m2, nbd);
454 }
455 }
456 }
457 }
458 }
459 } else {
460 // Conventional LDA or collinear magnetism
461 for (int nbd = 0; nbd < n__; nbd++) {
462 for (int m1 = 0; m1 < lmax_at; m1++) {
463 for (int m2 = 0; m2 < lmax_at; m2++) {
464 Up(um__.offset(ia) + m1, nbd) += um__(um__.offset(ia) + m2, um__.offset(ia) + m1, spins__()) *
465 dm(um__.offset(ia) + m2, nbd);
466 }
467 }
468 }
469 }
470 }
471 }
472
473 if (ctx__.processing_unit() == device_t::GPU) {
474 Up.allocate(memory_t::device);
475 Up.copy_to(memory_t::device);
476 }
477
478 transform<std::complex<T>, std::complex<T>>(ctx__.spla_context(), spins__(), 1.0, {&hub_wf__}, 0, hub_wf__.num_wf(),
479 Up, 0, 0, 1.0, {&hphi__}, N__, n__);
480}
