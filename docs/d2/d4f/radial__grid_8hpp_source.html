60  void init()
61  {
62  x_inv_ = mdarray<T, 1>(num_points(), memory_t::host, "Radial_grid::x_inv");
63  dx_ = mdarray<T, 1>(num_points() - 1, memory_t::host, "Radial_grid::dx");
64 
65  /* set x^{-1} */
66  for (int i = 0; i < num_points(); i++) {
67  x_inv_(i) = (x_(i) == 0) ? 0 : 1.0 / x_(i);
68  }
69 
70  /* set dx */
71  for (int i = 0; i < num_points() - 1; i++) {
72  dx_(i) = x_(i + 1) - x_(i);
73  }
74  }

95  int index_of(T x__) const
96  {
97  if (x__ < first() || x__ > last()) {
98  return -1;
99  }
100  int i0 = 0;
101  int i1 = num_points() - 1;
102 
103  while (i1 - i0 > 1) {
104  int i = (i1 + i0) >> 1;
105  if (x__ >= x_[i0] && x__ < x_[i]) {
106  i1 = i;
107  } else {
108  i0 = i;
109  }
110  }
111  return i0;
112  }

214 template <typename T>
215 class Radial_grid_pow : public Radial_grid<T>
216 {
217  public:
218  Radial_grid_pow(int num_points__, T rmin__, T rmax__, double p__)
219  : Radial_grid<T>(num_points__)
220  {
221  for (int i = 0; i < this->num_points(); i++) {
222  T t = static_cast<T>(i) / (this->num_points() - 1);
223  this->x_[i] = rmin__ + (rmax__ - rmin__) * std::pow(t, p__);
224  }
225  this->x_[0] = rmin__;
226  this->x_[num_points__ - 1] = rmax__;
227  this->init();
228  std::stringstream s;
229  s << p__;
230  this->name_ = "power" + s.str();
231  }
232 };

245 template <typename T>
246 class Radial_grid_exp : public Radial_grid<T>
247 {
248  public:
249  Radial_grid_exp(int num_points__, T rmin__, T rmax__, T p__ = 1.0)
250  : Radial_grid<T>(num_points__)
251  {
252  /* x_i = x_min * (x_max/x_min) ^ (i / (N - 1)) */
253  for (int i = 0; i < this->num_points(); i++) {
254  T t = static_cast<T>(i) / (this->num_points() - 1);
255  this->x_[i] = rmin__ * std::pow(rmax__ / rmin__, std::pow(t, p__));
256  }
257  this->x_[0] = rmin__;
258  this->x_[num_points__ - 1] = rmax__;
259  this->init();
260  this->name_ = "exponential";
261  }
262 };

264 template <typename T>
265 class Radial_grid_lin_exp : public Radial_grid<T>
266 {
267  public:
268  Radial_grid_lin_exp(int num_points__, T rmin__, T rmax__, T p__ = 6.0)
269  : Radial_grid<T>(num_points__)
270  {
271  /* x_i = x_min + (x_max - x_min) * A(t)
272  A(0) = 0; A(1) = 1
273  A(t) ~ b * t + Exp[t^a] - 1 */
274  T alpha = p__;
275  T beta = 1e-6 * this->num_points() / (rmax__ - rmin__);
276  T A = 1.0 / (std::exp(static_cast<T>(1)) + beta - static_cast<T>(1));
277  for (int i = 0; i < this->num_points(); i++) {
278  T t = static_cast<T>(i) / (this->num_points() - 1);
279  this->x_[i] =
280  rmin__ + (rmax__ - rmin__) * (beta * t + std::exp(std::pow(t, alpha)) - static_cast<T>(1)) * A;
281  }
282  // T beta = std::log((rmax__ - rmin__ * (num_points__ - 1)) / rmin__);

284  // T t = static_cast<T>(i) / (this->num_points() - 1);
285  // this->x_[i] = rmin__ * (i + std::exp(beta * std::pow(t, 1)));
286  //}
287 
288  this->x_[0] = rmin__;
289  this->x_[num_points__ - 1] = rmax__;
290  this->init();
291  this->name_ = "linear_exponential";
292  }

311 template <typename T>
312 Radial_grid<T> Radial_grid_factory(radial_grid_t grid_type__, int num_points__, T rmin__, T rmax__, double p__)
313 {
314  Radial_grid<T> rgrid;
315 
316  switch (grid_type__) {
317  case radial_grid_t::linear: {
318  rgrid = Radial_grid_lin<T>(num_points__, rmin__, rmax__);
319  break;
320  }
321  case radial_grid_t::exponential: {
322  rgrid = Radial_grid_exp<T>(num_points__, rmin__, rmax__, p__);
323  break;
324  }
325  case radial_grid_t::power: {
326  rgrid = Radial_grid_pow<T>(num_points__, rmin__, rmax__, p__);
327  break;
328  }
329  case radial_grid_t::lin_exp: {
330  rgrid = Radial_grid_lin_exp<T>(num_points__, rmin__, rmax__, p__);
331  break;
332  }
333  default: {
334  TERMINATE("wrong radial grid type");
335  }
336  }
337  return std::move(rgrid);
338 };

340 inline std::pair<radial_grid_t, double> get_radial_grid_t(std::string str__)
341 {
342  auto pos = str__.find(",");
343  if (pos == std::string::npos) {
344  std::stringstream s;
345  s << "wrong string for the radial grid type: " << str__;
346  throw std::runtime_error(s.str());
347  }
348 
349  std::string name = str__.substr(0, pos);
350  double p = std::stod(str__.substr(pos + 1));
351 
352  const std::map<std::string, radial_grid_t> map_to_type = {{"linear", radial_grid_t::linear},
353  {"exponential", radial_grid_t::exponential},
354  {"power", radial_grid_t::power},
355  {"lin_exp", radial_grid_t::lin_exp}};
356 
357  return std::make_pair(map_to_type.at(name), p);
358 }
