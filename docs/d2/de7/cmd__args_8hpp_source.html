47  void register_key(const std::string key__, const std::string description__)
48  {
49  key_desc_.push_back(std::pair<std::string, std::string>(key__, description__));
50 
51  int key_type = 0;
52  std::string key = key__.substr(2, key__.length());
53 
54  if (key[key.length() - 1] == '=') {
55  key = key.substr(0, key.length() - 1);
56  key_type = 1;
57  }
58 
59  if (known_keys_.count(key) != 0)
60  TERMINATE("key is already added");
61 
62  known_keys_[key] = key_type;
63  }

65  void parse_args(int argn__, char** argv__)
66  {
67  for (int i = 1; i < argn__; i++) {
68  std::string str(argv__[i]);
69  if (str.length() < 3 || str[0] != '-' || str[1] != '-') {
70  std::stringstream s;
71  s << "wrong key: " << str;
72  TERMINATE(s);
73  }
74 
75  size_t k = str.find("=");
76 
77  std::string key, val;
78  if (k != std::string::npos) {
79  key = str.substr(2, k - 2);
80  val = str.substr(k + 1, str.length());
81  } else {
82  key = str.substr(2, str.length());
83  }
84 
85  if (known_keys_.count(key) != 1) {
86  std::stringstream s;
87  s << "key " << key << " is not found";
88  TERMINATE(s);
89  }
90 
91  if (known_keys_[key] == 0 && k != std::string::npos) {
92  TERMINATE("this key must not have a value");
93  }
94 
95  if (known_keys_[key] == 1 && k == std::string::npos) {
96  TERMINATE("this key must have a value");
97  }
98 
99  if (keys_.count(key) != 0)
100  TERMINATE("key is already added");
101 
102  keys_[key] = val;
103  }
104  }

106  void print_help()
107  {
108  int max_key_width = 0;
109  for (int i = 0; i < (int)key_desc_.size(); i++)
110  max_key_width = std::max(max_key_width, (int)key_desc_[i].first.length());
111 
112  printf("Options:\n");
113 
114  for (int i = 0; i < (int)key_desc_.size(); i++) {
115  printf(" %s", key_desc_[i].first.c_str());
116  int k = (int)key_desc_[i].first.length();
117 
118  for (int j = 0; j < max_key_width - k + 1; j++)
119  printf(" ");
120 
121  printf("%s\n", key_desc_[i].second.c_str());
122  }
123  }
