30template <typename T>
31void K_point<T>::generate_spinor_wave_functions()
32{
33 PROFILE("sirius::K_point::generate_spinor_wave_functions");
34
35 if (ctx_.cfg().control().use_second_variation()) {
36 int nfv = ctx_.num_fv_states();
37
38 if (!ctx_.need_sv()) {
39 /* copy eigen-states and exit */
40 spinor_wave_functions().copy_from(device_t::CPU, ctx_.num_fv_states(), fv_states(), 0, 0, 0, 0);
41 return;
42 }
43
44 int nbnd = (ctx_.num_mag_dims() == 3) ? ctx_.num_bands() : nfv;
45
46 if (ctx_.processing_unit() == device_t::GPU) {
47 fv_states().allocate(spin_range(0), ctx_.mem_pool(memory_t::device));
48 fv_states().copy_to(spin_range(0), memory_t::device, 0, nfv);
49 sv_eigen_vectors_[0].allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
50 if (ctx_.num_mag_dims() == 1) {
51 sv_eigen_vectors_[1].allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
52 }
53 if (is_device_memory(ctx_.preferred_memory_t())) {
54 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
55 spinor_wave_functions().allocate(spin_range(ispn), ctx_.mem_pool(memory_t::device));
56 spinor_wave_functions().copy_to(spin_range(ispn), memory_t::device, 0, nbnd);
57 }
58 }
59 }
60
61 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
62 int s, o;
63
64 if (ctx_.num_mag_dims() == 3) {
65 /* in case of non-collinear magnetism sv_eigen_vectors is a single 2Nx2N matrix */
66 s = 0;
67 o = ispn * nfv; /* offset for spin up is 0, for spin dn is nfv */
68 } else {
69 /* sv_eigen_vectors is composed of two NxN matrices */
70 s = ispn;
71 o = 0;
72 }
73 /* multiply consecutively up and dn blocks */
74 sddk::transform<complex_type<T>, complex_type<T>>(ctx_.spla_context(), ispn, fv_states(), 0, nfv, sv_eigen_vectors_[s], o, 0,
75 spinor_wave_functions(), 0, nbnd);
76 }
77
78 if (ctx_.processing_unit() == device_t::GPU) {
79 fv_states().deallocate(spin_range(0), memory_t::device);
80 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
81 spinor_wave_functions().copy_to(spin_range(ispn), memory_t::host, 0, nbnd);
82 }
83 sv_eigen_vectors_[0].deallocate(memory_t::device);
84 if (ctx_.num_mag_dims() == 3) {
85 sv_eigen_vectors_[1].deallocate(memory_t::device);
86 }
87 }
88 } else {
89 throw std::runtime_error("not implemented");
90 }
91}
