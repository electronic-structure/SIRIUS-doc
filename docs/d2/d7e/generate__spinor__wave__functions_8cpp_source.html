30 void K_point::generate_spinor_wave_functions()
31 {
32  PROFILE("sirius::K_point::generate_spinor_wave_functions");
33 
34  if (ctx_.cfg().control().use_second_variation()) {
35  int nfv = ctx_.num_fv_states();
36 
37  if (!ctx_.need_sv()) {
38  /* copy eigen-states and exit */
39  spinor_wave_functions().copy_from(device_t::CPU, ctx_.num_fv_states(), fv_states(), 0, 0, 0, 0);
40  return;
41  }
42 
43  int nbnd = (ctx_.num_mag_dims() == 3) ? ctx_.num_bands() : nfv;
44 
45  if (ctx_.processing_unit() == device_t::GPU) {
46  fv_states().allocate(spin_range(0), ctx_.mem_pool(memory_t::device));
47  fv_states().copy_to(spin_range(0), memory_t::device, 0, nfv);
48  sv_eigen_vectors_[0].allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
49  if (ctx_.num_mag_dims() == 1) {
50  sv_eigen_vectors_[1].allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
51  }
52  if (is_device_memory(ctx_.preferred_memory_t())) {
53  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
54  spinor_wave_functions().allocate(spin_range(ispn), ctx_.mem_pool(memory_t::device));
55  spinor_wave_functions().copy_to(spin_range(ispn), memory_t::device, 0, nbnd);
56  }
57  }
58  }
59 
60  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
61  int s, o;
62 
63  if (ctx_.num_mag_dims() == 3) {
64  /* in case of non-collinear magnetism sv_eigen_vectors is a single 2Nx2N matrix */
65  s = 0;
66  o = ispn * nfv; /* offset for spin up is 0, for spin dn is nfv */
67  } else {
68  /* sv_eigen_vectors is composed of two NxN matrices */
69  s = ispn;
70  o = 0;
71  }
72  /* multiply consecutively up and dn blocks */
73  transform(ctx_.spla_context(), ispn, fv_states(), 0, nfv, sv_eigen_vectors_[s], o, 0,
74  spinor_wave_functions(), 0, nbnd);
75  }
76 
77  if (ctx_.processing_unit() == device_t::GPU) {
78  fv_states().deallocate(spin_range(0), memory_t::device);
79  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
80  spinor_wave_functions().copy_to(spin_range(ispn), memory_t::host, 0, nbnd);
81  }
82  sv_eigen_vectors_[0].deallocate(memory_t::device);
83  if (ctx_.num_mag_dims() == 3) {
84  sv_eigen_vectors_[1].deallocate(memory_t::device);
85  }
86  }
87  } else {
88  throw std::runtime_error("not implemented");
89  }
90 }
