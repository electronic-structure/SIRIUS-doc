87 void Force::ibs_force(Simulation_context_base& ctx__,
88  Band* band__,
89  K_point* kp__,
90  mdarray<double, 2>& ffac__,
91  mdarray<double, 2>& forcek__)
92 {
93  PROFILE("sirius::Force::ibs_force");
94 
95  auto& param = ctx__;
96  auto& uc = ctx__.unit_cell();
97  //auto rl = ctx__.reciprocal_lattice();
98 
99  forcek__.zero();
100 
101  dmatrix<double_complex> dm(ctx__.num_fv_states(), ctx__.num_fv_states(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
102  compute_dmat(param, kp__, dm);
103 
104  auto& fv_evec = kp__->fv_eigen_vectors();
105 
106  dmatrix<double_complex> h(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
107  dmatrix<double_complex> o(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
108 
109  dmatrix<double_complex> h1(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
110  dmatrix<double_complex> o1(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
111 
112  dmatrix<double_complex> zm1(kp__->gklo_basis_size(), ctx__.num_fv_states(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
113  dmatrix<double_complex> zf(ctx__.num_fv_states(), ctx__.num_fv_states(), ctx__.blacs_grid(), param.cyclic_block_size(), param.cyclic_block_size());
114 
115  mdarray<double_complex, 2> alm_row(kp__->num_gkvec_row(), uc.max_mt_aw_basis_size());
116  mdarray<double_complex, 2> alm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
117  mdarray<double_complex, 2> halm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
118 
119  for (int ia = 0; ia < uc.num_atoms(); ia++)
120  {
121  h.zero();
122  o.zero();
123 
124  auto& atom = uc.atom(ia);
125  auto& type = atom.type();
126 
127  /* generate matching coefficients for current atom */
128  kp__->alm_coeffs_row().generate(ia, alm_row);
129  kp__->alm_coeffs_col().generate(ia, alm_col);
130 
131  /* setup apw-lo and lo-apw blocks */
132  band__->set_fv_h_o_apw_lo(kp__, type, atom, ia, alm_row, alm_col, h, o);
133 
134  /* apply MT Hamiltonian to column coefficients */
135  band__->apply_hmt_to_apw<spin_block_t::nm>(atom, kp__->num_gkvec_col(), alm_col, halm_col);
136 
137  /* conjugate row (<bra|) matching coefficients */
138  for (int i = 0; i < type.mt_aw_basis_size(); i++)
139  {
140  for (int igk = 0; igk < kp__->num_gkvec_row(); igk++) alm_row(igk, i) = std::conj(alm_row(igk, i));
141  }
142 
143  /* apw-apw block of the overlap matrix */
144  linalg<CPU>::gemm(0, 1, kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
145  alm_row.at<CPU>(), alm_row.ld(), alm_col.at<CPU>(), alm_col.ld(), o.at<CPU>(), o.ld());
146 
147  /* apw-apw block of the Hamiltonian matrix */
148  linalg<CPU>::gemm(0, 1, kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
149  alm_row.at<CPU>(), alm_row.ld(), halm_col.at<CPU>(), halm_col.ld(), h.at<CPU>(), h.ld());
150 
151  int iat = type.id();
152 
153  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) // loop over columns
154  {
155  auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
156  auto gkvec_col_cart = kp__->gkvec().gkvec_cart(kp__->igk_col(igk_col));
157  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) // for each column loop over rows
158  {
159  auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
160  auto gkvec_row_cart = kp__->gkvec().gkvec_cart(kp__->igk_row(igk_row));
161 
162  int ig12 = ctx__.gvec().index_g12(gvec_row, gvec_col);
163 
164  int igs = ctx__.gvec().shell(ig12);
165 
166  double_complex zt = std::conj(ctx__.gvec_phase_factor(ig12, ia)) * ffac__(iat, igs) * fourpi / uc.omega();
167 
168  double t1 = 0.5 * (gkvec_row_cart * gkvec_col_cart);
169 
170  h(igk_row, igk_col) -= t1 * zt;
171  o(igk_row, igk_col) -= zt;
172  }
173  }
174 
175  for (int x = 0; x < 3; x++)
176  {
177  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) // loop over columns
178  {
179  auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
180  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) // for each column loop over rows
181  {
182  auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
183  int ig12 = ctx__.gvec().index_g12(gvec_row, gvec_col);
184 
185  vector3d<double> vg = ctx__.gvec().gvec_cart(ig12);
186  h1(igk_row, igk_col) = double_complex(0.0, vg[x]) * h(igk_row, igk_col);
187  o1(igk_row, igk_col) = double_complex(0.0, vg[x]) * o(igk_row, igk_col);
188  }
189  }
190 
191  for (int icol = 0; icol < kp__->num_lo_col(); icol++) {
192  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) {
193  auto gkvec_row_cart = kp__->gkvec().gkvec_cart(kp__->igk_row(igk_row));
194  h1(igk_row, icol + kp__->num_gkvec_col()) = double_complex(0.0, gkvec_row_cart[x]) * h(igk_row, icol + kp__->num_gkvec_col());
195  o1(igk_row, icol + kp__->num_gkvec_col()) = double_complex(0.0, gkvec_row_cart[x]) * o(igk_row, icol + kp__->num_gkvec_col());
196  }
197  }
198 
199  for (int irow = 0; irow < kp__->num_lo_row(); irow++)
200  {
201  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++)
202  {
203  auto gkvec_col_cart = kp__->gkvec().gkvec_cart(kp__->igk_col(igk_col));
204  h1(irow + kp__->num_gkvec_row(), igk_col) = double_complex(0.0, -gkvec_col_cart[x]) * h(irow + kp__->num_gkvec_row(), igk_col);
205  o1(irow + kp__->num_gkvec_row(), igk_col) = double_complex(0.0, -gkvec_col_cart[x]) * o(irow + kp__->num_gkvec_row(), igk_col);
206  }
207  }
208 
209  /* zm1 = H * V */
210  linalg<CPU>::gemm(0, 0, kp__->gklo_basis_size(), ctx__.num_fv_states(), kp__->gklo_basis_size(),
211  linalg_const<double_complex>::one(), h1, fv_evec, linalg_const<double_complex>::zero(), zm1);
212 
213  /* F = V^{+} * zm1 = V^{+} * H * V */
214  linalg<CPU>::gemm(2, 0, ctx__.num_fv_states(), ctx__.num_fv_states(), kp__->gklo_basis_size(),
215  linalg_const<double_complex>::one(), fv_evec, zm1, linalg_const<double_complex>::zero(), zf);
216 
217  /* zm1 = O * V */
218  linalg<CPU>::gemm(0, 0, kp__->gklo_basis_size(), ctx__.num_fv_states(), kp__->gklo_basis_size(),
219  linalg_const<double_complex>::one(), o1, fv_evec, linalg_const<double_complex>::zero(), zm1);
220 
221  STOP();
222  ///* multiply by energy */
223  //for (int i = 0; i < (int)kp__->spl_fv_states().local_size(); i++)
224  //{
225  // int ist = kp__->spl_fv_states(i);
226  // for (int j = 0; j < kp__->gklo_basis_size_row(); j++) zm1(j, i) = zm1(j, i) * kp__->fv_eigen_value(ist);
227  //}
228 
229  /* F = F - V^{+} * zm1 = F - V^{+} * O * (E*V) */
230  linalg<CPU>::gemm(2, 0, ctx__.num_fv_states(), ctx__.num_fv_states(), kp__->gklo_basis_size(),
231  double_complex(-1, 0), fv_evec, zm1, double_complex(1, 0), zf);
232 
233  for (int i = 0; i < dm.num_cols_local(); i++)
234  {
235  for (int j = 0; j < dm.num_rows_local(); j++)
236  forcek__(x, ia) += kp__->weight() * real(dm(j, i) * zf(j, i));
237  }
238  }
239  } //ia
240 }

242 void Force::total_force(Simulation_context_base& ctx__,
243  Potential* potential__,
244  Density* density__,
245  K_point_set* ks__,
246  mdarray<double, 2>& force__)
247 {
248  PROFILE("sirius::Force::total_force");
249 
250  auto& uc = ctx__.unit_cell();
251 
252  //auto ffac = ctx__.step_function().get_step_function_form_factors(ctx__.gvec().num_shells(), ctx__.unit_cell(), ctx__.gvec(), ctx__.comm());
253  STOP();
254 
255  force__.zero();
256 
257  mdarray<double, 2> forcek(3, uc.num_atoms());
258  for (int ikloc = 0; ikloc < ks__->spl_num_kpoints().local_size(); ikloc++)
259  {
260  //int ik = ks__->spl_num_kpoints(ikloc);
261  //ibs_force(ctx__, band_, (*ks__)[ik], ffac, forcek);
262  STOP();
263  for (int ia = 0; ia < uc.num_atoms(); ia++)
264  {
265  for (int x: {0, 1, 2}) force__(x, ia) += forcek(x, ia);
266  }
267  }
268  ctx__.comm().allreduce(&force__(0, 0), (int)force__.size());
269 
270  if (ctx__.control().verbosity_ > 2 && ctx__.comm().rank() == 0) {
271  printf("\n");
272  printf("Forces\n");
273  printf("\n");
274  for (int ia = 0; ia < uc.num_atoms(); ia++) {
275  printf("ia : %i, IBS : %12.6f %12.6f %12.6f\n", ia, force__(0, ia), force__(1, ia), force__(2, ia));
276  }
277  }
278 
279  mdarray<double, 2> forcehf(3, uc.num_atoms());
280 
281  forcehf.zero();
282  for (int ialoc = 0; ialoc < (int)uc.spl_num_atoms().local_size(); ialoc++)
283  {
284  int ia = uc.spl_num_atoms(ialoc);
285  auto g = gradient(potential__->hartree_potential_mt(ialoc));
286  for (int x = 0; x < 3; x++) forcehf(x, ia) = uc.atom(ia).zn() * g[x](0, 0) * y00;
287  }
288  ctx__.comm().allreduce(&forcehf(0, 0), (int)forcehf.size());
289 
290  if (ctx__.control().verbosity_ > 2 && ctx__.comm().rank() == 0) {
291  printf("\n");
292  for (int ia = 0; ia < uc.num_atoms(); ia++) {
293  printf("ia : %i, Hellmann–Feynman : %12.6f %12.6f %12.6f\n", ia, forcehf(0, ia), forcehf(1, ia), forcehf(2, ia));
294  }
295  }
296 
297  mdarray<double, 2> forcerho(3, uc.num_atoms());
298  forcerho.zero();
299  for (int ialoc = 0; ialoc < (int)uc.spl_num_atoms().local_size(); ialoc++)
300  {
301  int ia = uc.spl_num_atoms(ialoc);
302  auto g = gradient(density__->density_mt(ialoc));
303  for (int x = 0; x < 3; x++) forcerho(x, ia) = inner(potential__->effective_potential_mt(ialoc), g[x]);
304  }
305  ctx__.comm().allreduce(&forcerho(0, 0), (int)forcerho.size());
306 
307  if (ctx__.control().verbosity_ > 2 && ctx__.comm().rank() == 0) {
308  printf("\n");
309  printf("rho force\n");
310  for (int ia = 0; ia < uc.num_atoms(); ia++) {
311  printf("ia : %i, density contribution : %12.6f %12.6f %12.6f\n", ia, forcerho(0, ia), forcerho(1, ia), forcerho(2, ia));
312  }
313  }
314 
315  for (int ia = 0; ia < uc.num_atoms(); ia++) {
316  for (int x = 0; x < 3; x++) {
317  force__(x, ia) += (forcehf(x, ia) + forcerho(x, ia));
318  }
319  }
320 }
