39 /// Index of Rlm in QE in the block of lm coefficients for a given l.
40 static inline int idx_m_qe(int m__)
41 {
42  return (m__ > 0) ? 2 * m__ - 1 : -2 * m__;
43 }

45 /// Mapping of atomic indices from SIRIUS to QE order.
46 static inline std::vector<int> atomic_orbital_index_map_QE(sirius::Atom_type const& type__)
47 {
48  int nbf = type__.mt_basis_size();
49 
50  std::vector<int> idx_map(nbf);
51  for (int xi = 0; xi < nbf; xi++) {
52  int m = type__.indexb(xi).m;
53  int idxrf = type__.indexb(xi).idxrf;
54  idx_map[xi] = type__.indexb().index_by_idxrf(idxrf) + idx_m_qe(m); /* beginning of lm-block + new offset in lm block */
55  }
56  return idx_map;
57 }
