37 /// Index of Rlm in QE in the block of lm coefficients for a given l.
38 static inline int idx_m_qe(int m__)
39 {
40  return (m__ > 0) ? 2 * m__ - 1 : -2 * m__;

43 /// Mapping of atomic indices from SIRIUS to QE order.
44 static inline std::vector<int> atomic_orbital_index_map_QE(sirius::Atom_type const& type__)
45 {
46  int nbf = type__.mt_basis_size();
47 
48  std::vector<int> idx_map(nbf);
49  for (int xi = 0; xi < nbf; xi++) {
50  int m = type__.indexb(xi).m;
51  int idxrf = type__.indexb(xi).idxrf;
52  idx_map[xi] = type__.indexb().index_by_idxrf(idxrf) + idx_m_qe(m); /* beginning of lm-block + new offset in lm block */
53  }
54  return std::move(idx_map);
55 }

58 {
