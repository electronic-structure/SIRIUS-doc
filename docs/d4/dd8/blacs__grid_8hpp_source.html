53  public:
54  BLACS_grid(Communicator const& comm__, int num_ranks_row__, int num_ranks_col__)
55  : comm_(comm__)
56  {
57  mpi_grid_ = std::unique_ptr<MPI_grid>(new MPI_grid({num_ranks_row__, num_ranks_col__}, comm_));
58  rank_map_.resize(num_ranks_row__ * num_ranks_col__);
59 
60 #ifdef SIRIUS_SCALAPACK
61  /* create handler first */
62  blacs_handler_ = linalg_base::create_blacs_handler(mpi_grid_->communicator().mpi_comm());
63 
64  for (int j = 0; j < num_ranks_col__; j++) {
65  for (int i = 0; i < num_ranks_row__; i++) {
66  rank_map_[i + j * num_ranks_row__] = mpi_grid_->communicator().cart_rank({i, j});
67  }
68  }
69 
70  /* create context */
71  blacs_context_ = blacs_handler_;
72  linalg_base::gridmap(&blacs_context_, &rank_map_[0], num_ranks_row__, num_ranks_row__, num_ranks_col__);
73 
74  /* check the grid */
75  int nrow1, ncol1, irow1, icol1;
76  linalg_base::gridinfo(blacs_context_, &nrow1, &ncol1, &irow1, &icol1);
77 
78  if (rank_row() != irow1 || rank_col() != icol1 || num_ranks_row() != nrow1 || num_ranks_col() != ncol1) {
79  std::stringstream s;
80  s << "wrong grid" << std::endl
81  << " row | col | nrow | ncol " << std::endl
82  << " mpi_grid " << rank_row() << " " << rank_col() << " " << num_ranks_row() << " " << num_ranks_col()
83  << std::endl
84  << " blacs " << irow1 << " " << icol1 << " " << nrow1 << " " << ncol1;
85  TERMINATE(s);
86  }
87 #else
88  for (int i = 0; i < static_cast<int>(rank_map_.size()); i++) {
89  rank_map_[i] = i;
90  }
91 
92 #endif
93  }
94 
95  ~BLACS_grid()
96  {
97  int mpi_finalized;
98  MPI_Finalized(&mpi_finalized);
99  if (mpi_finalized == 0) {
100 #ifdef SIRIUS_SCALAPACK
101  linalg_base::gridexit(blacs_context_);
102  linalg_base::free_blacs_handler(blacs_handler_);
103 #endif
104  }
105  }
