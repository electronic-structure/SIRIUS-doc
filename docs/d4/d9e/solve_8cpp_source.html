30template <typename T>
31void
32Band::solve_full_potential(Hamiltonian_k<T>& Hk__) const
33{
34 ctx_.print_memory_usage(__FILE__, __LINE__);
35 if (ctx_.cfg().control().use_second_variation()) {
36 /* solve non-magnetic Hamiltonian (so-called first variation) */
37 auto& itso = ctx_.cfg().iterative_solver();
38 if (itso.type() == "exact") {
39 diag_full_potential_first_variation_exact(Hk__);
40 } else if (itso.type() == "davidson") {
41 diag_full_potential_first_variation_davidson(Hk__);
42 }
43 /* generate first-variational states */
44 Hk__.kp().generate_fv_states();
45 /* solve magnetic Hamiltonian */
46 diag_full_potential_second_variation(Hk__);
47 /* generate spinor wave-functions */
48 Hk__.kp().generate_spinor_wave_functions();
49 } else {
50 throw std::runtime_error("not implemented");
51 //diag_full_potential_single_variation();
52 }
53 ctx_.print_memory_usage(__FILE__, __LINE__);
54}

69template <typename T, typename F>
70int
71Band::solve_pseudo_potential(Hamiltonian_k<real_type<T>>& Hk__, double itsol_tol__, double empy_tol__) const
72{
73 ctx_.print_memory_usage(__FILE__, __LINE__);
74
75 int niter{0};
76
77 auto& itso = ctx_.cfg().iterative_solver();
78 if (itso.type() == "exact") {
79 if (ctx_.num_mag_dims() != 3) {
80 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
81 diag_pseudo_potential_exact<T>(ispn, Hk__);
82 }
83 } else {
84 STOP();
85 }
86 } else if (itso.type() == "davidson") {
87 auto& kp = Hk__.kp();
88
89 auto tolerance = [&](int j__, int ispn__) -> double {
90
91 /* tolerance for occupied states */
92 double tol = itsol_tol__;
93 /* if band is empty, make tolerance larger (in most cases we don't need high precision on
94 * unoccupied states) */
95 if (std::abs(kp.band_occupancy(j__, ispn__)) < ctx_.min_occupancy() * ctx_.max_occupancy()) {
96 tol += empy_tol__;
97 }
98
99 return tol;
100 };
101
102 auto result = davidson<T, F>(Hk__, ctx_.num_bands(), ctx_.num_mag_dims(), kp.spinor_wave_functions(),
103 [&](int i, int ispn){return kp.band_occupancy(i, ispn);}, tolerance,
104 ctx_.cfg().iterative_solver().residual_tolerance(), ctx_.cfg().iterative_solver().num_steps(),
105 ctx_.cfg().iterative_solver().locking(), ctx_.cfg().iterative_solver().subspace_size(),
106 ctx_.cfg().iterative_solver().converge_by_energy(), ctx_.cfg().iterative_solver().extra_ortho(), std::cout, 0);
107
108 niter = result.niter;
109 for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
110 for (int j = 0; j < ctx_.num_bands(); j++) {
111 kp.band_energy(j, ispn, result.eval(j, ispn));
112 }
113 }
114 } else {
115 RTE_THROW("unknown iterative solver type");
116 }
117
118 /* check residuals */
119 if (ctx_.cfg().control().verification() >= 2) {
120 check_residuals<T>(Hk__);
121 check_wave_functions<T>(Hk__);
122 }
123
124 ctx_.print_memory_usage(__FILE__, __LINE__);
125
126 return niter;
127}

129template <typename T, typename F>
130void
131Band::solve(K_point_set& kset__, Hamiltonian0<T>& H0__, bool precompute__, double itsol_tol__) const
132{
133 PROFILE("sirius::Band::solve");
134
135 if (precompute__ && ctx_.full_potential()) {
136 H0__.potential().generate_pw_coefs();
137 H0__.potential().update_atomic_potential();
138 unit_cell_.generate_radial_functions();
139 unit_cell_.generate_radial_integrals();
140 }
141
142 ctx_.print_memory_usage(__FILE__, __LINE__);
143
144 double empy_tol{itsol_tol__};
145 if (!ctx_.full_potential()) {
146 empy_tol = std::max(itsol_tol__ * ctx_.cfg().settings().itsol_tol_ratio(),
147 ctx_.cfg().iterative_solver().empty_states_tolerance());
148 ctx_.message(2, __function_name__, "iterative solver tolerance (occupied, empty) : %1.4e, %1.4e\n",
149 itsol_tol__, itsol_tol__ + empy_tol);
150 }
151
152 int num_dav_iter{0};
153 /* solve secular equation and generate wave functions */
154 for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
155 int ik = kset__.spl_num_kpoints(ikloc);
156 auto kp = kset__.get<T>(ik);
157
158 auto Hk = H0__(*kp);
159 if (ctx_.full_potential()) {
160 solve_full_potential<T>(Hk);
161 } else {
162 if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
163 num_dav_iter += solve_pseudo_potential<T, F>(Hk, itsol_tol__, empy_tol);
164 } else {
165 num_dav_iter += solve_pseudo_potential<std::complex<T>, std::complex<F>>(Hk, itsol_tol__, empy_tol);
166 }
167 }
168 }
169 kset__.comm().allreduce(&num_dav_iter, 1);
170 ctx_.num_itsol_steps(num_dav_iter);
171 if (!ctx_.full_potential()) {
172 ctx_.message(2, __function_name__, "average number of iterations: %12.6f\n",
173 static_cast<double>(num_dav_iter) / kset__.num_kpoints());
174 }
175
176 /* synchronize eigen-values */
177 kset__.sync_band<T, sync_band_t::energy>();
178
179 ctx_.message(2, __function_name__, "%s", "Lowest band energies\n");
180 if (ctx_.verbosity() >= 2 && ctx_.comm().rank() == 0) {
181 for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
182 std::printf("ik : %2i, ", ik);
183 for (int j = 0; j < std::min(ctx_.cfg().control().num_bands_to_print(), ctx_.num_bands()); j++) {
184 std::printf("%12.6f", kset__.get<T>(ik)->band_energy(j, 0));
185 }
186 if (ctx_.num_mag_dims() == 1) {
187 std::printf("\n ");
188 for (int j = 0; j < std::min(ctx_.cfg().control().num_bands_to_print(), ctx_.num_bands()); j++) {
189 std::printf("%12.6f", kset__.get<T>(ik)->band_energy(j, 1));
190 }
191 }
192 std::printf("\n");
193 }
194 }
195 ctx_.print_memory_usage(__FILE__, __LINE__);
196}
