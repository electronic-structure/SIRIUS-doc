30 Band::solve_full_potential(Hamiltonian_k& Hk__) const
31 {
32  if (ctx_.control().use_second_variation_) {
33  /* solve non-magnetic Hamiltonian (so-called first variation) */
34  auto& itso = ctx_.iterative_solver_input();
35  if (itso.type_ == "exact") {
36  diag_full_potential_first_variation_exact(Hk__);
37  } else if (itso.type_ == "davidson") {
38  diag_full_potential_first_variation_davidson(Hk__);
39  }
40  /* generate first-variational states */
41  Hk__.kp().generate_fv_states();
42  /* solve magnetic Hamiltonian */
43  diag_full_potential_second_variation(Hk__);
44  /* generate spinor wave-functions */
45  Hk__.kp().generate_spinor_wave_functions();
46  } else {
47  throw std::runtime_error("not implemented");
48  //diag_full_potential_single_variation();
49  }
50 }

103 void
104 Band::solve(K_point_set& kset__, Hamiltonian0& H0__, bool precompute__) const
105 {
106  PROFILE("sirius::Band::solve");
107 
108  if (precompute__ && ctx_.full_potential()) {
109  H0__.potential().generate_pw_coefs();
110  H0__.potential().update_atomic_potential();
111  unit_cell_.generate_radial_functions();
112  unit_cell_.generate_radial_integrals();
113  }
114 
115  ctx_.print_memory_usage(__FILE__, __LINE__);
116 
117  if (!ctx_.full_potential()) {
118  ctx_.message(1, __function_name__, "iterative solver tolerance: %18.12f\n", ctx_.iterative_solver_tolerance());
119  }
120 
121  int num_dav_iter{0};
122  /* solve secular equation and generate wave functions */
123  for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
124  int ik = kset__.spl_num_kpoints(ikloc);
125  auto kp = kset__[ik];
126 
127  auto Hk = H0__(*kp);
128  if (ctx_.full_potential()) {
129  solve_full_potential(Hk);
130  } else {
131  if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
132  num_dav_iter += solve_pseudo_potential<double>(Hk);
133  } else {
134  num_dav_iter += solve_pseudo_potential<double_complex>(Hk);
135  }
136  }
137  }
138  kset__.comm().allreduce(&num_dav_iter, 1);
139  ctx_.num_itsol_steps(num_dav_iter);
140  if (!ctx_.full_potential()) {
141  ctx_.message(1, __function_name__, "average number of iterations: %12.6f\n",
142  static_cast<double>(num_dav_iter) / kset__.num_kpoints());
143  }
144 
145  /* synchronize eigen-values */
146  kset__.sync_band_energies();
147 
148  ctx_.message(2, __function_name__, "%s", "Lowest band energies\n");
149  if (ctx_.control().verbosity_ >= 2 && ctx_.comm().rank() == 0) {
150  for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
151  std::printf("ik : %2i, ", ik);
152  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_bands()); j++) {
153  std::printf("%12.6f", kset__[ik]->band_energy(j, 0));
154  }
155  if (ctx_.num_mag_dims() == 1) {
156  std::printf("\n ");
157  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_bands()); j++) {
158  std::printf("%12.6f", kset__[ik]->band_energy(j, 1));
159  }
160  }
161  std::printf("\n");
162  }
163  }
164 }
