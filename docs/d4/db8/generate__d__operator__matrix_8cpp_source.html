42 void Potential::generate_D_operator_matrix()
43 {
44  PROFILE("sirius::Potential::generate_D_operator_matrix");
45 
46  auto spl_ngv_loc = ctx_.split_gvec_local();
47 
48  if (ctx_.augmentation_op(0)) {
49  ctx_.augmentation_op(0)->prepare(stream_id(0), &ctx_.mem_pool(memory_t::device));
50  }
51  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
52  auto& atom_type = unit_cell_.atom_type(iat);
53  int nbf = atom_type.mt_basis_size();
54 
55  /* start copy of Q(G) for the next atom type */
56  if (ctx_.processing_unit() == device_t::GPU) {
57  acc::sync_stream(stream_id(0));
58  if (iat + 1 != unit_cell_.num_atom_types() && ctx_.augmentation_op(iat + 1)) {
59  ctx_.augmentation_op(iat + 1)->prepare(stream_id(0), &ctx_.mem_pool(memory_t::device));
60  }
61  }
62 
63  /* trivial case */
64  if (!atom_type.augment() || atom_type.num_atoms() == 0) {
65  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
66  for (int i = 0; i < atom_type.num_atoms(); i++) {
67  int ia = atom_type.atom_id(i);
68  auto& atom = unit_cell_.atom(ia);
69 
70  for (int xi2 = 0; xi2 < nbf; xi2++) {
71  for (int xi1 = 0; xi1 < nbf; xi1++) {
72  atom.d_mtrx(xi1, xi2, iv) = 0;
73  }
74  }
75  }
76  }
77  continue;
78  }
79  matrix<double> d_tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms(), ctx_.mem_pool(memory_t::host));
80  if (ctx_.processing_unit() == device_t::GPU) {
81  d_tmp.allocate(ctx_.mem_pool(memory_t::device));
82  }
83 
84  ctx_.print_memory_usage(__FILE__, __LINE__);
85 
86  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
87  matrix<double> veff_a(2 * spl_ngv_loc.local_size(), atom_type.num_atoms(), ctx_.mem_pool(memory_t::host));
88 
89  auto la = linalg_t::blas;
90  auto mem = memory_t::host;
91 
92  d_tmp.zero();
93  if (ctx_.processing_unit() == device_t::GPU) {
94  la = linalg_t::gpublas;
95  mem = memory_t::device;
96  d_tmp.zero(memory_t::device);
97  veff_a.allocate(ctx_.mem_pool(memory_t::device));
98  }
99 
100  /* split a large loop over G-vectors into blocks */
101  for (int ib = 0; ib < spl_ngv_loc.num_ranks(); ib++) {
102  int g_begin = spl_ngv_loc.global_index(0, ib);
103  int g_end = g_begin + spl_ngv_loc.local_size(ib);
104 
105  switch (ctx_.processing_unit()) {
106  case device_t::CPU: {
107  #pragma omp parallel for schedule(static)
108  for (int i = 0; i < atom_type.num_atoms(); i++) {
109  int ia = atom_type.atom_id(i);
110 
111  for (int igloc = g_begin; igloc < g_end; igloc++) {
112  int ig = ctx_.gvec().offset() + igloc;
113  /* V(G) * exp(i * G * r_{alpha}) */
114  auto z = component(iv).f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia);
115  veff_a(2 * (igloc - g_begin), i) = z.real();
116  veff_a(2 * (igloc - g_begin) + 1, i) = z.imag();
117  }
118  }
119  break;
120  }
121  case device_t::GPU: {
122  /* wait for stream#1 to finish previous zgemm */
123  acc::sync_stream(stream_id(1));
124  /* copy plane wave coefficients of effective potential to GPU */
125  mdarray<double_complex, 1> veff(&component(iv).f_pw_local(g_begin), spl_ngv_loc.local_size(ib));
126  veff.allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
127 #if defined(__GPU)
128  mul_veff_with_phase_factors_gpu(atom_type.num_atoms(), spl_ngv_loc.local_size(ib),
129  veff.at(memory_t::device),
130  ctx_.gvec_coord().at(memory_t::device, g_begin, 0),
131  ctx_.gvec_coord().at(memory_t::device, g_begin, 1),
132  ctx_.gvec_coord().at(memory_t::device, g_begin, 2),
133  ctx_.unit_cell().atom_coord(iat).at(memory_t::device),
134  veff_a.at(memory_t::device), spl_ngv_loc.local_size(), 1);
135 #endif
136  break;
137  }
138  }
139  if (ctx_.control().print_checksum_) {
140  if (ctx_.processing_unit() == device_t::GPU) {
141  veff_a.copy_to(memory_t::host);
142  }
143  auto cs = veff_a.checksum();
144  std::stringstream s;
145  s << "Gvec_block_" << ib << "_veff_a";
146  utils::print_checksum(s.str(), cs);
147  }
148  linalg(la).gemm('N', 'N', nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * spl_ngv_loc.local_size(ib),
149  &linalg_const<double>::one(),
150  ctx_.augmentation_op(iat)->q_pw().at(mem, 0, 2 * g_begin),
151  ctx_.augmentation_op(iat)->q_pw().ld(),
152  veff_a.at(mem), veff_a.ld(),
153  &linalg_const<double>::one(),
154  d_tmp.at(mem), d_tmp.ld(),
155  stream_id(1));
156  } // ib (blocks of G-vectors)
157 
158  if (ctx_.processing_unit() == device_t::GPU) {
159  d_tmp.copy_to(memory_t::host);
160  }
161 
162  if (ctx_.gvec().reduced()) {
163  if (comm_.rank() == 0) {
164  for (int i = 0; i < atom_type.num_atoms(); i++) {
165  for (int j = 0; j < nbf * (nbf + 1) / 2; j++) {
166  d_tmp(j, i) = 2 * d_tmp(j, i) - component(iv).f_pw_local(0).real() *
167  ctx_.augmentation_op(iat)->q_pw(j, 0);
168  }
169  }
170  } else {
171  for (int i = 0; i < atom_type.num_atoms(); i++) {
172  for (int j = 0; j < nbf * (nbf + 1) / 2; j++) {
173  d_tmp(j, i) *= 2;
174  }
175  }
176  }
177  }
178 
179  /* sum from all ranks */
180  comm_.allreduce(d_tmp.at(memory_t::host), static_cast<int>(d_tmp.size()));
181 
182  if (ctx_.control().print_checksum_ && ctx_.comm().rank() == 0) {
183  for (int i = 0; i < atom_type.num_atoms(); i++) {
184  std::stringstream s;
185  s << "D_mtrx_val(atom_t" << iat << "_i" << i << "_c" << iv << ")";
186  auto cs = mdarray<double, 1>(&d_tmp(0, i), nbf * (nbf + 1) / 2).checksum();
187  utils::print_checksum(s.str(), cs);
188  }
189  }
190 
191  #pragma omp parallel for schedule(static)
192  for (int i = 0; i < atom_type.num_atoms(); i++) {
193  int ia = atom_type.atom_id(i);
194  auto& atom = unit_cell_.atom(ia);
195 
196  for (int xi2 = 0; xi2 < nbf; xi2++) {
197  for (int xi1 = 0; xi1 <= xi2; xi1++) {
198  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
199  /* D-matix is symmetric */
200  atom.d_mtrx(xi1, xi2, iv) = atom.d_mtrx(xi2, xi1, iv) = d_tmp(idx12, i) * unit_cell_.omega();
201  }
202  }
203  }
204  }
205  ctx_.augmentation_op(iat)->dismiss();
206 
207  }
208 }
