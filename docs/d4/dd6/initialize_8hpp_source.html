25 inline void K_point::initialize()
26 {
27  PROFILE("sirius::K_point::initialize");
28 
29  zil_.resize(ctx_.lmax_apw() + 1);
30  for (int l = 0; l <= ctx_.lmax_apw(); l++) {
31  zil_[l] = std::pow(double_complex(0, 1), l);
32  }
33 
34  l_by_lm_ = utils::l_by_lm(ctx_.lmax_apw());
35 
36  int bs = ctx_.cyclic_block_size();
37 
38  if (ctx_.control().use_second_variation_ && ctx_.full_potential()) {
39  assert(ctx_.num_fv_states() > 0);
40  fv_eigen_values_.resize(ctx_.num_fv_states());
41  }
42 
43  /* In case of collinear magnetism we store only non-zero spinor components.
44  *
45  * non magnetic case:
46  * .---.
47  * | |
48  * .---.
49  *
50  * collinear case:
51  * .---. .---.
52  * |uu | 0 |uu |
53  * .---.---. -> .---.
54  * 0 |dd | |dd |
55  * .---. .---.
56  *
57  * non collinear case:
58  * .-------.
59  * | |
60  * .-------.
61  * | |
62  * .-------.
63  */
64  int nst = ctx_.num_bands();
65 
66  auto mem_type_evp = (ctx_.std_evp_solver_type() == ev_solver_t::magma) ? memory_t::host_pinned : memory_t::host;
67  auto mem_type_gevp = (ctx_.gen_evp_solver_type() == ev_solver_t::magma) ? memory_t::host_pinned : memory_t::host;
68 
69  /* build a full list of G+k vectors for all MPI ranks */
70  generate_gkvec(ctx_.gk_cutoff());
71  /* build a list of basis functions */
72  generate_gklo_basis();
73 
74  if (ctx_.full_potential()) {
75  if (ctx_.control().use_second_variation_) {
76  if (ctx_.need_sv()) {
77  /* in case of collinear magnetism store pure up and pure dn components, otherwise store the full matrix */
78  for (int is = 0; is < ctx_.num_spin_dims(); is++) {
79  sv_eigen_vectors_[is] = dmatrix<double_complex>(nst, nst, ctx_.blacs_grid(), bs, bs, mem_type_evp);
80  }
81  }
82  /* allocate fv eien vectors */
83  fv_eigen_vectors_slab_ = std::unique_ptr<Wave_functions>(
84  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(),
85  [this](int ia){return unit_cell_.atom(ia).mt_lo_basis_size();}, ctx_.num_fv_states(), ctx_.preferred_memory_t()));
86 
87  fv_eigen_vectors_slab_->pw_coeffs(0).prime().zero();
88  fv_eigen_vectors_slab_->mt_coeffs(0).prime().zero();
89  /* starting guess for wave-functions */
90  for (int i = 0; i < ctx_.num_fv_states(); i++) {
91  for (int igloc = 0; igloc < gkvec().gvec_count(comm().rank()); igloc++) {
92  int ig = igloc + gkvec().gvec_offset(comm().rank());
93  if (ig == i) {
94  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 1.0;
95  }
96  if (ig == i + 1) {
97  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.5;
98  }
99  if (ig == i + 2) {
100  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.125;
101  }
102  }
103  }
104  if (ctx_.iterative_solver_input().type_ == "exact") {
105  /* ELPA needs a full matrix of eigen-vectors as it uses it as a work space */
106  fv_eigen_vectors_ = dmatrix<double_complex>(gklo_basis_size(), gklo_basis_size(), ctx_.blacs_grid(), bs, bs, mem_type_gevp);
107  } else {
108  int ncomp = ctx_.iterative_solver_input().num_singular_;
109  if (ncomp < 0) {
110  ncomp = ctx_.num_fv_states() / 2;
111  }
112 
113  singular_components_ = std::unique_ptr<Wave_functions>(new Wave_functions(gkvec_partition(), ncomp, ctx_.preferred_memory_t()));
114  singular_components_->pw_coeffs(0).prime().zero();
115  /* starting guess for wave-functions */
116  for (int i = 0; i < ncomp; i++) {
117  for (int igloc = 0; igloc < gkvec().count(); igloc++) {
118  int ig = igloc + gkvec().offset();
119  if (ig == i) {
120  singular_components_->pw_coeffs(0).prime(igloc, i) = 1.0;
121  }
122  if (ig == i + 1) {
123  singular_components_->pw_coeffs(0).prime(igloc, i) = 0.5;
124  }
125  if (ig == i + 2) {
126  singular_components_->pw_coeffs(0).prime(igloc, i) = 0.125;
127  }
128  }
129  }
130  if (ctx_.control().print_checksum_) {
131  auto cs = singular_components_->checksum_pw(CPU, 0, 0, ncomp);
132  if (comm().rank() == 0) {
133  utils::print_checksum("singular_components", cs);
134  }
135  }
136  }
137 
138  fv_states_ = std::unique_ptr<Wave_functions>(new Wave_functions(gkvec_partition(),
139  unit_cell_.num_atoms(),
140  [this](int ia)
141  {
142  return unit_cell_.atom(ia).mt_basis_size();
143  },
144  ctx_.num_fv_states(),
145  ctx_.preferred_memory_t()));
146 
147  spinor_wave_functions_ = std::unique_ptr<Wave_functions>(
148  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(), [this](int ia)
149  {
150  return unit_cell_.atom(ia).mt_basis_size();
151  }, nst, ctx_.preferred_memory_t(),
152  ctx_.num_spins()));
153  } else {
154  TERMINATE_NOT_IMPLEMENTED
155  }
156  } else {
157  spinor_wave_functions_ = std::unique_ptr<Wave_functions>(
158  new Wave_functions(gkvec_partition(), nst, ctx_.preferred_memory_t(), ctx_.num_spins()));
159  }
160 
161  update();
162 }
