37template <typename T>
38void Stress::calc_stress_nonloc_aux()
39{
40 PROFILE("sirius::Stress|nonloc");
41
42 mdarray<real_type<T>, 2> collect_result(9, ctx_.unit_cell().num_atoms());
43 collect_result.zero();
44
45 stress_nonloc_.zero();
46
47 /* if there are no beta projectors then get out there */
48 if (ctx_.unit_cell().mt_lo_basis_size() == 0) {
49 return;
50 }
51
52 ctx_.print_memory_usage(__FILE__, __LINE__);
53
54 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
55 int ik = kset_.spl_num_kpoints(ikloc);
56 auto kp = kset_.get<real_type<T>>(ik);
57 if (is_device_memory(ctx_.preferred_memory_t())) {
58 int nbnd = ctx_.num_bands();
59 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
60 /* allocate GPU memory */
61 kp->spinor_wave_functions().pw_coeffs(ispn).allocate(ctx_.mem_pool(memory_t::device));
62 kp->spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, nbnd);
63 }
64 }
65 Beta_projectors_strain_deriv<real_type<T>> bp_strain_deriv(ctx_, kp->gkvec(), kp->igk_loc());
66
67 Non_local_functor<T> nlf(ctx_, bp_strain_deriv);
68
69 nlf.add_k_point_contribution(*kp, collect_result);
70
71 if (is_device_memory(ctx_.preferred_memory_t())) {
72 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
73 /* deallocate GPU memory */
74 kp->spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
75 }
76 }
77 }
78
79 #pragma omp parallel
80 {
81 matrix3d<double> tmp_stress; // TODO: test pragma omp parallel for reduction(+:stress)
82
83 #pragma omp for
84 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
85 for (int i = 0; i < 3; i++) {
86 for (int j = 0; j < 3; j++) {
87 tmp_stress(i, j) -= collect_result(j * 3 + i, ia);
88 }
89 }
90 }
91
92 #pragma omp critical
93 stress_nonloc_ += tmp_stress;
94 }
95
96 ctx_.comm().allreduce(&stress_nonloc_(0, 0), 9);
97
98 stress_nonloc_ *= (1.0 / ctx_.unit_cell().omega());
99
100 symmetrize(stress_nonloc_);
101}

107matrix3d<double> Stress::calc_stress_total()
108{
109 calc_stress_kin();
110 calc_stress_har();
111 calc_stress_ewald();
112 calc_stress_vloc();
113 calc_stress_core();
114 calc_stress_xc();
115 calc_stress_us();
116 calc_stress_nonloc();
117 stress_hubbard_.zero();
118 if (ctx_.hubbard_correction()) {
119 calc_stress_hubbard();
120 }
121 stress_total_.zero();
122
123 for (int mu = 0; mu < 3; mu++) {
124 for (int nu = 0; nu < 3; nu++) {
125 stress_total_(mu, nu) = stress_kin_(mu, nu) + stress_har_(mu, nu) + stress_ewald_(mu, nu) +
126 stress_vloc_(mu, nu) + stress_core_(mu, nu) + stress_xc_(mu, nu) +
127 stress_us_(mu, nu) + stress_nonloc_(mu, nu) + stress_hubbard_(mu, nu);
128 }
129 }
130 return stress_total_;
131}

133matrix3d<double> Stress::calc_stress_hubbard()
134{
135 stress_hubbard_.zero();
136
137 /* if there are no beta projectors then get out there */
138 /* TODO : Need to fix the case where pp have no beta projectors */
139 if (ctx_.unit_cell().mt_lo_basis_size() == 0) {
140 TERMINATE("Hubbard forces : Your pseudo potentials do not have beta projectors. This need a proper fix");
141 return stress_hubbard_;
142 }
143
144 mdarray<double_complex, 5> dn(potential_.U().max_number_of_orbitals_per_atom(),
145 potential_.U().max_number_of_orbitals_per_atom(),
146 2, ctx_.unit_cell().num_atoms(), 9);
147
148 Q_operator<double> q_op(ctx_);
149
150 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
151 dn.zero();
152 int ik = kset_.spl_num_kpoints(ikloc);
153 auto kp = kset_.get<double>(ik);
154
155 kp->beta_projectors().prepare();
156
157 if (ctx_.num_mag_dims() == 3) {
158 TERMINATE("Hubbard stress correction is only implemented for the simple hubbard correction.");
159 }
160
161 /* compute the derivative of the occupancies numbers */
162 potential_.U().compute_occupancies_stress_derivatives(*kp, q_op, dn);
163 for (int dir1 = 0; dir1 < 3; dir1++) {
164 for (int dir2 = 0; dir2 < 3; dir2++) {
165 for (int ia1 = 0; ia1 < ctx_.unit_cell().num_atoms(); ia1++) {
166 const auto& atom = ctx_.unit_cell().atom(ia1);
167 if (atom.type().hubbard_correction()) {
168 const int lmax_at = 2 * atom.type().lo_descriptor_hub(0).l + 1;
169 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
170 for (int m1 = 0; m1 < lmax_at; m1++) {
171 for (int m2 = 0; m2 < lmax_at; m2++) {
172 stress_hubbard_(dir1, dir2) -= (potential_.hubbard_potential().local(ia1)(m2, m1, ispn) *
173 dn(m1, m2, ispn, ia1, dir1 + 3 * dir2)).real() /
174 ctx_.unit_cell().omega();
175 }
176 }
177 }
178 }
179 }
180 }
181 }
182 kp->beta_projectors().dismiss();
183 }
184
185 /* global reduction */
186 kset_.comm().allreduce(&stress_hubbard_(0, 0), 9);
187 symmetrize(stress_hubbard_);
188
189 return stress_hubbard_;
190}

192matrix3d<double> Stress::calc_stress_core()
193{
194 stress_core_.zero();
195
196 bool empty = true;
197
198 /* check if the core atomic wave functions are set up or not */
199 /* if not then the core correction is simply ignored */
200
201 for (int ia = 0; (ia < ctx_.unit_cell().num_atoms()) && empty; ia++) {
202 Atom& atom = ctx_.unit_cell().atom(ia);
203 if (!atom.type().ps_core_charge_density().empty()) {
204 empty = false;
205 }
206 }
207
208 if (empty) {
209 return stress_core_;
210 }
211
212 potential_.xc_potential().fft_transform(-1);
213
214 auto q = ctx_.gvec().shells_len();
215 auto ff = ctx_.ps_core_ri_djl().values(q, ctx_.comm());
216 auto drhoc = ctx_.make_periodic_function<index_domain_t::local>(ff);
217
218 double sdiag{0};
219 int ig0 = ctx_.gvec().skip_g0();
220
221 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
222 auto G = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
223 auto g = G.length();
224
225 for (int mu : {0, 1, 2}) {
226 for (int nu : {0, 1, 2}) {
227 stress_core_(mu, nu) -= std::real(std::conj(potential_.xc_potential().f_pw_local(igloc)) *
228 drhoc[igloc]) * G[mu] * G[nu] / g;
229 }
230 }
231
232 sdiag += std::real(std::conj(potential_.xc_potential().f_pw_local(igloc)) *
233 density_.rho_pseudo_core().f_pw_local(igloc));
234 }
235
236 if (ctx_.gvec().reduced()) {
237 stress_core_ *= 2;
238 sdiag *= 2;
239 }
240 if (ctx_.comm().rank() == 0) {
241 sdiag += std::real(std::conj(potential_.xc_potential().f_pw_local(0)) *
242 density_.rho_pseudo_core().f_pw_local(0));
243 }
244
245 for (int mu : {0, 1, 2}) {
246 stress_core_(mu, mu) -= sdiag;
247 }
248
249 ctx_.comm().allreduce(&stress_core_(0, 0), 9);
250
251 symmetrize(stress_core_);
252
253 return stress_core_;
254}

256matrix3d<double> Stress::calc_stress_xc()
257{
258 stress_xc_.zero();
259
260 double e = sirius::energy_exc(density_, potential_) - sirius::energy_vxc(density_, potential_) -
261 sirius::energy_bxc(density_, potential_);
262
263 for (int l = 0; l < 3; l++) {
264 stress_xc_(l, l) = e / ctx_.unit_cell().omega();
265 }
266
267 if (potential_.is_gradient_correction()) {
268
269 matrix3d<double> t;
270
271 /* factor 2 in the expression for gradient correction comes from the
272 derivative of sigm (which is grad(rho) * grad(rho)) */
273
274 if (ctx_.num_spins() == 1) {
275 Smooth_periodic_function<double> rhovc(ctx_.spfft<double>(), ctx_.gvec_partition());
276 rhovc.zero();
277 rhovc.add(density_.rho());
278 rhovc.add(density_.rho_pseudo_core());
279
280 /* transform to PW domain */
281 rhovc.fft_transform(-1);
282
283 /* generate pw coeffs of the gradient */
284 auto grad_rho = gradient(rhovc);
285
286 /* gradient in real space */
287 for (int x : {0, 1, 2}) {
288 grad_rho[x].fft_transform(1);
289 }
290
291 for (int irloc = 0; irloc < ctx_.spfft<double>().local_slice_size(); irloc++) {
292 for (int mu = 0; mu < 3; mu++) {
293 for (int nu = 0; nu < 3; nu++) {
294 t(mu, nu) += 2 * grad_rho[mu].f_rg(irloc) * grad_rho[nu].f_rg(irloc) *
295 potential_.vsigma(0).f_rg(irloc);
296 }
297 }
298 }
299 } else {
300 auto result = get_rho_up_dn<true>(density_);
301 auto& rho_up = *result[0];
302 auto& rho_dn = *result[1];
303
304 /* transform to PW domain */
305 rho_up.fft_transform(-1);
306 rho_dn.fft_transform(-1);
307
308 /* generate pw coeffs of the gradient */
309 auto grad_rho_up = gradient(rho_up);
310 auto grad_rho_dn = gradient(rho_dn);
311
312 /* gradient in real space */
313 for (int x : {0, 1, 2}) {
314 grad_rho_up[x].fft_transform(1);
315 grad_rho_dn[x].fft_transform(1);
316 }
317
318 for (int irloc = 0; irloc < ctx_.spfft<double>().local_slice_size(); irloc++) {
319 for (int mu = 0; mu < 3; mu++) {
320 for (int nu = 0; nu < 3; nu++) {
321 t(mu, nu) += grad_rho_up[mu].f_rg(irloc) * grad_rho_up[nu].f_rg(irloc) * 2 *
322 potential_.vsigma(0).f_rg(irloc) +
323 (grad_rho_up[mu].f_rg(irloc) * grad_rho_dn[nu].f_rg(irloc) +
324 grad_rho_dn[mu].f_rg(irloc) * grad_rho_up[nu].f_rg(irloc)) *
325 potential_.vsigma(1).f_rg(irloc) +
326 grad_rho_dn[mu].f_rg(irloc) * grad_rho_dn[nu].f_rg(irloc) * 2 *
327 potential_.vsigma(2).f_rg(irloc);
328 }
329 }
330 }
331 }
332 Communicator(ctx_.spfft<double>().communicator()).allreduce(&t(0, 0), 9);
333 t *= (-1.0 / ctx_.fft_grid().num_points());
334 stress_xc_ += t;
335 }
336
337 symmetrize(stress_xc_);
338
339 return stress_xc_;
340}

342matrix3d<double> Stress::calc_stress_us()
343{
344 PROFILE("sirius::Stress|us");
345
346 stress_us_.zero();
347
348 /* check if we have beta projectors. Only for pseudo potentials */
349 if (ctx_.unit_cell().mt_lo_basis_size() == 0) {
350 return stress_us_;
351 }
352
353 auto& ri = ctx_.aug_ri();
354 auto& ri_dq = ctx_.aug_ri_djl();
355
356 potential_.fft_transform(-1);
357
358 Augmentation_operator_gvec_deriv q_deriv(ctx_, ctx_.unit_cell().lmax(), ctx_.gvec(), ctx_.gvec_tp());
359
360 linalg_t la{linalg_t::none};
361 memory_t qmem{memory_t::none};
362
363 memory_pool* mp{nullptr};
364 switch (ctx_.processing_unit()) {
365 case device_t::CPU: {
366 mp = &ctx_.mem_pool(memory_t::host);
367 la = linalg_t::blas;
368 qmem = memory_t::host;
369 break;
370 }
371 case device_t::GPU: {
372 mp = &ctx_.mem_pool(memory_t::host_pinned);
373 la = linalg_t::spla;
374 qmem = memory_t::device;
375 break;
376 }
377 }
378
379 for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
380 auto& atom_type = ctx_.unit_cell().atom_type(iat);
381 if (!atom_type.augment() || atom_type.num_atoms() == 0) {
382 continue;
383 }
384
385 q_deriv.prepare(atom_type, ri, ri_dq);
386
387 int nbf = atom_type.mt_basis_size();
388
389 /* get auxiliary density matrix */
390 auto dm = density_.density_matrix_aux(density_.density_matrix(), iat);
391
392 sddk::mdarray<double_complex, 2> phase_factors(atom_type.num_atoms(), ctx_.gvec().count(),
393 ctx_.mem_pool(memory_t::host));
394
395 PROFILE_START("sirius::Stress|us|phase_fac");
396 #pragma omp parallel for schedule(static)
397 for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
398 int ig = ctx_.gvec().offset() + igloc;
399 for (int i = 0; i < atom_type.num_atoms(); i++) {
400 phase_factors(i, igloc) = ctx_.gvec_phase_factor(ig, atom_type.atom_id(i));
401 }
402 }
403 PROFILE_STOP("sirius::Stress|us|phase_fac");
404
405 mdarray<double, 2> v_tmp(atom_type.num_atoms(), ctx_.gvec().count() * 2, *mp);
406 mdarray<double, 2> tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms(), *mp);
407 /* over spin components, can be from 1 to 4 */
408 for (int ispin = 0; ispin < ctx_.num_mag_dims() + 1; ispin++) {
409 for (int nu = 0; nu < 3; nu++) {
410 q_deriv.generate_pw_coeffs(atom_type, nu);
411
412 for (int mu = 0; mu < 3; mu++) {
413 PROFILE_START("sirius::Stress|us|prepare");
414 int igloc0{0};
415 if (ctx_.comm().rank() == 0) {
416 for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
417 v_tmp(ia, 0) = v_tmp(ia, 1) = 0;
418 }
419 igloc0 = 1;
420 }
421 #pragma omp parallel for schedule(static)
422 for (int igloc = igloc0; igloc < ctx_.gvec().count(); igloc++) {
423 auto gvc = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
424 double g = gvc.length();
425
426 for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
427 auto z = phase_factors(ia, igloc) * potential_.component(ispin).f_pw_local(igloc) *
428 (-gvc[mu] / g);
429 v_tmp(ia, 2 * igloc) = z.real();
430 v_tmp(ia, 2 * igloc + 1) = z.imag();
431 }
432 }
433 PROFILE_STOP("sirius::Stress|us|prepare");
434
435 PROFILE_START("sirius::Stress|us|gemm");
436 linalg(la).gemm('N', 'T', nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
437 &linalg_const<double>::one(),
438 q_deriv.q_pw().at(qmem), q_deriv.q_pw().ld(),
439 v_tmp.at(memory_t::host), v_tmp.ld(),
440 &linalg_const<double>::zero(),
441 tmp.at(memory_t::host), tmp.ld());
442 PROFILE_STOP("sirius::Stress|us|gemm");
443
444 for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
445 for (int i = 0; i < nbf * (nbf + 1) / 2; i++) {
446 stress_us_(mu, nu) += tmp(i, ia) * dm(i, ia, ispin) * q_deriv.sym_weight(i);
447 }
448 }
449 }
450 }
451 }
452 }
453
454 ctx_.comm().allreduce(&stress_us_(0, 0), 9);
455 if (ctx_.gvec().reduced()) {
456 stress_us_ *= 2;
457 }
458
459 stress_us_ *= (1.0 / ctx_.unit_cell().omega());
460
461 symmetrize(stress_us_);
462
463 return stress_us_;
464}

466matrix3d<double> Stress::calc_stress_ewald()
467{
468 PROFILE("sirius::Stress|ewald");
469
470 stress_ewald_.zero();
471
472 double lambda = ctx_.ewald_lambda();
473
474 auto& uc = ctx_.unit_cell();
475
476 int ig0 = ctx_.gvec().skip_g0();
477 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
478 int ig = ctx_.gvec().offset() + igloc;
479
480 auto G = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
481 double g2 = std::pow(G.length(), 2);
482 double g2lambda = g2 / 4.0 / lambda;
483
484 double_complex rho(0, 0);
485
486 for (int ia = 0; ia < uc.num_atoms(); ia++) {
487 rho += ctx_.gvec_phase_factor(ig, ia) * static_cast<double>(uc.atom(ia).zn());
488 }
489
490 double a1 = twopi * std::pow(std::abs(rho) / uc.omega(), 2) * std::exp(-g2lambda) / g2;
491
492 for (int mu : {0, 1, 2}) {
493 for (int nu : {0, 1, 2}) {
494 stress_ewald_(mu, nu) += a1 * G[mu] * G[nu] * 2 * (g2lambda + 1) / g2;
495 }
496 }
497
498 for (int mu : {0, 1, 2}) {
499 stress_ewald_(mu, mu) -= a1;
500 }
501 }
502
503 if (ctx_.gvec().reduced()) {
504 stress_ewald_ *= 2;
505 }
506
507 ctx_.comm().allreduce(&stress_ewald_(0, 0), 9);
508
509 for (int mu : {0, 1, 2}) {
510 stress_ewald_(mu, mu) += twopi * std::pow(uc.num_electrons() / uc.omega(), 2) / 4 / lambda;
511 }
512
513 for (int ia = 0; ia < uc.num_atoms(); ia++) {
514 for (int i = 1; i < uc.num_nearest_neighbours(ia); i++) {
515 auto ja = uc.nearest_neighbour(i, ia).atom_id;
516 auto d = uc.nearest_neighbour(i, ia).distance;
517 auto rc = uc.nearest_neighbour(i, ia).rc;
518
519 double a1 = (0.5 * uc.atom(ia).zn() * uc.atom(ja).zn() / uc.omega() / std::pow(d, 3)) *
520 (-2 * std::exp(-lambda * std::pow(d, 2)) * std::sqrt(lambda / pi) * d -
521 std::erfc(std::sqrt(lambda) * d));
522
523 for (int mu : {0, 1, 2}) {
524 for (int nu : {0, 1, 2}) {
525 stress_ewald_(mu, nu) += a1 * rc[mu] * rc[nu];
526 }
527 }
528 }
529 }
530
531 symmetrize(stress_ewald_);
532
533 return stress_ewald_;
534}

594matrix3d<double> Stress::calc_stress_har()
595{
596 PROFILE("sirius::Stress|har");
597
598 stress_har_.zero();
599
600 int ig0 = ctx_.gvec().skip_g0();
601 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
602 auto G = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
603 double g2 = std::pow(G.length(), 2);
604 auto z = density_.rho().f_pw_local(igloc);
605 double d = twopi * (std::pow(z.real(), 2) + std::pow(z.imag(), 2)) / g2;
606
607 for (int mu : {0, 1, 2}) {
608 for (int nu : {0, 1, 2}) {
609 stress_har_(mu, nu) += d * 2 * G[mu] * G[nu] / g2;
610 }
611 }
612 for (int mu : {0, 1, 2}) {
613 stress_har_(mu, mu) -= d;
614 }
615 }
616
617 if (ctx_.gvec().reduced()) {
618 stress_har_ *= 2;
619 }
620
621 ctx_.comm().allreduce(&stress_har_(0, 0), 9);
622
623 symmetrize(stress_har_);
624
625 return stress_har_;
626}

628template <typename T>
629void
630Stress::calc_stress_kin_aux()
631{
632 stress_kin_.zero();
633
634 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
635 int ik = kset_.spl_num_kpoints(ikloc);
636 auto kp = kset_.get<T>(ik);
637
638 double fact = kp->gkvec().reduced() ? 2.0 : 1.0;
639 fact *= kp->weight();
640
641 #pragma omp parallel
642 {
643 matrix3d<double> tmp;
644 for (int ispin = 0; ispin < ctx_.num_spins(); ispin++) {
645 #pragma omp for
646 for (int i = 0; i < kp->num_occupied_bands(ispin); i++) {
647 for (int igloc = 0; igloc < kp->num_gkvec_loc(); igloc++) {
648 auto Gk = kp->gkvec().template gkvec_cart<index_domain_t::local>(igloc);
649
650 double f = kp->band_occupancy(i, ispin);
651 auto z = kp->spinor_wave_functions().pw_coeffs(ispin).prime(igloc, i);
652 double d = fact * f * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
653 for (int mu : {0, 1, 2}) {
654 for (int nu : {0, 1, 2}) {
655 tmp(mu, nu) += Gk[mu] * Gk[nu] * d;
656 }
657 }
658 }
659 }
660 }
661 #pragma omp critical
662 stress_kin_ += tmp;
663 }
664 } // ikloc
665
666 ctx_.comm().allreduce(&stress_kin_(0, 0), 9);
667
668 stress_kin_ *= (-1.0 / ctx_.unit_cell().omega());
669
670 symmetrize(stress_kin_);
671}

686void Stress::symmetrize(matrix3d<double>& mtrx__) const
687{
688 if (!ctx_.use_symmetry()) {
689 return;
690 }
691
692 matrix3d<double> result;
693
694 for (int i = 0; i < ctx_.unit_cell().symmetry().size(); i++) {
695 auto R = ctx_.unit_cell().symmetry()[i].spg_op.Rcp;
696 result = result + dot(dot(transpose(R), mtrx__), R);
697 }
698
699 mtrx__ = result * (1.0 / ctx_.unit_cell().symmetry().size());
700
701 std::vector<std::array<int, 2>> idx = {{0, 1}, {0, 2}, {1, 2}};
702 for (auto e : idx) {
703 mtrx__(e[0], e[1]) = mtrx__(e[1], e[0]) = 0.5 * (mtrx__(e[0], e[1]) + mtrx__(e[1], e[0]));
704 }
705}

707matrix3d<double> Stress::calc_stress_vloc()
708{
709 PROFILE("sirius::Stress|vloc");
710
711 stress_vloc_.zero();
712
713 auto q = ctx_.gvec().shells_len();
714 auto ri_vloc = ctx_.vloc_ri().values(q, ctx_.comm());
715 auto ri_vloc_dg = ctx_.vloc_ri_djl().values(q, ctx_.comm());
716
717 auto v = ctx_.make_periodic_function<index_domain_t::local>(ri_vloc);
718 auto dv = ctx_.make_periodic_function<index_domain_t::local>(ri_vloc_dg);
719
720 double sdiag{0};
721
722 int ig0 = ctx_.gvec().skip_g0();
723 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
724
725 auto G = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
726
727 for (int mu : {0, 1, 2}) {
728 for (int nu : {0, 1, 2}) {
729 stress_vloc_(mu, nu) +=
730 std::real(std::conj(density_.rho().f_pw_local(igloc)) * dv[igloc]) * G[mu] * G[nu];
731 }
732 }
733
734 sdiag += std::real(std::conj(density_.rho().f_pw_local(igloc)) * v[igloc]);
735 }
736
737 if (ctx_.gvec().reduced()) {
738 stress_vloc_ *= 2;
739 sdiag *= 2;
740 }
741 if (ctx_.comm().rank() == 0) {
742 sdiag += std::real(std::conj(density_.rho().f_pw_local(0)) * v[0]);
743 }
744
745 for (int mu : {0, 1, 2}) {
746 stress_vloc_(mu, mu) -= sdiag;
747 }
748
749 ctx_.comm().allreduce(&stress_vloc_(0, 0), 9);
750
751 symmetrize(stress_vloc_);
752
753 return stress_vloc_;
754}
