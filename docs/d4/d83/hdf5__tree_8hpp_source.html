213  /// Write a multidimensional array.
214  template <typename T>
215  void write(const std::string& name, T const* data, const std::vector<int>& dims)
216  {
217  /* open group */
218  HDF5_group group(file_id_, path_);
219 
220  /* make dataspace */
221  HDF5_dataspace dataspace(dims);
222 
223  /* create new dataset */
224  HDF5_dataset dataset(group, dataspace, name, hdf5_type_wrapper<T>::type_id());
225 
226  /* write data */
227  if (H5Dwrite(dataset.id(), hdf5_type_wrapper<T>::type_id(), dataspace.id(), H5S_ALL, H5P_DEFAULT, data) < 0) {
228  TERMINATE("error in H5Dwrite()");
229  }
230  }

232  /// Read a multidimensional array.
233  template <typename T>
234  void read(const std::string& name, T* data, const std::vector<int>& dims)
235  {
236  HDF5_group group(file_id_, path_);
237 
238  HDF5_dataspace dataspace(dims);
239 
240  HDF5_dataset dataset(group.id(), name);
241 
242  if (H5Dread(dataset.id(), hdf5_type_wrapper<T>::type_id(), dataspace.id(), H5S_ALL, H5P_DEFAULT, data) < 0) {
243  TERMINATE("error in H5Dread()");
244  }
245  }

253  /// Constructor to create the HDF5 tree.
254  HDF5_tree(const std::string& file_name__, hdf5_access_t access__)
255  : file_name_(file_name__)
256  {
257  if (H5open() < 0) {
258  TERMINATE("error in H5open()");
259  }
260 
261  if (false) {
262  H5Eset_auto(H5E_DEFAULT, NULL, NULL);
263  }
264 
265  switch (access__) {
266  case hdf5_access_t::truncate: {
267  file_id_ = H5Fcreate(file_name_.c_str(), H5F_ACC_TRUNC | H5F_ACC_SWMR_WRITE, H5P_DEFAULT, H5P_DEFAULT);
268  if (file_id_ < 0) {
269  TERMINATE("error in H5Fcreate()");
270  }
271  break;
272  }
273  case hdf5_access_t::read_write: {
274  file_id_ = H5Fopen(file_name_.c_str(), H5F_ACC_RDWR | H5F_ACC_SWMR_WRITE, H5P_DEFAULT);
275  break;
276  }
277  case hdf5_access_t::read_only: {
278  file_id_ = H5Fopen(file_name_.c_str(), H5F_ACC_RDONLY | H5F_ACC_SWMR_READ, H5P_DEFAULT);
279  break;
280  }
281  }
282  if (file_id_ < 0) {
283  TERMINATE("H5Fopen() failed");
284  }
285 
286  path_ = "/";
287  }
288 
289  /// Destructor.
290  ~HDF5_tree()
291  {
292  if (root_node_) {
293  if (H5Fclose(file_id_) < 0) {
294  TERMINATE("error in H5Fclose()");
295  }
296  }
297  }
