373 /* @fortran begin function void sirius_set_pw_coeffs_v2 Set plane-wave coefficients of a periodic function.
374  @fortran argument in required void* handler Ground state handler.
375  @fortran argument in required string label Label of the function.
376  @fortran argument in required complex pw_coeffs Local array of plane-wave coefficients.
377  @fortran argument in optional int ngv Local number of G-vectors.
378  @fortran argument in optional int gvl List of G-vectors in lattice coordinates (Miller indices).
379  @fortran argument in optional int comm MPI communicator used in distribution of G-vectors
380  @fortran end */
381 void sirius_set_pw_coeffs_v2(void* const* handler__,
382  char const* label__,
383  std::complex<double> const* pw_coeffs__,
384  int const* ngv__,
385  int* gvl__,
386  int const* comm__)
387 {
388  PROFILE("sirius_api::sirius_set_pw_coeffs");
389 
390  auto& gs = static_cast<utils::any_ptr*>(*handler__)->get<sirius::DFT_ground_state>();
391 
392  std::string label(label__);
393 
394  if (gs.ctx().full_potential()) {
395  if (label == "veff") {
396  gs.potential().set_veff_pw(pw_coeffs__);
397  } else if (label == "rm_inv") {
398  gs.potential().set_rm_inv_pw(pw_coeffs__);
399  } else if (label == "rm2_inv") {
400  gs.potential().set_rm2_inv_pw(pw_coeffs__);
401  } else {
402  TERMINATE("wrong label");
403  }
404  } else {
405  assert(ngv__ != nullptr);
406  assert(gvl__ != nullptr);
407  assert(comm__ != nullptr);
408 
409  Communicator comm(MPI_Comm_f2c(*comm__));
410  mdarray<int, 2> gvec(gvl__, 3, *ngv__);
411 
412  std::vector<double_complex> v(gs.ctx().gvec().num_gvec(), 0);
413  #pragma omp parallel for schedule(static)
414  for (int i = 0; i < *ngv__; i++) {
415  vector3d<int> G(gvec(0, i), gvec(1, i), gvec(2, i));
416  auto gvc = gs.ctx().unit_cell().reciprocal_lattice_vectors() * vector3d<double>(G[0], G[1], G[2]);
417  if (gvc.length() > gs.ctx().pw_cutoff()) {
418  continue;
419  }
420  int ig = gs.ctx().gvec().index_by_gvec(G);
421  if (ig >= 0) {
422  v[ig] = pw_coeffs__[i];
423  } else {
424  if (gs.ctx().gamma_point()) {
425  ig = gs.ctx().gvec().index_by_gvec(G * (-1));
426  if (ig == -1) {
427  std::stringstream s;
428  auto gvc = gs.ctx().unit_cell().reciprocal_lattice_vectors() * vector3d<double>(G[0], G[1], G[2]);
429  s << "wrong index of G-vector" << std::endl
430  << "input G-vector: " << G << " (length: " << gvc.length() << " [a.u.^-1])" << std::endl;
431  TERMINATE(s);
432  } else {
433  v[ig] = std::conj(pw_coeffs__[i]);
434  }
435  }
436  }
437  }
438  comm.allreduce(v.data(), gs.ctx().gvec().num_gvec());
439 
440  // TODO: check if FFT transformation is necessary
441  if (label == "rho") {
442  gs.density().rho().scatter_f_pw(v);
443  gs.density().rho().fft_transform(1);
444  } else if (label == "magz") {
445  gs.density().magnetization(0).scatter_f_pw(v);
446  gs.density().magnetization(0).fft_transform(1);
447  } else if (label == "magx") {
448  gs.density().magnetization(1).scatter_f_pw(v);
449  gs.density().magnetization(1).fft_transform(1);
450  } else if (label == "magy") {
451  gs.density().magnetization(2).scatter_f_pw(v);
452  gs.density().magnetization(2).fft_transform(1);
453  } else if (label == "veff") {
454  gs.potential().effective_potential().scatter_f_pw(v);
455  gs.potential().effective_potential().fft_transform(1);
456  } else if (label == "bz") {
457  gs.potential().effective_magnetic_field(0).scatter_f_pw(v);
458  gs.potential().effective_magnetic_field(0).fft_transform(1);
459  } else if (label == "bx") {
460  gs.potential().effective_magnetic_field(1).scatter_f_pw(v);
461  gs.potential().effective_magnetic_field(1).fft_transform(1);
462  } else if (label == "by") {
463  gs.potential().effective_magnetic_field(2).scatter_f_pw(v);
464  gs.potential().effective_magnetic_field(2).fft_transform(1);
465  } else if (label == "vxc") {
466  gs.potential().xc_potential().scatter_f_pw(v);
467  gs.potential().xc_potential().fft_transform(1);
468  } else if (label == "vloc") {
469  gs.potential().local_potential().scatter_f_pw(v);
470  gs.potential().local_potential().fft_transform(1);
471  } else if (label == "dveff") {
472  gs.potential().dveff().scatter_f_pw(v);
473  } else {
474  std::stringstream s;
475  s << "wrong label in sirius_set_pw_coeffs()" << std::endl
476  << " label: " << label;
477  TERMINATE(s);
478  }
479  }

482 /* @fortran begin function void sirius_get_pw_coeffs_v2 Set plane-wave coefficients of a periodic function.
483  @fortran argument in required void* handler Ground state handler.
484  @fortran argument in required string label Label of the function.
485  @fortran argument in required complex pw_coeffs Local array of plane-wave coefficients.
486  @fortran argument in optional int ngv Local number of G-vectors.
487  @fortran argument in optional int gvl List of G-vectors in lattice coordinates (Miller indices).
488  @fortran argument in optional int comm MPI communicator used in distribution of G-vectors
489  @fortran end */
490 void sirius_get_pw_coeffs_v2(void* const* handler__,
491  char const* label__,
492  std::complex<double>* pw_coeffs__,
493  int const* ngv__,
494  int* gvl__,
495  int const* comm__)
496 {
497  PROFILE("sirius_api::sirius_get_pw_coeffs");
498 
499  auto& gs = static_cast<utils::any_ptr*>(*handler__)->get<sirius::DFT_ground_state>();
500 
501  std::string label(label__);
502  if (gs.ctx().full_potential()) {
503  STOP();
504  } else {
505  assert(ngv__ != NULL);
506  assert(gvl__ != NULL);
507  assert(comm__ != NULL);
508 
509  Communicator comm(MPI_Comm_f2c(*comm__));
510  mdarray<int, 2> gvec(gvl__, 3, *ngv__);
511 
512  std::map<std::string, sirius::Smooth_periodic_function<double>*> func = {
513  {"rho", &gs.density().rho()},
514  {"magz", &gs.density().magnetization(0)},
515  {"magx", &gs.density().magnetization(1)},
516  {"magy", &gs.density().magnetization(2)},
517  {"veff", &gs.potential().effective_potential()},
518  {"vloc", &gs.potential().local_potential()},
519  {"rhoc", &gs.density().rho_pseudo_core()}
520  };
521 
522  std::vector<double_complex> v;
523  try {
524  v = func.at(label)->gather_f_pw();
525  } catch(...) {
526  TERMINATE("wrong label");
527  }
528 
529  for (int i = 0; i < *ngv__; i++) {
530  vector3d<int> G(gvec(0, i), gvec(1, i), gvec(2, i));
531 
532  auto gvc = gs.ctx().unit_cell().reciprocal_lattice_vectors() * vector3d<double>(G[0], G[1], G[2]);
533  if (gvc.length() > gs.ctx().pw_cutoff()) {
534  pw_coeffs__[i] = 0;
535  continue;
536  }
537 
538  bool is_inverse{false};
539  int ig = gs.ctx().gvec().index_by_gvec(G);
540  if (ig == -1 && gs.ctx().gvec().reduced()) {
541  ig = gs.ctx().gvec().index_by_gvec(G * (-1));
542  is_inverse = true;
543  }
544  if (ig == -1) {
545  std::stringstream s;
546  auto gvc = gs.ctx().unit_cell().reciprocal_lattice_vectors() * vector3d<double>(G[0], G[1], G[2]);
