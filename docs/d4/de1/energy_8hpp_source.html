34 /// Compute the ion-ion electrostatic energy using Ewald method.
35 /** The following contribution (per unit cell) to the total energy has to be computed:
36  * \f[
37  * E^{ion-ion} = \frac{1}{N} \frac{1}{2} \sum_{i \neq j} \frac{Z_i Z_j}{|{\bf r}_i - {\bf r}_j|} =
38  * \frac{1}{2} \sideset{}{'} \sum_{\alpha \beta {\bf T}} \frac{Z_{\alpha} Z_{\beta}}{|{\bf r}_{\alpha} -
39  * {\bf r}_{\beta} + {\bf T}|}
40  * \f]
41  * where \f$ N \f$ is the number of unit cells in the crystal.
42  * Following the idea of Ewald the Coulomb interaction is split into two terms:
43  * \f[
44  * \frac{1}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} =
45  * \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} -
46  * {\bf r}_{\beta} + {\bf T}|} +
47  * \frac{{\rm erfc}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} -
48  * {\bf r}_{\beta} + {\bf T}|}
49  * \f]
50  * Second term is computed directly. First term is computed in the reciprocal space. Remembering that
51  * \f[
52  * \frac{1}{\Omega} \sum_{\bf G} e^{i{\bf Gr}} = \sum_{\bf T} \delta({\bf r - T})
53  * \f]
54  * we rewrite the first term as
55  * \f[
56  * \frac{1}{2} \sideset{}{'} \sum_{\alpha \beta {\bf T}} Z_{\alpha} Z_{\beta}
57  * \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} -
58  * {\bf r}_{\beta} + {\bf T}|} = \frac{1}{2} \sum_{\alpha \beta {\bf T}} Z_{\alpha} Z_{\beta}
59  * \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}
60  * {|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} -
61  * \frac{1}{2} \sum_{\alpha} Z_{\alpha}^2 2 \sqrt{\frac{\lambda}{\pi}} = \\
62  * \frac{1}{2} \sum_{\alpha \beta} Z_{\alpha} Z_{\beta} \frac{1}{\Omega} \sum_{\bf G} \int e^{i{\bf Gr}}
63  * \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha\beta} + {\bf r}|)}{|{\bf r}_{\alpha\beta} + {\bf r}|}
64  * d{\bf r} - \sum_{\alpha} Z_{\alpha}^2 \sqrt{\frac{\lambda}{\pi}}
65  * \f]
66  * The integral is computed using the \f$ \ell=0 \f$ term of the spherical expansion of the plane-wave:
67  * \f[
68  * \int e^{i{\bf Gr}} \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha\beta} + {\bf r}|)}{|{\bf r}_{\alpha\beta} +
69  * {\bf r}|} d{\bf r} =
70  * \int e^{-i{\bf r}_{\alpha \beta}{\bf G}} e^{i{\bf Gr}} \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}|)}{|{\bf r}|}
71  * d{\bf r} = e^{-i{\bf r}_{\alpha \beta}{\bf G}} 4 \pi \int_0^{\infty} \frac{\sin({G r})}{G}
72  * {\rm erf}(\sqrt{\lambda} r ) dr
73  * \f]
74  * We will split integral in two parts:
75  * \f[
76  * \int_0^{\infty} \sin({G r}) {\rm erf}(\sqrt{\lambda} r ) dr = \int_0^{b} \sin({G r})
77  * {\rm erf}(\sqrt{\lambda} r ) dr +
78  * \int_b^{\infty} \sin({G r}) dr = \frac{1}{G} e^{-\frac{G^2}{4 \lambda}}
79  * \f]
80  * where \f$ b \f$ is sufficiently large. To reproduce in Mathrmatica:
81  \verbatim
82  Limit[Limit[
83  Integrate[Sin[g*x]*Erf[Sqrt[nu] * x], {x, 0, b},
84  Assumptions -> {nu > 0, g >= 0, b > 0}] +
85  Integrate[Sin[g*(x + I*a)], {x, b, \[Infinity]},
86  Assumptions -> {a > 0, nu > 0, g >= 0, b > 0}], a -> 0],
87  b -> \[Infinity], Assumptions -> {nu > 0, g >= 0}]
88  \endverbatim
89  * The first term of the Ewald sum thus becomes:
90  * \f[
91  * \frac{2 \pi}{\Omega} \sum_{{\bf G}} \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \Big| \sum_{\alpha} Z_{\alpha}
92  * e^{-i{\bf r}_{\alpha}{\bf G}} \Big|^2 - \sum_{\alpha} Z_{\alpha}^2 \sqrt{\frac{\lambda}{\pi}}
93  * \f]
94  * For \f$ G=0 \f$ the following is done:
95  * \f[
96  * \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \approx \frac{1}{G^2}-\frac{1}{4 \lambda }
97  * \f]
98  * The term \f$ \frac{1}{G^2} \f$ is compensated together with the corresponding Hartree terms in electron-electron
99  * and electron-ion interactions (cell should be neutral) and we are left with the following conribution:
100  * \f[
101  * -\frac{2\pi}{\Omega}\frac{N_{el}^2}{4 \lambda}
102  * \f]
103  * Final expression for the Ewald energy:
104  * \f[
105  * E^{ion-ion} = \frac{1}{2} \sideset{}{'} \sum_{\alpha \beta {\bf T}} Z_{\alpha} Z_{\beta}
106  * \frac{{\rm erfc}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} -
107  * {\bf r}_{\beta} + {\bf T}|} + \frac{2 \pi}{\Omega} \sum_{{\bf G}\neq 0}
108  * \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \Big| \sum_{\alpha} Z_{\alpha} e^{-i{\bf r}_{\alpha}{\bf G}}
109  * \Big|^2 - \sum_{\alpha} Z_{\alpha}^2 \sqrt{\frac{\lambda}{\pi}} - \frac{2\pi}{\Omega}
110  * \frac{N_{el}^2}{4 \lambda}
111  * \f]
112  */
113 double ewald_energy(const Simulation_context& ctx, const Gvec& gvec, const Unit_cell& unit_cell);

150 /// Total energy of the electronic subsystem.
151 /** <b> Full potential total energy </b>
152  *
153  * From the definition of the density functional we have:
154  * \f[
155  * E[\rho] = T[\rho] + E^{H}[\rho] + E^{XC}[\rho] + E^{ext}[\rho]
156  * \f]
157  * where \f$ T[\rho] \f$ is the kinetic energy, \f$ E^{H}[\rho] \f$ - electrostatic energy of
158  * electron-electron density interaction, \f$ E^{XC}[\rho] \f$ - exchange-correlation energy
159  * and \f$ E^{ext}[\rho] \f$ - energy in the external field of nuclei.
160  *
161  * Electrostatic and external field energies are grouped in the following way:
162  * \f[
163  * \frac{1}{2} \int \int \frac{\rho({\bf r})\rho({\bf r'}) d{\bf r} d{\bf r'}}{|{\bf r} - {\bf r'}|} +
164  * \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r} = \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} +
165  * \frac{1}{2} \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r}
166  * \f]
167  * Here \f$ V^{H}({\bf r}) \f$ is the total (electron + nuclei) electrostatic potential returned by the
168  * poisson solver. Next we transform the remaining term:
169  * \f[
170  * \frac{1}{2} \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r} =
171  * \frac{1}{2} \int \int \frac{\rho({\bf r})\rho^{nuc}({\bf r'}) d{\bf r} d{\bf r'}}{|{\bf r} - {\bf r'}|} =
172  * \frac{1}{2} \int V^{H,el}({\bf r}) \rho^{nuc}({\bf r}) d{\bf r}
173  * \f]
174  *
175  * <b> Pseudopotential total energy </b>
176  *
177  * Total energy in PW-PP method has the following expression:
178  * \f[
179  * E_{tot} = \sum_{i} f_i \sum_{\sigma \sigma'} \langle \psi_i^{\sigma'} | \Big( \hat T + \sum_{\xi \xi'}
180  * |\beta_{\xi} \rangle D_{\xi \xi'}^{ion} \delta_{\sigma \sigma'} \langle \beta_{\xi'} |\Big) | \psi_i^{\sigma}
181  * \rangle + \int V^{ion}({\bf r})\rho({\bf r})d{\bf r} + \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} +
182  * E^{XC}[\rho + \rho_{core}, |{\bf m}|]
183  * \f]
184  * Ionic contribution to the non-local part of pseudopotential is diagonal in spin. The following rearrangement
185  * is performed next:
186  * \f[
187  * \int \rho({\bf r}) \Big( V^{ion}({\bf r}) + \frac{1}{2} V^{H}({\bf r}) \Big) d{\bf r} = \\
188  * \int \rho({\bf r}) \Big( V^{ion}({\bf r}) + V^{H}({\bf r}) + V^{XC}({\bf r}) \Big) d{\bf r} +
189  * \int {\bf m}({\bf r}) {\bf B}^{XC}({\bf r}) d{\bf r} -
190  * \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r})d{\bf r} -
191  * \int {\bf m}({\bf r}) {\bf B}^{XC}({\bf r}) d{\bf r} = \\
192  * \sum_{\sigma \sigma'}\int \rho_{\sigma \sigma'}({\bf r}) V_{\sigma' \sigma}^{eff}({\bf r}) d{\bf r} -
193  * \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r})d{\bf r} -
194  * \int {\bf m}({\bf r}) {\bf B}^{XC}({\bf r}) d{\bf r}
195  * \f]
196  * Where
197  * \f[
198  * \rho_{\sigma \sigma'}({\bf r}) = \sum_{i}^{occ} f_{i} \psi_{i}^{\sigma' *}({\bf r})\psi_{i}^{\sigma}({\bf r})
199  * \f]
200  * is a \f$ 2 \times 2 \f$ density matrix and
201  * \f[
202  * V_{\sigma\sigma'}^{eff}({\bf r})=\Big({\bf I}V^{eff}({\bf r})+{\boldsymbol \sigma}{\bf B}^{XC}({\bf r}) \Big)
203  * = \left( \begin{array}{cc} V^{eff}({\bf r})+B_z^{XC}({\bf r}) & B_x^{XC}({\bf r})-iB_y^{XC}({\bf r}) \\
204  * B_x^{XC}({\bf r})+iB_y^{XC}({\bf r}) & V^{eff}({\bf r})-B_z^{XC}({\bf r}) \end{array} \right)
205  * \f]
206  * is a \f$ 2 \times 2 \f$ matrix potential (see \ref dft for the full derivation).
207  *
208  * We are interested in this term:
209  * \f[
210  * \sum_{\sigma \sigma'}\int \rho_{\sigma \sigma'}({\bf r}) V_{\sigma' \sigma}^{eff}({\bf r}) d{\bf r} =
211  * \int V^{eff}({\bf r})\rho({\bf r})d{\bf r} + \int {\bf m}({\bf r}) {\bf B}^{XC}({\bf r}) d{\bf r}
212  * \f]
213  *
214  * We are going to split density into two contributions (sum of occupied bands \f$ \rho^{ps} \f$ and augmented
215  * charge \f$ \rho^{aug} \f$) and use the definition of \f$ \rho^{aug} \f$: \f[ \sum_{\sigma \sigma'}\int
216  * \rho_{\sigma \sigma'}^{aug}({\bf r}) V_{\sigma' \sigma}^{eff}({\bf r}) d{\bf r} = \sum_{\sigma \sigma'}\int
217  * \sum_{i} \sum_{\xi \xi'} f_i \langle \psi_i^{\sigma'} | \beta_{\xi} \rangle Q_{\xi \xi'}({\bf r}) \langle
218  * \beta_{\xi'} | \psi_i^{\sigma} \rangle V_{\sigma' \sigma}^{eff}({\bf r}) d{\bf r} = \sum_{\sigma \sigma'}
219  * \sum_{i}\sum_{\xi \xi'} f_i \langle \psi_i^{\sigma'} | \beta_{\xi} \rangle D_{\xi \xi', \sigma' \sigma}^{aug}
220  * \langle \beta_{\xi'} | \psi_i^{\sigma} \rangle \f] Now we can rewrite the total energy expression: \f[ E_{tot} =
221  * \sum_{i} f_i \sum_{\sigma \sigma'} \langle \psi_i^{\sigma'} | \Big( \hat T + \sum_{\xi \xi'} |\beta_{\xi} \rangle
222  * D_{\xi \xi'}^{ion} \delta_{\sigma \sigma'} + D_{\xi \xi', \sigma' \sigma}^{aug} \langle \beta_{\xi'} |\Big) |
223  * \psi_i^{\sigma} \rangle + \sum_{\sigma \sigma}
224  * \int V^{eff}_{\sigma' \sigma}({\bf r})\rho^{ps}_{\sigma \sigma'}({\bf r})d{\bf r} -
225  * \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r}) d{\bf r} -
226  * \int {\bf m}({\bf r}) {\bf B}^{XC}({\bf r}) d{\bf r} + E^{XC}[\rho + \rho_{core}, |{\bf m}|]
227  * \f]
228  * From the Kohn-Sham equations
229  * \f[
230  * \hat T |\psi_i^{\sigma} \rangle + \sum_{\sigma'} \sum_{\xi \xi'} \Big( |\beta_{\xi}
231  * \rangle D_{\xi \xi', \sigma' \sigma} \langle \beta_{\xi'}| + \hat V^{eff}_{\sigma' \sigma} \Big)
232  * | \psi_i^{\sigma'} \rangle = \varepsilon_i \Big( 1+\hat S \Big) |\psi_i^{\sigma} \rangle
233  * \f]
234  * we immediately obtain that
235  * \f[
236  * \sum_{i} f_i \varepsilon_i = \sum_{i} f_i \sum_{\sigma \sigma'} \langle \psi_i^{\sigma'} |
237  * \Big( \hat T + \sum_{\xi \xi'} |\beta_{\xi}
238  * \rangle D_{\xi \xi', \sigma' \sigma} \langle \beta_{\xi'} |\Big) |
239  * \psi_i^{\sigma} \rangle + \sum_{\sigma \sigma}
240  * \int V^{eff}_{\sigma' \sigma}({\bf r})\rho^{ps}_{\sigma \sigma'}({\bf r})d{\bf r}
241  * \f]
242  * and the total energy expression simplifies to:
243  * \f[
244  * E_{tot} = \sum_{i} f_i \varepsilon_i - \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} -
245  * \int V^{XC}({\bf r})\rho({\bf r}) d{\bf r} - \int {\bf m}({\bf r}) {\bf B}^{XC}({\bf r}) d{\bf r} +
246  * E^{XC}[\rho + \rho_{core}, |{\bf m}|]
247  * \f]
248  */
249 double total_energy(Simulation_context const& ctx, K_point_set const& kset, Density const& density,
250  Potential const& potential, double ewald_energy);
