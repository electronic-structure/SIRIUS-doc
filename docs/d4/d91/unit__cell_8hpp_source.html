332  /// Update the parameters that depend on atomic positions or lattice vectors.
333  inline void update()
334  {
335  PROFILE("sirius::Unit_cell::update");
336 
337  auto v0 = lattice_vector(0);
338  auto v1 = lattice_vector(1);
339  auto v2 = lattice_vector(2);
340 
341  double r = std::max(std::max(v0.length(), std::max(v1.length(), v2.length())),
342  parameters_.parameters_input().nn_radius_);
343 
344  find_nearest_neighbours(r);
345 
346  if (parameters_.full_potential()) {
347  /* find new MT radii and initialize radial grid */
348  if (parameters_.auto_rmt()) {
349  auto rg = get_radial_grid_t(parameters_.settings().radial_grid_);
350  std::vector<double> Rmt = find_mt_radii();
351  for (int iat = 0; iat < num_atom_types(); iat++) {
352  double r0 = atom_type(iat).radial_grid().first();
353 
354  atom_type(iat).set_radial_grid(rg.first, atom_type(iat).num_mt_points(), r0, Rmt[iat], rg.second);
355  }
356  }
357 
358  int ia, ja;
359  if (check_mt_overlap(ia, ja)) {
360  std::stringstream s;
361  s << "overlaping muffin-tin spheres for atoms " << ia << "(" << atom(ia).type().symbol() << ")"
362  << " and " << ja << "(" << atom(ja).type().symbol() << ")" << std::endl
363  << " radius of atom " << ia << " : " << atom(ia).mt_radius() << std::endl
364  << " radius of atom " << ja << " : " << atom(ja).mt_radius() << std::endl
365  << " distance : " << nearest_neighbours_[ia][1].distance << " " << nearest_neighbours_[ja][1].distance;
366  TERMINATE(s);
367  }
368 
369  min_mt_radius_ = 1e100;
370  max_mt_radius_ = 0;
371  for (int i = 0; i < num_atom_types(); i++) {
372  min_mt_radius_ = std::min(min_mt_radius_, atom_type(i).mt_radius());
373  max_mt_radius_ = std::max(max_mt_radius_, atom_type(i).mt_radius());
374  }
375  }
376 
377  if (parameters_.use_symmetry()) {
378  get_symmetry();
379  }
380 
381  spl_num_atom_symmetry_classes_ = splindex<block>(num_atom_symmetry_classes(), comm_.size(), comm_.rank());
382 
383  volume_mt_ = 0.0;
384  if (parameters_.full_potential()) {
385  for (int ia = 0; ia < num_atoms(); ia++) {
386  volume_mt_ += fourpi * std::pow(atom(ia).mt_radius(), 3) / 3.0;
387  }
388  }
389 
390  volume_it_ = omega() - volume_mt_;
391 
392  for (int iat = 0; iat < num_atom_types(); iat++) {
393  int nat = atom_type(iat).num_atoms();
394  if (nat > 0) {
395  for (int i = 0; i < nat; i++) {
396  int ia = atom_type(iat).atom_id(i);
397  for (int x: {0, 1, 2}) {
398  atom_coord_[iat](i, x) = atom(ia).position()[x];
399  }
400  }
401  if (parameters_.processing_unit() == GPU) {
402  atom_coord_[iat].copy<memory_t::host, memory_t::device>();
403  }
404  }
405  }
406  }

681  inline void import(Unit_cell_input const& inp__)
682  {
683  if (inp__.exist_) {
684  /* first, load all types */
685  for (int iat = 0; iat < (int)inp__.labels_.size(); iat++) {
686  auto label = inp__.labels_[iat];
687  auto fname = inp__.atom_files_.at(label);
688  add_atom_type(label, fname);
689  }
690  /* then load atoms */
691  for (int iat = 0; iat < (int)inp__.labels_.size(); iat++) {
692  auto label = inp__.labels_[iat];
693  auto fname = inp__.atom_files_.at(label);
694  for (size_t ia = 0; ia < inp__.coordinates_[iat].size(); ia++) {

847 inline std::vector<double> Unit_cell::find_mt_radii()
848 {
849  if (nearest_neighbours_.size() == 0) {
850  TERMINATE("array of nearest neighbours is empty");
851  }
852 
853  std::vector<double> Rmt(num_atom_types(), 1e10);
854 
855  if (parameters_.auto_rmt() == 1) {
856  for (int ia = 0; ia < num_atoms(); ia++) {
857  int id1 = atom(ia).type_id();
858  if (nearest_neighbours_[ia].size() > 1) {
859  int ja = nearest_neighbours_[ia][1].atom_id;
860  int id2 = atom(ja).type_id();
861  /* don't allow spheres to touch: take a smaller value than half a distance */
862  double R = std::min(parameters_.rmt_max(), 0.95 * nearest_neighbours_[ia][1].distance / 2);
863  /* take minimal R for the given atom type */
864  Rmt[id1] = std::min(R, Rmt[id1]);
865  Rmt[id2] = std::min(R, Rmt[id2]);
866  } else {
867  Rmt[id1] = parameters_.rmt_max();
868  }
869  }
870  }
871 
872  if (parameters_.auto_rmt() == 2) {
873  std::vector<double> scale(num_atom_types(), 1e10);
874 
875  for (int ia = 0; ia < num_atoms(); ia++) {
876  int id1 = atom(ia).type_id();
877  if (nearest_neighbours_[ia].size() > 1) {
878  int ja = nearest_neighbours_[ia][1].atom_id;
879  int id2 = atom(ja).type_id();
880 
881  double d = nearest_neighbours_[ia][1].distance;
882  double s = 0.95 * d / (atom_type(id1).mt_radius() + atom_type(id2).mt_radius());
883  scale[id1] = std::min(s, scale[id1]);
884  scale[id2] = std::min(s, scale[id2]);
885  } else {
886  scale[id1] = parameters_.rmt_max() / atom_type(id1).mt_radius();
887  }
888  }
889 
890  for (int iat = 0; iat < num_atom_types(); iat++) {
891  Rmt[iat] = std::min(parameters_.rmt_max(), atom_type(iat).mt_radius() * scale[iat]);
892  }
893  }
894 
895  /* Suppose we have 3 different atoms. First we determint Rmt between 1st and 2nd atom,
896  * then we determine Rmt between (let's say) 2nd and 3rd atom and at this point we reduce
897  * the Rmt of the 2nd atom. This means that the 1st atom gets a possibility to expand if
898  * it is far from the 3rd atom. */
899  bool inflate = true;
900 
901  if (inflate) {
902  std::vector<bool> scale_Rmt(num_atom_types(), true);
903  for (int ia = 0; ia < num_atoms(); ia++) {
904  int id1 = atom(ia).type_id();
905 
906  if (nearest_neighbours_[ia].size() > 1) {
907  int ja = nearest_neighbours_[ia][1].atom_id;
908  int id2 = atom(ja).type_id();
909  double dist = nearest_neighbours_[ia][1].distance;
910 
911  if (Rmt[id1] + Rmt[id2] > dist * 0.94) {
912  scale_Rmt[id1] = false;
913  scale_Rmt[id2] = false;
914  }
915  }
916  }
917 
918  for (int ia = 0; ia < num_atoms(); ia++) {
919  int id1 = atom(ia).type_id();
920  if (nearest_neighbours_[ia].size() > 1) {
921  int ja = nearest_neighbours_[ia][1].atom_id;
922  int id2 = atom(ja).type_id();
923  double dist = nearest_neighbours_[ia][1].distance;
924 
925  if (scale_Rmt[id1] && !scale_Rmt[id2]) {
926  Rmt[id1] = std::min(parameters_.rmt_max(), 0.95 * (dist - Rmt[id2]));
927  scale_Rmt[id1] = false;
928  }
929  }
930  }
931  }
932 
933  for (int i = 0; i < num_atom_types(); i++) {
934  if (Rmt[i] < 0.3) {
935  std::stringstream s;
936  s << "muffin-tin radius for atom type " << i << " (" << atom_types_[i].label()
937  << ") is too small: " << Rmt[i];
938  TERMINATE(s);
939  }
940  }
941 
942  return Rmt;
943 }
