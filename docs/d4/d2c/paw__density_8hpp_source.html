70  break;
71  }
72 
73  case 3:
74  case 1: {
75  double nm = (std::abs(magn[2]) < 1.0) ? magn[2] : std::copysign(1, magn[2]);
76  density_matrix_(xi, xi, 0, ia) = 0.5 * (1.0 + nm) * occ / double(2 * l + 1);

85 inline void Density::generate_paw_atom_density(paw_density_data_t& pdd)
86 {
87  int ia = pdd.ia;
88 
89  auto& atom_type = pdd.atom_->type();
90 
91  auto l_by_lm = utils::l_by_lm(2 * atom_type.indexr().lmax_lo());
92 
93  /* get gaunt coefficients */
94  Gaunt_coefficients<double> GC(atom_type.indexr().lmax_lo(), 2 * atom_type.indexr().lmax_lo(),
95  atom_type.indexr().lmax_lo(), SHT::gaunt_rlm);
96 
97  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
98  pdd.ae_density_[i].zero();
99  pdd.ps_density_[i].zero();
100  }
101 
102  /* get radial grid to divide density over r^2 */
103  auto& grid = atom_type.radial_grid();
104 
105  auto& paw_ae_wfs = atom_type.ae_paw_wfs_array();
106  auto& paw_ps_wfs = atom_type.ps_paw_wfs_array();
107 
108  /* iterate over local basis functions (or over lm1 and lm2) */
109  for (int xi2 = 0; xi2 < atom_type.indexb().size(); xi2++) {
110  int lm2 = atom_type.indexb(xi2).lm;
111  int irb2 = atom_type.indexb(xi2).idxrf;
112 
113  for (int xi1 = 0; xi1 <= xi2; xi1++) {
114  int lm1 = atom_type.indexb(xi1).lm;
115  int irb1 = atom_type.indexb(xi1).idxrf;
116 
117  /* get num of non-zero GC */
118  int num_non_zero_gk = GC.num_gaunt(lm1, lm2);
119 
120  double diag_coef = (xi1 == xi2) ? 1.0 : 2.0;
121 
122  /* store density matrix in aux form */
123  double dm[4] = {0, 0, 0, 0};
124  switch (ctx_.num_mag_dims()) {
125  case 3: {
126  dm[2] = 2 * std::real(density_matrix_(xi1, xi2, 2, ia));
127  dm[3] = -2 * std::imag(density_matrix_(xi1, xi2, 2, ia));
128  }
129  case 1: {
130  dm[0] = std::real(density_matrix_(xi1, xi2, 0, ia) + density_matrix_(xi1, xi2, 1, ia));
131  dm[1] = std::real(density_matrix_(xi1, xi2, 0, ia) - density_matrix_(xi1, xi2, 1, ia));
132  break;
133  }
134  case 0: {
135  dm[0] = std::real(density_matrix_(xi1, xi2, 0, ia));
136  }
137  }
138 
139  for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
140  auto& ae_dens = pdd.ae_density_[imagn];
141  auto& ps_dens = pdd.ps_density_[imagn];
142 
143  /* add nonzero coefficients */
144  for (int inz = 0; inz < num_non_zero_gk; inz++) {
145  auto& lm3coef = GC.gaunt(lm1, lm2, inz);
146 
147  /* iterate over radial points */
148  for (int irad = 0; irad < grid.num_points(); irad++) {
149 
150  /* we need to divide density over r^2 since wave functions are stored multiplied by r */
151  double inv_r2 = diag_coef / (grid[irad] * grid[irad]);
152 
153  /* calculate unified density/magnetization
154  * dm_ij * GauntCoef * ( phi_i phi_j + Q_ij) */
155  ae_dens(lm3coef.lm3, irad) += dm[imagn] * inv_r2 * lm3coef.coef * paw_ae_wfs(irad, irb1) * paw_ae_wfs(irad, irb2);
156  ps_dens(lm3coef.lm3, irad) += dm[imagn] * inv_r2 * lm3coef.coef *
157  (paw_ps_wfs(irad, irb1) * paw_ps_wfs(irad, irb2) + atom_type.q_radial_function(irb1, irb2, l_by_lm[lm3coef.lm3])(irad));
158  }
159  }
160  }
161  }
162  }
