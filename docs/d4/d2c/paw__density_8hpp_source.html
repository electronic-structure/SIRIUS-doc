46  paw_density(Simulation_context& ctx__)
47  : ctx_(&ctx__)
48  {
49  auto& uc = ctx__.unit_cell();
50  if (!uc.num_paw_atoms()) {
51  return;
52  }
53 
54  using sf = Spheric_function<function_domain_t::spectral, double>;
55 
56  ae_density_ = sddk::mdarray<sf, 2>(ctx__.num_mag_dims() + 1, uc.spl_num_paw_atoms().local_size());
57  ps_density_ = sddk::mdarray<sf, 2>(ctx__.num_mag_dims() + 1, uc.spl_num_paw_atoms().local_size());
58 
59  for (int i = 0; i < uc.spl_num_paw_atoms().local_size(); i++) {
60  int ia_paw = uc.spl_num_paw_atoms(i);
61  int ia = uc.paw_atom_index(ia_paw);
62  auto& atom = uc.atom(ia);
63  auto& atom_type = atom.type();
64 
65  int l_max = 2 * atom_type.indexr().lmax_lo();
66  int lm_max_rho = utils::lmmax(l_max);
67 
68  atoms_.push_back(ia);
69 
70  /* allocate density arrays */
71  for (int j = 0; j < ctx__.num_mag_dims() + 1; j++) {
72  ae_density_(j, i) = sf(lm_max_rho, atom.radial_grid());
73  ps_density_(j, i) = sf(lm_max_rho, atom.radial_grid());
74  }
75  }
76  }
