30 template<typename T>
31 void Non_local_functor<T>::add_k_point_contribution(K_point& kpoint__, sddk::mdarray<double, 2>& collect_res__)
32 {
33  PROFILE("sirius::Non_local_functor::add_k_point");
34 
35  auto& unit_cell = ctx_.unit_cell();
36 
37  if (ctx_.unit_cell().mt_lo_basis_size() == 0) {
38  return;
39  }
40 
41  auto& bp = kpoint__.beta_projectors();
42 
43  double main_two_factor{-2};
44 
45  for (int icnk = 0; icnk < bp_base_.num_chunks(); icnk++) {
46 
47  bp.prepare();
48  /* generate chunk for inner product of beta */
49  bp.generate(icnk);
50 
51  /* store <beta|psi> for spin up and down */
52  matrix<T> beta_phi_chunks[2];
53 
54  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
55  int nbnd = kpoint__.num_occupied_bands(ispn);
56  beta_phi_chunks[ispn] = bp.inner<T>(icnk, kpoint__.spinor_wave_functions(), ispn, 0, nbnd);
57  }
58  bp.dismiss();
59 
60  bp_base_.prepare();
61  for (int x = 0; x < bp_base_.num_comp(); x++) {
62  /* generate chunk for inner product of beta gradient */
63  bp_base_.generate(icnk, x);
64 
65  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
66  int spin_factor = (ispn == 0 ? 1 : -1);
67 
68  int nbnd = kpoint__.num_occupied_bands(ispn);
69 
70  /* inner product of beta gradient and WF */
71  auto bp_base_phi_chunk = bp_base_.template inner<T>(icnk, kpoint__.spinor_wave_functions(), ispn, 0,
72  nbnd);
73 
74  splindex<splindex_t::block> spl_nbnd(nbnd, kpoint__.comm().size(), kpoint__.comm().rank());
75 
76  int nbnd_loc = spl_nbnd.local_size();
77 
78  #pragma omp parallel for
79  for (int ia_chunk = 0; ia_chunk < bp_base_.chunk(icnk).num_atoms_; ia_chunk++) {
80  int ia = bp_base_.chunk(icnk).desc_(static_cast<int>(beta_desc_idx::ia), ia_chunk);
81  int offs = bp_base_.chunk(icnk).desc_(static_cast<int>(beta_desc_idx::offset), ia_chunk);
82  int nbf = bp_base_.chunk(icnk).desc_(static_cast<int>(beta_desc_idx::nbf), ia_chunk);
83  int iat = unit_cell.atom(ia).type_id();
84 
85  if (unit_cell.atom(ia).type().spin_orbit_coupling()) {
86  TERMINATE("stress and forces with SO coupling are not upported");
87  }
88 
89  /* helper lambda to calculate for sum loop over bands for different beta_phi and dij combinations*/
90  auto for_bnd = [&](int ibf, int jbf, double_complex dij, double_complex qij,
91  matrix<T> &beta_phi_chunk) {
92  /* gather everything = - 2 Re[ occ(k,n) weight(k) beta_phi*(i,n) [Dij - E(n)Qij] beta_base_phi(j,n) ]*/
93  for (int ibnd_loc = 0; ibnd_loc < nbnd_loc; ibnd_loc++) {
94  int ibnd = spl_nbnd[ibnd_loc];
95 
96  double_complex scalar_part =
97  main_two_factor * kpoint__.band_occupancy(ibnd, ispn) * kpoint__.weight() *
98  std::conj(beta_phi_chunk(offs + jbf, ibnd)) *
99  bp_base_phi_chunk(offs + ibf, ibnd) *
100  (dij - kpoint__.band_energy(ibnd, ispn) * qij);
101 
102  /* get real part and add to the result array*/
103  collect_res__(x, ia) += scalar_part.real();
104  }
105  };
106 
107  for (int ibf = 0; ibf < nbf; ibf++) {
108  int lm2 = unit_cell.atom(ia).type().indexb(ibf).lm;
109  int idxrf2 = unit_cell.atom(ia).type().indexb(ibf).idxrf;
110  for (int jbf = 0; jbf < nbf; jbf++) {
111  int lm1 = unit_cell.atom(ia).type().indexb(jbf).lm;
112  int idxrf1 = unit_cell.atom(ia).type().indexb(jbf).idxrf;
113 
114  /* Qij exists only in the case of ultrasoft/PAW */
115  double qij{0};
116  if (unit_cell.atom(ia).type().augment()) {
117  qij = ctx_.augmentation_op(iat)->q_mtrx(ibf, jbf);
118  }
119  double_complex dij{0};
120 
121  /* get non-magnetic or collinear spin parts of dij*/
122  switch (ctx_.num_spins()) {
123  case 1: {
124  dij = unit_cell.atom(ia).d_mtrx(ibf, jbf, 0);
125  if (lm1 == lm2) {
126  dij += unit_cell.atom(ia).type().d_mtrx_ion()(idxrf1, idxrf2);
127  }
128  break;
129  }
130 
131  case 2: {
132  /* Dij(00) = dij + dij_Z ; Dij(11) = dij - dij_Z*/
133  dij = (unit_cell.atom(ia).d_mtrx(ibf, jbf, 0) +
134  spin_factor * unit_cell.atom(ia).d_mtrx(ibf, jbf, 1));
135  if (lm1 == lm2) {
136  dij += unit_cell.atom(ia).type().d_mtrx_ion()(idxrf1, idxrf2);
137  }
138  break;
139  }
140 
141  default: {
142  TERMINATE("Error in non_local_functor, D_aug_mtrx. ");
143  break;
144  }
145  }
146 
147  /* add non-magnetic or diagonal spin components (or collinear part) */
148  for_bnd(ibf, jbf, dij, double_complex(qij, 0.0), beta_phi_chunks[ispn]);
149 
150  /* for non-collinear case*/
151  if (ctx_.num_mag_dims() == 3) {
152  /* Dij(10) = dij_X + i dij_Y ; Dij(01) = dij_X - i dij_Y */
153  dij = double_complex(unit_cell.atom(ia).d_mtrx(ibf, jbf, 2),
154  spin_factor * unit_cell.atom(ia).d_mtrx(ibf, jbf, 3));
155  /* add non-diagonal spin components*/
156  for_bnd(ibf, jbf, dij, double_complex(0.0, 0.0), beta_phi_chunks[ispn + spin_factor]);
157  }
158  } // jbf
159  } // ibf
160  } // ia_chunk
161  } // ispn
162  } // x
163  }
164 
165  bp_base_.dismiss();
166 }
