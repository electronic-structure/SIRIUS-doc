33 /// Base class for all kinds of radial integrals.
34 template <int N>
35 class Radial_integrals_base
36 {
37  protected:
38  /// Unit cell.
39  Unit_cell const& unit_cell_;
40 
41  /// Linear grid of q-points on which the interpolation of radial integrals is done.
42  Radial_grid<double> grid_q_;
43 
44  /// Split index of q-points.
45  splindex<splindex_t::block> spl_q_;
46 
47  /// Array with integrals.
48  mdarray<Spline<double>, N> values_;
49 
50  public:
51  /// Constructor.
52  Radial_integrals_base(Unit_cell const& unit_cell__, double qmax__, int np__)
53  : unit_cell_(unit_cell__)
54  {
55  /* Add extra length to the cutoffs in order to interpolate radial integrals for q > cutoff.
56  This is needed for the variable cell relaxation when lattice changes and the G-vectors in
57  Cartiesin coordinates exceed the initial cutoff length. Do not remove this extra delta! */
58 
59  /* add extra length in [a.u.^-1] */
60  qmax__ += std::max(5.0, qmax__ * 0.1);
61 
62  grid_q_ = Radial_grid_lin<double>(static_cast<int>(np__ * qmax__), 0, qmax__);
63  spl_q_ = splindex<splindex_t::block>(grid_q_.num_points(), unit_cell_.comm().size(), unit_cell_.comm().rank());
64  }
65 
66  /// Get starting index iq and delta dq for the q-point on the linear grid.
67  /** The following condition is satisfied: q = grid_q[iq] + dq */
68  inline std::pair<int, double> iqdq(double q__) const
69  {
70  if (q__ > grid_q_.last()) {
71  std::stringstream s;
72  s << "[sirius::Radial_integrals_base::iqdq] q-point is out of range" << std::endl
73  << " q : " << q__ << std::endl
74  << " last point of the q-grid : " << grid_q_.last();
75  TERMINATE(s);
76  }
77  std::pair<int, double> result;
78  /* find index of q-point */
79  result.first = static_cast<int>((grid_q_.num_points() - 1) * q__ / grid_q_.last());
80  /* delta q = q - q_i */
81  result.second = q__ - grid_q_[result.first];
82  return result;
83  }
84 
85  template <typename... Args>
86  inline double value(Args... args, double q__) const
87  {
88  auto idx = iqdq(q__);
89  return values_(args...)(idx.first, idx.second);
90  }
91 
92  inline int nq() const
93  {
94  return grid_q_.num_points();
95  }
96 };

98 /// Radial integrals of the atomic centered orbitals.
99 /** Used in initialize_subspace and in the hubbard correction. */
100 template<bool jl_deriv>
101 class Radial_integrals_atomic_wf : public Radial_integrals_base<2>
102 {
103  private:
104  void generate();
105 
106  public:
107  Radial_integrals_atomic_wf(Unit_cell const& unit_cell__, double qmax__, int np__)
108  : Radial_integrals_base<2>(unit_cell__, qmax__, np__)
109  {
110  int no_max{0};
111  for (int iat = 0; iat < unit_cell__.num_atom_types(); iat++) {
112  no_max = std::max(no_max, unit_cell__.atom_type(iat).num_ps_atomic_wf());
113  }
114 
115  values_ = mdarray<Spline<double>, 2>(no_max, unit_cell_.num_atom_types());
116 
117  generate();
118  }
119 
120  /// retrieve a given orbital from an atom type
121  inline Spline<double> const& values(int iwf__, int iat__) const
122  {
123  return values_(iwf__, iat__);
124  }
125 
126  /// Get all values for a given atom type and q-point.
127  inline mdarray<double, 1> values(int iat__, double q__) const
128  {
129  auto idx = iqdq(q__);
130  auto& atom_type = unit_cell_.atom_type(iat__);
131  mdarray<double, 1> val(atom_type.num_ps_atomic_wf());
132  for (int i = 0; i < atom_type.num_ps_atomic_wf(); i++) {
133  val(i) = values_(i, iat__)(idx.first, idx.second);
134  }
135  return val;
136  }
137 };

158  inline mdarray<double, 2> values(int iat__, double q__) const
159  {
160  auto idx = iqdq(q__);
161 
162  auto& atom_type = unit_cell_.atom_type(iat__);
163  int lmax = atom_type.indexr().lmax();
164  int nbrf = atom_type.mt_radial_basis_size();
165 
166  mdarray<double, 2> val(nbrf * (nbrf + 1) / 2, 2 * lmax + 1);
167  for (int l = 0; l <= 2 * lmax; l++) {
168  for (int i = 0; i < nbrf * (nbrf + 1) / 2; i++) {
169  val(i, l) = values_(i, l, iat__)(idx.first, idx.second);
170  }
171  }
172  return val;
173  }

268 class Radial_integrals_vloc : public Radial_integrals_base<1>
269 {
270  private:

272 
273  public:
274  Radial_integrals_vloc(Unit_cell const& unit_cell__, double qmax__, int np__)
275  : Radial_integrals_base<1>(unit_cell__, qmax__, np__)
276  {

278  generate();
279  }
280 
281  /// Special implementation to recover the true radial integral value.
282  inline double value(int iat__, double q__) const
283  {
284  auto idx = iqdq(q__);
285  if (std::abs(q__) < 1e-12) {
286  if (jl_deriv) {
287  return 0;
288  } else {
289  return values_(iat__)(0);
290  }
291  } else {
292  auto& atom_type = unit_cell_.atom_type(iat__);
293 
294  auto q2 = std::pow(q__, 2);
295  if (jl_deriv) {
296  if (!unit_cell_.parameters().parameters_input().enable_esm_ ||
297  unit_cell_.parameters().parameters_input().esm_bc_ == "pbc") {
298  return values_(iat__)(idx.first, idx.second) / q2 / q__ -
299  atom_type.zn() * std::exp(-q2 / 4) * (4 + q2) / 2 / q2 / q2;
300  } else {
301  return values_(iat__)(idx.first, idx.second) / q2 / q__;
302  }
303  } else {
304  if (!unit_cell_.parameters().parameters_input().enable_esm_ ||
305  unit_cell_.parameters().parameters_input().esm_bc_ == "pbc") {
306  return values_(iat__)(idx.first, idx.second) / q__ - atom_type.zn() * std::exp(-q2 / 4) / q2;
307  } else {
308  return values_(iat__)(idx.first, idx.second) / q__;
309  }
310  }
311  }
312  }
313 };
