33 /// Base class for all kinds of radial integrals.
34 template <int N>
35 class Radial_integrals_base
36 {
37  protected:
38  /// Unit cell.
39  Unit_cell const& unit_cell_;
40 
41  /// Linear grid of q-points on which the interpolation of radial integrals is done.
42  Radial_grid<double> grid_q_;
43 
44  /// Split index of q-points.
45  splindex<block> spl_q_;
46 
47  /// Array with integrals.
48  mdarray<Spline<double>, N> values_;
49 
50  public:
51  /// Constructor.
52  Radial_integrals_base(Unit_cell const& unit_cell__, double qmax__, int np__)
53  : unit_cell_(unit_cell__)
54  {
55  grid_q_ = Radial_grid_lin<double>(static_cast<int>(np__ * qmax__), 0, qmax__);
56  spl_q_ = splindex<block>(grid_q_.num_points(), unit_cell_.comm().size(), unit_cell_.comm().rank());
57  }
58 
59  /// Get starting index iq and delta dq for the q-point on the linear grid.
60  /** The following condition is satisfied: q = grid_q[iq] + dq */
61  inline std::pair<int, double> iqdq(double q__) const
62  {
63  if (q__ > grid_q_.last()) {
64  std::stringstream s;
65  s << "[sirius::Radial_integrals_base::iddq] q-point is out of range" << std::endl
66  << " q : " << q__ << std::endl
67  << " last point of the q-grid : " << grid_q_.last();
68  std::cout << s.str() << "\n";
69  printf("%s\n", s.str().c_str());
70  TERMINATE(s);
71  }
72  std::pair<int, double> result;
73  /* find index of q-point */
74  result.first = static_cast<int>((grid_q_.num_points() - 1) * q__ / grid_q_.last());
75  /* delta q = q - q_i */
76  result.second = q__ - grid_q_[result.first];
77  return std::move(result);
78  }
79 
80  template <typename... Args>
81  inline double value(Args... args, double q__) const
82  {
83  auto idx = iqdq(q__);
84  return values_(args...)(idx.first, idx.second);
85  }
86 
87  inline int nq() const
88  {
89  return grid_q_.num_points();
90  }
91 };

93 /// Radial integrals of the atomic centered orbitals.
94 /** Used in initialize_subspace and in the hubbard correction. */
95 template <bool jl_deriv>
96 class Radial_integrals_atomic_wf : public Radial_integrals_base<2>
97 {
98  private:
99 
100  void generate()
101  {
102  PROFILE("sirius::Radial_integrals|atomic_centered_wfc");
103 
104  /* spherical Bessel functions jl(qx) */
105  mdarray<Spherical_Bessel_functions, 1> jl(nq());
106 
107  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
108 
109  auto& atom_type = unit_cell_.atom_type(iat);
110 
111  int nwf = atom_type.num_ps_atomic_wf();
112  if (!nwf) {
113  continue;
114  }
115 
116  /* create jl(qx) */
117  #pragma omp parallel for
118  for (int iq = 0; iq < nq(); iq++) {
119  jl(iq) = Spherical_Bessel_functions(atom_type.lmax_ps_atomic_wf(), atom_type.radial_grid(), grid_q_[iq]);
120  }
121 
122  /* loop over all pseudo wave-functions */
123  for (int i = 0; i < nwf; i++) {
124  values_(i, iat) = Spline<double>(grid_q_);
125  auto& wf = atom_type.ps_atomic_wf(i);
126  const int l = std::abs(wf.first);
127 
128  #pragma omp parallel for
129  for (int iq = 0; iq < nq(); iq++) {
130  if (jl_deriv) {
131  auto s = jl(iq).deriv_q(l);
132  values_(i, iat)(iq) = sirius::inner(s, wf.second, 1);
133  } else {
134  values_(i, iat)(iq) = sirius::inner(jl(iq)[l], wf.second, 1);
135  }
136  }
137 
138  values_(i, iat).interpolate();
139  }
140  }
141  }
142 
143  public:
144  Radial_integrals_atomic_wf(Unit_cell const& unit_cell__, double qmax__, int np__)
145  : Radial_integrals_base<2>(unit_cell__, qmax__, np__)
146  {
147  int no_max{0};
148  for (int iat = 0; iat < unit_cell__.num_atom_types(); iat++) {
149  no_max = std::max(no_max, unit_cell__.atom_type(iat).num_ps_atomic_wf());
150  }
151 
152  values_ = mdarray<Spline<double>, 2>(no_max, unit_cell_.num_atom_types());
153 
154  generate();
155  }
156 
157  /// retrieve a given orbital from an atom type
158  inline Spline<double> const& values(int iwf__, int iat__) const
159  {
160  return values_(iwf__, iat__);
161  }
162 
163  /// Get all values for a given atom type and q-point.
164  inline mdarray<double, 1> values(int iat__, double q__) const
165  {
166  auto idx = iqdq(q__);
167  auto& atom_type = unit_cell_.atom_type(iat__);
168  mdarray<double, 1> val(atom_type.num_ps_atomic_wf());
169  for (int i = 0; i < atom_type.num_ps_atomic_wf(); i++) {
170  val(i) = values_(i, iat__)(idx.first, idx.second);
171  }
172  return std::move(val);
173  }
174 };

176 /// Radial integrals of the augmentation operator.
177 template <bool jl_deriv>
178 class Radial_integrals_aug : public Radial_integrals_base<3>
179 {
180  private:
181  void generate()
182  {
183  PROFILE("sirius::Radial_integrals|aug");
184 
185  /* interpolate <j_{l_n}(q*x) | Q_{xi,xi'}^{l}(x) > with splines */
186  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
187  auto& atom_type = unit_cell_.atom_type(iat);
188 
189  if (!atom_type.augment()) {
190  continue;
191  }
192 
193  /* number of radial beta-functions */
194  int nbrf = atom_type.mt_radial_basis_size();
195  /* maximum l of beta-projectors */
196  int lmax_beta = atom_type.indexr().lmax();
197 
198  for (int l = 0; l <= 2 * lmax_beta; l++) {
199  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
200  values_(idx, l, iat) = Spline<double>(grid_q_);
201  }
202  }
203 
204  #pragma omp parallel for
205  for (int iq_loc = 0; iq_loc < spl_q_.local_size(); iq_loc++) {
206  int iq = spl_q_[iq_loc];
207 
208  Spherical_Bessel_functions jl(2 * lmax_beta, atom_type.radial_grid(), grid_q_[iq]);
209 
210  for (int l3 = 0; l3 <= 2 * lmax_beta; l3++) {
211  for (int idxrf2 = 0; idxrf2 < nbrf; idxrf2++) {
212  int l2 = atom_type.indexr(idxrf2).l;
213  for (int idxrf1 = 0; idxrf1 <= idxrf2; idxrf1++) {
214  int l1 = atom_type.indexr(idxrf1).l;
215 
216  int idx = idxrf2 * (idxrf2 + 1) / 2 + idxrf1;
217 
218  if (l3 >= std::abs(l1 - l2) && l3 <= (l1 + l2) && (l1 + l2 + l3) % 2 == 0) {
219  if (jl_deriv) {
220  auto s = jl.deriv_q(l3);
221  values_(idx, l3, iat)(iq) =
222  sirius::inner(s, atom_type.q_radial_function(idxrf1, idxrf2, l3), 0);
223  } else {
224  values_(idx, l3, iat)(iq) =
225  sirius::inner(jl[l3], atom_type.q_radial_function(idxrf1, idxrf2, l3), 0);
226  }
227  }
228  }
229  }
230  }
231  }
232  for (int l = 0; l <= 2 * lmax_beta; l++) {
233  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
234  unit_cell_.comm().allgather(&values_(idx, l, iat)(0), spl_q_.global_offset(), spl_q_.local_size());
235  }
236  }
237 
238  #pragma omp parallel for
239  for (int l = 0; l <= 2 * lmax_beta; l++) {
240  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
241  values_(idx, l, iat).interpolate();
242  }
243  }
244  }
245  }
246 
247  public:
248  Radial_integrals_aug(Unit_cell const& unit_cell__, double qmax__, int np__)
249  : Radial_integrals_base<3>(unit_cell__, qmax__, np__)
250  {
251  int nmax = unit_cell_.max_mt_radial_basis_size();
252  int lmax = unit_cell_.lmax();
253 
254  values_ = mdarray<Spline<double>, 3>(nmax * (nmax + 1) / 2, 2 * lmax + 1, unit_cell_.num_atom_types());
255 
256  generate();
257  }
258 
259  inline mdarray<double, 2> values(int iat__, double q__) const
260  {
261  auto idx = iqdq(q__);
262 
263  auto& atom_type = unit_cell_.atom_type(iat__);
264  int lmax = atom_type.indexr().lmax();
265  int nbrf = atom_type.mt_radial_basis_size();
266 
267  mdarray<double, 2> val(nbrf * (nbrf + 1) / 2, 2 * lmax + 1);
268  for (int l = 0; l <= 2 * lmax; l++) {
269  for (int i = 0; i < nbrf * (nbrf + 1) / 2; i++) {
270  val(i, l) = values_(i, l, iat__)(idx.first, idx.second);
271  }
272  }
273  return std::move(val);
274  }
275 };

371 template <bool jl_deriv>
372 class Radial_integrals_beta : public Radial_integrals_base<2>
373 {
374  private:
375  void generate()
376  {
377  PROFILE("sirius::Radial_integrals|beta");
378 
379  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
380  auto& atom_type = unit_cell_.atom_type(iat);
381  int nrb = atom_type.num_beta_radial_functions();
382 
383  if (!nrb) {
384  continue;
385  }
386 
387  for (int idxrf = 0; idxrf < nrb; idxrf++) {
388  values_(idxrf, iat) = Spline<double>(grid_q_);
389  }
390 
391  #pragma omp parallel for
392  for (int iq_loc = 0; iq_loc < spl_q_.local_size(); iq_loc++) {
393  int iq = spl_q_[iq_loc];
394  Spherical_Bessel_functions jl(unit_cell_.lmax(), atom_type.radial_grid(), grid_q_[iq]);
395  for (int idxrf = 0; idxrf < nrb; idxrf++) {
396  int l = atom_type.indexr(idxrf).l;
397  /* compute \int j_l(q * r) beta_l(r) r^2 dr or \int d (j_l(q*r) / dq) beta_l(r) r^2 */
398  /* remeber that beta(r) are defined as miltiplied by r */
399  if (jl_deriv) {
400  auto s = jl.deriv_q(l);
401  values_(idxrf, iat)(iq) = sirius::inner(s, atom_type.beta_radial_function(idxrf), 1);
402  } else {
403  values_(idxrf, iat)(iq) = sirius::inner(jl[l], atom_type.beta_radial_function(idxrf), 1);
404  }
405  }
406  }
407 
408  for (int idxrf = 0; idxrf < nrb; idxrf++) {
409  unit_cell_.comm().allgather(&values_(idxrf, iat)(0), spl_q_.global_offset(), spl_q_.local_size());
410  values_(idxrf, iat).interpolate();
411  }
412  }
413  }
414 
415  public:
416  Radial_integrals_beta(Unit_cell const& unit_cell__, double qmax__, int np__)
417  : Radial_integrals_base<2>(unit_cell__, qmax__, np__)
418  {
419  /* create space for <j_l(qr)|beta> or <d j_l(qr) / dq|beta> radial integrals */
420  values_ = mdarray<Spline<double>, 2>(unit_cell_.max_mt_radial_basis_size(), unit_cell_.num_atom_types());
421  generate();
422  }
423 
424  /// Get all values for a given atom type and q-point.
425  inline mdarray<double, 1> values(int iat__, double q__) const
426  {
427  auto idx = iqdq(q__);
428  auto& atom_type = unit_cell_.atom_type(iat__);
429  mdarray<double, 1> val(atom_type.mt_radial_basis_size());
430  for (int i = 0; i < atom_type.mt_radial_basis_size(); i++) {
431  val(i) = values_(i, iat__)(idx.first, idx.second);
432  }
433  return std::move(val);
434  }

494 template <bool jl_deriv>
495 class Radial_integrals_vloc : public Radial_integrals_base<1>
496 {
497  private:
498  void generate()
499  {
500  PROFILE("sirius::Radial_integrals|vloc");
501 
502  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
503  auto& atom_type = unit_cell_.atom_type(iat);
504 
505  if (atom_type.local_potential().empty()) {
506  continue;
507  }
508 
509  values_(iat) = Spline<double>(grid_q_);
510 
511  auto& vloc = atom_type.local_potential();
512 
513  int np = atom_type.num_mt_points();
514  //if (std::abs(vloc.back() * atom_type.radial_grid().last() + atom_type.zn()) > 1e-10) {
515  // std::stringstream s;
516  // s << "Wrong asymptotics of local potential for atom type " << iat << std::endl
517  // << "hack with 10 a.u. cutoff is activated";
518  // WARNING(s);
519  if (true) {
520  int np1 = atom_type.radial_grid().index_of(10);
521  if (np1 != -1) {
522  np = np1;
523  }
524  }
525 
526  auto rg = atom_type.radial_grid().segment(np);
527 
528  #pragma omp parallel for
529  for (int iq_loc = 0; iq_loc < spl_q_.local_size(); iq_loc++) {
530  int iq = spl_q_[iq_loc];
531  Spline<double> s(rg);
532  double g = grid_q_[iq];
533 
534  if (jl_deriv) { /* integral with derivative of j0(q*r) over q */
535  for (int ir = 0; ir < rg.num_points(); ir++) {
536  double x = rg[ir];
537  s(ir) = (x * vloc[ir] + atom_type.zn() * std::erf(x)) *
538  (std::sin(g * x) - g * x * std::cos(g * x));
539  }
540  } else { /* integral with j0(q*r) */
541  if (iq == 0) { /* q=0 case */
542  if (unit_cell_.parameters().parameters_input().enable_esm_ &&
543  unit_cell_.parameters().parameters_input().esm_bc_ != "pbc") {
544  for (int ir = 0; ir < rg.num_points(); ir++) {
545  double x = rg[ir];
546  s(ir) = (x * vloc[ir] + atom_type.zn() * std::erf(x)) * x;
547  }
548  } else {
549  for (int ir = 0; ir < rg.num_points(); ir++) {
550  double x = rg[ir];
551  s(ir) = (x * vloc[ir] + atom_type.zn()) * x;
552  }
553  }
554  } else {
555  for (int ir = 0; ir < rg.num_points(); ir++) {
556  double x = rg[ir];
557  s(ir) = (x * vloc[ir] + atom_type.zn() * std::erf(x)) * std::sin(g * x);
558  }
559  }
560  }
561  values_(iat)(iq) = s.interpolate().integrate(0);
562  }
563  unit_cell_.comm().allgather(&values_(iat)(0), spl_q_.global_offset(), spl_q_.local_size());
564  values_(iat).interpolate();
565  }
566  }

568  public:
569  Radial_integrals_vloc(Unit_cell const& unit_cell__, double qmax__, int np__)
570  : Radial_integrals_base<1>(unit_cell__, qmax__, np__)
571  {
572  values_ = mdarray<Spline<double>, 1>(unit_cell_.num_atom_types());

576  /// Special implementation to recover the true radial integral value.
577  inline double value(int iat__, double q__) const
578  {
579  auto idx = iqdq(q__);
580  if (std::abs(q__) < 1e-12) {
581  if (jl_deriv) {
582  return 0;
583  } else {
584  return values_(iat__)(0);
585  }
586  } else {
