34 /// Base class for all kinds of radial integrals.
35 template <int N>
36 class Radial_integrals_base
37 {
38  protected:
39  /// Unit cell.
40  Unit_cell const& unit_cell_;
41 
42  /// Linear grid of q-points on which the interpolation of radial integrals is done.
43  Radial_grid<double> grid_q_;
44 
45  /// Split index of q-points.
46  splindex<splindex_t::block> spl_q_;
47 
48  /// Array with integrals.
49  sddk::mdarray<Spline<double>, N> values_;
50 
51  /// Maximum length of the reciprocal wave-vector.
52  double qmax_{0};
53 
54  public:
55  /// Constructor.
56  Radial_integrals_base(Unit_cell const& unit_cell__, double const qmax__, int const np__)
57  : unit_cell_(unit_cell__)
58  {
59  /* Add extra length to the cutoffs in order to interpolate radial integrals for q > cutoff.
60  This is needed for the variable cell relaxation when lattice changes and the G-vectors in
61  Cartiesin coordinates exceed the initial cutoff length. Do not remove this extra delta! */
62 
63  /* add extra length in [a.u.^-1] */
64  qmax_ = qmax__ + std::max(10.0, qmax__ * 0.1);
65 
66  grid_q_ = Radial_grid_lin<double>(static_cast<int>(np__ * qmax_), 0, qmax_);
67  spl_q_ = splindex<splindex_t::block>(grid_q_.num_points(), unit_cell_.comm().size(), unit_cell_.comm().rank());
68  }
69 
70  /// Get starting index iq and delta dq for the q-point on the linear grid.
71  /** The following condition is satisfied: q = grid_q[iq] + dq */
72  inline std::pair<int, double> iqdq(double q__) const
73  {
74  if (q__ > grid_q_.last()) {
75  std::stringstream s;
76  s << "q-point is out of range" << std::endl
77  << " q : " << q__ << std::endl
78  << " last point of the q-grid : " << grid_q_.last() << std::endl;
79  auto uc = unit_cell_.serialize();
80  s << "unit cell: " << uc;
81  RTE_THROW(s);
82  }
83  std::pair<int, double> result;
84  /* find index of q-point */
85  result.first = static_cast<int>((grid_q_.num_points() - 1) * q__ / grid_q_.last());
86  /* delta q = q - q_i */
87  result.second = q__ - grid_q_[result.first];
88  return result;
89  }
90 
91  /// Return value of the radial integral with specific indices.
92  template <typename... Args>
93  inline double value(Args... args, double q__) const
94  {
95  auto idx = iqdq(q__);
96  return values_(args...)(idx.first, idx.second);
97  }
98 
99  inline int nq() const
100  {
101  return grid_q_.num_points();
102  }
103 
104  inline double qmax() const
105  {
106  return qmax_;
107  }
108 };

110 /// Radial integrals of the atomic centered orbitals.
111 /** Used in initialize_subspace and in the hubbard correction. */
112 template<bool jl_deriv>
113 class Radial_integrals_atomic_wf : public Radial_integrals_base<2>
114 {
115  private:
116  /// Return radial basis index for a given atom type.
117  std::function<sirius::experimental::radial_functions_index const&(int)> indexr_;
118  /// Generate radial integrals.
119  void generate(std::function<Spline<double> const&(int, int)> fl__);
120 
121  public:
122  /// Constructor.
123  Radial_integrals_atomic_wf(Unit_cell const& unit_cell__, double qmax__, int np__,
124  std::function<sirius::experimental::radial_functions_index const&(int)> indexr__,
125  std::function<Spline<double> const&(int, int)> fl__)
126  : Radial_integrals_base<2>(unit_cell__, qmax__, np__)
127  , indexr_(indexr__)
128  {
129  int nrf_max{0};
130  for (int iat = 0; iat < unit_cell__.num_atom_types(); iat++) {
131  nrf_max = std::max(nrf_max, static_cast<int>(indexr_(iat).size()));
132  }
133 
134  values_ = sddk::mdarray<Spline<double>, 2>(nrf_max, unit_cell_.num_atom_types());
135 
136  generate(fl__);
137  }
138 
139  /// Retrieve a value for a given orbital of an atom type.
140  inline Spline<double> const& values(int iwf__, int iat__) const
141  {
142  return values_(iwf__, iat__);
143  }
144 
145  /// Get all values for a given atom type and q-point.
146  inline sddk::mdarray<double, 1> values(int iat__, double q__) const
147  {
148  auto idx = iqdq(q__);
149  auto& atom_type = unit_cell_.atom_type(iat__);
150  int nrf = indexr_(atom_type.id()).size();
151 
152  sddk::mdarray<double, 1> val(nrf);
153  for (int i = 0; i < nrf; i++) {
154  val(i) = values_(i, iat__)(idx.first, idx.second);
155  }
156  return val;
157  }

186  {
187  auto& atom_type = unit_cell_.atom_type(iat__);
188  int lmax = atom_type.indexr().lmax();
189  int nbrf = atom_type.mt_radial_basis_size();
190 
191  sddk::mdarray<double, 2> val(nbrf * (nbrf + 1) / 2, 2 * lmax + 1);
192 
193  if (ri_callback_ == nullptr) {
194  auto idx = iqdq(q__);
195 
196  for (int l = 0; l <= 2 * lmax; l++) {
197  for (int i = 0; i < nbrf * (nbrf + 1) / 2; i++) {
198  val(i, l) = values_(i, l, iat__)(idx.first, idx.second);
199  }
200  }
201  } else {
202  ri_callback_(iat__ + 1, q__, &val[0], nbrf * (nbrf + 1) / 2, 2 * lmax + 1);
203  }
204  return val;
205  }

237  /// Compute all values of the raial integrals.
238  inline sddk::mdarray<double, 2> values(std::vector<double>& q__, sddk::Communicator const& comm__) const
239  {
240  int nq = static_cast<int>(q__.size());
241  splindex<splindex_t::block> splq(nq, comm__.size(), comm__.rank());
242  sddk::mdarray<double, 2> result(nq, unit_cell_.num_atom_types());
243  result.zero();
244  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
245  if (!unit_cell_.atom_type(iat).ps_total_charge_density().empty()) {
246  #pragma omp parallel for
247  for (int iqloc = 0; iqloc < splq.local_size(); iqloc++) {
248  int iq = splq[iqloc];
249  if (ri_callback_) {
250  ri_callback_(iat + 1, 1, &q__[iq], &result(iq, iat));
251  } else {
252  result(iq, iat) = this->value<int>(iat, q__[iq]);
253  }
254  }
255  comm__.allgather(&result(0, iat), splq.local_size(), splq.global_offset());
256  }
257  }
258  return result;
259  }

348 template <bool jl_deriv>
349 class Radial_integrals_vloc : public Radial_integrals_base<1>
350 {
351  private:
352  std::function<void(int, int, double*, double*)> ri_callback_{nullptr};
353  void generate();
354 
355  public:
356  Radial_integrals_vloc(Unit_cell const& unit_cell__, double qmax__, int np__,
357  std::function<void(int, int, double*, double*)> ri_callback__)
358  : Radial_integrals_base<1>(unit_cell__, qmax__, np__)
359  , ri_callback_(ri_callback__)
360  {
361  if (ri_callback_ == nullptr) {
362  values_ = mdarray<Spline<double>, 1>(unit_cell_.num_atom_types());
363  generate();
364  }
365  }
366 
367  /// Special implementation to recover the true radial integral value.
368  inline double value(int iat__, double q__) const
369  {
370  if (unit_cell_.atom_type(iat__).local_potential().empty()) {
371  return 0;
372  }
373  auto idx = iqdq(q__);
374  if (std::abs(q__) < 1e-12) {
375  if (jl_deriv) {
376  return 0;
377  } else {
378  return values_(iat__)(0);
379  }
380  } else {
381  auto& atom_type = unit_cell_.atom_type(iat__);
382 
383  auto q2 = std::pow(q__, 2);
384  if (jl_deriv) {
385  return values_(iat__)(idx.first, idx.second) / q2 / q__ -
386  atom_type.zn() * std::exp(-q2 / 4) * (4 + q2) / 2 / q2 / q2;
387  } else {
388  return values_(iat__)(idx.first, idx.second) / q__ - atom_type.zn() * std::exp(-q2 / 4) / q2;
389  }
390  }
391  }
392 
393  /// Compute all values of the raial integrals.
394  inline sddk::mdarray<double, 2> values(std::vector<double>& q__, sddk::Communicator const& comm__) const
395  {
396  int nq = static_cast<int>(q__.size());
397  splindex<splindex_t::block> splq(nq, comm__.size(), comm__.rank());
398  sddk::mdarray<double, 2> result(nq, unit_cell_.num_atom_types());
399  result.zero();
400  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
401  if (!unit_cell_.atom_type(iat).local_potential().empty()) {
402  #pragma omp parallel for
403  for (int iqloc = 0; iqloc < splq.local_size(); iqloc++) {
404  int iq = splq[iqloc];
405  if (ri_callback_) {
406  ri_callback_(iat + 1, 1, &q__[iq], &result(iq, iat));
407  } else {
408  result(iq, iat) = this->value(iat, q__[iq]);
409  }
410  }
411  comm__.allgather(&result(0, iat), splq.local_size(), splq.global_offset());
412  }
413  }
414  return result;
415  }
