28std::unique_ptr<Eigensolver> Eigensolver_factory(std::string name__, memory_pool* mpd__)
29{
30 std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
31
32 Eigensolver* ptr = nullptr;
33 switch (get_ev_solver_t(name__)) {
34 case ev_solver_t::lapack: {
35 ptr = new Eigensolver_lapack();
36 break;
37 }
38 case ev_solver_t::scalapack: {
39 ptr = new Eigensolver_scalapack();
40 break;
41 }
42 case ev_solver_t::elpa: {
43 if (name__ == "elpa1") {
44 ptr = new Eigensolver_elpa(1);
45 } else {
46 ptr = new Eigensolver_elpa(2);
47 }
48 break;
49 }
50 case ev_solver_t::magma: {
51 ptr = new Eigensolver_magma();
52 break;
53 }
54 case ev_solver_t::magma_gpu: {
55 ptr = new Eigensolver_magma_gpu();
56 break;
57 }
58 case ev_solver_t::cusolver: {
59 ptr = new Eigensolver_cuda(mpd__);
60 break;
61 }
62 default: {
63 TERMINATE("not implemented");
64 }
65 }
66 return std::unique_ptr<Eigensolver>(ptr);
67}
