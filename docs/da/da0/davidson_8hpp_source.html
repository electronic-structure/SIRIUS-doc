67template <typename T>
68inline int
69remove_linearly_dependent(::spla::Context& spla_ctx__, spin_range spins__, Wave_functions<real_type<T>>& phi__,
70 int N__, int n__, sddk::dmatrix<T>& o__)
71
72{
73 PROFILE("sirius::remove_linearly_dependent");
74
75 /* compute <phi | phi> */
76 inner(spla_ctx__, spins__, phi__, N__, n__, phi__, N__, n__, o__, 0, 0);
77
78 auto la = (o__.comm().size() == 1) ? linalg_t::lapack : linalg_t::scalapack;
79 linalg(la).geqrf(n__, n__, o__, 0, 0);
80 auto diag = o__.get_diag(n__);
81
82 auto eps = std::numeric_limits<real_type<T>>::epsilon();
83
84 int n{0};
85 for (int i = 0; i < n__; i++) {
86 if (std::abs(diag[i]) >= eps * 10) {
87 /* shift linearly independent basis functions to the beginning of phi */
88 if (n != i) {
89 for (int ispn: spins__) {
90 phi__.copy_from(phi__, 1, ispn, N__ + i, ispn, N__ + n);
91 }
92 }
93 n++;
94 }
95 }
96 return n;
97}

100/// Solve the eigen-problem using Davidson iterative method.
101/**
102\tparam T Type of the wave-functions in real space (one of float, double, complex<float>, complex<double>).
103\tparam F Type of the subspace matrices.
104\param [in] Hk Hamiltonian for a given k-point.
105\param [in] num_bands Number of eigen-states (bands) to compute.
106\param [in] num_mag_dims Number of magnetic dimensions (0, 1 or 3).
107\param [in,out] psi Wave-functions. On input they are used for the starting guess of the subspace basis.
108 On output they are the solutions of Hk|psi> = e S|psi> eigen-problem.
109\param [in] occupancy Lambda-function for the band occupancy numbers.
110\param [in] tolerance Lambda-function for the band energy tolerance.
111\param [in] res_tol Residual tolerance.
112\param [in] num_stpes Number of iterative steps.
113\param [in] locking Lock and do not update of the converged wave-functions.
114\param [in] subspace_size Size of the diagonalziation subspace.
115\param [in] estimate_eval Estimate eigen-values to get the converrged rersiduals.
116\param [in] extra_ortho Orthogonalize new subspace basis one extra time.
117\param [out] out Output stream.
118\param [in] verbosity Verbosity level.
119\return List of eigen-values.
120*/
121template <typename T, typename F>
122inline davidson_result_t
123davidson(Hamiltonian_k<real_type<T>>& Hk__, int num_bands__, int num_mag_dims__, Wave_functions<real_type<T>>& psi__,
124 std::function<double(int, int)> occupancy__, std::function<double(int, int)> tolerance__, double res_tol__,
125 int num_steps__, bool locking__, int subspace_size__, bool estimate_eval__, bool extra_ortho__,
126 std::ostream& out__, int verbosity__ = 0)
127{
128 PROFILE("sirius::davidson");
129
130 auto& ctx = Hk__.H0().ctx();
131 ctx.print_memory_usage(__FILE__, __LINE__);
132
133
134 auto& kp = Hk__.kp();
135 auto& gkvecp = kp.gkvec_partition();
136
137 auto& itso = ctx.cfg().iterative_solver();
138
139 /* true if this is a non-collinear case */
140 const bool nc_mag = (num_mag_dims__ == 3);
141
142 /* number of spin components, treated simultaneously
143 * 1 - in case of non-magnetic or collinear calculation
144 * 2 - in case of non-collinear calculation
145 */
146 const int num_sc = nc_mag ? 2 : 1;
147
148 /* number of spinor components stored under the same band index */
149 const int num_spinors = (num_mag_dims__ == 1) ? 2 : 1;
150
151 /* number of spins */
152 const int num_spins = (num_mag_dims__ == 0) ? 1 : 2;
153
154 /* maximum subspace size */
155 const int num_phi = subspace_size__ * num_bands__;
156
157 if (num_phi > kp.num_gkvec()) {
158 std::stringstream s;
159 s << "subspace size is too large!";
160 RTE_THROW(s);
161 }
162
163 /* alias for memory pool */
164 auto& mp = ctx.mem_pool(ctx.host_memory_t());
165
166 /* allocate wave-functions */
167
168 /* auxiliary wave-functions */
169 Wave_functions<real_type<T>> phi(mp, gkvecp, num_phi, ctx.aux_preferred_memory_t(), num_sc);
170
171 /* Hamiltonian, applied to auxiliary wave-functions */
172 Wave_functions<real_type<T>> hphi(mp, gkvecp, num_phi, ctx.preferred_memory_t(), num_sc);
173
174 /* S operator, applied to auxiliary wave-functions */
175 Wave_functions<real_type<T>> sphi(mp, gkvecp, num_phi, ctx.preferred_memory_t(), num_sc);
176
177 /* Hamiltonain, applied to new Psi wave-functions */
178 Wave_functions<real_type<T>> hpsi(mp, gkvecp, num_bands__, ctx.preferred_memory_t(), num_sc);
179
180 /* S operator, applied to new Psi wave-functions */
181 Wave_functions<real_type<T>> spsi(mp, gkvecp, num_bands__, ctx.preferred_memory_t(), num_sc);
182
183 /* residuals */
184 Wave_functions<real_type<T>> res(mp, gkvecp, num_bands__, ctx.preferred_memory_t(), num_sc);
185
186 const int bs = ctx.cyclic_block_size();
187
188 dmatrix<F> hmlt(num_phi, num_phi, ctx.blacs_grid(), bs, bs, mp);
189 dmatrix<F> ovlp(num_phi, num_phi, ctx.blacs_grid(), bs, bs, mp);
190 dmatrix<F> evec(num_phi, num_phi, ctx.blacs_grid(), bs, bs, mp);
191 dmatrix<F> hmlt_old(num_phi, num_phi, ctx.blacs_grid(), bs, bs, mp);
192 dmatrix<F> ovlp_old(num_phi, num_phi, ctx.blacs_grid(), bs, bs, mp);
193
194 if (is_device_memory(ctx.aux_preferred_memory_t())) {
195 auto& mpd = ctx.mem_pool(memory_t::device);
196
197 for (int i = 0; i < num_sc; i++) {
198 phi.pw_coeffs(i).allocate(mpd);
199 }
200 }
201
202 if (is_device_memory(ctx.preferred_memory_t())) {
203 auto& mpd = ctx.mem_pool(memory_t::device);
204
205 for (int ispn = 0; ispn < psi__.num_sc(); ispn++) {
206 psi__.pw_coeffs(ispn).allocate(mpd);
207 psi__.pw_coeffs(ispn).copy_to(memory_t::device, 0, num_bands__);
208 }
209
210 for (int i = 0; i < num_sc; i++) {
211 res.pw_coeffs(i).allocate(mpd);
212
213 hphi.pw_coeffs(i).allocate(mpd);
214 sphi.pw_coeffs(i).allocate(mpd);
215
216 hpsi.pw_coeffs(i).allocate(mpd);
217 spsi.pw_coeffs(i).allocate(mpd);
218 }
219
220 if (ctx.blacs_grid().comm().size() == 1) {
221 evec.allocate(mpd);
222 ovlp.allocate(mpd);
223 hmlt.allocate(mpd);
224 }
225 }
226
227 kp.copy_hubbard_orbitals_on_device();
228
229 ctx.print_memory_usage(__FILE__, __LINE__);
230
231 /* get diagonal elements for preconditioning */
232 auto h_o_diag = Hk__.template get_h_o_diag_pw<T, 3>();
233
234 if (ctx.print_checksum()) {
235 auto cs1 = h_o_diag.first.checksum();
236 auto cs2 = h_o_diag.second.checksum();
237 kp.comm().allreduce(&cs1, 1);
238 kp.comm().allreduce(&cs2, 1);
239 if (kp.comm().rank() == 0) {
240 utils::print_checksum("h_diag", cs1);
241 utils::print_checksum("o_diag", cs2);
242 }
243 }
244
245 auto& std_solver = ctx.std_evp_solver();
246
247 if (ctx.print_checksum()) {
248 for (int ispn = 0; ispn < num_spins; ispn++) {
249 auto cs = psi__.checksum_pw(get_device_t(psi__.preferred_memory_t()), ispn, 0, num_bands__);
250 std::stringstream s;
251 s << "input spinor_wave_functions_" << ispn;
252 if (kp.comm().rank() == 0) {
253 utils::print_checksum(s.str(), cs);
254 }
255 }
256 }
257
258 davidson_result_t result{0, sddk::mdarray<double, 2>(num_bands__, num_spinors)};
259
260 PROFILE_START("sirius::davidson|iter");
261 for (int ispin_step = 0; ispin_step < num_spinors; ispin_step++) {
262 ctx.print_memory_usage(__FILE__, __LINE__);
263
264 /* converged vectors */
265 int num_locked = 0;
266
267 sddk::mdarray<real_type<F>, 1> eval(num_bands__);
268 sddk::mdarray<real_type<F>, 1> eval_old(num_bands__);
269
270 /* check if band energy is converged */
271 auto is_converged = [&](int j__, int ispn__) -> bool {
272 return std::abs(eval[j__] - eval_old[j__]) <= tolerance__(j__ + num_locked, ispn__);
273 };
274
275 if (itso.init_eval_old()) {
276 eval_old = [&](int64_t j) { return kp.band_energy(j, ispin_step); };
277 } else {
278 eval_old = []() { return 1e10; };
279 }
280
281 /* trial basis functions */
282 for (int ispn = 0; ispn < num_sc; ispn++) {
283 phi.copy_from(psi__, num_bands__, nc_mag ? ispn : ispin_step, 0, ispn, 0);
284 }
285
286 if (ctx.print_checksum()) {
287 for (int ispn = 0; ispn < num_sc; ispn++) {
288 auto cs = phi.checksum_pw(get_device_t(phi.preferred_memory_t()), ispn, 0, num_bands__);
289 std::stringstream s;
290 s << "input phi" << ispn;
291 if (kp.comm().rank() == 0) {
292 utils::print_checksum(s.str(), cs);
293 }
294 }
295 }
296
297 /* first phase: setup and diagonalize reduced Hamiltonian and get eigen-values;
298 * this is done before the main itertive loop */
299
300 /* apply Hamiltonian and S operators to the basis functions */
301 Hk__.template apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), 0, num_bands__, phi, &hphi, &sphi);
302
303 /* DEBUG */
304 if (ctx.cfg().control().verification() >= 1) {
305 /* setup eigen-value problem */
306 Band(ctx).set_subspace_mtrx<T, F>(0, num_bands__, 0, phi, hphi, hmlt, &hmlt_old);
307 Band(ctx).set_subspace_mtrx<T, F>(0, num_bands__, 0, phi, sphi, ovlp, &ovlp_old);
308
309 auto max_diff = check_hermitian(hmlt, num_bands__);
310 if (max_diff > (std::is_same<real_type<T>, double>::value ? 1e-12 : 1e-6)) {
311 std::stringstream s;
312 s << "H matrix is not Hermitian, max_err = " << max_diff << std::endl
313 << " happened before entering the iterative loop" << std::endl;
314 WARNING(s);
315 if (num_bands__ <= 20) {
316 hmlt.serialize("davidson:H_first", num_bands__);
317 }
318 }
319 max_diff = check_hermitian(ovlp, num_bands__);
320 if (max_diff > (std::is_same<real_type<T>, double>::value ? 1e-12 : 1e-6)) {
321 std::stringstream s;
322 s << "O matrix is not Hermitian, max_err = " << max_diff << std::endl
323 << " happened before entering the iterative loop" << std::endl;
324 WARNING(s);
325 if (num_bands__ <= 20) {
326 hmlt.serialize("davidson:O_first", num_bands__);
327 }
328 }
329 }
330 /* END DEBUG */
331
332 orthogonalize<T>(ctx.spla_context(), ctx.preferred_memory_t(), ctx.blas_linalg_t(), spin_range(nc_mag ? 2 : 0),
333 phi, hphi, sphi, 0, num_bands__, ovlp, res);
334
335 /* setup eigen-value problem */
336 Band(ctx).set_subspace_mtrx<T>(0, num_bands__, 0, phi, hphi, hmlt, &hmlt_old);
337
338 if (ctx.cfg().control().verification() >= 1) {
339 auto max_diff = check_hermitian(hmlt, num_bands__);
340 if (max_diff > (std::is_same<real_type<T>, double>::value ? 1e-12 : 1e-6)) {
341 std::stringstream s;
342 s << "H matrix is not Hermitian, max_err = " << max_diff << std::endl
343 << " happened before entering the iterative loop" << std::endl;
344 WARNING(s);
345 if (num_bands__ <= 20) {
346 hmlt.serialize("davidson:H", num_bands__);
347 }
348 }
349 }
350
351 /* current subspace size */
352 int N = num_bands__;
353
354 // Seems like a smaller block size is not always improving time to solution much,
355 // so keep it num_bands.
356 int block_size = num_bands__;
357
358 /* solve generalized eigen-value problem with the size N and get lowest num_bands eigen-vectors */
359 if (std_solver.solve(N, num_bands__, hmlt, &eval[0], evec)) {
360 std::stringstream s;
361 s << "error in diagonalziation";
362 RTE_THROW(s);
363 }
364
365 ctx.evp_work_count(1);
366
367 for (int i = 0; i < num_bands__; i++) {
368 kp.message(4, __function_name__, "eval[%i]=%20.16f\n", i, eval[i]);
369 }
370
371 /* number of newly added basis functions */
372 int num_unconverged{0};
373
374 /* tolerance for the norm of L2-norms of the residuals, used for
375 * relative convergence criterion. We can only compute this after
376 * we have the first residual norms available */
377 real_type<T> relative_frobenius_tolerance{0};
378 real_type<T> current_frobenius_norm{0};
379
380 /* second phase: start iterative diagonalization */
381 for (int iter_step = 0; iter_step < num_steps__; iter_step++) {
382 int num_lockable = 0;
383
384 bool last_iteration = iter_step == (num_steps__ - 1);
385
386 int num_ritz = num_bands__ - num_locked;
387
388 /* don't compute residuals on last iteration */
389 if (!last_iteration) {
390 /* get new preconditionined residuals, and also hpsi and opsi as a by-product */
391 auto result = residuals<T>(ctx, ctx.preferred_memory_t(), ctx.blas_linalg_t(),
392 spin_range(nc_mag ? 2 : ispin_step), N, num_ritz, num_locked, eval, evec,
393 hphi, sphi, hpsi, spsi, res, h_o_diag.first, h_o_diag.second,
394 estimate_eval__, res_tol__, is_converged);
395
396 num_unconverged = result.unconverged_residuals;
397 num_lockable = result.num_consecutive_smallest_converged;
398 current_frobenius_norm = result.frobenius_norm;
399
400 /* set the relative tolerance convergence criterion */
401 if (iter_step == 0) {
402 relative_frobenius_tolerance = current_frobenius_norm * itso.relative_tolerance();
403 }
404 }
405
406 /* verify convergence criteria */
407 int num_converged = num_ritz - num_unconverged;
408 bool converged_by_relative_tol = iter_step > 0 && current_frobenius_norm < relative_frobenius_tolerance;
409 bool converged_by_absolute_tol = num_locked + num_converged + itso.min_num_res() >= num_bands__;
410
411 bool converged = converged_by_relative_tol || converged_by_absolute_tol;
412
413 /* Todo: num_unconverged might be very small at some point slowing down convergence
414 can we add more? */
415 int expand_with = std::min(num_unconverged, block_size);
416 bool should_restart = ((N + expand_with) > num_phi) ||
417 (num_lockable > 5 && num_unconverged < itso.early_restart() * num_lockable);
418
419 kp.message(3, __function_name__,
420 "Restart = %s. Locked = %d. Converged = %d. Wanted = %d. Lockable = %d. "
421 "Num ritz = %d. Expansion size = %d\n",
422 should_restart ? "yes" : "no", num_locked, num_converged, num_bands__, num_lockable, num_ritz,
423 expand_with);
424
425 /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
426 if (should_restart || converged || last_iteration) {
427 PROFILE("sirius::davidson|update_phi");
428 /* recompute wave-functions */
429 /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
430
431 /* No need to recompute the wave functions when converged in the first iteration */
432 if (iter_step != 0 || num_unconverged != 0 || ctx.cfg().settings().always_update_wf()) {
433 /* in case of non-collinear magnetism transform two components */
434 transform<T>(ctx.spla_context(), nc_mag ? 2 : ispin_step, {&phi}, num_locked, N - num_locked, evec,
435 0, 0, {&psi__}, num_locked, num_ritz);
436
437 /* update eigen-values */
438 for (int j = num_locked; j < num_bands__; j++) {
439 result.eval(j, ispin_step) = eval[j - num_locked];
440 }
441
442 } else {
443 kp.message(3, __function_name__, "%s", "wave-functions are not recomputed\n");
444 for (int j = 0; j < num_bands__; j++) {
445 result.eval(j, ispin_step) = kp.band_energy(j, ispin_step);
446 }
447 }
448
449 if (last_iteration && !converged) {
450 kp.message(3, __function_name__,
451 "Warning: maximum number of iterations reached, but %i "
452 "residual(s) did not converge for k-point %f %f %f\n",
453 num_unconverged, kp.vk()[0], kp.vk()[1], kp.vk()[2]);
454 }
455
456 /* exit the loop if the eigen-vectors are converged or this is a last iteration */
457 if (converged || last_iteration) {
458 kp.message(3, __function_name__, "end of iterative diagonalization; n=%i, k=%i\n",
459 num_unconverged, iter_step);
460 break;
461 } else { /* otherwise, set Psi as a new trial basis */
462 kp.message(3, __function_name__, "%s", "subspace size limit reached\n");
463
464 // TODO: consider keeping more than num_bands when nearly all Ritz vectors have converged.
465 int keep = num_bands__;
466
467 /* need to compute all hpsi and opsi states (not only unconverged) */
468 if (estimate_eval__) {
469 transform<T, F>(ctx.spla_context(), nc_mag ? 2 : ispin_step, 1.0,
470 std::vector<Wave_functions<real_type<T>>*>({&hphi, &sphi}), num_locked,
471 N - num_locked, evec, 0, 0, 0.0, {&hpsi, &spsi}, 0, num_ritz);
472 }
473
474 /* update basis functions, hphi and ophi */
475 for (int ispn = 0; ispn < num_sc; ispn++) {
476 phi.copy_from(psi__, keep - num_locked, nc_mag ? ispn : ispin_step, num_locked,
477 nc_mag ? ispn : 0, num_locked);
478 hphi.copy_from(hpsi, keep - num_locked, ispn, 0, ispn, num_locked);
479 sphi.copy_from(spsi, keep - num_locked, ispn, 0, ispn, num_locked);
480 }
481
482 // Remove locked Ritz values so indexing starts at unconverged eigenpairs
483 if (locking__ && num_lockable > 0) {
484 for (int i = num_lockable; i < num_ritz; ++i) {
485 eval[i - num_lockable] = eval[i];
486 }
487 }
488
489 /* remove the locked block from the projected matrix too */
490 hmlt_old.zero();
491 for (int i = 0; i < keep - num_locked; i++) {
492 hmlt_old.set(i, i, eval[i]);
493 }
494
495 /* number of basis functions that we already have */
496 N = keep;
497
498 /* only when we do orthogonalization we can lock vecs */
499 if (locking__) {
500 num_locked += num_lockable;
501 }
502 }
503 }
504
505 /* expand variational subspace with new basis vectors obtatined from residuals */
506 for (int ispn = 0; ispn < num_sc; ispn++) {
507 phi.copy_from(res, expand_with, ispn, 0, ispn, N);
508 }
509
510 project_out_subspace<T, F>(ctx.spla_context(), spin_range(nc_mag ? 2 : 0), phi, sphi, N, expand_with, ovlp);
511
512 //std::cout << "expand_with before = " << expand_with << std::endl;
513 //expand_with = remove_linearly_dependent(ctx.spla_context(), spin_range(nc_mag ? 2 : 0), phi, N,
514 // expand_with, ovlp);
515 //std::cout << "expand_with after = " << expand_with << std::endl;
516
517 /* apply Hamiltonian and S operators to the new basis functions */
518 Hk__.template apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), N, expand_with, phi, &hphi, &sphi);
519
520 kp.message(3, __function_name__, "Orthogonalize %d to %d\n", expand_with, N);
521
522 orthogonalize<T>(ctx.spla_context(), ctx.preferred_memory_t(), ctx.blas_linalg_t(),
523 spin_range(nc_mag ? 2 : 0), phi, hphi, sphi, N, expand_with, ovlp, res, false);
524
525 if (extra_ortho__) {
526 orthogonalize<T>(ctx.spla_context(), ctx.preferred_memory_t(), ctx.blas_linalg_t(),
527 spin_range(nc_mag ? 2 : 0), phi, hphi, sphi, N, expand_with, ovlp, res, false);
528 }
529
530 /* setup eigen-value problem
531 * N is the number of previous basis functions
532 * expand_with is the number of new basis functions */
533 Band(ctx).set_subspace_mtrx<T, F>(N, expand_with, num_locked, phi, hphi, hmlt, &hmlt_old);
534
535 if (ctx.cfg().control().verification() >= 1) {
536 real_type<T> max_diff = check_hermitian(hmlt, N + expand_with - num_locked);
537 if (max_diff > (std::is_same<real_type<T>, double>::value ? 1e-12 : 1e-6)) {
538 std::stringstream s;

567
568 } /* loop over iterative steps k */

576 psi__.pw_coeffs(ispn).deallocate(memory_t::device);
577 }
578 }
579
580 kp.release_hubbard_orbitals_on_device();
581 ctx.print_memory_usage(__FILE__, __LINE__);
