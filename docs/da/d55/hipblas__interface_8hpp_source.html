39 inline void error_message(hipblasStatus_t status)
40 {
41  switch (status) {
42  case HIPBLAS_STATUS_NOT_INITIALIZED: {
43  printf("the library was not initialized\n");
44  break;
45  }
46  case HIPBLAS_STATUS_INVALID_VALUE: {
47  printf("the parameters m,n,k<0\n");
48  break;
49  }
50  case HIPBLAS_STATUS_ARCH_MISMATCH: {
51  printf("the device does not support double-precision\n");
52  break;
53  }
54  case HIPBLAS_STATUS_EXECUTION_FAILED: {
55  printf("the function failed to launch on the GPU\n");
56  break;
57  }
58  default: {
59  printf("hipblas status unknown");
60  }
61  }
62 }

184 inline void create_stream_handles()
185 {
186  CALL_HIPBLAS(hipblasCreate, (&null_stream_handle()));
187 
188  stream_handles() = std::vector<hipblasHandle_t>(acc::num_streams());
189  for (int i = 0; i < acc::num_streams(); i++) {
190  CALL_HIPBLAS(hipblasCreate, (&stream_handles()[i]));
191 
192  CALL_HIPBLAS(hipblasSetStream, (stream_handles()[i], acc::stream(stream_id(i))));
193  }
194 }
195 
196 inline void destroy_stream_handles()
197 {
198  CALL_HIPBLAS(hipblasDestroy, (null_stream_handle()));
199  for (int i = 0; i < acc::num_streams(); i++) {
200  CALL_HIPBLAS(hipblasDestroy, (stream_handles()[i]));
201  }
202 }
