117  /// Check wave-functions for orthonormalization.
118  template <typename T>
119  void check_wave_functions(K_point& kp__, Hamiltonian& H__) const
120  {
121  if (kp__.comm().rank() == 0) {
122  printf("checking wave-functions\n");
123  }
124 
125  if (!ctx_.full_potential()) {
126 
127  dmatrix<T> ovlp(ctx_.num_bands(), ctx_.num_bands(), ctx_.blacs_grid(), ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
128 
129  const bool nc_mag = (ctx_.num_mag_dims() == 3);
130  const int num_sc = nc_mag ? 2 : 1;
131 
132  auto& psi = kp__.spinor_wave_functions();
133  Wave_functions spsi(kp__.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
134 
135  if (is_device_memory(ctx_.preferred_memory_t())) {
136  auto& mpd = ctx_.mem_pool(memory_t::device);
137  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
138  psi.pw_coeffs(ispn).allocate(mpd);
139  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
140  }
141  for (int i = 0; i < num_sc; i++) {
142  spsi.pw_coeffs(i).allocate(mpd);
143  }
144  ovlp.allocate(memory_t::device);
145  }
146  kp__.beta_projectors().prepare();
147  /* compute residuals */
148  for (int ispin_step = 0; ispin_step < ctx_.num_spin_dims(); ispin_step++) {
149  /* apply Hamiltonian and S operators to the wave-functions */
150  H__.apply_h_s<T>(&kp__, nc_mag ? 2 : ispin_step, 0, ctx_.num_bands(), psi, nullptr, &spsi);
151  inner(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), nc_mag ? 2 : ispin_step, psi, 0, ctx_.num_bands(),
152  spsi, 0, ctx_.num_bands(), ovlp, 0, 0);
153 
154  double diff = check_identity(ovlp, ctx_.num_bands());
155 
156  if (kp__.comm().rank() == 0) {
157  if (diff > 1e-12) {
158  printf("overlap matrix is not identity, maximum error : %20.12f\n", diff);
159  } else {
160  printf("OK! Wave functions are orthonormal.\n");
161  }
162  }
163  }
164  if (is_device_memory(ctx_.preferred_memory_t())) {
165  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
166  psi.pw_coeffs(ispn).deallocate(memory_t::device);
167  }
168  }
169  kp__.beta_projectors().dismiss();
170  }
171  }
