59double
60unit_step_function_form_factors(double R__, double g__)
61{
62 if (g__ < 1e-12) {
63 return std::pow(R__, 3) / 3.0;
64 } else {
65 return (std::sin(g__ * R__) - g__ * R__ * std::cos(g__ * R__)) / std::pow(g__, 3);
66 }
67}

131void
132Simulation_context::init_fft_grid()
133{
134 if (!(cfg().control().fft_mode() == "serial" || cfg().control().fft_mode() == "parallel")) {
135 RTE_THROW("wrong FFT mode");
136 }
137
138 auto rlv = unit_cell().reciprocal_lattice_vectors();
139
140 /* create FFT driver for dense mesh (density and potential) */
141 auto fft_grid = cfg().settings().fft_grid_size();
142 if (fft_grid[0] * fft_grid[1] * fft_grid[2] == 0) {
143 fft_grid_ = get_min_fft_grid(pw_cutoff(), rlv);
144 cfg().settings().fft_grid_size(fft_grid_);
145 } else {
146 /* else create a grid with user-specified dimensions */
147 fft_grid_ = sddk::FFT3D_grid(fft_grid);
148 }
149
150 /* create FFT grid for coarse mesh */
151 fft_coarse_grid_ = get_min_fft_grid(2 * gk_cutoff(), rlv);
152}

154mdarray<double, 3>
155Simulation_context::generate_sbessel_mt(int lmax__) const
156{
157 PROFILE("sirius::Simulation_context::generate_sbessel_mt");
158
159 mdarray<double, 3> sbessel_mt(lmax__ + 1, gvec().count(), unit_cell().num_atom_types());
160 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
161 #pragma omp parallel for schedule(static)
162 for (int igloc = 0; igloc < gvec().count(); igloc++) {
163 auto gv = gvec().gvec_cart<index_domain_t::local>(igloc);
164 gsl_sf_bessel_jl_array(lmax__, gv.length() * unit_cell().atom_type(iat).mt_radius(),
165 &sbessel_mt(0, igloc, iat));
166 }
167 }
168 return sbessel_mt;
169}

185mdarray<double_complex, 2>
186Simulation_context::sum_fg_fl_yg(int lmax__, double_complex const* fpw__, mdarray<double, 3>& fl__,
187 matrix<double_complex>& gvec_ylm__)
188{
189 PROFILE("sirius::Simulation_context::sum_fg_fl_yg");
190
191 int ngv_loc = gvec().count();
192
193 int na_max{0};
194 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
195 na_max = std::max(na_max, unit_cell().atom_type(iat).num_atoms());
196 }
197
198 int lmmax = utils::lmmax(lmax__);
199 /* resuling matrix */
200 mdarray<double_complex, 2> flm(lmmax, unit_cell().num_atoms());
201
202 matrix<double_complex> phase_factors;
203 matrix<double_complex> zm;
204 matrix<double_complex> tmp;
205
206 switch (processing_unit()) {
207 case device_t::CPU: {
208 auto& mp = mem_pool(memory_t::host);
209 phase_factors = matrix<double_complex>(ngv_loc, na_max, mp);
210 zm = matrix<double_complex>(lmmax, ngv_loc, mp);
211 tmp = matrix<double_complex>(lmmax, na_max, mp);
212 break;
213 }
214 case device_t::GPU: {
215 auto& mp = mem_pool(memory_t::host);
216 auto& mpd = mem_pool(memory_t::device);
217 phase_factors = matrix<double_complex>(nullptr, ngv_loc, na_max);
218 phase_factors.allocate(mpd);
219 zm = matrix<double_complex>(lmmax, ngv_loc, mp);
220 zm.allocate(mpd);
221 tmp = matrix<double_complex>(lmmax, na_max, mp);
222 tmp.allocate(mpd);
223 break;
224 }
225 }
226
227 std::vector<double_complex> zil(lmax__ + 1);
228 for (int l = 0; l <= lmax__; l++) {
229 zil[l] = std::pow(double_complex(0, 1), l);
230 }
231
232 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
233 int na = unit_cell().atom_type(iat).num_atoms();
234 generate_phase_factors(iat, phase_factors);
235 PROFILE_START("sirius::Simulation_context::sum_fg_fl_yg|zm");
236 #pragma omp parallel for schedule(static)
237 for (int igloc = 0; igloc < ngv_loc; igloc++) {
238 for (int l = 0, lm = 0; l <= lmax__; l++) {
239 double_complex z = fourpi * fl__(l, igloc, iat) * zil[l] * fpw__[igloc];
240 for (int m = -l; m <= l; m++, lm++) {
241 zm(lm, igloc) = z * std::conj(gvec_ylm__(lm, igloc));
242 }
243 }
244 }
245 PROFILE_STOP("sirius::Simulation_context::sum_fg_fl_yg|zm");
246 PROFILE_START("sirius::Simulation_context::sum_fg_fl_yg|mul");
247 switch (processing_unit()) {
248 case device_t::CPU: {
249 linalg(linalg_t::blas)
250 .gemm('N', 'N', lmmax, na, ngv_loc, &linalg_const<double_complex>::one(), zm.at(memory_t::host),
251 zm.ld(), phase_factors.at(memory_t::host), phase_factors.ld(),
252 &linalg_const<double_complex>::zero(), tmp.at(memory_t::host), tmp.ld());
253 break;
254 }
255 case device_t::GPU: {
256 zm.copy_to(memory_t::device);
257 linalg(linalg_t::gpublas)
258 .gemm('N', 'N', lmmax, na, ngv_loc, &linalg_const<double_complex>::one(), zm.at(memory_t::device),
259 zm.ld(), phase_factors.at(memory_t::device), phase_factors.ld(),
260 &linalg_const<double_complex>::zero(), tmp.at(memory_t::device), tmp.ld());
261 tmp.copy_to(memory_t::host);
262 break;
263 }
264 }
265 PROFILE_STOP("sirius::Simulation_context::sum_fg_fl_yg|mul");
266
267 for (int i = 0; i < na; i++) {
268 int ia = unit_cell().atom_type(iat).atom_id(i);
269 std::copy(&tmp(0, i), &tmp(0, i) + lmmax, &flm(0, ia));
270 }
271 }
272
273 comm().allreduce(&flm(0, 0), (int)flm.size());
274
275 return flm;
276}

278double
279Simulation_context::ewald_lambda() const
280{
281 /* alpha = 1 / (2*sigma^2), selecting alpha here for better convergence */
282 double lambda{1};
283 double gmax = pw_cutoff();
284 double upper_bound{0};
285 double charge = unit_cell().num_electrons();
286
287 /* iterate to find lambda */
288 do {
289 lambda += 0.1;
290 upper_bound =
291 charge * charge * std::sqrt(2.0 * lambda / twopi) * std::erfc(gmax * std::sqrt(1.0 / (4.0 * lambda)));
292 } while (upper_bound < 1e-8);
293
294 if (lambda < 1.5 && comm().rank() == 0) {
295 std::stringstream s;
296 s << "ewald_lambda(): pw_cutoff is too small";
297 WARNING(s);
298 }
299 return lambda;
300}

302splindex<splindex_t::block>
303Simulation_context::split_gvec_local() const
304{
305 /* local number of G-vectors for this MPI rank */
306 int ngv_loc = gvec().count();
307 /* estimate number of G-vectors in a block */
308 int ld{-1};
309 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
310 int nat = unit_cell().atom_type(iat).num_atoms();
311 int nbf = unit_cell().atom_type(iat).mt_basis_size();
312
313 ld = std::max(ld, std::max(nbf * (nbf + 1) / 2, nat));
314 }
315 /* limit the size of relevant array to ~1Gb */
316 int ngv_b = (1 << 30) / sizeof(double_complex) / ld;
317 ngv_b = std::max(1, std::min(ngv_loc, ngv_b));
318 /* number of blocks of G-vectors */
319 int nb = ngv_loc / ngv_b;
320 /* split local number of G-vectors between blocks */
321 return splindex<splindex_t::block>(ngv_loc, nb, 0);
322}

324void
325Simulation_context::initialize()
326{
327 PROFILE("sirius::Simulation_context::initialize");
328
329 /* can't initialize twice */
330 if (initialized_) {
331 RTE_THROW("Simulation parameters are already initialized.");
332 }
333
334 auto verb_lvl = utils::get_env<int>("SIRIUS_VERBOSITY");
335 if (verb_lvl) {
336 this->verbosity(*verb_lvl);
337 }
338
339 electronic_structure_method(cfg().parameters().electronic_structure_method());
340 core_relativity(cfg().parameters().core_relativity());
341 valence_relativity(cfg().parameters().valence_relativity());
342
343 /* can't run fp-lapw with Gamma point trick */
344 if (full_potential()) {
345 gamma_point(false);
346 }
347
348 /* Gamma-point calculation and non-collinear magnetism are not compatible */
349 if (num_mag_dims() == 3) {
350 gamma_point(false);
351 }
352
353 /* set processing unit type */
354 processing_unit(cfg().control().processing_unit());
355
356 /* check if we can use a GPU device */
357 if (processing_unit() == device_t::GPU) {
358#if !defined(SIRIUS_GPU)
359 RTE_THROW("not compiled with GPU support!");
360#endif
361 }
362
363 /* initialize MPI communicators */
364 init_comm();
365
366 auto print_mpi_layout = utils::get_env<int>("SIRIUS_PRINT_MPI_LAYOUT");
367
368 if (verbosity() >= 3 || (print_mpi_layout && *print_mpi_layout)) {
369 pstdout pout(comm());
370 if (comm().rank() == 0) {
371 pout.printf("MPI rank placement\n");
372 pout.printf("------------------\n");
373 }
374 auto name = sddk::Communicator::processor_name();
375 pout.printf("rank: %3i, comm_band_rank: %3i, comm_k_rank: %3i, hostname: %s, mpi processor name: %s\n",
376 comm().rank(), comm_band().rank(), comm_k().rank(), utils::hostname().c_str(), name.c_str());
377 }
378
379 switch (processing_unit()) {
380 case device_t::CPU: {
381 host_memory_t_ = memory_t::host;
382 break;
383 }
384 case device_t::GPU: {
385 host_memory_t_ = memory_t::host_pinned;
386 break;
387 }
388 }
389
390 switch (processing_unit()) {
391 case device_t::CPU: {
392 preferred_memory_t_ = memory_t::host;
393 break;
394 }
395 case device_t::GPU: {
396 if (cfg().control().memory_usage() == "high") {
397 preferred_memory_t_ = memory_t::device;
398 }
399 if (cfg().control().memory_usage() == "low" || cfg().control().memory_usage() == "medium") {
400 preferred_memory_t_ = memory_t::host_pinned;
401 }
402 break;
403 }
404 }
405
406 switch (processing_unit()) {
407 case device_t::CPU: {
408 aux_preferred_memory_t_ = memory_t::host;
409 break;
410 }
411 case device_t::GPU: {
412 if (cfg().control().memory_usage() == "high" || cfg().control().memory_usage() == "medium") {
413 aux_preferred_memory_t_ = memory_t::device;
414 }
415 if (cfg().control().memory_usage() == "low") {
416 aux_preferred_memory_t_ = memory_t::host_pinned;
417 }
418 break;
419 }
420 }
421
422 switch (processing_unit()) {
423 case device_t::CPU: {
424 blas_linalg_t_ = linalg_t::blas;
425 break;
426 }
427 case device_t::GPU: {
428 if (cfg().control().memory_usage() == "high") {
429 blas_linalg_t_ = linalg_t::gpublas;
430 }
431 if (cfg().control().memory_usage() == "low" || cfg().control().memory_usage() == "medium") {
432#if defined(SIRIUS_ROCM)
433 blas_linalg_t_ = linalg_t::gpublas;
434#else
435 blas_linalg_t_ = linalg_t::cublasxt;
436#endif
437 }
438 break;
439 }
440 }
441
442 if (processing_unit() == device_t::GPU) {
443 spla_ctx_.reset(new spla::Context{SPLA_PU_GPU});
444 spla_ctx_->set_tile_size_gpu(1688); // limit GPU memory usage to around 500MB
445 }
446 // share context for blas operations to reduce memory consumption
447 splablas::get_handle_ptr() = spla_ctx_;
448
449 /* can't use reduced G-vectors in LAPW code */
450 if (full_potential()) {
451 cfg().control().reduce_gvec(false);
452 }
453
454 if (!cfg().iterative_solver().type().size() || (cfg().iterative_solver().type() == "auto")) {
455 if (full_potential()) {
456 cfg().iterative_solver().type("exact");
457 } else {
458 cfg().iterative_solver().type("davidson");
459 }
460 }
461 /* set default values for the G-vector cutoff */
462 if (pw_cutoff() <= 0) {
463 pw_cutoff(full_potential() ? 12 : 20);
464 }
465
466 /* initialize variables related to the unit cell */
467 unit_cell().initialize();
468 /* save the volume of the initial unit cell */
469 omega0_ = unit_cell().omega();
470 /* save initial lattice vectors */
471 lattice_vectors0_ = unit_cell().lattice_vectors();
472
473 /* check the lattice symmetries */
474 if (use_symmetry()) {
475 auto lv = matrix3d<double>(unit_cell().lattice_vectors());
476
477 auto lat_sym = find_lat_sym(lv, 1e-6);
478
479 #pragma omp parallel for
480 for (int i = 0; i < unit_cell().symmetry().size(); i++) {
481 auto& spgR = unit_cell().symmetry()[i].spg_op.R;
482 bool found{false};
483 for (size_t i = 0; i < lat_sym.size(); i++) {
484 auto latR = lat_sym[i];
485 found = true;
486 for (int x : {0, 1, 2}) {
487 for (int y : {0, 1, 2}) {
488 found = found && (spgR(x, y) == latR(x, y));
489 }
490 }
491 if (found) {
492 break;
493 }
494 }
495 if (!found) {
496 RTE_THROW("spglib lattice symmetry was not found in the list of SIRIUS generated symmetries");
497 }
498 }
499 }
500
501 /* force global spin-orbit correction flag if one of the species has spin-orbit */
502 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
503 if (unit_cell().atom_type(iat).spin_orbit_coupling()) {
504 this->so_correction(true);
505 }
506 }
507
508 /* set default values for the G+k-vector cutoff */
509 if (full_potential()) {
510 /* find the cutoff for G+k vectors (derived from rgkmax (aw_cutoff here) and minimum MT radius) */
511 if (aw_cutoff() > 0) {
512 gk_cutoff(aw_cutoff() / unit_cell().min_mt_radius());
513 }
514 if (gk_cutoff() <= 0) {
515 gk_cutoff(3);
516 }
517 } else {
518 /* in pseudopotential case */
519 if (gk_cutoff() <= 0) {
520 gk_cutoff(7);
521 }
522 }
523
524 /* check the G+k cutoff */
525 if (gk_cutoff() * 2 > pw_cutoff()) {
526 std::stringstream s;
527 s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
528 << " pw cutoff : " << pw_cutoff() << std::endl
529 << " doubled G+k cutoff : " << gk_cutoff() * 2;
530 RTE_THROW(s);
531 }
532
533 if (!full_potential()) {
534 set_lmax_rho(unit_cell().lmax() * 2);
535 set_lmax_pot(unit_cell().lmax() * 2);
536 set_lmax_apw(-1);
537 }
538
539 /* initialize FFT grid dimensions */
540 init_fft_grid();
541
542 int nbnd = static_cast<int>(unit_cell().num_valence_electrons() / 2.0) +
543 std::max(10, static_cast<int>(0.1 * unit_cell().num_valence_electrons()));
544 if (full_potential()) {
545 /* take 10% of empty non-magnetic states */
546 if (num_fv_states() < 0) {
547 num_fv_states(nbnd);
548 }
549 if (num_fv_states() < static_cast<int>(unit_cell().num_valence_electrons() / 2.0)) {
550 std::stringstream s;
551 s << "not enough first-variational states : " << num_fv_states();
552 RTE_THROW(s);
553 }
554 } else {
555 if (num_mag_dims() == 3) {
556 nbnd *= 2;
557 }
558 /* if number of bands was not set by the host code, set it here */
559 if (num_bands() < 0) {
560 num_bands(nbnd);
561 }
562 }
563
564 std::string evsn[] = {std_evp_solver_name(), gen_evp_solver_name()};
565#if defined(SIRIUS_CUDA)
566 bool is_cuda{true};
567#else
568 bool is_cuda{false};
569#endif
570#if defined(SIRIUS_MAGMA)
571 bool is_magma{true};
572#else
573 bool is_magma{false};
574#endif
575#if defined(SIRIUS_SCALAPACK)
576 bool is_scalapack{true};
577#else
578 bool is_scalapack{false};
579#endif
580#if defined(SIRIUS_ELPA)
581 bool is_elpa{true};
582#else
583 bool is_elpa{false};
584#endif
585
586 if (processing_unit() == device_t::CPU || acc::num_devices() == 0) {
587 is_cuda = false;
588 is_magma = false;
589 }
590
591 int npr = cfg().control().mpi_grid_dims()[0];
592 int npc = cfg().control().mpi_grid_dims()[1];
593
594 /* deduce the default eigen-value solver */
595 for (int i : {0, 1}) {
596 if (evsn[i] == "auto") {
597 /* conditions for sequential diagonalization */
598 if (comm_band().size() == 1 || npc == 1 || npr == 1 || !is_scalapack) {
599 if (full_potential()) {
600 if (is_magma) {
601 evsn[i] = "magma";
602 } else if (is_cuda) {
603 evsn[i] = "cusolver";
604 } else {
605 evsn[i] = "lapack";
606 }
607 } else {
608 if (is_cuda) {
609 evsn[i] = "cusolver";
610 } else if (is_magma && num_bands() > 200) {
611 evsn[i] = "magma";
612 } else {
613 evsn[i] = "lapack";
614 }
615 }
616 } else {
617 if (is_scalapack) {
618 evsn[i] = "scalapack";
619 }
620 if (is_elpa) {
621 evsn[i] = "elpa1";
622 }
623 }
624 }
625 }
626
627 auto pstr = utils::get_env<std::string>("SIRIUS_EV_SOLVER");
628 if (pstr) {
629 evsn[0] = *pstr;
630 evsn[1] = *pstr;
631 }
632
633 std_evp_solver_name(evsn[0]);
634 gen_evp_solver_name(evsn[1]);
635
636 std_evp_solver_ = Eigensolver_factory(std_evp_solver_name(), &mem_pool(memory_t::device));
637 gen_evp_solver_ = Eigensolver_factory(gen_evp_solver_name(), &mem_pool(memory_t::device));
638
639 auto& std_solver = std_evp_solver();
640 auto& gen_solver = gen_evp_solver();
641
642 if (std_solver.is_parallel() != gen_solver.is_parallel()) {
643 RTE_THROW("both solvers must be sequential or parallel");
644 }
645
646 /* setup BLACS grid */
647 if (std_solver.is_parallel()) {
648 blacs_grid_ = std::unique_ptr<BLACS_grid>(new BLACS_grid(comm_band(), npr, npc));
649 } else {
650 blacs_grid_ = std::unique_ptr<BLACS_grid>(new BLACS_grid(Communicator::self(), 1, 1));
651 }
652

1028/** The update of the lattice vectors or atomic positions has an impact on many quantities which have to be
1029 recomputed in the correct order. First, the unit cell is updated and the new reciprocal lattice vectors
1030 are obtained. Then the G-vectors are computed (if this is the first call to update()) or recomputed with a
1031 new reciprocal lattice, but without rebuilding a new list. On the first call the spfft objects are created
1032 after G-vectors. */

1034Simulation_context::update()

1036 PROFILE("sirius::Simulation_context::update");

1038 /* update unit cell (reciprocal lattice, etc.) */

1044 auto spfft_pu = this->processing_unit() == device_t::CPU ? SPFFT_PU_HOST : SPFFT_PU_GPU;
1045
1046 /* create a list of G-vectors for corase FFT grid; this is done only once,
1047 the next time only reciprocal lattice of the G-vectors is updated */
1048 if (!gvec_coarse_) {
1049 /* create list of coarse G-vectors */
1050 gvec_coarse_ = std::unique_ptr<Gvec>(new Gvec(rlv, 2 * gk_cutoff(), comm(), cfg().control().reduce_gvec()));
1051 /* create FFT friendly partiton */
1052 gvec_coarse_partition_ = std::unique_ptr<Gvec_partition>(
1053 new Gvec_partition(*gvec_coarse_, comm_fft_coarse(), comm_ortho_fft_coarse()));
1054
1055 auto spl_z = split_fft_z(fft_coarse_grid_[2], comm_fft_coarse());
1056
1057 /* create spfft buffer for coarse transform */
1058 spfft_grid_coarse_ = std::unique_ptr<spfft::Grid>(
1059 new spfft::Grid(fft_coarse_grid_[0], fft_coarse_grid_[1], fft_coarse_grid_[2],
1060 gvec_coarse_partition_->zcol_count_fft(), spl_z.local_size(), spfft_pu, -1,
1061 comm_fft_coarse().mpi_comm(), SPFFT_EXCH_DEFAULT));
1062#ifdef USE_FP32
1063 spfft_grid_coarse_float_ = std::unique_ptr<spfft::GridFloat>(
1064 new spfft::GridFloat(fft_coarse_grid_[0], fft_coarse_grid_[1], fft_coarse_grid_[2],
1065 gvec_coarse_partition_->zcol_count_fft(), spl_z.local_size(), spfft_pu, -1,
1066 comm_fft_coarse().mpi_comm(), SPFFT_EXCH_DEFAULT));
1067#endif

1070
1071 auto gv = gvec_coarse_partition_->get_gvec();

1078#ifdef USE_FP32

1080 spfft_pu, fft_type_coarse, fft_coarse_grid_[0], fft_coarse_grid_[1], fft_coarse_grid_[2],
1081 spl_z.local_size(), gvec_coarse_partition_->gvec_count_fft(), SPFFT_INDEX_TRIPLETS,

1084 } else {

1089 if (!gvec_) {

1091 gvec_partition_ = std::unique_ptr<Gvec_partition>(new Gvec_partition(*gvec_, comm_fft(), comm_ortho_fft()));

1097 new spfft::Grid(fft_grid_[0], fft_grid_[1], fft_grid_[2],
1098 gvec_partition_->zcol_count_fft(), spl_z.local_size(), spfft_pu, -1,
1099 comm_fft().mpi_comm(), SPFFT_EXCH_DEFAULT));
1100#if defined(USE_FP32)
1101 spfft_grid_float_ = std::unique_ptr<spfft::GridFloat>(
1102 new spfft::GridFloat(fft_grid_[0], fft_grid_[1], fft_grid_[2],
1103 gvec_partition_->zcol_count_fft(), spl_z.local_size(), spfft_pu, -1,
1104 comm_fft().mpi_comm(), SPFFT_EXCH_DEFAULT));
1105#endif
1106 const auto fft_type = gvec().reduced() ? SPFFT_TRANS_R2C : SPFFT_TRANS_C2C;
1107
1108 auto gv = gvec_partition_->get_gvec();
1109
1110 spfft_transform_.reset(new spfft::Transform(spfft_grid_->create_transform(
1111 spfft_pu, fft_type, fft_grid_[0], fft_grid_[1], fft_grid_[2],
1112 spl_z.local_size(), gvec_partition_->gvec_count_fft(), SPFFT_INDEX_TRIPLETS, gv.at(memory_t::host))));
1113#if defined(USE_FP32)

1115 spfft_pu, fft_type, fft_grid_[0], fft_grid_[1], fft_grid_[2],
1116 spl_z.local_size(), gvec_partition_->gvec_count_fft(), SPFFT_INDEX_TRIPLETS, gv.at(memory_t::host))));
1117#endif

1120 do not change during the execution */
1121 switch (this->processing_unit()) {
1122 case device_t::CPU: {
1123 break;
1124 }
1125 case device_t::GPU: {
1126 gvec_coord_ = sddk::mdarray<int, 2>(gvec().count(), 3, memory_t::host, "gvec_coord_");
1127 #pragma omp parallel for schedule(static)
1128 for (int igloc = 0; igloc < gvec().count(); igloc++) {
1129 int ig = gvec().offset() + igloc;
1130 auto G = gvec().gvec(ig);
1131 for (int x : {0, 1, 2}) {
1132 gvec_coord_(igloc, x) = G[x];
1133 }
1134 }
1135 gvec_coord_.allocate(memory_t::device).copy_to(memory_t::device);
1136 break;
1137 }
1138 }

1144 * Always update the mapping between the canonical FFT distribution and "local G-shells"
1145 * distribution which is used in symmetriezation of lattice periodic functions. */

1148 /* check symmetry of G-vectors */
1149 if (unit_cell().num_atoms() != 0 && use_symmetry() && cfg().control().verification() >= 1) {
1150 check_gvec(gvec(), unit_cell().symmetry());
1151 if (!full_potential()) {
1152 check_gvec(gvec_coarse(), unit_cell().symmetry());
1153 }
1154 check_gvec(*remap_gvec_, unit_cell().symmetry());
1155 }
1156
1157 /* check if FFT grid is OK; this check is especially needed if the grid is set as external parameter */
1158 if (cfg().control().verification() >= 0) {
1159 #pragma omp parallel for
1160 for (int igloc = 0; igloc < gvec().count(); igloc++) {
1161 int ig = gvec().offset() + igloc;
1162
1163 auto gv = gvec().gvec(ig);
1164 /* check limits */
1165 for (int x : {0, 1, 2}) {
1166 auto limits = fft_grid().limits(x);
1167 /* check boundaries */
1168 if (gv[x] < limits.first || gv[x] > limits.second) {
1169 std::stringstream s;
1170 s << "G-vector is outside of grid limits\n"
1171 << " G: " << gv << ", length: " << gvec().gvec_cart<index_domain_t::global>(ig).length() << "\n"
1172 << " FFT grid limits: " << fft_grid().limits(0).first << " " << fft_grid().limits(0).second
1173 << " " << fft_grid().limits(1).first << " " << fft_grid().limits(1).second << " "
1174 << fft_grid().limits(2).first << " " << fft_grid().limits(2).second << "\n"
1175 << " FFT grid is not compatible with G-vector cutoff (" << this->pw_cutoff() << ")";
1176 RTE_THROW(s);
1177 }
1178 }
1179 }
1180 }
1181
1182 init_atoms_to_grid_idx(cfg().control().rmt_max());
1183
1184 std::pair<int, int> limits(0, 0);
1185 for (int x : {0, 1, 2}) {
1186 limits.first = std::min(limits.first, fft_grid().limits(x).first);
1187 limits.second = std::max(limits.second, fft_grid().limits(x).second);
1188 }
1189
1190 /* recompute phase factors for atoms */
1191 phase_factors_ = mdarray<double_complex, 3>(3, limits, unit_cell().num_atoms(), memory_t::host, "phase_factors_");
1192 #pragma omp parallel for
1193 for (int i = limits.first; i <= limits.second; i++) {
1194 for (int ia = 0; ia < unit_cell().num_atoms(); ia++) {
1195 auto pos = unit_cell().atom(ia).position();
1196 for (int x : {0, 1, 2}) {
1197 phase_factors_(x, i, ia) = std::exp(double_complex(0.0, twopi * (i * pos[x])));
1198 }
1199 }
1200 }
1201
1202 /* recompute phase factors for atom types */
1203 phase_factors_t_ = mdarray<double_complex, 2>(gvec().count(), unit_cell().num_atom_types());
1204 #pragma omp parallel for schedule(static)
1205 for (int igloc = 0; igloc < gvec().count(); igloc++) {
1206 /* global index of G-vector */
1207 int ig = gvec().offset() + igloc;
1208 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
1209 double_complex z(0, 0);
1210 for (int ia = 0; ia < unit_cell().atom_type(iat).num_atoms(); ia++) {
1211 z += gvec_phase_factor(ig, unit_cell().atom_type(iat).atom_id(ia));
1212 }
1213 phase_factors_t_(igloc, iat) = z;
1214 }
1215 }
1216
1217 if (use_symmetry()) {
1218 sym_phase_factors_ = mdarray<double_complex, 3>(3, limits, unit_cell().symmetry().size());
1219
1220 #pragma omp parallel for
1221 for (int i = limits.first; i <= limits.second; i++) {
1222 for (int isym = 0; isym < unit_cell().symmetry().size(); isym++) {
1223 auto t = unit_cell().symmetry()[isym].spg_op.t;
1224 for (int x : {0, 1, 2}) {
1225 sym_phase_factors_(x, i, isym) = std::exp(double_complex(0.0, twopi * (i * t[x])));
1226 }
1227 }
1228 }
1229 }
1230
1231 /* precompute some G-vector related arrays */
1232 gvec_tp_ = sddk::mdarray<double, 2>(gvec().count(), 2, memory_t::host, "gvec_tp_");
1233 #pragma omp parallel for schedule(static)
1234 for (int igloc = 0; igloc < gvec().count(); igloc++) {
1235 auto rtp = SHT::spherical_coordinates(gvec().gvec_cart<index_domain_t::local>(igloc));
1236 gvec_tp_(igloc, 0) = rtp[1];
1237 gvec_tp_(igloc, 1) = rtp[2];
1238 }
1239
1240 switch (this->processing_unit()) {
1241 case device_t::CPU: {
1242 break;
1243 }
1244 case device_t::GPU: {
1245 gvec_tp_.allocate(memory_t::device).copy_to(memory_t::device);
1246 break;
1247 }
1248 }
1249
1250 /* create or update radial integrals */
1251 if (!full_potential()) {
1252 /* find the new maximum length of G-vectors */
1253 double new_pw_cutoff{this->pw_cutoff()};
1254 for (int igloc = 0; igloc < gvec().count(); igloc++) {
1255 int ig = gvec().offset() + igloc;
1256 new_pw_cutoff = std::max(new_pw_cutoff, gvec().gvec_len(ig));
1257 }
1258 gvec().comm().allreduce<double, mpi_op_t::max>(&new_pw_cutoff, 1);
1259 /* estimate new G+k-vectors cutoff */
1260 double new_gk_cutoff = this->gk_cutoff();
1261 if (new_pw_cutoff > this->pw_cutoff()) {
1262 new_gk_cutoff += (new_pw_cutoff - this->pw_cutoff());
1263 }
1264
1265 /* radial integrals with pw_cutoff */
1266 if (!aug_ri_ || aug_ri_->qmax() < new_pw_cutoff) {
1267 aug_ri_ = std::unique_ptr<Radial_integrals_aug<false>>(new Radial_integrals_aug<false>(
1268 unit_cell(), new_pw_cutoff, cfg().settings().nprii_aug(), aug_ri_callback_));
1269 }
1270
1271 if (!aug_ri_djl_ || aug_ri_djl_->qmax() < new_pw_cutoff) {
1272 aug_ri_djl_ = std::unique_ptr<Radial_integrals_aug<true>>(new Radial_integrals_aug<true>(
1273 unit_cell(), new_pw_cutoff, cfg().settings().nprii_aug(), aug_ri_djl_callback_));
1274 }
1275
1276 if (!ps_core_ri_ || ps_core_ri_->qmax() < new_pw_cutoff) {
1277 ps_core_ri_ =
1278 std::unique_ptr<Radial_integrals_rho_core_pseudo<false>>(new Radial_integrals_rho_core_pseudo<false>(
1279 unit_cell(), new_pw_cutoff, cfg().settings().nprii_rho_core(), rhoc_ri_callback_));
1280 }
1281
1282 if (!ps_core_ri_djl_ || ps_core_ri_djl_->qmax() < new_pw_cutoff) {
1283 ps_core_ri_djl_ =
1284 std::unique_ptr<Radial_integrals_rho_core_pseudo<true>>(new Radial_integrals_rho_core_pseudo<true>(
1285 unit_cell(), new_pw_cutoff, cfg().settings().nprii_rho_core(), rhoc_ri_djl_callback_));
1286 }
1287
1288 if (!ps_rho_ri_ || ps_rho_ri_->qmax() < new_pw_cutoff) {
1289 ps_rho_ri_ = std::unique_ptr<Radial_integrals_rho_pseudo>(
1290 new Radial_integrals_rho_pseudo(unit_cell(), new_pw_cutoff, 20, ps_rho_ri_callback_));
1291 }
1292
1293 if (!vloc_ri_ || vloc_ri_->qmax() < new_pw_cutoff) {
1294 vloc_ri_ = std::unique_ptr<Radial_integrals_vloc<false>>(new Radial_integrals_vloc<false>(
1295 unit_cell(), new_pw_cutoff, cfg().settings().nprii_vloc(), vloc_ri_callback_));
1296 }

1298 if (!vloc_ri_djl_ || vloc_ri_djl_->qmax() < new_pw_cutoff) {
1299 vloc_ri_djl_ = std::unique_ptr<Radial_integrals_vloc<true>>(new Radial_integrals_vloc<true>(
1300 unit_cell(), new_pw_cutoff, cfg().settings().nprii_vloc(), vloc_ri_djl_callback_));
1301 }
1302
1303 /* radial integrals with pw_cutoff */
1304 if (!beta_ri_ || beta_ri_->qmax() < new_gk_cutoff) {
1305 beta_ri_ = std::unique_ptr<Radial_integrals_beta<false>>(new Radial_integrals_beta<false>(
1306 unit_cell(), new_gk_cutoff, cfg().settings().nprii_beta(), beta_ri_callback_));
1307 }
1308
1309 if (!beta_ri_djl_ || beta_ri_djl_->qmax() < new_gk_cutoff) {
1310 beta_ri_djl_ = std::unique_ptr<Radial_integrals_beta<true>>(new Radial_integrals_beta<true>(
1311 unit_cell(), new_gk_cutoff, cfg().settings().nprii_beta(), beta_ri_djl_callback_));
1312 }

1315 return unit_cell().atom_type(iat).indexr_wfs();
1316 };
1317

1320 };
1321

1328 atomic_wf_ri_djl_ = std::unique_ptr<Radial_integrals_atomic_wf<true>>(
1329 new Radial_integrals_atomic_wf<true>(unit_cell(), new_gk_cutoff, 20, idxr_wf, ps_wf));
1330 }

1333 return unit_cell().atom_type(iat).indexr_hub();
1334 };
1335
1336 auto ps_wf_hub = [&](int iat, int i) -> Spline<double> const& {
1337 return unit_cell().atom_type(iat).hubbard_radial_function(i);
1338 };
1339
1340 if (!hubbard_wf_ri_ || hubbard_wf_ri_->qmax() < new_gk_cutoff) {
1341 hubbard_wf_ri_ = std::unique_ptr<Radial_integrals_atomic_wf<false>>(
1342 new Radial_integrals_atomic_wf<false>(unit_cell(), new_gk_cutoff, 20, idxr_wf_hub, ps_wf_hub));
1343 }
1344
1345 if (!hubbard_wf_ri_djl_ || hubbard_wf_ri_djl_->qmax() < new_gk_cutoff) {
1346 hubbard_wf_ri_djl_ = std::unique_ptr<Radial_integrals_atomic_wf<true>>(
1347 new Radial_integrals_atomic_wf<true>(unit_cell(), new_gk_cutoff, 20, idxr_wf_hub, ps_wf_hub));
1348 }
1349 /* update augmentation operator */
1350 memory_pool* mp{nullptr};
1351 memory_pool* mpd{nullptr};
1352 switch (this->processing_unit()) {
1353 case device_t::CPU: {
1354 mp = &mem_pool(memory_t::host);
1355 break;
1356 }
1357 case device_t::GPU: {
1358 mp = &mem_pool(memory_t::host_pinned);
1359 mpd = &mem_pool(memory_t::device);
1360 break;
1361 }
1362 }
1363 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
1364 if (unit_cell().atom_type(iat).augment() && unit_cell().atom_type(iat).num_atoms() > 0) {
1365 augmentation_op_[iat] = std::unique_ptr<Augmentation_operator>(
1366 new Augmentation_operator(unit_cell().atom_type(iat), gvec()));
1367 augmentation_op_[iat]->generate_pw_coeffs(aug_ri(), gvec_tp_, *mp, mpd);
1368 } else {
1369 augmentation_op_[iat] = nullptr;
1370 }
1371 }
1372 }

1374 if (full_potential()) { // TODO: add corresponging radial integarls of Theta
1375 init_step_function();
1376 }

1475void
1476Simulation_context::init_atoms_to_grid_idx(double R__)
1477{
1478 PROFILE("sirius::Simulation_context::init_atoms_to_grid_idx");
1479
1480 atoms_to_grid_idx_.resize(unit_cell().num_atoms());
1481
1482 vector3d<double> delta(1.0 / spfft<double>().dim_x(), 1.0 / spfft<double>().dim_y(), 1.0 / spfft<double>().dim_z());
1483
1484 int z_off = spfft<double>().local_z_offset();
1485 vector3d<int> grid_beg(0, 0, z_off);
1486 vector3d<int> grid_end(spfft<double>().dim_x(), spfft<double>().dim_y(), z_off + spfft<double>().local_z_length());
1487 std::vector<vector3d<double>> verts_cart{{-R__, -R__, -R__}, {R__, -R__, -R__}, {-R__, R__, -R__},
1488 {R__, R__, -R__}, {-R__, -R__, R__}, {R__, -R__, R__},
1489 {-R__, R__, R__}, {R__, R__, R__}};
1490
1491 auto bounds_box = [&](vector3d<double> pos) {
1492 std::vector<vector3d<double>> verts;
1493
1494 /* pos is a position of atom */
1495 for (auto v : verts_cart) {
1496 verts.push_back(pos + unit_cell().get_fractional_coordinates(v));
1497 }
1498
1499 std::pair<vector3d<int>, vector3d<int>> bounds_ind;
1500
1501 for (int x : {0, 1, 2}) {
1502 std::sort(verts.begin(), verts.end(),
1503 [x](vector3d<double>& a, vector3d<double>& b) { return a[x] < b[x]; });
1504 bounds_ind.first[x] = std::max(static_cast<int>(verts[0][x] / delta[x]) - 1, grid_beg[x]);
1505 bounds_ind.second[x] = std::min(static_cast<int>(verts[5][x] / delta[x]) + 1, grid_end[x]);
1506 }
1507
1508 return bounds_ind;
1509 };
1510
1511 #pragma omp parallel for
1512 for (int ia = 0; ia < unit_cell().num_atoms(); ia++) {
1513
1514 std::vector<std::pair<int, double>> atom_to_ind_map;
1515
1516 for (int t0 = -1; t0 <= 1; t0++) {
1517 for (int t1 = -1; t1 <= 1; t1++) {
1518 for (int t2 = -1; t2 <= 1; t2++) {
1519 auto pos = unit_cell().atom(ia).position() + vector3d<double>(t0, t1, t2);
1520
1521 /* find the small box around this atom */
1522 auto box = bounds_box(pos);
1523
1524 for (int j0 = box.first[0]; j0 < box.second[0]; j0++) {
1525 for (int j1 = box.first[1]; j1 < box.second[1]; j1++) {
1526 for (int j2 = box.first[2]; j2 < box.second[2]; j2++) {
1527 auto v = pos - vector3d<double>(delta[0] * j0, delta[1] * j1, delta[2] * j2);
1528 auto r = unit_cell().get_cartesian_coordinates(v).length();
1529 if (r < R__) {
1530 auto ir = fft_grid_.index_by_coord(j0, j1, j2 - z_off);
1531 atom_to_ind_map.push_back({ir, r});
1532 }
1533 }
1534 }
1535 }
1536 }
1537 }
1538 }
1539
1540 atoms_to_grid_idx_[ia] = std::move(atom_to_ind_map);
1541 }

1544void
1545Simulation_context::init_step_function()
1546{
1547 auto v = make_periodic_function<index_domain_t::global>([&](int iat, double g) {
1548 auto R = unit_cell().atom_type(iat).mt_radius();
1549 return unit_step_function_form_factors(R, g);
1550 });
1551
1552 theta_ = sddk::mdarray<double, 1>(spfft<double>().local_slice_size());
1553 theta_pw_ = sddk::mdarray<double_complex, 1>(gvec().num_gvec());
1554
1555 try {
1556 for (int ig = 0; ig < gvec().num_gvec(); ig++) {
1557 theta_pw_[ig] = -v[ig];
1558 }
1559 theta_pw_[0] += 1.0;
1560
1561 std::vector<double_complex> ftmp(gvec_partition().gvec_count_fft());
1562 for (int i = 0; i < gvec_partition().gvec_count_fft(); i++) {
1563 ftmp[i] = theta_pw_[gvec_partition().idx_gvec(i)];
1564 }
1565 spfft<double>().backward(reinterpret_cast<double const*>(ftmp.data()), SPFFT_PU_HOST);
1566 double* theta_ptr = spfft<double>().local_slice_size() == 0 ? nullptr : &theta_[0];
1567 spfft_output(spfft<double>(), theta_ptr);
1568 } catch (...) {
1569 std::stringstream s;
1570 s << "fft_grid = " << fft_grid_[0] << " " << fft_grid_[1] << " " << fft_grid_[2] << std::endl
1571 << "spfft<double>().local_slice_size() = " << spfft<double>().local_slice_size() << std::endl
1572 << "gvec_partition().gvec_count_fft() = " << gvec_partition().gvec_count_fft();
1573 RTE_THROW(s);
1574 }
1575
1576 double vit{0};
1577 for (int i = 0; i < spfft<double>().local_slice_size(); i++) {
1578 vit += theta_[i];
1579 }
1580 vit *= (unit_cell().omega() / fft_grid().num_points());
1581 Communicator(spfft<double>().communicator()).allreduce(&vit, 1);
1582
1583 if (std::abs(vit - unit_cell().volume_it()) > 1e-10) {
1584 std::stringstream s;
1585 s << "step function gives a wrong volume for IT region" << std::endl
1586 << " difference with exact value : " << std::abs(vit - unit_cell().volume_it());
1587 if (comm().rank() == 0) {
1588 WARNING(s);
1589 }
1590 }
1591 if (cfg().control().print_checksum()) {
1592 double_complex z1 = theta_pw_.checksum();
1593 double d1 = theta_.checksum();
1594 Communicator(spfft<double>().communicator()).allreduce(&d1, 1);
1595 if (comm().rank() == 0) {
1596 utils::print_checksum("theta", d1);
1597 utils::print_checksum("theta_pw", z1);
1598 }
1599 }

1602void
1603Simulation_context::init_comm()
1604{
1605 PROFILE("sirius::Simulation_context::init_comm");
1606
1607 /* check MPI grid dimensions and set a default grid if needed */
1608 if (!cfg().control().mpi_grid_dims().size()) {
1609 mpi_grid_dims({1, 1});
1610 }
1611 if (cfg().control().mpi_grid_dims().size() != 2) {
1612 RTE_THROW("wrong MPI grid");
1613 }
1614
1615 int npr = cfg().control().mpi_grid_dims()[0];
1616 int npc = cfg().control().mpi_grid_dims()[1];
1617 int npb = npr * npc;
1618 if (npb <= 0) {
1619 std::stringstream s;
1620 s << "wrong mpi grid dimensions : " << npr << " " << npc;
1621 RTE_THROW(s);
1622 }
1623 int npk = comm_.size() / npb;
1624 if (npk * npb != comm_.size()) {
1625 std::stringstream s;
1626 s << "Can't divide " << comm_.size() << " ranks into groups of size " << npb;
1627 RTE_THROW(s);
1628 }
1629
1630 /* create k- and band- communicators */
1631 if (comm_k_.is_null() && comm_band_.is_null()) {
1632 comm_band_ = comm_.split(comm_.rank() / npb);
1633 comm_k_ = comm_.split(comm_.rank() % npb);
1634 }
1635
1636 /* setup MPI grid */
1637 mpi_grid_ = std::unique_ptr<MPI_grid>(new MPI_grid({npc, npr}, comm_band_));
1638
1639 /* here we know the number of ranks for band parallelization */
