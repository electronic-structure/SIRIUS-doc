113 mdarray<double_complex, 2>
114 Simulation_context::sum_fg_fl_yg(int lmax__, double_complex const* fpw__, mdarray<double, 3>& fl__,
115  matrix<double_complex>& gvec_ylm__)
116 {
117  PROFILE("sirius::Simulation_context::sum_fg_fl_yg");
118 
119  int ngv_loc = gvec().count();
120 
121  int na_max{0};
122  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
123  na_max = std::max(na_max, unit_cell().atom_type(iat).num_atoms());
124  }
125 
126  int lmmax = utils::lmmax(lmax__);
127  /* resuling matrix */
128  mdarray<double_complex, 2> flm(lmmax, unit_cell().num_atoms());
129 
130  matrix<double_complex> phase_factors;
131  matrix<double_complex> zm;
132  matrix<double_complex> tmp;
133 
134  switch (processing_unit()) {
135  case device_t::CPU: {
136  auto& mp = mem_pool(memory_t::host);
137  phase_factors = matrix<double_complex>(ngv_loc, na_max, mp);
138  zm = matrix<double_complex>(lmmax, ngv_loc, mp);
139  tmp = matrix<double_complex>(lmmax, na_max, mp);
140  break;
141  }
142  case device_t::GPU: {
143  auto& mp = mem_pool(memory_t::host);
144  auto& mpd = mem_pool(memory_t::device);
145  phase_factors = matrix<double_complex>(nullptr, ngv_loc, na_max);
146  phase_factors.allocate(mpd);
147  zm = matrix<double_complex>(lmmax, ngv_loc, mp);
148  zm.allocate(mpd);
149  tmp = matrix<double_complex>(lmmax, na_max, mp);
150  tmp.allocate(mpd);
151  break;
152  }
153  }
154 
155  std::vector<double_complex> zil(lmax__ + 1);
156  for (int l = 0; l <= lmax__; l++) {
157  zil[l] = std::pow(double_complex(0, 1), l);
158  }
159 
160  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
161  int na = unit_cell().atom_type(iat).num_atoms();
162  generate_phase_factors(iat, phase_factors);
163  PROFILE_START("sirius::Simulation_context::sum_fg_fl_yg|zm");
164  #pragma omp parallel for schedule(static)
165  for (int igloc = 0; igloc < ngv_loc; igloc++) {
166  for (int l = 0, lm = 0; l <= lmax__; l++) {
167  double_complex z = fourpi * fl__(l, igloc, iat) * zil[l] * fpw__[igloc];
168  for (int m = -l; m <= l; m++, lm++) {
169  zm(lm, igloc) = z * std::conj(gvec_ylm__(lm, igloc));
170  }
171  }
172  }
173  PROFILE_STOP("sirius::Simulation_context::sum_fg_fl_yg|zm");
174  PROFILE_START("sirius::Simulation_context::sum_fg_fl_yg|mul");
175  switch (processing_unit()) {
176  case device_t::CPU: {
177  linalg(linalg_t::blas).gemm('N', 'N', lmmax, na, ngv_loc,
178  &linalg_const<double_complex>::one(), zm.at(memory_t::host), zm.ld(),
179  phase_factors.at(memory_t::host), phase_factors.ld(),
180  &linalg_const<double_complex>::zero(), tmp.at(memory_t::host), tmp.ld());
181  break;
182  }
183  case device_t::GPU: {
184  zm.copy_to(memory_t::device);
185  linalg(linalg_t::gpublas).gemm('N', 'N', lmmax, na, ngv_loc,
186  &linalg_const<double_complex>::one(), zm.at(memory_t::device), zm.ld(),
187  phase_factors.at(memory_t::device), phase_factors.ld(),
188  &linalg_const<double_complex>::zero(), tmp.at(memory_t::device), tmp.ld());
189  tmp.copy_to(memory_t::host);
190  break;
191  }
192  }
193  PROFILE_STOP("sirius::Simulation_context::sum_fg_fl_yg|mul");
194 
195  for (int i = 0; i < na; i++) {
196  int ia = unit_cell().atom_type(iat).atom_id(i);
197  std::copy(&tmp(0, i), &tmp(0, i) + lmmax, &flm(0, ia));
198  }
199  }
200 
201  comm().allreduce(&flm(0, 0), (int)flm.size());
202 
203  return flm;
204 }

206 double Simulation_context::ewald_lambda() const
207 {
208  /* alpha = 1 / (2*sigma^2), selecting alpha here for better convergence */
209  double lambda{1};
210  double gmax = pw_cutoff();
211  double upper_bound{0};
212  double charge = unit_cell().num_electrons();
213 
214  /* iterate to find lambda */
215  do {
216  lambda += 0.1;
217  upper_bound =
218  charge * charge * std::sqrt(2.0 * lambda / twopi) * std::erfc(gmax * std::sqrt(1.0 / (4.0 * lambda)));
219  } while (upper_bound < 1e-8);
220 
221  if (lambda < 1.5 && comm().rank() == 0) {
222  std::stringstream s;
223  s << "ewald_lambda(): pw_cutoff is too small";
224  WARNING(s);
225  }
226  return lambda;
227 }

229 splindex<splindex_t::block> Simulation_context::split_gvec_local() const
230 {
231  /* local number of G-vectors for this MPI rank */
232  int ngv_loc = gvec().count();
233  /* estimate number of G-vectors in a block */
234  int ld{-1};
235  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
236  int nat = unit_cell().atom_type(iat).num_atoms();
237  int nbf = unit_cell().atom_type(iat).mt_basis_size();
238 
239  ld = std::max(ld, std::max(nbf * (nbf + 1) / 2, nat));
240  }
241  /* limit the size of relevant array to ~1Gb */
242  int ngv_b = (1 << 30) / sizeof(double_complex) / ld;
243  ngv_b = std::max(1, std::min(ngv_loc, ngv_b));
244  /* number of blocks of G-vectors */
245  int nb = ngv_loc / ngv_b;
246  /* split local number of G-vectors between blocks */
247  return splindex<splindex_t::block>(ngv_loc, nb, 0);
248 }

250 void Simulation_context::initialize()
251 {
252  PROFILE("sirius::Simulation_context::initialize");
253 
254  /* can't initialize twice */
255  if (initialized_) {
256  TERMINATE("Simulation parameters are already initialized.");
257  }
258  electronic_structure_method(parameters_input().electronic_structure_method_);
259  set_core_relativity(parameters_input().core_relativity_);
260  set_valence_relativity(parameters_input().valence_relativity_);
261 
262  /* can't run fp-lapw with Gamma point trick */
263  if (full_potential()) {
264  gamma_point(false);
265  }
266 
267  /* Gamma-point calculation and non-collinear magnetism are not compatible */
268  if (num_mag_dims() == 3) {
269  gamma_point(false);
270  }
271 
272  /* set processing unit type */
273  set_processing_unit(control().processing_unit_);
274 
275  /* check if we can use a GPU device */
276  if (processing_unit() == device_t::GPU) {
277 #if !defined(__GPU)
278  throw std::runtime_error("not compiled with GPU support!");
279 #endif
280  }
281 
282  /* initialize MPI communicators */
283  init_comm();
284 
285  auto print_mpi_layout = utils::get_env<int>("SIRIUS_PRINT_MPI_LAYOUT");
286 
287  if (control().verbosity_ >= 3 || (print_mpi_layout && *print_mpi_layout)) {
288  pstdout pout(comm());
289  if (comm().rank() == 0) {
290  pout.printf("MPI rank placement\n");
291  pout.printf("------------------\n");
292  }
293  auto name = sddk::Communicator::processor_name();
294  pout.printf("rank: %3i, comm_band_rank: %3i, comm_k_rank: %3i, hostname: %s, mpi processor name: %s\n",
295  comm().rank(), comm_band().rank(), comm_k().rank(), utils::hostname().c_str(), name.c_str());
296  }
297 
298  switch (processing_unit()) {
299  case device_t::CPU: {
300  host_memory_t_ = memory_t::host;
301  break;
302  }
303  case device_t::GPU: {
304  host_memory_t_ = memory_t::host_pinned;
305  break;
306  }
307  }
308 
309  switch (processing_unit()) {
310  case device_t::CPU: {
311  preferred_memory_t_ = memory_t::host;
312  break;
313  }
314  case device_t::GPU: {
315  if (control_input_.memory_usage_ == "high") {
316  preferred_memory_t_ = memory_t::device;
317  }
318  if (control_input_.memory_usage_ == "low" || control_input_.memory_usage_ == "medium") {
319  preferred_memory_t_ = memory_t::host_pinned;
320  }
321  break;
322  }
323  }
324 
325  switch (processing_unit()) {
326  case device_t::CPU: {
327  aux_preferred_memory_t_ = memory_t::host;
328  break;
329  }
330  case device_t::GPU: {
331  if (control_input_.memory_usage_ == "high" || control_input_.memory_usage_ == "medium") {
332  aux_preferred_memory_t_ = memory_t::device;
333  }
334  if (control_input_.memory_usage_ == "low") {
335  aux_preferred_memory_t_ = memory_t::host_pinned;
336  }
337  break;
338  }
339  }
340 
341  switch (processing_unit()) {
342  case device_t::CPU: {
343  blas_linalg_t_ = linalg_t::blas;
344  break;
345  }
346  case device_t::GPU: {
347  if (control_input_.memory_usage_ == "high") {
348  blas_linalg_t_ = linalg_t::gpublas;
349  }
350  if (control_input_.memory_usage_ == "low" || control_input_.memory_usage_ == "medium") {
351 #ifdef __ROCM
352  blas_linalg_t_ = linalg_t::gpublas;
353 #else
354  blas_linalg_t_ = linalg_t::cublasxt;
355 #endif
356  }
357  break;
358  }
359  }
360 
361  /* can't use reduced G-vectors in LAPW code */
362  if (full_potential()) {
363  control_input_.reduce_gvec_ = false;
364  }
365 
366  if (!iterative_solver_input_.type_.size()) {
367  if (full_potential()) {
368  iterative_solver_input_.type_ = "exact";
369  } else {
370  iterative_solver_input_.type_ = "davidson";
371  }
372  }
373  /* set default values for the G-vector cutoff */
374  if (pw_cutoff() <= 0) {
375  pw_cutoff(full_potential() ? 12 : 20);
376  }
377 
378  /* initialize variables related to the unit cell */
379  unit_cell().initialize();
380  /* save the volume of the initial unit cell */
381  omega0_ = unit_cell().omega();
382 
383  /* check the lattice symmetries */
384  if (use_symmetry()) {
385  auto lv = matrix3d<double>(unit_cell().lattice_vectors());
386 
387  auto lat_sym = find_lat_sym(lv, 1e-6);
388 
389  #pragma omp parallel for
390  for (int i = 0; i < unit_cell().symmetry().num_mag_sym(); i++) {
391  auto& spgR = unit_cell().symmetry().magnetic_group_symmetry(i).spg_op.R;
392  bool found{false};
393  for (size_t i = 0; i < lat_sym.size(); i++) {
394  auto latR = lat_sym[i];
395  found = true;
396  for (int x : {0, 1, 2}) {
397  for (int y : {0, 1, 2}) {
398  found = found && (spgR(x, y) == latR(x, y));
399  }
400  }
401  if (found) {
402  break;
403  }
404  }
405  if (!found) {
406  TERMINATE("spglib lattice symmetry was not found in the list of SIRIUS generated symmetries");
407  }
408  }
409  }
410 
411  /* set default values for the G+k-vector cutoff */
412  if (full_potential()) {
413  /* find the cutoff for G+k vectors (derived from rgkmax (aw_cutoff here) and minimum MT radius) */
414  if (aw_cutoff() > 0) {
415  gk_cutoff(aw_cutoff() / unit_cell().min_mt_radius());
416  }
417  if (gk_cutoff() <= 0) {
418  gk_cutoff(3);
419  }
420  } else {
421  /* in pseudopotential case */
422  if (gk_cutoff() <= 0) {
423  gk_cutoff(7);
424  }
425  }
426 
427  /* check the G+k cutoff */
428  if (gk_cutoff() * 2 > pw_cutoff()) {
429  std::stringstream s;
430  s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
431  << " pw cutoff : " << pw_cutoff() << std::endl
432  << " doubled G+k cutoff : " << gk_cutoff() * 2;
433  TERMINATE(s);
434  }
435 
436  if (!full_potential()) {
437  set_lmax_rho(unit_cell().lmax() * 2);
438  set_lmax_pot(unit_cell().lmax() * 2);
439  set_lmax_apw(-1);
440  }
441 
442  /* initialize FFT grid dimensions */
443  init_fft_grid();
444 
445  int nbnd = static_cast<int>(unit_cell().num_valence_electrons() / 2.0) +
446  std::max(10, static_cast<int>(0.1 * unit_cell().num_valence_electrons()));
447  if (full_potential()) {
448  /* take 10% of empty non-magnetic states */
449  if (num_fv_states() < 0) {
450  num_fv_states(nbnd);
451  }
452  if (num_fv_states() < static_cast<int>(unit_cell().num_valence_electrons() / 2.0)) {
453  std::stringstream s;
454  s << "not enough first-variational states : " << num_fv_states();
455  TERMINATE(s);
456  }
457  } else {
458  if (num_mag_dims() == 3) {
459  nbnd *= 2;
460  }
461  /* if number of bands was not set by the host code, set it here */
462  if (num_bands() < 0) {
463  num_bands(nbnd);
464  }
465  }
466 
467  std::string evsn[] = {std_evp_solver_name(), gen_evp_solver_name()};
468 #if defined(__CUDA)
469  bool is_cuda{true};
470 #else
471  bool is_cuda{false};
472 #endif
473 #if defined(__MAGMA)
474  bool is_magma{true};
475 #else
476  bool is_magma{false};
477 #endif
478 #if defined(__SCALAPACK)
479  bool is_scalapack{true};
480 #else
481  bool is_scalapack{false};
482 #endif
483 #if defined(__ELPA)
484  bool is_elpa{true};
485 #else
486  bool is_elpa{false};
487 #endif
488 
489  if (processing_unit() == device_t::CPU || acc::num_devices() == 0) {
490  is_cuda = false;
491  is_magma = false;
492  }
493 
494  int npr = control_input_.mpi_grid_dims_[0];
495  int npc = control_input_.mpi_grid_dims_[1];
496 
497  /* deduce the default eigen-value solver */
498  for (int i : {0, 1}) {
499  if (evsn[i] == "") {
500  /* conditions for sequential diagonalization */
501  if (comm_band().size() == 1 || npc == 1 || npr == 1 || !is_scalapack) {
502  if (full_potential()) {
503  if (is_magma) {
504  evsn[i] = "magma";
505  } else if (is_cuda) {
506  evsn[i] = "cusolver";
507  } else {
508  evsn[i] = "lapack";
509  }
510  //if (is_cuda) {
511  // evsn[i] = "cusolver";
512  //} else if (is_magma) {
513  // evsn[i] = "magma";
514  //} else {
515  // evsn[i] = "lapack";
516  //}
517  } else {
518  if (is_cuda) {
519  evsn[i] = "cusolver";
520  } else if (is_magma && num_bands() > 200) {
521  evsn[i] = "magma";
522  } else {
523  evsn[i] = "lapack";
524  }
525  }
526  } else {
527  if (is_scalapack) {
528  evsn[i] = "scalapack";
529  }
530  if (is_elpa) {
531  evsn[i] = "elpa1";
532  }
533  }
534  }
535  }
536 
537  std_evp_solver_name(evsn[0]);
538  gen_evp_solver_name(evsn[1]);
539 
540  std_evp_solver_ = Eigensolver_factory(std_evp_solver_name(), &mem_pool(memory_t::device));
541  gen_evp_solver_ = Eigensolver_factory(gen_evp_solver_name(), &mem_pool(memory_t::device));
542 
543  auto& std_solver = std_evp_solver();
544  auto& gen_solver = gen_evp_solver();
545 
546  if (std_solver.is_parallel() != gen_solver.is_parallel()) {
547  TERMINATE("both solvers must be sequential or parallel");
548  }
549 
550  /* setup BLACS grid */
551  if (std_solver.is_parallel()) {
552  blacs_grid_ = std::unique_ptr<BLACS_grid>(new BLACS_grid(comm_band(), npr, npc));
553  } else {
554  blacs_grid_ = std::unique_ptr<BLACS_grid>(new BLACS_grid(Communicator::self(), 1, 1));
555  }
556 
557  /* setup the cyclic block size */
558  if (cyclic_block_size() < 0) {
559  double a = std::min(std::log2(double(num_bands()) / blacs_grid_->num_ranks_col()),
560  std::log2(double(num_bands()) / blacs_grid_->num_ranks_row()));
561  if (a < 1) {
562  control_input_.cyclic_block_size_ = 2;
563  } else {
564  control_input_.cyclic_block_size_ =
565  static_cast<int>(std::min(128.0, std::pow(2.0, static_cast<int>(a))) + 1e-12);
566  }
567  }
568 
569  /* placeholder for augmentation operator for each atom type */
570  augmentation_op_.resize(unit_cell().num_atom_types());
571 
572  /* create G-vectors on the first call to update() */
573  update();
574 
575  this->print_memory_usage(__FILE__, __LINE__);
576 
577  if (control().verbosity_ >= 1 && comm().rank() == 0) {
578  print_info();
579  }
580 
581  initialized_ = true;
582 }
