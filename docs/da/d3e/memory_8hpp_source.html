56/// Memory types where the code can store data.
57/** All memory types can be divided into two (possibly overlapping) groups: accessible by the CPU and accessible by the
58 * device. */
59enum class memory_t : unsigned int
60{
61 /// Nothing.
62 none = 0b0000,
63 /// Host memory.
64 host = 0b0001,
65 /// Pinned host memory. This is host memory + extra bit flag.
66 host_pinned = 0b0011,
67 /// Device memory.
68 device = 0b1000,
69 /// Managed memory (accessible from both host and device).
70 managed = 0b1101,
71};
72
73/// Check if this is a valid host memory (memory, accessible by the host).
74inline bool is_host_memory(memory_t mem__)
75{
76 return static_cast<unsigned int>(mem__) & 0b0001;
77}
78
79/// Check if this is a valid device memory (memory, accessible by the device).
80inline bool is_device_memory(memory_t mem__)
81{
82 return static_cast<unsigned int>(mem__) & 0b1000;
83}

85/// Get a memory type from a string.
86inline memory_t get_memory_t(std::string name__)
87{
88 std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
89 std::map<std::string, memory_t> const m = {
90 {"none", memory_t::none},
91 {"host", memory_t::host},
92 {"host_pinned", memory_t::host_pinned},
93 {"managed", memory_t::managed},
94 {"device", memory_t::device}
95 };
96
97 if (m.count(name__) == 0) {
98 std::stringstream s;
99 s << "get_memory_t(): wrong label of the memory_t enumerator: " << name__;
100 throw std::runtime_error(s.str());
101 }
102 return m.at(name__);
103}

151/// Allocate n elements in a specified memory.
152/** Allocate a memory block of the memory_t type. Return a nullptr if this memory is not available, otherwise
153 * return a pointer to an allocated block. */
154template <typename T>
155inline T* allocate(size_t n__, memory_t M__)
156{
157 switch (M__) {
158 case memory_t::none: {
159 return nullptr;
160 }
161 case memory_t::host: {
162 return static_cast<T*>(std::malloc(n__ * sizeof(T)));
163 }
164 case memory_t::host_pinned: {
165#ifdef SIRIUS_GPU
166 return acc::allocate_host<T>(n__);
167#else
168 return nullptr;
169#endif
170 }
171 case memory_t::device: {
172#ifdef SIRIUS_GPU
173 return acc::allocate<T>(n__);
174#else
175 return nullptr;
176#endif
177 }
178 default: {
179 throw std::runtime_error("allocate(): unknown memory type");
180 }
181 }
182}

189 break;

213/// Copy between different memory types.
214template <typename T, typename F>
215inline void copy(memory_t from_mem__, T const* from_ptr__, memory_t to_mem__, F* to_ptr__, size_t n__)
216{
217 if (is_host_memory(to_mem__) && is_host_memory(from_mem__)) {
218 std::copy(from_ptr__, from_ptr__ + n__, to_ptr__);
219 return;
220 }
221#if defined(SIRIUS_GPU)
222 throw std::runtime_error("Copy mixed precision type not supported in device memory");
223 return;
224#endif
225}

227template <typename T>
228inline void copy(memory_t from_mem__, T const* from_ptr__, memory_t to_mem__, T* to_ptr__, size_t n__)
229{
230 if (is_host_memory(to_mem__) && is_host_memory(from_mem__)) {
231 std::memcpy(to_ptr__, from_ptr__, n__ * sizeof(T));
232 return;
233 }
234#if defined(SIRIUS_GPU)
235 if (is_device_memory(to_mem__) && is_device_memory(from_mem__)) {
236 acc::copy(to_ptr__, from_ptr__, n__);
237 return;
238 }
239 if (is_device_memory(to_mem__) && is_host_memory(from_mem__)) {
240 acc::copyin(to_ptr__, from_ptr__, n__);
241 return;
242 }
243 if (is_host_memory(to_mem__) && is_device_memory(from_mem__)) {
244 acc::copyout(to_ptr__, from_ptr__, n__);
245 return;
246 }
247#endif

331/// Descriptor of the allocated memory block.
332/** Internally the block might be split into sub-blocks. */
333struct memory_block_descriptor
334{
335 /// Storage buffer for the memory blocks.
336 std::unique_ptr<uint8_t, memory_t_deleter_base> buffer_;
337 /// Size of the storage buffer.
338 size_t size_{0};
339 /// List of <offset, size> pairs of the free subblocks.
340 /** The list is ordered, i.e. offset of the next free block is greater or equal to the offset + size of the
341 * previous block. */
342 std::list<std::pair<size_t, size_t>> free_subblocks_;
343
344 /// Create a new empty memory block.
345 memory_block_descriptor(size_t size__, memory_t M__)
346 : buffer_(get_unique_ptr<uint8_t>(size__, M__))
347 , size_(size__)
348 {
349 free_subblocks_.push_back(std::make_pair(0, size_));
350 }
351
352 /// Check if the memory block is empty.
353 inline bool is_empty() const
354 {
355 return (free_subblocks_.size() == 1 &&
356 free_subblocks_.front().first == 0 &&
357 free_subblocks_.front().second == size_);
358 }
359
360 /// Return total size of the block.
361 inline size_t size() const
362 {
363 return size_;
364 }
365
366 /// Try to allocate a subblock of memory.
367 /** Return a valid pointer in case of success and nullptr if empty space can't be found in this memory block.
368 The returned pointer is not aligned. */
369 uint8_t* allocate_subblock(size_t size__)
370 {
371 uint8_t* ptr{nullptr};
372 for (auto it = free_subblocks_.begin(); it != free_subblocks_.end(); it++) {
373 /* if this free subblock can fit the "size" elements */
374 if (size__ <= it->second) {
375 /* pointer to the beginning of subblock */
376 ptr = buffer_.get() + it->first;
377 it->first += size__;
378 it->second -= size__;
379 if (it->second == 0) {
380 free_subblocks_.erase(it);
381 }
382 break;
383 }
384 }
385 return ptr;
386 }
387
388 /// Free the pointer and its memory to the list of free subblocks.
389 void free_subblock(uint8_t* ptr__, size_t size__)
390 {
391 /* offset from the beginning of the memory buffer */
392 size_t offset = static_cast<size_t>(ptr__ - buffer_.get());
393 auto check_free_subblocks = [&]()
394 {
395#ifndef NDEBUG
396 if (free_subblocks_.size() <= 1) {
397 return;
398 }
399 auto it = free_subblocks_.begin();
400 auto it1 = it;
401 it1++;
402 for (; it1 != free_subblocks_.end(); it1++) {
403 /* if offse + size of the previous free block is larger than the offset of next block
404 this is an error */
405 if (it->first + it->second > it1->first) {
406 throw std::runtime_error("wrong order of free memory blocks");
407 }
408 }
409#endif
410 };
411
412 for (auto it = free_subblocks_.begin(); it != free_subblocks_.end(); it++) {
413 /* check if we can attach released subblock before this subblock */
414 if (it->first == offset + size__) {
415 it->first = offset;
416 it->second += size__;
417 check_free_subblocks();
418 return;
419 }
420 /* check if we can attach released subblock after this subblock */
421 if (it->first + it->second == offset) {
422 it->second += size__;
423 /* now check if we can attach this subblock to the top of the next one */
424 auto it1 = it;
425 it1++;
426 if (it1 != free_subblocks_.end()) {
427 if (it->first + it->second == it1->first) {
428 /* merge second block into first and erase it */
429 it->second += it1->second;
430 free_subblocks_.erase(it1);
431 }
432 }
433 check_free_subblocks();
434 return;
435 }
436 /* finally, check if the released subblock is before this subblock, but not touching it */
437 if (offset + size__ < it->first) {
438 free_subblocks_.insert(it, std::make_pair(offset, size__));
439 check_free_subblocks();
440 return;
441 }
442 }
443 /* otherwise this will be the last free subblock */
444 free_subblocks_.push_back(std::make_pair(offset, size__));
445 check_free_subblocks();
446 }
447
448 /// Return the total size of the free subblocks.
449 size_t get_free_size() const
450 {
451 size_t sz{0};
452 for (auto& e: free_subblocks_) {
453 sz += e.second;
454 }
455 return sz;
456 }
457};

459/// Store information about the allocated subblock: iterator in the list of memory blocks and subblock size;
460struct memory_subblock_descriptor
461{
462 /// Iterator in the list of block descriptors stored by memory pool.
463 /** Iterator points to a memory block in which this sub-block was allocated */
464 std::list<memory_block_descriptor>::iterator it_;
465 /// Size of the sub-block.
466 size_t size_;
467 /// This is the precise beginning of the memory sub-block.
468 /** Used to compute the exact location of the sub-block inside a memory block. */
469 uint8_t* unaligned_ptr_;
470};

472//// Memory pool.
473/** This class stores list of allocated memory blocks. Each of the blocks can be divided into subblocks. When subblock
474 * is deallocated it is merged with previous or next free subblock in the memory block. If this was the last subblock
475 * in the block of memory, the (now) free block of memory is merged with the neighbours (if any are available).
476 */
477class memory_pool
478{
479 private:
480 /// Type of memory that is handeled by this pool.
481 memory_t M_;
482 /// List of blocks of allocated memory.
483 std::list<memory_block_descriptor> memory_blocks_;
484 /// Mapping between an allocated pointer and a subblock descriptor.
485 std::map<uint8_t*, memory_subblock_descriptor> map_ptr_;
486
487 public:
488
489 /// Constructor
490 memory_pool(memory_t M__, size_t initial_size__ = 0)
491 : M_(M__)
492 {
493 if (initial_size__) {
494 memory_blocks_.push_back(memory_block_descriptor(initial_size__, M_));
495 }
496 }
497
498 /// Return a pointer to a memory block for n elements of type T.
499 template <typename T>
500 T* allocate(size_t num_elements__)
501 {
502#if defined(SIRIUS_USE_MEMORY_POOL)
503 /* memory block descriptor returns an unaligned memory; here we compute the the aligment value */
504 size_t align_size = std::max(size_t(SIRIUS_GPU_MEMORY_ALIGMENT), alignof(T));
505 /* size of the memory block in bytes */
506 size_t size = num_elements__ * sizeof(T) + align_size;
507
508 uint8_t* ptr{nullptr};
509
510 /* iterate over existing blocks */
511 auto it = memory_blocks_.begin();
512 for (; it != memory_blocks_.end(); it++) {
513 /* try to allocate a block */
514 ptr = it->allocate_subblock(size);
515 /* break if this memory block can store the subblock */
516 if (ptr) {
517 break;
518 }
519 }
520
521 /* if memory chunk was not found in the list of available blocks, add a new memory block with enough capacity */
522 if (!ptr) {
523 /* free all empty blocks and get their total size */
524 size_t new_size{0};
525 auto i = memory_blocks_.begin();
526 while (i != memory_blocks_.end()) {
527 if (i->is_empty()) {
528 new_size += i->size();
529 memory_blocks_.erase(i++);
530 } else {
531 ++i;
532 }
533 }
534 /* get upper limit for the size of the new block */
535 new_size = std::max(new_size, size);
536
537 memory_blocks_.push_back(memory_block_descriptor(new_size, M_));
538 it = memory_blocks_.end();
539 it--;
540 ptr = it->allocate_subblock(size);
541 }
542 if (!ptr) {
543 throw std::runtime_error("memory allocation failed");
544 }
545 /* save the information about the allocated memory sub-block */
546 memory_subblock_descriptor msb;
547 /* location in the list of blocks */
548 msb.it_ = it;
549 /* total size including the aligment */
550 msb.size_ = size;
551 /* beginning of the block (unaligned) */
552 msb.unaligned_ptr_ = ptr;
553 auto uip = reinterpret_cast<std::uintptr_t>(ptr);
554 /* align the pointer */
555 if (uip % align_size) {
556 uip += (align_size - uip % align_size);
557 }
558 auto aligned_ptr = reinterpret_cast<uint8_t*>(uip);
559 /* add to the hash table */
560 map_ptr_[aligned_ptr] = msb;
561 return reinterpret_cast<T*>(aligned_ptr);
562#else
563 return sddk::allocate<T>(num_elements__, M_);
564#endif
565 }
566
567 /// Delete a pointer and add its memory back to the pool.
568 void free(void* ptr__)
569 {
570#if defined(SIRIUS_USE_MEMORY_POOL)
571 auto ptr = reinterpret_cast<uint8_t*>(ptr__);
572 /* get a descriptor of this pointer */
573 auto& msb = map_ptr_.at(ptr);
574 /* free the sub-block */
575 msb.it_->free_subblock(msb.unaligned_ptr_, msb.size_);
576 /* remove this pointer from the hash table */
577 map_ptr_.erase(ptr);
578#else
579 sddk::deallocate(ptr__, M_);
580#endif
581 }

757/// Multidimensional array with the column-major (Fortran) order.
758/** The implementation supports two memory pointers: one is accessible by CPU and second is accessible by a device.
759 The following constructors are implemented:
760 \code{.cpp}
761 // wrap a host memory pointer and create 2D array 10 x 20.
762 mdarray<T, 2>(ptr, 10, 20);
763
764 // wrap a host and device pointers
765 mdarray<T, 2>(ptr, ptr_d, 10, 20);
766
767 // wrap a device pointers only
768 mdarray<T, 2>(nullptr, ptr_d, 10, 20);
769
770 // create 10 x 20 2D array in main memory
771 mdarray<T, 2>(10, 20);
772
773 // create 10 x 20 2D array in device memory
774 mdarray<T, 2>(10, 20, memory_t::device);
775

811 /// Initialize the offsets used to compute the index of the elements.
812 void init_dimensions(std::array<mdarray_index_descriptor, N> const dims__)

814 dims_ = dims__;

817 size_t ld{1};

825 /// Return linear index in the range [0, size) by the N-dimensional indices (i0, i1, ...)

829 static_assert(N == sizeof...(args), "wrong number of dimensions");
830 std::array<index_type, N> i = {args...};
831
832 for (int j = 0; j < N; j++) {
833 mdarray_assert(i[j] >= dims_[j].begin() && i[j] <= dims_[j].end());
834 }
835
836 size_t idx = offsets_[0] + i[0];
837 for (int j = 1; j < N; j++) {
838 idx += i[j] * offsets_[j];
839 }
840 mdarray_assert(idx >= 0 && idx < size());
841 return idx;
842 }
