36 inline const char*
37 error_message(::acc::blas::status_t status)
38 {
39  switch (status) {
40  case CUBLAS_STATUS_NOT_INITIALIZED: {
41  return "the library was not initialized";
42  break;
43  }
44  case CUBLAS_STATUS_INVALID_VALUE: {
45  return "the parameters m,n,k<0";
46  break;
47  }
48  case CUBLAS_STATUS_ARCH_MISMATCH: {
49  return "the device does not support double-precision";
50  break;
51  }
52  case CUBLAS_STATUS_EXECUTION_FAILED: {
53  return "the function failed to launch on the GPU";
54  break;
55  }
56  default: {
57  return "gpublas status unknown";
58  }
59  }
60 }

69 inline ::acc::blas::operation_t
70 get_gpublasOperation_t(char c)
71 {
72  switch (c) {
73  case 'n':
74  case 'N': {
75  return ::acc::blas::operation::None;
76  }
77  case 't':
78  case 'T': {
79  return ::acc::blas::operation::Transpose;
80  }
81  case 'c':
82  case 'C': {
83  return ::acc::blas::operation::ConjugateTranspose;
84  }
85  default: {
86  throw std::runtime_error("get_gpublasOperation_t(): wrong operation");
87  }
88  }
89  return ::acc::blas::operation::None; // make compiler happy
90 }

149 #define CALL_GPU_BLAS(func__, args__) \
150  { \
151  ::acc::blas::status_t status; \
152  if ((status = func__ args__) != ::acc::blas::status::Success) { \
153  error_message(status); \
154  char nm[1024]; \
155  gethostname(nm, 1024); \
156  std::printf("hostname: %s\n", nm); \
157  std::printf("Error in %s at line %i of file %s\n", #func__, __LINE__, __FILE__); \
158  stack_backtrace(); \
159  } \
160  }

169 inline void
170 create_stream_handles()
171 {
172  // acc::set_device();
173  CALL_GPU_BLAS(::acc::blas::create, (&null_stream_handle()));
174 
175  stream_handles() = std::vector<::acc::blas::handle_t>(acc::num_streams());
176  for (int i = 0; i < acc::num_streams(); i++) {
177  CALL_GPU_BLAS(::acc::blas::create, (&stream_handles()[i]));
178 
179  CALL_GPU_BLAS(::acc::blas::set_stream, (stream_handles()[i], acc::stream(stream_id(i))));
180  }
181 }
182 
183 inline void
184 destroy_stream_handles()
185 {
186  // acc::set_device();
187  CALL_GPU_BLAS(::acc::blas::destroy, (null_stream_handle()));
188  for (int i = 0; i < acc::num_streams(); i++) {
189  CALL_GPU_BLAS(::acc::blas::destroy, (stream_handles()[i]));
190  }
191 }
