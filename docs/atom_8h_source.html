117  /// Generate radial Hamiltonian and effective magnetic field integrals
118  /** Hamiltonian operator has the following representation inside muffin-tins:
119  * \f[
120  * \hat H = -\frac{1}{2}\nabla^2 + \sum_{\ell m} V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) =
121  * \underbrace{-\frac{1}{2} \nabla^2+V_{00}(r)R_{00}}_{H_{s}(r)} +\sum_{\ell=1} \sum_{m=-\ell}^{\ell}
122  * V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) = \sum_{\ell m} \widetilde V_{\ell m}(r) R_{\ell m}(\hat {\bf r})
123  * \f]
124  * where
125  * \f[
126  * \widetilde V_{\ell m}(r) = \left\{ \begin{array}{ll}
127  * \frac{H_{s}(r)}{R_{00}} & \ell = 0 \\
128  * V_{\ell m}(r) & \ell > 0 \end{array} \right.
129  * \f]
130  */
131  inline void generate_radial_integrals(device_t pu__, Communicator const& comm__);

245  /** Compute the following kinds of sums for different spin-blocks of the Hamiltonian:
246  * \f[
247  * \sum_{L_3} \langle Y_{L_1} u_{\ell_1 \nu_1} | R_{L_3} h_{L_3} | Y_{L_2} u_{\ell_2 \nu_2} \rangle =
248  * \sum_{L_3} \langle u_{\ell_1 \nu_1} | h_{L_3} | u_{\ell_2 \nu_2} \rangle
249  * \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
250  * \f]
251  */
252  template <spin_block_t sblock>
253  inline double_complex radial_integrals_sum_L3(int idxrf1__,
254  int idxrf2__,
255  std::vector<gaunt_L3<double_complex> > const& gnt__) const
256  {
257  double_complex zsum(0, 0);
258 
259  for (size_t i = 0; i < gnt__.size(); i++) {
260  switch (sblock) {
261  case spin_block_t::nm: {
262  /* just the Hamiltonian */
263  zsum += gnt__[i].coef * h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__);
264  break;
265  }
266  case spin_block_t::uu: {
267  /* h + Bz */
268  zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) +
269  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
270  break;
271  }
272  case spin_block_t::dd: {
273  /* h - Bz */
274  zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) -
275  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
276  break;
277  }
278  case spin_block_t::ud: {
279  /* Bx - i By */
280  zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
281  -b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
282  break;
283  }
284  case spin_block_t::du: {
285  /* Bx + i By */
286  zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
287  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
288  break;
289  }
290  }
291  }
292  return zsum;
293  }
