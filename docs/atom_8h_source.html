159  /// Generate radial Hamiltonian and effective magnetic field integrals
160  /** Hamiltonian operator has the following representation inside muffin-tins:
161  * \f[
162  * \hat H = -\frac{1}{2}\nabla^2 + \sum_{\ell m} V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) =
163  * \underbrace{-\frac{1}{2} \nabla^2+V_{00}(r)R_{00}}_{H_{s}(r)} +\sum_{\ell=1} \sum_{m=-\ell}^{\ell}
164  * V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) = \sum_{\ell m} \widetilde V_{\ell m}(r) R_{\ell m}(\hat {\bf r})
165  * \f]
166  * where
167  * \f[
168  * \widetilde V_{\ell m}(r) = \left\{ \begin{array}{ll}
169  * \frac{H_{s}(r)}{R_{00}} & \ell = 0 \\
170  * V_{\ell m}(r) & \ell > 0 \end{array} \right.
171  * \f]
172  */
173  inline void generate_radial_integrals(device_t pu__, Communicator const& comm__)
174  {
175  PROFILE("sirius::Atom::generate_radial_integrals");
176 
177  int lmmax = Utils::lmmax(lmax_pot_);
178  int nmtp = type().num_mt_points();
179  int nrf = type().indexr().size();
180  int num_mag_dims = type().parameters().num_mag_dims();
181 
182  if (comm__.size() != 1) {
183  TERMINATE("not yet mpi parallel");
184  }
185 
186  splindex<block> spl_lm(lmmax, comm__.size(), comm__.rank());
187 
188  auto l_by_lm = Utils::l_by_lm(lmax_pot_);
189 
190  h_radial_integrals_.zero();
191  if (num_mag_dims) {
192  b_radial_integrals_.zero();
193  }
194 
195  /* copy radial functions to spline objects */
196  std::vector< Spline<double> > rf_spline(nrf);
197  #pragma omp parallel for
198  for (int i = 0; i < nrf; i++) {
199  rf_spline[i] = Spline<double>(type().radial_grid());
200  for (int ir = 0; ir < nmtp; ir++) {
201  rf_spline[i][ir] = symmetry_class().radial_function(ir, i);
202  }
203  }
204 
205  /* copy effective potential components to spline objects */
206  std::vector<Spline<double>> v_spline(lmmax * (1 + num_mag_dims));
207  #pragma omp parallel for
208  for (int lm = 0; lm < lmmax; lm++) {
209  v_spline[lm] = Spline<double>(type().radial_grid());
210  for (int ir = 0; ir < nmtp; ir++) {
211  v_spline[lm][ir] = veff_(lm, ir);
212  }
213 
214  for (int j = 0; j < num_mag_dims; j++) {
215  v_spline[lm + (j + 1) * lmmax] = Spline<double>(type().radial_grid());
216  for (int ir = 0; ir < nmtp; ir++) {
217  v_spline[lm + (j + 1) * lmmax][ir] = beff_[j](lm, ir);
218  }
219  }
220  }
221 
222  /* interpolate potential multiplied by a radial function */
223  std::vector<Spline<double>> vrf_spline(lmmax * nrf * (1 + num_mag_dims));
224 
225  auto& idx_ri = type().idx_radial_integrals();
226 
227  mdarray<double, 1> result(idx_ri.size(1));
228 
229  if (pu__ == GPU) {
230  #ifdef __GPU
231  auto& rgrid = type().radial_grid();
232  auto& rf_coef = type().rf_coef();
233  auto& vrf_coef = type().vrf_coef();
234 
235  sddk::timer t1("sirius::Atom::generate_radial_integrals|interp");
236  #pragma omp parallel
237  {
238  //int tid = Platform::thread_id();
239  #pragma omp for
240  for (int i = 0; i < nrf; i++) {
241  rf_spline[i].interpolate();
242  std::memcpy(rf_coef.at<CPU>(0, 0, i), rf_spline[i].coeffs().at<CPU>(), nmtp * 4 * sizeof(double));
243  //cuda_async_copy_to_device(rf_coef.at<GPU>(0, 0, i), rf_coef.at<CPU>(0, 0, i), nmtp * 4 * sizeof(double), tid);
244  }
245  #pragma omp for
246  for (int i = 0; i < lmmax * (1 + num_mag_dims); i++) {
247  v_spline[i].interpolate();
248  }
249  }
250  rf_coef.async_copy_to_device();
251 
252  #pragma omp parallel for
253  for (int lm = 0; lm < lmmax; lm++) {
254  for (int i = 0; i < nrf; i++) {
255  for (int j = 0; j < num_mag_dims + 1; j++) {
256  int idx = lm + lmmax * i + lmmax * nrf * j;
257  vrf_spline[idx] = rf_spline[i] * v_spline[lm + j * lmmax];
258  std::memcpy(vrf_coef.at<CPU>(0, 0, idx), vrf_spline[idx].coeffs().at<CPU>(), nmtp * 4 * sizeof(double));
259  //cuda_async_copy_to_device(vrf_coef.at<GPU>(0, 0, idx), vrf_coef.at<CPU>(0, 0, idx), nmtp * 4 *sizeof(double), tid);
260  }
261  }
262  }
263  vrf_coef.copy_to_device();
264  t1.stop();
265 
266  result.allocate(memory_t::device);
267  sddk::timer t2("sirius::Atom::generate_radial_integrals|inner");
268  spline_inner_product_gpu_v3(idx_ri.at<GPU>(), (int)idx_ri.size(1), nmtp, rgrid.x().at<GPU>(), rgrid.dx().at<GPU>(),
269  rf_coef.at<GPU>(), vrf_coef.at<GPU>(), result.at<GPU>());
270  acc::sync();
271  if (type().parameters().control().print_performance_) {
272  double tval = t2.stop();
273  DUMP("spline GPU integration performance: %12.6f GFlops", 1e-9 * double(idx_ri.size(1)) * nmtp * 85 / tval);
274  }
275  result.copy_to_host();
276  result.deallocate_on_device();
277  #else
278  TERMINATE_NO_GPU
279  #endif
280  }
281  if (pu__ == CPU) {
282  sddk::timer t1("sirius::Atom::generate_radial_integrals|interp");
283  #pragma omp parallel
284  {
285  #pragma omp for
286  for (int i = 0; i < nrf; i++) {
287  rf_spline[i].interpolate();
288  }
289  #pragma omp for
290  for (int i = 0; i < lmmax * (1 + num_mag_dims); i++) {
291  v_spline[i].interpolate();
292  }
293 
294  #pragma omp for
295  for (int lm = 0; lm < lmmax; lm++) {
296  for (int i = 0; i < nrf; i++) {
297  for (int j = 0; j < num_mag_dims + 1; j++) {
298  vrf_spline[lm + lmmax * i + lmmax * nrf * j] = rf_spline[i] * v_spline[lm + j * lmmax];
299  }
300  }
301  }
302  }
303  t1.stop();
304 
305  sddk::timer t2("sirius::Atom::generate_radial_integrals|inner");
306  #pragma omp parallel for
307  for (int j = 0; j < (int)idx_ri.size(1); j++) {
308  result(j) = inner(rf_spline[idx_ri(0, j)], vrf_spline[idx_ri(1, j)], 2);
309  }
310  if (type().parameters().control().print_performance_) {
311  double tval = t2.stop();
312  DUMP("spline CPU integration performance: %12.6f GFlops", 1e-9 * double(idx_ri.size(1)) * nmtp * 85 / tval);
313  }
314  }
315 
316  int n{0};
317  for (int lm = 0; lm < lmmax; lm++) {
318  int l = l_by_lm[lm];
319 
320  for (int i2 = 0; i2 < type().indexr().size(); i2++) {
321  int l2 = type().indexr(i2).l;
322 
323  for (int i1 = 0; i1 <= i2; i1++) {
324  int l1 = type().indexr(i1).l;
325  if ((l + l1 + l2) % 2 == 0) {
326  if (lm) {
327  h_radial_integrals_(lm, i1, i2) = h_radial_integrals_(lm, i2, i1) = result(n++);
328  } else {
329  h_radial_integrals_(0, i1, i2) = symmetry_class().h_spherical_integral(i1, i2);
330  h_radial_integrals_(0, i2, i1) = symmetry_class().h_spherical_integral(i2, i1);
331  }
332  for (int j = 0; j < num_mag_dims; j++) {
333  b_radial_integrals_(lm, i1, i2, j) = b_radial_integrals_(lm, i2, i1, j) = result(n++);
334  }
335  }
336  }
337  }
338  }
339 
340  if (type().parameters().control().print_checksum_) {
341  DUMP("checksum(h_radial_integrals): %18.10f", h_radial_integrals_.checksum());
342  }
343  }

457  /** Compute the following kinds of sums for different spin-blocks of the Hamiltonian:
458  * \f[
459  * \sum_{L_3} \langle Y_{L_1} u_{\ell_1 \nu_1} | R_{L_3} h_{L_3} | Y_{L_2} u_{\ell_2 \nu_2} \rangle =
460  * \sum_{L_3} \langle u_{\ell_1 \nu_1} | h_{L_3} | u_{\ell_2 \nu_2} \rangle
461  * \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
462  * \f]
463  */
464  template <spin_block_t sblock>
465  inline double_complex radial_integrals_sum_L3(int idxrf1__,
466  int idxrf2__,
467  std::vector<gaunt_L3<double_complex> > const& gnt__) const
468  {
469  double_complex zsum(0, 0);
470 
471  for (size_t i = 0; i < gnt__.size(); i++) {
472  switch (sblock) {
473  case spin_block_t::nm: {
474  /* just the Hamiltonian */
475  zsum += gnt__[i].coef * h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__);
476  break;
477  }
478  case spin_block_t::uu: {
479  /* h + Bz */
480  zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) +
481  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
482  break;
483  }
484  case spin_block_t::dd: {
485  /* h - Bz */
486  zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) -
487  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
488  break;
489  }
490  case spin_block_t::ud: {
491  /* Bx - i By */
492  zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
493  -b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
494  break;
495  }
496  case spin_block_t::du: {
497  /* Bx + i By */
498  zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
499  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
500  break;
501  }
502  }
503  }
504  return zsum;
505  }
