58  Non_local_operator(Beta_projectors& beta__, device_t pu__) : beta_(beta__), pu_(pu__)
59  {
60  PROFILE("sirius::Non_local_operator::Non_local_operator");
61 
62  auto& uc = beta_.unit_cell();
63  packed_mtrx_offset_ = mdarray<int, 1>(uc.num_atoms());
64  packed_mtrx_size_ = 0;
65  for (int ia = 0; ia < uc.num_atoms(); ia++) {
66  int nbf = uc.atom(ia).mt_basis_size();
67  packed_mtrx_offset_(ia) = packed_mtrx_size_;
68  packed_mtrx_size_ += nbf * nbf;
69  }
70 
71  if (pu_ == GPU) {
72  packed_mtrx_offset_.allocate(memory_t::device);
73  packed_mtrx_offset_.template copy<memory_t::host, memory_t::device>();
74  }
75  }

95 template<>
96 inline void Non_local_operator<double_complex>::apply(int chunk__,
97  int ispn__,
98  wave_functions& op_phi__,
99  int idx0__,
100  int n__,
101  matrix<double_complex>& beta_phi__)
102 {
103  PROFILE("sirius::Non_local_operator::apply");
104 
105  if (is_null_) {
106  return;
107  }
108 
109  assert(op_phi__.pw_coeffs().num_rows_loc() == beta_.num_gkvec_loc());
110 
111  auto& beta_gk = beta_.pw_coeffs_a();
112  int num_gkvec_loc = beta_.num_gkvec_loc();
113  auto& bp_chunks = beta_.beta_projector_chunks();
114  int nbeta = bp_chunks(chunk__).num_beta_;
115 
116  if (static_cast<size_t>(nbeta * n__) > work_.size()) {
117  work_ = mdarray<double_complex, 1>(nbeta * n__);
118  if (pu_ == GPU) {
119  work_.allocate(memory_t::device);
120  }
121  }
122  /* compute O * <beta|phi> for atoms in a chunk */
123  #pragma omp parallel for
124  for (int i = 0; i < bp_chunks(chunk__).num_atoms_; i++) {
125  /* number of beta functions for a given atom */
126  int nbf = bp_chunks(chunk__).desc_(beta_desc_idx::nbf, i);
127  int offs = bp_chunks(chunk__).desc_(beta_desc_idx::offset, i);
128  int ia = bp_chunks(chunk__).desc_(beta_desc_idx::ia, i);
129  switch (pu_) {
130  case CPU: {
131  linalg<CPU>::gemm(0, 0, nbf, n__, nbf,
132  op_.at<CPU>(packed_mtrx_offset_(ia), ispn__), nbf,
133  beta_phi__.at<CPU>(offs, 0), nbeta,
134  work_.at<CPU>(offs), nbeta);
135 
136  break;
137  }
138  case GPU: {
139  #ifdef __GPU
140  linalg<GPU>::gemm(0, 0, nbf, n__, nbf,
141  op_.at<GPU>(packed_mtrx_offset_(ia), ispn__), nbf,
142  beta_phi__.at<GPU>(offs, 0), nbeta,
143  work_.at<GPU>(offs), nbeta,
144  omp_get_thread_num());
145  #endif
146  break;
147  }
148  }
149  }
150 
151  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
152  switch (pu_) {
153  case CPU: {
154  linalg<CPU>::gemm(0, 0, num_gkvec_loc, n__, nbeta,
155  linalg_const<double_complex>::one(),
156  beta_gk.at<CPU>(), num_gkvec_loc,
157  work_.at<CPU>(), nbeta,
158  linalg_const<double_complex>::one(),
159  op_phi__.pw_coeffs().prime().at<CPU>(0, idx0__), op_phi__.pw_coeffs().prime().ld());
160  break;
161  }
162  case GPU: {
163  #ifdef __GPU
164  /* wait for previous zgemms */
165  #pragma omp parallel
166  acc::sync_stream(omp_get_thread_num());
167 
168  linalg<GPU>::gemm(0, 0, num_gkvec_loc, n__, nbeta,
169  &linalg_const<double_complex>::one(),
170  beta_gk.at<GPU>(), beta_gk.ld(),
171  work_.at<GPU>(), nbeta,
172  &linalg_const<double_complex>::one(),
173  op_phi__.pw_coeffs().prime().at<GPU>(0, idx0__), op_phi__.pw_coeffs().prime().ld());
174  acc::sync_stream(-1);
175  #endif
176  break;
177  }
178  }
179 }

257  reinterpret_cast<double*>(beta_gk.at<GPU>()), 2 * num_gkvec_loc,
258  work_.at<GPU>(), nbeta,
259  &linalg_const<double>::one(),
260  reinterpret_cast<double*>(op_phi__.pw_coeffs().prime().at<GPU>(0, idx0__)), 2 * num_gkvec_loc);

268 template <typename T>
269 class D_operator: public Non_local_operator<T>
270 {
271  public:
272 
273  D_operator(Simulation_context const& ctx__, Beta_projectors& beta__)
274  : Non_local_operator<T>(beta__, ctx__.processing_unit())
275  {
276  this->op_ = mdarray<T, 2>(this->packed_mtrx_size_, ctx__.num_mag_dims() + 1);
277  this->op_.zero();
278  /* D-matrix is complex in non-collinear case */
279  if (ctx__.num_mag_dims() == 3) {
280  assert((std::is_same<T, double_complex>::value));
281  }
282 
283  auto& uc = this->beta_.unit_cell();
284 
285  for (int ia = 0; ia < uc.num_atoms(); ia++) {
286  int nbf = uc.atom(ia).mt_basis_size();
287  for (int xi2 = 0; xi2 < nbf; xi2++) {
288  for (int xi1 = 0; xi1 < nbf; xi1++) {
289  int idx = xi2 * nbf + xi1;
290  switch (ctx__.num_mag_dims()) {
291  case 3: {
292  double bx = uc.atom(ia).d_mtrx(xi1, xi2, 2);
293  double by = uc.atom(ia).d_mtrx(xi1, xi2, 3);
294  this->op_(this->packed_mtrx_offset_(ia) + idx, 2) = type_wrapper<T>::bypass(double_complex(bx, -by));
295  this->op_(this->packed_mtrx_offset_(ia) + idx, 3) = type_wrapper<T>::bypass(double_complex(bx, by));
296  }
297  case 1: {
298  double v = uc.atom(ia).d_mtrx(xi1, xi2, 0);
299  double bz = uc.atom(ia).d_mtrx(xi1, xi2, 1);
300  this->op_(this->packed_mtrx_offset_(ia) + idx, 0) = v + bz;
301  this->op_(this->packed_mtrx_offset_(ia) + idx, 1) = v - bz;
302  break;
303  }
304  case 0: {
305  this->op_(this->packed_mtrx_offset_(ia) + idx, 0) = uc.atom(ia).d_mtrx(xi1, xi2, 0);
306  break;
307  }
308  default: {
309  TERMINATE("wrong number of magnetic dimensions");
310  }
311  }
312  }
313  }
314  }
315 
316  if (this->pu_ == GPU) {
317  this->op_.allocate(memory_t::device);
318  this->op_.template copy<memory_t::host, memory_t::device>();
319  }
320  }
321 };

323 template <typename T>
324 class Q_operator: public Non_local_operator<T>
325 {
326  public:
327 
328  Q_operator(Simulation_context const& ctx__, Beta_projectors& beta__)
329  : Non_local_operator<T>(beta__, ctx__.processing_unit())
330  {
331  /* Q-operator is independent of spin */
332  this->op_ = mdarray<T, 2>(this->packed_mtrx_size_, 1);
333  this->op_.zero();
334 
335  auto& uc = this->beta_.unit_cell();
336  for (int ia = 0; ia < uc.num_atoms(); ia++) {
337  int iat = uc.atom(ia).type().id();
338  if (!uc.atom_type(iat).pp_desc().augment) {
339  continue;
340  }
341  int nbf = uc.atom(ia).mt_basis_size();
342  for (int xi2 = 0; xi2 < nbf; xi2++) {
343  for (int xi1 = 0; xi1 < nbf; xi1++) {
344  if (ctx__.unit_cell().atom_type(iat).pp_desc().augment) {
345  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 0) = ctx__.augmentation_op(iat).q_mtrx(xi1, xi2);
346  }
347  }
348  }
349  }
350  if (this->pu_ == GPU) {
351  this->op_.allocate(memory_t::device);
352  this->op_.template copy<memory_t::host, memory_t::device>();
353  }
354  }
