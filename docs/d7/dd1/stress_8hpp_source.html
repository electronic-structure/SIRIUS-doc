33 /// Stress tensor.
34 /** The following referenceces were particularly useful in the derivation of the stress tensor components:
35  * - Hutter, D. M. A. J. (2012). Ab Initio Molecular Dynamics (pp. 1–580).
36  * - Marx, D., & Hutter, J. (2000). Ab initio molecular dynamics: Theory and implementation.
37  * Modern Methods and Algorithms of Quantum Chemistry.
38  * - Knuth, F., Carbogno, C., Atalla, V., & Blum, V. (2015). All-electron formalism for total energy strain derivatives
39  * and stress tensor components for numeric atom-centered orbitals. Computer Physics Communications.
40  * - Willand, A., Kvashnin, Y. O., Genovese, L., Vázquez-Mayagoitia, Á., Deb, A. K., Sadeghi, A., et al. (2013).
41  * Norm-conserving pseudopotentials with chemical accuracy compared to all-electron calculations.
42  * The Journal of Chemical Physics, 138(10), 104109. http://doi.org/10.1103/PhysRevB.50.4327
43  * - Corso, A. D., & Resta, R. (1994). Density-functional theory of macroscopic stress: Gradient-corrected calculations
44  * for crystalline Se. Physical Review B.
45  *
46  * Stress tensor describes a reaction of crystall to a strain:
47  * \f[
48  * \sigma_{\mu \nu} = \frac{1}{\Omega} \frac{\partial E}{\partial \varepsilon_{\mu \nu}}
49  * \f]
50  * where \f$ \varepsilon_{\mu \nu} \f$ is a symmetric strain tensor which describes the infinitesimal deformation of a crystal:
51  * \f[
52  * r_{\mu} \rightarrow \sum_{\nu} (\delta_{\mu \nu} + \varepsilon_{\mu \nu}) r_{\nu}
53  * \f]
54  * The following expressions are helpful:
55  * - strain derivative of a general position vector component:
56  * \f[
57  * \frac{\partial r_{\tau}}{\partial \varepsilon_{\mu \nu}} = \delta_{\tau \mu} r_{\nu}
58  * \f]
59  * - strain derivative of the lengths of a general position vector:
60  * \f[
61  * \frac{\partial |{\bf r}|}{\partial \varepsilon_{\mu \nu}} = \sum_{\tau} \frac{\partial |{\bf r}|}{\partial r_{\tau}}
62  * \frac{\partial r_{\tau}}{\partial \varepsilon_{\mu \nu}} = \sum_{\tau} \frac{ r_{\tau}}{|{\bf r}|}
63  * \delta_{\tau \mu} r_{\nu} = \frac{r_{\mu} r_{\nu}}{|{\bf r}|}
64  * \f]
65  * - strain derivative of the unit cell volume:
66  * \f[
67  * \frac{\partial \Omega}{\partial \varepsilon_{\mu \nu}} = \delta_{\mu \nu} \Omega
68  * \f]
69  * - strain derivative of the inverse square root of the unit cell volume:
70  * \f[
71  * \frac{\partial}{\partial \varepsilon_{\mu \nu}} \frac{1}{\sqrt{\Omega}} = -\frac{1}{2}\frac{1}{\sqrt{\Omega}} \delta_{\mu \nu}
72  * \f]
73  * - strain derivative of a reciprocal vector:
74  * \f[
75  * \frac{\partial G_{\tau}}{\partial \varepsilon_{\mu \nu}} = -\delta_{\tau \nu} G_{\mu}
76  * \f]
77  * - strain derivative of the length of a reciprocal vector:
78  * \f[
79  * \frac{\partial |{\bf G}|}{\partial \varepsilon_{\mu \nu}} = \sum_{\tau} \frac{\partial |{\bf G}|}{\partial G_{\tau}}
80  * \frac{\partial G_{\tau}}{\partial \varepsilon_{\mu \nu}} = -\frac{1}{|{\bf G}|}G_{\nu}G_{\mu}
81  * \f]
82  * In the derivation of the stress tensor contributions it is important to know which variables are invariant under lattice
83  * distortion. This are:
84  * - scalar product of the real-space (in the Bravais lattice frame) and reciprocal (in the reciprocal lattice frame) vectors
85  * - normalized plane-wave coefficients of the wave-functions
86  * \f[
87  * \psi({\bf G}) = \frac{1}{\sqrt{\Omega}} \int e^{-i {\bf G}{\bf r}} \psi({\bf r}) d{\bf r}
88  * \f]
89  * - unnormalized plane-wave coefficients of the charge density
90  * \f[
91  * \tilde \rho({\bf G}) = \int e^{-i {\bf G}{\bf r}} \sum_{j} |\psi_j({\bf r})|^{2} d{\bf r}
92  * \f]
93  */

127  /// Non-local contribution to stress.
128  /** Energy contribution from the non-local part of pseudopotential:
129  * \f[
130  * E^{nl} = \sum_{i{\bf k}} \sum_{\alpha}\sum_{\xi \xi'}P_{\xi}^{\alpha,i{\bf k}} D_{\xi\xi'}^{\alpha}P_{\xi'}^{\alpha,i{\bf k}*}
131  * \f]
132  * where
133  * \f[
134  * P_{\xi}^{\alpha,i{\bf k}} = \langle \psi_{i{\bf k}} | \beta_{\xi}^{\alpha} \rangle =
135  * \frac{1}{\Omega} \sum_{\bf G} \langle \psi_{i{\bf k}} | {\bf G+k} \rangle \langle {\bf G+k} | \beta_{\xi}^{\alpha} \rangle =
136  * \sum_{\bf G} \psi_i^{*}({\bf G+k}) \beta_{\xi}^{\alpha}({\bf G+k})
137  * \f]
138  * where
139  * \f[
140  * \beta_{\xi}^{\alpha}({\bf G+k}) = \frac{1}{\sqrt{\Omega}} \int e^{-i({\bf G+k}){\bf r}} \beta_{\xi}^{\alpha}({\bf r}) d{\bf r} =
141  * \frac{4\pi}{\sqrt{\Omega}} e^{-i{\bf (G+k) r_{\alpha}}}(-i)^{\ell} R_{\ell m}(\theta_{G+k}, \phi_{G+k})
142  * \int \beta_{\ell}(r) j_{\ell}(|{\bf G+k}|r) r^2 dr
143  * \f]
144  * Contribution to stress tensor:
145  * \f[
146  * \sigma_{\mu \nu}^{nl} = \frac{1}{\Omega} \frac{\partial E^{nl}}{\partial \varepsilon_{\mu \nu}} =
147  * \frac{1}{\Omega} \sum_{i{\bf k}} \sum_{\xi \xi'} \Bigg(
148  * \frac{\partial P_{\xi}^{\alpha,i{\bf k}}}{\partial \varepsilon_{\mu \nu}} D_{\xi\xi'}^{\alpha}P_{\xi'}^{\alpha,i{\bf k}*} +
149  * P_{\xi}^{\alpha,i{\bf k}} D_{\xi\xi'}^{\alpha} \frac{\partial P_{\xi'}^{\alpha,i{\bf k}*}}{\partial \varepsilon_{\mu \nu}} \Bigg)
150  * \f]
151  * We need to compute strain derivatives of \f$ P_{\xi}^{\alpha,i{\bf k}} \f$:
152  * \f[
153  * \frac{\partial}{\partial \varepsilon_{\mu \nu}} P_{\xi}^{\alpha,i{\bf k}} =
154  * \sum_{\bf G} \psi_i^{*}({\bf G+k}) \frac{\partial}{\partial \varepsilon_{\mu \nu}} \beta_{\xi}^{\alpha}({\bf G+k})
155  * \f]
156  * First way to take strain derivative of beta-projectors (here and below \f$ {\bf G+k} = {\bf q} \f$):
157  * \f[
158  * \frac{\partial}{\partial \varepsilon_{\mu \nu}} \beta_{\xi}^{\alpha}({\bf q}) =
159  * -\frac{4\pi}{2\sqrt{\Omega}} \delta_{\mu \nu} e^{-i{\bf q r_{\alpha}}}(-i)^{\ell} R_{\ell m}(\theta_q, \phi_q)
160  * \int \beta_{\ell}(r) j_{\ell}(q r) r^2 dr + \\
161  * \frac{4\pi}{\sqrt{\Omega}} e^{-i{\bf q r_{\alpha}}}(-i)^{\ell} \frac{\partial R_{\ell m}(\theta_q, \phi_q)}{\partial \varepsilon_{\mu \nu}}
162  * \int \beta_{\ell}(r) j_{\ell}(q r) r^2 dr + \\
163  * \frac{4\pi}{\sqrt{\Omega}} e^{-i{\bf q r_{\alpha}}}(-i)^{\ell} R_{\ell m}(\theta_q, \phi_q)
164  * \int \beta_{\ell}(r) \frac{\partial j_{\ell}(q r)}{\partial \varepsilon_{\mu \nu}} r^2 dr = \\
165  * \frac{4\pi}{\sqrt{\Omega}} e^{-i{\bf q r_{\alpha}}}(-i)^{\ell} \Bigg[ \int \beta_{\ell}(r) j_{\ell}(q r) r^2 dr
166  * \Big(\frac{\partial R_{\ell m}(\theta_q, \phi_q)}{\partial \varepsilon_{\mu \nu}} - \frac{1}{2} R_{\ell m}(\theta_q, \phi_q) \delta_{\mu \nu}\Big) +
167  * R_{\ell m}(\theta_q, \phi_q) \int \beta_{\ell}(r) \frac{\partial j_{\ell}(q r)}{\partial \varepsilon_{\mu \nu}} r^2 dr
168  * \Bigg]
169  * \f]
170  *
171  * Strain derivative of the real spherical harmonics:
172  * \f[
173  * \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \varepsilon_{\mu \nu}} =
174  * \sum_{\tau} \frac{\partial R_{\ell m}(\theta, \phi)}{\partial q_{\tau}} \frac{\partial q_{\tau}}{\partial \varepsilon_{\mu \nu}} =
175  * -q_{\mu} \frac{\partial R_{\ell m}(\theta, \phi)}{\partial q_{\nu}}
176  * \f]
177  * For the derivatives of spherical harmonics over Cartesian components of vector please refer to the SHT::dRlm_dr function.
178  *
179  * Strain derivative of spherical Bessel function integral:
180  * \f[
181  * \int \beta_{\ell}(r) \frac{\partial j_{\ell}(qr) }{\partial \varepsilon_{\mu \nu}} r^2 dr =
182  * \int \beta_{\ell}(r) \frac{\partial j_{\ell}(qr) }{\partial q} \frac{-q_{\mu} q_{\nu}}{q} r^2 dr
183  * \f]
184  * The second way to compute strain derivative of beta-projectors is trough Gaunt coefficients:
185  * \f[
186  * \frac{\partial}{\partial \varepsilon_{\mu \nu}} \beta_{\xi}^{\alpha}({\bf q}) =
187  * -\frac{1}{2\sqrt{\Omega}} \delta_{\mu \nu} \int e^{-i{\bf q r}} \beta_{\xi}^{\alpha}({\bf r}) d{\bf r} +
188  * \frac{1}{\sqrt{\Omega}} \int i r_{\nu} q_{\mu} e^{-i{\bf q r}} \beta_{\xi}^{\alpha}({\bf r}) d{\bf r}
189  * \f]
190  * (second term comes from the strain derivative of \f$ e^{-i{\bf q r}} \f$). Remembering that \f$ {\bf r} \f$ is
191  * proportional to p-like real spherical harmonics, we can rewrite the second part of beta-projector derivative as:
192  * \f[
193  * \frac{1}{\sqrt{\Omega}} \int i r_{\nu} q_{\mu} e^{-i{\bf q r}} \beta_{\xi}^{\alpha}({\bf r}) d{\bf r} =
194  * \frac{1}{\sqrt{\Omega}} i q_{\mu} \int r \bar R_{1 \nu}(\theta, \phi) 4\pi \sum_{\ell_3 m_3} (-i)^{\ell_3} R_{\ell_3 m_3}(\theta_q, \phi_q)
195  * R_{\ell_3 m_3}(\theta, \phi) j_{\ell_3}(q r) \beta_{\ell_2}^{\alpha}(r) R_{\ell_2 m_2}(\theta, \phi) d{\bf r} = \\
196  * \frac{4 \pi}{\sqrt{\Omega}} i q_{\mu} \sum_{\ell_3 m_3} (-i)^{\ell_3} R_{\ell_3 m_3}(\theta_q, \phi_q)
197  * \langle \bar R_{1\nu} | R_{\ell_3 m_3} | R_{\ell_2 m_2} \rangle
198  * \int j_{\ell_3}(q r) \beta_{\ell_2}^{\alpha}(r) r^3 dr
199  * \f]
200  * where
201  * \f[
202  * \bar R_{1 x}(\theta, \phi) = -2 \sqrt{\frac{\pi}{3}} R_{11}(\theta, \phi) = \sin(\theta) \cos(\phi) \\
203  * \bar R_{1 y}(\theta, \phi) = -2 \sqrt{\frac{\pi}{3}} R_{1-1}(\theta,\phi) = \sin(\theta) \sin(\phi) \\
204  * \bar R_{1 z}(\theta, \phi) = 2 \sqrt{\frac{\pi}{3}} R_{10}(\theta, \phi) = \cos(\theta)
205  * \f]
206  */

208  inline void calc_stress_nonloc_aux()
209  {
210  PROFILE("sirius::Stress|nonloc");
211 
212  mdarray<double, 2> collect_result(9, ctx_.unit_cell().num_atoms());
213  collect_result.zero();
214 
215  stress_nonloc_.zero();
216 
217  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
218  int ik = kset_.spl_num_kpoints(ikloc);
219  auto kp = kset_[ik];
220  if (is_device_memory(ctx_.preferred_memory_t())) {
221  int nbnd = ctx_.num_bands();
222  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
223  /* allocate GPU memory */
224  kp->spinor_wave_functions().pw_coeffs(ispn).allocate(memory_t::device);
225  kp->spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, nbnd);
226  }
227  kp->spinor_wave_functions().preferred_memory_t(ctx_.preferred_memory_t());
228  }
229  Beta_projectors_strain_deriv bp_strain_deriv(ctx_, kp->gkvec(), kp->igk_loc());
230 
231  Non_local_functor<T> nlf(ctx_, bp_strain_deriv);
232 
233  nlf.add_k_point_contribution(*kp, collect_result);
234 
235  if (is_device_memory(ctx_.preferred_memory_t())) {
236  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
237  /* deallocate GPU memory */
238  kp->spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
239  }
240  kp->spinor_wave_functions().preferred_memory_t(memory_t::host);
241  }
242  }
243 
244  #pragma omp parallel
245  {
246  matrix3d<double> tmp_stress; // TODO: test pragma omp paralell for reduction(+:stress)
247 
248  #pragma omp for
249  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
250  for (int i = 0; i < 3; i++) {
251  for (int j = 0; j < 3; j++) {
252  tmp_stress(i, j) -= collect_result(j * 3 + i, ia);
253  }
254  }
255  }
256 
257  #pragma omp critical
258  stress_nonloc_ += tmp_stress;
259  }
260 
261  ctx_.comm().allreduce(&stress_nonloc_(0, 0), 9);
262 
263  stress_nonloc_ *= (1.0 / ctx_.unit_cell().omega());
264 
265  symmetrize(stress_nonloc_);
266  }

268  inline void symmetrize(matrix3d<double>& mtrx__) const
269  {
270  if (!ctx_.use_symmetry()) {
271  return;
272  }
273 
274  matrix3d<double> result;
275 
276  for (int i = 0; i < ctx_.unit_cell().symmetry().num_mag_sym(); i++) {
277  auto R = ctx_.unit_cell().symmetry().magnetic_group_symmetry(i).spg_op.rotation;
278  result = result + transpose(R) * mtrx__ * R;
279  }
280 
281  mtrx__ = result * (1.0 / ctx_.unit_cell().symmetry().num_mag_sym());
282 
283  std::vector<std::array<int, 2>> idx = {{0, 1}, {0, 2}, {1, 2}};
284  for (auto e: idx) {
285  mtrx__(e[0], e[1]) = mtrx__(e[1], e[0]) = 0.5 * (mtrx__(e[0], e[1]) + mtrx__(e[1], e[0]));
286  }

303  /// Local potential contribution to stress.
304  /** Energy contribution from the local part of pseudopotential:
305  * \f[
306  * E^{loc} = \int \rho({\bf r}) V^{loc}({\bf r}) d{\bf r} = \frac{1}{\Omega} \sum_{\bf G} \langle \rho | {\bf G} \rangle \langle {\bf G}| V^{loc} \rangle =
307  * \frac{1}{\Omega} \sum_{\bf G} \tilde \rho^{*}({\bf G}) \tilde V^{loc}({\bf G})
308  * \f]
309  * where
310  * \f[
311  * \tilde \rho({\bf G}) = \langle {\bf G} | \rho \rangle = \int e^{-i{\bf Gr}}\rho({\bf r}) d {\bf r}
312  * \f]
313  * and
314  * \f[
315  * \tilde V^{loc}({\bf G}) = \langle {\bf G} | V^{loc} \rangle = \int e^{-i{\bf Gr}}V^{loc}({\bf r}) d {\bf r}
316  * \f]
317  * Using the expression for \f$ \tilde V^{loc}({\bf G}) \f$, the local contribution to the total energy is rewritten as
318  * \f[
319  * E^{loc} = \frac{1}{\Omega} \sum_{\bf G} \tilde \rho^{*}({\bf G}) \sum_{\alpha} e^{-{\bf G\tau}_{\alpha}} 4 \pi
320  * \int V_{\alpha}^{loc}(r)\frac{\sin(Gr)}{Gr} r^2 dr =
321  * \frac{4\pi}{\Omega}\sum_{\bf G}\tilde \rho^{*}({\bf G})\sum_{\alpha} e^{-{\bf G\tau}_{\alpha}}
322  * \Bigg( \int \Big(V_{\alpha}(r) r + Z_{\alpha}^p {\rm erf}(r) \Big) \frac{\sin(Gr)}{G} dr - Z_{\alpha}^p \frac{e^{-\frac{G^2}{4}}}{G^2} \Bigg)
323  * \f]
324  * (see \link sirius::Potential::generate_local_potential \endlink for details).
325  *
326  * Contribution to stress tensor:
327  * \f[
328  * \sigma_{\mu \nu}^{loc} = \frac{1}{\Omega} \frac{\partial E^{loc}}{\partial \varepsilon_{\mu \nu}} =
329  * \frac{1}{\Omega} \frac{-1}{\Omega} \delta_{\mu \nu} \sum_{\bf G}\tilde \rho^{*}({\bf G}) \tilde V^{loc}({\bf G}) +
330  * \frac{4\pi}{\Omega^2} \sum_{\bf G}\tilde \rho^{*}({\bf G}) \sum_{\alpha} e^{-{\bf G\tau}_{\alpha}}
331  * \Bigg( \int \Big(V_{\alpha}(r) r + Z_{\alpha}^p {\rm erf}(r) \Big) \Big( \frac{r \cos (G r)}{G}-\frac{\sin (G r)}{G^2} \Big)
332  * \Big( -\frac{G_{\mu}G_{\nu}}{G} \Big) dr - Z_{\alpha}^p \Big( -\frac{e^{-\frac{G^2}{4}}}{2 G}-\frac{2 e^{-\frac{G^2}{4}}}{G^3} \Big)
333  * \Big( -\frac{G_{\mu}G_{\nu}}{G} \Big) \Bigg) = \\
334  * -\delta_{\mu \nu} \sum_{\bf G}\rho^{*}({\bf G}) V^{loc}({\bf G}) + \sum_{\bf G} \rho^{*}({\bf G}) \Delta V^{loc}({\bf G}) G_{\mu}G_{\nu}
335  * \f]
336  * where \f$ \Delta V^{loc}({\bf G}) \f$ is built from the following radial integrals:
337  * \f[
338  * \int \Big(V_{\alpha}(r) r + Z_{\alpha}^p {\rm erf}(r) \Big) \Big( \frac{\sin (G r)}{G^3} - \frac{r \cos (G r)}{G^2}\Big) dr -
339  * Z_{\alpha}^p \Big( \frac{e^{-\frac{G^2}{4}}}{2 G^2} + \frac{2 e^{-\frac{G^2}{4}}}{G^4} \Big)
340  * \f]
341  */
342  inline matrix3d<double> calc_stress_vloc()
343  {
344  PROFILE("sirius::Stress|vloc");
345 
346  stress_vloc_.zero();
347 
348  auto& ri_vloc = ctx_.vloc_ri();
349  auto& ri_vloc_dg = ctx_.vloc_ri_djl();
350 
351  auto v = ctx_.make_periodic_function<index_domain_t::local>([&](int iat, double g)
352  {
353  return ri_vloc.value(iat, g);
354  });
355 
356  auto dv = ctx_.make_periodic_function<index_domain_t::local>([&](int iat, double g)
357  {
358  return ri_vloc_dg.value(iat, g);
359  });
360 
361  double sdiag{0};
362 
363  int ig0 = (ctx_.comm().rank() == 0) ? 1 : 0;
364  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
365 
366  auto G = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
367 
368  for (int mu: {0, 1, 2}) {
369  for (int nu: {0, 1, 2}) {
370  stress_vloc_(mu, nu) += std::real(std::conj(density_.rho().f_pw_local(igloc)) * dv[igloc]) * G[mu] * G[nu];
371  }
372  }
373 
374  sdiag += std::real(std::conj(density_.rho().f_pw_local(igloc)) * v[igloc]);
375  }
376 
377  if (ctx_.gvec().reduced()) {
378  stress_vloc_ *= 2;
379  sdiag *= 2;
380  }
381  if (ctx_.comm().rank() == 0) {
382  sdiag += std::real(std::conj(density_.rho().f_pw_local(0)) * v[0]);
383  }
384 
385  for (int mu: {0, 1, 2}) {
386  stress_vloc_(mu, mu) -= sdiag;
387  }
388 
389  ctx_.comm().allreduce(&stress_vloc_(0, 0), 9);
390 
391  symmetrize(stress_vloc_);
392 
393  return stress_vloc_;
394  }

401  /// Hartree energy contribution to stress.
402  /** Hartree energy:
403  * \f[
404  * E^{H} = \frac{1}{2} \int_{\Omega} \rho({\bf r}) V^{H}({\bf r}) d{\bf r} =
405  * \frac{1}{2} \frac{1}{\Omega} \sum_{\bf G} \langle \rho | {\bf G} \rangle \langle {\bf G}| V^{H} \rangle =
406  * \frac{2 \pi}{\Omega} \sum_{\bf G} \frac{|\tilde \rho({\bf G})|^2}{G^2}
407  * \f]
408  * where
409  * \f[
410  * \langle {\bf G} | \rho \rangle = \int e^{-i{\bf Gr}}\rho({\bf r}) d {\bf r} = \tilde \rho({\bf G})
411  * \f]
412  * and
413  * \f[
414  * \langle {\bf G} | V^{H} \rangle = \int e^{-i{\bf Gr}}V^{H}({\bf r}) d {\bf r} = \frac{4\pi}{G^2} \tilde \rho({\bf G})
415  * \f]
416  *
417  * Hartree energy contribution to stress tensor:
418  * \f[
419  * \sigma_{\mu \nu}^{H} = \frac{1}{\Omega} \frac{\partial E^{H}}{\partial \varepsilon_{\mu \nu}} =
420  * \frac{1}{\Omega} 2\pi \Big( \big( \frac{\partial}{\partial \varepsilon_{\mu \nu}} \frac{1}{\Omega} \big)
421  * \sum_{{\bf G}} \frac{|\tilde \rho({\bf G})|^2}{G^2} +
422  * \frac{1}{\Omega} \sum_{{\bf G}} |\tilde \rho({\bf G})|^2 \frac{\partial}{\partial \varepsilon_{\mu \nu}} \frac{1}{G^2} \Big) = \\
423  * \frac{1}{\Omega} 2\pi \Big( -\frac{1}{\Omega} \delta_{\mu \nu} \sum_{{\bf G}} \frac{|\tilde \rho({\bf G})|^2}{G^2} +
424  * \frac{1}{\Omega} \sum_{\bf G} |\tilde \rho({\bf G})|^2 \sum_{\tau} \frac{-2 G_{\tau}}{G^4} \frac{\partial G_{\tau}}{\partial \varepsilon_{\mu \nu}} \Big) = \\
425  * 2\pi \sum_{\bf G} \frac{|\rho({\bf G})|^2}{G^2} \Big( -\delta_{\mu \nu} + \frac{2}{G^2} G_{\nu} G_{\mu} \Big)
426  * \f]
427  */
428  inline matrix3d<double> calc_stress_har()
429  {
430  PROFILE("sirius::Stress|har");
431 
432  stress_har_.zero();

490  * \f[

492  * \Big| \sum_{\alpha} Z_{\alpha} e^{-i{\bf r}_{\alpha}{\bf G}} \Big|^2 =
493  * -\frac{2\pi}{\Omega^2} \delta_{\mu \nu} \sum_{{\bf G}} \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \Big|
