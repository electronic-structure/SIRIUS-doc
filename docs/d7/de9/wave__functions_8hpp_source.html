51 /// Wave-functions representation.
52 /** Wave-functions consist of two parts: plane-wave part and mufin-tin part. Both are the matrix_storage objects
53  * with the slab distribution. Wave-functions have one or two spin components. In case of collinear magnetism
54  * each component represents a pure (up- or dn-) spinor state and they are independent. In non-collinear case
55  * the two components represent a full spinor state.
56  *
57  * In case of collinear magnetism we can work with auxiliary scalar wave-functions and update up- or dn- components
58  * of pure spinor wave-functions independently. We can also apply uu or dd block of Hamiltonian. In this case it is
59  * reasonable to implement the following convention: for scalar wave-function (num_sc = 1) it's value is returned
60  * for any spin index (ispn = 0 or ispn = 1).
61  */

127  inline mdarray<double, 1> sumsqr(device_t pu__, int ispn__, int n__) const
128  {
129  mdarray<double, 1> s(n__, memory_t::host, "sumsqr");
130  s.zero();
131  if (pu__ == GPU) {
132  s.allocate(memory_t::device);
133  s.zero<memory_t::device>();
134  }
135 
136  for (int is = s0(ispn__); is <= s1(ispn__); is++) {
137  switch (pu__) {
138  case CPU: {
139  #pragma omp parallel for
140  for (int i = 0; i < n__; i++) {
141  for (int ig = 0; ig < pw_coeffs(is).num_rows_loc(); ig++) {
142  s[i] += (std::pow(pw_coeffs(is).prime(ig, i).real(), 2) + std::pow(pw_coeffs(is).prime(ig, i).imag(), 2));
143  }
144  if (gkvecp_.gvec().reduced()) {
145  if (comm_.rank() == 0) {
146  s[i] = 2 * s[i] - std::pow(pw_coeffs(is).prime(0, i).real(), 2);
147  } else {
148  s[i] *= 2;
149  }
150  }
151  if (has_mt()) {
152  for (int j = 0; j < mt_coeffs(is).num_rows_loc(); j++) {
153  s[i] += (std::pow(mt_coeffs(is).prime(j, i).real(), 2) + std::pow(mt_coeffs(is).prime(j, i).imag(), 2));
154  }
155  }
156  }
157  break;
158  }
159  case GPU: {
160  #ifdef __GPU
161  add_square_sum_gpu(pw_coeffs(is).prime().at<GPU>(), pw_coeffs(is).num_rows_loc(), n__,
162  gkvecp_.gvec().reduced(), comm_.rank(), s.at<GPU>());
163  if (has_mt()) {
164  add_square_sum_gpu(mt_coeffs(is).prime().at<GPU>(), mt_coeffs(is).num_rows_loc(), n__,
165  0, comm_.rank(), s.at<GPU>());
166  }
167  #endif
168  break;
169  }
170  }
171  }
172  if (pu__ == GPU) {
173  s.copy<memory_t::device, memory_t::host>();
174  }
175  comm_.allreduce(s.at<CPU>(), n__);
176  return std::move(s);
177  }

328  /// Copy values from another wave-function.
329  /** \param [in] pu Type of processging unit which copies data.
330  * \param [in] n Number of wave-functions to copy.
331  * \param [in] src Input wave-function.
332  * \param [in] ispn Spin component on source wave-functions.
333  * \param [in] i0 Starting index of wave-functions in src.
334  * \param [in] jspn Spin component on destination wave-functions.
335  * \param [in] j0 Starting index of wave-functions in destination. */
336  inline void copy_from(device_t pu__,
337  int n__,
338  Wave_functions const& src__,
339  int ispn__,
340  int i0__,
341  int jspn__,
342  int j0__)
343  {
344  assert(ispn__ == 0 || ispn__ == 1);
345  assert(jspn__ == 0 || jspn__ == 1);
346 
347  int ngv = pw_coeffs(jspn__).num_rows_loc();
348  int nmt = has_mt() ? mt_coeffs(jspn__).num_rows_loc() : 0;
349 
350  switch (pu__) {
351  case CPU: {
352  /* copy PW part */
353  std::copy(src__.pw_coeffs(ispn__).prime().at<CPU>(0, i0__),
354  src__.pw_coeffs(ispn__).prime().at<CPU>(0, i0__) + ngv * n__,
355  pw_coeffs(jspn__).prime().at<CPU>(0, j0__));
356  /* copy MT part */
357  if (has_mt()) {
358  std::copy(src__.mt_coeffs(ispn__).prime().at<CPU>(0, i0__),
359  src__.mt_coeffs(ispn__).prime().at<CPU>(0, i0__) + nmt * n__,
360  mt_coeffs(jspn__).prime().at<CPU>(0, j0__));
361  }
362  break;
363  }
364  case GPU: {
365  #ifdef __GPU
366  /* copy PW part */
367  acc::copy(pw_coeffs(jspn__).prime().at<GPU>(0, j0__),
368  src__.pw_coeffs(ispn__).prime().at<GPU>(0, i0__),
369  ngv * n__);
370  /* copy MT part */
371  if (has_mt()) {
372  acc::copy(mt_coeffs(jspn__).prime().at<GPU>(0, j0__),
373  src__.mt_coeffs(ispn__).prime().at<GPU>(0, i0__),
374  nmt * n__);
375  }
376  #endif
377  break;
378  }
379  }
380  }

382  /// Compute the checksum of the spin-components.
383  /** Checksum of the n wave-function spin components is computed starting from i0.
384  * Only plane-wave coefficients are considered. */
385  inline double_complex checksum_pw(device_t pu__,
386  int ispn__,
387  int i0__,
388  int n__)
389  {
390  assert(n__ != 0);
391  double_complex cs(0, 0);
392  for (int s = s0(ispn__); s <= s1(ispn__); s++) {
393  cs += pw_coeffs(s).checksum(pu__, i0__, n__);
394  }
395  comm_.allreduce(&cs, 1);
396  return cs;
397  }

399  /// Checksum of muffin-tin coefficients.
400  inline double_complex checksum_mt(device_t pu__, int ispn__, int i0__, int n__)
401  {
402  assert(n__ != 0);
403  double_complex cs(0, 0);
404  if (!has_mt()) {
405  return cs;
406  }
407  for (int s = s0(ispn__); s <= s1(ispn__); s++) {
408  cs += mt_coeffs(s).checksum(pu__, i0__, n__);
409  }
410  comm_.allreduce(&cs, 1);
411  return cs;
412  }

532  /// Compute the overlap matrix between two sets of wave functions
533  /// and return the result in an 2D array
534  /// \f[ O_{ij} = \left< phi_i | psi_j\right> \f]
535 
536  template <typename T> inline mdarray<T, 2> overlap(device_t pu,
537  Wave_functions &phi_,
538  const int start_idx_psi_,
539  const int num_of_wf_psi_,
540  const int start_idx_phi_,
541  const int num_of_wf_phi_)
542  {
543  assert(((start_idx_phi_ + num_of_wf_phi_) <= phi_.num_wf())&&(num_of_wf_phi_ > 0));
544  assert(start_idx_phi_ >= 0);
545  assert(((start_idx_psi_ + num_of_wf_psi_) <= this->num_wf())&&(num_of_wf_psi_ > 0));
546  assert(start_idx_psi_ >= 0);
547 
548  mdarray <T, 2> dm_(num_of_wf_psi_, num_of_wf_phi_);
549 
550  dm_.zero();
551 
552  if (this->num_sc() != phi_.num_sc()) {
553  TERMINATE("Inconsistency between the two wave functions !!!!. they do not have the same number of components");
554  }
555 
556  if (pu == CPU) {
557  for (int ispn = 0; ispn < this->num_sc(); ispn++) {
558  linalg<CPU>::gemm(2, 0,
559  num_of_wf_psi_,
560  num_of_wf_phi_,
561  this->pw_coeffs(ispn).num_rows_loc(),
562  linalg_const<T>::one(),
563  this->pw_coeffs(ispn).prime().at<CPU>(0, start_idx_psi_),
564  this->pw_coeffs(ispn).prime().ld(),
565  phi_.pw_coeffs(ispn).prime().at<CPU>(0, start_idx_phi_),
566  phi_.pw_coeffs(ispn).prime().ld(),
567  linalg_const<T>::one(),
568  dm_.template at<CPU>(),
569  dm_.ld());
570 
571  if (this->has_mt()) {
572  linalg<CPU>::gemm(2, 0,
573  num_of_wf_psi_,
574  num_of_wf_phi_,
575  this->mt_coeffs(ispn).num_rows_loc(),
576  linalg_const<T>::one(),
577  this->mt_coeffs(ispn).prime().at<CPU>(0, start_idx_psi_),
578  this->mt_coeffs(ispn).prime().ld(),
579  phi_.mt_coeffs(ispn).prime().at<CPU>(0, start_idx_phi_),
580  phi_.mt_coeffs(ispn).prime().ld(),
581  linalg_const<T>::one(),
582  dm_.template at<CPU>(),
583  dm_.ld());
584  }
585  }
586  } else {
587 #ifdef __GPU
588  dm_.allocate(memory_t::device);
589  for (int ispn = 0; ispn < this->num_sc(); ispn++) {
590 
591  this->pw_coeffs(ispn).prime().allocate(memory_t::device);
592  this->pw_coeffs(ispn).copy_to_device(0, this->num_wf());
593 
594  phi_.pw_coeffs(ispn).prime().allocate(memory_t::device);
595  phi_.pw_coeffs(ispn).copy_to_device(0, phi_.num_wf());
596 
597  linalg<GPU>::gemm(2, 0,
598  num_of_wf_psi_,
599  num_of_wf_phi_,
600  this->pw_coeffs(ispn).num_rows_loc(),
601  &linalg_const<T>::one(),
602  this->pw_coeffs(ispn).prime().at<GPU>(0, start_idx_psi_),
603  this->pw_coeffs(ispn).prime().ld(),
604  phi_.pw_coeffs(ispn).prime().at<GPU>(0, start_idx_phi_),
605  phi_.pw_coeffs(ispn).prime().ld(),
606  &linalg_const<T>::one(),
607  dm_.template at<GPU>(),
608  dm_.ld());
609 
610  phi_.pw_coeffs(ispn).prime().deallocate(memory_t::device);
