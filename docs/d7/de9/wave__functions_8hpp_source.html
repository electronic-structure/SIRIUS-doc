48 /// Helper class to wrap spin index range.
49 /** Depending on the collinear or non-collinear case, the spin index range of the wave-functions is either
50  * [0, 0] or [1, 1] (trivial cases of single spin channel) or [0, 1] (spinor wave-functions). */
51 class spin_range : public std::vector<int>
52 {
53  private:
54  int idx_;
55  public:
56  explicit spin_range(int idx__)
57  : idx_(idx__)
58  {
59  if (!(idx_ == 0 || idx_ == 1 || idx_ == 2)) {
60  throw std::runtime_error("wrong spin index");
61  }
62  if (idx_ == 2) {
63  this->reserve(2);
64  this->push_back(0);
65  this->push_back(1);
66  } else {
67  this->reserve(1);
68  this->push_back(idx_);
69  }
70  }
71  inline int operator()() const
72  {
73  return idx_;
74  }
75 };

79 /// Wave-functions representation.
80 /** Wave-functions consist of two parts: plane-wave part and mufin-tin part. Both are the matrix_storage objects
81  * with the slab distribution. Wave-functions have one or two spin components. In case of collinear magnetism
82  * each component represents a pure (up- or dn-) spinor state and they are independent. In non-collinear case
83  * the two components represent a full spinor state.
84  *
85  * In case of collinear magnetism we can work with auxiliary scalar wave-functions and update up- or dn- components
86  * of pure spinor wave-functions independently. We can also apply uu or dd block of Hamiltonian. In this case it is
87  * reasonable to implement the following convention: for scalar wave-function (num_sc = 1) it's value is returned
88  * for any spin index (ispn = 0 or ispn = 1).
89  *
90  * Example below shows how the wave-functions are used:
91 
92  \code{.cpp}
93  // alias for wave-functions
94  auto& psi = kp__->spinor_wave_functions();
95  // create hpsi
96  Wave_functions hpsi(kp__->gkvec_partition(), ctx_.num_bands(), num_sc);
97  // create hpsi
98  Wave_functions spsi(kp__->gkvec_partition(), ctx_.num_bands(), num_sc);
99 
100  // if preferred memory is on GPU
101  if (is_device_memory(ctx_.preferred_memory_t())) {
102  // alias for memory pool
103  auto& mpd = ctx_.mem_pool(memory_t::device);
104  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
105  // allocate GPU memory
106  psi.pw_coeffs(ispn).allocate(mpd);
107  // copy to GPU
108  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
109  }
110  // set the preferred memory type
111  psi.preferred_memory_t(ctx_.preferred_memory_t());
112  // allocate hpsi and spsi on GPU
113  for (int i = 0; i < num_sc; i++) {
114  hpsi.pw_coeffs(i).allocate(mpd);
115  spsi.pw_coeffs(i).allocate(mpd);
116  }
117  // set preferred memory for hpsi
118  hpsi.preferred_memory_t(ctx_.preferred_memory_t());
119  // set preferred memory for spsi
120  spsi.preferred_memory_t(ctx_.preferred_memory_t());
121  }
122  // prepare beta projectors
123  kp__->beta_projectors().prepare();
124  for (int ispin_step = 0; ispin_step < ctx_.num_spin_dims(); ispin_step++) {
125  if (nc_mag) {
126  // apply Hamiltonian and S operators to both components of wave-functions
127  H__.apply_h_s<T>(kp__, 2, 0, ctx_.num_bands(), psi, &hpsi, &spsi);
128  } else {
129  // apply Hamiltonian and S operators to a single components of wave-functions
130  H__.apply_h_s<T>(kp__, ispin_step, 0, ctx_.num_bands(), psi, &hpsi, &spsi);
131  }
132 
133  for (int ispn = 0; ispn < num_sc; ispn++) {
134  // copy to host if needed
135  if (is_device_memory(ctx_.preferred_memory_t())) {
136  hpsi.copy_to(ispn, memory_t::host, 0, ctx_.num_bands());
137  spsi.copy_to(ispn, memory_t::host, 0, ctx_.num_bands());
138  }
139  }
140  // do something with hpsi and spsi
141  }
142  // free beta-projectors
143  kp__->beta_projectors().dismiss();
144  if (is_device_memory(ctx_.preferred_memory_t())) {
145  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
146  // deallocate wave-functions on GPU
147  psi.pw_coeffs(ispn).deallocate(memory_t::device);
148  }
149  // set preferred memory to CPU
150  psi.preferred_memory_t(memory_t::host);
151  }
152  \endcode
153  */
