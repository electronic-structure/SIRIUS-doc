25 /** \param [in] kp Pointer to k-point.
26  * \param [in] ispn Index of spin.
27  * \param [in] N Starting index of wave-functions.
28  * \param [in] n Number of wave-functions to which H and S are applied.
29  * \param [in] phi Input wave-functions [storage: CPU && GPU].
30  * \param [out] hphi Hamiltonian, applied to wave-functions [storage: CPU || GPU].
31  * \param [out] sphi Overlap operator, applied to wave-functions [storage: CPU || GPU].
32  *
33  * In non-collinear case (ispn = 2) the Hamiltonian and S operator are applied to both components of spinor
34  * wave-functions. Otherwise they are applied to a single component.
35  */
36 template <typename T>
37 void Hamiltonian::apply_h_s(K_point* kp__,
38  int ispn__,
39  int N__,
40  int n__,
41  Wave_functions& phi__,
42  Wave_functions* hphi__,
43  Wave_functions* sphi__) const
44 {
45  PROFILE("sirius::Hamiltonian::apply_h_s");
46 
47  if (hphi__ != nullptr) {
48  if (phi__.num_sc() != hphi__->num_sc()) {
49  TERMINATE("wrong number of spin components");
50  }
51  }
52 
53  if (sphi__ != nullptr) {
54  if (phi__.num_sc() != sphi__->num_sc()) {
55  TERMINATE("wrong number of spin components");
56  }
57  }
58 
59  double t1 = -omp_get_wtime();
60 
61  if (hphi__ != nullptr) {
62  /* apply local part of Hamiltonian */
63  local_op_->apply_h(ispn__, phi__, *hphi__, N__, n__);
64  }
65 
66  t1 += omp_get_wtime();
67 
68  if (kp__->comm().rank() == 0 && ctx_.control().print_performance_) {
69  printf("hloc performace: %12.6f bands/sec", n__ / t1);
70  }
71 
72  int nsc = (ispn__ == 2) ? 2 : 1;
73 
74  if (ctx_.control().print_checksum_ && (hphi__ != nullptr)) {
75  for (int ispn = 0; ispn < nsc; ispn++) {
76  auto cs1 = phi__.checksum(get_device_t(phi__.preferred_memory_t()), ispn, N__, n__);
77  auto cs2 = hphi__->checksum(get_device_t(hphi__->preferred_memory_t()), ispn, N__, n__);
78  if (kp__->comm().rank() == 0) {
79  std::stringstream s;
80  s << "phi_" << ispn;
81  utils::print_checksum(s.str(), cs1);
82  s.str("");
83  s << "hphi_" << ispn;
84  utils::print_checksum(s.str(), cs2);
85  }
86  }
87  }
88 
89  /* set intial sphi */
90  for (int ispn = 0; (ispn < nsc) && (sphi__ != nullptr); ispn++) {
91  sphi__->copy_from(phi__, n__, ispn, N__, ispn, N__);
92  }
93 
94  /* return if there are no beta-projectors */
95  if (!ctx_.unit_cell().mt_lo_basis_size()) {
96  return;
97  }
98 
99  for (int i = 0; i < kp__->beta_projectors().num_chunks(); i++) {
100  /* generate beta-projectors for a block of atoms */
101  kp__->beta_projectors().generate(i);
102  /* non-collinear case */
103  if (ispn__ == 2) {
104  for (int ispn = 0; ispn < 2; ispn++) {
105 
106  auto beta_phi = kp__->beta_projectors().inner<T>(i, phi__, ispn, N__, n__);
107 
108  if (hphi__) {
109  /* apply diagonal spin blocks */
110  D<T>().apply(i, ispn, *hphi__, N__, n__, kp__->beta_projectors(), beta_phi);
111  /* apply non-diagonal spin blocks */
112  /* xor 3 operator will map 0 to 3 and 1 to 2 */
113  D<T>().apply(i, ispn ^ 3, *hphi__, N__, n__, kp__->beta_projectors(), beta_phi);
114  }
115 
116  if (sphi__) {
117  /* apply Q operator (diagonal in spin) */
118  Q<T>().apply(i, ispn, *sphi__, N__, n__, kp__->beta_projectors(), beta_phi);
119  /* apply non-diagonal spin blocks */
120  if (ctx_.so_correction()) {
121  Q<T>().apply(i, ispn ^ 3, *sphi__, N__, n__, kp__->beta_projectors(), beta_phi);
122  }
123  }
124  }
125  } else { /* non-magnetic or collinear case */
126 
127  auto beta_phi = kp__->beta_projectors().inner<T>(i, phi__, ispn__, N__, n__);
128  if (hphi__) {
129  D<T>().apply(i, ispn__, *hphi__, N__, n__, kp__->beta_projectors(), beta_phi);
130  }
131 
132  if (sphi__) {
133  Q<T>().apply(i, ispn__, *sphi__, N__, n__, kp__->beta_projectors(), beta_phi);
134  }
135  }
136  }
137 
138  /* apply the hubbard potential if relevant */
139  if (ctx_.hubbard_correction() && !ctx_.gamma_point() && (hphi__ != NULL)) {
140 
141  // copy the hubbard wave functions on GPU (if needed) and
142  // return afterwards, or if they are not already calculated
143  // compute the wave functions and copy them on GPU (if needed)
144 
145  this->U().generate_atomic_orbitals(*kp__, Q<T>());
146 
147  // Apply the hubbard potential and deallocate the hubbard wave
148  // functions on GPU (if needed)
149  this->U().apply_hubbard_potential(*kp__, ispn__, N__, n__, phi__, *hphi__);
150 
151  if (ctx_.processing_unit() == device_t::GPU) {
152  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
153  kp__->hubbard_wave_functions().deallocate(ispn, memory_t::device);
154  }
155  }
156  }
157 
158  if ((ctx_.control().print_checksum_) && (hphi__ != nullptr) && (sphi__ != nullptr)) {
159  for (int ispn = 0; ispn < nsc; ispn++) {
160  auto cs1 = hphi__->checksum(get_device_t(hphi__->preferred_memory_t()), ispn, N__, n__);
161  auto cs2 = sphi__->checksum(get_device_t(sphi__->preferred_memory_t()), ispn, N__, n__);
162  if (kp__->comm().rank() == 0) {
163  std::stringstream s;
164  s << "hphi_" << ispn;
165  utils::print_checksum(s.str(), cs1);
166  s.str("");
167  s << "sphi_" << ispn;
168  utils::print_checksum(s.str(), cs2);
169  }
170  }
171  }
172 }

174 inline void Hamiltonian::apply_fv_h_o(K_point* kp__,
175  bool apw_only__,
176  bool phi_is_lo__,
177  int N__,
178  int n__,
179  Wave_functions& phi__,
180  Wave_functions* hphi__,
181  Wave_functions* ophi__) const
182 {
183  PROFILE("sirius::Hamiltonian::apply_fv_h_o");
184 
185  /* trivial case */
186  if (hphi__ == nullptr && ophi__ == nullptr) {
187  return;
188  }
189 
190  if (!apw_only__) {
191  if (hphi__ != nullptr) {
192  /* zero the local-orbital part */
193  hphi__->mt_coeffs(0).zero(memory_t::host, N__, n__);
194  }
195  if (ophi__ != nullptr) {
196  /* zero the local-orbital part */
197  ophi__->mt_coeffs(0).zero(memory_t::host, N__, n__);
198  }
199  }
200 
201  if (!phi_is_lo__) {
202  /* interstitial part */
203  local_op_->apply_h_o(N__, n__, phi__, hphi__, ophi__);
204  } else {
205  /* zero the APW part */
206  switch (ctx_.processing_unit()) {
207  case device_t::CPU: {
208  if (hphi__ != nullptr) {
209  hphi__->pw_coeffs(0).zero(memory_t::host, N__, n__);
210  }
211  if (ophi__ != nullptr) {
212  ophi__->pw_coeffs(0).zero(memory_t::host, N__, n__);
213  }
214  break;
215  }
216  case device_t::GPU: {
217  if (hphi__ != nullptr) {
218  hphi__->pw_coeffs(0).zero(memory_t::device, N__, n__);
219  }
220  if (ophi__ != nullptr) {
221  ophi__->pw_coeffs(0).zero(memory_t::device, N__, n__);
222  }
223  break;
224  }
225  }
226  }
227 
228  if (ctx_.processing_unit() == GPU && !apw_only__) {
229  phi__.mt_coeffs(0).copy_to(memory_t::host, N__, n__);
230  }
231 
232  /* short name for local number of G+k vectors */
233  int ngv = kp__->num_gkvec_loc();
234 
235  /* split atoms in blocks */
236  int num_atoms_in_block = 2 * omp_get_max_threads();
237  int nblk = utils::num_blocks(unit_cell_.num_atoms(), num_atoms_in_block);
238 
239  /* maximum number of AW radial functions in a block of atoms */
240  int max_mt_aw = num_atoms_in_block * unit_cell_.max_mt_aw_basis_size();
241  /* maximum number of LO radial functions in a block of atoms */
242  int max_mt_lo = num_atoms_in_block * unit_cell_.max_mt_lo_basis_size();
243 
244  utils::timer t0("sirius::Hamiltonian::apply_fv_h_o|alloc");
245 
246  /* matching coefficients for a block of atoms */
247  matrix<double_complex> alm_block;
248  matrix<double_complex> halm_block;
249 
250  switch (ctx_.processing_unit()) {
251  case device_t::CPU: {
252  alm_block = matrix<double_complex>(ctx_.mem_pool(memory_t::host), ngv, max_mt_aw);
253  if (hphi__ != nullptr) {
254  halm_block = matrix<double_complex>(ctx_.mem_pool(memory_t::host), ngv, std::max(max_mt_aw, max_mt_lo));
255  }
256  break;
257  }
258  case GPU: {
259  alm_block = matrix<double_complex>(ctx_.mem_pool(memory_t::host_pinned), ngv, max_mt_aw);
260  alm_block.allocate(ctx_.mem_pool(memory_t::device));
261  if (hphi__ != nullptr) {
262  halm_block = matrix<double_complex>(ctx_.mem_pool(memory_t::host_pinned),
263  ngv, std::max(max_mt_aw, max_mt_lo));
264  halm_block.allocate(ctx_.mem_pool(memory_t::device));
265  }
266  break;
267  }
268  }
269  size_t sz = max_mt_aw * n__;
270  /* buffers for alm_phi and halm_phi */
271  mdarray<double_complex, 1> alm_phi_buf;
272  if (ophi__ != nullptr) {
273  switch (ctx_.processing_unit()) {
274  case CPU: {
275  alm_phi_buf = mdarray<double_complex, 1>(ctx_.mem_pool(memory_t::host), sz);
276  break;
277  }
278  case GPU: {
279  alm_phi_buf = mdarray<double_complex, 1>(ctx_.mem_pool(memory_t::host_pinned), sz);
280  alm_phi_buf.allocate(ctx_.mem_pool(memory_t::device));
281  break;
282  }
283  }
284  }
285  mdarray<double_complex, 1> halm_phi_buf;
286  if (hphi__ != nullptr) {
287  switch (ctx_.processing_unit()) {
288  case CPU: {
289  halm_phi_buf = mdarray<double_complex, 1>(ctx_.mem_pool(memory_t::host), sz);
290  break;
291  }
292  case GPU: {
293  size_t sz = max_mt_aw * n__;
294  halm_phi_buf = mdarray<double_complex, 1>(ctx_.mem_pool(memory_t::host_pinned), sz);
295  halm_phi_buf.allocate(ctx_.mem_pool(memory_t::device));
296  break;
297  }
298  }
299  }
300  t0.stop();

302  auto generate_alm = [&](int atom_begin, int atom_end, std::vector<int>& offsets_aw)
303  {
304  utils::timer t1("sirius::Hamiltonian::apply_fv_h_o|alm");
305  #pragma omp parallel
306  {
307  int tid = omp_get_thread_num();
308  for (int ia = atom_begin; ia < atom_end; ia++) {
309  if (ia % omp_get_num_threads() == tid) {
310  int ialoc = ia - atom_begin;
311  auto& atom = unit_cell_.atom(ia);
312  auto& type = atom.type();
313 
314  /* wrapper for matching coefficients for a given atom */
315  mdarray<double_complex, 2> alm_tmp;
316  mdarray<double_complex, 2> halm_tmp;
317  switch (ctx_.processing_unit()) {
318  case CPU: {
319  alm_tmp = mdarray<double_complex, 2>(alm_block.at(memory_t::host, 0, offsets_aw[ialoc]),
320  ngv, type.mt_aw_basis_size());
321  if (hphi__ != nullptr) {
322  halm_tmp = mdarray<double_complex, 2>(halm_block.at(memory_t::host, 0, offsets_aw[ialoc]),
323  ngv, type.mt_aw_basis_size());
324  }
325  break;
326  }
327  case GPU: {
328  alm_tmp = mdarray<double_complex, 2>(alm_block.at(memory_t::host, 0, offsets_aw[ialoc]),
329  alm_block.at(memory_t::device, 0, offsets_aw[ialoc]),
330  ngv, type.mt_aw_basis_size());
331  if (hphi__ != nullptr) {
332  halm_tmp = mdarray<double_complex, 2>(halm_block.at(memory_t::host, 0, offsets_aw[ialoc]),
333  halm_block.at(memory_t::device, 0, offsets_aw[ialoc]),
334  ngv, type.mt_aw_basis_size());
335  }
336  break;
337  }
338  }
339 
340  /* generate LAPW matching coefficients on the CPU */
341  kp__->alm_coeffs_loc().generate(ia, alm_tmp);
342  /* conjugate alm */
343  for (int xi = 0; xi < type.mt_aw_basis_size(); xi++) {
344  for (int igk = 0; igk < ngv; igk++) {
345  alm_tmp(igk, xi) = std::conj(alm_tmp(igk, xi));
346  }
347  }
348 #if defined(__GPU)
349  if (ctx_.processing_unit() == device_t::GPU) {
350  alm_tmp.copy_to(memory_t::device, stream_id(tid));
351  }
352 #endif
353  if (hphi__ != nullptr) {
354  apply_hmt_to_apw<spin_block_t::nm>(atom, ngv, alm_tmp, halm_tmp);
355 #if defined(__GPU)
356  if (ctx_.processing_unit() == device_t::GPU) {
357  halm_tmp.copy_to(memory_t::device, stream_id(tid));
358  }
359 #endif
360  }
361  }
362  }
363 #if defined(__GPU)
364  if (ctx_.processing_unit() == device_t::GPU) {
365  acc::sync_stream(stream_id(tid));
366  }
367 #endif
368  }
369  };

371  auto compute_alm_phi = [&](matrix<double_complex>& alm_phi, matrix<double_complex>& halm_phi, int num_mt_aw)
372  {
373  utils::timer t1("sirius::Hamiltonian::apply_fv_h_o|alm_phi");
374 
375  /* first zgemm: A(G, lm)^{T} * C(G, i) and hA(G, lm)^{T} * C(G, i) */
376  switch (ctx_.processing_unit()) {
377  case CPU: {
378  if (ophi__ != nullptr) {
379  /* create resulting array with proper dimensions from the already allocated chunk of memory */
380  alm_phi = matrix<double_complex>(alm_phi_buf.at(memory_t::host), num_mt_aw, n__);
381  /* alm_phi(lm, i) = A(G, lm)^{T} * C(G, i), remember that Alm was conjugated */
382  linalg<CPU>::gemm(2, 0, num_mt_aw, n__, ngv,
383  alm_block.at(memory_t::host), alm_block.ld(),
384  phi__.pw_coeffs(0).prime().at(memory_t::host, 0, N__), phi__.pw_coeffs(0).prime().ld(),
385  alm_phi.at(memory_t::host), alm_phi.ld());
386  }
387  if (hphi__ != nullptr) {
388  /* create resulting array with proper dimensions from the already allocated chunk of memory */
389  halm_phi = matrix<double_complex>(halm_phi_buf.at(memory_t::host), num_mt_aw, n__);
390  /* halm_phi(lm, i) = H_{mt}A(G, lm)^{T} * C(G, i) */
391  linalg<CPU>::gemm(2, 0, num_mt_aw, n__, ngv,
392  halm_block.at(memory_t::host), halm_block.ld(),
393  phi__.pw_coeffs(0).prime().at(memory_t::host, 0, N__), phi__.pw_coeffs(0).prime().ld(),
394  halm_phi.at(memory_t::host), halm_phi.ld());
395  }
396  break;
397  }
398  case GPU: {
399 #if defined(__GPU)
400  if (ophi__ != nullptr) {
401  /* create resulting array with proper dimensions from the already allocated chunk of memory */
402  alm_phi = matrix<double_complex>(alm_phi_buf.at(memory_t::host), alm_phi_buf.at(memory_t::device), num_mt_aw, n__);
403  /* alm_phi(lm, i) = A(G, lm)^{T} * C(G, i) */
404  linalg<GPU>::gemm(2, 0, num_mt_aw, n__, ngv,
405  alm_block.at(memory_t::device), alm_block.ld(),
406  phi__.pw_coeffs(0).prime().at(memory_t::device, 0, N__), phi__.pw_coeffs(0).prime().ld(),
407  alm_phi.at(memory_t::device), alm_phi.ld());
408  alm_phi.copy_to(memory_t::host);
409  }
410  if (hphi__ != nullptr) {
411  /* create resulting array with proper dimensions from the already allocated chunk of memory */
412  halm_phi = matrix<double_complex>(halm_phi_buf.at(memory_t::host), halm_phi_buf.at(memory_t::device), num_mt_aw, n__);
413  /* halm_phi(lm, i) = H_{mt}A(G, lm)^{T} * C(G, i) */
414  linalg<GPU>::gemm(2, 0, num_mt_aw, n__, ngv,
415  halm_block.at(memory_t::device), halm_block.ld(),
416  phi__.pw_coeffs(0).prime().at(memory_t::device, 0, N__), phi__.pw_coeffs(0).prime().ld(),
417  halm_phi.at(memory_t::device), halm_phi.ld());
418 
419  halm_phi.copy_to(memory_t::host);
420  }
421 #endif
422  break;
423  }
424  }
425 
426  if (hphi__ != nullptr) {
427  kp__->comm().allreduce(halm_phi.at(memory_t::host), num_mt_aw * n__);
428  if (ctx_.processing_unit() == device_t::GPU) {
429  halm_phi.copy_to(memory_t::device);
430  }
431  }
432 
433  if (ophi__ != nullptr) {
434  kp__->comm().allreduce(alm_phi.at(memory_t::host), num_mt_aw * n__);
435  if (ctx_.processing_unit() == device_t::GPU) {
436  alm_phi.copy_to(memory_t::device);
437  }
438  }
439  };

441  auto compute_apw_apw = [&](matrix<double_complex>& alm_phi, matrix<double_complex>& halm_phi, int num_mt_aw)
442  {

444  /* second zgemm: Alm^{*} (Alm * C) */
445  switch (ctx_.processing_unit()) {
446  case CPU: {
447  if (ophi__ != nullptr) {
448  /* APW-APW contribution to overlap */

458  /* APW-APW contribution to Hamiltonian */

461  alm_block.at(memory_t::host), alm_block.ld(),

485  halm_phi.at(memory_t::device), halm_phi.ld(),

487  hphi__->pw_coeffs(0).prime().at(memory_t::device, 0, N__), hphi__->pw_coeffs(0).prime().ld());
488  }

495  auto collect_lo = [&](int atom_begin, int atom_end, std::vector<int>& offsets_lo, matrix<double_complex>& phi_lo_block)
496  {
497  utils::timer t1("sirius::Hamiltonian::apply_fv_h_o|phi_lo");
498  /* broadcast local orbital coefficients */
499  for (int ia = atom_begin; ia < atom_end; ia++) {
500  int ialoc = ia - atom_begin;
501  auto& atom = unit_cell_.atom(ia);
502  auto& type = atom.type();
503  auto ia_location = phi__.spl_num_atoms().location(ia);
504 
505  /* lo coefficients for a given atom and all bands */
506  matrix<double_complex> phi_lo_ia(type.mt_lo_basis_size(), n__);
507 
508  if (ia_location.rank == kp__->comm().rank()) {
509  #pragma omp parallel for schedule(static)
510  for (int i = 0; i < n__; i++) {
511  std::memcpy(&phi_lo_ia(0, i),
512  phi__.mt_coeffs(0).prime().at(memory_t::host, phi__.offset_mt_coeffs(ia_location.local_index), N__ + i),
513  type.mt_lo_basis_size() * sizeof(double_complex));
514  }
515  }
516  /* broadcast from a rank */
517  kp__->comm().bcast(phi_lo_ia.at(memory_t::host), static_cast<int>(phi_lo_ia.size()), ia_location.rank);
518  /* wrtite into a proper position in a block */
519  #pragma omp parallel for schedule(static)
520  for (int i = 0; i < n__; i++) {

548  int xi_lo = naw + ilo;

566  }
567  case GPU: {

569  hmt.allocate(memory_t::device);

576  break;

578  }
579  }
580  } //ia

582  case CPU: {
583  linalg<CPU>::gemm(0, 0, ngv, n__, num_mt_lo,
584  linalg_const<double_complex>::one(),
585  halm_block.at(memory_t::host), halm_block.ld(),
586  phi_lo_block.at(memory_t::host), phi_lo_block.ld(),

589  break;
590 
591  }
592  case GPU: {
593 #if defined(__GPU)
594  linalg<GPU>::gemm(0, 0, ngv, n__, num_mt_lo,
595  &linalg_const<double_complex>::one(),
596  halm_block.at(memory_t::device), halm_block.ld(),
597  phi_lo_block.at(memory_t::device), phi_lo_block.ld(),
598  &linalg_const<double_complex>::one(),
599  hphi__->pw_coeffs(0).prime().at(memory_t::device, 0, N__), hphi__->pw_coeffs(0).prime().ld());
600 #endif
601  break;
602  }
603  }

606  /* apw-lo block for ophi */
607  if (ophi__ != nullptr) {
608  halm_block.zero();

610  int ialoc = ia - atom_begin;
611  auto& atom = unit_cell_.atom(ia);
612  auto& type = atom.type();
613  int naw = type.mt_aw_basis_size();
614  int nlo = type.mt_lo_basis_size();
615 
616  #pragma omp parallel for schedule(static)
617  for (int ilo = 0; ilo < nlo; ilo++) {
618  int xi_lo = naw + ilo;
619  /* local orbital indices */
620  int l_lo = type.indexb(xi_lo).l;
621  int lm_lo = type.indexb(xi_lo).lm;
622  int order_lo = type.indexb(xi_lo).order;
623  /* use halm as temporary buffer to compute alm*o */
624  for (int order_aw = 0; order_aw < (int)type.aw_descriptor(l_lo).size(); order_aw++) {
625  for (int igloc = 0; igloc < ngv; igloc++) {
626  halm_block(igloc, offsets_lo[ialoc] + ilo) +=
627  alm_block(igloc, offsets_aw[ialoc] + type.indexb_by_lm_order(lm_lo, order_aw)) *
628  atom.symmetry_class().o_radial_integral(l_lo, order_aw, order_lo);
629  } // TODO: block copy to GPU
630  }
631  }
632  } //ia
633  switch (ctx_.processing_unit()) {
634  case CPU: {
635  linalg<CPU>::gemm(0, 0, ngv, n__, num_mt_lo,
636  linalg_const<double_complex>::one(),
637  halm_block.at(memory_t::host), halm_block.ld(),
638  phi_lo_block.at(memory_t::host), phi_lo_block.ld(),
639  linalg_const<double_complex>::one(),

641  break;

644  case GPU: {
645 #if defined(__GPU)
646  halm_block.copy_to(memory_t::device, 0, ngv * num_mt_lo);
647  linalg<GPU>::gemm(0, 0, ngv, n__, num_mt_lo,
648  &linalg_const<double_complex>::one(),
649  halm_block.at(memory_t::device), halm_block.ld(),
650  phi_lo_block.at(memory_t::device), phi_lo_block.ld(),
651  &linalg_const<double_complex>::one(),
652  ophi__->pw_coeffs(0).prime().at(memory_t::device, 0, N__), ophi__->pw_coeffs(0).prime().ld());
653 #endif
654  break;
655  }

660  utils::timer t2("sirius::Hamiltonian::apply_fv_h_o|mt");
661  /* loop over blocks of atoms */
662  for (int iblk = 0; iblk < nblk; iblk++) {
663  int atom_begin = iblk * num_atoms_in_block;
664  int atom_end = std::min(unit_cell_.num_atoms(), (iblk + 1) * num_atoms_in_block);
665  /* actual number of AW radial functions in a block of atoms */
666  int num_mt_aw{0};
667  /* actual number of local orbitals in a block of atoms */
668  int num_mt_lo{0};
669  std::vector<int> offsets_aw(num_atoms_in_block);
670  std::vector<int> offsets_lo(num_atoms_in_block);
671  for (int ia = atom_begin; ia < atom_end; ia++) {
672  int ialoc = ia - atom_begin;
673  auto& atom = unit_cell_.atom(ia);
674  auto& type = atom.type();
675  offsets_aw[ialoc] = num_mt_aw;
676  offsets_lo[ialoc] = num_mt_lo;
677  num_mt_aw += type.mt_aw_basis_size();
678  num_mt_lo += type.mt_lo_basis_size();
679  }
680 
681  /* created alm and halm for a block of atoms */
682  generate_alm(atom_begin, atom_end, offsets_aw);
683 
684  matrix<double_complex> alm_phi;
685  matrix<double_complex> halm_phi;
686 
687  compute_alm_phi(alm_phi, halm_phi, num_mt_aw);
688 
689  if (!phi_is_lo__) {
690  compute_apw_apw(alm_phi, halm_phi, num_mt_aw);
691  }
692 
693  if (!apw_only__) {
694  /* local orbital coefficients for a block of atoms and all states */
695  matrix<double_complex> phi_lo_block(num_mt_lo, n__);
696  if (ctx_.processing_unit() == device_t::GPU) {
697  phi_lo_block.allocate(memory_t::device);
698  }
699  collect_lo(atom_begin, atom_end, offsets_lo, phi_lo_block);
700 
701  compute_apw_lo(atom_begin, atom_end, num_mt_lo, offsets_aw, offsets_lo, phi_lo_block);
702 
703  utils::timer t3("sirius::Hamiltonian::apply_fv_h_o|lo-lo-apw");
704  /* lo-APW contribution */
705  for (int ia = atom_begin; ia < atom_end; ia++) {
706  int ialoc = ia - atom_begin;
707  auto& atom = unit_cell_.atom(ia);
708  auto& type = atom.type();
709 
710  auto ia_location = phi__.spl_num_atoms().location(ia);
711 
712  if (ia_location.rank == kp__->comm().rank()) {
713  int offset_mt_coeffs = phi__.offset_mt_coeffs(ia_location.local_index);
714 
715  #pragma omp parallel for schedule(static)
716  for (int ilo = 0; ilo < type.mt_lo_basis_size(); ilo++) {
717  int xi_lo = type.mt_aw_basis_size() + ilo;
718  /* local orbital indices */
719  int l_lo = type.indexb(xi_lo).l;
720  int lm_lo = type.indexb(xi_lo).lm;
721  int order_lo = type.indexb(xi_lo).order;
722  int idxrf_lo = type.indexb(xi_lo).idxrf;
723 
724  /* lo-lo contribution */
725  for (int jlo = 0; jlo < type.mt_lo_basis_size(); jlo++) {
726  int xi_lo1 = type.mt_aw_basis_size() + jlo;
727  int lm1 = type.indexb(xi_lo1).lm;
728  int order1 = type.indexb(xi_lo1).order;
729  int idxrf1 = type.indexb(xi_lo1).idxrf;
730  if (lm_lo == lm1 && ophi__ != nullptr) {
731  for (int i = 0; i < n__; i++) {
732  ophi__->mt_coeffs(0).prime(offset_mt_coeffs + ilo, N__ + i) +=
733  phi_lo_block(offsets_lo[ialoc] + jlo, i) *
734  atom.symmetry_class().o_radial_integral(l_lo, order_lo, order1);
735  }
736  }
737  if (hphi__ != nullptr) {
738  auto& gc = gaunt_coefs_->gaunt_vector(lm_lo, lm1);
739  for (int i = 0; i < n__; i++) {
740  hphi__->mt_coeffs(0).prime(offset_mt_coeffs + ilo, N__ + i) +=
741  phi_lo_block(offsets_lo[ialoc] + jlo, i) *
742  atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf_lo, idxrf1, gc);
743  }
744  }
745  }
746 
747  /* lo-APW contribution */
748  if (!phi_is_lo__) {
749  if (ophi__ != nullptr) {
750  for (int i = 0; i < n__; i++) {
751  /* lo-APW contribution to ophi */
752  for (int order_aw = 0; order_aw < (int)type.aw_descriptor(l_lo).size(); order_aw++) {
753  ophi__->mt_coeffs(0).prime(offset_mt_coeffs + ilo, N__ + i) +=
754  atom.symmetry_class().o_radial_integral(l_lo, order_lo, order_aw) *
755  alm_phi(offsets_aw[ialoc] + type.indexb_by_lm_order(lm_lo, order_aw), i);
756  }
757  }
758  }
759 
760  if (hphi__ != nullptr) {
761  for (int i = 0; i < n__; i++) {
762  double_complex z(0, 0);
763  for (int xi = 0; xi < type.mt_aw_basis_size(); xi++) {
764  int lm_aw = type.indexb(xi).lm;
765  int idxrf_aw = type.indexb(xi).idxrf;
766  auto& gc = gaunt_coefs_->gaunt_vector(lm_lo, lm_aw);
767  z += atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf_lo, idxrf_aw, gc) *
768  alm_phi(offsets_aw[ialoc] + xi, i);
769  }
770  /* lo-APW contribution to hphi */
771  hphi__->mt_coeffs(0).prime(offset_mt_coeffs + ilo, N__ + i) += z;
772  }
773  }
774  }
775  }
776  }
777  }
778  t3.stop();
779  }
780  }
781  t2.stop();
782  if (ctx_.processing_unit() == GPU && !apw_only__) {
783  if (hphi__ != nullptr) {
784  hphi__->mt_coeffs(0).copy_to(memory_t::device, N__, n__);
785  }
786  if (ophi__ != nullptr) {
787  ophi__->mt_coeffs(0).copy_to(memory_t::device, N__, n__);
788  }
789  }
790  if (ctx_.control().print_checksum_) {
791  if (hphi__) {
792  auto cs1 = hphi__->checksum_pw(ctx_.processing_unit(), 0, N__, n__);
793  auto cs2 = hphi__->checksum(ctx_.processing_unit(), 0, N__, n__);
794  if (kp__->comm().rank() == 0) {
795  utils::print_checksum("hphi_pw", cs1);
796  utils::print_checksum("hphi", cs2);
797  }
798  }
799  if (ophi__) {
800  auto cs1 = ophi__->checksum_pw(ctx_.processing_unit(), 0, N__, n__);
801  auto cs2 = ophi__->checksum(ctx_.processing_unit(), 0, N__, n__);
802  if (kp__->comm().rank() == 0) {
803  utils::print_checksum("ophi_pw", cs1);
804  utils::print_checksum("ophi", cs2);
805  }
806  }
807  }

810 // TODO: port to GPU
811 inline void Hamiltonian::apply_magnetic_field(K_point* kp__,
812  Wave_functions& fv_states__,
813  std::vector<Wave_functions>& hpsi__) const
814 {
815  PROFILE("sirius::Hamiltonian::apply_magnetic_field");
816 
817  assert(hpsi__.size() == 2 || hpsi__.size() == 3);
818 
819  local_op_->apply_b(0, ctx_.num_fv_states(), fv_states__, hpsi__);
820 
821  mdarray<double_complex, 3> zm(unit_cell_.max_mt_basis_size(), unit_cell_.max_mt_basis_size(), ctx_.num_mag_dims());
822 
823  for (int ialoc = 0; ialoc < fv_states__.spl_num_atoms().local_size(); ialoc++) {
824  int ia = fv_states__.spl_num_atoms()[ialoc];
825  auto& atom = unit_cell_.atom(ia);
826  int offset = fv_states__.offset_mt_coeffs(ialoc);
827  int mt_basis_size = atom.type().mt_basis_size();
828 
829  zm.zero();
830 
831  /* only upper triangular part of zm is computed because it is a hermitian matrix */
832  #pragma omp parallel for default(shared)
833  for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
834  int lm2 = atom.type().indexb(xi2).lm;
835  int idxrf2 = atom.type().indexb(xi2).idxrf;
836 
837  for (int i = 0; i < ctx_.num_mag_dims(); i++) {
838  for (int xi1 = 0; xi1 <= xi2; xi1++) {
839  int lm1 = atom.type().indexb(xi1).lm;
840  int idxrf1 = atom.type().indexb(xi1).idxrf;
841 
842  zm(xi1, xi2, i) = gaunt_coefs_->sum_L3_gaunt(lm1, lm2, atom.b_radial_integrals(idxrf1, idxrf2, i));
843  }
844  }
845  }
846  /* compute bwf = B_z*|wf_j> */
847  linalg<CPU>::hemm(0, 0, mt_basis_size, ctx_.num_fv_states(),
848  linalg_const<double_complex>::one(),
849  zm.at(memory_t::host),
850  zm.ld(),
851  fv_states__.mt_coeffs(0).prime().at(memory_t::host, offset, 0),
852  fv_states__.mt_coeffs(0).prime().ld(),
853  linalg_const<double_complex>::zero(),
854  hpsi__[0].mt_coeffs(0).prime().at(memory_t::host, offset, 0),
855  hpsi__[0].mt_coeffs(0).prime().ld());
856 
857  /* compute bwf = (B_x - iB_y)|wf_j> */
858  if (hpsi__.size() == 3) {
859  /* reuse first (z) component of zm matrix to store (B_x - iB_y) */
860  for (int xi2 = 0; xi2 < mt_basis_size; xi2++) {
861  for (int xi1 = 0; xi1 <= xi2; xi1++) {
862  zm(xi1, xi2, 0) = zm(xi1, xi2, 1) - double_complex(0, 1) * zm(xi1, xi2, 2);
863  }
864 
865  /* remember: zm for x,y,z, components of magnetic field is hermitian and we computed
866  * only the upper triangular part */
867  for (int xi1 = xi2 + 1; xi1 < mt_basis_size; xi1++) {
868  zm(xi1, xi2, 0) = std::conj(zm(xi2, xi1, 1)) - double_complex(0, 1) * std::conj(zm(xi2, xi1, 2));
869  }
870  }
871 
872  linalg<CPU>::gemm(0, 0, mt_basis_size, ctx_.num_fv_states(), mt_basis_size,
873  zm.at(memory_t::host),
874  zm.ld(),
875  fv_states__.mt_coeffs(0).prime().at(memory_t::host, offset, 0),
876  fv_states__.mt_coeffs(0).prime().ld(),
877  hpsi__[2].mt_coeffs(0).prime().at(memory_t::host, offset, 0),
878  hpsi__[2].mt_coeffs(0).prime().ld());
879  }
880  }
881 
882  /* copy Bz|\psi> to -Bz|\psi> */
883  for (int i = 0; i < ctx_.num_fv_states(); i++) {
884  for (int j = 0; j < fv_states__.pw_coeffs(0).num_rows_loc(); j++) {
885  hpsi__[1].pw_coeffs(0).prime(j, i) = -hpsi__[0].pw_coeffs(0).prime(j, i);
886  }
887  for (int j = 0; j < fv_states__.mt_coeffs(0).num_rows_loc(); j++) {
888  hpsi__[1].mt_coeffs(0).prime(j, i) = -hpsi__[0].mt_coeffs(0).prime(j, i);
889  }
890  }
891 }
