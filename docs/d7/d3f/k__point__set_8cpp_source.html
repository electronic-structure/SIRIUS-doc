55 void K_point_set::create_k_mesh(vector3d<int> k_grid__, vector3d<int> k_shift__, int use_symmetry__)
56 {
57  PROFILE("sirius::K_point_set::create_k_mesh");
58 
59  int nk;
60  mdarray<double, 2> kp;
61  std::vector<double> wk;
62  if (use_symmetry__) {
63  auto result = get_irreducible_reciprocal_mesh(ctx_.unit_cell().symmetry(), k_grid__, k_shift__);
64  nk = std::get<0>(result);
65  wk = std::get<1>(result);
66  auto tmp = std::get<2>(result);
67  kp = mdarray<double, 2>(3, nk);
68  for (int i = 0; i < nk; i++) {
69  for (int x : {0, 1, 2}) {
70  kp(x, i) = tmp[i][x];
71  }
72  }
73  } else {
74  nk = k_grid__[0] * k_grid__[1] * k_grid__[2];
75  wk = std::vector<double>(nk, 1.0 / nk);
76  kp = mdarray<double, 2>(3, nk);
77 
78  int ik = 0;
79  for (int i0 = 0; i0 < k_grid__[0]; i0++) {
80  for (int i1 = 0; i1 < k_grid__[1]; i1++) {
81  for (int i2 = 0; i2 < k_grid__[2]; i2++) {
82  kp(0, ik) = double(i0 + k_shift__[0] / 2.0) / k_grid__[0];
83  kp(1, ik) = double(i1 + k_shift__[1] / 2.0) / k_grid__[1];
84  kp(2, ik) = double(i2 + k_shift__[2] / 2.0) / k_grid__[2];
85  ik++;
86  }
87  }
88  }
89  }
90 
91  for (int ik = 0; ik < nk; ik++) {
92  add_kpoint(&kp(0, ik), wk[ik]);
93  }
94 
95  initialize();
96 }

99 {

102  }

107  spl_num_kpoints_ = splindex<splindex_t::chunk>(num_kpoints(), comm().size(), comm().rank(), spl_tmp.counts());
108  } else {
109  spl_num_kpoints_ = splindex<splindex_t::chunk>(num_kpoints(), comm().size(), comm().rank(), counts);

111 
112  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
113  kpoints_[spl_num_kpoints_[ikloc]]->initialize();

115 
116  if (ctx_.control().verbosity_ > 0) {
117  print_info();

119  ctx_.print_memory_usage(__FILE__, __LINE__);
120  this->initialized_ = true;

123 void K_point_set::sync_band_occupancies()
124 {
125  int nranks = comm().size();
126  int pid = comm().rank();
127  std::vector<int> nk_per_rank(nranks);
128  for (int i = 0; i < nranks; ++i) {
129  nk_per_rank[i] = spl_num_kpoints_.local_size(i);
130  }
131  int ns = 1;
132  int num_bands = ctx_.num_bands();
133  if (ctx_.num_spin_dims() > 1)
134  ns = 2;
135 
136  std::vector<int> offsets(nranks);
137  std::vector<int> sizes(nranks);
138  int offset = 0;
139  for (int i = 0; i < nranks; ++i) {
140 
141  offsets[i] = offset;
142  int lsize = nk_per_rank[i] * num_bands * ns;
143  sizes[i] = lsize;
144  offset += lsize;
145  }
146  int size = offset;
147 
148  std::vector<double> tmp(sizes[pid]);
149  for (int i = 0; i < sizes[pid]; ++i) {
150  int gi = offsets[pid] + i;
151  int k = gi / (num_bands * ns);
152  int spin = (gi % (num_bands * ns)) / num_bands;
153  int n = (gi % (num_bands * ns)) % num_bands;
154  tmp[i] = kpoints_[k]->band_occupancy(n, spin);
155  }
156 
157  std::vector<double> occupancies(size);
158  comm().allgather(tmp.data(), sizes[pid], occupancies.data(), sizes.data(), offsets.data());
159 
160  for (int i = 0; i < size; ++i) {
161  int k = i / (num_bands * ns);
162  int spin = (i % (num_bands * ns)) / num_bands;
163  int n = (i % (num_bands * ns)) % num_bands;
164  kpoints_[k]->band_occupancy(n, spin, occupancies[i]);
165  }
166 }

168 void K_point_set::find_band_occupancies()
169 {
170  PROFILE("sirius::K_point_set::find_band_occupancies");
171 
172  auto band_occ_callback = ctx_.band_occ_callback();
173  if (band_occ_callback) {
174  band_occ_callback();
175  return;
176  }
177 
178  double ef{0};
179  double de{0.1};
180 
181  int s{1};
182  int sp;
183 
184  mdarray<double, 3> bnd_occ(ctx_.num_bands(), ctx_.num_spin_dims(), num_kpoints());
185 
186  double ne{0};
187 
188  /* target number of electrons */
189  double ne_target = ctx_.unit_cell().num_valence_electrons() - ctx_.parameters_input().extra_charge_;
190 
191  if (std::abs(ctx_.num_fv_states() * double(ctx_.max_occupancy()) - ne_target) < 1e-10) {
192  // this is an insulator, skip search for band occupancies
193  this->band_gap_ = -1;
194 
195  // determine fermi energy as max occupied band energy.
196  double efermi = std::numeric_limits<double>::min();
197  for (int ik = 0; ik < num_kpoints(); ik++) {
198  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
199  for (int j = 0; j < ctx_.num_bands(); j++) {
200  efermi = std::max(efermi, kpoints_[ik]->band_energy(j, ispn));
201  }
202  }
203  }
204  energy_fermi_ = efermi;
205  return;
206  }
207 
208  int step{0};
209  /* calculate occupations */
210  while (std::abs(ne - ne_target) >= 1e-11) {
211  /* update Efermi */
212  ef += de;
213  /* compute total number of electrons */
214  ne = 0.0;
215  for (int ik = 0; ik < num_kpoints(); ik++) {
216  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
217  for (int j = 0; j < ctx_.num_bands(); j++) {
218  bnd_occ(j, ispn, ik) =
219  smearing::gaussian(kpoints_[ik]->band_energy(j, ispn) - ef, ctx_.smearing_width()) *
220  ctx_.max_occupancy();
221  ne += bnd_occ(j, ispn, ik) * kpoints_[ik]->weight();
222  }
223  }
224  }
225 
226  sp = s;
227  s = (ne > ne_target) ? -1 : 1;
228  /* reduce de step if we change the direction, otherwise increase the step */
229  de = (s != sp) ? (-de * 0.5) : (de * 1.25);
230 
231  if (step > 10000) {
232  std::stringstream s;
233  s << "search of band occupancies failed after 10000 steps";
234  TERMINATE(s);
235  }
236  step++;
237  }
238 
239  energy_fermi_ = ef;
240 
241  for (int ik = 0; ik < num_kpoints(); ik++) {
242  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
243  for (int j = 0; j < ctx_.num_bands(); j++) {
244  kpoints_[ik]->band_occupancy(j, ispn, bnd_occ(j, ispn, ik));
245  }
246  }
247  }
248 
249  band_gap_ = 0.0;
250 
251  int nve = static_cast<int>(ne_target + 1e-12);
252  if (ctx_.num_spins() == 2 || (std::abs(nve - ne_target) < 1e-12 && nve % 2 == 0)) {
253  /* find band gap */
254  std::vector<std::pair<double, double>> eband;
255  std::pair<double, double> eminmax;
256 
257  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
258  for (int j = 0; j < ctx_.num_bands(); j++) {
259  eminmax.first = 1e10;
260  eminmax.second = -1e10;
261 
262  for (int ik = 0; ik < num_kpoints(); ik++) {
263  eminmax.first = std::min(eminmax.first, kpoints_[ik]->band_energy(j, ispn));
264  eminmax.second = std::max(eminmax.second, kpoints_[ik]->band_energy(j, ispn));
265  }
266 
267  eband.push_back(eminmax);
268  }
269  }
270 
271  std::sort(eband.begin(), eband.end());
272 
273  int ist = nve;
274  if (ctx_.num_spins() == 1) {
275  ist /= 2;
276  }
277 
278  if (eband[ist].first > eband[ist - 1].second) {
279  band_gap_ = eband[ist].first - eband[ist - 1].second;
280  }
281  }
282 }

320 void K_point_set::save(std::string const& name__) const
321 {
322  if (ctx_.comm().rank() == 0) {
323  if (!utils::file_exists(name__)) {
324  HDF5_tree(name__, hdf5_access_t::truncate);
325  }
326  HDF5_tree fout(name__, hdf5_access_t::read_write);
327  fout.create_node("K_point_set");
328  fout["K_point_set"].write("num_kpoints", num_kpoints());
329  }
330  ctx_.comm().barrier();
331  for (int ik = 0; ik < num_kpoints(); ik++) {
332  /* check if this ranks stores the k-point */
333  if (ctx_.comm_k().rank() == spl_num_kpoints_.local_rank(ik)) {
334  kpoints_[ik]->save(name__, ik);
335  }
336  /* wait for all */
337  ctx_.comm().barrier();
338  }
339 }
