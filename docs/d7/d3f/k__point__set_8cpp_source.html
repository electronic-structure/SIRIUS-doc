28 template <sync_band_t what>
29 void K_point_set::sync_band()
30 {
31  PROFILE("sirius::K_point_set::sync_band");
32 
33  sddk::mdarray<double, 3> data(ctx_.num_bands(), ctx_.num_spinors(), num_kpoints(), ctx_.mem_pool(memory_t::host),
34  "K_point_set::sync_band.data");
35 
36  int nb = ctx_.num_bands() * ctx_.num_spinors();
37  #pragma omp parallel
38  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
39  int ik = spl_num_kpoints_[ikloc];
40  switch (what) {
41  case sync_band_t::energy: {
42  std::copy(&kpoints_[ik]->band_energies_(0, 0), &kpoints_[ik]->band_energies_(0, 0) + nb,
43  &data(0, 0, ik));
44  break;
45  }
46  case sync_band_t::occupancy: {
47  std::copy(&kpoints_[ik]->band_occupancies_(0, 0), &kpoints_[ik]->band_occupancies_(0, 0) + nb,
48  &data(0, 0, ik));
49  break;
50  }
51  }
52  }
53 
54  comm().allgather(data.at(memory_t::host), nb * spl_num_kpoints_.local_size(),
55  nb * spl_num_kpoints_.global_offset());
56 
57  #pragma omp parallel for
58  for (int ik = 0; ik < num_kpoints(); ik++) {
59  switch (what) {
60  case sync_band_t::energy: {
61  std::copy(&data(0, 0, ik), &data(0, 0, ik) + nb, &kpoints_[ik]->band_energies_(0, 0));
62  break;
63  }
64  case sync_band_t::occupancy: {
65  std::copy(&data(0, 0, ik), &data(0, 0, ik) + nb, &kpoints_[ik]->band_occupancies_(0, 0));
66  break;
67  }
68  }
69  }
70 }

79 void K_point_set::create_k_mesh(vector3d<int> k_grid__, vector3d<int> k_shift__, int use_symmetry__)
80 {
81  PROFILE("sirius::K_point_set::create_k_mesh");
82 
83  int nk;
84  mdarray<double, 2> kp;
85  std::vector<double> wk;
86  if (use_symmetry__) {
87  auto result = get_irreducible_reciprocal_mesh(ctx_.unit_cell().symmetry(), k_grid__, k_shift__);
88  nk = std::get<0>(result);
89  wk = std::get<1>(result);
90  auto tmp = std::get<2>(result);
91  kp = mdarray<double, 2>(3, nk);
92  for (int i = 0; i < nk; i++) {
93  for (int x : {0, 1, 2}) {
94  kp(x, i) = tmp[i][x];
95  }
96  }
97  } else {
98  nk = k_grid__[0] * k_grid__[1] * k_grid__[2];
99  wk = std::vector<double>(nk, 1.0 / nk);
100  kp = mdarray<double, 2>(3, nk);
101 
102  int ik = 0;
103  for (int i0 = 0; i0 < k_grid__[0]; i0++) {
104  for (int i1 = 0; i1 < k_grid__[1]; i1++) {
105  for (int i2 = 0; i2 < k_grid__[2]; i2++) {
106  kp(0, ik) = double(i0 + k_shift__[0] / 2.0) / k_grid__[0];
107  kp(1, ik) = double(i1 + k_shift__[1] / 2.0) / k_grid__[1];
108  kp(2, ik) = double(i2 + k_shift__[2] / 2.0) / k_grid__[2];
109  ik++;
110  }
111  }
112  }
113  }
114 
115  for (int ik = 0; ik < nk; ik++) {
116  add_kpoint(&kp(0, ik), wk[ik]);
117  }
118 
119  initialize();
120 }

125  TERMINATE("K-point set is already initialized");
126  }
127  PROFILE("sirius::K_point_set::initialize");

129  if (counts.empty()) {
130  splindex<splindex_t::block> spl_tmp(num_kpoints(), comm().size(), comm().rank());
131  spl_num_kpoints_ = splindex<splindex_t::chunk>(num_kpoints(), comm().size(), comm().rank(), spl_tmp.counts());
132  } else {
133  spl_num_kpoints_ = splindex<splindex_t::chunk>(num_kpoints(), comm().size(), comm().rank(), counts);
134  }
135 
136  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
137  kpoints_[spl_num_kpoints_[ikloc]]->initialize();
138  }
139 
140  if (ctx_.verbosity() > 0) {
141  print_info();
142  }
143  ctx_.print_memory_usage(__FILE__, __LINE__);
144  this->initialized_ = true;

147 void K_point_set::find_band_occupancies()
148 {
149  PROFILE("sirius::K_point_set::find_band_occupancies");
150 
151  auto band_occ_callback = ctx_.band_occ_callback();
152  if (band_occ_callback) {
153  band_occ_callback();
154  return;
155  }
156 
157  /* target number of electrons */
158  double ne_target = ctx_.unit_cell().num_valence_electrons() - ctx_.cfg().parameters().extra_charge();
159 
160  /* this is a special case when there are no empty states */
161  if (ctx_.num_mag_dims() != 1 && std::abs(ctx_.num_bands() * ctx_.max_occupancy() - ne_target) < 1e-10) {
162  /* this is an insulator, skip search for band occupancies */
163  this->band_gap_ = 0;
164 
165  /* determine fermi energy as max occupied band energy. */
166  energy_fermi_ = std::numeric_limits<double>::lowest();
167  for (int ik = 0; ik < num_kpoints(); ik++) {
168  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
169  for (int j = 0; j < ctx_.num_bands(); j++) {
170  energy_fermi_ = std::max(energy_fermi_, kpoints_[ik]->band_energy(j, ispn));
171  }
172  }
173  }
174  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
175  int ik = spl_num_kpoints_[ikloc];
176  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
177  #pragma omp parallel for
178  for (int j = 0; j < ctx_.num_bands(); j++) {
179  kpoints_[ik]->band_occupancy(j, ispn, ctx_.max_occupancy());
180  }
181  }
182  }
183 
184  this->sync_band<sync_band_t::occupancy>();
185  return;
186  }
187 
188  if (ctx_.smearing_width() == 0) {
189  throw std::runtime_error("[find_band_occupancies] zero smearing width");
190  }
191 
192  /* get minimum and maximum band energies */
193 
194  auto emin = std::numeric_limits<double>::max();
195  auto emax = std::numeric_limits<double>::lowest();
196 
197  #pragma omp parallel for reduction(min:emin) reduction(max:emax)
198  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
199  int ik = spl_num_kpoints_[ikloc];
200  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
201  emin = std::min(emin, kpoints_[ik]->band_energy(0, ispn));
202  emax = std::max(emax, kpoints_[ik]->band_energy(ctx_.num_bands() - 1, ispn));
203  }
204  }
205  comm().allreduce<double, sddk::mpi_op_t::min>(&emin, 1);
206  comm().allreduce<double, sddk::mpi_op_t::max>(&emax, 1);
207 
208  double ne{0};
209 
210  /* smearing function */
211  auto f = smearing::occupancy(ctx_.smearing(), ctx_.smearing_width());
212 
213  splindex<splindex_t::block> splb(ctx_.num_bands(), ctx_.comm_band().size(), ctx_.comm_band().rank());
214 
215  int step{0};
216  /* calculate occupations */
217  while (std::abs(ne - ne_target) >= 1e-11) {
218  energy_fermi_ = (emin + emax) / 2.0;
219  /* compute total number of electrons */
220  ne = 0.0;
221  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
222  int ik = spl_num_kpoints_[ikloc];
223  double tmp{0};
224  #pragma omp parallel reduction(+:tmp)
225  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
226  #pragma omp for
227  for (int j = 0; j < splb.local_size(); j++) {
228  tmp += f(energy_fermi_ - kpoints_[ik]->band_energy(splb[j], ispn)) * ctx_.max_occupancy();
229  }
230  }
231  ne += tmp * kpoints_[ik]->weight();
232  }
233  ctx_.comm().allreduce(&ne, 1);
234  if (ne > ne_target) {
235  emax = energy_fermi_;
236  } else {
237  emin = energy_fermi_;
238  }
239 
240  if (step > 10000) {
241  std::stringstream s;
242  s << "search of band occupancies failed after 10000 steps";
243  TERMINATE(s);
244  }
245  step++;
246  }
247 
248  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
249  int ik = spl_num_kpoints_[ikloc];
250  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
251  #pragma omp parallel for
252  for (int j = 0; j < ctx_.num_bands(); j++) {
253  auto o = f(energy_fermi_ - kpoints_[ik]->band_energy(j, ispn)) * ctx_.max_occupancy();
254  kpoints_[ik]->band_occupancy(j, ispn, o);
255  }
256  }
257  }
258 
259  this->sync_band<sync_band_t::occupancy>();
260 
261  band_gap_ = 0.0;
262 
263  int nve = static_cast<int>(ne_target + 1e-12);
264  if (ctx_.num_spins() == 2 || (std::abs(nve - ne_target) < 1e-12 && nve % 2 == 0)) {
265  /* find band gap */
266  std::vector<std::pair<double, double>> eband(ctx_.num_bands() * ctx_.num_spinors());
267 
268  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
269  #pragma omp for
270  for (int j = 0; j < ctx_.num_bands(); j++) {
271  std::pair<double, double> eminmax;
272  eminmax.first = std::numeric_limits<double>::max();
273  eminmax.second = std::numeric_limits<double>::lowest();
274 
275  for (int ik = 0; ik < num_kpoints(); ik++) {
276  eminmax.first = std::min(eminmax.first, kpoints_[ik]->band_energy(j, ispn));
277  eminmax.second = std::max(eminmax.second, kpoints_[ik]->band_energy(j, ispn));
278  }
279 
280  eband[j + ispn * ctx_.num_bands()] = eminmax;
281  }
282  }
283 
284  std::sort(eband.begin(), eband.end());
285 
286  int ist = nve;
287  if (ctx_.num_spins() == 1) {
288  ist /= 2;
289  }
290 
291  if (eband[ist].first > eband[ist - 1].second) {
292  band_gap_ = eband[ist].first - eband[ist - 1].second;
293  }
294  }
295 }

297 double K_point_set::valence_eval_sum() const
298 {
299  double eval_sum{0};
300 
301  splindex<splindex_t::block> splb(ctx_.num_bands(), ctx_.comm_band().size(), ctx_.comm_band().rank());
302 
303  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
304  auto ik = spl_num_kpoints_[ikloc];
305  auto const& kp = kpoints_[ik];
306  double tmp{0};
307  #pragma omp parallel for reduction(+:tmp)
308  for (int j = 0; j < splb.local_size(); j++) {
309  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
310  tmp += kp->band_energy(splb[j], ispn) * kp->band_occupancy(splb[j], ispn);
311  }
312  }
313  eval_sum += kp->weight() * tmp;
314  }
315  ctx_.comm().allreduce(&eval_sum, 1);
316 
317  return eval_sum;
318 }

320 double K_point_set::entropy_sum() const
321 {
322  double s_sum{0};
323 
324  double ne_target = ctx_.unit_cell().num_valence_electrons() - ctx_.cfg().parameters().extra_charge();
325 
326  bool only_occ = (ctx_.num_mag_dims() != 1 &&
327  std::abs(ctx_.num_bands() * ctx_.max_occupancy() - ne_target) < 1e-10);
328 
329  if (only_occ) {
330  return 0;
331  }
332 
333  auto f = smearing::entropy(ctx_.smearing(), ctx_.smearing_width());
334 
335  splindex<splindex_t::block> splb(ctx_.num_bands(), ctx_.comm_band().size(), ctx_.comm_band().rank());
336 
337  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
338  auto ik = spl_num_kpoints_[ikloc];
339  auto const& kp = kpoints_[ik];
340  double tmp{0};
341  #pragma omp parallel for reduction(+:tmp)
342  for (int j = 0; j < splb.local_size(); j++) {
343  for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
344  tmp += ctx_.max_occupancy() * f(energy_fermi_ - kp->band_energy(splb[j], ispn));
345  }
346  }
347  s_sum += kp->weight() * tmp;
348  }
349  ctx_.comm().allreduce(&s_sum, 1);
350 
351  return s_sum;
352 }

390 void K_point_set::save(std::string const& name__) const
391 {
392  if (ctx_.comm().rank() == 0) {
393  if (!utils::file_exists(name__)) {
394  HDF5_tree(name__, hdf5_access_t::truncate);
395  }
396  HDF5_tree fout(name__, hdf5_access_t::read_write);
397  fout.create_node("K_point_set");
398  fout["K_point_set"].write("num_kpoints", num_kpoints());
399  }
400  ctx_.comm().barrier();
401  for (int ik = 0; ik < num_kpoints(); ik++) {
402  /* check if this ranks stores the k-point */
403  if (ctx_.comm_k().rank() == spl_num_kpoints_.local_rank(ik)) {
404  kpoints_[ik]->save(name__, ik);
405  }
406  /* wait for all */
407  ctx_.comm().barrier();
408  }
409 }
