52 void K_point_set::create_k_mesh(vector3d<int> k_grid__, vector3d<int> k_shift__, int use_symmetry__)
53 {
54  PROFILE("sirius::K_point_set::create_k_mesh");
55 
56  int nk;
57  mdarray<double, 2> kp;
58  std::vector<double> wk;
59  if (use_symmetry__) {
60  auto result = get_irreducible_reciprocal_mesh(unit_cell_.symmetry(), k_grid__, k_shift__);
61  nk = std::get<0>(result);
62  wk = std::get<1>(result);
63  auto tmp = std::get<2>(result);
64  kp = mdarray<double, 2>(3, nk);
65  for (int i = 0; i < nk; i++) {
66  for (int x : {0, 1, 2}) {
67  kp(x, i) = tmp[i][x];
68  }
69  }
70  } else {
71  nk = k_grid__[0] * k_grid__[1] * k_grid__[2];
72  wk = std::vector<double>(nk, 1.0 / nk);
73  kp = mdarray<double, 2>(3, nk);
74 
75  int ik = 0;
76  for (int i0 = 0; i0 < k_grid__[0]; i0++) {
77  for (int i1 = 0; i1 < k_grid__[1]; i1++) {
78  for (int i2 = 0; i2 < k_grid__[2]; i2++) {
79  kp(0, ik) = double(i0 + k_shift__[0] / 2.0) / k_grid__[0];
80  kp(1, ik) = double(i1 + k_shift__[1] / 2.0) / k_grid__[1];
81  kp(2, ik) = double(i2 + k_shift__[2] / 2.0) / k_grid__[2];
82  ik++;
83  }
84  }
85  }
86  }
87 
88  for (int ik = 0; ik < nk; ik++) {
89  add_kpoint(&kp(0, ik), wk[ik]);
90  }
91 
92  initialize();
93 }

119 void K_point_set::sync_band_occupancies()
120 {
121  int nranks = comm().size();
122  int pid = comm().rank();
123  std::vector<int> nk_per_rank(nranks);
124  for (int i = 0; i < nranks; ++i) {
125  nk_per_rank[i] = spl_num_kpoints_.local_size(i);
126  }
127  int ns = 1;
128  int num_bands = ctx_.num_bands();
129  if (ctx_.num_spin_dims() > 1)
130  ns = 2;
131 
132  std::vector<int> offsets(nranks);
133  std::vector<int> sizes(nranks);
134  int offset = 0;
135  for (int i = 0; i < nranks; ++i) {
136 
137  offsets[i] = offset;
138  int lsize = nk_per_rank[i] * num_bands * ns;
139  sizes[i] = lsize;
140  offset += lsize;
141  }
142  int size = offset;
143 
144  std::vector<double> tmp(sizes[pid]);
145  for (int i = 0; i < sizes[pid]; ++i) {
146  int gi = offsets[pid] + i;
147  int k = gi / (num_bands * ns);
148  int spin = (gi % (num_bands * ns)) / num_bands;
149  int n = (gi % (num_bands * ns)) % num_bands;
150  tmp[i] = kpoints_[k]->band_occupancy(n, spin);
151  }
152 
153  std::vector<double> occupancies(size);
154  comm().allgather(tmp.data(), sizes[pid], occupancies.data(), sizes.data(), offsets.data());
155 
156  for (int i = 0; i < size; ++i) {
157  int k = i / (num_bands * ns);
158  int spin = (i % (num_bands * ns)) / num_bands;
159  int n = (i % (num_bands * ns)) % num_bands;
160  kpoints_[k]->band_occupancy(n, spin, occupancies[i]);
161  }
162 }

164 void K_point_set::find_band_occupancies()
165 {
166  PROFILE("sirius::K_point_set::find_band_occupancies");
167 
168  double ef{0};
169  double de{0.1};
170 
171  int s{1};
172  int sp;
173 
174  mdarray<double, 3> bnd_occ(ctx_.num_bands(), ctx_.num_spin_dims(), num_kpoints());
175 
176  double ne{0};
177 
178  /* target number of electrons */
179  double ne_target = unit_cell_.num_valence_electrons() - ctx_.parameters_input().extra_charge_;
180 
181  int step{0};
182  /* calculate occupations */
183  while (std::abs(ne - ne_target) >= 1e-11) {
184  /* update Efermi */
185  ef += de;
186  /* compute total number of electrons */
187  ne = 0.0;
188  for (int ik = 0; ik < num_kpoints(); ik++) {
189  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
190  for (int j = 0; j < ctx_.num_bands(); j++) {
191  bnd_occ(j, ispn, ik) =
192  smearing::gaussian(kpoints_[ik]->band_energy(j, ispn) - ef, ctx_.smearing_width()) *
193  ctx_.max_occupancy();
194  ne += bnd_occ(j, ispn, ik) * kpoints_[ik]->weight();
195  }
196  }
197  }
198 
199  sp = s;
200  s = (ne > ne_target) ? -1 : 1;
201  /* reduce de step if we change the direction, otherwise increase the step */
202  de = (s != sp) ? (-de * 0.5) : (de * 1.25);
203 
204  if (step > 10000) {
205  std::stringstream s;
206  s << "search of band occupancies failed after 10000 steps";
207  TERMINATE(s);
208  }
209  step++;
210  }
211 
212  energy_fermi_ = ef;
213 
214  for (int ik = 0; ik < num_kpoints(); ik++) {
215  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
216  for (int j = 0; j < ctx_.num_bands(); j++) {
217  kpoints_[ik]->band_occupancy(j, ispn, bnd_occ(j, ispn, ik));
218  }
219  }
220  }
221 
222  band_gap_ = 0.0;
223 
224  int nve = static_cast<int>(ne_target + 1e-12);
225  if (ctx_.num_spins() == 2 || (std::abs(nve - ne_target) < 1e-12 && nve % 2 == 0)) {
226  /* find band gap */
227  std::vector<std::pair<double, double>> eband;
228  std::pair<double, double> eminmax;
229 
230  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
231  for (int j = 0; j < ctx_.num_bands(); j++) {
232  eminmax.first = 1e10;
233  eminmax.second = -1e10;
234 
235  for (int ik = 0; ik < num_kpoints(); ik++) {
236  eminmax.first = std::min(eminmax.first, kpoints_[ik]->band_energy(j, ispn));
237  eminmax.second = std::max(eminmax.second, kpoints_[ik]->band_energy(j, ispn));
238  }
239 
240  eband.push_back(eminmax);
241  }
242  }
243 
244  std::sort(eband.begin(), eband.end());
245 
246  int ist = nve;
247  if (ctx_.num_spins() == 1) {
248  ist /= 2;
249  }
250 
251  if (eband[ist].first > eband[ist - 1].second) {
252  band_gap_ = eband[ist].first - eband[ist - 1].second;
253  }
254  }
255 }

293 void K_point_set::save(std::string const& name__) const
294 {
295  if (ctx_.comm().rank() == 0) {
296  if (!utils::file_exists(name__)) {
297  HDF5_tree(name__, hdf5_access_t::truncate);
298  }
299  HDF5_tree fout(name__, hdf5_access_t::read_write);
300  fout.create_node("K_point_set");
301  fout["K_point_set"].write("num_kpoints", num_kpoints());
302  }
303  ctx_.comm().barrier();
304  for (int ik = 0; ik < num_kpoints(); ik++) {
305  /* check if this ranks stores the k-point */
306  if (ctx_.comm_k().rank() == spl_num_kpoints_.local_rank(ik)) {
307  kpoints_[ik]->save(name__, ik);
308  }
309  /* wait for all */
310  ctx_.comm().barrier();
311  }
312 }
