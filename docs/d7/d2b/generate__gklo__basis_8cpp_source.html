29template <typename T>
30void
31K_point<T>::generate_gklo_basis()
32{
33 /* find local number of row G+k vectors */
34 splindex<splindex_t::block_cyclic> spl_ngk_row(num_gkvec(), num_ranks_row_, rank_row_, ctx_.cyclic_block_size());
35 num_gkvec_row_ = spl_ngk_row.local_size();
36
37 igk_row_.resize(num_gkvec_row_);
38 for (int i = 0; i < num_gkvec_row_; i++) {
39 igk_row_[i] = spl_ngk_row[i];
40 }
41
42 /* find local number of column G+k vectors */
43 splindex<splindex_t::block_cyclic> spl_ngk_col(num_gkvec(), num_ranks_col_, rank_col_, ctx_.cyclic_block_size());
44 num_gkvec_col_ = spl_ngk_col.local_size();
45
46 igk_col_.resize(num_gkvec_col_);
47 for (int i = 0; i < num_gkvec_col_; i++) {
48 igk_col_[i] = spl_ngk_col[i];
49 }
50
51 /* mapping between local and global G+k vecotor indices */
52 igk_loc_.resize(num_gkvec_loc());
53 for (int i = 0; i < num_gkvec_loc(); i++) {
54 igk_loc_[i] = gkvec().offset() + i;
55 }
56
57 if (ctx_.full_potential()) {
58 splindex<splindex_t::block_cyclic> spl_nlo_row(num_gkvec() + unit_cell_.mt_lo_basis_size(), num_ranks_row_,
59 rank_row_, ctx_.cyclic_block_size());
60 splindex<splindex_t::block_cyclic> spl_nlo_col(num_gkvec() + unit_cell_.mt_lo_basis_size(), num_ranks_col_,
61 rank_col_, ctx_.cyclic_block_size());
62
63 lo_basis_descriptor lo_desc;
64
65 int idx{0};
66 /* local orbital basis functions */
67 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
68 auto& atom = unit_cell_.atom(ia);
69 auto& type = atom.type();
70
71 int lo_index_offset = type.mt_aw_basis_size();
72
73 for (int j = 0; j < type.mt_lo_basis_size(); j++) {
74 int l = type.indexb(lo_index_offset + j).l;
75 int lm = type.indexb(lo_index_offset + j).lm;
76 int order = type.indexb(lo_index_offset + j).order;
77 int idxrf = type.indexb(lo_index_offset + j).idxrf;
78 lo_desc.ia = static_cast<uint16_t>(ia);
79 lo_desc.l = static_cast<uint8_t>(l);
80 lo_desc.lm = static_cast<uint16_t>(lm);
81 lo_desc.order = static_cast<uint8_t>(order);
82 lo_desc.idxrf = static_cast<uint8_t>(idxrf);
83
84 if (spl_nlo_row.local_rank(num_gkvec() + idx) == rank_row_) {
85 lo_basis_descriptors_row_.push_back(lo_desc);
86 }
87 if (spl_nlo_col.local_rank(num_gkvec() + idx) == rank_col_) {
88 lo_basis_descriptors_col_.push_back(lo_desc);
89 }
90
91 idx++;
92 }
93 }
94 assert(idx == unit_cell_.mt_lo_basis_size());
95
96 atom_lo_cols_.clear();
97 atom_lo_cols_.resize(unit_cell_.num_atoms());
98 for (int i = 0; i < num_lo_col(); i++) {
99 int ia = lo_basis_descriptor_col(i).ia;
100 atom_lo_cols_[ia].push_back(i);
101 }
102
103 atom_lo_rows_.clear();
104 atom_lo_rows_.resize(unit_cell_.num_atoms());
105 for (int i = 0; i < num_lo_row(); i++) {
106 int ia = lo_basis_descriptor_row(i).ia;
107 atom_lo_rows_[ia].push_back(i);
108 }
109 }
110}
