29 void K_point::generate_gklo_basis()
30 {
31  /* find local number of row G+k vectors */
32  splindex<splindex_t::block_cyclic> spl_ngk_row(num_gkvec(), num_ranks_row_, rank_row_, ctx_.cyclic_block_size());
33  num_gkvec_row_ = spl_ngk_row.local_size();
34 
35  igk_row_.resize(num_gkvec_row_);
36  for (int i = 0; i < num_gkvec_row_; i++) {
37  igk_row_[i] = spl_ngk_row[i];
38  }
39 
40  /* find local number of column G+k vectors */
41  splindex<splindex_t::block_cyclic> spl_ngk_col(num_gkvec(), num_ranks_col_, rank_col_, ctx_.cyclic_block_size());
42  num_gkvec_col_ = spl_ngk_col.local_size();
43 
44  igk_col_.resize(num_gkvec_col_);
45  for (int i = 0; i < num_gkvec_col_; i++) {
46  igk_col_[i] = spl_ngk_col[i];
47  }
48 
49  /* mapping between local and global G+k vecotor indices */
50  igk_loc_.resize(num_gkvec_loc());
51  for (int i = 0; i < num_gkvec_loc(); i++) {
52  igk_loc_[i] = gkvec().offset() + i;
53  }
54 
55  if (ctx_.full_potential()) {
56  splindex<splindex_t::block_cyclic> spl_nlo_row(num_gkvec() + unit_cell_.mt_lo_basis_size(), num_ranks_row_,
57  rank_row_, ctx_.cyclic_block_size());
58  splindex<splindex_t::block_cyclic> spl_nlo_col(num_gkvec() + unit_cell_.mt_lo_basis_size(), num_ranks_col_,
59  rank_col_, ctx_.cyclic_block_size());
60 
61  lo_basis_descriptor lo_desc;
62 
63  int idx{0};
64  /* local orbital basis functions */
65  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
66  auto& atom = unit_cell_.atom(ia);
67  auto& type = atom.type();
68 
69  int lo_index_offset = type.mt_aw_basis_size();
70 
71  for (int j = 0; j < type.mt_lo_basis_size(); j++) {
72  int l = type.indexb(lo_index_offset + j).l;
73  int lm = type.indexb(lo_index_offset + j).lm;
74  int order = type.indexb(lo_index_offset + j).order;
75  int idxrf = type.indexb(lo_index_offset + j).idxrf;
76  lo_desc.ia = static_cast<uint16_t>(ia);
77  lo_desc.l = static_cast<uint8_t>(l);
78  lo_desc.lm = static_cast<uint16_t>(lm);
79  lo_desc.order = static_cast<uint8_t>(order);
80  lo_desc.idxrf = static_cast<uint8_t>(idxrf);
81 
82  if (spl_nlo_row.local_rank(num_gkvec() + idx) == rank_row_) {
83  lo_basis_descriptors_row_.push_back(lo_desc);
84  }
85  if (spl_nlo_col.local_rank(num_gkvec() + idx) == rank_col_) {
86  lo_basis_descriptors_col_.push_back(lo_desc);
87  }
88 
89  idx++;
90  }
91  }
92  assert(idx == unit_cell_.mt_lo_basis_size());
93 
94  atom_lo_cols_.clear();
95  atom_lo_cols_.resize(unit_cell_.num_atoms());
96  for (int i = 0; i < num_lo_col(); i++) {
97  int ia = lo_basis_descriptor_col(i).ia;
98  atom_lo_cols_[ia].push_back(i);
99  }
100 
101  atom_lo_rows_.clear();
102  atom_lo_rows_.resize(unit_cell_.num_atoms());
103  for (int i = 0; i < num_lo_row(); i++) {
104  int ia = lo_basis_descriptor_row(i).ia;
105  atom_lo_rows_[ia].push_back(i);
106  }
107  }
108 }
