30 // BLAS Level 1 includes routines and functions performing vector-vector
31 // operations. Most BLAS 1 routines are about reduction: compute the norm,
32 // calculate the dot production of two vectors, find the maximum/minimum index
33 // of the element of the vector. As you may observed, although the computation
34 // type is different, the core algorithm is the same: scan all element of the
35 // vector(s) and reduce to one single result.
36 //
37 // The reduction algorithm on GPU is called [parallel
38 // reduction](https://raw.githubusercontent.com/mateuszbuda/GPUExample/master/reduce3.png)
39 // which is adopted in rocBLAS. At the beginning, all the threads in the thread
40 // block participate. After each step of reduction (like a tree), the number of
41 // participating threads decrease by half. At the end of the parallel reduction,
42 // only one thread (usually thread 0) owns the result in its thread block.
43 //
44 // Classically, the BLAS 1 reduction needs more than one GPU kernel to finish,
45 // because the lack of global synchronization of thread blocks without exiting
46 // the kernel. The first kernels gather partial results, write into a temporary
47 // working buffer. The second kernel finishes the final reduction.
48 //
49 // For example, BLAS 1 routine i*amax is to find index of the maximum absolute
50 // value element of a vector. In this routine:
51 //
52 // Kernel 1: launch many thread block as needed. Each thread block works on a
53 // subset of the vector. Each thread block use the parallel reduction to find a
54 // local index with the maximum absolute value of the subset. There are
55 // number-of-the-thread-blocks local results.The results are written into a
56 // temporary working buffer. The working buffer has number-of-the-thread-blocks
57 // elements.
58 //
59 // Kernel 2: launch only one thread block which reads the temporary work buffer and
60 // reduces to final result still with the parallel reduction.
61 //
62 // As you may see, if there is a mechanism to synchronize all the thread blocks
63 // after local index is obtained in kernel 1 (without ending the kernel), then
64 // Kernel 2's computation can be merged into Kernel 1. One such mechanism is called
65 // atomic operation. However, atomic operation is new and is not used in rocBLAS
66 // yet. rocBLAS still use the classic standard parallel reduction right now.

68 namespace {
69 // Recursively compute reduction
70 template <rocblas_int k, typename REDUCE, typename T>
71 struct rocblas_reduction_s
72 {
73  __forceinline__ __device__ void operator()(rocblas_int tx, T* x)
74  {
75  // Reduce the lower half with the upper half
76  if(tx < k)
77  REDUCE{}(x[tx], x[tx + k]);
78  __syncthreads();
79 
80  // Recurse down with k / 2
81  rocblas_reduction_s<k / 2, REDUCE, T>{}(tx, x);
82  }
83 };
84 
85 // leaf node for terminating recursion
86 template <typename REDUCE, typename T>
87 struct rocblas_reduction_s<0, REDUCE, T>
88 {
89  __forceinline__ __device__ void operator()(rocblas_int tx, T* x) {}
90 };

92 /*! \brief general parallel reduction
93 
94  \details
95 
96  @param[in]
97  n rocblas_int. assume a power of 2
98  @param[in]
99  T element type of vector x
100  @param[in]
101  REDUCE reduction functor
102  @param[in]
103  tx rocblas_int. thread id
104  @param[inout]
105  x pointer storing vector x on the GPU.
106  usually x is stored in shared memory;
107  x[0] store the final result.
108  ********************************************************************/
109 template <rocblas_int NB, typename REDUCE, typename T>
110 __attribute__((flatten)) __device__ void rocblas_reduction(rocblas_int tx, T* x)
111 {
112  static_assert(NB > 1 && !(NB & (NB - 1)), "NB must be a power of 2");
113  __syncthreads();
114  rocblas_reduction_s<NB / 2, REDUCE, T>{}(tx, x);
115 }

163 // kennel 1 writes partial results per thread block in workspace; number of partial results is
164 // blocks
165 template <rocblas_int NB,
166  typename FETCH,
167  typename REDUCE = rocblas_reduce_sum,
168  typename Ti,
169  typename To>
170 __global__ void
171 rocblas_reduction_kernel_part1(rocblas_int n, const Ti* x, rocblas_int incx, To* workspace)
172 {
173  ssize_t tx = hipThreadIdx_x;
174  ssize_t tid = hipBlockIdx_x * hipBlockDim_x + tx;
175  __shared__ To tmp[NB];
176 
177  // bound
178  if(tid < n)
179  tmp[tx] = FETCH{}(x[tid * incx], tid);
180  else
181  tmp[tx] = default_value<To>{}(); // pad with default value
182 
183  rocblas_reduction<NB, REDUCE>(tx, tmp);
184 
185  if(tx == 0)
186  workspace[hipBlockIdx_x] = tmp[0];
187 }
188 
189 // kernel 2 gathers all the partial results in workspace and finishes the final reduction;
190 // number of threads (NB) loop blocks
191 template <rocblas_int NB,
192  typename REDUCE = rocblas_reduce_sum,
193  typename FINALIZE = rocblas_finalize_identity,
194  typename To,
195  typename Tr>
196 __global__ void rocblas_reduction_kernel_part2(rocblas_int nblocks, To* workspace, Tr* result)
197 {
198  rocblas_int tx = hipThreadIdx_x;
199  __shared__ To tmp[NB];
200 
201  if(tx < nblocks)
202  {
203  tmp[tx] = workspace[tx];
204 
205  // bound, loop
206  for(rocblas_int i = tx + NB; i < nblocks; i += NB)
207  REDUCE{}(tmp[tx], workspace[i]);
208  }
209  else
210  { // pad with default value
211  tmp[tx] = default_value<To>{}();
212  }
213 
214  if(nblocks < 32)
215  {
216  // no need parallel reduction
217  __syncthreads();
218 
219  if(tx == 0)
220  for(rocblas_int i = 1; i < nblocks; i++)
221  REDUCE{}(tmp[0], tmp[i]);
222  }
223  else
224  {
225  // parallel reduction
226  rocblas_reduction<NB, REDUCE>(tx, tmp);
227  }
228 
229  // Store result on device or in workspace
230  if(tx == 0)
231  *result = FINALIZE{}(tmp[0]);
232 }

234 // At least two kernels are needed to finish the reduction
235 // kennel 1 write partial result per thread block in workspace, blocks partial results
236 // kernel 2 gathers all the partial result in workspace and finishes the final reduction.
237 template <rocblas_int NB,
238  typename FETCH,
239  typename REDUCE = rocblas_reduce_sum,
240  typename FINALIZE = rocblas_finalize_identity,
241  typename Ti,
242  typename To,
243  typename Tr>
244 rocblas_status rocblas_reduction_kernel(rocblas_handle __restrict__ handle,
245  rocblas_int n,
246  const Ti* x,
247  rocblas_int incx,
248  Tr* result,
249  To* workspace,
250  rocblas_int blocks)
251 {
252  hipLaunchKernelGGL((rocblas_reduction_kernel_part1<NB, FETCH, REDUCE>),
253  blocks,
254  NB,
255  0,
256  handle->rocblas_stream,
257  n,
258  x,
259  incx,
260  workspace);
261 
262  if(handle->pointer_mode == rocblas_pointer_mode_device)
263  {
264  hipLaunchKernelGGL((rocblas_reduction_kernel_part2<NB, REDUCE, FINALIZE>),
265  1,
266  NB,
267  0,
268  handle->rocblas_stream,
269  blocks,
270  workspace,
271  result);
272  }
273  else
274  {
275  // If in host pointer mode, workspace is converted to Tr* and the result is
276  // placed there, and then copied from device to host. If To is a class type,
277  // it must be a standard layout type and its first member must be of type Tr.
278  static_assert(std::is_standard_layout<To>(), "To must be a standard layout type");
279 
280  if(blocks > 1)
281  {
282  hipLaunchKernelGGL((rocblas_reduction_kernel_part2<NB, REDUCE, FINALIZE>),
283  1,
284  NB,
285  0,
286  handle->rocblas_stream,
287  blocks,
288  workspace,
289  (Tr*)workspace);
290  }
291 
292  if(std::is_same<FINALIZE, rocblas_finalize_identity>() || blocks > 1)
293  {
294  // If FINALIZE is trivial or kernel part2 was called, result is in the
295  // beginning of workspace[0], and can be copied directly.
296  RETURN_IF_HIP_ERROR(hipMemcpy(result, workspace, sizeof(Tr), hipMemcpyDeviceToHost));
297  }
298  else
299  {
300  // If FINALIZE is not trivial and kernel part2 was not called, then
301  // workspace[0] needs to be finalized on host.
302  To res;
303  RETURN_IF_HIP_ERROR(hipMemcpy(&res, workspace, sizeof(To), hipMemcpyDeviceToHost));
304  *result = FINALIZE{}(res);
305  }
306  }
307 
308  return rocblas_status_success;
309 }
