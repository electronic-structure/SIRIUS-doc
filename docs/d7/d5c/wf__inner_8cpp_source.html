41// If scalar type, g-vector 0 contribution must be scaled before / after inner product to aboid counting twice
42template <typename T>
43void scale_gamma_wf(spin_range spins, int m, int i0, real_type<T> alpha, Wave_functions<real_type<T>>& bra) {
44 for (auto s : spins) {
45 const int incx = bra.pw_coeffs(s).prime().ld() * 2; // complex matrix is read as scalar
46 if (bra.preferred_memory_t() == memory_t::device) {
47#if defined(SIRIUS_GPU)
48 if (std::is_same<T, double>::value) {
49 accblas::dscal(m, reinterpret_cast<double*>(&alpha),
50 reinterpret_cast<double*>(bra.pw_coeffs(s).prime().at(bra.preferred_memory_t(), 0, i0)),
51 incx);
52 } else if (std::is_same<T, float>::value) {
53 accblas::sscal(m, reinterpret_cast<float*>(&alpha),
54 reinterpret_cast<float*>(bra.pw_coeffs(s).prime().at(bra.preferred_memory_t(), 0, i0)),
55 incx);
56 }
57#else
58 throw std::runtime_error("not compiled with GPU support!");
59#endif
60 } else {
61 if (std::is_same<T, double>::value) {
62 FORTRAN(dscal)
63 (&m, reinterpret_cast<double*>(&alpha),
64 reinterpret_cast<double*>(bra.pw_coeffs(s).prime().at(bra.preferred_memory_t(), 0, i0)), &incx);
65 } else if (std::is_same<T, float>::value) {
66 FORTRAN(sscal)
67 (&m, reinterpret_cast<float*>(&alpha),
68 reinterpret_cast<float*>(bra.pw_coeffs(s).prime().at(bra.preferred_memory_t(), 0, i0)), &incx);
69 }
70 }
71 }
72}

93// special for complex type
94template <typename T>
95void
96inner_mt(::spla::Context& spla_ctx__, ::spla::MatrixDistribution& spla_mat_dist__, spin_range ispn__,
97 Wave_functions<T>& bra__, int i0__, int m__, Wave_functions<T>& ket__,
98 int j0__, int n__, dmatrix<std::complex<T>>& result__, int irow0__, int jcol0__)
99{
100 bool local_has_mt = bra__.has_mt();
101 bool global_has_mt = false;
102
103 // Not all ranks may have mt, but all must call spla if at least one does
104 MPI_Allreduce(&local_has_mt, &global_has_mt, 1, MPI_C_BOOL, MPI_LOR, bra__.comm().mpi_comm());
105 if (global_has_mt) {
106 std::complex<T>* result_ptr = result__.size_local() ? result__.at(memory_t::host, 0, 0) : nullptr;
107 auto spins = spin_range(ispn__);
108 for (auto s : spins) {
109 PROFILE("sddk::wf_inner|mt");
110 if (local_has_mt) {
111 spla::pgemm_ssb(
112 m__, n__, bra__.mt_coeffs(s).num_rows_loc(), SPLA_OP_CONJ_TRANSPOSE, 1.0,
113 bra__.mt_coeffs(s).prime().at(bra__.preferred_memory_t(), 0, i0__), bra__.mt_coeffs(s).prime().ld(),
114 ket__.mt_coeffs(s).prime().at(ket__.preferred_memory_t(), 0, j0__), ket__.mt_coeffs(s).prime().ld(),
115 1.0, result_ptr, result__.ld(), irow0__, jcol0__, spla_mat_dist__, spla_ctx__);
116 } else {
117 spla::pgemm_ssb(m__, n__, 0, SPLA_OP_CONJ_TRANSPOSE, 1.0, nullptr, 0, nullptr, 0, 1.0,
118 result__.at(memory_t::host, 0, 0), result__.ld(), irow0__, jcol0__, spla_mat_dist__,
119 spla_ctx__);
120 }
121 }
122 }

127template <typename T>
128void
129inner(::spla::Context& spla_ctx__, ::sddk::spin_range spins__, Wave_functions<real_type<T>>& bra__, int i0__, int m__,
130 Wave_functions<real_type<T>>& ket__, int j0__, int n__, dmatrix<T>& result__, int irow0__, int jcol0__)
131{
132 PROFILE("sddk::wf_inner");
133
134 spla::MatrixDistribution spla_mat_dist = bra__.comm().size() > result__.comm().size()
135 ? spla::MatrixDistribution::create_mirror(bra__.comm().mpi_comm())
136 : result__.spla_distribution();
137
138 using precision_type = real_type<T>;
139 precision_type alpha = 1.0;
140 int size_factor = 1;
141 if (std::is_same<T, precision_type>::value) {
142 alpha = 2.0;
143 size_factor = 2;
144 }
145
146 // For gamma case, contribution of g = 0 vector must not be counted double -> multiply by 0.5
147 if (bra__.comm().rank() == 0) {
148 PROFILE("sddk::wf_inner|scale");
149 scale_gamma_wf<T>(spins__, m__, i0__, 0.5, bra__);
150 }
151
152 precision_type beta = 0.0;
153
154 T* result_ptr = reinterpret_cast<T*>(result__.size_local() ? result__.at(memory_t::host, 0, 0) : nullptr);
155
156 for (auto s : spins__) {
157 PROFILE("sddk::wf_inner|pw");
158 spla::pgemm_ssb(m__, n__, size_factor * bra__.pw_coeffs(s).num_rows_loc(), SPLA_OP_CONJ_TRANSPOSE, alpha,
159 reinterpret_cast<const T*>(bra__.pw_coeffs(s).prime().at(bra__.preferred_memory_t(), 0, i0__)),
160 size_factor * bra__.pw_coeffs(s).prime().ld(),
161 reinterpret_cast<const T*>(ket__.pw_coeffs(s).prime().at(ket__.preferred_memory_t(), 0, j0__)),
162 size_factor * ket__.pw_coeffs(s).prime().ld(), beta, result_ptr, result__.ld(), irow0__,
163 jcol0__, spla_mat_dist, spla_ctx__);
164 beta = 1.0;
165 }
166
167 // For gamma case, g = 0 vector is rescaled back
168 if (bra__.comm().rank() == 0) {
169 PROFILE("sddk::wf_inner|scale_back");
170 scale_gamma_wf<T>(spins__, m__, i0__, 2.0, bra__);
171 }
172
173 // add mt contribution
174 inner_mt(spla_ctx__, spla_mat_dist, spins__, bra__, i0__, m__, ket__, j0__, n__, result__, irow0__, jcol0__);
175
176 // make sure result is updated on device as well
177 if (result__.on_device()) {
178 result__.copy_to(memory_t::device);
179 }
180}
