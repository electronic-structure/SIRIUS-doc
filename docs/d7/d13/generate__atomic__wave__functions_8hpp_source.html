29 inline void K_point::generate_atomic_wave_functions_aux(const int num_ao__,
30  Wave_functions& phi,
31  std::vector<int>& offset,
32  const bool hubbard)
33 {
34  if (!num_ao__) {
35  return;
36  }
37 
38  int lmax{0};
39  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
40  auto& atom_type = unit_cell_.atom_type(iat);
41  lmax = std::max(lmax, atom_type.lmax_ps_atomic_wf());
42  }
43  lmax = std::max(lmax, unit_cell_.lmax());
44 
45  #pragma omp parallel for schedule(static)
46  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
47  /* global index of G+k vector */
48  int igk = this->idxgk(igk_loc);
49  /* vs = {r, theta, phi} */
50  auto vs = SHT::spherical_coordinates(this->gkvec().gkvec_cart<index_domain_t::local>(igk_loc));
51  /* compute real spherical harmonics for G+k vector */
52  std::vector<double> rlm(utils::lmmax(lmax));
53  SHT::spherical_harmonics(lmax, vs[1], vs[2], &rlm[0]);
54  /* get values of radial integrals for a given G+k vector length */
55  std::vector<mdarray<double, 1>> ri_values(unit_cell_.num_atom_types());
56  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
57  ri_values[iat] = ctx_.atomic_wf_ri().values(iat, vs[0]);
58  }
59 
60  int n{0};
61  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
62  auto phase = twopi * dot(gkvec().gkvec(igk), unit_cell_.atom(ia).position());
63  auto phase_factor = std::exp(double_complex(0.0, phase));
64  auto& atom_type = unit_cell_.atom(ia).type();
65  if (!hubbard) {
66  for (int i = 0; i < atom_type.num_ps_atomic_wf(); i++) {
67  auto l = std::abs(atom_type.ps_atomic_wf(i).first);
68  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
69  for (int m = -l; m <= l; m++) {
70  int lm = utils::lm(l, m);
71  phi.pw_coeffs(0).prime(igk_loc, n) = z * std::conj(phase_factor) * rlm[lm] * ri_values[atom_type.id()][i];
72  n++;
73  }
74  } // i
75  } else {
76  if (atom_type.hubbard_correction()) {
77  if (atom_type.spin_orbit_coupling()) {
78  // one channel only now
79  for (int i = 0; i < 2; i++) {
80  auto &orb = atom_type.hubbard_orbital(i);
81  const int l = std::abs(orb.hubbard_l());
82  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
83  for (int m = -l; m <= l; m++) {
84  int lm = utils::lm(l, m);
85  phi.pw_coeffs(0).prime(igk_loc, offset[ia] + l + m) += 0.5 * z * std::conj(phase_factor) * rlm[lm] * ri_values[atom_type.id()][orb.rindex()];
86  phi.pw_coeffs(1).prime(igk_loc, offset[ia] + 3 * l + m + 1) += 0.5 * z * std::conj(phase_factor) * rlm[lm] * ri_values[atom_type.id()][orb.rindex()];
87  }
88  }
89  } else {
90  // add the loop over different channels. need to compute the offsets accordingly
91  for (int channel = 0, offset__ = 0; channel < atom_type.number_of_hubbard_channels(); channel++) {
92  auto &orb = atom_type.hubbard_orbital(channel);
93  const int l = std::abs(orb.hubbard_l());
94  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
95  for (int m = -l; m <= l; m++) {
96  int lm = utils::lm(l, m);
97  phi.pw_coeffs(0).prime(igk_loc, offset[ia] + offset__ + l + m) = z * std::conj(phase_factor) * rlm[lm] * ri_values[atom_type.id()][orb.rindex()];
98  if (ctx_.num_mag_dims() == 3) {
99  phi.pw_coeffs(1).prime(igk_loc, offset[ia] + offset__ + 3 * l + m + 1) = z * std::conj(phase_factor) * rlm[lm] * ri_values[atom_type.id()][orb.rindex()];
100  }
101  }
102  offset__ += (ctx_.num_mag_dims() == 3) ? (2 * (2 * l + 1)) : (2 * l + 1);
103  }
104  }
105  }
106  }
107  }
108  } // igk_loc
109 }

117 inline void K_point::compute_gradient_wave_functions(Wave_functions& phi,
118  const int starting_position_i,
119  const int num_wf,
120  Wave_functions& dphi,
121  const int starting_position_j,
122  const int direction) {
123  std::vector<double_complex> qalpha(this->num_gkvec_loc());
124 
125  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
126  auto G = this->gkvec().gkvec_cart<index_domain_t::local>(igk_loc);
127 
128  qalpha[igk_loc] = double_complex(0.0, -G[direction]);
129  }
130 
131  #pragma omp parallel for schedule(static)
132  for (int nphi = 0; nphi < num_wf; nphi++) {
133  for (int ispn = 0; ispn < phi.num_sc(); ispn++) {
134  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
135  dphi.pw_coeffs(ispn).prime(igk_loc, nphi + starting_position_j) = qalpha[igk_loc] *
136  phi.pw_coeffs(ispn).prime(igk_loc, nphi + starting_position_i);
137  }
138  }
139  }
