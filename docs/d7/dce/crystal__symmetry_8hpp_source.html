42/// Descriptor of the space group symmetry operation.
43struct space_group_symmetry_descriptor
44{

46 matrix3d<int> R;

48 /// Inverse of R.
49 matrix3d<int> invR;

51 /// Inverse transposed of R.
52 matrix3d<int> invRT;

58 matrix3d<double> Rc;

274/** \page sym Symmetry
275 * \section section1 Definition of symmetry operation
276 *
277 * SIRIUS uses Spglib to find the spacial symmetry operations. Spglib defines symmetry operation in fractional
278 * coordinates:
279 * \f[
280 * {\bf x'} = \{ {\bf R} | {\bf t} \} {\bf x} \equiv {\bf R}{\bf x} + {\bf t}
281 * \f]
282 * where \b R is the proper or improper rotation matrix with elements equal to -1,0,1 and determinant of 1
283 * (pure rotation) or -1 (rotoreflection) and \b t is the fractional translation, associated with the symmetry
284 * operation. The inverse of the symmetry operation is:
285 * \f[
286 * {\bf x} = \{ {\bf R} | {\bf t} \}^{-1} {\bf x'} = {\bf R}^{-1} ({\bf x'} - {\bf t}) =
287 * {\bf R}^{-1} {\bf x'} - {\bf R}^{-1} {\bf t}
288 * \f]
289 *
290 * We will always use an \a active transformation (transformation of vectors or functions) and never a passive
291 * transformation (transformation of coordinate system). However one should remember definition of the function
292 * transformation:
293 * \f[
294 * \hat {\bf P} f({\bf r}) \equiv f(\hat {\bf P}^{-1} {\bf r})
295 * \f]
296 *
297 * It is straightforward to get the rotation matrix in Cartesian coordinates. We know how the vector in Cartesian
298 * coordinates is obtained from the vector in fractional coordinates:
299 * \f[
300 * {\bf v} = {\bf L} {\bf x}
301 * \f]
302 * where \b L is the 3x3 matrix which clomuns are three lattice vectors. The backward transformation is simply
303 * \f[
304 * {\bf x} = {\bf L}^{-1} {\bf v}
305 * \f]
306 * Now we write rotation operation in fractional coordinates and apply the backward transformation to Cartesian
307 * coordinates:
308 * \f[
309 * {\bf x'} = {\bf R}{\bf x} \rightarrow {\bf L}^{-1} {\bf v'} = {\bf R} {\bf L}^{-1} {\bf v}
310 * \f]
311 * from which we derive the rotation operation in Cartesian coordinates:
312 * \f[
313 * {\bf v'} = {\bf L} {\bf R} {\bf L}^{-1} {\bf v}
314 * \f]
315 */
316
