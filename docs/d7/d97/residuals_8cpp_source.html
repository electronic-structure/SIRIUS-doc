33 static void
34 compute_residuals(sddk::memory_t mem_type__, sddk::spin_range spins__, int num_bands__, sddk::mdarray<double, 1>& eval__,
35  sddk::Wave_functions& hpsi__, sddk::Wave_functions& opsi__, sddk::Wave_functions& res__)
36 {
37  for (int ispn: spins__) {
38  if (is_host_memory(mem_type__)) {
39  /* compute residuals r_{i} = H\Psi_{i} - E_{i}O\Psi_{i} */
40  #pragma omp parallel for
41  for (int i = 0; i < num_bands__; i++) {
42  for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
43  res__.pw_coeffs(ispn).prime(ig, i) = hpsi__.pw_coeffs(ispn).prime(ig, i) -
44  eval__[i] * opsi__.pw_coeffs(ispn).prime(ig, i);
45  }
46  if (res__.has_mt()) {
47  for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
48  res__.mt_coeffs(ispn).prime(j, i) = hpsi__.mt_coeffs(ispn).prime(j, i) -
49  eval__[i] * opsi__.mt_coeffs(ispn).prime(j, i);
50  }
51  }
52  }
53  } else {
54 #if defined(__GPU)
55  compute_residuals_gpu(hpsi__.pw_coeffs(ispn).prime().at(memory_t::device),
56  opsi__.pw_coeffs(ispn).prime().at(memory_t::device),
57  res__.pw_coeffs(ispn).prime().at(memory_t::device),
58  res__.pw_coeffs(ispn).num_rows_loc(),
59  num_bands__,
60  eval__.at(memory_t::device));
61  if (res__.has_mt()) {
62  compute_residuals_gpu(hpsi__.mt_coeffs(ispn).prime().at(memory_t::device),
63  opsi__.mt_coeffs(ispn).prime().at(memory_t::device),
64  res__.mt_coeffs(ispn).prime().at(memory_t::device),
65  res__.mt_coeffs(ispn).num_rows_loc(),
66  num_bands__,
67  eval__.at(memory_t::device));
68  }
69 #endif
70  }
71  }
72 }

74 /// Apply preconditioner to the residuals.
75 static void
76 apply_preconditioner(memory_t mem_type__, spin_range spins__, int num_bands__, Wave_functions& res__,
77  mdarray<double, 2> const& h_diag__, mdarray<double, 2> const& o_diag__, mdarray<double, 1>& eval__)
78 {
79  for (int ispn: spins__) {
80  if (is_host_memory(mem_type__)) {
81  #pragma omp parallel for schedule(static)
82  for (int i = 0; i < num_bands__; i++) {
83  for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
84  double p = h_diag__(ig, ispn) - o_diag__(ig, ispn) * eval__[i];
85  p = 0.5 * (1 + p + std::sqrt(1 + (p - 1) * (p - 1)));
86  res__.pw_coeffs(ispn).prime(ig, i) /= p;
87  }
88  if (res__.has_mt()) {
89  for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
90  double p = h_diag__(res__.pw_coeffs(ispn).num_rows_loc() + j, ispn) -
91  o_diag__(res__.pw_coeffs(ispn).num_rows_loc() + j, ispn) * eval__[i];
92  p = 0.5 * (1 + p + std::sqrt(1 + (p - 1) * (p - 1)));
93  res__.mt_coeffs(ispn).prime(j, i) /= p;
94  }
95  }
96  }
97  } else {
98 #if defined(__GPU)
99  apply_preconditioner_gpu(res__.pw_coeffs(ispn).prime().at(memory_t::device),
100  res__.pw_coeffs(ispn).num_rows_loc(),
101  num_bands__,
102  eval__.at(memory_t::device),
103  h_diag__.at(memory_t::device, 0, ispn),
104  o_diag__.at(memory_t::device, 0, ispn));
105  if (res__.has_mt()) {
106  apply_preconditioner_gpu(res__.mt_coeffs(ispn).prime().at(memory_t::device),
107  res__.mt_coeffs(ispn).num_rows_loc(),
108  num_bands__,
109  eval__.at(memory_t::device),
110  h_diag__.at(memory_t::device, res__.pw_coeffs(ispn).num_rows_loc(), ispn),
111  o_diag__.at(memory_t::device, res__.pw_coeffs(ispn).num_rows_loc(), ispn));
112  }
113 #endif
114  }
115  }

118 template <typename T>
119 static inline int
120 normalized_preconditioned_residuals(memory_t mem_type__, spin_range spins__, int num_bands__, mdarray<double,1>& eval__,
121  Wave_functions& hpsi__, Wave_functions& opsi__, Wave_functions& res__,
122  mdarray<double, 2> const& h_diag__, mdarray<double, 2> const& o_diag__,
123  double norm_tolerance__)
124 {
125  PROFILE("sirius::normalized_preconditioned_residuals");
126 
127  assert(num_bands__ != 0);
128 
129  auto pu = get_device_t(mem_type__);
130 
131  /* compute "raw" residuals */
132  compute_residuals(mem_type__, spins__, num_bands__, eval__, hpsi__, opsi__, res__);
133 
134  /* compute norm of the "raw" residuals */
135  auto res_norm = res__.l2norm(pu, spins__, num_bands__);
136 
137  /* apply preconditioner */
138  apply_preconditioner(mem_type__, spins__, num_bands__, res__, h_diag__, o_diag__, eval__);
139 
140  /* this not strictly necessary as the wave-function orthoronormalization can take care of this;
141  however, normalization of residuals is harmless and gives a better numerical stability */
142  res__.normalize(pu, spins__, num_bands__);
143 
144  int n{0};
145  for (int i = 0; i < num_bands__; i++) {
146  /* take the residual if it's norm is above the threshold */
147  if (res_norm[i] > norm_tolerance__) {
148  /* shift unconverged residuals to the beginning of array */
149  if (n != i) {
150  for (int ispn: spins__) {
151  res__.copy_from(res__, 1, ispn, i, ispn, n);
152  }
153  }
154  n++;
155  }
156  }
157 
158  /* prevent numerical noise */
159  /* this only happens for real wave-functions (Gamma-point case), non-magnetic or collinear magnetic */
160  if (std::is_same<T, double>::value && res__.comm().rank() == 0 && n != 0 && spins__() != 2) {
161  if (is_device_memory(res__.preferred_memory_t())) {
162 #if defined(__GPU)
163  make_real_g0_gpu(res__.pw_coeffs(spins__()).prime().at(memory_t::device), res__.pw_coeffs(spins__()).prime().ld(), n);
164 #endif
165  } else {
166  for (int i = 0; i < n; i++) {
167  res__.pw_coeffs(spins__()).prime(0, i) = res__.pw_coeffs(spins__()).prime(0, i).real();
168  }
169  }
170  }
171 
172  return n;
173 }

175 /// Compute residuals from eigen-vectors.
176 template <typename T>
177 int
178 residuals(memory_t mem_type__, linalg_t la_type__, int ispn__, int N__, int num_bands__, mdarray<double, 1>& eval__,
179  dmatrix<T>& evec__, Wave_functions& hphi__, Wave_functions& ophi__, Wave_functions& hpsi__,
180  Wave_functions& opsi__, Wave_functions& res__, mdarray<double, 2> const& h_diag__,
181  mdarray<double, 2> const& o_diag__, bool estimate_eval__, double norm_tolerance__,
182  std::function<bool(int, int)> is_converged__)
183 {
184  PROFILE("sirius::residuals");
185 
186  assert(N__ != 0);
187 
188  mdarray<double, 1> res_norm;
189  dmatrix<T> evec_tmp;
190  mdarray<double, 1> eval_tmp;
191 
192  dmatrix<T>* evec_ptr{nullptr};
193  mdarray<double, 1>* eval_ptr{nullptr};
194 
195  int n{0};
196  if (estimate_eval__) {
197  std::vector<int> ev_idx;
198  for (int j = 0; j < num_bands__; j++) {
199  if (!is_converged__(j, ispn__)) {
200  ev_idx.push_back(j);
201  }
202  }
203 
204  n = static_cast<int>(ev_idx.size());
205 
206  if (n) {
207  eval_tmp = mdarray<double, 1>(n);
208  eval_ptr = &eval_tmp;
209  evec_tmp = dmatrix<T>(N__, n, evec__.blacs_grid(), evec__.bs_row(), evec__.bs_col());
210  evec_ptr = &evec_tmp;
211 
212  int num_rows_local = evec_tmp.num_rows_local();
213  for (int j = 0; j < n; j++) {
214  eval_tmp[j] = eval__[ev_idx[j]];
215  if (evec__.blacs_grid().comm().size() == 1) {
216  /* do a local copy */
217  std::copy(&evec__(0, ev_idx[j]), &evec__(0, ev_idx[j]) + num_rows_local, &evec_tmp(0, j));
218  } else {
219  auto pos_src = evec__.spl_col().location(ev_idx[j]);
220  auto pos_dest = evec_tmp.spl_col().location(j);
221  /* do MPI send / recieve */
222  if (pos_src.rank == evec__.blacs_grid().comm_col().rank()) {
223  evec__.blacs_grid().comm_col().isend(&evec__(0, pos_src.local_index), num_rows_local, pos_dest.rank, ev_idx[j]);
224  }
225  if (pos_dest.rank == evec__.blacs_grid().comm_col().rank()) {
226  evec__.blacs_grid().comm_col().recv(&evec_tmp(0, pos_dest.local_index), num_rows_local, pos_src.rank, ev_idx[j]);
227  }
228  }
229  }
230  if (is_device_memory(mem_type__) && evec_tmp.blacs_grid().comm().size() == 1) {
231  evec_tmp.allocate(memory_t::device);
232  }
233  if (is_device_memory(mem_type__)) {
234  eval_tmp.allocate(memory_t::device).copy_to(memory_t::device);
235  }
236  }
237  } else { /* compute all residuals first */
238  if (is_device_memory(mem_type__)) {
239  eval__.allocate(memory_t::device).copy_to(memory_t::device);
240  }
241  evec_ptr = &evec__;
242  eval_ptr = &eval__;
243  n = num_bands__;
244  }
245  if (!n) {
246  return 0;
247  }
248 
249  /* compute H\Psi_{i} = \sum_{mu} H\phi_{mu} * Z_{mu, i} and O\Psi_{i} = \sum_{mu} O\phi_{mu} * Z_{mu, i} */
250  transform<T>(mem_type__, la_type__, ispn__, {&hphi__, &ophi__}, 0, N__, *evec_ptr, 0, 0, {&hpsi__, &opsi__}, 0, n);
251 
252  n = normalized_preconditioned_residuals<T>(mem_type__, spin_range(ispn__), n, *eval_ptr, hpsi__, opsi__, res__,
253  h_diag__, o_diag__, norm_tolerance__);
254 
255  return n;
256 }
