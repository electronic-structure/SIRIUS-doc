71template <typename T, typename = std::enable_if_t<std::is_scalar<T>::value>>
72static void
73compute_residuals(sddk::memory_t mem_type__, sddk::spin_range spins__, int num_bands__, sddk::mdarray<T, 1>& eval__,
74 sddk::Wave_functions<T>& hpsi__, sddk::Wave_functions<T>& opsi__, sddk::Wave_functions<T>& res__)
75{
76 for (int ispn: spins__) {
77 if (is_host_memory(mem_type__)) {
78 /* compute residuals r_{i} = H\Psi_{i} - E_{i}O\Psi_{i} */
79 #pragma omp parallel for
80 for (int i = 0; i < num_bands__; i++) {
81 for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
82 res__.pw_coeffs(ispn).prime(ig, i) = hpsi__.pw_coeffs(ispn).prime(ig, i) -
83 eval__[i] * opsi__.pw_coeffs(ispn).prime(ig, i);
84 }
85 if (res__.has_mt()) {
86 for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
87 res__.mt_coeffs(ispn).prime(j, i) = hpsi__.mt_coeffs(ispn).prime(j, i) -
88 eval__[i] * opsi__.mt_coeffs(ispn).prime(j, i);
89 }
90 }
91 }
92 } else {
93#if defined(SIRIUS_GPU)
94 compute_residuals_gpu(hpsi__.pw_coeffs(ispn).prime().at(sddk::memory_t::device),
95 opsi__.pw_coeffs(ispn).prime().at(sddk::memory_t::device),
96 res__.pw_coeffs(ispn).prime().at(sddk::memory_t::device),
97 res__.pw_coeffs(ispn).num_rows_loc(),
98 num_bands__,
99 eval__.at(sddk::memory_t::device));
100 if (res__.has_mt()) {
101 compute_residuals_gpu(hpsi__.mt_coeffs(ispn).prime().at(sddk::memory_t::device),
102 opsi__.mt_coeffs(ispn).prime().at(sddk::memory_t::device),
103 res__.mt_coeffs(ispn).prime().at(sddk::memory_t::device),
104 res__.mt_coeffs(ispn).num_rows_loc(),
105 num_bands__,
106 eval__.at(sddk::memory_t::device));
107 }
108#endif
109 }
110 }
111}

113/// Apply preconditioner to the residuals.
114template <typename T, typename = std::enable_if_t<std::is_scalar<T>::value>>
115static void
116apply_preconditioner(sddk::memory_t mem_type__, sddk::spin_range spins__, int num_bands__, sddk::Wave_functions<T>& res__,
117 sddk::mdarray<T, 2> const& h_diag__, sddk::mdarray<T, 2> const& o_diag__,
118 sddk::mdarray<T, 1>& eval__)
119{
120 for (int ispn: spins__) {
121 if (is_host_memory(mem_type__)) {
122 #pragma omp parallel for schedule(static)
123 for (int i = 0; i < num_bands__; i++) {
124 for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
125 T p = h_diag__(ig, ispn) - o_diag__(ig, ispn) * eval__[i];
126 p = 0.5 * (1 + p + std::sqrt(1 + (p - 1) * (p - 1)));
127 res__.pw_coeffs(ispn).prime(ig, i) /= p;
128 }
129 if (res__.has_mt()) {
130 for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
131 T p = h_diag__(res__.pw_coeffs(ispn).num_rows_loc() + j, ispn) -
132 o_diag__(res__.pw_coeffs(ispn).num_rows_loc() + j, ispn) * eval__[i];
133 p = 0.5 * (1 + p + std::sqrt(1 + (p - 1) * (p - 1)));
134 res__.mt_coeffs(ispn).prime(j, i) /= p;
135 }
136 }
137 }
138 } else {
139#if defined(SIRIUS_GPU)
140 apply_preconditioner_gpu(res__.pw_coeffs(ispn).prime().at(sddk::memory_t::device),
141 res__.pw_coeffs(ispn).num_rows_loc(),
142 num_bands__,
143 eval__.at(sddk::memory_t::device),
144 h_diag__.at(sddk::memory_t::device, 0, ispn),
145 o_diag__.at(sddk::memory_t::device, 0, ispn));
146 if (res__.has_mt()) {
147 apply_preconditioner_gpu(res__.mt_coeffs(ispn).prime().at(sddk::memory_t::device),
148 res__.mt_coeffs(ispn).num_rows_loc(),
149 num_bands__,
150 eval__.at(sddk::memory_t::device),
151 h_diag__.at(sddk::memory_t::device, res__.pw_coeffs(ispn).num_rows_loc(), ispn),
152 o_diag__.at(sddk::memory_t::device, res__.pw_coeffs(ispn).num_rows_loc(), ispn));
153 }
154#endif
155 }
156 }

159template <typename T>
160static int
161normalized_preconditioned_residuals(sddk::memory_t mem_type__, sddk::spin_range spins__, int num_bands__,
162 sddk::mdarray<real_type<T>,1>& eval__, sddk::Wave_functions<real_type<T>>& hpsi__,
163 sddk::Wave_functions<real_type<T>>& opsi__, sddk::Wave_functions<real_type<T>>& res__,
164 sddk::mdarray<real_type<T>, 2> const& h_diag__, sddk::mdarray<real_type<T>, 2> const& o_diag__,
165 real_type<T> norm_tolerance__, sddk::mdarray<real_type<T>, 1> &residual_norms__)
166{
167 PROFILE("sirius::normalized_preconditioned_residuals");
168
169 assert(num_bands__ != 0);
170
171 auto pu = get_device_t(mem_type__);
172
173 /* compute "raw" residuals */
174 compute_residuals<real_type<T>>(mem_type__, spins__, num_bands__, eval__, hpsi__, opsi__, res__);
175
176 /* compute norm of the "raw" residuals */
177 residual_norms__ = res__.l2norm(pu, spins__, num_bands__);
178
179 /* apply preconditioner */
180 apply_preconditioner<real_type<T>>(mem_type__, spins__, num_bands__, res__, h_diag__, o_diag__, eval__);
181
182 /* this not strictly necessary as the wave-function orthoronormalization can take care of this;
183 however, normalization of residuals is harmless and gives a better numerical stability */
184 res__.normalize(pu, spins__, num_bands__);
185
186 int num_unconverged{0};
187
188 for (int i = 0; i < num_bands__; i++) {
189 /* take the residual if it's norm is above the threshold */
190 if (residual_norms__[i] > norm_tolerance__) {
191 /* shift unconverged residuals to the beginning of array */
192 /* note: we can just keep them where they were */
193 if (num_unconverged != i) {
194 for (int ispn: spins__) {
195 res__.copy_from(res__, 1, ispn, i, ispn, num_unconverged);
196 }
197 }
198 num_unconverged++;
199 }
200 }
201
202 /* prevent numerical noise */
203 /* this only happens for real wave-functions (Gamma-point case), non-magnetic or collinear magnetic */
204 if (std::is_same<T, real_type<T>>::value && res__.comm().rank() == 0 && num_unconverged != 0 && spins__() != 2) {
205 if (is_device_memory(res__.preferred_memory_t())) {
206#if defined(SIRIUS_GPU)
207 make_real_g0_gpu(res__.pw_coeffs(spins__()).prime().at(sddk::memory_t::device), res__.pw_coeffs(spins__()).prime().ld(), num_unconverged);
208#endif
209 } else {
210 for (int i = 0; i < num_unconverged; i++) {
211 res__.pw_coeffs(spins__()).prime(0, i) = res__.pw_coeffs(spins__()).prime(0, i).real();
212 }
213 }
214 }
215
216 return num_unconverged;
217}

219/// Compute residuals from eigen-vectors.
220template <typename T, typename F>
221residual_result
222residuals(Simulation_context& ctx__, sddk::memory_t mem_type__, sddk::linalg_t la_type__, sddk::spin_range ispn__, int N__,
223 int num_bands__, int num_locked, sddk::mdarray<real_type<F>, 1>& eval__, sddk::dmatrix<F>& evec__,
224 sddk::Wave_functions<real_type<T>>& hphi__, sddk::Wave_functions<real_type<T>>& ophi__, sddk::Wave_functions<real_type<T>>& hpsi__,
225 sddk::Wave_functions<real_type<T>>& opsi__, sddk::Wave_functions<real_type<T>>& res__, sddk::mdarray<real_type<T>, 2> const& h_diag__,
226 sddk::mdarray<real_type<T>, 2> const& o_diag__, bool estimate_eval__, real_type<T> norm_tolerance__,
227 std::function<bool(int, int)> is_converged__)
228{
229 PROFILE("sirius::residuals");
230
231 assert(N__ != 0);
232
233 sddk::mdarray<real_type<T>, 1> res_norm;
234 sddk::dmatrix<F> evec_tmp;
235
236 sddk::mdarray<real_type<T>, 1> eval(num_bands__);
237 eval = [&](size_t j) -> real_type<T> { return eval__[j]; };
238
239 sddk::dmatrix<F>* evec_ptr{nullptr};
240
241 /* total number of residuals to be computed */
242 int num_residuals{0};
243
244 /* number of lockable eigenvectors */
245 int num_consecutive_converged{0};
246
247 /* number of residuals that do not meet any convergence criterion */
248 int num_unconverged{0};
249
250 int ispn = (ispn__() == 2) ? 0 : ispn__();
251
252 // When estimate_eval is set we only compute true residuals of unconverged eigenpairs
253 // where convergence is determined just on the change in the eigenvalues.
254 if (estimate_eval__) {
255 // Locking is only based on the is_converged__ criterion, not on the actual
256 // residual norms. We could lock more by considering the residual norm criterion
257 // later, but since we're reordering eigenvectors too, this becomes messy.
258 while (num_consecutive_converged < num_bands__ && is_converged__(num_consecutive_converged, ispn)) {
259 ++num_consecutive_converged;
260 }
261
262 // Collect indices of unconverged eigenpairs.
263 std::vector<int> ev_idx;
264 for (int j = 0; j < num_bands__; j++) {
265 if (!is_converged__(j, ispn)) {
266 ev_idx.push_back(j);
267 }
268 }
269
270 // If everything is converged, return early.
271 if (ev_idx.empty()) {
272 return residual_result{num_bands__, 0, 0};
273 }
274
275 // Otherwise copy / reorder the unconverged eigenpairs
276 num_residuals = static_cast<int>(ev_idx.size());
277
278 evec_tmp = sddk::dmatrix<F>(N__, num_residuals, evec__.blacs_grid(), evec__.bs_row(), evec__.bs_col());
279 evec_ptr = &evec_tmp;
280
281 int num_rows_local = evec_tmp.num_rows_local();
282 for (int j = 0; j < num_residuals; j++) {
283 eval[j] = eval[ev_idx[j]];
284 if (evec__.blacs_grid().comm().size() == 1) {
285 /* do a local copy */
286 std::copy(&evec__(0, ev_idx[j]), &evec__(0, ev_idx[j]) + num_rows_local, &evec_tmp(0, j));
287 } else {
288 auto pos_src = evec__.spl_col().location(ev_idx[j]);
289 auto pos_dest = evec_tmp.spl_col().location(j);
290 /* do MPI send / receive */
291 if (pos_src.rank == evec__.blacs_grid().comm_col().rank() && num_rows_local) {
292 evec__.blacs_grid().comm_col().isend(&evec__(0, pos_src.local_index), num_rows_local, pos_dest.rank, ev_idx[j]);
293 }
294 if (pos_dest.rank == evec__.blacs_grid().comm_col().rank() && num_rows_local) {
295 evec__.blacs_grid().comm_col().recv(&evec_tmp(0, pos_dest.local_index), num_rows_local, pos_src.rank, ev_idx[j]);
296 }
297 }
298 }
299 if (is_device_memory(mem_type__) && evec_tmp.blacs_grid().comm().size() == 1) {
300 evec_tmp.allocate(sddk::memory_t::device);
301 }
302 } else {
303 evec_ptr = &evec__;
304 num_residuals = num_bands__;
305 }
306 if (is_device_memory(mem_type__)) {
307 eval.allocate(sddk::memory_t::device).copy_to(sddk::memory_t::device);
308 }
309
310 /* compute H\Psi_{i} = \sum_{mu} H\phi_{mu} * Z_{mu, i} and O\Psi_{i} = \sum_{mu} O\phi_{mu} * Z_{mu, i} */
311 sddk::transform<T, F>(ctx__.spla_context(), ispn__(), {&hphi__, &ophi__}, num_locked, N__ - num_locked,
312 *evec_ptr, 0, 0, {&hpsi__, &opsi__}, 0, num_residuals);
313
314 num_unconverged = normalized_preconditioned_residuals<T>(mem_type__, ispn__, num_residuals, eval, hpsi__, opsi__,
315 res__, h_diag__, o_diag__, norm_tolerance__, res_norm);
316
317 // In case we're not using the delta in eigenvalues as a convergence criterion,
318 // we lock eigenpairs using residual norms.
319 if (!estimate_eval__) {
320 while (num_consecutive_converged < num_residuals && res_norm[num_consecutive_converged] <= norm_tolerance__) {
321 ++num_consecutive_converged;
322 }
323 }
324
325 auto frobenius_norm = 0.0;
326 for (int i = 0; i < num_residuals; i++) {
327 frobenius_norm += res_norm[i] * res_norm[i];
328 }
329 frobenius_norm = std::sqrt(frobenius_norm);
330 return {
331 num_consecutive_converged,
332 num_unconverged,
333 frobenius_norm
334 };
335}
