109void Energy::compute()
110{
111 auto& ctx = kset.ctx();
112 int num_spins = ctx.num_spins();
113 int num_bands = ctx.num_bands();
114 int nk = kset.spl_num_kpoints().local_size();
115
116 // // // transfer from device to host (only if data on GPU is present)
117 // if(is_device_memory(ctx.preferred_memory_t())) {
118 // for (int ik = 0; ik < nk; ++ik) {
119 // for (int ispn = 0; ispn < num_spins; ++ispn) {
120 // int num_wf = cphis[ik]->num_wf();
121 // if (cphis[ik]->pw_coeffs(ispn).prime().on_device()) {
122 // // std::cout << "copying wfc from DEVICE -> HOST" << "\n";
123 // cphis[ik]->pw_coeffs(ispn).copy_to(memory_t::host, 0, num_wf);
124 // }
125
126 // }
127 // }
128 // }
129
130 density.generate<double>(kset, ctx.use_symmetry(), true /* add core */, true /* transform to rg */);
131
132 potential.generate(density, ctx.use_symmetry(), true);
133
134 /* compute H@X and new band energies */
135 auto H0 = Hamiltonian0<double>(potential);
136 // apply Hamiltonian
137 for (int i = 0; i < nk; ++i) {
138 auto& kp = *kset.get<double>(kset.spl_num_kpoints(i));
139 std::vector<double> band_energies(num_bands);
140
141 if (is_device_memory(ctx.preferred_memory_t())) {
142 auto& mpd = ctx.mem_pool(memory_t::device);
143 for (int ispn = 0; ispn < num_spins; ispn++) {
144 cphis[i]->pw_coeffs(ispn).allocate(mpd);
145 // copy to device
146 int num_wf = cphis[i]->num_wf();
147 cphis[i]->pw_coeffs(ispn).copy_to(memory_t::device, 0, num_wf);
148 }
149 }
150
151 assert(cphis[i] == kp.spinor_wave_functions_ptr());
152 apply_hamiltonian(H0, kp, *hphis[i], kp.spinor_wave_functions(), sphis[i]);
153 // compute band energies
154 for (int ispn = 0; ispn < num_spins; ++ispn) {
155 for (int jj = 0; jj < num_bands; ++jj) {
156 dmatrix<std::complex<double>> dmat(1, 1, memory_t::host);
157 dmat.allocate(memory_t::device);
158 sddk::inner(ctx.spla_context(), spin_range(ispn),
159 /* bra */ kp.spinor_wave_functions(), jj, 1,
160 /* ket */ *hphis[i], jj, 1,
161 /* out */ dmat, 0, 0);
162 // deal with memory...
163 // assert(std::abs(dmat(0, 0).imag()) < 1e-10);
164 kp.band_energy(jj, ispn, dmat(0, 0).real());
165 }
166 }
167 }
168 kset.sync_band<double, sync_band_t::energy>();
169
170 // evaluate total energy
171 double eewald = ewald_energy(ctx, ctx.gvec(), ctx.unit_cell());
172 this->etot = total_energy(ctx, kset, density, potential, eewald);
173}

201std::shared_ptr<nlcglib::VectorBaseZ> Energy::get_fn()
202{
203 auto nk = kset.spl_num_kpoints().local_size();
204 const int ns = kset.ctx().num_spins();
205 int nbands = kset.ctx().num_bands();
206 std::vector<std::vector<double>> fn;

217 fn.push_back(std::move(fn_local));

219 }
220 }

224void Energy::set_fn(const std::vector<std::pair<int, int>>& keys, const std::vector<std::vector<double>>& fn)
225{
226 const int nbands = kset.ctx().num_bands();
227 #ifndef NDEBUG
228 const int ns = kset.ctx().num_spins();
229 auto nk = kset.spl_num_kpoints().local_size();
230 const double max_occ = ns == 1 ? 2.0 : 1.0;
231 #endif
232
233 assert(static_cast<int>(fn.size()) == nk*ns);
234 for (size_t iloc = 0; iloc < fn.size(); ++iloc) {
235 // global k-point index
236 int gidk= keys[iloc].first;
237 int ispn = keys[iloc].second;
238 auto& kp = *kset.get<double>(gidk);
239 const auto& fn_loc = fn[iloc];
240 assert(static_cast<int>(fn_loc.size()) == nbands);
241 for (int i = 0; i < nbands; ++i) {
242 assert(fn_loc[i] >= 0 && fn_loc[i] <= max_occ);
243 kp.band_occupancy(i, ispn, fn_loc[i]);
244 }
245 }
246 kset.sync_band<double, sync_band_t::occupancy>();
247}
