26 inline void Density::add_k_point_contribution_dm(K_point* kp__, mdarray<double_complex, 4>& density_matrix__)
27 {
28  PROFILE("sirius::Density::add_k_point_contribution_dm");
29 
30  if (ctx_.full_potential()) {
31  /* non-magnetic or spin-collinear case */
32  if (ctx_.num_mag_dims() != 3) {
33  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
34  int nbnd = kp__->num_occupied_bands(ispn);
35 
36  mdarray<double_complex, 2> wf1(unit_cell_.max_mt_basis_size(), nbnd);
37  mdarray<double_complex, 2> wf2(unit_cell_.max_mt_basis_size(), nbnd);
38 
39  for (int ialoc = 0; ialoc < kp__->spinor_wave_functions().spl_num_atoms().local_size(); ialoc++) {
40  int ia = kp__->spinor_wave_functions().spl_num_atoms()[ialoc];
41  int mt_basis_size = unit_cell_.atom(ia).type().mt_basis_size();
42  int offset_wf = kp__->spinor_wave_functions().offset_mt_coeffs(ialoc);
43 
44  for (int i = 0; i < nbnd; i++) {
45  for (int xi = 0; xi < mt_basis_size; xi++) {
46  auto c = kp__->spinor_wave_functions().mt_coeffs(ispn).prime(offset_wf + xi, i);
47  wf1(xi, i) = std::conj(c);
48  wf2(xi, i) = c * kp__->band_occupancy(i, ispn) * kp__->weight();
49  }
50  }
51  /* add |psi_j> n_j <psi_j| to density matrix */
52  linalg<CPU>::gemm(0, 1, mt_basis_size, mt_basis_size, nbnd, linalg_const<double_complex>::one(),
53  &wf1(0, 0), wf1.ld(), &wf2(0, 0), wf2.ld(), linalg_const<double_complex>::one(),
54  density_matrix__.at<CPU>(0, 0, ispn, ia), density_matrix__.ld());
55  }
56  }
57  } else {
58  int nbnd = kp__->num_occupied_bands();
59 
60  mdarray<double_complex, 3> wf1(unit_cell_.max_mt_basis_size(), nbnd, ctx_.num_spins());
61  mdarray<double_complex, 3> wf2(unit_cell_.max_mt_basis_size(), nbnd, ctx_.num_spins());
62 
63  for (int ialoc = 0; ialoc < kp__->spinor_wave_functions().spl_num_atoms().local_size(); ialoc++) {
64  int ia = kp__->spinor_wave_functions().spl_num_atoms()[ialoc];
65  int mt_basis_size = unit_cell_.atom(ia).type().mt_basis_size();
66  int offset_wf = kp__->spinor_wave_functions().offset_mt_coeffs(ialoc);
67 
68  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
69  for (int i = 0; i < nbnd; i++) {
70 
71  for (int xi = 0; xi < mt_basis_size; xi++) {
72  auto c = kp__->spinor_wave_functions().mt_coeffs(ispn).prime(offset_wf + xi, i);
73  wf1(xi, i, ispn) = std::conj(c);
74  wf2(xi, i, ispn) = c * kp__->band_occupancy(i, 0) * kp__->weight();
75  }
76  }
77  }
78  /* compute diagonal terms */
79  for (int ispn = 0; ispn < 2; ispn++) {
80  linalg<CPU>::gemm(0, 1, mt_basis_size, mt_basis_size, nbnd, linalg_const<double_complex>::one(),
81  &wf1(0, 0, ispn), wf1.ld(), &wf2(0, 0, ispn), wf2.ld(),
82  linalg_const<double_complex>::one(), density_matrix__.at<CPU>(0, 0, ispn, ia),
83  density_matrix__.ld());
84  }
85  /* offdiagonal term */
86  linalg<CPU>::gemm(0, 1, mt_basis_size, mt_basis_size, nbnd, linalg_const<double_complex>::one(),
87  &wf1(0, 0, 1), wf1.ld(), &wf2(0, 0, 0), wf2.ld(), linalg_const<double_complex>::one(),
88  density_matrix__.at<CPU>(0, 0, 2, ia), density_matrix__.ld());
89  }
90  }
91  } else { /* pseudopotential */
92  if (!ctx_.unit_cell().mt_lo_basis_size()) {
93  return;
94  }
95 
96  kp__->beta_projectors().prepare();
97 
98  if (ctx_.num_mag_dims() != 3) {
99  for (int chunk = 0; chunk < kp__->beta_projectors().num_chunks(); chunk++) {
100  kp__->beta_projectors().generate(chunk);
101 
102  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
103  /* total number of occupied bands for this spin */
104  int nbnd = kp__->num_occupied_bands(ispn);
105  /* compute <beta|psi> */
106  auto beta_psi = kp__->beta_projectors().inner<T>(chunk, kp__->spinor_wave_functions(), ispn, 0, nbnd);
107 
108  /* number of beta projectors */
109  int nbeta = kp__->beta_projectors().chunk(chunk).num_beta_;
110 
111  splindex<block> spl_nbnd(nbnd, kp__->comm().size(), kp__->comm().rank());
112 
113  int nbnd_loc = spl_nbnd.local_size();
114  if (nbnd_loc) { // TODO: this part can also be moved to GPU
115  #pragma omp parallel
116  {
117  /* auxiliary arrays */
118  mdarray<double_complex, 2> bp1(nbeta, nbnd_loc);
119  mdarray<double_complex, 2> bp2(nbeta, nbnd_loc);
120  #pragma omp for
121  for (int ia = 0; ia < kp__->beta_projectors().chunk(chunk).num_atoms_; ia++) {
122  int nbf = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::nbf, ia);
123  int offs = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::offset, ia);
124  int ja = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::ia, ia);
125 
126  for (int i = 0; i < nbnd_loc; i++) {
127  int j = spl_nbnd[i];
128 
129  for (int xi = 0; xi < nbf; xi++) {
130  bp1(xi, i) = beta_psi(offs + xi, j);
131  bp2(xi, i) = std::conj(bp1(xi, i)) * kp__->weight() *
132  kp__->band_occupancy(j, ispn);
133  }
134  }
135 
136  linalg<CPU>::gemm(0, 1, nbf, nbf, nbnd_loc, linalg_const<double_complex>::one(),
137  &bp1(0, 0), bp1.ld(), &bp2(0, 0), bp2.ld(),
138  linalg_const<double_complex>::one(),
139  &density_matrix__(0, 0, ispn, ja),
140  density_matrix__.ld());
141  }
142  }
143  }
144  }
145  }
146  } else {
147  for (int chunk = 0; chunk < kp__->beta_projectors().num_chunks(); chunk++) {
148  kp__->beta_projectors().generate(chunk);
149 
150  /* number of beta projectors */
151  int nbeta = kp__->beta_projectors().chunk(chunk).num_beta_;
152 
153  /* total number of occupied bands */
154  int nbnd = kp__->num_occupied_bands();
155 
156  splindex<block> spl_nbnd(nbnd, kp__->comm().size(), kp__->comm().rank());
157  int nbnd_loc = spl_nbnd.local_size();
158 
159  /* auxiliary arrays */
160  mdarray<double_complex, 3> bp1(nbeta, nbnd_loc, ctx_.num_spins());
161  mdarray<double_complex, 3> bp2(nbeta, nbnd_loc, ctx_.num_spins());
162 
163  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
164  /* compute <beta|psi> */
165  auto beta_psi = kp__->beta_projectors().inner<T>(chunk, kp__->spinor_wave_functions(), ispn, 0, nbnd);
166  #pragma omp parallel for schedule(static)
167  for (int i = 0; i < nbnd_loc; i++) {
168  int j = spl_nbnd[i];
169 
170  for (int m = 0; m < nbeta; m++) {
171  bp1(m, i, ispn) = beta_psi(m, j);
172  bp2(m, i, ispn) = std::conj(beta_psi(m, j)) * kp__->weight() * kp__->band_occupancy(j, 0);
173  }
174  }
175  }
176  for (int ia = 0; ia < kp__->beta_projectors().chunk(chunk).num_atoms_; ia++) {
177  int nbf = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::nbf, ia);
178  int offs = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::offset, ia);
179  int ja = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::ia, ia);
180  if (ctx_.unit_cell().atom(ja).type().spin_orbit_coupling()) {
181  mdarray<double_complex, 3> bp3(nbf, nbnd_loc, 2);
182  bp3.zero();
183  /* We already have the <beta|psi> but we need to rotate
184  * them when the spin orbit interaction is included in the
185  * pseudo potential.
186  *
187  * We rotate \f[\langle\beta|\psi\rangle\f] accordingly by multiplying it with
188  * the \f[f^{\sigma\sigma^{'}}_{\xi,\xi^'}\f]
189  */
190 
191  for (int xi1 = 0; xi1 < nbf; xi1++) {
192  for (int i = 0; i < nbnd_loc; i++) {
193  for (int xi1p = 0; xi1p < nbf; xi1p++) {
194  if (ctx_.unit_cell().atom(ja).type().compare_index_beta_functions(xi1, xi1p)) {
195  bp3(xi1, i, 0) +=
196  bp1(offs + xi1p, i, 0) *
197  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 0, 0) +
198  bp1(offs + xi1p, i, 1) *
199  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 0, 1);
200  bp3(xi1, i, 1) +=
201  bp1(offs + xi1p, i, 0) *
202  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 1, 0) +
203  bp1(offs + xi1p, i, 1) *
204  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 1, 1);
205  }
206  }
207  }
208  }
209 
210  for (int xi1 = 0; xi1 < nbf; xi1++) {
211  for (int i = 0; i < nbnd_loc; i++) {
212  bp1(offs + xi1, i, 0) = bp3(xi1, i, 0);
213  bp1(offs + xi1, i, 1) = bp3(xi1, i, 1);
214  }
215  }
216 
217  bp3.zero();
218 
219  for (int xi1 = 0; xi1 < nbf; xi1++) {
220  for (int i = 0; i < nbnd_loc; i++) {
221  for (int xi1p = 0; xi1p < nbf; xi1p++) {
222  if (ctx_.unit_cell().atom(ja).type().compare_index_beta_functions(xi1, xi1p)) {
223  bp3(xi1, i, 0) +=
224  bp2(offs + xi1p, i, 0) *
225  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 0, 0) +
226  bp2(offs + xi1p, i, 1) *
227  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 1, 0);
228  bp3(xi1, i, 1) +=
229  bp2(offs + xi1p, i, 0) *
230  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 0, 1) +
231  bp2(offs + xi1p, i, 1) *
232  ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 1, 1);
233  }
234  }
235  }
236  }
237 
238  for (int xi1 = 0; xi1 < nbf; xi1++) {
239  for (int i = 0; i < nbnd_loc; i++) {
240  bp2(offs + xi1, i, 0) = bp3(xi1, i, 0);
241  bp2(offs + xi1, i, 1) = bp3(xi1, i, 1);
242  }
243  }
244  }
245  }
246 
247  if (nbnd_loc) {
248  #pragma omp parallel for
249  for (int ia = 0; ia < kp__->beta_projectors().chunk(chunk).num_atoms_; ia++) {
250  int nbf = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::nbf, ia);
251  int offs = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::offset, ia);
252  int ja = kp__->beta_projectors().chunk(chunk).desc_(beta_desc_idx::ia, ia);
253  /* compute diagonal spin blocks */
254  for (int ispn = 0; ispn < 2; ispn++) {
255  linalg<CPU>::gemm(0, 1, nbf, nbf, nbnd_loc, linalg_const<double_complex>::one(),
256  &bp1(offs, 0, ispn), bp1.ld(), &bp2(offs, 0, ispn), bp2.ld(),
257  linalg_const<double_complex>::one(), &density_matrix__(0, 0, ispn, ja),
258  density_matrix__.ld());
259  }
260  /* off-diagonal spin block */
261  linalg<CPU>::gemm(0, 1, nbf, nbf, nbnd_loc, linalg_const<double_complex>::one(),
262  &bp1(offs, 0, 0), bp1.ld(), &bp2(offs, 0, 1), bp2.ld(),
263  linalg_const<double_complex>::one(), &density_matrix__(0, 0, 2, ja),
264  density_matrix__.ld());
265  }
266  }
267  }
268  }
269  kp__->beta_projectors().dismiss();
270  }
