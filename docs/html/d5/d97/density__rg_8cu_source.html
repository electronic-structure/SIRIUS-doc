27 __global__ void update_density_rg_1_gpu_kernel(int size__,
28  cuDoubleComplex const* psi_rg__,
29  double wt__,
30  double* density_rg__)
31 {
32  int ir = blockIdx.x * blockDim.x + threadIdx.x;
33  if (ir < size__)
34  {
35  cuDoubleComplex z = psi_rg__[ir];
36  density_rg__[ir] += (z.x * z.x + z.y * z.y) * wt__;
37  }
38 }

59 __global__ void update_density_rg_2_gpu_kernel(int size__,
60  cuDoubleComplex const* psi_up_rg__,
61  cuDoubleComplex const* psi_dn_rg__,
62  double wt__,
63  double* density_x_rg__,
64  double* density_y_rg__)
65 {
66  int ir = blockIdx.x * blockDim.x + threadIdx.x;
67  if (ir < size__) {
68  cuDoubleComplex z = cuCmul(psi_up_rg__[ir], cuConj(psi_dn_rg__[ir]));
69  density_x_rg__[ir] += 2 * z.x * wt__;
70  density_y_rg__[ir] -= 2 * z.y * wt__;
71  }
72 }
