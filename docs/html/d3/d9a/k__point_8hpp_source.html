213  /// Find G+k vectors within the cutoff.
214  inline void generate_gkvec(double gk_cutoff__)
215  {
216  PROFILE("sirius::K_point::generate_gkvec");
217 
218  if (ctx_.full_potential() && (gk_cutoff__ * unit_cell_.max_mt_radius() > ctx_.lmax_apw()) &&
219  comm_.rank() == 0 && ctx_.control().verbosity_ >= 0) {
220  std::stringstream s;
221  s << "G+k cutoff (" << gk_cutoff__ << ") is too large for a given lmax ("
222  << ctx_.lmax_apw() << ") and a maximum MT radius (" << unit_cell_.max_mt_radius() << ")" << std::endl
223  << "suggested minimum value for lmax : " << int(gk_cutoff__ * unit_cell_.max_mt_radius()) + 1;
224  WARNING(s);
225  }
226 
227  if (gk_cutoff__ * 2 > ctx_.pw_cutoff()) {
228  std::stringstream s;
229  s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
230  << " pw cutoff : " << ctx_.pw_cutoff() << std::endl
231  << " doubled G+k cutoff : " << gk_cutoff__ * 2;
232  TERMINATE(s);
233  }
234 
235  /* create G+k vectors; communicator of the coarse FFT grid is used because wave-functions will be transformed
236  * only on the coarse grid; G+k-vectors will be distributed between MPI ranks assigned to the k-point */
237  gkvec_ = std::unique_ptr<Gvec>(new Gvec(vk_, ctx_.unit_cell().reciprocal_lattice_vectors(), gk_cutoff__, comm(),
238  ctx_.gamma_point()));
239 
240  gkvec_partition_ = std::unique_ptr<Gvec_partition>(new Gvec_partition(*gkvec_, ctx_.comm_fft_coarse(),
241  ctx_.comm_band_ortho_fft_coarse()));
242 
243  gkvec_offset_ = gkvec().gvec_offset(comm().rank());
244  }

416  * \varphi_{j{\bf k}}({\bf r}) \} \f$ is used to expand first-variational wave-functions:
417  *
418  * \f[
419  * \psi_{i{\bf k}}({\bf r}) = \sum_{\mu} c_{\mu i}^{\bf k} \varphi_{\mu \bf k}({\bf r}) =
420  * \sum_{{\bf G}}c_{{\bf G} i}^{\bf k} \varphi_{\bf G+k}({\bf r}) +
421  * \sum_{j}c_{j i}^{\bf k}\varphi_{j{\bf k}}({\bf r})
422  * \f]
423  *
424  * Inside muffin-tins the expansion is converted into the following form:
425  * \f[
426  * \psi_{i {\bf k}}({\bf r})= \begin{array}{ll}

430  * \f]
431  *
432  * Thus, the total number of coefficients representing a wave-funstion is equal
433  * to the number of muffin-tin basis functions of the form \f$ f_{\ell \lambda}^{\alpha}(r)

1288 /** \page basis Basis functions for Kohn-Sham wave-functions expansion
1289  *
1290  * \section basis1 LAPW+lo basis
1291  *
1292  * LAPW+lo basis consists of two different sets of functions: LAPW functions \f$ \varphi_{{\bf G+k}} \f$ defined over
1293  * entire unit cell:
1294  * \f[
1295  * \varphi_{{\bf G+k}}({\bf r}) = \left\{ \begin{array}{ll}
1296  * \displaystyle \sum_{L} \sum_{\nu=1}^{O_{\ell}^{\alpha}} a_{L\nu}^{\alpha}({\bf G+k})u_{\ell \nu}^{\alpha}(r)
1297  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in {\rm MT} \alpha \\
1298  * \displaystyle \frac{1}{\sqrt \Omega} e^{i({\bf G+k}){\bf r}} & {\bf r} \in {\rm I} \end{array} \right.
1299  * \f]
1300  * and Bloch sums of local orbitals defined inside muffin-tin spheres only:
1301  * \f[
1302  * \begin{array}{ll} \displaystyle \varphi_{j{\bf k}}({\bf r})=\sum_{{\bf T}} e^{i{\bf kT}}
1303  * \varphi_{j}({\bf r - T}) & {\rm {\bf r} \in MT} \end{array}
1304  * \f]
1305  * Each local orbital is composed of radial and angular parts:
1306  * \f[
1307  * \varphi_{j}({\bf r}) = \phi_{\ell_j}^{\zeta_j,\alpha_j}(r) Y_{\ell_j m_j}(\hat {\bf r})
1308  * \f]
1309  * Radial part of local orbital is defined as a linear combination of radial functions (minimum two radial functions
1310  * are required) such that local orbital vanishes at the sphere boundary:
1311  * \f[
1312  * \phi_{\ell}^{\zeta, \alpha}(r) = \sum_{p}\gamma_{p}^{\zeta,\alpha} u_{\ell \nu_p}^{\alpha}(r)
1313  * \f]
1314  *
1315  * Arbitrary number of local orbitals can be introduced for each angular quantum number (this is highlighted by
1316  * the index \f$ \zeta \f$).
1317  *
1318  * Radial functions are m-th order (with zero-order being a function itself) energy derivatives of the radial
1319  * Schrödinger equation:
1320  * \f[
1321  * u_{\ell \nu}^{\alpha}(r) = \frac{\partial^{m_{\nu}}}{\partial^{m_{\nu}}E}u_{\ell}^{\alpha}(r,E)\Big|_{E=E_{\nu}}
1322  * \f]
1323  */
