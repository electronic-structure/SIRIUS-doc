75  json ground_state(double energy_tol, double charge_tol, bool rel)
76  {
77  PROFILE("sirius::Free_atom::ground_state");
78 
79  int np = radial_grid().num_points();
80  assert(np > 0);
81 
82  free_atom_orbital_density_ = mdarray<double, 2>(np, num_atomic_levels());
83  free_atom_wave_functions_ = mdarray<double, 2>(np, num_atomic_levels());
84 
85  XC_functional *Ex = nullptr;
86  XC_functional Ec("XC_LDA_C_VWN", 1);;
87  if (rel) {
88  TERMINATE("Fixme : the libxc staring with version 4 changed the way to set relativitic LDA exchange");
89  Ex = new XC_functional("XC_LDA_REL_X", 1);
90  } else {
91  Ex = new XC_functional("XC_LDA_X", 1);
92  }
93 
94  std::vector<double> veff(np);
95  std::vector<double> vrho(np);
96  std::vector<double> vnuc(np);
97  for (int i = 0; i < np; i++) {
98  vnuc[i] = -zn() * radial_grid().x_inv(i);
99  veff[i] = vnuc[i];
100  vrho[i] = 0;
101  }
102 
103  Mixer<double>* mixer = new Broyden1<double>(0, np, 12, 0.8, 0.1, 1.0, Communicator::self());
104  for (int i = 0; i < np; i++) {
105  mixer->input_local(i, vrho[i]);
106  }
107  mixer->initialize();
108 
109  Spline<double> rho(radial_grid());
110 
111  Spline<double> f(radial_grid());
112 
113  std::vector<double> vh(np);
114  std::vector<double> vxc(np);
115  std::vector<double> exc(np);
116  std::vector<double> vx(np);
117  std::vector<double> vc(np);
118  std::vector<double> ex(np);
119  std::vector<double> ec(np);
120  std::vector<double> g1;
121  std::vector<double> g2;
122  std::vector<double> rho_old;
123 
124  enu_.resize(num_atomic_levels());
125 
126  double energy_tot = 0.0;
127  double energy_tot_old;
128  double charge_rms;
129  double energy_diff;
130  double energy_enuc = 0;
131  double energy_xc = 0;
132  double energy_kin = 0;
133  double energy_coul = 0;
134 
135  /* starting values for E_{nu} */
136  for (int ist = 0; ist < num_atomic_levels(); ist++) {
137  enu_[ist] = -1.0 * zn() / 2 / std::pow(double(atomic_level(ist).n), 2);
138  }
139 
140  int num_iter{-1};
141 
142  for (int iter = 0; iter < 200; iter++) {
143  rho_old = rho.values();
144 
145  std::memset(&rho(0), 0, rho.num_points() * sizeof(double));
146  #pragma omp parallel default(shared)
147  {
148  std::vector<double> rho_t(rho.num_points());
149  std::memset(&rho_t[0], 0, rho.num_points() * sizeof(double));
150 
151  #pragma omp for
152  for (int ist = 0; ist < num_atomic_levels(); ist++) {
153  // relativity_t rt = (rel) ? relativity_t::koelling_harmon : relativity_t::none;
154  relativity_t rt = (rel) ? relativity_t::dirac : relativity_t::none;
155  Bound_state bound_state(rt, zn(), atomic_level(ist).n, atomic_level(ist).l, atomic_level(ist).k,
156  radial_grid(), veff, enu_[ist]);
157  enu_[ist] = bound_state.enu();
158  auto& bs_rho = bound_state.rho();
159  auto& bs_u = bound_state.u();
160 
161  /* assume a spherical symmetry */
162  for (int i = 0; i < np; i++) {
163  free_atom_orbital_density_(i, ist) = bs_rho(i);
164  free_atom_wave_functions_(i, ist) = bs_u(i);
165  /* sum of squares of spherical harmonics for angular momentm l is (2l+1)/4pi */
166  rho_t[i] += atomic_level(ist).occupancy * free_atom_orbital_density_(i, ist) / fourpi;
167  }
168  }
169 
170  #pragma omp critical
171  for (int i = 0; i < rho.num_points(); i++) {
172  rho(i) += rho_t[i];
173  }
174  }
175 
176  charge_rms = 0.0;
177  for (int i = 0; i < np; i++) {
178  charge_rms += std::pow(rho(i) - rho_old[i], 2);
179  }
180  charge_rms = sqrt(charge_rms / np);
181 
182  rho.interpolate();
183 
184  /* compute Hartree potential */
185  rho.integrate(g2, 2);
186  double t1 = rho.integrate(g1, 1);
187 
188  for (int i = 0; i < np; i++) {
189  vh[i] = fourpi * (g2[i] / radial_grid(i) + t1 - g1[i]);
190  }
191 
192  /* compute XC potential and energy */
193  Ex->get_lda(rho.num_points(), &rho(0), &vx[0], &ex[0]);
194  Ec.get_lda(rho.num_points(), &rho(0), &vc[0], &ec[0]);
195  for (int ir = 0; ir < rho.num_points(); ir++) {
196  vxc[ir] = (vx[ir] + vc[ir]);
197  exc[ir] = (ex[ir] + ec[ir]);
198  }
199 
200  /* mix old and new effective potential */
201  for (int i = 0; i < np; i++) {
202  mixer->input_local(i, vh[i] + vxc[i]);
203  }
204  mixer->mix(1e-16);
205  for (int i = 0; i < np; i++) {
206  vrho[i] = mixer->output_local(i);
207  veff[i] = vrho[i] + vnuc[i];
208  }
209 
210  //= /* mix old and new effective potential */
211  //= for (int i = 0; i < np; i++)
212  //= veff[i] = (1 - beta) * veff[i] + beta * (vnuc[i] + vh[i] + vxc[i]);
213 
214  /* sum of occupied eigen values */
215  double eval_sum = 0.0;
216  for (int ist = 0; ist < num_atomic_levels(); ist++) {
217  eval_sum += atomic_level(ist).occupancy * enu_[ist];
218  }
219 
220  for (int i = 0; i < np; i++) {
221  f(i) = vrho[i] * rho(i);
222  }
223  /* kinetic energy */
224  energy_kin = eval_sum - fourpi * (f.interpolate().integrate(2) - zn() * rho.integrate(1));
225 
226  /* XC energy */
227  for (int i = 0; i < np; i++) {
228  f(i) = exc[i] * rho(i);
229  }
230  energy_xc = fourpi * f.interpolate().integrate(2);
231 
232  /* electron-nuclear energy: \int vnuc(r) * rho(r) r^2 dr */
233  energy_enuc = -fourpi * zn() * rho.integrate(1);
234 
235  /* Coulomb energy */
236  for (int i = 0; i < np; i++) {
237  f(i) = vh[i] * rho(i);
238  }
239  energy_coul = 0.5 * fourpi * f.interpolate().integrate(2);
240 
241  energy_tot_old = energy_tot;
242 
243  energy_tot = energy_kin + energy_xc + energy_coul + energy_enuc;
244 
245  energy_diff = std::abs(energy_tot - energy_tot_old);
246 
247  if (energy_diff < energy_tol && charge_rms < charge_tol) {
248  num_iter = iter;
249  break;
250  }
251  }
252 
253  json dict;
254  if (num_iter >= 0) {
255  dict["converged"] = true;
256  dict["num_scf_iterations"] = num_iter;
257  } else {
258  dict["converged"] = false;
259  }
260  dict["energy_diff"] = energy_diff;
261  dict["charge_rms"] = charge_rms;
262  dict["energy_tot"] = energy_tot;
263 
264  free_atom_density_spline_ = Spline<double>(radial_grid_, rho.values());
265 
266  free_atom_potential_ = Spline<double>(radial_grid_, vrho);
267 
268  return std::move(dict);
