252  /// Convert form Rlm to Ylm representation.
253  static void convert(int lmax__, double const* f_rlm__, double_complex* f_ylm__)
254  {
255  int lm = 0;
256  for (int l = 0; l <= lmax__; l++) {
257  for (int m = -l; m <= l; m++) {
258  if (m == 0) {
259  f_ylm__[lm] = f_rlm__[lm];
260  } else {
261  int lm1 = Utils::lm_by_l_m(l, -m);
262  f_ylm__[lm] = ylm_dot_rlm(l, m, m) * f_rlm__[lm] + ylm_dot_rlm(l, m, -m) * f_rlm__[lm1];
263  }
264  lm++;
265  }
266  }

321  /// Transform Cartesian coordinates [x,y,z] to spherical coordinates [r,theta,phi]
322  static vector3d<double> spherical_coordinates(vector3d<double> vc)
323  {
324  vector3d<double> vs;
325 
326  const double eps{1e-12};
327 
328  vs[0] = vc.length();
329 
330  if (vs[0] <= eps) {
331  vs[1] = 0.0;
332  vs[2] = 0.0;
333  } else {
334  vs[1] = std::acos(vc[2] / vs[0]); // theta = cos^{-1}(z/r)
335 
336  if (std::abs(vc[0]) > eps || std::abs(vc[1]) > eps) {
337  vs[2] = std::atan2(vc[1], vc[0]); // phi = tan^{-1}(y/x)
338  if (vs[2] < 0.0) {
339  vs[2] += twopi;
340  }
341  } else {
342  vs[2] = 0.0;
343  }
344  }
345 
346  return vs;
347  }

349  /// Generate complex spherical harmonics Ylm
350  static void spherical_harmonics(int lmax, double theta, double phi, double_complex* ylm)
351  {
352  double x = std::cos(theta);
353  std::vector<double> result_array(lmax + 1);
354 
355  for (int l = 0; l <= lmax; l++) {
356  for (int m = 0; m <= l; m++) {
357  double_complex z = std::exp(double_complex(0.0, m * phi));
358  ylm[Utils::lm_by_l_m(l, m)] = gsl_sf_legendre_sphPlm(l, m, x) * z;
359  if (m % 2) {
360  ylm[Utils::lm_by_l_m(l, -m)] = -std::conj(ylm[Utils::lm_by_l_m(l, m)]);
361  } else {
362  ylm[Utils::lm_by_l_m(l, -m)] = std::conj(ylm[Utils::lm_by_l_m(l, m)]);
363  }
364  }
365  }
366  }

368  /// Generate real spherical harmonics Rlm
369  /** Mathematica code:
370  * \verbatim
371  * R[l_, m_, th_, ph_] :=
372  * If[m > 0, std::sqrt[2]*ComplexExpand[Re[SphericalHarmonicY[l, m, th, ph]]],
373  * If[m < 0, std::sqrt[2]*ComplexExpand[Im[SphericalHarmonicY[l, m, th, ph]]],
374  * If[m == 0, ComplexExpand[Re[SphericalHarmonicY[l, 0, th, ph]]]]]]
375  * \endverbatim
376  */
377  static void spherical_harmonics(int lmax, double theta, double phi, double* rlm)
378  {
379  int lmmax = (lmax + 1) * (lmax + 1);
380  std::vector<double_complex> ylm(lmmax);
381  spherical_harmonics(lmax, theta, phi, &ylm[0]);
382 
383  double const t = std::sqrt(2.0);
384 
385  rlm[0] = y00;
386 
387  for (int l = 1; l <= lmax; l++) {
388  for (int m = -l; m < 0; m++) {
389  rlm[Utils::lm_by_l_m(l, m)] = t * ylm[Utils::lm_by_l_m(l, m)].imag();
390  }
391 
392  rlm[Utils::lm_by_l_m(l, 0)] = ylm[Utils::lm_by_l_m(l, 0)].real();
393 
394  for (int m = 1; m <= l; m++) {
395  rlm[Utils::lm_by_l_m(l, m)] = t * ylm[Utils::lm_by_l_m(l, m)].real();
396  }
397  }

400  /// Compute element of the transformation matrix from complex to real spherical harmonics.
401  /** Real spherical harmonic can be written as a linear combination of complex harmonics:
402  *
403  * \f[
404  * R_{\ell m}(\theta, \phi) = \sum_{m'} a^{\ell}_{m' m}Y_{\ell m'}(\theta, \phi)
405  * \f]
406  * where
407  * \f[
408  * a^{\ell}_{m' m} = \langle Y_{\ell m'} | R_{\ell m} \rangle
409  * \f]
410  *
411  * Transformation from real to complex spherical harmonics is conjugate transpose:
412  *
413  * \f[
414  * Y_{\ell m}(\theta, \phi) = \sum_{m'} a^{\ell*}_{m m'}R_{\ell m'}(\theta, \phi)
415  * \f]
416  *
417  * Mathematica code:
418  * \verbatim
419  * b[m1_, m2_] :=
420  * If[m1 == 0, 1,
421  * If[m1 < 0 && m2 < 0, -I/std::sqrt[2],
422  * If[m1 > 0 && m2 < 0, (-1)^m1*I/std::sqrt[2],
423  * If[m1 < 0 && m2 > 0, (-1)^m2/std::sqrt[2],
424  * If[m1 > 0 && m2 > 0, 1/std::sqrt[2]]]]]]
425  *
426  * a[m1_, m2_] := If[Abs[m1] == Abs[m2], b[m1, m2], 0]
427  *
428  * R[l_, m_, t_, p_] := Sum[a[m1, m]*SphericalHarmonicY[l, m1, t, p], {m1, -l, l}]
429  * \endverbatim
430  */
431  static inline double_complex ylm_dot_rlm(int l, int m1, int m2)
432  {
433  double const isqrt2 = 1.0 / std::sqrt(2);
434 
435  assert(l >= 0 && std::abs(m1) <= l && std::abs(m2) <= l);
436 
437  if (!((m1 == m2) || (m1 == -m2))) {
438  return double_complex(0, 0);
439  }
440 
441  if (m1 == 0) {
442  return double_complex(1, 0);
443  }
444 
445  if (m1 < 0) {
446  if (m2 < 0) {
447  return -double_complex(0, isqrt2);
448  } else {
449  return std::pow(-1.0, m2) * double_complex(isqrt2, 0);
450  }
451  } else {
452  if (m2 < 0) {
453  return pow(-1.0, m1) * double_complex(0, isqrt2);
454  } else {
455  return double_complex(isqrt2, 0);
456  }
457  }
458  }

465  /// Gaunt coefficent of three complex spherical harmonics.
466  /**
467  * \f[
468  * \langle Y_{\ell_1 m_1} | Y_{\ell_2 m_2} | Y_{\ell_3 m_3} \rangle
469  * \f]
470  */
471  static double gaunt_ylm(int l1, int l2, int l3, int m1, int m2, int m3)
472  {
473  assert(l1 >= 0);
474  assert(l2 >= 0);
475  assert(l3 >= 0);
476  assert(m1 >= -l1 && m1 <= l1);
477  assert(m2 >= -l2 && m2 <= l2);
478  assert(m3 >= -l3 && m3 <= l3);
479 
480  return std::pow(-1.0, std::abs(m1)) * std::sqrt(double(2 * l1 + 1) * double(2 * l2 + 1) * double(2 * l3 + 1) / fourpi) *
481  gsl_sf_coupling_3j(2 * l1, 2 * l2, 2 * l3, 0, 0, 0) *
482  gsl_sf_coupling_3j(2 * l1, 2 * l2, 2 * l3, -2 * m1, 2 * m2, 2 * m3);
483  }

485  /// Gaunt coefficent of three real spherical harmonics.
486  /**
487  * \f[
488  * \langle R_{\ell_1 m_1} | R_{\ell_2 m_2} | R_{\ell_3 m_3} \rangle
489  * \f]
490  */
491  static double gaunt_rlm(int l1, int l2, int l3, int m1, int m2, int m3)
492  {
493  assert(l1 >= 0);
494  assert(l2 >= 0);
495  assert(l3 >= 0);
496  assert(m1 >= -l1 && m1 <= l1);
497  assert(m2 >= -l2 && m2 <= l2);
498  assert(m3 >= -l3 && m3 <= l3);
499 
500  double d = 0;
501  for (int k1 = -l1; k1 <= l1; k1++) {
502  for (int k2 = -l2; k2 <= l2; k2++) {
503  for (int k3 = -l3; k3 <= l3; k3++) {
504  d += std::real(std::conj(SHT::ylm_dot_rlm(l1, k1, m1)) *
505  SHT::ylm_dot_rlm(l2, k2, m2) *
506  SHT::ylm_dot_rlm(l3, k3, m3)) * SHT::gaunt_ylm(l1, l2, l3, k1, k2, k3);
507  }
508  }
509  }
510  return d;
511  }

518  */
519  static double_complex gaunt_hybrid(int l1, int l2, int l3, int m1, int m2, int m3)

614  static void wigner_d_matrix(int l, double beta, mdarray<double, 2>& d_mtrx__)
615  {

617  long double sin_b2 = std::sin((long double)beta / 2.0L);
618 
619  for (int m1 = -l; m1 <= l; m1++) {
620  for (int m2 = -l; m2 <= l; m2++) {
621  long double d = 0;
622  for (int j = 0; j <= std::min(l + m1, l - m2); j++) {
623  if ((l - m2 - j) >= 0 && (l + m1 - j) >= 0 && (j + m2 - m1) >= 0) {
624  long double g = (std::sqrt(Utils::factorial(l + m1)) / Utils::factorial(l - m2 - j)) *
625  (std::sqrt(Utils::factorial(l - m1)) / Utils::factorial(l + m1 - j)) *

1082  /** The following derivative is computed:
1083  * \f[
1084  * \frac{\partial R_{\ell m}(\theta_r, \phi_r)}{\partial r_{\mu}} =
1085  * \frac{\partial R_{\ell m}(\theta_r, \phi_r)}{\partial \theta_r} \frac{\partial \theta_r}{\partial r_{\mu}} +
1086  * \frac{\partial R_{\ell m}(\theta_r, \phi_r)}{\partial \phi_r} \frac{\partial \phi_r}{\partial r_{\mu}}
1087  * \f]
1088  * The derivatives of angles are:
1089  * \f[
1090  * \frac{\partial \theta_r}{\partial r_{x}} = \frac{\cos(\phi_r) \cos(\theta_r)}{r} \\
1091  * \frac{\partial \theta_r}{\partial r_{y}} = \frac{\cos(\theta_r) \sin(\phi_r)}{r} \\
1092  * \frac{\partial \theta_r}{\partial r_{z}} = -\frac{\sin(\theta_r)}{r}
1093  * \f]
1094  * and
1095  * \f[
1096  * \frac{\partial \phi_r}{\partial r_{x}} = -\frac{\sin(\phi_r)}{\sin(\theta_r) r} \\
1097  * \frac{\partial \phi_r}{\partial r_{y}} = \frac{\cos(\phi_r)}{\sin(\theta_r) r} \\
1098  * \frac{\partial \phi_r}{\partial r_{z}} = 0
1099  * \f]
1100  * The derivative of \f$ \phi \f$ has discontinuities at \f$ \theta = 0, \theta=\pi \f$. This, however, is not a problem, because
1101  * multiplication by the the derivative of \f$ R_{\ell m} \f$ removes it. The following functions have to be hardcoded:
1102  * \f[
1103  * \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \\
1104  * \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{1}{\sin(\theta)}
1105  * \f]
1106  *
1107  * Mathematica script for spherical harmonic derivatives:
1108  \verbatim
1109  Rlm[l_, m_, th_, ph_] :=
1110  If[m > 0, Sqrt[2]*ComplexExpand[Re[SphericalHarmonicY[l, m, th, ph]]],
1111  If[m < 0, Sqrt[2]*ComplexExpand[Im[SphericalHarmonicY[l, m, th, ph]]],
1112  If[m == 0, ComplexExpand[Re[SphericalHarmonicY[l, 0, th, ph]]]]
1113  ]
