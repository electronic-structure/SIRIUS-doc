300  /// Compute element of the transformation matrix from complex to real spherical harmonics.
301  /** Real spherical harmonic can be written as a linear combination of complex harmonics:
302  *
303  * \f[
304  * R_{\ell m}(\theta, \phi) = \sum_{m'} a^{\ell}_{m' m}Y_{\ell m'}(\theta, \phi)
305  * \f]
306  * where
307  * \f[
308  * a^{\ell}_{m' m} = \langle Y_{\ell m'} | R_{\ell m} \rangle
309  * \f]
310  *
311  * Transformation from real to complex spherical harmonics is conjugate transpose:
312  *
313  * \f[
314  * Y_{\ell m}(\theta, \phi) = \sum_{m'} a^{\ell*}_{m m'}R_{\ell m'}(\theta, \phi)
315  * \f]
316  *
317  * Mathematica code:
318  * \verbatim
319  * b[m1_, m2_] :=
320  * If[m1 == 0, 1,
321  * If[m1 < 0 && m2 < 0, -I/std::sqrt[2],
322  * If[m1 > 0 && m2 < 0, (-1)^m1*I/std::sqrt[2],
323  * If[m1 < 0 && m2 > 0, (-1)^m2/std::sqrt[2],
324  * If[m1 > 0 && m2 > 0, 1/std::sqrt[2]]]]]]
325  *
326  * a[m1_, m2_] := If[Abs[m1] == Abs[m2], b[m1, m2], 0]
327  *
328  * R[l_, m_, t_, p_] := Sum[a[m1, m]*SphericalHarmonicY[l, m1, t, p], {m1, -l, l}]
329  * \endverbatim
330  */
331  static inline double_complex ylm_dot_rlm(int l, int m1, int m2)
332  {
333  const double isqrt2 = 0.70710678118654752440;
334 
335  assert(l >= 0 && std::abs(m1) <= l && std::abs(m2) <= l);
336 
337  if (!((m1 == m2) || (m1 == -m2))) {
338  return double_complex(0, 0);
339  }
340 
341  if (m1 == 0) {
342  return double_complex(1, 0);
343  }
344 
345  if (m1 < 0) {
346  if (m2 < 0) {
347  return -double_complex(0, isqrt2);
348  } else {
349  return std::pow(-1.0, m2) * double_complex(isqrt2, 0);
350  }
351  } else {
352  if (m2 < 0) {
353  return pow(-1.0, m1) * double_complex(0, isqrt2);
354  } else {
355  return double_complex(isqrt2, 0);
356  }
357  }
358  }

392  /** Clebsch-Gordan coefficients arise when two angular momenta are combined into a

395  static inline double clebsch_gordan(int l1, int l2, int l3, int m1, int m2, int m3)

438  static void wigner_d_matrix(int l, double beta, mdarray<double, 2>& d_mtrx__)
439  {
440  long double cos_b2 = std::cos((long double)beta / 2.0L);
441  long double sin_b2 = std::sin((long double)beta / 2.0L);
442 
443  for (int m1 = -l; m1 <= l; m1++)
444  {
445  for (int m2 = -l; m2 <= l; m2++)
446  {
447  long double d = 0;
448  for (int j = 0; j <= std::min(l + m1, l - m2); j++)
449  {
450  if ((l - m2 - j) >= 0 && (l + m1 - j) >= 0 && (j + m2 - m1) >= 0)
451  {
452  long double g = (std::sqrt(Utils::factorial(l + m1)) / Utils::factorial(l - m2 - j)) *
453  (std::sqrt(Utils::factorial(l - m1)) / Utils::factorial(l + m1 - j)) *
454  (std::sqrt(Utils::factorial(l - m2)) / Utils::factorial(j + m2 - m1)) *
455  (std::sqrt(Utils::factorial(l + m2)) / Utils::factorial(j));
456  d += g * std::pow(-1, j) * std::pow(cos_b2, 2 * l + m1 - m2 - 2 * j) * std::pow(sin_b2, 2 * j + m2 - m1);
457  }
458  }
459  d_mtrx__(m1 + l, m2 + l) = (double)d;
460  }
461  }

464  static void rotation_matrix_l(int l, vector3d<double> euler_angles, int proper_rotation,
465  double_complex* rot_mtrx__, int ld)
466  {
467  mdarray<double_complex, 2> rot_mtrx(rot_mtrx__, ld, 2 * l + 1);
468 
469  mdarray<double, 2> d_mtrx(2 * l + 1, 2 * l + 1);
470  wigner_d_matrix(l, euler_angles[1], d_mtrx);
471 
472  for (int m1 = -l; m1 <= l; m1++)
473  {
474  for (int m2 = -l; m2 <= l; m2++)
475  {
476  rot_mtrx(m1 + l, m2 + l) = std::exp(double_complex(0, -euler_angles[0] * m1 - euler_angles[2] * m2)) *
477  d_mtrx(m1 + l, m2 + l) * std::pow(proper_rotation, l);
478  }
479  }

482  static void rotation_matrix_l(int l, vector3d<double> euler_angles, int proper_rotation,
483  double* rot_mtrx__, int ld)
484  {
485  mdarray<double, 2> rot_mtrx_rlm(rot_mtrx__, ld, 2 * l + 1);
486  mdarray<double_complex, 2> rot_mtrx_ylm(2 * l + 1, 2 * l + 1);
487 
488  mdarray<double, 2> d_mtrx(2 * l + 1, 2 * l + 1);
489  wigner_d_matrix(l, euler_angles[1], d_mtrx);
490 
491  for (int m1 = -l; m1 <= l; m1++)
492  {
493  for (int m2 = -l; m2 <= l; m2++)
494  {
495  rot_mtrx_ylm(m1 + l, m2 + l) = std::exp(double_complex(0, -euler_angles[0] * m1 - euler_angles[2] * m2)) *
496  d_mtrx(m1 + l, m2 + l) * std::pow(proper_rotation, l);
497  }
498  }
499  for (int m1 = -l; m1 <= l; m1++)
500  {
501  auto i13 = (m1 == 0) ? std::vector<int>({0}) : std::vector<int>({-m1, m1});
502 
503  for (int m2 = -l; m2 <= l; m2++)
504  {
505  auto i24 = (m2 == 0) ? std::vector<int>({0}) : std::vector<int>({-m2, m2});
506 
507  for (int m3: i13)
508  {
509  for (int m4: i24)
510  {
511  rot_mtrx_rlm(m1 + l, m2 + l) += std::real(rlm_dot_ylm(l, m1, m3) *
512  rot_mtrx_ylm(m3 + l, m4 + l) *
513  ylm_dot_rlm(l, m4, m2));
514  }
515  }
516  }
517  }

542  /// Compute derivative of real-spherical harmonic with respect to theta angle.
543  static double dRlm_dtheta(int lm, double theta, double phi)
544  {
545  switch (lm) {
546  case 0: return 0;
547 
548  case 1: return -(std::sqrt(3/pi)*std::cos(theta)*std::sin(phi))/2.;
549 
550  case 2: return -(std::sqrt(3/pi)*std::sin(theta))/2.;
551 
552  case 3: return -(std::sqrt(3/pi)*std::cos(phi)*std::cos(theta))/2.;
553 
554  case 4: return -(std::sqrt(15/pi)*std::cos(phi)*std::cos(theta)*std::sin(phi)*std::sin(theta));
555 
556  case 5: return -(std::sqrt(15/pi)*std::cos(2*theta)*std::sin(phi))/2.;
557 
558  case 6: return (-3*std::sqrt(5/pi)*std::cos(theta)*std::sin(theta))/2.;
559 
560  case 7: return -(std::sqrt(15/pi)*std::cos(phi)*std::cos(2*theta))/2.;
561 
562  case 8: return (std::sqrt(15/pi)*std::cos(2*phi)*std::sin(2*theta))/4.;
563 
564  case 9: return (-3*std::sqrt(35/(2.*pi))*std::cos(theta)*std::sin(3*phi)*std::pow(std::sin(theta),2))/4.;
565

758 {
759  vector3d<double> vs;

761  double eps = 1e-12;
762 
763  vs[0] = vc.length();
764

766  {
767  vs[1] = 0.0;
768  vs[2] = 0.0;
769  }
770  else

775  {
