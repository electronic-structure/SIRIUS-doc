238  void generate_beta_gk_t_lat_deriv(Simulation_context const& ctx__)
239  {
240  PROFILE("sirius::Beta_projectors::generate_beta_gk_t_lat_deriv");
241 
242  auto beta_dG_ri = Radial_integrals_beta_dg(unit_cell_, ctx_.gk_cutoff(), 20);
243 
244  int mu = 0;
245  int nu = 0;
246 
247  /* allocate array */
248  matrix<double_complex> beta_gk_t_lat_deriv(gkvec_.gvec_count(comm_.rank()), num_beta_t_);
249 
250  /* compute dG_tau / da_{mu,nu} */
251  auto dG_da = [this](vector3d<double>& gvc, int tau, int mu, int nu)
252  {
253  return -unit_cell_.inverse_lattice_vectors()(nu, tau) * gvc[mu];
254  };
255 
256  /* compute derivative of theta angle with respect to lattice vectors d theta / da_{mu,nu} */
257  auto dtheta_da = [this, dG_da](vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
258  {
259  double g = gvs[0];
260  double theta = gvs[1];
261  double phi = gvs[2];
262 
263  double result = std::cos(theta) * std::cos(phi) * dG_da(gvc, 0, mu, nu) +
264  std::cos(theta) * std::sin(phi) * dG_da(gvc, 1, mu, nu) -
265  std::sin(theta) * dG_da(gvc, 2, mu, nu);
266  return result / g;
267  };
268 
269  auto dphi_da = [this, dG_da](vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
270  {
271  double g = gvs[0];
272  double phi = gvs[2];
273 
274  double result = -std::sin(phi) * dG_da(gvc, 0, mu, nu) + std::cos(phi) * dG_da(gvc, 0, mu, nu);
275  return result / g;
276  };
277 
278  auto dRlm_da = [this, dtheta_da, dphi_da](int lm, vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
279  {
280  double theta = gvs[1];
281  double phi = gvs[2];
282 
283  return SHT::dRlm_dtheta(lm, theta, phi) * dtheta_da(gvc, gvs, mu, nu) +
284  SHT::dRlm_dphi_sin_theta(lm, theta, phi) * dphi_da(gvc, gvs, mu, nu);
285  };
286 
287  auto djl_da = [this, dG_da](vector3d<double>& gvc, vector3d<double>& gvs, int mu, int nu)
288  {
289  double theta = gvs[1];
290  double phi = gvs[2];
291 
292  return std::sin(theta) * std::cos(phi) * dG_da(gvc, 0, mu, nu) +
293  std::sin(theta) * std::sin(phi) * dG_da(gvc, 1, mu, nu) +
294  std::cos(theta) * dG_da(gvc, 2, mu, nu);
295  };
296 
297  /* compute d <G+k|beta> / a_{mu, nu} */
298  #pragma omp parallel for
299  for (int igkloc = 0; igkloc < gkvec_.gvec_count(comm_.rank()); igkloc++) {
300  int igk = gkvec_.gvec_offset(comm_.rank()) + igkloc;
301  auto gvc = gkvec_.gkvec_cart(igk);
302  /* vs = {r, theta, phi} */
303  auto gvs = SHT::spherical_coordinates(gvc);
304 
305  if (gvs[0] < 1e-10) {
306  for (int i = 0; i < num_beta_t_; i++) {
307  beta_gk_t_lat_deriv(igkloc, i) = 0;
308  }
309  continue;
310  }
311 
312  /* compute real spherical harmonics for G+k vector */
313  std::vector<double> gkvec_rlm(Utils::lmmax(lmax_beta_));
314  std::vector<double> gkvec_drlm(Utils::lmmax(lmax_beta_));
315 
316  SHT::spherical_harmonics(lmax_beta_, gvs[1], gvs[2], &gkvec_rlm[0]);
317 
318  for (int lm = 0; lm < Utils::lmmax(lmax_beta_); lm++) {
319  gkvec_drlm[lm] = dRlm_da(lm, gvc, gvs, mu, nu);
320  }
321 
322  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
323  auto& atom_type = unit_cell_.atom_type(iat);
324  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
325  int l = atom_type.indexb(xi).l;
326  int lm = atom_type.indexb(xi).lm;
327  int idxrf = atom_type.indexb(xi).idxrf;
328 
329  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
330 
331  auto d1 = beta_radial_integrals_->value(idxrf, iat, gvs[0]) *
332  (gkvec_drlm[lm] - 0.5 * gkvec_rlm[lm] * unit_cell_.inverse_lattice_vectors()(nu, mu));
333 
334  auto d2 = gkvec_rlm[lm] * djl_da(gvc, gvs, mu, nu) * beta_dG_ri.value(idxrf, iat, gvs[0]);
335 
336  beta_gk_t_lat_deriv(igkloc, atom_type.offset_lo() + xi) = z * (d1 + d2);
337  }
338  }
339  }
340 
341  //if (unit_cell_.parameters().control().print_checksum_) {
342  // auto c1 = beta_gk_t_.checksum();
343  // comm_.allreduce(&c1, 1);
344  // if (comm_.rank() == 0) {
345  // DUMP("checksum(beta_gk_t) : %18.10f %18.10f", c1.real(), c1.imag())
346  // }
347  //}
348  }

351 inline Beta_projectors::Beta_projectors(Simulation_context_base const& ctx__,
352  Communicator const& comm__,
353  Gvec const& gkvec__)
354  : ctx_(ctx__)
355  , comm_(comm__)
356  , unit_cell_(ctx__.unit_cell())
357  , gkvec_(gkvec__)
358  , lmax_beta_(unit_cell_.lmax())
359  , pu_(ctx__.processing_unit())
360 {
361  PROFILE("sirius::Beta_projectors::Beta_projectors");
362 
363  num_gkvec_loc_ = gkvec_.gvec_count(comm_.rank());
364 
365  beta_radial_integrals_ = std::unique_ptr<Radial_integrals_beta>(new Radial_integrals_beta(unit_cell_, ctx_.gk_cutoff(), 20));
366 
367  split_in_chunks();
368 
369  generate_beta_gk_t(ctx_);
370 
371  if (pu_ == GPU) {
372  gkvec_coord_ = mdarray<double, 2>(3, num_gkvec_loc_, ctx__.dual_memory_t());
373  /* copy G+k vectors */
374  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
375  int igk = gkvec_.gvec_offset(comm_.rank()) + igk_loc;
376  auto vgk = gkvec_.gkvec(igk);
377  for (auto x: {0, 1, 2}) {
378  gkvec_coord_(x, igk_loc) = vgk[x];
379  }
380  }
381  gkvec_coord_.copy<memory_t::host, memory_t::device>();
382 
383  beta_gk_t_.allocate(memory_t::device);
384  beta_gk_t_.copy<memory_t::host, memory_t::device>();
385  }
386  beta_gk_gpu_ = matrix<double_complex>(num_gkvec_loc_, max_num_beta_, memory_t::none);
387 
388  beta_gk_a_ = matrix<double_complex>(num_gkvec_loc_, unit_cell_.mt_lo_basis_size());
389 
390  #pragma omp for
391  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
392  double phase = twopi * (gkvec_.vk() * unit_cell_.atom(ia).position());
393  double_complex phase_k = std::exp(double_complex(0.0, phase));
394 
395  std::vector<double_complex> phase_gk(num_gkvec_loc_);
396  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
397  int igk = gkvec_.gvec_offset(comm_.rank()) + igk_loc;
398  auto G = gkvec_.gvec(igk);
399  phase_gk[igk_loc] = std::conj(ctx__.gvec_phase_factor(G, ia) * phase_k);
400  }
401 
402  for (int xi = 0; xi < unit_cell_.atom(ia).mt_lo_basis_size(); xi++) {
403  for (int igk_loc = 0; igk_loc < num_gkvec_loc_; igk_loc++) {
404  beta_gk_a_(igk_loc, unit_cell_.atom(ia).offset_lo() + xi) =
405  beta_gk_t_(igk_loc, unit_cell_.atom(ia).type().offset_lo() + xi) * phase_gk[igk_loc];
406  }
407  }
408  }

411 inline void Beta_projectors::generate_beta_gk_t(Simulation_context_base const& ctx__)
412 {
413  PROFILE("sirius::Beta_projectors::generate_beta_gk_t");
414 
415  if (!num_beta_t_) {
416  return;
417  }
418 
419  /* allocate array */
420  beta_gk_t_ = matrix<double_complex>(gkvec_.gvec_count(comm_.rank()), num_beta_t_);
421 
422  /* compute <G+k|beta> */
423  #pragma omp parallel for
424  for (int igkloc = 0; igkloc < gkvec_.gvec_count(comm_.rank()); igkloc++) {
425  int igk = gkvec_.gvec_offset(comm_.rank()) + igkloc;
426  double gk = gkvec_.gvec_len(igk);
427  /* vs = {r, theta, phi} */
428  auto vs = SHT::spherical_coordinates(gkvec_.gkvec_cart(igk));
429  /* compute real spherical harmonics for G+k vector */
430  std::vector<double> gkvec_rlm(Utils::lmmax(lmax_beta_));
431  SHT::spherical_harmonics(lmax_beta_, vs[1], vs[2], &gkvec_rlm[0]);
432  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
433  auto& atom_type = unit_cell_.atom_type(iat);
434  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
435  int l = atom_type.indexb(xi).l;
436  int lm = atom_type.indexb(xi).lm;
437  int idxrf = atom_type.indexb(xi).idxrf;
438 
439  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
440  beta_gk_t_(igkloc, atom_type.offset_lo() + xi) = z * gkvec_rlm[lm] * beta_radial_integrals_->value(idxrf, iat, gk);
441  }
442  }
443  }
444 
445  if (unit_cell_.parameters().control().print_checksum_) {
446  auto c1 = beta_gk_t_.checksum();
447  comm_.allreduce(&c1, 1);
448  if (comm_.rank() == 0) {
449  DUMP("checksum(beta_gk_t) : %18.10f %18.10f", c1.real(), c1.imag())
450  }
451  }

454 inline void Beta_projectors::split_in_chunks()
455 {
456  /* split beta-projectors into chunks */
457  int num_atoms_in_chunk = (comm_.size() == 1) ? unit_cell_.num_atoms() : std::min(unit_cell_.num_atoms(), 256);
458  int num_beta_chunks = unit_cell_.num_atoms() / num_atoms_in_chunk + std::min(1, unit_cell_.num_atoms() % num_atoms_in_chunk);
459  splindex<block> spl_beta_chunks(unit_cell_.num_atoms(), num_beta_chunks, 0);
460 
461  beta_chunks_ = mdarray<beta_chunk_t, 1>(num_beta_chunks);
462 
463  int offset_in_beta_gk = 0;
464 
465  for (int ib = 0; ib < num_beta_chunks; ib++) {
466  /* number of atoms in chunk */
467  int na = spl_beta_chunks.local_size(ib);
468  beta_chunks_(ib).num_atoms_ = na;
469  beta_chunks_(ib).desc_ = mdarray<int, 2>(4, na);
470  beta_chunks_(ib).atom_pos_ = mdarray<double, 2>(3, na);
471 
472  int num_beta{0};
473 
474  for (int i = 0; i < na; i++) {
475  /* global index of atom by local index and chunk */
476  int ia = spl_beta_chunks.global_index(i, ib);
477  auto pos = unit_cell_.atom(ia).position();
478  auto& type = unit_cell_.atom(ia).type();
479  /* atom fractional coordinates */
480  for (int x: {0, 1, 2}) {
481  beta_chunks_(ib).atom_pos_(x, i) = pos[x];
482  }
483  /* number of beta functions for atom */
484  beta_chunks_(ib).desc_(beta_desc_idx::nbf, i) = type.mt_basis_size();
485  /* offset in beta_gk*/
486  beta_chunks_(ib).desc_(beta_desc_idx::offset, i) = num_beta;
487  /* offset in beta_gk_t */
488  beta_chunks_(ib).desc_(beta_desc_idx::offset_t, i) = type.offset_lo();
489  /* global index of atom */
490  beta_chunks_(ib).desc_(beta_desc_idx::ia, i) = ia;
491 
492  num_beta += type.mt_basis_size();
493  }
494  /* number of beta-projectors in this chunk */
495  beta_chunks_(ib).num_beta_ = num_beta;
496  beta_chunks_(ib).offset_ = offset_in_beta_gk;
497  offset_in_beta_gk += num_beta;
498 
499  if (pu_ == GPU) {
500  beta_chunks_[ib].desc_.allocate(memory_t::device);
501  beta_chunks_[ib].desc_.copy<memory_t::host, memory_t::device>();
502 
503  beta_chunks_[ib].atom_pos_.allocate(memory_t::device);
504  beta_chunks_[ib].atom_pos_.copy<memory_t::host, memory_t::device>();
505  }
506  }
507 
508  max_num_beta_ = 0;
509  for (int ib = 0; ib < num_beta_chunks; ib++) {
510  max_num_beta_ = std::max(max_num_beta_, beta_chunks_(ib).num_beta_);
511  }
512 
513  num_beta_t_ = 0;
514  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
515  num_beta_t_ += unit_cell_.atom_type(iat).mt_lo_basis_size();
516  }
517 }

544 template<>
545 inline void Beta_projectors::inner<double_complex>(int chunk__,
546  wave_functions& phi__,
547  int idx0__,
548  int n__,
549  mdarray<double_complex, 2> &beta_gk,
550  mdarray<double, 1> &beta_phi)
551 {
552  PROFILE("sirius::Beta_projectors::inner");
553 
554  assert(num_gkvec_loc_ == phi__.pw_coeffs().num_rows_loc());
555 
556  int nbeta = beta_chunk(chunk__).num_beta_;
557 
558  if (static_cast<size_t>(nbeta * n__) > beta_phi.size()) {
559  beta_phi = mdarray<double, 1>(2 * nbeta * n__);
560  if (pu_ == GPU) {
561  beta_phi.allocate(memory_t::device);
562  }
563  }
564 
565  switch (pu_) {
566  case CPU: {
567  /* compute <beta|phi> */
568  linalg<CPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc_,
569  beta_gk.at<CPU>(), num_gkvec_loc_,
570  phi__.pw_coeffs().prime().at<CPU>(0, idx0__), phi__.pw_coeffs().prime().ld(),
571  (double_complex*)beta_phi.at<CPU>(), nbeta);
572  break;
573  }
574  case GPU: {
575  #ifdef __GPU
576  linalg<GPU>::gemm(2, 0, nbeta, n__, num_gkvec_loc_, beta_gk.at<GPU>(), num_gkvec_loc_,
577  phi__.pw_coeffs().prime().at<GPU>(0, idx0__), phi__.pw_coeffs().prime().ld(),
578  (double_complex*)beta_phi.at<GPU>(), nbeta);
