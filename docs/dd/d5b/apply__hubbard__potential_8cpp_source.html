29 /** This function computes the Hubbard contribution to the hamiltonian and add it to ophi.
30  * The S matrix is already applied to phi_i */
31 void Hubbard::apply_hubbard_potential(Wave_functions& hub_wf, spin_range spins__, const int idx__, const int n__,
32  Wave_functions& phi, Wave_functions& hphi)
33 {
34  dmatrix<double_complex> dm(hub_wf.num_wf(), n__);
35  dm.zero();
36 
37  if (ctx_.processing_unit() == device_t::GPU) {
38  dm.allocate(memory_t::device);
39  }
40 
41  /* First calculate the local part of the projections
42  dm(i, n) = <phi_i| S |psi_{nk}> */
43  inner(ctx_.spla_context(), spins__, hub_wf, 0, hub_wf.num_wf(), phi, idx__, n__, dm, 0, 0);
44 
45  dmatrix<double_complex> Up(hub_wf.num_wf(), n__);
46  Up.zero();
47 
48  #pragma omp parallel for schedule(static)
49  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ++ia) {
50  const auto& atom = ctx_.unit_cell().atom(ia);
51  if (atom.type().hubbard_correction()) {
52  const int lmax_at = 2 * atom.type().indexr_hub().am(0).l() + 1;
53  // we apply the hubbard correction. For now I have no papers
54  // giving me the formula for the SO case so I rely on QE for it
55  // but I do not like it at all
56  if (ctx_.num_mag_dims() == 3) {
57  for (int s1 = 0; s1 < ctx_.num_spins(); s1++) {
58  for (int s2 = 0; s2 < ctx_.num_spins(); s2++) {
59  const int ind = (s1 == s2) * s1 + (1 + 2 * s2 + s1) * (s1 != s2);
60 
61  // TODO: replace this with matrix matrix multiplication
62 
63  for (int nbd = 0; nbd < n__; nbd++) {
64  for (int m1 = 0; m1 < lmax_at; m1++) {
65  for (int m2 = 0; m2 < lmax_at; m2++) {
66  Up(this->number_of_hubbard_orbitals_ * s1 + this->offset_[ia] + m1, nbd) +=
67  this->hubbard_potential_(m2, m1, ind, ia) *
68  dm(this->number_of_hubbard_orbitals_ * s2 + this->offset_[ia] + m2, nbd);
69  }
70  }
71  }
72  }
73  }
74  } else {
75  // Conventional LDA or colinear magnetism
76  for (int nbd = 0; nbd < n__; nbd++) {
77  for (int m1 = 0; m1 < lmax_at; m1++) {
78  for (int m2 = 0; m2 < lmax_at; m2++) {
79  Up(this->offset_[ia] + m1, nbd) += this->hubbard_potential_(m2, m1, spins__(), ia) *
80  dm(this->offset_[ia] + m2, nbd);
81  }
82  }
83  }
84  }
85  }
86  }
87 
88  if (ctx_.processing_unit() == device_t::GPU) {
89  Up.allocate(memory_t::device);
90  Up.copy_to(memory_t::device);
91  }
92 
93  transform<double_complex>(ctx_.spla_context(), spins__(), 1.0, {&hub_wf}, 0, hub_wf.num_wf(),
94  Up, 0, 0, 1.0, {&hphi}, idx__, n__);
95 }
96 
97 }
