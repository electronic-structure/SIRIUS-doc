36void
37Band::diag_full_potential_first_variation_exact(Hamiltonian_k<double>& Hk__) const
38{
39 PROFILE("sirius::Band::diag_fv_exact");
40
41 auto& kp = Hk__.kp();
42
43 auto& solver = ctx_.gen_evp_solver();
44
45 /* total eigen-value problem size */
46 int ngklo = kp.gklo_basis_size();
47
48 /* block size of scalapack 2d block-cyclic distribution */
49 int bs = ctx_.cyclic_block_size();
50
51 sddk::dmatrix<double_complex> h(ngklo, ngklo, ctx_.blacs_grid(), bs, bs, ctx_.mem_pool(solver.host_memory_t()));
52 sddk::dmatrix<double_complex> o(ngklo, ngklo, ctx_.blacs_grid(), bs, bs, ctx_.mem_pool(solver.host_memory_t()));
53
54 /* setup Hamiltonian and overlap */
55 Hk__.set_fv_h_o(h, o);
56
57 if (ctx_.gen_evp_solver().type() == ev_solver_t::cusolver) {
58 auto& mpd = ctx_.mem_pool(memory_t::device);
59 h.allocate(mpd);
60 o.allocate(mpd);
61 kp.fv_eigen_vectors().allocate(mpd);
62 }
63
64 if (ctx_.cfg().control().verification() >= 1) {
65 double max_diff = check_hermitian(h, ngklo);
66 if (max_diff > 1e-12) {
67 std::stringstream s;
68 s << "H matrix is not hermitian" << std::endl
69 << "max error: " << max_diff;
70 TERMINATE(s);
71 }
72 max_diff = check_hermitian(o, ngklo);
73 if (max_diff > 1e-12) {
74 std::stringstream s;
75 s << "O matrix is not hermitian" << std::endl
76 << "max error: " << max_diff;
77 TERMINATE(s);
78 }
79 }
80
81 if (ctx_.print_checksum()) {
82 auto z1 = h.checksum();
83 auto z2 = o.checksum();
84 kp.comm().allreduce(&z1, 1);
85 kp.comm().allreduce(&z2, 1);
86 if (kp.comm().rank() == 0) {
87 utils::print_checksum("h_lapw", z1);
88 utils::print_checksum("o_lapw", z2);
89 }
90 }
91
92 assert(kp.gklo_basis_size() > ctx_.num_fv_states());
93
94 std::vector<double> eval(ctx_.num_fv_states());
95
96 ctx_.print_memory_usage(__FILE__, __LINE__);
97 if (solver.solve(kp.gklo_basis_size(), ctx_.num_fv_states(), h, o, eval.data(), kp.fv_eigen_vectors())) {
98 RTE_THROW("error in generalized eigen-value problem");
99 }
100 ctx_.print_memory_usage(__FILE__, __LINE__);
101
102 if (ctx_.gen_evp_solver().type() == ev_solver_t::cusolver) {
103 h.deallocate(memory_t::device);
104 o.deallocate(memory_t::device);
105 kp.fv_eigen_vectors().deallocate(memory_t::device);
106 }
107 kp.set_fv_eigen_values(&eval[0]);
108
109 for (int i = 0; i < ctx_.num_fv_states(); i++) {
110 kp.message(4, __function_name__, "eval[%i]=%20.16f\n", i, eval[i]);
111 }
112
113 if (ctx_.print_checksum()) {
114 auto z1 = kp.fv_eigen_vectors().checksum();
115 kp.comm().allreduce(&z1, 1);
116 if (kp.comm().rank() == 0) {
117 utils::print_checksum("fv_eigen_vectors", z1);
118 }
119 }
120
121 /* remap to slab */
122 kp.fv_eigen_vectors_slab().pw_coeffs(0).remap_from(kp.fv_eigen_vectors(), 0);
123 kp.fv_eigen_vectors_slab().mt_coeffs(0).remap_from(kp.fv_eigen_vectors(), kp.num_gkvec());
124
125 /* renormalize wave-functions */
126 if (ctx_.valence_relativity() == relativity_t::iora) {
127 Wave_functions<double> ofv(kp.gkvec_partition(), unit_cell_.num_atoms(),
128 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, ctx_.num_fv_states(),
129 ctx_.preferred_memory_t(), 1);
130 if (ctx_.processing_unit() == device_t::GPU) {
131 kp.fv_eigen_vectors_slab().allocate(spin_range(0), memory_t::device);
132 kp.fv_eigen_vectors_slab().copy_to(spin_range(0), memory_t::device, 0, ctx_.num_fv_states());
133 ofv.allocate(spin_range(0), memory_t::device);
134 }
135
136 Hk__.apply_fv_h_o(false, false, 0, ctx_.num_fv_states(), kp.fv_eigen_vectors_slab(), nullptr, &ofv);
137
138 if (ctx_.processing_unit() == device_t::GPU) {
139 kp.fv_eigen_vectors_slab().deallocate(spin_range(0), memory_t::device);
140 }
141
142 //if (true) {
143 // Wave_functions phi(kp.gkvec_partition(), unit_cell_.num_atoms(),
144 // [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, ctx_.num_fv_states(),
145 // ctx_.preferred_memory_t(), 1);
146 // Wave_functions ofv(kp.gkvec_partition(), unit_cell_.num_atoms(),
147 // [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, ctx_.num_fv_states(),
148 // ctx_.preferred_memory_t(), 1);
149 // phi.allocate(spin_range(0), memory_t::device);
150 // ofv.allocate(spin_range(0), memory_t::device);
151
152 // for (int i = 0; i < kp.num_gkvec(); i++) {
153 // phi.zero(device_t::CPU, 0, 0, ctx_.num_fv_states());
154 // for (int j = 0; j < ctx_.num_fv_states(); j++) {
155 // phi.pw_coeffs(0).prime(i, j) = 1.0;
156 // }
157 // phi.copy_to(spin_range(0), memory_t::device, 0, ctx_.num_fv_states());
158 // Hk__.apply_fv_h_o(false, false, 0, ctx_.num_fv_states(), phi, nullptr, &ofv);
159 // }
160
161 // for (int i = 0; i < unit_cell_.mt_lo_basis_size(); i++) {
162 // phi.zero(device_t::CPU, 0, 0, ctx_.num_fv_states());
163 // for (int j = 0; j < ctx_.num_fv_states(); j++) {
164 // phi.mt_coeffs(0).prime(i, j) = 1.0;
165 // }
166 // phi.copy_to(spin_range(0), memory_t::device, 0, ctx_.num_fv_states());
167 // Hk__.apply_fv_h_o(false, false, 0, ctx_.num_fv_states(), phi, nullptr, &ofv);
168 // }
169 //}
170
171 std::vector<double> norm(ctx_.num_fv_states(), 0);
172 #pragma omp parallel for schedule(static)
173 for (int i = 0; i < ctx_.num_fv_states(); i++) {
174 for (int j = 0; j < ofv.pw_coeffs(0).num_rows_loc(); j++) {
175 norm[i] += std::real(std::conj(kp.fv_eigen_vectors_slab().pw_coeffs(0).prime(j, i)) * ofv.pw_coeffs(0).prime(j, i));
176 }
177 for (int j = 0; j < ofv.mt_coeffs(0).num_rows_loc(); j++) {
178 norm[i] += std::real(std::conj(kp.fv_eigen_vectors_slab().mt_coeffs(0).prime(j, i)) * ofv.mt_coeffs(0).prime(j, i));
179 }
180 }
181 kp.comm().allreduce(norm);
182 if (ctx_.verbosity() >= 2) {
183 for (int i = 0; i < ctx_.num_fv_states(); i++) {
184 kp.message(2, __function_name__, "norm(%i)=%18.12f\n", i, norm[i]);
185 }
186 }
187 #pragma omp parallel for schedule(static)
188 for (int i = 0; i < ctx_.num_fv_states(); i++) {
189 norm[i] = 1 / std::sqrt(norm[i]);
190 for (int j = 0; j < ofv.pw_coeffs(0).num_rows_loc(); j++) {
191 kp.fv_eigen_vectors_slab().pw_coeffs(0).prime(j, i) *= norm[i];
192 }
193 for (int j = 0; j < ofv.mt_coeffs(0).num_rows_loc(); j++) {
194 kp.fv_eigen_vectors_slab().mt_coeffs(0).prime(j, i) *= norm[i];
195 }
196 }
197 }
198
199 if (ctx_.cfg().control().verification() >= 2) {
200 kp.message(1, __function_name__, "%s", "checking application of H and O\n");
201 /* check application of H and O */
202 Wave_functions<double> hphi(kp.gkvec_partition(), unit_cell_.num_atoms(),
203 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, ctx_.num_fv_states(),
204 ctx_.preferred_memory_t());
205 Wave_functions<double> ophi(kp.gkvec_partition(), unit_cell_.num_atoms(),
206 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, ctx_.num_fv_states(),
207 ctx_.preferred_memory_t());
208
209 if (ctx_.processing_unit() == device_t::GPU) {
210 kp.fv_eigen_vectors_slab().allocate(spin_range(0), memory_t::device);
211 kp.fv_eigen_vectors_slab().copy_to(spin_range(0), memory_t::device, 0, ctx_.num_fv_states());
212 hphi.allocate(spin_range(0), memory_t::device);
213 ophi.allocate(spin_range(0), memory_t::device);
214 }
215
216 Hk__.apply_fv_h_o(false, false, 0, ctx_.num_fv_states(), kp.fv_eigen_vectors_slab(), &hphi, &ophi);
217
218 dmatrix<double_complex> hmlt(ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.blacs_grid(),
219 ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
220 dmatrix<double_complex> ovlp(ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.blacs_grid(),
221 ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
222
223 inner(ctx_.spla_context(), spin_range(0), kp.fv_eigen_vectors_slab(), 0, ctx_.num_fv_states(),
224 hphi, 0, ctx_.num_fv_states(), hmlt, 0, 0);
225 inner(ctx_.spla_context(), spin_range(0), kp.fv_eigen_vectors_slab(), 0, ctx_.num_fv_states(),
226 ophi, 0, ctx_.num_fv_states(), ovlp, 0, 0);
227
228 double max_diff{0};
229 for (int i = 0; i < hmlt.num_cols_local(); i++) {
230 int icol = hmlt.icol(i);
231 for (int j = 0; j < hmlt.num_rows_local(); j++) {
232 int jrow = hmlt.irow(j);
233 if (icol == jrow) {
234 max_diff = std::max(max_diff, std::abs(hmlt(j, i) - eval[icol]));
235 } else {
236 max_diff = std::max(max_diff, std::abs(hmlt(j, i)));
237 }
238 }
239 }
240 if (max_diff > 1e-9) {
241 std::stringstream s;
242 s << "application of Hamiltonian failed, maximum error: " << max_diff;
243 WARNING(s);
244 }
245
246 max_diff = 0;
247 for (int i = 0; i < ovlp.num_cols_local(); i++) {
248 int icol = ovlp.icol(i);
249 for (int j = 0; j < ovlp.num_rows_local(); j++) {
250 int jrow = ovlp.irow(j);
251 if (icol == jrow) {
252 max_diff = std::max(max_diff, std::abs(ovlp(j, i) - 1.0));
253 } else {
254 max_diff = std::max(max_diff, std::abs(ovlp(j, i)));
255 }
256 }
257 }
258 if (max_diff > 1e-9) {
259 std::stringstream s;
260 s << "application of overlap failed, maximum error: " << max_diff;
261 WARNING(s);
262 }
263 }

266void Band::get_singular_components(Hamiltonian_k<double>& Hk__, mdarray<double, 2>& o_diag__) const
267{
268 PROFILE("sirius::Band::get_singular_components");
269
270 auto& kp = Hk__.kp();
271
272 mdarray<double, 2> diag1(kp.num_gkvec_loc(), 1, memory_t::host, "diag1");
273 for (int ig = 0; ig < kp.num_gkvec_loc(); ig++) {
274 diag1[ig] = 1;
275 }
276
277 if (ctx_.processing_unit() == device_t::GPU) {
278 diag1.allocate(memory_t::device).copy_to(memory_t::device);
279 }
280
281 auto& psi = kp.singular_components();
282
283 int ncomp = psi.num_wf();
284
285 ctx_.message(3, __function_name__, "number of singular components: %i\n", ncomp);
286
287 auto& itso = ctx_.cfg().iterative_solver();
288
289 int num_phi = itso.subspace_size() * ncomp;
290
291 Wave_functions<double> phi(kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t());
292 Wave_functions<double> ophi(kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t());
293 Wave_functions<double> opsi(kp.gkvec_partition(), ncomp, ctx_.preferred_memory_t());
294 Wave_functions<double> res(kp.gkvec_partition(), ncomp, ctx_.preferred_memory_t());
295
296 int bs = ctx_.cyclic_block_size();
297
298 dmatrix<double_complex> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
299 dmatrix<double_complex> ovlp_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
300 dmatrix<double_complex> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
301
302 if (ctx_.processing_unit() == device_t::GPU) {
303 psi.pw_coeffs(0).allocate(memory_t::device);
304 psi.pw_coeffs(0).copy_to(memory_t::device, 0, ncomp);
305 phi.pw_coeffs(0).allocate(memory_t::device);
306 res.pw_coeffs(0).allocate(memory_t::device);
307 ophi.pw_coeffs(0).allocate(memory_t::device);
308 opsi.pw_coeffs(0).allocate(memory_t::device);
309 if (ctx_.blacs_grid().comm().size() == 1) {
310 evec.allocate(memory_t::device);
311 ovlp.allocate(memory_t::device);
312 }
313 }
314
315 mdarray<double, 1> eval(ncomp);
316 mdarray<double, 1> eval_old(ncomp);
317 eval = [](){return -1;};
318
319 phi.copy_from(ctx_.processing_unit(), ncomp, psi, 0, 0, 0, 0);
320
321 if (ctx_.print_checksum()) {
322 phi.print_checksum(ctx_.processing_unit(), "phi", 0, ncomp);
323 }
324
325 /* current subspace size */
326 int N{0};
327
328 /* number of newly added basis functions */
329 int n = ncomp;
330
331 ctx_.print_memory_usage(__FILE__, __LINE__);
332
333 auto& std_solver = ctx_.std_evp_solver();
334
335 /* tolerance for the norm of L2-norms of the residuals, used for
336 * relative convergence criterion. We can only compute this after
337 * we have the first residual norms available */
338 double relative_frobenius_tolerance{0};
339 double current_frobenius_norm{0};
340
341 /* start iterative diagonalization */
342 for (int k = 0; k < itso.num_steps(); k++) {
343 /* apply Hamiltonian and overlap operators to the new basis functions */
344 Hk__.apply_fv_h_o(true, false, N, n, phi, nullptr, &ophi);
345 if (ctx_.processing_unit() == device_t::GPU) {
346 ophi.copy_to(spin_range(0), memory_t::device, N, n);
347 }
348
349 if (ctx_.cfg().control().verification() >= 1) {
350 set_subspace_mtrx<double_complex, double_complex>(0, N + n, 0, phi, ophi, ovlp);
351
352 if (ctx_.cfg().control().verification() >= 2) {
353 ovlp.serialize("overlap", N + n);
354 }
355
356 double max_diff = check_hermitian(ovlp, N + n);
357 if (max_diff > 1e-12) {
358 std::stringstream s;
359 s << "overlap matrix is not hermitian, max_err = " << max_diff;
360 TERMINATE(s);
361 }
362 }
363
364 orthogonalize<std::complex<double>>(ctx_.spla_context(), ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), spin_range(0), phi, ophi,
365 N, n, ovlp, res);
366
367 /* setup eigen-value problem
368 * N is the number of previous basis functions
369 * n is the number of new basis functions */
370 set_subspace_mtrx<double_complex, double_complex>(N, n, 0, phi, ophi, ovlp, &ovlp_old);
371
372 if (ctx_.cfg().control().verification() >= 1) {
373 if (ctx_.cfg().control().verification() >= 2) {
374 ovlp.serialize("overlap_ortho", N + n);
375 }
376
377 double max_diff = check_hermitian(ovlp, N + n);
378 if (max_diff > 1e-12) {
379 std::stringstream s;
380 s << "overlap matrix is not hermitian, max_err = " << max_diff;
381 TERMINATE(s);
382 }
383 }
384
385 /* increase size of the variation space */
386 N += n;
387
388 eval >> eval_old;
389
390 /* solve standard eigen-value problem with the size N */
391 if (std_solver.solve(N, ncomp, ovlp, &eval[0], evec)) {
392 std::stringstream s;
393 s << "[sirius::Band::get_singular_components] error in diagonalization";
394 TERMINATE(s);
395 }
396
397 for (int i = 0; i < ncomp; i++) {
398 if (eval[i] < 0) {
399 std::stringstream s;
400 s << "[sirius::Band::get_singular_components] overlap matrix is not positively defined";
401 TERMINATE(s);
402 }
403 }
404
405 kp.message(3, __function_name__, "step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
406 for (int i = 0; i < ncomp; i++) {
407 kp.message(4, __function_name__, "eval[%i]=%20.16f, diff=%20.16f\n", i, eval[i], std::abs(eval[i] - eval_old[i]));
408 }
409
410 bool last_iteration = k == (itso.num_steps() - 1);
411
412 /* don't compute residuals on last iteration */
413 if (!last_iteration) {
414 /* get new preconditionined residuals, and also opsi and psi as a by-product */
415 auto result = sirius::residuals<double_complex, double_complex>(
416 ctx_, ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), spin_range(0), N, ncomp, 0, eval, evec, ophi, phi, opsi, psi,
417 res, o_diag__, diag1, itso.converge_by_energy(), itso.residual_tolerance(),
418 [&](int i, int ispn) { return std::abs(eval[i] - eval_old[i]) < itso.energy_tolerance(); });
419 n = result.unconverged_residuals;
420 current_frobenius_norm = result.frobenius_norm;
421
422 /* set the relative tolerance convergence criterion */
423 if (k == 0) {
424 relative_frobenius_tolerance = current_frobenius_norm * itso.relative_tolerance();
425 }
426
427 kp.message(3, __function_name__, "number of added residuals: %i\n", n);
428 if (ctx_.print_checksum()) {
429 res.print_checksum(ctx_.processing_unit(), "res", 0, n);
430 }
431 }
432 /* verify convergence criteria */
433 bool converged_by_relative_tol = k > 0 && current_frobenius_norm < relative_frobenius_tolerance ;
434 bool converged_by_absolute_tol = n <= itso.min_num_res();
435 bool converged = converged_by_absolute_tol || converged_by_relative_tol;
436
437 /* check if running out of space */
438 bool should_restart = N + n > num_phi;
439
440 /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
441 if (should_restart || converged || last_iteration) {
442 PROFILE("sirius::Band::get_singular_components|update_phi");
443 /* recompute wave-functions */
444 /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
445 transform<double_complex, double_complex>(ctx_.spla_context(), 0, phi, 0, N, evec, 0, 0, psi, 0, ncomp);
446
447 /* exit the loop if the eigen-vectors are converged or this is a last iteration */
448 if (converged || last_iteration) {
449 break;
450 } else { /* otherwise, set Psi as a new trial basis */
451 kp.message(3, __function_name__, "%s", "subspace size limit reached\n");
452
453 if (itso.converge_by_energy()) {
454 transform<double_complex, double_complex>(ctx_.spla_context(), 0, ophi, 0, N, evec, 0, 0, opsi, 0, ncomp);
455 }
456
457 ovlp_old.zero();
458 for (int i = 0; i < ncomp; i++) {
459 ovlp_old.set(i, i, eval[i]);
460 }
461 /* update basis functions */
462 phi.copy_from(ctx_.processing_unit(), ncomp, psi, 0, 0, 0, 0);
463 ophi.copy_from(ctx_.processing_unit(), ncomp, opsi, 0, 0, 0, 0);
464 /* number of basis functions that we already have */
465 N = ncomp;
466 }
467 }
468 /* expand variational subspace with new basis vectors obtatined from residuals */
469 phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, N);
470 }
471
472 if (ctx_.processing_unit() == device_t::GPU) {
473 psi.pw_coeffs(0).copy_to(memory_t::host, 0, ncomp);
474 psi.pw_coeffs(0).deallocate(memory_t::device);
475 }
476
477 kp.message(2, __function_name__, "smallest eigen-value of the singular components: %20.16f\n", eval[0]);
478}

480void Band::diag_full_potential_first_variation_davidson(Hamiltonian_k<double>& Hk__) const
481{
482 PROFILE("sirius::Band::diag_fv_davidson");
483
484 auto& kp = Hk__.kp();
485
486 auto h_o_diag = Hk__.get_h_o_diag_lapw<3>();
487
488 if (ctx_.print_checksum()) {
489 auto cs1 = h_o_diag.first.checksum();
490 auto cs2 = h_o_diag.second.checksum();
491 if (kp.comm().rank() == 0) {
492 utils::print_checksum("h_dial_lapw", cs1);
493 utils::print_checksum("o_diag_lapw", cs2);
494 }
495 }
496
497 get_singular_components(Hk__, h_o_diag.second);
498
499 /* short notation for number of target wave-functions */
500 int num_bands = ctx_.num_fv_states();
501
502 auto& itso = ctx_.cfg().iterative_solver();
503
504 /* short notation for target wave-functions */
505 auto& psi = kp.fv_eigen_vectors_slab();
506
507 /* total number of local orbitals */
508 int nlo = ctx_.unit_cell().mt_lo_basis_size();
509
510 /* number of singular components */
511 int ncomp = kp.singular_components().num_wf();
512
513 /* number of auxiliary basis functions */
514 int num_phi = nlo + ncomp + itso.subspace_size() * num_bands;
515 /* sanity check */
516 if (num_phi >= kp.num_gkvec()) {
517 TERMINATE("subspace is too big");
518 }
519
520 /* allocate wave-functions */
521 Wave_functions<double> phi(kp.gkvec_partition(), unit_cell_.num_atoms(),
522 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi,
523 ctx_.preferred_memory_t());
524 Wave_functions<double> hphi(kp.gkvec_partition(), unit_cell_.num_atoms(),
525 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi,
526 ctx_.preferred_memory_t());

528 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi,
529 ctx_.preferred_memory_t());
530 Wave_functions<double> hpsi(kp.gkvec_partition(), unit_cell_.num_atoms(),
531 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_bands,
532 ctx_.preferred_memory_t());
533 Wave_functions<double> opsi(kp.gkvec_partition(), unit_cell_.num_atoms(),
534 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_bands,
535 ctx_.preferred_memory_t());
536
537 /* residuals */
538 /* res is also used as a temporary array in orthogonalize() and the first time nlo + ncomp + num_bands
539 * states will be orthogonalized */
540 Wave_functions<double> res(kp.gkvec_partition(), unit_cell_.num_atoms(),
541 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, nlo + ncomp + num_bands,
542 ctx_.preferred_memory_t());
543
544 //auto mem_type = (gen_evp_solver_->type() == ev_magma) ? memory_t::host_pinned : memory_t::host;
545
546 int bs = ctx_.cyclic_block_size();
547
548 dmatrix<double_complex> hmlt(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
549 dmatrix<double_complex> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
550 dmatrix<double_complex> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
551 dmatrix<double_complex> hmlt_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
552
553 /* add pure local orbitals to the basis */
554 if (nlo) {
555 phi.pw_coeffs(0).zero(memory_t::host, 0, nlo);
556 phi.mt_coeffs(0).zero(memory_t::host, 0, nlo);
557 for (int ialoc = 0; ialoc < phi.spl_num_atoms().local_size(); ialoc++) {
558 int ia = phi.spl_num_atoms()[ialoc];
559 for (int xi = 0; xi < unit_cell_.atom(ia).mt_lo_basis_size(); xi++) {
560 phi.mt_coeffs(0).prime(phi.offset_mt_coeffs(ialoc) + xi, unit_cell_.atom(ia).offset_lo() + xi) = 1.0;
561 }
562 }
563 }
564
565 /* add singular components to the basis */
566 if (ncomp != 0) {
567 phi.mt_coeffs(0).zero(memory_t::host, nlo, ncomp);
568 for (int j = 0; j < ncomp; j++) {
569 std::memcpy(phi.pw_coeffs(0).prime().at(memory_t::host, 0, nlo + j),
570 kp.singular_components().pw_coeffs(0).prime().at(memory_t::host, 0, j),
571 phi.pw_coeffs(0).num_rows_loc() * sizeof(double_complex));
572 }
573 }
574
575 if (ctx_.processing_unit() == device_t::GPU) {
576 psi.allocate(spin_range(0), memory_t::device);
577 psi.copy_to(spin_range(0), memory_t::device, 0, num_bands);
578
579 phi.allocate(spin_range(0), memory_t::device);
580 phi.copy_to(spin_range(0), memory_t::device, 0, nlo + ncomp);
581
582 res.allocate(spin_range(0), memory_t::device);
583
584 hphi.allocate(spin_range(0), memory_t::device);
585 ophi.allocate(spin_range(0), memory_t::device);
586
587 hpsi.allocate(spin_range(0), memory_t::device);
588 opsi.allocate(spin_range(0), memory_t::device);
589
590 if (ctx_.blacs_grid().comm().size() == 1) {
591 evec.allocate(memory_t::device);
592 ovlp.allocate(memory_t::device);
593 hmlt.allocate(memory_t::device);
594 }
595 }
596
597 mdarray<double, 1> eval(num_bands);
598 mdarray<double, 1> eval_old(num_bands);
599 eval_old = [](){return -1.0;};
600
601 for (int i = 0; i < num_bands; i++) {
602 eval[i] = kp.fv_eigen_value(i);
603 }
604
605 /* trial basis functions */
606 phi.copy_from(ctx_.processing_unit(), num_bands, psi, 0, 0, 0, nlo + ncomp);
607
608 if (ctx_.print_checksum()) {

610 psi.print_checksum(ctx_.processing_unit(), "psi", 0, num_bands);
611 phi.print_checksum(ctx_.processing_unit(), "phi", 0, nlo + ncomp + num_bands);
612 }
613
614 /* current subspace size */
615 int N = 0;
616
617 /* number of newly added basis functions */
618 int n = nlo + ncomp + num_bands;
619
620 ctx_.print_memory_usage(__FILE__, __LINE__);
621
622 auto& std_solver = ctx_.std_evp_solver();
623
624 /* tolerance for the norm of L2-norms of the residuals, used for
625 * relative convergence criterion. We can only compute this after
626 * we have the first residual norms available */
627 double relative_frobenius_tolerance{0};
628 double current_frobenius_norm{0};
629
630 /* start iterative diagonalization */
631 for (int k = 0; k < itso.num_steps(); k++) {
632
633 bool last_iteration = k == (itso.num_steps() - 1);
634
635 /* apply Hamiltonian and overlap operators to the new basis functions */
636 if (k == 0) {
637 Hk__.apply_fv_h_o(false, true, 0, nlo, phi, &hphi, &ophi);
638 Hk__.apply_fv_h_o(false, false, nlo, ncomp + num_bands, phi, &hphi, &ophi);
639 } else {
640 Hk__.apply_fv_h_o(false, false, N, n, phi, &hphi, &ophi);
641 }
642
643 orthogonalize<std::complex<double>>(ctx_.spla_context(), ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), spin_range(0), phi,
644 hphi, ophi, N, n, ovlp, res);
645
646 /* setup eigen-value problem
647 * N is the number of previous basis functions
648 * n is the number of new basis functions */
649 set_subspace_mtrx<double_complex, double_complex>(N, n, 0, phi, hphi, hmlt, &hmlt_old);
650
651 /* increase size of the variation space */
652 N += n;
653
654 eval >> eval_old;
655
656 /* solve standard eigen-value problem with the size N */
657 if (std_solver.solve(N, num_bands, hmlt, &eval[0], evec)) {
658 std::stringstream s;
659 s << "[sirius::Band::diag_full_potential_first_variation_davidson] error in diagonalziation";
660 TERMINATE(s);
661 }
662 kp.message(2, __function_name__, "step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
663 for (int i = 0; i < num_bands; i++) {
664 kp.message(4, __function_name__, "eval[%i]=%20.16f, diff=%20.16f\n", i, eval[i], std::abs(eval[i] - eval_old[i]));
665 }
666
667 /* don't compute residuals on last iteration */
668 if (!last_iteration) {
669 /* get new preconditionined residuals, and also hpsi and opsi as a by-product */
670 auto result = sirius::residuals<double_complex, double_complex>(
671 ctx_, ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), spin_range(0), N, num_bands, 0, eval, evec, hphi, ophi, hpsi,
672 opsi, res, h_o_diag.first, h_o_diag.second, itso.converge_by_energy(), itso.residual_tolerance(),
673 [&](int i, int ispn) { return std::abs(eval[i] - eval_old[i]) < itso.energy_tolerance(); });
674 n = result.unconverged_residuals;
675 current_frobenius_norm = result.frobenius_norm;
676
677 /* set the relative tolerance convergence criterion */
678 if (k == 0) {
679 relative_frobenius_tolerance = current_frobenius_norm * itso.relative_tolerance();
680 }
681 }
682
683 /* verify convergence criteria */
684 bool converged_by_relative_tol = k > 0 && current_frobenius_norm < relative_frobenius_tolerance ;
685 bool converged_by_absolute_tol = n <= itso.min_num_res();
686 bool converged = converged_by_absolute_tol || converged_by_relative_tol;
687
688 /* check if running out of space */
689 bool should_restart = N + n > num_phi;
690
691 if (converged) {
692 kp.message(3, __function_name__, "converged by %s tolerance\n", converged_by_relative_tol ? "relative" : "absolute");
693 }
694
695 /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
696 if (should_restart || converged || last_iteration) {
697 PROFILE("sirius::Band::diag_fv_davidson|update_phi");
698 /* recompute wave-functions */
699 /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
700 transform<double_complex, double_complex>(ctx_.spla_context(), 0, phi, 0, N, evec, 0, 0, psi, 0, num_bands);
701
702 /* exit the loop if the eigen-vectors are converged or this is a last iteration */
703 if (converged || last_iteration) {
704 break;
705 } else { /* otherwise, set Psi as a new trial basis */
706 kp.message(3, __function_name__, "%s", "subspace size limit reached\n");
707 /* update basis functions */
708 /* first nlo + ncomp functions are fixed, don't update them */
709 phi.copy_from(ctx_.processing_unit(), num_bands, psi, 0, 0, 0, nlo + ncomp);
710 phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, nlo + ncomp + num_bands);
711 /* number of basis functions that we already have */
712 N = nlo + ncomp;
713 n += num_bands;
714 }
715 } else {
716 /* expand variational subspace with new basis vectors obtatined from residuals */
717 phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, N);
718 }
719 }
720
721 if (ctx_.processing_unit() == device_t::GPU) {
722 psi.pw_coeffs(0).copy_to(memory_t::host, 0, num_bands);
723 psi.mt_coeffs(0).copy_to(memory_t::host, 0, num_bands);
724 psi.deallocate(spin_range(0), memory_t::device);
725 }
726 kp.set_fv_eigen_values(&eval[0]);
727}
