160  #pragma omp parallel for schedule(static)

207  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());

209  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());

220  int jrow = hmlt.irow(j);
221  if (icol == jrow) {
222  max_diff = std::max(max_diff, std::abs(hmlt(j, i) - eval[icol]));
223  } else {
224  max_diff = std::max(max_diff, std::abs(hmlt(j, i)));
225  }
226  }
227  }

229  std::stringstream s;
230  s << "application of Hamiltonian failed, maximum error: " << max_diff;
231  WARNING(s);
232  }
233 
234  max_diff = 0;
235  for (int i = 0; i < ovlp.num_cols_local(); i++) {

237  for (int j = 0; j < ovlp.num_rows_local(); j++) {
238  int jrow = ovlp.irow(j);
239  if (icol == jrow) {
240  max_diff = std::max(max_diff, std::abs(ovlp(j, i) - 1.0));
241  } else {
242  max_diff = std::max(max_diff, std::abs(ovlp(j, i)));
243  }
244  }
245  }
246  if (max_diff > 1e-9) {
247  std::stringstream s;

254 void Band::get_singular_components(Hamiltonian_k& Hk__, mdarray<double, 2>& o_diag__) const
255 {
256  PROFILE("sirius::Band::get_singular_components");
257 
258  auto& kp = Hk__.kp();
259 
260  mdarray<double, 2> diag1(kp.num_gkvec_loc(), 1, memory_t::host, "diag1");
261  for (int ig = 0; ig < kp.num_gkvec_loc(); ig++) {
262  diag1[ig] = 1;
263  }
264 
265  if (ctx_.processing_unit() == device_t::GPU) {
266  diag1.allocate(memory_t::device).copy_to(memory_t::device);
267  }
268 
269  auto& psi = kp.singular_components();
270 
271  int ncomp = psi.num_wf();
272 
273  ctx_.message(3, __func__, "number of singular components: %i\n", ncomp);
274 
275  auto& itso = ctx_.iterative_solver_input();
276 
277  int num_phi = itso.subspace_size_ * ncomp;
278 
279  Wave_functions phi(kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t());
280  Wave_functions ophi(kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t());
281  Wave_functions opsi(kp.gkvec_partition(), ncomp, ctx_.preferred_memory_t());
282  Wave_functions res(kp.gkvec_partition(), ncomp, ctx_.preferred_memory_t());
283 
284  int bs = ctx_.cyclic_block_size();
285 
286  dmatrix<double_complex> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
287  dmatrix<double_complex> ovlp_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
288  dmatrix<double_complex> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
289 
290  if (ctx_.processing_unit() == device_t::GPU) {
291  psi.pw_coeffs(0).allocate(memory_t::device);
292  psi.pw_coeffs(0).copy_to(memory_t::device, 0, ncomp);
293  phi.pw_coeffs(0).allocate(memory_t::device);
294  res.pw_coeffs(0).allocate(memory_t::device);
295  ophi.pw_coeffs(0).allocate(memory_t::device);
296  opsi.pw_coeffs(0).allocate(memory_t::device);
297  if (ctx_.blacs_grid().comm().size() == 1) {
298  evec.allocate(memory_t::device);
299  ovlp.allocate(memory_t::device);
300  }
301  }
302 
303  mdarray<double, 1> eval(ncomp);
304  mdarray<double, 1> eval_old(ncomp);
305  eval_old = [](){return 1e10;};
306 
307  phi.copy_from(ctx_.processing_unit(), ncomp, psi, 0, 0, 0, 0);
308 
309  if (ctx_.control().print_checksum_) {
310  phi.print_checksum(ctx_.processing_unit(), "phi", 0, ncomp);
311  }
312 
313  /* current subspace size */
314  int N{0};
315 
316  /* number of newly added basis functions */
317  int n = ncomp;
318 
319  ctx_.message(3, __func__, "iterative solver tolerance: %18.12f\n", ctx_.iterative_solver_tolerance());
320 
321  if (kp.comm().rank() == 0 && ctx_.control().print_memory_usage_) {
322  MEMORY_USAGE_INFO();
323  }
324 
325  auto& std_solver = ctx_.std_evp_solver();
326 
327  /* start iterative diagonalization */
328  for (int k = 0; k < itso.num_steps_; k++) {
329  /* apply Hamiltonian and overlap operators to the new basis functions */
330  Hk__.apply_fv_h_o(true, false, N, n, phi, nullptr, &ophi);
331  if (ctx_.processing_unit() == device_t::GPU) {
332  ophi.copy_to(spin_range(0), memory_t::device, N, n);
333  }
334 
335  if (ctx_.control().verification_ >= 1) {
336  set_subspace_mtrx(0, N + n, phi, ophi, ovlp);
337 
338  if (ctx_.control().verification_ >= 2) {
339  ovlp.serialize("overlap", N + n);
340  }
341 
342  double max_diff = check_hermitian(ovlp, N + n);
343  if (max_diff > 1e-12) {
344  std::stringstream s;
345  s << "overlap matrix is not hermitian, max_err = " << max_diff;
346  TERMINATE(s);
347  }
348  }
349 
350  orthogonalize(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), 0, phi, ophi, N, n, ovlp, res);
351 
352  /* setup eigen-value problem
353  * N is the number of previous basis functions
354  * n is the number of new basis functions */
355  set_subspace_mtrx(N, n, phi, ophi, ovlp, &ovlp_old);
356 
357  if (ctx_.control().verification_ >= 1) {
358  if (ctx_.control().verification_ >= 2) {
359  ovlp.serialize("overlap_ortho", N + n);
360  }
361 
362  double max_diff = check_hermitian(ovlp, N + n);
363  if (max_diff > 1e-12) {
364  std::stringstream s;
365  s << "overlap matrix is not hermitian, max_err = " << max_diff;
366  TERMINATE(s);
367  }
368  }
369 
370  /* increase size of the variation space */
371  N += n;
372 
373  eval >> eval_old;
374 
375  /* solve standard eigen-value problem with the size N */
376  if (std_solver.solve(N, ncomp, ovlp, &eval[0], evec)) {
377  std::stringstream s;
378  s << "error in diagonalziation";
379  TERMINATE(s);
380  }
381 
382  for (int i = 0; i < ncomp; i++) {
383  if (eval[i] < 0) {
384  std::stringstream s;
385  s << "overlap matrix is not positively defined";
386  TERMINATE(s);
387  }
388  }
389 
390  kp.message(3, __func__, "step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
391  for (int i = 0; i < ncomp; i++) {
392  kp.message(4, __func__, "eval[%i]=%20.16f, diff=%20.16f\n", i, eval[i], std::abs(eval[i] - eval_old[i]));
393  }
394 
395  /* don't compute residuals on last iteration */
396  if (k != itso.num_steps_ - 1) {
397  /* get new preconditionined residuals, and also opsi and psi as a by-product */
398  n = sirius::residuals(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), 0,
399  N, ncomp, eval, evec, ophi, phi, opsi, psi, res, o_diag__, diag1,
400  itso.converge_by_energy_, itso.residual_tolerance_,
401  [&](int i, int ispn){return std::abs(eval[i] - eval_old[i]) < itso.energy_tolerance_;});
402  }
403 
404  /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
405  if (N + n > num_phi || n <= itso.min_num_res_ || k == (itso.num_steps_ - 1)) {
406  utils::timer t1("sirius::Band::get_singular_components|update_phi");
407  /* recompute wave-functions */
408  /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
409  transform(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), 0, phi, 0, N, evec, 0, 0, psi, 0, ncomp);
410 
411  /* exit the loop if the eigen-vectors are converged or this is a last iteration */
412  if (n <= itso.min_num_res_ || k == (itso.num_steps_ - 1)) {
413  break;
414  } else { /* otherwise, set Psi as a new trial basis */
415  kp.message(3, __func__, "subspace size limit reached\n");
416 
417  if (itso.converge_by_energy_) {
418  transform(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), 0, ophi, 0, N, evec, 0, 0, opsi, 0, ncomp);
419  }
420 
421  ovlp_old.zero();
422  for (int i = 0; i < ncomp; i++) {
423  ovlp_old.set(i, i, eval[i]);
424  }
425  /* update basis functions */
426  phi.copy_from(ctx_.processing_unit(), ncomp, psi, 0, 0, 0, 0);
427  ophi.copy_from(ctx_.processing_unit(), ncomp, opsi, 0, 0, 0, 0);
428  /* number of basis functions that we already have */
429  N = ncomp;
430  }
431  }
432  /* expand variational subspace with new basis vectors obtatined from residuals */
433  phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, N);
434  }
435 
436  if (ctx_.processing_unit() == device_t::GPU) {
437  psi.pw_coeffs(0).copy_to(memory_t::host, 0, ncomp);
438  psi.pw_coeffs(0).deallocate(memory_t::device);
439  }
440 
441  kp.message(2, __func__, "smallest eigen-value of the singular components: %20.16f\n", eval[0]);
442 }

445 {
446  PROFILE("sirius::Band::diag_fv_davidson");
447 
448  auto& kp = Hk__.kp();
449 
450  auto h_o_diag = Hk__.get_h_o_diag_lapw<3>();
451 
452  get_singular_components(Hk__, h_o_diag.second);
453 
454  /* short notation for number of target wave-functions */
455  int num_bands = ctx_.num_fv_states();
456

458

480  ctx_.preferred_memory_t());

485  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi,
486  ctx_.preferred_memory_t());
487  Wave_functions hpsi(kp.gkvec_partition(), unit_cell_.num_atoms(),
488  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_bands,
489  ctx_.preferred_memory_t());
490  Wave_functions opsi(kp.gkvec_partition(), unit_cell_.num_atoms(),
491  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_bands,

643  }

648  phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, nlo + ncomp + num_bands);
649  /* number of basis functions that we already have */
650  N = nlo + ncomp;
651  n += num_bands;
652  }
653  } else {

655  phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, N);
656  }
657  }
658 
659  if (ctx_.processing_unit() == device_t::GPU) {

661  psi.mt_coeffs(0).copy_to(memory_t::host, 0, num_bands);
662  psi.deallocate(spin_range(0), memory_t::device);
663  }
