31 void Field4D::symmetrize(Periodic_function<double>* f__, Periodic_function<double>* gz__,
32  Periodic_function<double>* gx__, Periodic_function<double>* gy__)
33 {
34  PROFILE("sirius::Field4D::symmetrize");
35 
36  /* quick exit: the only symmetry operation is identity */
37  if (ctx_.unit_cell().symmetry().num_mag_sym() == 1) {
38  return;
39  }
40 
41  auto& comm = ctx_.comm();
42 
43  if (ctx_.cfg().control().print_hash()) {
44  auto h = f__->hash_f_pw();
45  if (ctx_.comm().rank() == 0) {
46  utils::print_hash("f_unsymmetrized(G)", h);
47  }
48  }
49 
50  /* symmetrize PW components */
51  switch (ctx_.num_mag_dims()) {
52  case 0: {
53  sirius::symmetrize(ctx_.unit_cell().symmetry(), ctx_.remap_gvec(), ctx_.sym_phase_factors(),
54  &f__->f_pw_local(0), nullptr, nullptr, nullptr);
55  if (ctx_.cfg().control().print_hash()) {
56  auto h = f__->hash_f_pw();
57  if (ctx_.comm().rank() == 0) {
58  utils::print_hash("f_symmetrized(G)", h);
59  }
60  }
61  break;
62  }
63  case 1: {
64  sirius::symmetrize(ctx_.unit_cell().symmetry(), ctx_.remap_gvec(), ctx_.sym_phase_factors(),
65  &f__->f_pw_local(0), nullptr, nullptr, &gz__->f_pw_local(0));
66  break;
67  }
68  case 3: {
69  if (ctx_.cfg().control().print_hash()) {
70  auto h1 = gx__->hash_f_pw();
71  auto h2 = gy__->hash_f_pw();
72  auto h3 = gz__->hash_f_pw();
73  if (ctx_.comm().rank() == 0) {
74  utils::print_hash("fx_unsymmetrized(G)", h1);
75  utils::print_hash("fy_unsymmetrized(G)", h2);
76  utils::print_hash("fz_unsymmetrized(G)", h3);
77  }
78  }
79 
80  sirius::symmetrize(ctx_.unit_cell().symmetry(), ctx_.remap_gvec(), ctx_.sym_phase_factors(),
81  &f__->f_pw_local(0), &gx__->f_pw_local(0), &gy__->f_pw_local(0), &gz__->f_pw_local(0));
82 
83  if (ctx_.cfg().control().print_hash()) {
84  auto h1 = gx__->hash_f_pw();
85  auto h2 = gy__->hash_f_pw();
86  auto h3 = gz__->hash_f_pw();
87  if (ctx_.comm().rank() == 0) {
88  utils::print_hash("fx_symmetrized(G)", h1);
89  utils::print_hash("fy_symmetrized(G)", h2);
90  utils::print_hash("fz_symmetrized(G)", h3);
91  }
92  }
93  break;
94  }
95  }
96 
97  if (ctx_.full_potential()) {
98  /* symmetrize MT components */
99  symmetrize_function(ctx_.unit_cell().symmetry(), comm, f__->f_mt());
100  switch (ctx_.num_mag_dims()) {
101  case 1: {
102  symmetrize_vector_function(ctx_.unit_cell().symmetry(), comm, gz__->f_mt());
103  break;
104  }
105  case 3: {
106  symmetrize_vector_function(ctx_.unit_cell().symmetry(), comm, gx__->f_mt(), gy__->f_mt(), gz__->f_mt());
107  break;
108  }
109  }
110  }
111 }
