32 void Field4D::symmetrize(Periodic_function<double>* f__, Periodic_function<double>* gz__,
33  Periodic_function<double>* gx__, Periodic_function<double>* gy__)
34 {
35  PROFILE("sirius::Field4D::symmetrize");
36 
37  /* quick exit: the only symmetry operation is identity */
38  if (ctx_.unit_cell().symmetry().num_mag_sym() == 1) {
39  return;
40  }
41 
42  auto& comm = ctx_.comm();
43 
44  auto& remap_gvec = ctx_.remap_gvec();
45 
46  if (ctx_.control().print_hash_) {
47  auto h = f__->hash_f_pw();
48  if (ctx_.comm().rank() == 0) {
49  utils::print_hash("f_unsymmetrized(G)", h);
50  }
51  }
52 
53  symmetrize_function(ctx_.unit_cell().symmetry(), remap_gvec, ctx_.sym_phase_factors(), &f__->f_pw_local(0));
54 
55  if (ctx_.control().print_hash_) {
56  auto h = f__->hash_f_pw();
57  if (ctx_.comm().rank() == 0) {
58  utils::print_hash("f_symmetrized(G)", h);
59  }
60  }
61 
62  /* symmetrize PW components */
63  switch (ctx_.num_mag_dims()) {
64  case 1: {
65  symmetrize_vector_function(ctx_.unit_cell().symmetry(), remap_gvec, ctx_.sym_phase_factors(),
66  &gz__->f_pw_local(0));
67  break;
68  }
69  case 3: {
70  if (ctx_.control().print_hash_) {
71  auto h1 = gx__->hash_f_pw();
72  auto h2 = gy__->hash_f_pw();
73  auto h3 = gz__->hash_f_pw();
74  if (ctx_.comm().rank() == 0) {
75  utils::print_hash("fx_unsymmetrized(G)", h1);
76  utils::print_hash("fy_unsymmetrized(G)", h2);
77  utils::print_hash("fz_unsymmetrized(G)", h3);
78  }
79  }
80 
81  symmetrize_vector_function(ctx_.unit_cell().symmetry(), remap_gvec, ctx_.sym_phase_factors(),
82  &gx__->f_pw_local(0), &gy__->f_pw_local(0), &gz__->f_pw_local(0));
83 
84  if (ctx_.control().print_hash_) {
85  auto h1 = gx__->hash_f_pw();
86  auto h2 = gy__->hash_f_pw();
87  auto h3 = gz__->hash_f_pw();
88  if (ctx_.comm().rank() == 0) {
89  utils::print_hash("fx_symmetrized(G)", h1);
90  utils::print_hash("fy_symmetrized(G)", h2);
91  utils::print_hash("fz_symmetrized(G)", h3);
92  }
93  }
94  break;
95  }
96  }
97 
98  if (ctx_.full_potential()) {
99  /* symmetrize MT components */
100  symmetrize_function(ctx_.unit_cell().symmetry(), comm, f__->f_mt());
101  switch (ctx_.num_mag_dims()) {
102  case 1: {
103  symmetrize_vector_function(ctx_.unit_cell().symmetry(), comm, gz__->f_mt());
104  break;
105  }
106  case 3: {
107  symmetrize_vector_function(ctx_.unit_cell().symmetry(), comm, gx__->f_mt(), gy__->f_mt(), gz__->f_mt());
108  break;
109  }
110  }
111  }
112 }

148 void sirius::Field4D::mixer_input()
149 {
150  /* split real-space points between available ranks */
151  splindex<splindex_t::block> spl_np(ctx_.spfft().local_slice_size(), ctx_.comm_ortho_fft().size(),
152  ctx_.comm_ortho_fft().rank());

154  int k{0};
155 
156  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {

158  for (int i = 0; i < static_cast<int>(component(j).f_mt(ialoc).size()); i++) {
159  mixer_->input_local(k++, component(j).f_mt(ialoc)[i]);

163  for (int i = 0; i < spl_np.local_size(); i++) {

169 void sirius::Field4D::mixer_output()
170 {
171  /* split real-space points between available ranks */
172  splindex<splindex_t::block> spl_np(ctx_.spfft().local_slice_size(), ctx_.comm_ortho_fft().size(),
173  ctx_.comm_ortho_fft().rank());
174 
175  int k{0};
176 
177  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
178  for (int ialoc = 0; ialoc < ctx_.unit_cell().spl_num_atoms().local_size(); ialoc++) {
179  auto& f_mt = const_cast<Spheric_function<function_domain_t::spectral, double>&>(component(j).f_mt(ialoc));
180  for (int i = 0; i < static_cast<int>(component(j).f_mt(ialoc).size()); i++) {
181  f_mt[i] = mixer_->output_local(k++);
182  }
183  }
184  //for (int i = 0; i < ctx_.fft().local_size(); i++) {
185  for (int i = 0; i < spl_np.local_size(); i++) {
186  component(j).f_rg(spl_np[i]) = mixer_->output_local(k++);
187  }
188  ctx_.comm_ortho_fft().allgather(&component(j).f_rg(0), spl_np.global_offset(), spl_np.local_size());
189  component(j).sync_mt();
190  }
191 }

193 void sirius::Field4D::mixer_init(Mixer_input mixer_cfg__)
194 {
195  int sz{0};
196  for (int ialoc = 0; ialoc < ctx_.unit_cell().spl_num_atoms().local_size(); ialoc++) {
197  sz += static_cast<int>(scalar().f_mt(ialoc).size());
198  }
199  sz += ctx_.spfft().local_slice_size();
200 
201  mixer_ = Mixer_factory<double>(0, (ctx_.num_mag_dims() + 1) * sz, mixer_cfg__, ctx_.comm());
202  mixer_input();
203  mixer_->initialize();
204 }
