41 /// Implementation of FFT3D.
42 /** FFT convention:
43  * \f[
44  * f({\bf r}) = \sum_{{\bf G}} e^{i{\bf G}{\bf r}} f({\bf G})
45  * \f]
46  * is a \em backward transformation from a set of pw coefficients to a function.
47  *
48  * \f[
49  * f({\bf G}) = \frac{1}{\Omega} \int e^{-i{\bf G}{\bf r}} f({\bf r}) d {\bf r} =
50  * \frac{1}{N} \sum_{{\bf r}_j} e^{-i{\bf G}{\bf r}_j} f({\bf r}_j)
51  * \f]
52  * is a \em forward transformation from a function to a set of coefficients.
53  *
54  * The following cases are handeled by the FFT driver:
55  * - transformation of a single real / complex function (serial / parallel, cpu / gpu)
56  * - transformation of two real functions (serial / parallel, cpu / gpu)
57  * - input / ouput data buffer pointer (cpu / gpu). GPU input pointer works only in serial.
58  *
59  * The transformation of two real functions is done as one transformation of complex function:
60  * \f[
61  * \Psi({\bf r}) = \psi_1({\bf r}) + i \psi_2({\bf r})
62  * \f]
63  * For each of the real wave functions the following condition is fulfilled:
64  * \f[
65  * \psi_{1,2}({\bf r}) = \psi_{1,2}^{*}({\bf r})
66  * \f]
67  * from which it follows that
68  * \f[
69  * \psi_{1,2}({\bf G}) = \psi_{1,2}^{*}(-{\bf G})
70  * \f]
71  * When z-direction is transformed
72  * \f[
73  * \psi_{1,2}(G_x, G_y, z) = \sum_{G_z} e^{izG_z} \psi_{1,2}(G_x, G_y, G_z)
74  * \f]
75  * it leads to the following symmetry
76  * \f[
77  * \psi_{1,2}^{*}(G_x, G_y, z) = \sum_{G_z} e^{-izG_z} \psi_{1,2}^{*}(G_x, G_y, G_z) =
78  * \sum_{G_z} e^{-izG_z} \psi_{1,2}(-G_x, -G_y, -G_z) = \psi_{1,2}(-G_x, -G_y, z)
79  * \f]
80  *
81  * \todo GPU input ponter for parallel FFT
82  * \todo decompose 3D fft into three consecutive 1D ffts
83  */

314  /* data is host memory */
315  if (is_host_memory(mem)) {
316  utils::timer t("sddk::FFT3D::transform_z_serial|cpu");
317  #pragma omp parallel for schedule(dynamic, 1)
318  for (int i = 0; i < num_zcol_local; i++) {
319  int tid = omp_get_thread_num();
320  /* global index of column */
321  int icol = gvec_partition_->idx_zcol<index_domain_t::local>(i);
322  /* offset of the PW coeffs in the input/output data buffer */
323  int data_offset = gvec_partition_->zcol_offs(icol);
324 
325  switch (direction) {
326  case 1: {
327  /* clear z buffer */
328  std::fill(fftw_buffer_z_[tid], fftw_buffer_z_[tid] + size(2), 0);
329  /* load z column of PW coefficients into buffer */
330  for (size_t j = 0; j < gvec_partition_->gvec().zcol(icol).z.size(); j++) {
331  int z = coord_by_freq<2>(gvec_partition_->gvec().zcol(icol).z[j]);
332  fftw_buffer_z_[tid][z] = data__[data_offset + j];

334 
335  /* column with {x,y} = {0,0} has only non-negative z components */
336  if (is_reduced && !icol) {

338  for (size_t j = 0; j < gvec_partition_->gvec().zcol(icol).z.size(); j++) {
339  int z = coord_by_freq<2>(-gvec_partition_->gvec().zcol(icol).z[j]);
340  fftw_buffer_z_[tid][z] = std::conj(data__[data_offset + j]);
341  }
342  }
343

345  fftw_execute(plan_backward_z_[tid]);
346 
347  /* redistribute z-column for a forthcoming all-to-all or just load the
348  * full column into auxiliary buffer in serial case */
349  for (int r = 0; r < comm_.size(); r++) {
350  int lsz = spl_z_.local_size(r);
351  int offs = spl_z_.global_offset(r);
352 
353  std::copy(&fftw_buffer_z_[tid][offs], &fftw_buffer_z_[tid][offs] + lsz,
354  &fft_buffer_aux__[offs * num_zcol_local + i * lsz]);
355  }
356  break;
357  }
358  case -1: {
359  /* collect full z-column or just load it from the auxiliary buffer is serial case */
360  for (int r = 0; r < comm_.size(); r++) {
361  int lsz = spl_z_.local_size(r);
362  int offs = spl_z_.global_offset(r);
363 
364  std::copy(&fft_buffer_aux__[offs * num_zcol_local + i * lsz],
365  &fft_buffer_aux__[offs * num_zcol_local + i * lsz] + lsz,
366  &fftw_buffer_z_[tid][offs]);
367  }
368 
369  /* perform local FFT transform of a column */
370  fftw_execute(plan_forward_z_[tid]);
371 
372  /* save z column of PW coefficients */
373  for (size_t j = 0; j < gvec_partition_->gvec().zcol(icol).z.size(); j++) {
374  int z = coord_by_freq<2>(gvec_partition_->gvec().zcol(icol).z[j]);
375  data__[data_offset + j] = fftw_buffer_z_[tid][z] * norm;
376  }
377 
378  break;
379  }
380  default: {
381  TERMINATE("wrong direction");
382  }
383  }
384  }
385  }

389  template <int direction, memory_t mem>
390  void transform_z(double_complex* data__, mdarray<double_complex, 1>& fft_buffer_aux__, cufftHandle cufft_plan_z__)
391  {
392  PROFILE("sddk::FFT3D::transform_z");
393 
394  int rank = comm_.rank();
395 
396  if (direction == -1) {
397  /* copy z-sticks to CPU; we need to copy to CPU in two cases:
398  - when data location is on host (in this case z-transfrom is done by CPU) but the processing
399  unit is GPU (xy-transform is done by GPU)
400  - or when we don't use GPU-direct but the transformation is parallel; in this case we are
401  going to do mpi_alltoall in host memory
402  */
403  if ((is_host_memory(mem) && pu_ == device_t::GPU) ||
404  (is_device_memory(mem) && !is_gpu_direct_ && comm_.size() > 1)) {
405  fft_buffer_aux__.copy_to(memory_t::host, 0, local_size_z_ * gvec_partition_->gvec().num_zcol());
406  }
407 
408  /* collect full sticks */
409  if (comm_.size() > 1) {
410  utils::timer t("sddk::FFT3D::transform_z|comm");
411 
412  block_data_descriptor send(comm_.size());
413  block_data_descriptor recv(comm_.size());
414  for (int r = 0; r < comm_.size(); r++) {
415  send.counts[r] = spl_z_.local_size(rank) * gvec_partition_->zcol_count_fft(r);
416  recv.counts[r] = spl_z_.local_size(r) * gvec_partition_->zcol_count_fft(rank);
417  }
418  send.calc_offsets();
419  recv.calc_offsets();
420 
421  if (is_host_memory(mem) || !is_gpu_direct_) {
422  /* copy auxiliary buffer because it will be use as the output buffer in the following mpi_a2a */
423  std::copy(&fft_buffer_aux__[0],
424  &fft_buffer_aux__[0] + gvec_partition_->gvec().num_zcol() * local_size_z_,
425  &fft_buffer_[0]);
426 
427  comm_.alltoall(&fft_buffer_[0], &send.counts[0], &send.offsets[0], &fft_buffer_aux__[0],
428  &recv.counts[0], &recv.offsets[0]);
429  }
430 #ifdef __GPU
431  if (is_device_memory(mem) && is_gpu_direct_) {
432  /* copy auxiliary buffer because it will be use as the output buffer in the following mpi_a2a */
433  acc::copy<double_complex>(fft_buffer_.at(memory_t::device), fft_buffer_aux__.at(memory_t::device),
434  gvec_partition_->gvec().num_zcol() * local_size_z_);
435 
436  comm_.alltoall(fft_buffer_.at(memory_t::device), &send.counts[0], &send.offsets[0], fft_buffer_aux__.at(memory_t::device),
437  &recv.counts[0], &recv.offsets[0]);
438  }
439 
440  /* buffer is on CPU after mpi_a2a and has to be copied to GPU */
441  if (is_device_memory(mem) && !is_gpu_direct_) {
442  fft_buffer_aux__.copy_to(memory_t::device, 0, gvec_partition_->zcol_count_fft() * size(2));
443  }
444 #endif
445  }
446  }
447 
448  transform_z_serial<direction, mem>(data__, fft_buffer_aux__, cufft_plan_z__);
449 
450  if (direction == 1) {
451  /* scatter z-columns between slabs of FFT buffer */
452  if (comm_.size() > 1) {
453  utils::timer t("sddk::FFT3D::transform_z|comm");
454 #ifdef __GPU
455  if (is_device_memory(mem) && !is_gpu_direct_) {
456  fft_buffer_aux__.copy_to(memory_t::host, 0, gvec_partition_->zcol_count_fft() * size(2));
457  }
458 #endif
459  block_data_descriptor send(comm_.size());
460  block_data_descriptor recv(comm_.size());
461  for (int r = 0; r < comm_.size(); r++) {
462  send.counts[r] = spl_z_.local_size(r) * gvec_partition_->zcol_count_fft(rank);
463  recv.counts[r] = spl_z_.local_size(rank) * gvec_partition_->zcol_count_fft(r);
464  }
465  send.calc_offsets();
466  recv.calc_offsets();
467 
468  if (is_host_memory(mem) || !is_gpu_direct_) {
469  /* scatter z-columns */
470  comm_.alltoall(&fft_buffer_aux__[0], &send.counts[0], &send.offsets[0], &fft_buffer_[0],
471  &recv.counts[0], &recv.offsets[0]);
472  /* copy local fractions of z-columns into auxiliary buffer */
473  std::copy(&fft_buffer_[0], &fft_buffer_[0] + gvec_partition_->gvec().num_zcol() * local_size_z_,
474  &fft_buffer_aux__[0]);
475  }
476 
477 #ifdef __GPU
478  if (is_device_memory(mem) && is_gpu_direct_) {
479  /* scatter z-columns */
480  comm_.alltoall(fft_buffer_aux__.at(memory_t::device), &send.counts[0], &send.offsets[0], fft_buffer_.at(memory_t::device),
481  &recv.counts[0], &recv.offsets[0]);
482  /* copy local fractions of z-columns into auxiliary buffer */
483  acc::copy<double_complex>(fft_buffer_aux__.at(memory_t::device), fft_buffer_.at(memory_t::device),
484  gvec_partition_->gvec().num_zcol() * local_size_z_);
485  }
486 #endif
487  }
488  /* copy back to device memory */
489  if ((is_host_memory(mem) && pu_ == device_t::GPU) ||
490  (is_device_memory(mem) && !is_gpu_direct_ && comm_.size() > 1)) {
491  fft_buffer_aux__.copy_to(memory_t::device, 0, local_size_z_ * gvec_partition_->gvec().num_zcol());
492  }
493  }

508  case device_t::GPU: {
509 #ifdef __GPU

513  /* srteam #0 unpacks z-columns into proper position of FFT buffer */
514  unpack_z_cols_gpu((cuDoubleComplex*)fft_buffer_aux__.at(memory_t::device),
515  (cuDoubleComplex*)fft_buffer_.at(memory_t::device), size(0), size(1), local_size_z_,
516  gvec_partition_->gvec().num_zcol(), z_col_pos_.at(memory_t::device), is_reduced,
517  cufft_stream_id);
518  /* stream #0 executes FFT */
519  cufft::backward_transform(cufft_plan_xy_, (cuDoubleComplex*)fft_buffer_.at(memory_t::device));
520  break;
521  }
522  case -1: {
523  /* stream #0 executes FFT */
524  cufft::forward_transform(cufft_plan_xy_, (cuDoubleComplex*)fft_buffer_.at(memory_t::device));
525  /* stream #0 packs z-columns */
526  pack_z_cols_gpu((cuDoubleComplex*)fft_buffer_aux__.at(memory_t::device),
527  (cuDoubleComplex*)fft_buffer_.at(memory_t::device), size(0), size(1), local_size_z_,
528  gvec_partition_->gvec().num_zcol(), z_col_pos_.at(memory_t::device), cufft_stream_id);
529  break;
530  }
531  }
532  acc::sync_stream(stream_id(cufft_stream_id));
533 #endif
534  break;
535  }
536  case device_t::CPU: {
537  #pragma omp parallel for schedule(static)
538  for (int iz = 0; iz < local_size_z_; iz++) {
539  int tid = omp_get_thread_num();
540  switch (direction) {
541  case 1: {
542  /* clear xy-buffer */
543  std::fill(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, 0);
544  /* load z-columns into proper location */
545  for (int i = 0; i < gvec_partition_->gvec().num_zcol(); i++) {
546  fftw_buffer_xy_[tid][z_col_pos_(i, 0)] = fft_buffer_aux__[iz + i * local_size_z_];
547 
548  if (is_reduced && i) {
549  fftw_buffer_xy_[tid][z_col_pos_(i, 1)] =
550  std::conj(fftw_buffer_xy_[tid][z_col_pos_(i, 0)]);
551  }
552  }
553 
554  /* execute local FFT transform */
555  fftw_execute(plan_backward_xy_[tid]);
556 
557  /* copy xy plane to the main FFT buffer */
558  std::copy(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, &fft_buffer_[iz * size_xy]);
559 
560  break;
561  }
562  case -1: {
563  /* copy xy plane from the main FFT buffer */
564  std::copy(&fft_buffer_[iz * size_xy], &fft_buffer_[iz * size_xy] + size_xy,
565  fftw_buffer_xy_[tid]);
566 
567  /* execute local FFT transform */
568  fftw_execute(plan_forward_xy_[tid]);
569 
570  /* get z-columns */
571  for (int i = 0; i < gvec_partition_->gvec().num_zcol(); i++) {
572  fft_buffer_aux__[iz + i * local_size_z_] = fftw_buffer_xy_[tid][z_col_pos_(i, 0)];
573  }
574 
575  break;
576  }
577  default: {
578  TERMINATE("wrong direction");
579  }
580  }
581  }
582  break;
583  }
584  }

587  /// Apply 2D FFT transformation to z-columns of two real functions.
588  /** The transformation is always done in the memory of processing unit. */
589  template <int direction>
590  void transform_xy(mdarray<double_complex, 1>& fft_buffer_aux1__, mdarray<double_complex, 1>& fft_buffer_aux2__)
591  {
592  PROFILE("sddk::FFT3D::transform_xy");
593 
594  if (!gvec_partition_->gvec().reduced()) {
595  TERMINATE("reduced set of G-vectors is required");
596  }
597 
598  int size_xy = size(0) * size(1);
599 
600 #ifdef __GPU
601  if (pu_ == GPU) {
602  /* stream #0 will be doing cuFFT */
603  switch (direction) {
604  case 1: {
605  /* srteam #0 unpacks z-columns into proper position of FFT buffer */
606  unpack_z_cols_2_gpu((cuDoubleComplex*)fft_buffer_aux1__.at(memory_t::device),
607  (cuDoubleComplex*)fft_buffer_aux2__.at(memory_t::device),
608  (cuDoubleComplex*)fft_buffer_.at(memory_t::device), size(0), size(1), local_size_z_,
609  gvec_partition_->gvec().num_zcol(), z_col_pos_.at(memory_t::device), cufft_stream_id);
610  /* stream #0 executes FFT */
611  cufft::backward_transform(cufft_plan_xy_, (cuDoubleComplex*)fft_buffer_.at(memory_t::device));
612  break;
613  }
614  case -1: {
615  /* stream #0 executes FFT */
616  cufft::forward_transform(cufft_plan_xy_, (cuDoubleComplex*)fft_buffer_.at(memory_t::device));
617  /* stream #0 packs z-columns */
618  pack_z_cols_2_gpu((cuDoubleComplex*)fft_buffer_aux1__.at(memory_t::device),
619  (cuDoubleComplex*)fft_buffer_aux2__.at(memory_t::device),
620  (cuDoubleComplex*)fft_buffer_.at(memory_t::device), size(0), size(1), local_size_z_,
621  gvec_partition_->gvec().num_zcol(), z_col_pos_.at(memory_t::device), cufft_stream_id);
622  break;
623  }
624  }
625  acc::sync_stream(stream_id(cufft_stream_id));
626  }
627 #endif
628 
629  if (pu_ == CPU) {
630  #pragma omp parallel
631  {
632  int tid = omp_get_thread_num();
633  #pragma omp for schedule(static)
634  for (int iz = 0; iz < local_size_z_; iz++) {
635  switch (direction) {
636  case 1: {
637  /* clear xy-buffer */
638  std::fill(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, 0);
639 
640  /* load first z-column into proper location */
641  fftw_buffer_xy_[tid][z_col_pos_(0, 0)] =
642  fft_buffer_aux1__[iz] + double_complex(0, 1) * fft_buffer_aux2__[iz];
643 
644  /* load remaining z-columns into proper location */
645  for (int i = 1; i < gvec_partition_->gvec().num_zcol(); i++) {
646  /* {x, y} part */
647  fftw_buffer_xy_[tid][z_col_pos_(i, 0)] =
648  fft_buffer_aux1__[iz + i * local_size_z_] +
649  double_complex(0, 1) * fft_buffer_aux2__[iz + i * local_size_z_];
650 
651  /* {-x, -y} part */
652  fftw_buffer_xy_[tid][z_col_pos_(i, 1)] =
653  std::conj(fft_buffer_aux1__[iz + i * local_size_z_]) +
654  double_complex(0, 1) * std::conj(fft_buffer_aux2__[iz + i * local_size_z_]);
655  }
656 
657  /* execute local FFT transform */
658  fftw_execute(plan_backward_xy_[tid]);
659 
660  /* copy xy plane to the main FFT buffer */
661  std::copy(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, &fft_buffer_[iz * size_xy]);
662 
663  break;
664  }
665  case -1: {
666  /* copy xy plane from the main FFT buffer */
667  std::copy(&fft_buffer_[iz * size_xy], &fft_buffer_[iz * size_xy] + size_xy,
668  fftw_buffer_xy_[tid]);
669 
670  /* execute local FFT transform */
671  fftw_execute(plan_forward_xy_[tid]);
672 
673  /* get z-columns */
674  for (int i = 0; i < gvec_partition_->gvec().num_zcol(); i++) {
675  fft_buffer_aux1__[iz + i * local_size_z_] =
676  0.5 * (fftw_buffer_xy_[tid][z_col_pos_(i, 0)] +
677  std::conj(fftw_buffer_xy_[tid][z_col_pos_(i, 1)]));
678 
679  fft_buffer_aux2__[iz + i * local_size_z_] =
680  double_complex(0, -0.5) * (fftw_buffer_xy_[tid][z_col_pos_(i, 0)] -
681  std::conj(fftw_buffer_xy_[tid][z_col_pos_(i, 1)]));
682  }
683 
684  break;
685  }
686  default: {
687  TERMINATE("wrong direction");
688  }
689  }
690  }
691  }
692  }

904  /// Prepare FFT driver to transfrom functions with the specific G-vector distribution.
905  /** Because the full 3D transformtation is decomposed into 1D and 2D transformtations we need to
906  * make a preparatory step in order to find locations of non-zero z-sticks in a given FFT buffer.
907  * The following steps are performed here:
908  * - address of G-vector partition object is saved in the internal class variable
909  * - positions of non-zero z-columns are stored in a buffer; this is actually a reason to make a preparatory
910  * step: non-zero columns are different for different G-vector sets
911  *
912  * In case of GPU the following additional steps are performed:
913  * - a mapping between G-vector index an a position in FFT buffer for 1D z-transforms is created
914  * - cuFFT plan for 1D transforms is re-created if the number of columns has increased since the previous
915  * call
916  * - work buffer is allocated on GPU and attached to z- and xy- cuFFT plans
917  * - main FFT buffer and two auxiliary buffers are allocated on GPU
918  */
919  void prepare(Gvec_partition const& gvp__)
920  {
921  PROFILE("sddk::FFT3D::prepare");
922 
923  if (gvec_partition_) {
924  TERMINATE("FFT3D is already prepared for another G-vector partition");
925  }
926 
927  gvec_partition_ = &gvp__;
928 
929  int nc = gvp__.gvec().reduced() ? 2 : 1;
930 
931  utils::timer t1("sddk::FFT3D::prepare|cpu");
932  /* get positions of z-columns in xy plane */
933  z_col_pos_ = mdarray<int, 2>(gvp__.gvec().num_zcol(), nc, memory_t::host, "FFT3D.z_col_pos_");
934  #pragma omp parallel for schedule(static)
935  for (int i = 0; i < gvp__.gvec().num_zcol(); i++) {
936  int icol = gvp__.idx_zcol<index_domain_t::global>(i);
937  int x = coord_by_freq<0>(gvp__.gvec().zcol(icol).x);
938  int y = coord_by_freq<1>(gvp__.gvec().zcol(icol).y);
939  assert(x >= 0 && x < size(0));
940  assert(y >= 0 && y < size(1));
941  z_col_pos_(i, 0) = x + y * size(0);
942  if (gvp__.gvec().reduced()) {
943  x = coord_by_freq<0>(-gvp__.gvec().zcol(icol).x);
944  y = coord_by_freq<1>(-gvp__.gvec().zcol(icol).y);
945  assert(x >= 0 && x < size(0));
946  assert(y >= 0 && y < size(1));
947  z_col_pos_(i, 1) = x + y * size(0);
948  }
949  }
950  t1.stop();
951 
952 #ifdef __GPU
953  switch (pu_) {
954  case device_t::GPU: {
955  utils::timer t2("sddk::FFT3D::prepare|gpu");
956  size_t work_size;
957  map_gvec_to_fft_buffer_ = mdarray<int, 1>(gvp__.gvec_count_fft(), memory_t::host,
958  "FFT3D.map_zcol_to_fft_buffer_");
959  map_gvec_to_fft_buffer_.allocate(memory_t::device);
960  /* loop over local set of columns */
961  #pragma omp parallel for schedule(static)
962  for (int i = 0; i < gvp__.zcol_count_fft(); i++) {
963  /* global index of z-column */
964  int icol = gvec_partition_->idx_zcol<index_domain_t::local>(i);
965  /* loop over z-colmn */
966  for (size_t j = 0; j < gvp__.gvec().zcol(icol).z.size(); j++) {
967  /* local index of the G-vector */
968  size_t ig = gvp__.zcol_offs(icol) + j;
969  /* coordinate inside FFT 1D bufer */
970  int z = coord_by_freq<2>(gvp__.gvec().zcol(icol).z[j]);
971  assert(z >= 0 && z < size(2));
972  map_gvec_to_fft_buffer_[ig] = i * size(2) + z;
973  }
974  }
975  map_gvec_to_fft_buffer_.copy_to(memory_t::device);
976 
977  /* for the rank that stores {x=0,y=0} column we need to create a small second mapping */
978  if (gvp__.gvec().reduced() && comm_.rank() == 0) {
979  map_gvec_to_fft_buffer_x0y0_ =
980  mdarray<int, 1>(gvp__.gvec().zcol(0).z.size(), memory_t::host,
981  "FFT3D.map_zcol_to_fft_buffer_x0y0_");
982  map_gvec_to_fft_buffer_x0y0_.allocate(memory_t::device);
983  for (size_t j = 0; j < gvp__.gvec().zcol(0).z.size(); j++) {
984  int z = coord_by_freq<2>(-gvp__.gvec().zcol(0).z[j]);
985  assert(z >= 0 && z < size(2));
986  map_gvec_to_fft_buffer_x0y0_[j] = z;
987  }
988  map_gvec_to_fft_buffer_x0y0_.copy_to(memory_t::device);
989  }
990 
991  int dim_z[] = {size(2)};
992  int dims_xy[] = {size(1), size(0)};
993 
994  int zcol_count_max{0};
995  if (gvp__.gvec().bare()) {
996  init_cufft_plan_z(gvp__, zcol_gvec_count_max_, cufft_plan_z_gvec_created_, cufft_plan_z_gvec_);
997  zcol_count_max = zcol_gvec_count_max_;
998  } else {
999  init_cufft_plan_z(gvp__, zcol_gkvec_count_max_, cufft_plan_z_gkvec_created_, cufft_plan_z_gkvec_);
1000  zcol_count_max = zcol_gkvec_count_max_;
1001  }
1002 
1003  /* maximum worksize of z and xy transforms */
1004  work_size = std::max(cufft::get_work_size(2, dims_xy, local_size_z_),
1005  cufft::get_work_size(1, dim_z, zcol_count_max));
1006 
1007  /* use as temp array also after z-transform*/
1008  work_size = std::max(work_size, sizeof(double_complex) * size(2) * local_size_z_);
1009 
1010  /* allocate cufft work buffer */
1011  cufft_work_buf_ = mdarray<char, 1>(work_size, memory_t::device, "FFT3D.cufft_work_buf_");
1012 
1013  /* set work area for cufft */
1014  cufft::set_work_area(cufft_plan_xy_, cufft_work_buf_.at(memory_t::device));
1015  if (gvp__.gvec().bare()) {
1016  cufft::set_work_area(cufft_plan_z_gvec_, cufft_work_buf_.at(memory_t::device));
1017  } else {
1018  cufft::set_work_area(cufft_plan_z_gkvec_, cufft_work_buf_.at(memory_t::device));
1019  }

1021  fft_buffer_aux1_.allocate(memory_t::device);
1022  fft_buffer_aux2_.allocate(memory_t::device);
1023 
1024  fft_buffer_.allocate(memory_t::device);
1025 
1026  z_col_pos_.allocate(memory_t::device).copy_to(memory_t::device);
1027  break;
1028  }
1029  case device_t::CPU: break;
1030  }
1031 #endif
1032  }

1144 /** \page ft_pw Fourier transform and plane wave normalization
1145  *
1146  * We use plane waves in two different cases: a) plane waves (or augmented plane waves in the case of APW+lo method)
1147  * as a basis for expanding Kohn-Sham wave functions and b) plane waves are used to expand charge density and
1148  * potential. When we are dealing with plane wave basis functions it is convenient to adopt the following
1149  * normalization:
1150  * \f[
1151  * \langle {\bf r} |{\bf G+k} \rangle = \frac{1}{\sqrt \Omega} e^{i{\bf (G+k)r}}
1152  * \f]
1153  * such that
1154  * \f[
1155  * \langle {\bf G+k} |{\bf G'+k} \rangle_{\Omega} = \delta_{{\bf GG'}}
1156  * \f]
1157  * in the unit cell. However, for the expansion of periodic functions such as density or potential, the following
1158  * convention is more appropriate:
1159  * \f[
1160  * \rho({\bf r}) = \sum_{\bf G} e^{i{\bf Gr}} \rho({\bf G})
1161  * \f]
1162  * where
1163  * \f[
1164  * \rho({\bf G}) = \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} \rho({\bf r}) d{\bf r} =
1165  * \frac{1}{\Omega} \sum_{{\bf r}_i} e^{-i{\bf Gr}_i} \rho({\bf r}_i) \frac{\Omega}{N} =
1166  * \frac{1}{N} \sum_{{\bf r}_i} e^{-i{\bf Gr}_i} \rho({\bf r}_i)
1167  * \f]
1168  * i.e. with such convention the plane-wave expansion coefficients are obtained with a normalized FFT.
1169  */
