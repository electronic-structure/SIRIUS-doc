34 __global__ void spherical_harmonics_ylm_gpu_kernel(int lmax__, int ntp__, double const* tp__,
35  acc_complex_double_t* ylm__, int ld__)
36 {
37  int itp = blockDim.x * blockIdx.x + threadIdx.x;
38 
39  if (itp < ntp__) {
40  double theta = tp__[2 * itp];
41  double phi = tp__[2 * itp + 1];
42  double sint = sin(theta);
43  double cost = cos(theta);
44 
45  acc_complex_double_t* ylm = &ylm__[array2D_offset(0, itp, ld__)];
46 
47  ylm[0].x = 1.0 / sqrt(2 * twopi);
48  ylm[0].y = 0;
49 
50  for (int l = 1; l <= lmax__; l++) {
51  ylm[lmidx(l, l)].x = -sqrt(1 + 1.0 / 2 / l) * sint * ylm[lmidx(l - 1, l - 1)].x;
52  ylm[lmidx(l, l)].y = 0;
53  }
54  for (int l = 0; l < lmax__; l++) {
55  ylm[lmidx(l + 1, l)].x = sqrt(2.0 * l + 3) * cost * ylm[lmidx(l, l)].x;
56  ylm[lmidx(l + 1, l)].y = 0;
57  }
58  for (int m = 0; m <= lmax__ - 2; m++) {
59  for (int l = m + 2; l <= lmax__; l++) {
60  double alm = sqrt(static_cast<double>((2 * l - 1) * (2 * l + 1)) / (l * l - m * m));
61  double blm = sqrt(static_cast<double>((l - 1 - m) * (l - 1 + m)) / ((2 * l - 3) * (2 * l - 1)));
62  ylm[lmidx(l, m)].x = alm * (cost * ylm[lmidx(l - 1, m)].x - blm * ylm[lmidx(l - 2, m)].x);
63  ylm[lmidx(l, m)].y = 0;
64  }
65  }

92 __global__ void spherical_harmonics_rlm_gpu_kernel(int lmax__, int ntp__, double const* tp__,
93  double* rlm__, int ld__)
94 {
95  int itp = blockDim.x * blockIdx.x + threadIdx.x;
96 
97  if (itp < ntp__) {
98  double theta = tp__[itp];
99  double phi = tp__[ntp__ + itp];
100  double sint = sin(theta);
101  double cost = cos(theta);
102 
103  double* rlm = &rlm__[array2D_offset(0, itp, ld__)];
104 
105  rlm[0] = 1.0 / sqrt(2 * twopi);
106 
107  for (int l = 1; l <= lmax__; l++) {
108  rlm[lmidx(l, l)] = -sqrt(1 + 1.0 / 2 / l) * sint * rlm[lmidx(l - 1, l - 1)];
109  }
110  for (int l = 0; l < lmax__; l++) {
111  rlm[lmidx(l + 1, l)] = sqrt(2.0 * l + 3) * cost * rlm[lmidx(l, l)];
112  }
113  for (int m = 0; m <= lmax__ - 2; m++) {
114  for (int l = m + 2; l <= lmax__; l++) {
115  double alm = sqrt(static_cast<double>((2 * l - 1) * (2 * l + 1)) / (l * l - m * m));
116  double blm = sqrt(static_cast<double>((l - 1 - m) * (l - 1 + m)) / ((2 * l - 3) * (2 * l - 1)));
117  rlm[lmidx(l, m)] = alm * (cost * rlm[lmidx(l - 1, m)] - blm * rlm[lmidx(l - 2, m)]);
118  }
119  }
120 
121  double c0 = cos(phi);
122  double c1 = 1;
123  double s0 = -sin(phi);
124  double s1 = 0;
125  double c2 = 2 * c0;
126 
127  double const t = sqrt(2.0);
128 
129  for (int m = 1; m <= lmax__; m++) {
130  double c = c2 * c1 - c0;
131  c0 = c1;
132  c1 = c;
133  double s = c2 * s1 - s0;
134  s0 = s1;
135  s1 = s;
136  for (int l = m; l <= lmax__; l++) {
137  double p = rlm[lmidx(l, m)];
138  rlm[lmidx(l, m)] = t * p * c;
139  if (m % 2) {
140  rlm[lmidx(l, -m)] = t * p * s;
141  } else {
142  rlm[lmidx(l, -m)] = -t * p * s;
143  }
144  }
145  }
146  }
147 }
