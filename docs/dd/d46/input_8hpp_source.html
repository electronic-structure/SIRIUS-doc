41 /// Parse unit cell input section.
42 /** The following part of the input file is parsed:
43  \code{.json}
44  "unit_cell" : {
45  "lattice_vectors" : [
46  [a1_x, a1_y, a1_z],
47  [a2_x, a2_y, a2_z],
48  [a3_x, a3_y, a3_z]
49  ],
50 
51  "lattice_vectors_scale" : (float) scale,
52 
53  "atom_types" : ["label_A", "label_B", ...],
54 
55  "atom_files" : {
56  "label_A" : "file_A",
57  "label_B" : "file_B",
58  ...
59  },
60 
61  "atom_coordinate_units" : units,
62 
63  "atoms" : {
64  "label_A": [
65  coordinates_A_1,
66  coordinates_A_2,
67  ...
68  ],
69  "label_B" : [
70  coordinates_B_1,
71  coordinates_B_2,
72  ...
73  ]
74  }
75  }
76  \endcode
77 
78  The "atom_coordinate_units" string is optional. By default it is assumed to be "lattice" which means that the
79  atomic coordinates are provided in lattice (fractional) units. It can also be specified in "A" or "au" which
80  means that the input atomic coordinates are Cartesian and provided in Angstroms or atomic units of length.
81  This is useful in setting up the molecule calculation.
82  */
83 struct Unit_cell_input
84 {
85  /// First vector of the unit cell.
86  vector3d<double> a0_;
87 
88  /// Second vector of the unit cell.
89  vector3d<double> a1_;
90 
91  /// Third vector of the unit cell.
92  vector3d<double> a2_;
93 
94  /// Labels of the atom types.
95  std::vector<std::string> labels_;
96 
97  /// Mapping between a label of atom type and corresponding atomic species file.
98  std::map<std::string, std::string> atom_files_;
99 
100  /// Atomic coordinates.
101  /** Outer vector size is equal to the number of atom types. */
102  std::vector<std::vector<std::vector<double>>> coordinates_;
103 
104  /// True if this section exists in the input file.
105  bool exist_{false};
106 
107  /// Read the \b unit_cell input section.
108  void read(json const& parser)
109  {
110  if (parser.count("unit_cell")) {
111  exist_ = true;
112 
113  auto section = parser["unit_cell"];
114 
115  auto a0 = section["lattice_vectors"][0].get<std::vector<double>>();
116  auto a1 = section["lattice_vectors"][1].get<std::vector<double>>();
117  auto a2 = section["lattice_vectors"][2].get<std::vector<double>>();
118 
119  if (a0.size() != 3 || a1.size() != 3 || a2.size() != 3) {
120  throw std::runtime_error("wrong lattice vectors");
121  }
122 
123  double scale = section.value("lattice_vectors_scale", 1.0);
124 
125  for (int x : {0, 1, 2}) {
126  a0_[x] = a0[x] * scale;
127  a1_[x] = a1[x] * scale;
128  a2_[x] = a2[x] * scale;
129  }
130 
131  matrix3d<double> lv;
132  for (int x : {0, 1, 2}) {
133  lv(x, 0) = a0_[x];
134  lv(x, 1) = a1_[x];
135  lv(x, 2) = a2_[x];
136  }
137  auto ilv = inverse(lv);
138 
139  labels_.clear();
140  coordinates_.clear();
141 
142  std::string units = section.value("atom_coordinate_units", "lattice");
143 
144  for (auto& label : section["atom_types"]) {
145  if (std::find(std::begin(labels_), std::end(labels_), label) != std::end(labels_)) {
146  throw std::runtime_error("duplicate atom type label");
147  }
148  labels_.push_back(label);
149  }
150 
151  if (section.count("atom_files")) {
152  for (auto& label : labels_) {
153  atom_files_[label] = section["atom_files"].value(label, "");
154  }
155  }
156 
157  for (int iat = 0; iat < (int)labels_.size(); iat++) {
158  coordinates_.push_back(std::vector<std::vector<double>>());
159  for (size_t ia = 0; ia < section["atoms"][labels_[iat]].size(); ia++) {
160  auto v = section["atoms"][labels_[iat]][ia].get<std::vector<double>>();
161 
162  if (!(v.size() == 3 || v.size() == 6)) {
163  throw std::runtime_error("wrong coordinates size");
164  }
165  if (v.size() == 3) {
166  v.resize(6, 0.0);
167  }
168 
169  vector3d<double> v1(v[0], v[1], v[2]);
170  if (units == "A") {
171  for (int x : {0, 1, 2}) {
172  v1[x] /= bohr_radius;
173  }
174  }
175  if (units == "au" || units == "A") {
176  v1 = ilv * v1;
177  auto rv1 = reduce_coordinates(v1);
178  for (int x : {0, 1, 2}) {
179  v[x] = rv1.first[x];
180  }
181  }
182 
183  coordinates_[iat].push_back(v);
184  }
185  }
186  }
187  }
188 };

244 /// Parse the parameters of iterative solver.

246 {
247  /// Type of the iterative solver.
248  std::string type_{""};
249 
250  /// Number of steps (iterations) of the solver.
251  int num_steps_{20};
252 
253  /// Size of the variational subspace is this number times the number of bands.
254  int subspace_size_{2};
255 
256  /// Lock eigenvectors of the smallest eigenvalues when they have converged at restart
257  bool locking_{true};
258 
259  /// Tolerance for the eigen-energy difference \f$ |\epsilon_i^{old} - \epsilon_i^{new} | \f$.
260  /** This parameter is reduced during the SCF cycle to reach the high accuracy of the wave-functions. */

262 
263  /// Tolerance for the residual L2 norm.
264  double residual_tolerance_{1e-6};
265 
266  /// Relative tolerance for the residual L2 norm. (0 means this criterion is effectively not used)
267  double relative_tolerance_{0};
268 
269  /// Additional tolerance for empty states.
270  /** Setting this variable to 0 will treat empty states with the same tolerance as occupied states. */
271  double empty_states_tolerance_{0};
272 
273  /// Defines the flavour of the iterative solver.
274  /** If converge_by_energy is set to 0, then the residuals are estimated by their norm. If converge_by_energy
275  is set to 1 then the residuals are estimated by the eigen-energy difference. This allows to estimate the
276  unconverged residuals and then compute only the unconverged ones. */
277  int converge_by_energy_{0}; // TODO: rename, this is meaningless
278 
279  /// Minimum number of residuals to continue iterative diagonalization process.
280  int min_num_res_{0};
281 
282  /// Number of singular components for the LAPW Davidson solver.
283  int num_singular_{-1};
284 
285  /// Initialize eigen-values with previous (old) values.
286  bool init_eval_old_{true};
287 
288  /// Tell how to initialize the subspace.
289  /** It can be either "lcao", i.e. start from the linear combination of atomic orbitals or "random" - start from
290  the randomized wave functions. */
291  std::string init_subspace_{"lcao"};
292

299  subspace_size_ = section.value("subspace_size", subspace_size_);
300  locking_ = section.value("locking", locking_);
301  energy_tolerance_ = section.value("energy_tolerance", energy_tolerance_);
302  residual_tolerance_ = section.value("residual_tolerance", residual_tolerance_);
303  relative_tolerance_ = section.value("relative_tolerance", relative_tolerance_);
304  empty_states_tolerance_ = section.value("empty_states_tolerance", empty_states_tolerance_);
305  converge_by_energy_ = section.value("converge_by_energy", converge_by_energy_);
306  min_num_res_ = section.value("min_num_res", min_num_res_);
307  num_singular_ = section.value("num_singular", num_singular_);
308  init_eval_old_ = section.value("init_eval_old", init_eval_old_);

311  }
312  }

315 /// Parse control input section.
316 /** The following part of the input file is parsed:
317  \code{.json}
318  "control" : {
319  "mpi_grid_dims" : (1- 2- or 3-dimensional vector<int>) MPI grid layout
320  "cyclic_block_size" : (int) PBLAS / ScaLAPACK block size
321  "reduce_gvec" : (bool) use reduced G-vector set (reduce_gvec = true) or full set (reduce_gvec = false)
322  "std_evp_solver_type" : (string) type of eigen-solver for the standard eigen-problem
323  "gen_evp_solver_type" : (string) type of eigen-solver for the generalized eigen-problem
324  "processing_unit" : (string) primary processing unit
325  "fft_mode" : (string) serial or parallel FFT
326  }
327  \endcode
328  Parameters of the control input sections do not in general change the numerics, but instead control how the
329  results are obtained. Changing parameters in control section should not change the significant digits in final
330  results.
331  */
332 struct Control_input
333 {
334  /// Dimensions of the MPI grid (if used).
335  std::vector<int> mpi_grid_dims_;
336 
337  /// Block size for ScaLAPACK and ELPA.
338  int cyclic_block_size_{-1};
339 
340  /// Reduce G-vectors by inversion symmetry.
341  /** For real-valued functions like density and potential it is sufficient to store only half of the G-vectors
342  * and use the relation f(G) = f^{*}(-G) to recover second half of the plane-wave expansion coefficients. */
343  bool reduce_gvec_{true};
344 
345  /// Standard eigen-value solver to use.
346  std::string std_evp_solver_name_{""};
347 
348  /// Generalized eigen-value solver to use.
349  std::string gen_evp_solver_name_{""};
350 
351  /// Coarse grid FFT mode ("serial" or "parallel").
352  std::string fft_mode_{"serial"};
353 
354  /// Main processing unit to run on.
355  std::string processing_unit_{""};
356 
357  /// Maximum allowed muffin-tin radius in case of LAPW.
358  double rmt_max_{2.2};
359 
360  /// Tolerance of the spglib in finding crystal symmetries.
361  double spglib_tolerance_{1e-4};
362 
363  /// Level of verbosity.
364  /** The following convention in proposed:
365  * - 0: silent mode (no output is printed) \n
366  * - 1: basic output (low level of output) \n
367  * - 2: extended output (medium level of output) \n
368  * - 3: extensive output (high level of output) */
369  int verbosity_{0};
370 
371  /// Level of internal verification.
372  int verification_{0};
373 
374  /// Number of eigen-values that are printed to the standard output.
375  int num_bands_to_print_{10};
376 
377  /// If true then performance of some compute-intensive kernels will be printed to the standard output.
378  bool print_performance_{false};
379 
380  /// If true then memory usage will be printed to the standard output.
381  bool print_memory_usage_{false};
382 
383  /// If true then the checksums of some arrays will be printed (useful during debug).
384  bool print_checksum_{false};
385 
386  /// If true then the hashsums of some arrays will be printed.
387  bool print_hash_{false};
388 
389  /// If true then the stress tensor components are printed at the end of SCF run.
390  bool print_stress_{false};
391 
392  /// If true then the atomic forces are printed at the end of SCF run.
393  bool print_forces_{false};
394 
395  /// If true then the timer statistics is printed at the end of SCF run.
396  bool print_timers_{true};
397 
398  /// If true then the list of nearest neighbours for each atom is printed to the standard output.
399  bool print_neighbors_{false};
400 
401  /// True if second-variational diagonalization is used in LAPW method.
402  bool use_second_variation_{true};
403 
404  /// Control the usage of the GPU memory.
405  /** Possible values are: "low", "medium" and "high". */
406  std::string memory_usage_{"high"};
407 
408  /// Number of atoms in the beta-projectors chunk.
409  int beta_chunk_size_{256};

577

579  std::transform(valence_relativity_.begin(), valence_relativity_.end(), valence_relativity_.begin(),
580  ::tolower);
581 
582  num_fv_states_ = section.value("num_fv_states", num_fv_states_);
583  smearing_width_ = section.value("smearing_width", smearing_width_);
584  pw_cutoff_ = section.value("pw_cutoff", pw_cutoff_);
585  aw_cutoff_ = section.value("aw_cutoff", aw_cutoff_);
586  gk_cutoff_ = section.value("gk_cutoff", gk_cutoff_);
587  lmax_apw_ = section.value("lmax_apw", lmax_apw_);
588  lmax_rho_ = section.value("lmax_rho", lmax_rho_);
589  lmax_pot_ = section.value("lmax_pot", lmax_pot_);
590  num_mag_dims_ = section.value("num_mag_dims", num_mag_dims_);
591  auto_rmt_ = section.value("auto_rmt", auto_rmt_);
592  use_symmetry_ = section.value("use_symmetry", use_symmetry_);
593  gamma_point_ = section.value("gamma_point", gamma_point_);
594  ngridk_ = section.value("ngridk", ngridk_);
595  shiftk_ = section.value("shiftk", shiftk_);
596  num_dft_iter_ = section.value("num_dft_iter", num_dft_iter_);
597  auto vk = section.value("vk", std::vector<std::vector<double>>{});
598  for (auto& vki : vk) {
599  if (vki.size() != 3) {
600  throw std::runtime_error("parameters.vk expected to be of size 3");
601  }

603  }
