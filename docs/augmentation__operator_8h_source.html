46  void generate_pw_coeffs(double omega__, Gvec const& gvec__, Radial_integrals_aug const& radial_integrals__)
47  {
48  PROFILE("sirius::Augmentation_operator::generate_pw_coeffs");
49 
50  double fourpi_omega = fourpi / omega__;
51 
52  /* maximum l of beta-projectors */
53  int lmax_beta = atom_type_.indexr().lmax();
54  int lmmax = Utils::lmmax(2 * lmax_beta);
55 
56  std::vector<int> l_by_lm = Utils::l_by_lm(2 * lmax_beta);
57 
58  std::vector<double_complex> zilm(lmmax);
59  for (int l = 0, lm = 0; l <= 2 * lmax_beta; l++) {
60  for (int m = -l; m <= l; m++, lm++) {
61  zilm[lm] = std::pow(double_complex(0, 1), l);
62  }
63  }
64 
65  /* Gaunt coefficients of three real spherical harmonics */
66  Gaunt_coefficients<double> gaunt_coefs(lmax_beta, 2 * lmax_beta, lmax_beta, SHT::gaunt_rlm);
67 
68  /* split G-vectors between ranks */
69  int gvec_count = gvec__.gvec_count(comm_.rank());
70  int gvec_offset = gvec__.gvec_offset(comm_.rank());
71 
72  /* array of real spherical harmonics for each G-vector */
73  mdarray<double, 2> gvec_rlm(Utils::lmmax(2 * lmax_beta), gvec_count);
74  for (int igloc = 0; igloc < gvec_count; igloc++) {
75  int ig = gvec_offset + igloc;
76  auto rtp = SHT::spherical_coordinates(gvec__.gvec_cart(ig));
77  SHT::spherical_harmonics(2 * lmax_beta, rtp[1], rtp[2], &gvec_rlm(0, igloc));
78  }
79 
80  /* number of beta-projectors */
81  int nbf = atom_type_.mt_basis_size();
82 
83  q_mtrx_ = mdarray<double, 2>(nbf, nbf);
84  q_mtrx_.zero();
85 
86  /* array of plane-wave coefficients */
87  q_pw_ = mdarray<double, 2>(nbf * (nbf + 1) / 2, 2 * gvec_count, memory_t::host_pinned, "q_pw_");
88  #pragma omp parallel for
89  for (int igloc = 0; igloc < gvec_count; igloc++) {
90  int ig = gvec_offset + igloc;
91  double g = gvec__.gvec_len(ig);
92 
93  std::vector<double_complex> v(lmmax);
94 
95  for (int xi2 = 0; xi2 < nbf; xi2++) {
96  int lm2 = atom_type_.indexb(xi2).lm;
97  int idxrf2 = atom_type_.indexb(xi2).idxrf;
98 
99  for (int xi1 = 0; xi1 <= xi2; xi1++) {
100  int lm1 = atom_type_.indexb(xi1).lm;
101  int idxrf1 = atom_type_.indexb(xi1).idxrf;
102 
103  /* packed orbital index */
104  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
105  /* packed radial-function index */
106  int idxrf12 = idxrf2 * (idxrf2 + 1) / 2 + idxrf1;
107 
108  for (int lm3 = 0; lm3 < lmmax; lm3++) {
109  v[lm3] = std::conj(zilm[lm3]) * gvec_rlm(lm3, igloc) * radial_integrals__.value(idxrf12, l_by_lm[lm3], atom_type_.id(), g);
110  }
111 
112  double_complex z = fourpi_omega * gaunt_coefs.sum_L3_gaunt(lm2, lm1, &v[0]);
113  q_pw_(idx12, 2 * igloc) = z.real();
114  q_pw_(idx12, 2 * igloc + 1) = z.imag();
115  }
116  }
117  }
118 
119  sym_weight_ = mdarray<double, 1>(nbf * (nbf + 1) / 2, memory_t::host_pinned, "sym_weight_");
120  for (int xi2 = 0; xi2 < nbf; xi2++) {
121  for (int xi1 = 0; xi1 <= xi2; xi1++) {
122  /* packed orbital index */
123  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
124  sym_weight_(idx12) = (xi1 == xi2) ? 1 : 2;
125  }
126  }
127 
128  if (comm_.rank() == 0) {
129  for (int xi2 = 0; xi2 < nbf; xi2++) {
130  for (int xi1 = 0; xi1 <= xi2; xi1++) {
131  /* packed orbital index */
132  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
133  q_mtrx_(xi1, xi2) = q_mtrx_(xi2, xi1) = omega__ * q_pw_(idx12, 0);
134  }
135  }
136  }
137  /* broadcast from rank#0 */
138  comm_.bcast(&q_mtrx_(0, 0), nbf * nbf , 0);
139 
140  #ifdef __PRINT_OBJECT_CHECKSUM
141  double cs = q_pw_.checksum();
142  comm_.allreduce(&cs, 1);
143  DUMP("checksum(q_pw) : %18.10f", cs);
144  #endif
145  }
