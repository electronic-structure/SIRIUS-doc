195  /// Find G+k vectors within the cutoff.
196  inline void generate_gkvec(double gk_cutoff__)
197  {
198  PROFILE("sirius::K_point::generate_gkvec");
199 
200  if (ctx_.full_potential() && (gk_cutoff__ * unit_cell_.max_mt_radius() > ctx_.lmax_apw())) {
201  std::stringstream s;
202  s << "G+k cutoff (" << gk_cutoff__ << ") is too large for a given lmax ("
203  << ctx_.lmax_apw() << ") and a maximum MT radius (" << unit_cell_.max_mt_radius() << ")" << std::endl
204  << "suggested minimum value for lmax : " << int(gk_cutoff__ * unit_cell_.max_mt_radius()) + 1;
205  WARNING(s);
206  }
207 
208  if (gk_cutoff__ * 2 > ctx_.pw_cutoff()) {
209  std::stringstream s;
210  s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
211  << " pw cutoff : " << ctx_.pw_cutoff() << std::endl
212  << " doubled G+k cutoff : " << gk_cutoff__ * 2;
213  TERMINATE(s);
214  }

295  /// Total number of muffin-tin and plane-wave expansion coefficients for the wave-functions.
296  /** APW+lo basis \f$ \varphi_{\mu {\bf k}}({\bf r}) = \{ \varphi_{\bf G+k}({\bf r}),
297  * \varphi_{j{\bf k}}({\bf r}) \} \f$ is used to expand first-variational wave-functions:
298  *
299  * \f[
300  * \psi_{i{\bf k}}({\bf r}) = \sum_{\mu} c_{\mu i}^{\bf k} \varphi_{\mu \bf k}({\bf r}) =
301  * \sum_{{\bf G}}c_{{\bf G} i}^{\bf k} \varphi_{\bf G+k}({\bf r}) +
302  * \sum_{j}c_{j i}^{\bf k}\varphi_{j{\bf k}}({\bf r})
303  * \f]
304  *
305  * Inside muffin-tins the expansion is converted into the following form:
306  * \f[
307  * \psi_{i {\bf k}}({\bf r})= \begin{array}{ll}

312  *

314  * to the number of muffin-tin basis functions of the form \f$ f_{\ell \lambda}^{\alpha}(r)

627 /** \page basis Basis functions for Kohn-Sham wave-functions expansion
628  *
629  * \section basis1 LAPW+lo basis
630  *
631  * LAPW+lo basis consists of two different sets of functions: LAPW functions \f$ \varphi_{{\bf G+k}} \f$ defined over
632  * entire unit cell:
633  * \f[
634  * \varphi_{{\bf G+k}}({\bf r}) = \left\{ \begin{array}{ll}
635  * \displaystyle \sum_{L} \sum_{\nu=1}^{O_{\ell}^{\alpha}} a_{L\nu}^{\alpha}({\bf G+k})u_{\ell \nu}^{\alpha}(r)
636  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in {\rm MT} \alpha \\
637  * \displaystyle \frac{1}{\sqrt \Omega} e^{i({\bf G+k}){\bf r}} & {\bf r} \in {\rm I} \end{array} \right.
638  * \f]
639  * and Bloch sums of local orbitals defined inside muffin-tin spheres only:
640  * \f[
641  * \begin{array}{ll} \displaystyle \varphi_{j{\bf k}}({\bf r})=\sum_{{\bf T}} e^{i{\bf kT}}
642  * \varphi_{j}({\bf r - T}) & {\rm {\bf r} \in MT} \end{array}
643  * \f]
644  * Each local orbital is composed of radial and angular parts:
645  * \f[
646  * \varphi_{j}({\bf r}) = \phi_{\ell_j}^{\zeta_j,\alpha_j}(r) Y_{\ell_j m_j}(\hat {\bf r})
647  * \f]
648  * Radial part of local orbital is defined as a linear combination of radial functions (minimum two radial functions
649  * are required) such that local orbital vanishes at the sphere boundary:
650  * \f[
651  * \phi_{\ell}^{\zeta, \alpha}(r) = \sum_{p}\gamma_{p}^{\zeta,\alpha} u_{\ell \nu_p}^{\alpha}(r)
652  * \f]
653  *
654  * Arbitrary number of local orbitals can be introduced for each angular quantum number (this is highlighted by
655  * the index \f$ \zeta \f$).
656  *
657  * Radial functions are m-th order (with zero-order being a function itself) energy derivatives of the radial
658  * Schrödinger equation:
659  * \f[
660  * u_{\ell \nu}^{\alpha}(r) = \frac{\partial^{m_{\nu}}}{\partial^{m_{\nu}}E}u_{\ell}^{\alpha}(r,E)\Big|_{E=E_{\nu}}
661  * \f]
662  */
663 
664 /** \page data_dist K-point data distribution
665  *
666  * \section data_dist1 "Panel" and "full" data storage
667  *
668  * We have to deal with big arrays (matching coefficients, eigen vectors, wave functions, etc.) which may not fit
669  * into the memory of a single node. For some operations we need a "panel" distribution of data, where each
