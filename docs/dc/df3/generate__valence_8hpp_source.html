25 inline void Density::generate_valence(K_point_set const& ks__)
26 {
27  PROFILE("sirius::Density::generate_valence");
28 
29  /* check weights */
30  double wt{0};
31  double occ_val{0};
32  for (int ik = 0; ik < ks__.num_kpoints(); ik++) {
33  wt += ks__[ik]->weight();
34  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
35  for (int j = 0; j < ctx_.num_bands(); j++) {
36  occ_val += ks__[ik]->weight() * ks__[ik]->band_occupancy(j, ispn);
37  }
38  }
39  }
40 
41  if (std::abs(wt - 1.0) > 1e-12) {
42  std::stringstream s;
43  s << "K_point weights don't sum to one" << std::endl
44  << " obtained sum: " << wt;
45  TERMINATE(s);
46  }
47 
48  if (std::abs(occ_val - unit_cell_.num_valence_electrons()) > 1e-8 && ctx_.comm().rank() == 0) {
49  std::stringstream s;
50  s << "wrong band occupancies" << std::endl
51  << " computed : " << occ_val << std::endl
52  << " required : " << unit_cell_.num_valence_electrons() << std::endl
53  << " difference : " << std::abs(occ_val - unit_cell_.num_valence_electrons());
54  WARNING(s);
55  }
56 
57  density_matrix_.zero();
58 
59  /* zero density and magnetization */
60  zero();
61  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
62  rho_mag_coarse_[i]->zero();
63  }
64 
65  /* start the main loop over k-points */
66  for (int ikloc = 0; ikloc < ks__.spl_num_kpoints().local_size(); ikloc++) {
67  int ik = ks__.spl_num_kpoints(ikloc);
68  auto kp = ks__[ik];
69 
70  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
71  int nbnd = kp->num_occupied_bands(ispn);
72  if (is_device_memory(ctx_.preferred_memory_t())) {
73  /* allocate GPU memory */
74  kp->spinor_wave_functions().pw_coeffs(ispn).prime().allocate(ctx_.mem_pool(memory_t::device));
75  kp->spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, nbnd); // TODO: copy this asynchronously
76  }
77  /* swap wave functions for the FFT transformation */
78  kp->spinor_wave_functions().pw_coeffs(ispn).remap_forward(nbnd, 0, &ctx_.mem_pool(memory_t::host));
79  kp->spinor_wave_functions().preferred_memory_t(ctx_.preferred_memory_t());
80  }
81 
82  if (ctx_.electronic_structure_method() == electronic_structure_method_t::full_potential_lapwlo) {
83  add_k_point_contribution_dm<double_complex>(kp, density_matrix_);
84  }
85 
86  if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {
87  if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
88  add_k_point_contribution_dm<double>(kp, density_matrix_);
89  } else {
90  add_k_point_contribution_dm<double_complex>(kp, density_matrix_);
91  }
92  }
93 
94  /* add contribution from regular space grid */
95  add_k_point_contribution_rg(kp);
96 
97  if (is_device_memory(ctx_.preferred_memory_t())) {
98  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
99  /* deallocate GPU memory */
100  kp->spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
101  }
102  kp->spinor_wave_functions().preferred_memory_t(memory_t::host);
103  }
104  }
105 
106  if (density_matrix_.size()) {
107  ctx_.comm().allreduce(density_matrix_.at(memory_t::host), static_cast<int>(density_matrix_.size()));
108  }
109 
110  ctx_.fft_coarse().prepare(ctx_.gvec_coarse_partition());
111  auto& comm = ctx_.gvec_coarse_partition().comm_ortho_fft();
112  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
113  /* reduce arrays; assume that each rank did its own fraction of the density */
114  /* comm_ortho_fft is idential to a product of column communicator inside k-point with k-point communicator */
115  comm.allreduce(&rho_mag_coarse_[j]->f_rg(0), ctx_.fft_coarse().local_size());
116  /* print checksum if needed */
117  if (ctx_.control().print_checksum_) {
118  auto cs = mdarray<double, 1>(&rho_mag_coarse_[j]->f_rg(0), ctx_.fft_coarse().local_size()).checksum();
119  ctx_.fft_coarse().comm().allreduce(&cs, 1);
120  if (ctx_.comm().rank() == 0) {
121  utils::print_checksum("rho_mag_coarse_rg", cs);
122  }
123  }
124  /* transform to PW domain */
125  rho_mag_coarse_[j]->fft_transform(-1);
126  /* map to fine G-vector grid */
127  for (int igloc = 0; igloc < ctx_.gvec_coarse().count(); igloc++) {
128  component(j).f_pw_local(ctx_.gvec().gvec_base_mapping(igloc)) = rho_mag_coarse_[j]->f_pw_local(igloc);
129  }
130  }
131  ctx_.fft_coarse().dismiss();
132 
133  if (!ctx_.full_potential()) {
134  augment();
135 
136  if (ctx_.control().print_hash_ && ctx_.comm().rank() == 0) {
137  auto h = mdarray<double_complex, 1>(&rho().f_pw_local(0), ctx_.gvec().count()).hash();
138  utils::print_hash("rho", h);
139  }
140 
141  double nel = rho().f_0().real() * unit_cell_.omega();
142  /* check the number of electrons */
143  if (std::abs(nel - unit_cell_.num_electrons()) > 1e-8 && ctx_.comm().rank() == 0) {
144  std::stringstream s;
145  s << "wrong unsymmetrized density" << std::endl
146  << " obtained value : " << std::scientific << nel << std::endl
147  << " target value : " << std::scientific << unit_cell_.num_electrons() << std::endl
148  << " difference : " << std::scientific << std::abs(nel - unit_cell_.num_electrons()) << std::endl;
149  WARNING(s);
150  }
151  }
152 
153  /* for muffin-tin part */
154  if (ctx_.full_potential()) {
155  generate_valence_mt();
156  }
157 }
