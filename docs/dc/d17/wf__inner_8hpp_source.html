31/// Inner product between wave-functions.
32/** This function computes the inner product using a moving window scheme plus allreduce.
33 * The input wave-functions data must be previously allocated on the GPU.
34 * The result is always returned in the CPU pointer.
35 *
36 * The following \f$ m \times n \f$ sub-matrix is computed:
37 * \f[
38 * S_{irow0+i,jcol0+j} = \langle \phi_{i0 + i} | \tilde \phi_{j0 + j} \rangle
39 * \f]
40 *
41 * \param [in] spla_ctx Spla context
42 * \param [in] ispn Index of spin (0, 1, or 2; 2 means the contribution from two spinor components).
43 * \param [in] bra "bra" wave-functions \f$ \phi \f$.
44 * \param [in] i0 Index of the first "bra" wave-function.
45 * \param [in] m Number of "bra" wave-functions.
46 * \param [in] ket "ket" wave-functions \f$ \tilde \phi \f$.
47 * \param [in] j0 Index of the first "ket" wave-function.
48 * \param [in] n Number of "ket" wave-functions.
49 * \param [out] result Resulting inner product matrix \f$ S \f$.
50 * \param [in] irow0 First row (in the global matrix) of the inner product sub-matrix.
51 * \param [in] jcol0 First column (in the global matix) of the inner product sub-matrix.
52 */
53template <typename T>
54void
55inner(::spla::Context& spla_ctx__, spin_range ispn__, Wave_functions<real_type<T>>& bra__, int i0__, int m__,
56 Wave_functions<real_type<T>>& ket__, int j0__, int n__, dmatrix<T>& result__, int irow0__, int jcol0__);

58inline void
59inner(::spla::Context& spla_ctx__, spin_range ispn__, Wave_functions<float>& bra__, int i0__, int m__,
60 Wave_functions<float>& ket__, int j0__, int n__, dmatrix<std::complex<double>>& result__, int irow0__, int jcol0__)
61{
62 for (int i = 0; i < m__; i++) {
63 for (int j = 0; j < n__; j++) {
64 result__(irow0__ + i, jcol0__ + j) = std::complex<double>(0, 0);
65 }
66 }
67
68 for (int s: ispn__) {
69 int nk = ket__.pw_coeffs(s).num_rows_loc();
70 for (int i = 0; i < m__; i++) {
71 for (int j = 0; j < n__; j++) {
72 std::complex<double> z(0, 0);
73 for (int k = 0; k < nk; k++) {
74 assert(bra__.pw_coeffs(s).prime(k, i0__ + i) == bra__.pw_coeffs(s).prime(k, i0__ + i));
75 assert(ket__.pw_coeffs(s).prime(k, j0__ + j) == ket__.pw_coeffs(s).prime(k, j0__ + j));
76 z += std::conj(bra__.pw_coeffs(s).prime(k, i0__ + i)) * ket__.pw_coeffs(s).prime(k, j0__ + j);
77 }
78 assert(z == z);
79 result__(irow0__ + i, jcol0__ + j) += z;
80 assert(result__(irow0__ + i, jcol0__ + j) == result__(irow0__ + i, jcol0__ + j));
81 }
82 }
83 }
84}

86inline void
87inner(::spla::Context& spla_ctx__, spin_range ispn__, Wave_functions<float>& bra__, int i0__, int m__,
88 Wave_functions<float>& ket__, int j0__, int n__, dmatrix<double>& result__, int irow0__, int jcol0__)
89{
90 for (int i = 0; i < m__; i++) {
91 for (int j = 0; j < n__; j++) {
92 result__(irow0__ + i, jcol0__ + j) = 0.0;
93 }
94 }
95
96 for (int s: ispn__) {
97 int nk = ket__.pw_coeffs(s).num_rows_loc();
98 for (int i = 0; i < m__; i++) {
99 for (int j = 0; j < n__; j++) {
100 double z{0};
101 z = std::real(std::conj(bra__.pw_coeffs(s).prime(0, i0__ + i)) * ket__.pw_coeffs(s).prime(0, j0__ + j));
102
103 for (int k = 1; k < nk; k++) {
104 auto a = bra__.pw_coeffs(s).prime(k, i0__ + i);
105 auto b = ket__.pw_coeffs(s).prime(k, j0__ + j);
106 z += 2 * (std::real(a) * std::real(b) + std::imag(a) * std::imag(b));
107 }
108 result__(irow0__ + i, jcol0__ + j) += z;
109 }
110 }
111 }
