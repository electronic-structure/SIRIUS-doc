50 double utils::timer::stop()
51 {
52 #if defined(__USE_TIMER)
53  if (!active_) {
54  return 0;
55  }
56 
57  /* remove this timer name from the list; now last element contains
58  the name of the parent timer */
59  stack().pop_back();
60 
61  /* measure the time difference */
62  auto t2 = std::chrono::high_resolution_clock::now();
63  auto tdiff = std::chrono::duration_cast<std::chrono::duration<double>>(t2 - starting_time_);
64  double val = tdiff.count();
65 
66  auto& ts = timer_values(label_);
67 #ifdef __TIMER_SEQUENCE
68  ts.sequence.push_back(starting_time_);
69  ts.sequence.push_back(t2);
70 #endif
71  ts.min_val = std::min(ts.min_val, val);
72  ts.max_val = std::max(ts.max_val, val);
73  ts.tot_val += val;
74  ts.count++;
75 
76  if (stack().size() != 0) {
77  /* last element contains the name of the parent timer */
78  auto parent_label = stack().back();
79  /* add value to the parent timer */
80  if (timer_values_ex().count(parent_label) == 0) {
81  timer_values_ex()[parent_label] = std::map<std::string, double>();
82  }
83  if (timer_values_ex()[parent_label].count(label_) == 0) {
84  timer_values_ex()[parent_label][label_] = 0;
85  }
86  timer_values_ex()[parent_label][label_] += val;
87  }
88  active_ = false;
89  return val;
90 #else
91  return 0;
92 #endif
93 }

127 void utils::timer::print_tree()
128 {

131  return;
132  }

139 
140  for (auto& it : timer_values()) {

142  /* collect external times */
143  double te{0};

145  te += it2.second;
146  }
147  double f = it.second.tot_val / ttot;
148  if (f > 0.01) {
149  printf("%s (%10.4fs, %.2f %% of self, %.2f %% of total)\n", it.first.c_str(), it.second.tot_val,
150  (it.second.tot_val - te) / it.second.tot_val * 100, f * 100);
151 
152  std::vector<std::pair<double, std::string>> tmp;
153 
154  for (auto& it2 : timer_values_ex()[it.first]) {
155  tmp.push_back(std::pair<double, std::string>(it2.second / it.second.tot_val, it2.first));
156  }
157  std::sort(tmp.rbegin(), tmp.rend());
158  for (auto& e : tmp) {
159  printf("|--%s (%10.4fs, %.2f %%) \n", e.second.c_str(), timer_values_ex()[it.first][e.second],
160  e.first * 100);

162  }
163  }
164  }
165 #endif

168 nlohmann::json utils::timer::serialize()

188 #endif

190  }

195 nlohmann::json utils::timer::serialize_tree()
196 {
197  nlohmann::json dict;
198 #if defined(__USE_TIMER)
199  if (!timer_values().count(main_timer_label)) {
200  return {};
201  }
202  /* total execution time */
203  double ttot = timer_values()[main_timer_label].tot_val;
204 
205  /* loop over the timer; iterator `it` is a <key, valu> pair */
206  for (auto& it : timer_values()) {
207  /* if this timer is a parent timer for somebody and timer has a non-negligible contribution */
208  if (timer_values_ex().count(it.first) && (it.second.tot_val / ttot) > 0.01) {
209  /* collect child (external) times */
210  double te{0};
211  for (auto& it2 : timer_values_ex()[it.first]) {
212  te += it2.second;
213  }
214  nlohmann::json node;
215 
216  node["total_time"] = it.second.tot_val;
217  node["child_time"] = te;
218  node["self_time"] = it.second.tot_val - te;
219  node["fraction_of_global_time"] = it.second.tot_val / ttot;
220  node["call"] = {};
221 
222  /* add all children values */
223  for (auto& it2 : timer_values_ex()[it.first]) {
224  nlohmann::json n;
225  n["time"] = timer_values_ex()[it.first][it2.first];
226  n["fraction_of_parent"] = it2.second / it.second.tot_val;
227  node["call"][it2.first] = n;
228  }
229  dict[it.first] = node;
230  }
231  }
232 #endif
233  return dict;
234 }
