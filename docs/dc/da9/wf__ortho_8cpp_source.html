35template <typename T, typename F>
36int
37orthogonalize(::spla::Context& spla_ctx__, memory_t mem__, linalg_t la__, spin_range spins__,
38 int idx_bra__, int idx_ket__, std::vector<Wave_functions<real_type<T>>*> wfs__, int N__,
39 int n__, dmatrix<F>& o__, Wave_functions<real_type<T>>& tmp__, bool project_out__)
40{
41 PROFILE("sddk::orthogonalize");
42
43 const char* sddk_pp_raw = std::getenv("SDDK_PRINT_PERFORMANCE");
44 int sddk_pp = (sddk_pp_raw == NULL) ? 0 : std::atoi(sddk_pp_raw);
45
46 auto& comm = wfs__[0]->comm();
47
48 int K{0};
49
50 if (sddk_pp) {
51 K = wfs__[0]->gkvec().num_gvec() + wfs__[0]->num_mt_coeffs();
52 if (std::is_same<T, real_type<T>>::value) {
53 K *= 2;
54 }
55 }
56
57 auto sddk_debug_ptr = utils::get_env<int>("SDDK_DEBUG");
58 int sddk_debug = (sddk_debug_ptr) ? (*sddk_debug_ptr) : 0;
59
60 /* prefactor for the matrix multiplication in complex or double arithmetic (in Giga-operations) */
61 double ngop{8e-9}; // default value for complex type
62 if (std::is_same<T, real_type<T>>::value) { // change it if it is real type
63 ngop = 2e-9;
64 }
65
66 if (sddk_pp) {
67 comm.barrier();
68 }
69 // double time = -omp_get_wtime();
70
71 double gflops{0};
72
73 /* project out the old subspace:
74 * |\tilda phi_new> = |phi_new> - |phi_old><phi_old|phi_new>
75 * H|\tilda phi_new> = H|phi_new> - H|phi_old><phi_old|phi_new>
76 * S|\tilda phi_new> = S|phi_new> - S|phi_old><phi_old|phi_new> */
77 if (N__ > 0 && project_out__) {
78 inner(spla_ctx__, spins__, *wfs__[idx_bra__], 0, N__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
79 transform<T, F>(spla_ctx__, spins__(), -1.0, wfs__, 0, N__, o__, 0, 0, 1.0, wfs__, N__, n__);
80
81 if (sddk_pp) {
82 /* inner and transform have the same number of flops */
83 gflops += static_cast<int>(1 + wfs__.size()) * ngop * N__ * n__ * K;
84 }
85 }
86
87 if (sddk_debug >= 2) {
88 if (o__.comm().rank() == 0) {
89 std::printf("check QR decomposition, matrix size : %i\n", n__);
90 }
91 inner(spla_ctx__, spins__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
92
93 linalg(linalg_t::scalapack).geqrf(n__, n__, o__, 0, 0);
94 auto diag = o__.get_diag(n__);
95 if (o__.comm().rank() == 0) {
96 for (int i = 0; i < n__; i++) {
97 if (std::abs(diag[i]) < std::numeric_limits<real_type<T>>::epsilon() * 10) {
98 std::cout << "small norm: " << i << " " << diag[i] << std::endl;
99 }
100 }
101 }
102
103 if (o__.comm().rank() == 0) {
104 std::printf("check eigen-values, matrix size : %i\n", n__);
105 }
106 inner(spla_ctx__, spins__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
107
108 // if (sddk_debug >= 3) {
109 // save_to_hdf5("nxn_overlap.h5", o__, n__);
110 //}
111
112 std::vector<real_type<F>> eo(n__);
113 dmatrix<F> evec(o__.num_rows(), o__.num_cols(), o__.blacs_grid(), o__.bs_row(), o__.bs_col());
114
115 auto solver = Eigensolver_factory("lapack", nullptr);
116 solver->solve(n__, o__, eo.data(), evec);
117
118 if (o__.comm().rank() == 0) {
119 for (int i = 0; i < n__; i++) {
120 if (eo[i] < 1e-6) {
121 std::cout << "small eigen-value " << i << " " << eo[i] << std::endl;
122 }
123 }
124 }
125 }
126
127 /* orthogonalize new n__ x n__ block */
128 inner(spla_ctx__, spins__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
129
130 if (sddk_debug >= 1) {
131 if (o__.comm().rank() == 0) {
132 std::printf("check diagonal\n");
133 }
134 auto diag = o__.get_diag(n__);
135 for (int i = 0; i < n__; i++) {
136 if (std::real(diag[i]) <= 0 || std::imag(diag[i]) > 1e-12) {
137 std::cout << "wrong diagonal: " << i << " " << diag[i] << std::endl;
138 }
139 }
140 if (o__.comm().rank() == 0) {
141 std::printf("check hermitian\n");
142 }
143 real_type<T> d = check_hermitian(o__, n__);
144 if (d > 1e-12 && o__.comm().rank() == 0) {
145 std::stringstream s;
146 s << "matrix is not hermitian, max diff = " << d;
147 WARNING(s);
148 }
149 }
150
151 if (sddk_pp) {
152 gflops += ngop * n__ * n__ * K;
153 }
154
155 /* single MPI rank */
156 if (o__.comm().size() == 1 && std::is_same<real_type<T>, real_type<F>>::value) {
157 bool use_magma{false};
158
159 // MAGMA performance for Cholesky and inversion is not good enough; use lapack for the moment
160 //#if defined(SIRIUS_GPU) && defined(SIRIUS_MAGMA)
161 // if (pu__ == GPU) {
162 // use_magma = true;
163 // }
164 //#endif
165
166 PROFILE_START("sddk::orthogonalize|tmtrx");
167 if (use_magma) {
168 /* Cholesky factorization */
169 if (int info = linalg(linalg_t::magma).potrf(n__, o__.at(memory_t::device), o__.ld())) {
170 std::stringstream s;
171 s << "error in GPU factorization, info = " << info;
172 TERMINATE(s);
173 }
174 /* inversion of triangular matrix */
175 if (linalg(linalg_t::magma).trtri(n__, o__.at(memory_t::device), o__.ld())) {
176 TERMINATE("error in inversion");
177 }
178 } else { /* CPU version */
179 /* Cholesky factorization */
180 if (int info = linalg(linalg_t::lapack).potrf(n__, &o__(0, 0), o__.ld())) {
181 std::stringstream s;
182 s << "error in factorization, info = " << info << std::endl
183 << "number of existing states: " << N__ << std::endl
184 << "number of new states: " << n__ << std::endl
185 << "number of wave_functions: " << wfs__.size() << std::endl
186 << "idx_bra: " << idx_bra__ << " "
187 << "idx_ket:" << idx_ket__;
188 TERMINATE(s);
189 }
190 /* inversion of triangular matrix */
191 if (linalg(linalg_t::lapack).trtri(n__, &o__(0, 0), o__.ld())) {
192 TERMINATE("error in inversion");
193 }
194 if (is_device_memory(mem__)) {
195 acc::copyin(o__.at(memory_t::device), o__.ld(), o__.at(memory_t::host), o__.ld(), n__, n__);
196 }
197 }
198 PROFILE_STOP("sddk::orthogonalize|tmtrx");
199
200 PROFILE_START("sddk::orthogonalize|transform");
201
202 int sid{0};
203 for (int s : spins__) {
204 /* multiplication by triangular matrix */
205 for (auto& e : wfs__) {
206 /* wave functions are complex, transformation matrix is complex */
207 if (!std::is_scalar<T>::value) {
208 linalg(la__).trmm('R', 'U', 'N', e->pw_coeffs(s).num_rows_loc(), n__, &linalg_const<T>::one(),
209 reinterpret_cast<T*>(o__.at(mem__)), o__.ld(),
210 reinterpret_cast<T*>(e->pw_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__)),
211 e->pw_coeffs(s).prime().ld(), stream_id(sid++));
212
213 if (e->has_mt()) {
214 linalg(la__).trmm(
215 'R', 'U', 'N', e->mt_coeffs(s).num_rows_loc(), n__, &linalg_const<T>::one(),
216 reinterpret_cast<T*>(o__.at(mem__)), o__.ld(),
217 reinterpret_cast<T*>(e->mt_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__)),
218 e->mt_coeffs(s).prime().ld(), stream_id(sid++));
219 }
220 }
221 /* wave functions are real (psi(G) = psi^{*}(-G)), transformation matrix is real */
222 if (std::is_scalar<T>::value) {
223 linalg(la__).trmm('R', 'U', 'N', 2 * e->pw_coeffs(s).num_rows_loc(), n__, &linalg_const<T>::one(),
224 reinterpret_cast<T*>(o__.at(mem__)), o__.ld(),
225 reinterpret_cast<T*>(e->pw_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__)),
226 2 * e->pw_coeffs(s).prime().ld(), stream_id(sid++));
227
228 if (e->has_mt()) {
229 linalg(la__).trmm(
230 'R', 'U', 'N', 2 * e->mt_coeffs(s).num_rows_loc(), n__, &linalg_const<T>::one(),
231 reinterpret_cast<T*>(o__.at(mem__)), o__.ld(),
232 reinterpret_cast<T*>(e->mt_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__)),
233 2 * e->mt_coeffs(s).prime().ld(), stream_id(sid++));
234 }
235 }
236 }
237 }
238 if (la__ == linalg_t::gpublas || la__ == linalg_t::cublasxt || la__ == linalg_t::magma) {
239 // sync stream only if processing unit is gpu
240 for (int i = 0; i < sid; i++) {
241 acc::sync_stream(stream_id(i));
242 }
243 }
244 PROFILE_STOP("sddk::orthogonalize|transform");
245 } else { /* parallel transformation */
246 PROFILE_START("sddk::orthogonalize|potrf");
247 mdarray<F, 1> diag;
248 o__.make_real_diag(n__);
249 if (sddk_debug >= 1) {
250 diag = o__.get_diag(n__);
251 }
252 if (int info = linalg(linalg_t::scalapack).potrf(n__, o__.at(memory_t::host), o__.ld(), o__.descriptor())) {
253 std::stringstream s;
254 s << "error in Cholesky factorization, info = " << info << ", matrix size = " << n__;
255 if (sddk_debug >= 1) {
256 s << std::endl << " diag = " << diag[info - 1];
257 }
258 TERMINATE(s);
259 }
260 PROFILE_STOP("sddk::orthogonalize|potrf");
261
262 PROFILE_START("sddk::orthogonalize|trtri");
263 if (linalg(linalg_t::scalapack).trtri(n__, o__.at(memory_t::host), o__.ld(), o__.descriptor())) {
264 TERMINATE("error in inversion");
265 }
266 PROFILE_STOP("sddk::orthogonalize|trtri");
267
268 /* o is upper triangular matrix */
269 for (int i = 0; i < n__; i++) {
270 for (int j = i + 1; j < n__; j++) {
271 o__.set(j, i, 0);
272 }
273 }
274
275 /* phi is transformed into phi, so we can't use it as the output buffer; use tmp instead and then overwrite phi
276 */
277 for (auto& e : wfs__) {
278 transform<T, F>(spla_ctx__, spins__(), *e, N__, n__, o__, 0, 0, tmp__, 0, n__);
279 for (int s : spins__) {
280 e->copy_from(tmp__, n__, s, 0, s, N__);
281 }
282 }
283 }
284 if (sddk_debug >= 2) {
285 inner(spla_ctx__, spins__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
286 auto err = check_identity(o__, n__);
287 std::cout << "wf_ortho: error in (n, n) overlap matrix : " << err << std::endl;
288 }
289 return 0;
290}
