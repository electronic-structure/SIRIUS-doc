34 template <typename T, int idx_bra__, int idx_ket__>
35 void orthogonalize(memory_t mem__, linalg_t la__, int ispn__, std::vector<Wave_functions*> wfs__, int N__, int n__,
36  dmatrix<T>& o__, Wave_functions& tmp__)
37 {
38  PROFILE("sddk::orthogonalize");
39 
40  const char* sddk_pp_raw = std::getenv("SDDK_PRINT_PERFORMANCE");
41  int sddk_pp = (sddk_pp_raw == NULL) ? 0 : std::atoi(sddk_pp_raw);
42 
43  auto& comm = wfs__[0]->comm();
44 
45  auto spins = (ispn__ == 2) ? std::vector<int>({0, 1}) : std::vector<int>({ispn__});
46 
47  int K{0};
48  if (sddk_pp) {
49  K = wfs__[0]->gkvec().num_gvec() + wfs__[0]->num_mt_coeffs();
50  if (std::is_same<T, double>::value) {
51  K *= 2;
52  }
53  }
54 
55  auto sddk_debug_ptr = utils::get_env<int>("SDDK_DEBUG");
56  int sddk_debug = (sddk_debug_ptr) ? (*sddk_debug_ptr) : 0;
57 
58  double ngop{0};
59  if (std::is_same<T, double>::value) {
60  ngop = 2e-9;
61  }
62  if (std::is_same<T, double_complex>::value) {
63  ngop = 8e-9;
64  }
65 
66  if (sddk_pp) {
67  comm.barrier();
68  }
69  // double time = -omp_get_wtime();
70 
71  double gflops{0};
72 
73  /* project out the old subspace:
74  * |\tilda phi_new> = |phi_new> - |phi_old><phi_old|phi_new> */
75  if (N__ > 0) {
76  inner(mem__, la__, ispn__, *wfs__[idx_bra__], 0, N__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
77  transform(mem__, la__, ispn__, -1.0, wfs__, 0, N__, o__, 0, 0, 1.0, wfs__, N__, n__);
78 
79  if (sddk_pp) {
80  gflops += static_cast<int>(1 + wfs__.size()) * ngop * N__ * n__ *
81  K; // inner and transform have the same number of flops
82  }
83  }
84 
85  if (sddk_debug >= 2) {
86  //if (o__.comm().rank() == 0) {
87  // std::printf("check QR decomposition, matrix size : %i\n", n__);
88  //}
89  //inner(mem__, la__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
90 
91  //linalg<device_t::CPU>::geqrf(n__, n__, o__, 0, 0);
92  //auto diag = o__.get_diag(n__);
93  //if (o__.comm().rank() == 0) {
94  // for (int i = 0; i < n__; i++) {
95  // if (std::abs(diag[i]) < 1e-6) {
96  // std::cout << "small norm: " << i << " " << diag[i] << std::endl;
97  // }
98  // }
99  //}
100 
101  if (o__.comm().rank() == 0) {
102  std::printf("check eigen-values, matrix size : %i\n", n__);
103  }
104  inner(mem__, la__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
105 
106  // if (sddk_debug >= 3) {
107  // save_to_hdf5("nxn_overlap.h5", o__, n__);
108  //}
109 
110  std::vector<double> eo(n__);
111  dmatrix<T> evec(o__.num_rows(), o__.num_cols(), o__.blacs_grid(), o__.bs_row(), o__.bs_col());
112 
113  auto solver = Eigensolver_factory("scalapack", nullptr);
114  solver->solve(n__, o__, eo.data(), evec);
115 
116  if (o__.comm().rank() == 0) {
117  for (int i = 0; i < n__; i++) {
118  if (eo[i] < 1e-6) {
119  std::cout << "small eigen-value " << i << " " << eo[i] << std::endl;
120  }
121  }
122  }
123  }
124 
125  /* orthogonalize new n__ x n__ block */
126  inner(mem__, la__, ispn__, *wfs__[idx_bra__], N__, n__, *wfs__[idx_ket__], N__, n__, o__, 0, 0);
127 
128  if (sddk_debug >= 1) {
129  if (o__.comm().rank() == 0) {
130  std::printf("check diagonal\n");
131  }
132  auto diag = o__.get_diag(n__);
133  for (int i = 0; i < n__; i++) {
134  if (std::real(diag[i]) <= 0 || std::imag(diag[i]) > 1e-12) {
135  std::cout << "wrong diagonal: " << i << " " << diag[i] << std::endl;
136  }
137  }
138  if (o__.comm().rank() == 0) {
139  std::printf("check hermitian\n");
140  }
141  double d = check_hermitian(o__, n__);
142  if (d > 1e-12 && o__.comm().rank() == 0) {
143  std::stringstream s;
144  s << "matrix is not hermitian, max diff = " << d;
145  WARNING(s);
146  }
147  }
148 
149  if (sddk_pp) {
150  gflops += ngop * n__ * n__ * K;
151  }
152 
153  /* single MPI rank */
154  if (o__.comm().size() == 1) {
155  bool use_magma{false};
156 
157  // MAGMA performance for Cholesky and inversion is not good enough; use lapack for the moment
158  //#if defined(__GPU) && defined(__MAGMA)
159  // if (pu__ == GPU) {
160  // use_magma = true;
161  // }
162  //#endif
163 
164  PROFILE_START("sddk::orthogonalize|tmtrx");
165  if (use_magma) {
166  /* Cholesky factorization */
167  if (int info = linalg(linalg_t::magma).potrf(n__, o__.at(memory_t::device), o__.ld())) {
168  std::stringstream s;
169  s << "error in GPU factorization, info = " << info;
170  TERMINATE(s);
171  }
172  /* inversion of triangular matrix */
173  if (linalg(linalg_t::magma).trtri(n__, o__.at(memory_t::device), o__.ld())) {
174  TERMINATE("error in inversion");
175  }
176  } else { /* CPU version */
177  /* Cholesky factorization */
178  if (int info = linalg(linalg_t::lapack).potrf(n__, &o__(0, 0), o__.ld())) {
179  std::stringstream s;
180  s << "error in factorization, info = " << info << std::endl
181  << "number of existing states: " << N__ << std::endl
182  << "number of new states: " << n__ << std::endl
183  << "number of wave_functions: " << wfs__.size() << std::endl
184  << "idx_bra: " << idx_bra__ << " "
185  << "idx_ket:" << idx_ket__;
186  TERMINATE(s);
187  }
188  /* inversion of triangular matrix */
189  if (linalg(linalg_t::lapack).trtri(n__, &o__(0, 0), o__.ld())) {
190  TERMINATE("error in inversion");
191  }
192  if (is_device_memory(mem__)) {
193  acc::copyin(o__.at(memory_t::device), o__.ld(), o__.at(memory_t::host), o__.ld(), n__, n__);
194  }
195  }
196  PROFILE_STOP("sddk::orthogonalize|tmtrx");
197 
198  PROFILE_START("sddk::orthogonalize|transform");
199 
200  int sid{0};
201  for (int s : spins) {
202  /* multiplication by triangular matrix */
203  for (auto& e : wfs__) {
204  /* wave functions are complex, transformation matrix is complex */
205  if (std::is_same<T, double_complex>::value) {
206  linalg(la__).trmm('R', 'U', 'N', e->pw_coeffs(s).num_rows_loc(), n__,
207  &linalg_const<double_complex>::one(),
208  reinterpret_cast<double_complex*>(o__.at(mem__)), o__.ld(),
209  e->pw_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__),
210  e->pw_coeffs(s).prime().ld(), stream_id(sid++));
211 
212  if (e->has_mt()) {
213  linalg(la__).trmm('R', 'U', 'N', e->mt_coeffs(s).num_rows_loc(), n__,
214  &linalg_const<double_complex>::one(),
215  reinterpret_cast<double_complex*>(o__.at(mem__)), o__.ld(),
216  e->mt_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__),
217  e->mt_coeffs(s).prime().ld(), stream_id(sid++));
218  }
219  }
220  /* wave functions are real (psi(G) = psi^{*}(-G)), transformation matrix is real */
221  if (std::is_same<T, double>::value) {
222  linalg(la__).trmm(
223  'R', 'U', 'N', 2 * e->pw_coeffs(s).num_rows_loc(), n__, &linalg_const<double>::one(),
224  reinterpret_cast<double*>(o__.at(mem__)), o__.ld(),
225  reinterpret_cast<double*>(e->pw_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__)),
226  2 * e->pw_coeffs(s).prime().ld(), stream_id(sid++));
227 
228  if (e->has_mt()) {
229  linalg(la__).trmm(
230  'R', 'U', 'N', 2 * e->mt_coeffs(s).num_rows_loc(), n__, &linalg_const<double>::one(),
231  reinterpret_cast<double*>(o__.at(mem__)), o__.ld(),
232  reinterpret_cast<double*>(e->mt_coeffs(s).prime().at(e->preferred_memory_t(), 0, N__)),
233  2 * e->mt_coeffs(s).prime().ld(), stream_id(sid++));
234  }
235  }
236  }
237  }
238  if (la__ == linalg_t::gpublas || la__ == linalg_t::cublasxt || la__ == linalg_t::magma) {
239  // sync stream only if processing unit is gpu
240  for (int i = 0; i < sid; i++) {
241  acc::sync_stream(stream_id(i));
242  }
243  }
244  PROFILE_STOP("sddk::orthogonalize|transform");
245  } else { /* parallel transformation */
246  PROFILE_START("sddk::orthogonalize|potrf");
247  mdarray<T, 1> diag;
248  o__.make_real_diag(n__);
249  if (sddk_debug >= 1) {
250  diag = o__.get_diag(n__);
251  }
252  if (int info = linalg(linalg_t::scalapack).potrf(n__, o__.at(memory_t::host), o__.ld(), o__.descriptor())) {
253  std::stringstream s;
254  s << "error in Cholesky factorization, info = " << info << ", matrix size = " << n__;
255  if (sddk_debug >= 1) {
256  s << std::endl << " diag = " << diag[info - 1];
257  }
258  TERMINATE(s);
259  }
260  PROFILE_STOP("sddk::orthogonalize|potrf");
261 
262  PROFILE_START("sddk::orthogonalize|trtri");
263  if (linalg(linalg_t::scalapack).trtri(n__, o__.at(memory_t::host), o__.ld(), o__.descriptor())) {
264  TERMINATE("error in inversion");
265  }
266  PROFILE_STOP("sddk::orthogonalize|trtri");
267 
268  /* o is upper triangular matrix */
269  for (int i = 0; i < n__; i++) {
270  for (int j = i + 1; j < n__; j++) {
271  o__.set(j, i, 0);
272  }
273  }
274 
275  /* phi is transformed into phi, so we can't use it as the output buffer; use tmp instead and then overwrite phi
276  */
277  for (auto& e : wfs__) {
278  transform(mem__, la__, ispn__, *e, N__, n__, o__, 0, 0, tmp__, 0, n__);
279  for (int s : spins) {
280  e->copy_from(tmp__, n__, s, 0, s, N__);
281  }
282  }
283  }
