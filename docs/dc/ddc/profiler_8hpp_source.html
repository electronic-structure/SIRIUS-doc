76  public:
77  /// Constructor.
78  profiler(char const* function_name__, char const* file__, int line__, char const* label__)
79  : label_(std::string(label__))
80  , function_name_(std::string(function_name__))
81  , file_(std::string(file__))
82  , line_(line__)
83  {
84 #if defined(__PROFILE_STACK) || defined(__PROFILE_FUNC)
85  char str[2048];
86  snprintf(str, 2048, "%s at %s:%i", function_name__, file__, line__);
87 #endif
88 
89 #if defined(__PROFILE_STACK)
90  call_stack().push_back(std::string(str));
91 #endif
92 
93 #if defined(__PROFILE_FUNC)
94  int tab{0};
95 #if defined(__PROFILE_STACK)
96  tab = static_cast<int>(call_stack().size()) - 1;
97 #endif
98  for (int i = 0; i < tab; i++) {
99  printf(" ");
100  }
101  int rank;
102  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
103  printf("[rank%04i] + %s\n", rank, label_.c_str());
104 #endif
105 
106 #if defined(__PROFILE_TIME)
107  timer_ = std::unique_ptr<utils::timer>(new utils::timer(label_));
108 #endif
109 
110 #if defined(__GPU) && defined(__CUDA_NVTX)
111  acc::begin_range_marker(label_.c_str());
112 #endif
113 #if defined(__APEX)
114  apex_p_ = apex::start(label_);
115 #endif
116  }
117 
118  ~profiler()
119  {
120 #ifdef __PROFILE_FUNC
121  int tab{0};
122 #ifdef __PROFILE_STACK
123  tab = static_cast<int>(call_stack().size()) - 1;
124 #endif
125  for (int i = 0; i < tab; i++) {
126  printf(" ");
127  }
128  int rank;
129  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
130  printf("[rank%04i] - %s\n", rank, label_.c_str());
131 #endif
132 
133 #ifdef __PROFILE_STACK
134  call_stack().pop_back();
135 #endif
136 
137 #if defined(__GPU) && defined(__CUDA_NVTX)
138  acc::end_range_marker();
139 #endif
140 #if defined(__APEX)
141  apex::stop(apex_p_);
142 #endif
143  }

155  }
156 #endif
157  }
