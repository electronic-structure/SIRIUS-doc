59 /// Compute r_{i} = H\Psi_{i} - E_{i}O\Psi_{i} */
60 static void compute_res(device_t pu__,
61  int ispn__,
62  int num_bands__,
63  mdarray<double, 1>& eval__,
64  Wave_functions& hpsi__,
65  Wave_functions& opsi__,
66  Wave_functions& res__)
67 {
68  auto spins = get_spins(ispn__);
69 
70  for (int ispn: spins) {
71  switch (pu__) {
72  case CPU: {
73  /* compute residuals r_{i} = H\Psi_{i} - E_{i}O\Psi_{i} */
74  #pragma omp parallel for
75  for (int i = 0; i < num_bands__; i++) {
76  for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
77  res__.pw_coeffs(ispn).prime(ig, i) = hpsi__.pw_coeffs(ispn).prime(ig, i) -
78  eval__[i] * opsi__.pw_coeffs(ispn).prime(ig, i);
79  }
80  if (res__.has_mt()) {
81  for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
82  res__.mt_coeffs(ispn).prime(j, i) = hpsi__.mt_coeffs(ispn).prime(j, i) -
83  eval__[i] * opsi__.mt_coeffs(ispn).prime(j, i);
84  }
85  }
86  }
87  break;
88  }
89  case GPU: {
90 #if defined(__GPU)
91  compute_residuals_gpu(hpsi__.pw_coeffs(ispn).prime().at(memory_t::device),
92  opsi__.pw_coeffs(ispn).prime().at(memory_t::device),
93  res__.pw_coeffs(ispn).prime().at(memory_t::device),
94  res__.pw_coeffs(ispn).num_rows_loc(),
95  num_bands__,
96  eval__.at(memory_t::device));
97  if (res__.has_mt()) {
98  compute_residuals_gpu(hpsi__.mt_coeffs(ispn).prime().at(memory_t::device),
99  opsi__.mt_coeffs(ispn).prime().at(memory_t::device),
100  res__.mt_coeffs(ispn).prime().at(memory_t::device),
101  res__.mt_coeffs(ispn).num_rows_loc(),
102  num_bands__,
103  eval__.at(memory_t::device));
104  }
105 #endif
106  }
107  }
108  }
109 }

111 /// Apply preconditioner to the residuals.
112 static void apply_p(device_t pu__,
113  int ispn__,
114  int num_bands__,
115  Wave_functions& res__,
116  mdarray<double, 2>& h_diag__,
117  mdarray<double, 1>& o_diag__,
118  mdarray<double, 1>& eval__)
119 {
120  int s0 = (ispn__ == 2) ? 0 : ispn__;
121  int s1 = (ispn__ == 2) ? 1 : ispn__;
122 
123  for (int ispn = s0; ispn <= s1; ispn++) {
124  switch (pu__) {
125  case CPU: {
126  #pragma omp parallel for schedule(static)
127  for (int i = 0; i < num_bands__; i++) {
128  for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
129  double p = h_diag__(ig, ispn) - o_diag__[ig] * eval__[i];
130  p = 0.5 * (1 + p + std::sqrt(1 + (p - 1) * (p - 1)));
131  res__.pw_coeffs(ispn).prime(ig, i) /= p;
132  }
133  if (res__.has_mt()) {
134  for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
135  double p = h_diag__(res__.pw_coeffs(ispn).num_rows_loc() + j, ispn) -
136  o_diag__[res__.pw_coeffs(ispn).num_rows_loc() + j] * eval__[i];
137  p = 0.5 * (1 + p + std::sqrt(1 + (p - 1) * (p - 1)));
138  res__.mt_coeffs(ispn).prime(j, i) /= p;
139  }
140  }
141  }
142  break;
143  }
144  case GPU: {
145 #if defined(__GPU)
146  apply_preconditioner_gpu(res__.pw_coeffs(ispn).prime().at(memory_t::device),
147  res__.pw_coeffs(ispn).num_rows_loc(),
148  num_bands__,
149  eval__.at(memory_t::device),
150  h_diag__.at(memory_t::device, 0, ispn),
151  o_diag__.at(memory_t::device));
152  if (res__.has_mt()) {
153  apply_preconditioner_gpu(res__.mt_coeffs(ispn).prime().at(memory_t::device),
154  res__.mt_coeffs(ispn).num_rows_loc(),
155  num_bands__,
156  eval__.at(memory_t::device),
157  h_diag__.at(memory_t::device, res__.pw_coeffs(ispn).num_rows_loc(), ispn),
158  o_diag__.at(memory_t::device, res__.pw_coeffs(ispn).num_rows_loc()));
159  }
160  break;
161 #endif
162  }
163  }
164  }
165 }

167 /// Normalize residuals.
168 /** This not strictly necessary as the wave-function orthoronormalization can take care of this.
169  * However, normalization of residuals is harmless and gives a better numerical stability. */
170 static void normalize_res(device_t pu__,
171  int ispn__,
172  int num_bands__,
173  Wave_functions& res__,
174  mdarray<double, 1>& p_norm__)
175 {
176  auto spins = get_spins(ispn__);
177 
178  for (int ispn: spins) {
179  switch (pu__) {
180  case CPU: {
181  #pragma omp parallel for schedule(static)
182  for (int i = 0; i < num_bands__; i++) {
183  for (int ig = 0; ig < res__.pw_coeffs(ispn).num_rows_loc(); ig++) {
184  res__.pw_coeffs(ispn).prime(ig, i) *= p_norm__[i];
185  }
186  if (res__.has_mt()) {
187  for (int j = 0; j < res__.mt_coeffs(ispn).num_rows_loc(); j++) {
188  res__.mt_coeffs(ispn).prime(j, i) *= p_norm__[i];
189  }
190  }
191  }
192  break;
193  }
194  case GPU: {
195  #ifdef __GPU
196  scale_matrix_columns_gpu(res__.pw_coeffs(ispn).num_rows_loc(), num_bands__,
197  (acc_complex_double_t*)res__.pw_coeffs(ispn).prime().at(memory_t::device),
198  p_norm__.at(memory_t::device));
199 
200  if (res__.has_mt()) {
201  scale_matrix_columns_gpu(res__.mt_coeffs(ispn).num_rows_loc(),
202  num_bands__,
203  (acc_complex_double_t *)res__.mt_coeffs(ispn).prime().at(memory_t::device),
204  p_norm__.at(memory_t::device));
205  }
206  #endif
207  break;
208  }
209  }
210  }

213 inline mdarray<double, 1>
214 Band::residuals_aux(K_point* kp__,
215  int ispn__,
216  int num_bands__,
217  std::vector<double>& eval__,
218  Wave_functions& hpsi__,
219  Wave_functions& opsi__,
220  Wave_functions& res__,
221  mdarray<double, 2>& h_diag__,
222  mdarray<double, 1>& o_diag__) const
223 {
224  PROFILE("sirius::Band::residuals_aux");
225 
226  assert(num_bands__ != 0);
227 
228  auto pu = get_device_t(ctx_.preferred_memory_t());
229 
230  mdarray<double, 1> eval(eval__.data(), num_bands__, "residuals_aux::eval");
231  if (pu == device_t::GPU) {
232  eval.allocate(memory_t::device).copy_to(memory_t::device);
233  }
234 
235  /* compute residuals */
236  compute_res(pu, ispn__, num_bands__, eval, hpsi__, opsi__, res__);
237 
238  /* compute norm */
239  auto res_norm = res__.l2norm(pu, ispn__, num_bands__);
240 
241  apply_p(pu, ispn__, num_bands__, res__, h_diag__, o_diag__, eval);
242 
243  auto p_norm = res__.l2norm(pu, ispn__, num_bands__);
244  for (int i = 0; i < num_bands__; i++) {
245  p_norm[i] = 1.0 / p_norm[i];
246  }
247  if (pu == device_t::GPU) {
248  p_norm.copy_to(memory_t::device);
249  }
250 
251  /* normalize preconditioned residuals */
252  normalize_res(pu, ispn__, num_bands__, res__, p_norm);
253 
254  if (ctx_.control().verbosity_ >= 5) {
255  auto n_norm = res__.l2norm(pu, ispn__, num_bands__);
256  if (kp__->comm().rank() == 0) {
257  for (int i = 0; i < num_bands__; i++) {
258  printf("norms of residual %3i: %18.14f %24.14f %18.14f", i, res_norm[i], p_norm[i], n_norm[i]);
259  if (res_norm[i] > ctx_.iterative_solver_input().residual_tolerance_) {
260  printf(" +");
261  }
262  printf("\n");
263  }
264  }
265  }
266 
267  return std::move(res_norm);
268 }

270 template <typename T>
271 inline int Band::residuals(K_point* kp__,
272  int ispn__,
273  int N__,
274  int num_bands__,
275  std::vector<double>& eval__,
276  std::vector<double>& eval_old__,
277  dmatrix<T>& evec__,
278  Wave_functions& hphi__,
279  Wave_functions& ophi__,
280  Wave_functions& hpsi__,
281  Wave_functions& opsi__,
282  Wave_functions& res__,
283  mdarray<double, 2>& h_diag__,
284  mdarray<double, 1>& o_diag__) const
285 {
286  PROFILE("sirius::Band::residuals");
287 
288  assert(N__ != 0);
289 
290  auto& itso = ctx_.iterative_solver_input();
291  bool converge_by_energy = (itso.converge_by_energy_ == 1);
292 
293  auto spins = get_spins(ispn__);
294 
295  int n{0};
296  if (converge_by_energy) {
297 
298  /* main trick here: first estimate energy difference, and only then compute unconverged residuals */
299  auto get_ev_idx = [&](double tol__)
300  {
301  std::vector<int> ev_idx;
302  int s = ispn__ == 2 ? 0 : ispn__;
303  for (int i = 0; i < num_bands__; i++) {
304  double o1 = std::abs(kp__->band_occupancy(i, s) / ctx_.max_occupancy());
305  double o2 = std::abs(1 - o1);
306 
307  double tol = o1 * tol__ + o2 * (tol__ + itso.empty_states_tolerance_);
308  if (std::abs(eval__[i] - eval_old__[i]) > tol) {
309  ev_idx.push_back(i);
310  }
311  }
312  return std::move(ev_idx);
313  };
314 
315  auto ev_idx = get_ev_idx(itso.energy_tolerance_);
316 
317  n = static_cast<int>(ev_idx.size());
318 
319  if (n) {
320  std::vector<double> eval_tmp(n);
321 
322  int bs = ctx_.cyclic_block_size();
323  dmatrix<T> evec_tmp(N__, n, ctx_.blacs_grid(), bs, bs);
324  int num_rows_local = evec_tmp.num_rows_local();
325  for (int j = 0; j < n; j++) {
326  eval_tmp[j] = eval__[ev_idx[j]];
327  if (ctx_.blacs_grid().comm().size() == 1) {
328  /* do a local copy */
329  std::copy(&evec__(0, ev_idx[j]), &evec__(0, ev_idx[j]) + num_rows_local, &evec_tmp(0, j));
330  } else {
331  auto pos_src = evec__.spl_col().location(ev_idx[j]);
332  auto pos_dest = evec_tmp.spl_col().location(j);
333  /* do MPI send / recieve */
334  if (pos_src.rank == kp__->comm_col().rank()) {
335  kp__->comm_col().isend(&evec__(0, pos_src.local_index), num_rows_local, pos_dest.rank, ev_idx[j]);
336  }
337  if (pos_dest.rank == kp__->comm_col().rank()) {
338  kp__->comm_col().recv(&evec_tmp(0, pos_dest.local_index), num_rows_local, pos_src.rank, ev_idx[j]);
339  }
340  }
341  }
342  if (ctx_.processing_unit() == device_t::GPU && evec__.blacs_grid().comm().size() == 1) {
343  evec_tmp.allocate(memory_t::device);
344  }
345  /* compute H\Psi_{i} = \sum_{mu} H\phi_{mu} * Z_{mu, i} and O\Psi_{i} = \sum_{mu} O\phi_{mu} * Z_{mu, i} */
346  transform<T>(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), ispn__, {&hphi__, &ophi__}, 0, N__,
347  evec_tmp, 0, 0, {&hpsi__, &opsi__}, 0, n);
348 
349  /* print checksums */
350  if (ctx_.control().print_checksum_ && n != 0) {
351  for (int ispn: spins) {
352  auto cs1 = hpsi__.checksum(get_device_t(hpsi__.preferred_memory_t()), ispn, 0, n);
353  auto cs2 = opsi__.checksum(get_device_t(opsi__.preferred_memory_t()), ispn, 0, n);
354  if (kp__->comm().rank() == 0) {
355  std::stringstream s;
356  s.str("");
357  s << "hpsi_" << ispn;
358  utils::print_checksum(s.str(), cs1);
359  s.str("");
360  s << "opsi_" << ispn;
361  utils::print_checksum(s.str(), cs2);
362  }
363  }
364  }
365 
366  auto res_norm = residuals_aux(kp__, ispn__, n, eval_tmp, hpsi__, opsi__, res__, h_diag__, o_diag__);
367 
368  int nmax = n;
369  n = 0;
370  for (int i = 0; i < nmax; i++) {
371  /* take the residual if it's norm is above the threshold */
372  if (res_norm[i] > itso.residual_tolerance_) {
373  /* shift unconverged residuals to the beginning of array */
374  if (n != i) {
375  for (int ispn: spins) {
376  res__.copy_from(res__, 1, ispn, i, ispn, n);
377  }
378  }
379  n++;
380  }
381  }
382  if (ctx_.control().verbosity_ >= 3 && kp__->comm().rank() == 0) {
383  printf("initial and final number of residuals : %i %i\n", nmax, n);
384  }
385  }
386  } else { /* compute all residuals first */
387  /* compute H\Psi_{i} = \sum_{mu} H\phi_{mu} * Z_{mu, i} and O\Psi_{i} = \sum_{mu} O\phi_{mu} * Z_{mu, i} */
388  transform<T>(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), ispn__, {&hphi__, &ophi__}, 0, N__,
389  evec__, 0, 0, {&hpsi__, &opsi__}, 0, num_bands__);
390 
391  auto res_norm = residuals_aux(kp__, ispn__, num_bands__, eval__, hpsi__, opsi__, res__, h_diag__, o_diag__);
392 
393  for (int i = 0; i < num_bands__; i++) {
394  double tol = itso.residual_tolerance_;// + 1e-3 * std::abs(kp__->band_occupancy(i + s * ctx_.num_fv_states()) / ctx_.max_occupancy() - 1);
395  /* take the residual if its norm is above the threshold */
396  if (res_norm[i] > tol) {
397  /* shift unconverged residuals to the beginning of array */
398  if (n != i) {
399  for (int ispn: spins) {
400  res__.copy_from(res__, 1, ispn, i, ispn, n);
401  }
402  }
403  n++;
404  }
405  }
406  if (ctx_.control().verbosity_ >= 3 && kp__->comm().rank() == 0) {
407  printf("number of residuals : %i\n", n);
408  }
409  }
410 
411  /* prevent numerical noise */
412  /* this only happens for real wave-functions (Gamma-point case), non-magnetic or collinear magnetic */
413  if (std::is_same<T, double>::value && kp__->comm().rank() == 0 && n != 0) {
414  assert(ispn__ == 0 || ispn__ == 1);
415  if (is_device_memory(res__.preferred_memory_t())) {
416 #if defined(__GPU)
417  make_real_g0_gpu(res__.pw_coeffs(ispn__).prime().at(memory_t::device), res__.pw_coeffs(ispn__).prime().ld(), n);
418 #endif
419  } else {
420  for (int i = 0; i < n; i++) {
421  res__.pw_coeffs(ispn__).prime(0, i) = res__.pw_coeffs(ispn__).prime(0, i).real();
422  }
423  }
424  }
425 
426  /* print checksums */
427  if (ctx_.control().print_checksum_ && n != 0) {
428  for (int ispn: spins) {
429  auto cs = res__.checksum(get_device_t(res__.preferred_memory_t()), ispn, 0, n);
430  if (kp__->comm().rank() == 0) {
431  std::stringstream s;
432  s << "res_" << ispn;
433  utils::print_checksum(s.str(), cs);
434  }
435  }
436  }
437 
438  return n;
439 }

441 template <typename T>
442 void Band::check_residuals(K_point& kp__, Hamiltonian& H__) const
443 {
444  if (kp__.comm().rank() == 0) {
445  printf("checking residuals\n");
446  }
447 
448  const bool nc_mag = (ctx_.num_mag_dims() == 3);
449  const int num_sc = nc_mag ? 2 : 1;
450 
451  auto& psi = kp__.spinor_wave_functions();
452  Wave_functions hpsi(kp__.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
453  Wave_functions spsi(kp__.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
454  Wave_functions res(kp__.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
455 
456  if (is_device_memory(ctx_.preferred_memory_t())) {
457  auto& mpd = ctx_.mem_pool(memory_t::device);
458  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
459  psi.pw_coeffs(ispn).allocate(mpd);
460  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
461  }
462  for (int i = 0; i < num_sc; i++) {
463  res.pw_coeffs(i).allocate(mpd);
464  hpsi.pw_coeffs(i).allocate(mpd);
465  spsi.pw_coeffs(i).allocate(mpd);
466  }
467  }
468  kp__.beta_projectors().prepare();
469  /* compute residuals */
470  for (int ispin_step = 0; ispin_step < ctx_.num_spin_dims(); ispin_step++) {
471  if (nc_mag) {
472  /* apply Hamiltonian and S operators to the wave-functions */
473  H__.apply_h_s<T>(&kp__, 2, 0, ctx_.num_bands(), psi, &hpsi, &spsi);
474  } else {
475  /* apply Hamiltonian and S operators to the wave-functions */
476  H__.apply_h_s<T>(&kp__, ispin_step, 0, ctx_.num_bands(), psi, &hpsi, &spsi);
477  }
478 
479  for (int ispn = 0; ispn < num_sc; ispn++) {
480  if (is_device_memory(ctx_.preferred_memory_t())) {
481  hpsi.copy_to(spin_idx(ispn), memory_t::host, 0, ctx_.num_bands());
482  spsi.copy_to(spin_idx(ispn), memory_t::host, 0, ctx_.num_bands());
483  }
484  #pragma omp parallel for schedule(static)
485  for (int j = 0; j < ctx_.num_bands(); j++) {
486  for (int ig = 0; ig < kp__.num_gkvec_loc(); ig++) {
487  res.pw_coeffs(ispn).prime(ig, j) = hpsi.pw_coeffs(ispn).prime(ig, j) -
488  spsi.pw_coeffs(ispn).prime(ig, j) *
489  kp__.band_energy(j, ispin_step);
490  }
491  }
492  }
493  /* get the norm */
494  auto l2norm = res.l2norm(device_t::CPU, nc_mag ? 2 : 0, ctx_.num_bands());
495 
496  if (kp__.comm().rank() == 0) {
497  for (int j = 0; j < ctx_.num_bands(); j++) {
498  printf("band: %3i, residual l2norm: %18.12f\n", j, l2norm[j]);
499  }
500  }
501  }
502  kp__.beta_projectors().dismiss();
503  if (is_device_memory(ctx_.preferred_memory_t())) {
504  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
505  psi.pw_coeffs(ispn).deallocate(memory_t::device);
506  }
507  }
