37/// Symmetrize scalar or vector function.
38/** The following operation is performed:
39 \f[
40 f_{\mathrm{sym}}({\bf r}) = \frac{1}{N_{\mathrm{sym}}}
41 \sum_{\hat{\bf S}\hat{\bf P}} \hat {\bf S} \hat {\bf P}f({\bf r})
42 \f]
43 where \f$ f({\bf r}) \f$ has to be understood as an unsymmetrized scalar or vector function.
44 In the case of a scalar function \f$ \hat {\bf S} = 1 \f$. In the case of vector function
45 \f$ \hat {\bf S} \f$ is rotation matrix acting on the Cartesian components of the function.
46 \f$ \hat {\bf P} = \{{\bf R}|{\bf t}\} \f$ is a spacial part of the full magentic symmetry operatoin acting
47 on the real-space coordinates.
48
49 For the function expanded in plane-waves we have:
50 \f[
51 f_{\mathrm{sym}}({\bf r}) = \frac{1}{N_{\mathrm{sym}}}
52 \sum_{\hat{\bf S}\hat{\bf P}} \hat {\bf S} \hat {\bf P}f({\bf r}) =
53 \frac{1}{N_{\mathrm{sym}}} \sum_{\hat{\bf S}\hat{\bf P}} \hat {\bf S} \sum_{\bf G}
54 f({\bf G}) e^{i{\bf G}\hat{\bf P}^{-1}{\bf r}} = \\
55 \frac{1}{N_{\mathrm{sym}}} \sum_{\hat{\bf S}\hat{\bf P}} \sum_{\bf G} \hat {\bf S} f({\bf G})
56 e^{i{\bf G}({\bf R}^{-1}{\bf r} - {\bf R}^{-1}{\bf t})} =
57 \frac{1}{N_{\mathrm{sym}}} \sum_{\hat{\bf S}\hat{\bf P}} \sum_{\bf G} \hat {\bf S} f({\bf G})
58 e^{i{\bf G}'{\bf r}} e^{-i{\bf G}'{\bf t}}
59 \f]
60 where \f$ {\bf G}' = {\bf G}{\bf R}^{-1} = {\bf R}^{-T}{\bf G} \f$. The last expression establishes the link
61 between unsymmetrized plane-wave coefficient at <b>G</b>-vector and symmetrized coefficient at <b>G</b>'. We will
62 rewrite the expression using inverse relation \f$ {\bf G} = {\bf R}^{T}{\bf G'} \f$ and summing over <b>G</b>'
63 (which is just a permutaion of <b>G</b>):
64 \f[
65 f_{\mathrm{sym}}({\bf r}) =
66 \sum_{\bf G'} e^{i{\bf G}'{\bf r}} \frac{1}{N_{\mathrm{sym}}} \sum_{\hat{\bf S}\hat{\bf P}}
67 \hat {\bf S} f({\bf R}^{T}{\bf G'}) e^{-i{\bf G}'{\bf t}}
68 \f]
69 That gives an expression for the symmetrized plane-wave coefficient at <b>G</b>':
70 \f[
71 f_{\mathrm{sym}}({\bf G}') = \frac{1}{N_{\mathrm{sym}}} \sum_{\hat{\bf S}\hat{\bf P}}
72 \hat {\bf S} f({\bf R}^{T}{\bf G'}) e^{-i{\bf G}'{\bf t}}
73 \f]
74
75 Once \f$ f_{\mathrm{sym}}({\bf G}) \f$ has been calculated for a single <b>G</b>, its values at a
76 star of <b>G</b> can be calculated using the following relation:
77 \f[
78 f_{\mathrm{sym}}({\bf r}) = \hat{\bf S}\hat{\bf P} f_{\mathrm{sym}}({\bf r}) =
79 \hat{\bf S} f_{\mathrm{sym}}(\hat {\bf P}^{-1}{\bf r})
80 \f]
81 which leads to the following relation for the plane-wave coefficient:
82 \f[
83 \sum_{\bf G} f_{\mathrm{sym}}({\bf G})e^{i{\bf G}{\bf r}} =
84 \sum_{\bf G} \hat{\bf S}f_{\mathrm{sym}}({\bf G})e^{i{\bf G}\hat{\bf P}^{-1}{\bf r}} =
85 \sum_{\bf G} \hat{\bf S}f_{\mathrm{sym}}({\bf G})e^{i{\bf G}{\bf R}^{-1}{\bf r}}
86 e^{-i{\bf G}{\bf R}^{-1}{\bf t}} =
87 \sum_{\bf G'} \hat{\bf S}f_{\mathrm{sym}}({\bf G})e^{i{\bf G}'{\bf r}}
88 e^{-i{\bf G}'{\bf t}} =
89 \sum_{\bf G'} \hat{\bf S}f_{\mathrm{sym}}({\bf G'})e^{i{\bf G}'{\bf r}}
90 \f]
91 and so
92 \f[
93 f_{\mathrm{sym}}({\bf G}') = \hat{\bf S}f_{\mathrm{sym}}({\bf G})e^{-i{\bf G'}{\bf t}}
94 \f]
95 */
96inline void symmetrize(Crystal_symmetry const& sym__, Gvec_shells const& gvec_shells__,
97 sddk::mdarray<double_complex, 3> const& sym_phase_factors__, double_complex* f_pw__,
98 double_complex* x_pw__, double_complex* y_pw__, double_complex* z_pw__)
99{
100 PROFILE("sirius::symmetrize|fpw");
101
102 auto f_pw = f_pw__ ? gvec_shells__.remap_forward(f_pw__) : std::vector<double_complex>();
103 auto x_pw = x_pw__ ? gvec_shells__.remap_forward(x_pw__) : std::vector<double_complex>();
104 auto y_pw = y_pw__ ? gvec_shells__.remap_forward(y_pw__) : std::vector<double_complex>();
105 auto z_pw = z_pw__ ? gvec_shells__.remap_forward(z_pw__) : std::vector<double_complex>();
106
107 /* local number of G-vectors in a distribution with complete G-vector shells */
108 int ngv = gvec_shells__.gvec_count_remapped();
109
110 auto sym_f_pw = f_pw__ ? std::vector<double_complex>(ngv, 0) : std::vector<double_complex>();
111 auto sym_x_pw = x_pw__ ? std::vector<double_complex>(ngv, 0) : std::vector<double_complex>();
112 auto sym_y_pw = y_pw__ ? std::vector<double_complex>(ngv, 0) : std::vector<double_complex>();
113 auto sym_z_pw = z_pw__ ? std::vector<double_complex>(ngv, 0) : std::vector<double_complex>();
114
115 bool is_non_collin = ((x_pw__ != nullptr) && (y_pw__ != nullptr) && (z_pw__ != nullptr));
116
117 std::vector<bool> is_done(ngv, false);
118
119 double norm = 1 / double(sym__.size());
120
121 auto phase_factor = [&](int isym, vector3d<int> G)
122 {
123 return sym_phase_factors__(0, G[0], isym) *
124 sym_phase_factors__(1, G[1], isym) *
125 sym_phase_factors__(2, G[2], isym);
126 };
127
128 double const eps{1e-9};
129
130 PROFILE_START("sirius::symmetrize|fpw|local");
131
132 #pragma omp parallel
133 {
134 int nt = omp_get_max_threads();
135 int tid = omp_get_thread_num();
136
137 for (int igloc = 0; igloc < gvec_shells__.gvec_count_remapped(); igloc++) {
138 auto G = gvec_shells__.gvec_remapped(igloc);
139
140 int igsh = gvec_shells__.gvec_shell_remapped(igloc);
141
142#if !defined(NDEBUG)
143 if (igsh != gvec_shells__.gvec().shell(G)) {
144 throw std::runtime_error("wrong index of G-shell");
145 }
146#endif
147 /* each thread is working on full shell of G-vectors */
148 if (igsh % nt == tid && !is_done[igloc]) {
149
150 double_complex symf(0, 0);
151 double_complex symx(0, 0);
152 double_complex symy(0, 0);
153 double_complex symz(0, 0);
154
155 /* find the symmetrized PW coefficient */
156
157 for (int i = 0; i < sym__.size(); i++) {
158 auto G1 = dot(G, sym__[i].spg_op.R);
159
160 auto S = sym__[i].spin_rotation;
161
162 auto phase = std::conj(phase_factor(i, G));
163
164 /* local index of a rotated G-vector */
165 int ig1 = gvec_shells__.index_by_gvec(G1);
166
167 if (ig1 == -1) {
168 G1 = G1 * (-1);
169#if !defined(NDEBUG)
170 if (igsh != gvec_shells__.gvec().shell(G1)) {
171 throw std::runtime_error("wrong index of G-shell");
172 }
173#endif
174 ig1 = gvec_shells__.index_by_gvec(G1);
175 assert(ig1 >= 0 && ig1 < ngv);
176 if (f_pw__) {
177 symf += std::conj(f_pw[ig1]) * phase;
178 }
179 if (!is_non_collin && z_pw__) {
180 symz += std::conj(z_pw[ig1]) * phase * S(2, 2);
181 }
182 if (is_non_collin) {
183 auto v = dot(S, vector3d<double_complex>({x_pw[ig1], y_pw[ig1], z_pw[ig1]}));
184 symx += std::conj(v[0]) * phase;
185 symy += std::conj(v[1]) * phase;
186 symz += std::conj(v[2]) * phase;
187 }
188 } else {
189#if !defined(NDEBUG)
190 if (igsh != gvec_shells__.gvec().shell(G1)) {
191 throw std::runtime_error("wrong index of G-shell");
192 }
193#endif
194 assert(ig1 >= 0 && ig1 < ngv);
195 if (f_pw__) {
196 symf += f_pw[ig1] * phase;
197 }
198 if (!is_non_collin && z_pw__) {
199 symz += z_pw[ig1] * phase * S(2, 2);
200 }
201 if (is_non_collin) {
202 auto v = dot(S, vector3d<double_complex>({x_pw[ig1], y_pw[ig1], z_pw[ig1]}));
203 symx += v[0] * phase;
204 symy += v[1] * phase;
205 symz += v[2] * phase;
206 }
207 }
208 } /* loop over symmetries */
209
210 symf *= norm;
211 symx *= norm;
212 symy *= norm;
213 symz *= norm;
214
215 /* apply symmetry operation and get all other plane-wave coefficients */
216
217 for (int isym = 0; isym < sym__.size(); isym++) {
218 auto S = sym__[isym].spin_rotation;
219
220 auto G1 = dot(sym__[isym].spg_op.invRT, G);
221 /* index of a rotated G-vector */
222 int ig1 = gvec_shells__.index_by_gvec(G1);
223
224 if (ig1 != -1) {
225 assert(ig1 >= 0 && ig1 < ngv);
226 auto phase = std::conj(phase_factor(isym, G1));
227 double_complex symf1, symx1, symy1, symz1;
228 if (f_pw__) {
229 symf1 = symf * phase;
230 }
231 if (!is_non_collin && z_pw__) {
232 symz1 = symz * phase * S(2, 2);
233 }
234 if (is_non_collin) {
235 auto v = dot(S, vector3d<double_complex>({symx, symy, symz}));
236 symx1 = v[0] * phase;
237 symy1 = v[1] * phase;
238 symz1 = v[2] * phase;
239 }
240
241 if (is_done[ig1]) {
242 if (f_pw__) {
243 /* check that another symmetry operation leads to the same coefficient */
244 if (utils::abs_diff(sym_f_pw[ig1], symf1) > eps) {
245 std::stringstream s;
246 s << "inconsistent symmetry operation" << std::endl
247 << " existing value : " << sym_f_pw[ig1] << std::endl
248 << " computed value : " << symf1 << std::endl
249 << " difference: " << std::abs(sym_f_pw[ig1] - symf1) << std::endl;
250 RTE_THROW(s);
251 }
252 }
253 if (!is_non_collin && z_pw__) {
254 if (utils::abs_diff(sym_z_pw[ig1], symz1) > eps) {
255 std::stringstream s;
256 s << "inconsistent symmetry operation" << std::endl
257 << " existing value : " << sym_z_pw[ig1] << std::endl
258 << " computed value : " << symz1 << std::endl
259 << " difference: " << std::abs(sym_z_pw[ig1] - symz1) << std::endl;
260 RTE_THROW(s);
261 }
262 }
263 if (is_non_collin) {
264 if (utils::abs_diff(sym_x_pw[ig1], symx1) > eps ||
265 utils::abs_diff(sym_y_pw[ig1], symy1) > eps ||
266 utils::abs_diff(sym_z_pw[ig1], symz1) > eps) {
267
268 RTE_THROW("inconsistent symmetry operation");
269 }
270 }
271 } else {
272 if (f_pw__) {
273 sym_f_pw[ig1] = symf1;
274 }
275 if (!is_non_collin && z_pw__) {
276 sym_z_pw[ig1] = symz1;
277 }
278 if (is_non_collin) {
279 sym_x_pw[ig1] = symx1;
280 sym_y_pw[ig1] = symy1;
281 sym_z_pw[ig1] = symz1;
282
283 }
284 is_done[ig1] = true;
285 }
286 }
287 } /* loop over symmetries */
288 }
289 } /* loop over igloc */
290 }
291 PROFILE_STOP("sirius::symmetrize|fpw|local");
292
293#if !defined(NDEBUG)
294 for (int igloc = 0; igloc < gvec_shells__.gvec_count_remapped(); igloc++) {
295 auto G = gvec_shells__.gvec_remapped(igloc);
296 for (int isym = 0; isym < sym__.size(); isym++) {
297 auto S = sym__[isym].spin_rotation;
298 auto gv_rot = dot(sym__[isym].spg_op.invRT, G);
299 /* index of a rotated G-vector */
300 int ig_rot = gvec_shells__.index_by_gvec(gv_rot);
301 double_complex phase = std::conj(phase_factor(isym, gv_rot));
302
303 if (f_pw__ && ig_rot != -1) {
304 if (utils::abs_diff(sym_f_pw[ig_rot], sym_f_pw[igloc] * phase) > eps) {
305 std::stringstream s;
306 s << "check of symmetrized PW coefficients failed" << std::endl
307 << "difference : " << utils::abs_diff(sym_f_pw[ig_rot], sym_f_pw[igloc] * phase);
308 RTE_THROW(s);
309 }
310 }
311 if (!is_non_collin && z_pw__ && ig_rot != -1) {
312 if (utils::abs_diff(sym_z_pw[ig_rot], sym_z_pw[igloc] * phase * S(2, 2)) > eps) {
313 std::stringstream s;
314 s << "check of symmetrized PW coefficients failed" << std::endl
315 << "difference : " << utils::abs_diff(sym_z_pw[ig_rot], sym_z_pw[igloc] * phase * S(2, 2));
316 RTE_THROW(s);
317 }
318 }
319 }
320 }
321#endif
322
323 if (f_pw__) {
324 gvec_shells__.remap_backward(sym_f_pw, f_pw__);
325 }
326 if (x_pw__) {
327 gvec_shells__.remap_backward(sym_x_pw, x_pw__);
328 }
329 if (y_pw__) {
330 gvec_shells__.remap_backward(sym_y_pw, y_pw__);
331 }
332 if (z_pw__) {
333 gvec_shells__.remap_backward(sym_z_pw, z_pw__);
334 }
335}

337inline void symmetrize_function(Crystal_symmetry const& sym__, Communicator const& comm__, mdarray<double, 3>& frlm__)
338{
339 PROFILE("sirius::symmetrize_function|flm");
340
341 int lmmax = (int)frlm__.size(0);
342 int nrmax = (int)frlm__.size(1);
343 if (sym__.num_atoms() != (int)frlm__.size(2)) {
344 TERMINATE("wrong number of atoms");
345 }
346
347 splindex<splindex_t::block> spl_atoms(sym__.num_atoms(), comm__.size(), comm__.rank());
348
349 int lmax = utils::lmax(lmmax);
350
351 mdarray<double, 2> rotm(lmmax, lmmax);
352
353 mdarray<double, 3> fsym(lmmax, nrmax, spl_atoms.local_size());
354 fsym.zero();
355
356 double alpha = 1.0 / double(sym__.size());
357
358 for (int i = 0; i < sym__.size(); i++) {
359 /* full space-group symmetry operation is {R|t} */
360 int pr = sym__[i].spg_op.proper;
361 auto eang = sym__[i].spg_op.euler_angles;
362 sht::rotation_matrix(lmax, eang, pr, rotm);
363
364 for (int ialoc = 0; ialoc < spl_atoms.local_size(); ialoc++) {
365 int ia = spl_atoms[ialoc];
366 int ja = sym__[i].spg_op.inv_sym_atom[ia];
367 linalg(linalg_t::blas).gemm('N', 'N', lmmax, nrmax, lmmax, &alpha, rotm.at(memory_t::host), rotm.ld(),
368 frlm__.at(memory_t::host, 0, 0, ja), frlm__.ld(), &linalg_const<double>::one(),
369 fsym.at(memory_t::host, 0, 0, ialoc), fsym.ld());
370 }
371 }
372 double* sbuf = spl_atoms.local_size() ? fsym.at(memory_t::host) : nullptr;
373 comm__.allgather(sbuf, frlm__.at(memory_t::host), lmmax * nrmax * spl_atoms.local_size(),
374 lmmax * nrmax * spl_atoms.global_offset());
375}

377inline void symmetrize_vector_function(Crystal_symmetry const& sym__, Communicator const& comm__,
378 mdarray<double, 3>& vz_rlm__)
379{
380 PROFILE("sirius::symmetrize_function|vzlm");
381
382 int lmmax = (int)vz_rlm__.size(0);
383 int nrmax = (int)vz_rlm__.size(1);
384
385 splindex<splindex_t::block> spl_atoms(sym__.num_atoms(), comm__.size(), comm__.rank());
386
387 if (sym__.num_atoms() != (int)vz_rlm__.size(2)) {
388 TERMINATE("wrong number of atoms");
389 }
390
391 int lmax = utils::lmax(lmmax);
392
393 mdarray<double, 2> rotm(lmmax, lmmax);
394
395 mdarray<double, 3> fsym(lmmax, nrmax, spl_atoms.local_size());
396 fsym.zero();

422inline void symmetrize_vector_function(Crystal_symmetry const& sym__, Communicator const& comm__,
423 mdarray<double, 3>& vx_rlm__, mdarray<double, 3>& vy_rlm__,
424 mdarray<double, 3>& vz_rlm__)
425{
426 PROFILE("sirius::symmetrize_function|vlm");
427
428 int lmmax = (int)vx_rlm__.size(0);
429 int nrmax = (int)vx_rlm__.size(1);
430
431 splindex<splindex_t::block> spl_atoms(sym__.num_atoms(), comm__.size(), comm__.rank());
432
433 int lmax = utils::lmax(lmmax);
434
435 mdarray<double, 2> rotm(lmmax, lmmax);
436
437 mdarray<double, 4> v_sym(lmmax, nrmax, spl_atoms.local_size(), 3);
438 v_sym.zero();
439
440 mdarray<double, 3> vtmp(lmmax, nrmax, 3);
441
442 double alpha = 1.0 / double(sym__.size());
443
444 std::vector<mdarray<double, 3>*> vrlm({&vx_rlm__, &vy_rlm__, &vz_rlm__});
445
446 for (int i = 0; i < sym__.size(); i++) {
447 /* full space-group symmetry operation is {R|t} */
448 int pr = sym__[i].spg_op.proper;
449 auto eang = sym__[i].spg_op.euler_angles;
450 auto S = sym__[i].spin_rotation;
451 sht::rotation_matrix(lmax, eang, pr, rotm);
452
453 for (int ialoc = 0; ialoc < spl_atoms.local_size(); ialoc++) {
454 int ia = spl_atoms[ialoc];
455 int ja = sym__[i].spg_op.inv_sym_atom[ia];
456 for (int k: {0, 1, 2}) {
457 linalg(linalg_t::blas).gemm('N', 'N', lmmax, nrmax, lmmax, &alpha, rotm.at(memory_t::host), rotm.ld(),
458 vrlm[k]->at(memory_t::host, 0, 0, ja), vrlm[k]->ld(),
459 &linalg_const<double>::zero(), vtmp.at(memory_t::host, 0, 0, k), vtmp.ld());
460 }
461 #pragma omp parallel
462 for (int k: {0, 1, 2}) {
463 for (int j: {0, 1, 2}) {
464 #pragma omp for
465 for (int ir = 0; ir < nrmax; ir++) {
466 for (int lm = 0; lm < lmmax; lm++) {
467 v_sym(lm, ir, ialoc, k) += S(k, j) * vtmp(lm, ir, j);
468 }
469 }
470 }
471 }
472 }
473 }
474
475 for (int k: {0, 1, 2}) {
476 double* sbuf = spl_atoms.local_size() ? v_sym.at(memory_t::host, 0, 0, 0, k) : nullptr;
477 comm__.allgather(sbuf, vrlm[k]->at(memory_t::host), lmmax * nrmax * spl_atoms.local_size(),
478 lmmax * nrmax * spl_atoms.global_offset());
479 }
480}

537/// Symmetrize density or occupancy matrix according to a given list of basis functions.
538/** Density matrix arises in LAPW or PW methods. In PW it is computed in the basis of beta-projectors. Occupancy
539 * matrix is computed for the Hubbard-U correction. In both cases the matrix has the same structure and is
540 * symmetrized in the same way The symmetrization does depend explicitly on the beta or wfc. The last
541 * parameter is on when the atom has spin-orbit coupling and hubbard correction in
542 * that case, we must skip half of the indices because of the averaging of the
543 * radial integrals over the total angular momentum
544 */
545inline void symmetrize(const mdarray<double_complex, 4> &ns_,
546 const basis_functions_index &indexb,
547 const int ia,
548 const int ja,
549 const int ndm,
550 const mdarray<double, 2> &rotm,
551 const mdarray<double_complex, 2> &spin_rot_su2,
552 mdarray<double_complex, 4> &dm_,
553 const bool hubbard_)
554{
555 for (int xi1 = 0; xi1 < indexb.size(); xi1++) {
556 int l1 = indexb[xi1].l;
557 int lm1 = indexb[xi1].lm;
558 int o1 = indexb[xi1].order;
559
560 if ((hubbard_)&&(xi1 >= (2 * l1 + 1))) {
561 break;
562 }
563
564 for (int xi2 = 0; xi2 < indexb.size(); xi2++) {
565 int l2 = indexb[xi2].l;
566 int lm2 = indexb[xi2].lm;
567 int o2 = indexb[xi2].order;
568 std::array<double_complex, 3> dm_rot_spatial = {0, 0, 0};
569
570 //} the hubbard treatment when spin orbit coupling is present is
571 // foundamentally wrong since we consider the full hubbard
572 // correction with a averaged wave function (meaning we neglect the
573 // L.S correction within hubbard). A better option (although still
574 // wrong from physics pov) would be to consider a multi orbital case.
575
576 if ((hubbard_)&&(xi2 >= (2 * l2 + 1))) {
577 break;
578 }
579
580 // if (l1 == l2) {
581 // the rotation matrix of the angular momentum is block
582 // diagonal and does not couple different l.
583 for (int j = 0; j < ndm; j++) {
584 for (int m3 = -l1; m3 <= l1; m3++) {
585 int lm3 = utils::lm(l1, m3);
586 int xi3 = indexb.index_by_lm_order(lm3, o1);
587 for (int m4 = -l2; m4 <= l2; m4++) {
588 int lm4 = utils::lm(l2, m4);
589 int xi4 = indexb.index_by_lm_order(lm4, o2);
590 dm_rot_spatial[j] += ns_(xi3, xi4, j, ia) *
591 rotm(lm1, lm3) * rotm(lm2, lm4);
592 }
593 }
594 }
595
596 /* magnetic symmetrization */
597 if (ndm == 1) {
598 dm_(xi1, xi2, 0, ja) += dm_rot_spatial[0];
599 } else {
600 double_complex spin_dm[2][2] = {
601 {dm_rot_spatial[0], dm_rot_spatial[2]},
602 {std::conj(dm_rot_spatial[2]), dm_rot_spatial[1]}};
603
604 /* spin blocks of density matrix are: uu, dd, ud
605 the mapping from linear index (0, 1, 2) of density matrix components is:
606 for the first spin index: k & 1, i.e. (0, 1, 2) -> (0, 1, 0)
607 for the second spin index: min(k, 1), i.e. (0, 1, 2) -> (0, 1, 1)
608 */
609 for (int k = 0; k < ndm; k++) {
610 for (int is = 0; is < 2; is++) {
611 for (int js = 0; js < 2; js++) {
612 dm_(xi1, xi2, k, ja) += spin_rot_su2(k & 1, is) * spin_dm[is][js] * std::conj(spin_rot_su2(std::min(k, 1), js));
613 }
614 }
615 }
616 }
617 }
618 }
619}

621inline void
622symmetrize(std::function<sddk::mdarray<double_complex, 3>&(int ia__)> dm__,
623 int num_mag_comp__, Crystal_symmetry const& sym__,
624 std::function<sirius::experimental::basis_functions_index const*(int)> indexb__)
625{
626 /* quick exit */
627 if (sym__.size() == 1) {
628 return;
629 }
630
631 std::vector<sddk::mdarray<double_complex, 3>> dmsym(sym__.num_atoms());
632 for (int ia = 0; ia < sym__.num_atoms(); ia++) {
633 int iat = sym__.atom_type(ia);
634 if (indexb__(iat)) {
635 dmsym[ia] = sddk::mdarray<double_complex, 3>(indexb__(iat)->size(), indexb__(iat)->size(), 4);
636 dmsym[ia].zero();
637 }
638 }
639
640 int lmax{0};
641 for (int iat = 0; iat < sym__.num_atom_types(); iat++) {
642 if (indexb__(iat)) {
643 lmax = std::max(lmax, indexb__(iat)->indexr().lmax());
644 }
645 }
646
647 /* loop over symmetry operations */
648 for (int isym = 0; isym < sym__.size(); isym++) {
649 int pr = sym__[isym].spg_op.proper;
650 auto eang = sym__[isym].spg_op.euler_angles;
651 auto rotm = sht::rotation_matrix<double>(lmax, eang, pr);
652 auto spin_rot_su2 = rotation_matrix_su2(sym__[isym].spin_rotation);
653
654 for (int ia = 0; ia < sym__.num_atoms(); ia++) {
655 int iat = sym__.atom_type(ia);
656
657 if (!indexb__(iat)) {
658 continue;
659 }
660
661 int ja = sym__[isym].spg_op.inv_sym_atom[ia];
662
663 auto& indexb = *indexb__(iat);
664 auto& indexr = indexb.indexr();
665
666 int mmax = 2 * indexb.indexr().lmax() + 1;
667 sddk::mdarray<double_complex, 3> dm_ia(mmax, mmax, num_mag_comp__);
668
669 /* loop over radial functions */
670 for (int idxrf1 = 0; idxrf1 < indexr.size(); idxrf1++) {
671 /* angular momentum of radial function */
672 auto am1 = indexr.am(idxrf1);
673 auto ss1 = am1.subshell_size();
674 auto offset1 = indexb.offset(idxrf1);
675 for (int idxrf2 = 0; idxrf2 < indexr.size(); idxrf2++) {
676 /* angular momentum of radial function */
677 auto am2 = indexr.am(idxrf2);
678 auto ss2 = am2.subshell_size();
679 auto offset2 = indexb.offset(idxrf2);
680
681 dm_ia.zero();
682 for (int j = 0; j < num_mag_comp__; j++) {
683 /* apply spatial rotation */
684 for (int m1 = 0; m1 < ss1; m1++) {
685 for (int m2 = 0; m2 < ss2; m2++) {
686 for (int m1p = 0; m1p < ss1; m1p++) {
687 for (int m2p = 0; m2p < ss2; m2p++) {
688 dm_ia(m1, m2, j) += rotm[am1.l()](m1, m1p) *
689 dm__(ja)(offset1 + m1p, offset2 + m2p, j) *
690 rotm[am2.l()](m2, m2p);
691 }
692 }
693 }
694 }
695 }
696 /* magnetic symmetry */
697 if (num_mag_comp__ == 1) { /* trivial non-magnetic case */
698 for (int m1 = 0; m1 < ss1; m1++) {
699 for (int m2 = 0; m2 < ss2; m2++) {
700 dmsym[ia](m1 + offset1, m2 + offset2, 0) += dm_ia(m1, m2, 0);
701 }
702 }
703 } else {
704 int const map_s[3][2] = {{0, 0}, {1, 1}, {0, 1}};
705 for (int j = 0; j < num_mag_comp__; j++) {
706 int s1 = map_s[j][0];
707 int s2 = map_s[j][1];
708
709 for (int m1 = 0; m1 < ss1; m1++) {
710 for (int m2 = 0; m2 < ss2; m2++) {
711 double_complex dm[2][2] = {{dm_ia(m1, m2, 0), 0}, {0, dm_ia(m1, m2, 1)}};
712 if (num_mag_comp__ == 3) {
713 dm[0][1] = dm_ia(m1, m2, 2);
714 dm[1][0] = std::conj(dm[0][1]);
715 }
716
717 for (int s1p = 0; s1p < 2; s1p++) {
718 for (int s2p = 0; s2p < 2; s2p++) {
719 dmsym[ia](m1 + offset1, m2 + offset2, j) +=
720 spin_rot_su2(s1, s1p) * dm[s1p][s2p] *
721 std::conj(spin_rot_su2(s2, s2p));
722 }
723 }
724 }
725 }
726 }
727 if (num_mag_comp__ == 3) {
728 for (int m1 = 0; m1 < ss1; m1++) {
729 for (int m2 = 0; m2 < ss2; m2++) {
730 dmsym[ia](m1 + offset1, m2 + offset2, 3) =
731 std::conj(dmsym[ia](m1 + offset1, m2 + offset2, 2));
732 }
733 }
734 }
735 }
736
737 }
738 }
739 }
740 }
741
742 double alpha = 1.0 / sym__.size();
743
744 for (int ia = 0; ia < sym__.num_atoms(); ia++) {
745 int iat = sym__.atom_type(ia);
746 if (indexb__(iat)) {
747 for (size_t i = 0; i < dm__(ia).size(); i++) {
748 dm__(ia)[i] = dmsym[ia][i] * alpha;
749 }
750
751 }
752 }
753}
