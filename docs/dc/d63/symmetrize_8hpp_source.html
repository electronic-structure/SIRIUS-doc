37 /// Symmetrize scalar function.
38 /** The following operation is performed:
39  \f[
40  f_{\mathrm{sym}}({\bf x}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} f({\bf \hat P x})
41  \f]
42  For the function expanded in plane-waves we have:
43  \f[
44  f_{\mathrm{sym}}({\bf x}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} \sum_{\bf G}
45  e^{i{\bf G \hat P x}} \hat f({\bf G})
46  = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G (Rx +
47  t)}} \hat f({\bf G})
48  = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G t}}
49  e^{i{\bf R^T G x}} \hat f({\bf G})
50  \f]
51  Substitute \f$\bf \tilde G = \bf R^T \bf G\f$
52  \f[
53  f_{\mathrm{sym}}({\bf x}) =
54  \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \tilde G}} e^{i {\bf \tilde G} }\sum_{{\bf \hat P}}
55  e^{i {\bf R}^{-T} {\bf t}} \hat f ({\bf R}^{-T} {\bf \tilde G}) \,,
56  \f]
57  to find the Fourier coefficients \f$ \hat f_{\mathrm{sym}} \f$ of \f$f_{\mathrm{sym}}\f$ in
58  terms of \f$ \hat f \f$:
59  \f[
60  \hat f_{\mathrm{sym}}({\bf G}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} e^{i {\bf R}^{-T} {\bf t}}
61  \hat f ({\bf R}^{-T} {\bf G})\,.
62  \f]
63  Once \f$\hat f_{\mathrm{sym}} \f$ has been calculated by the above formula for a single \f$\bf G \f$, its values at
64  points \f${\bf R}^{-T} {\bf G}\f$, \f$\forall \, {\bf R}\f$ are given by the update formula
65  \f[
66  \hat f_{\mathrm{sym}} ({\bf R}^{-T} {\bf G}) = e^{-i {\bf R}^{-T} {\bf G} {\bf t}}
67  \hat{f}_{\mathrm{sym}} ({\bf G})\,,
68  \f]
69 
70  which follows by using that \f$f_{\mathrm{sym}}({\bf \hat P} {\bf x}) = f_{\mathrm{sym}}({\bf x})\f$:
71  \f{eqnarray*}{
72  f_{\mathrm{sym}}(\hat{P}{\bf x}) &=& \sum_G e^{i G ({\bf R}{\bf x} + {\bf t} )} \hat{f}_{\mathrm{sym}}({\bf G}) \ \
73  &=& \sum_G e^{i {\bf G} {\bf x}} e^{i {\bf R}^{-T} {\bf G} {\bf t}}
74  \hat{f}_{\mathrm{sym}}({\bf R}^{-T} {\bf G}) \,.
75  \f}
76  */
77 inline void symmetrize_function(Unit_cell_symmetry const& sym__, Gvec_shells const& gvec_shells__,
78  mdarray<double_complex, 3> const& sym_phase_factors__, double_complex* f_pw__)
79 {
80  PROFILE("sirius::symmetrize_function|fpw");
81 
82  auto v = gvec_shells__.remap_forward(f_pw__);
83 
84  std::vector<double_complex> sym_f_pw(v.size(), 0);
85  std::vector<bool> is_done(v.size(), false);
86 
87  double norm = 1 / double(sym__.num_mag_sym());
88 
89  auto phase_factor = [&](int isym, vector3d<int> G)
90  {
91  return sym_phase_factors__(0, G[0], isym) *
92  sym_phase_factors__(1, G[1], isym) *
93  sym_phase_factors__(2, G[2], isym);
94  };
95 
96  PROFILE_START("sirius::symmetrize_function|fpw|local");
97 
98  #pragma omp parallel
99  {
100  int nt = omp_get_max_threads();
101  int tid = omp_get_thread_num();
102 
103  for (int igloc = 0; igloc < gvec_shells__.gvec_count_remapped(); igloc++) {
104  auto G = gvec_shells__.gvec_remapped(igloc);
105 
106  int igsh = gvec_shells__.gvec_shell_remapped(igloc);
107 
108  /* each thread is working on full shell of G-vectors */
109  if (igsh % nt == tid && !is_done[igloc]) {
110  double_complex zsym(0, 0);
111 
112  for (int i = 0; i < sym__.num_mag_sym(); i++) {
113  auto& invRT = sym__.magnetic_group_symmetry(i).spg_op.invRT;
114  auto gv_rot = invRT * G;
115  double_complex phase = phase_factor(i, gv_rot);
116 
117  /* local index of a rotated G-vector */
118  int ig_rot = gvec_shells__.index_by_gvec(gv_rot);
119 
120  if (ig_rot == -1) {
121  gv_rot = gv_rot * (-1);
122  ig_rot = gvec_shells__.index_by_gvec(gv_rot);
123  assert(ig_rot >= 0 && ig_rot < (int)v.size());
124  zsym += std::conj(v[ig_rot]) * phase;
125  } else {
126  assert(ig_rot >= 0 && ig_rot < (int)v.size());
127  zsym += v[ig_rot] * phase;
128  }
129  } /* loop over symmetries */
130 
131  zsym *= norm;
132 
133  for (int i = 0; i < sym__.num_mag_sym(); i++) {
134  auto& invRT = sym__.magnetic_group_symmetry(i).spg_op.invRT;
135  auto gv_rot = invRT * G;
136  /* index of a rotated G-vector */
137  int ig_rot = gvec_shells__.index_by_gvec(gv_rot);
138  double_complex phase = std::conj(phase_factor(i, gv_rot));
139 
140  if (ig_rot == -1) {
141  /* skip */
142  } else {
143  assert(ig_rot >= 0 && ig_rot < int(v.size()));
144  sym_f_pw[ig_rot] = zsym * phase;
145  is_done[ig_rot] = true;
146  }
147  } /* loop over symmetries */
148  }
149  } /* loop over igloc */
150  }
151  PROFILE_STOP("sirius::symmetrize_function|fpw|local");
152 
153 
154  gvec_shells__.remap_backward(sym_f_pw, f_pw__);
155 }

158  mdarray<double_complex, 3> const& sym_phase_factors__, double_complex* fz_pw__)
159 {
160  PROFILE("sirius::symmetrize_vector_function|vzpw");

201  assert(ig_rot >= 0 && ig_rot < (int)v.size());
202  zsym += v[ig_rot] * phase;
203  }
204  } /* loop over symmetries */
205 
206  zsym *= norm;
207 
208  for (int i = 0; i < sym__.num_mag_sym(); i++) {
209  auto& invRT = sym__.magnetic_group_symmetry(i).spg_op.invRT;
210  auto& S = sym__.magnetic_group_symmetry(i).spin_rotation;
211  auto gv_rot = invRT * G;
212  /* index of rotated G-vector */
213  int ig_rot = gvec_shells__.index_by_gvec(gv_rot);
214  double_complex phase = std::conj(phase_factor(i, gv_rot)) / S(2, 2) ;
215 
216  if (ig_rot == -1) {
217  /* skip */
218  } else {
219  assert(ig_rot >= 0 && ig_rot < int(v.size()));
220  sym_f_pw[ig_rot] = zsym * phase;
221  is_done[ig_rot] = true;
222  }
223  } /* loop over symmetries */
224  }
225  }
226  }
227 
228  gvec_shells__.remap_backward(sym_f_pw, fz_pw__);

231 /// Symmetrize vector valued function.
232 /** The following operations are performed.
233  *
234  * Fourier coefficient of symmetrized function:
235  * \f[
236  * \hat f_{\mathrm{sym}}({\bf G}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} e^{i {\bf R}^{-T} {\bf t}} {\bf S} \hat f ({\bf R}^{-T} {\bf G})\,.
237  * \f]
238  *
239  * Update formula when \f$\hat f({\bf G})\f$ is known:
240  * \f[
241  * \hat f_{\mathrm{sym}} ({\bf R}^{-T} {\bf G}) = e^{-i {\bf R}^{-T} {\bf G} {\bf t}}
242  * {\bf S}^{-1} \hat{f}_{\mathrm{sym}} ({\bf G})\,,
243  * \f]
244  *
245  * The derivation works similarly to the one for symmetrize_function().
246  */
247 inline void symmetrize_vector_function(Unit_cell_symmetry const& sym__, Gvec_shells const& gvec_shells__,
248  mdarray<double_complex, 3> const& sym_phase_factors__,
249  double_complex* fx_pw__, double_complex* fy_pw__, double_complex* fz_pw__)
250 {
251  PROFILE("sirius::symmetrize_vector_function|vpw");
252 
253  auto vx = gvec_shells__.remap_forward(fx_pw__);
254  auto vy = gvec_shells__.remap_forward(fy_pw__);
255  auto vz = gvec_shells__.remap_forward(fz_pw__);
256 
257  std::vector<double_complex> sym_fx_pw(vx.size(), 0);
258  std::vector<double_complex> sym_fy_pw(vx.size(), 0);
259  std::vector<double_complex> sym_fz_pw(vx.size(), 0);
260  std::vector<bool> is_done(vx.size(), false);
261 
262  double norm = 1 / double(sym__.num_mag_sym());
263 
264  auto phase_factor = [&](int isym, const vector3d<int>& G)
265  {
266  return sym_phase_factors__(0, G[0], isym) *
267  sym_phase_factors__(0, G[1], isym) *
268  sym_phase_factors__(0, G[2], isym);
269  };
270 
271  auto vrot = [&](vector3d<double_complex> const& v, matrix3d<double> const& S) -> vector3d<double_complex>
272  {
273  return S * v;
274  };
275 
276  #pragma omp parallel
277  {
278  int nt = omp_get_max_threads();
279  int tid = omp_get_thread_num();
280 
281  for (int igloc = 0; igloc < gvec_shells__.gvec_count_remapped(); igloc++) {
282  auto G = gvec_shells__.gvec_remapped(igloc);
283 
284  int igsh = gvec_shells__.gvec_shell_remapped(igloc);
285 
286  if (igsh % nt == tid && !is_done[igloc]) {
287  double_complex xsym(0, 0);
288  double_complex ysym(0, 0);
289  double_complex zsym(0, 0);
290 
291  for (int i = 0; i < sym__.num_mag_sym(); i++) {
292  /* full space-group symmetry operation is {R|t} */
293  auto& invRT = sym__.magnetic_group_symmetry(i).spg_op.invRT;
294  auto& S = sym__.magnetic_group_symmetry(i).spin_rotation;
295  double_complex phase = phase_factor(i, G);
296  auto gv_rot = invRT * G;
297  /* index of a rotated G-vector */
298  int ig_rot = gvec_shells__.index_by_gvec(gv_rot);
299 
300  if (ig_rot == -1) {
301  gv_rot = gv_rot * (-1);
302  ig_rot = gvec_shells__.index_by_gvec(gv_rot);
303  auto v_rot = vrot({vx[ig_rot], vy[ig_rot], vz[ig_rot]}, S);
304  assert(ig_rot >=0 && ig_rot < (int)vx.size());
305  xsym += std::conj(v_rot[0]) * phase;
306  ysym += std::conj(v_rot[1]) * phase;
307  zsym += std::conj(v_rot[2]) * phase;
308  } else {
309  assert(ig_rot >=0 && ig_rot < (int)vx.size());
310  auto v_rot = vrot({vx[ig_rot], vy[ig_rot], vz[ig_rot]}, S);
311  xsym += v_rot[0] * phase;
312  ysym += v_rot[1] * phase;
313  zsym += v_rot[2] * phase;
314  }
315  } /* loop over symmetries */
316 
317  xsym *= norm;
318  ysym *= norm;
319  zsym *= norm;
320 
321  for (int i = 0; i < sym__.num_mag_sym(); i++) {
322  const auto& invRT = sym__.magnetic_group_symmetry(i).spg_op.invRT;
323  const auto& invS = sym__.magnetic_group_symmetry(i).spin_rotation_inv;
324  auto gv_rot = invRT * G;
325  /* index of a rotated G-vector */
326  int ig_rot = gvec_shells__.index_by_gvec(gv_rot);
327  auto v_rot = vrot({xsym, ysym, zsym}, invS);
328  double_complex phase = std::conj(phase_factor(i, gv_rot));
329 
330  if (ig_rot == -1) {
331  /* skip */
332  } else {
333  assert(ig_rot >= 0 && ig_rot < int(vz.size()));
334  sym_fx_pw[ig_rot] = v_rot[0] * phase;
335  sym_fy_pw[ig_rot] = v_rot[1] * phase;
336  sym_fz_pw[ig_rot] = v_rot[2] * phase;
337  is_done[ig_rot] = true;
338  }
339  } /* loop over symmetries */
340  }
341  }
342  }
343 
344  gvec_shells__.remap_backward(sym_fx_pw, fx_pw__);
345  gvec_shells__.remap_backward(sym_fy_pw, fy_pw__);
346  gvec_shells__.remap_backward(sym_fz_pw, fz_pw__);
347 }

349 inline void symmetrize_function(Unit_cell_symmetry const& sym__, Communicator const& comm__, mdarray<double, 3>& frlm__)
350 {
351  PROFILE("sirius::symmetrize_function|flm");
352 
353  int lmmax = (int)frlm__.size(0);
354  int nrmax = (int)frlm__.size(1);
355  if (sym__.num_atoms() != (int)frlm__.size(2)) {
356  TERMINATE("wrong number of atoms");
357  }
358 
359  splindex<splindex_t::block> spl_atoms(sym__.num_atoms(), comm__.size(), comm__.rank());
360 
361  int lmax = utils::lmax(lmmax);
362 
363  mdarray<double, 2> rotm(lmmax, lmmax);
364 
365  mdarray<double, 3> fsym(lmmax, nrmax, spl_atoms.local_size());
366  fsym.zero();
367 
368  double alpha = 1.0 / double(sym__.num_mag_sym());
369 
370  for (int i = 0; i < sym__.num_mag_sym(); i++) {
371  /* full space-group symmetry operation is {R|t} */
372  int pr = sym__.magnetic_group_symmetry(i).spg_op.proper;
373  auto eang = sym__.magnetic_group_symmetry(i).spg_op.euler_angles;
374  int isym = sym__.magnetic_group_symmetry(i).isym;
375  SHT::rotation_matrix(lmax, eang, pr, rotm);
376 
377  for (int ia = 0; ia < sym__.num_atoms(); ia++) {
378  int ja = sym__.sym_table(ia, isym);
379  auto location = spl_atoms.location(ja);
380  if (location.rank == comm__.rank()) {
381  linalg(linalg_t::blas).gemm('N', 'N', lmmax, nrmax, lmmax, &alpha, rotm.at(memory_t::host), rotm.ld(),
382  frlm__.at(memory_t::host, 0, 0, ia), frlm__.ld(), &linalg_const<double>::one(),
383  fsym.at(memory_t::host, 0, 0, location.local_index), fsym.ld());
384  }
385  }
386  }
387  double* sbuf = spl_atoms.local_size() ? fsym.at(memory_t::host) : nullptr;
388  comm__.allgather(sbuf, frlm__.at(memory_t::host),
389  lmmax * nrmax * spl_atoms.global_offset(),
390  lmmax * nrmax * spl_atoms.local_size());
391 }

442 inline void symmetrize_vector_function(Unit_cell_symmetry const& sym__, Communicator const& comm__,
443  mdarray<double, 3>& vx_rlm__, mdarray<double, 3>& vy_rlm__,
444  mdarray<double, 3>& vz_rlm__)
445 {
446  PROFILE("sirius::symmetrize_function|vlm");
447 
448  int lmmax = (int)vx_rlm__.size(0);
449  int nrmax = (int)vx_rlm__.size(1);
450 
451  splindex<splindex_t::block> spl_atoms(sym__.num_atoms(), comm__.size(), comm__.rank());
452 
453  int lmax = utils::lmax(lmmax);
454 
455  mdarray<double, 2> rotm(lmmax, lmmax);
456 
457  mdarray<double, 4> v_sym(lmmax, nrmax, spl_atoms.local_size(), 3);
458  v_sym.zero();
459 
460  mdarray<double, 3> vtmp(lmmax, nrmax, 3);
461 
462  double alpha = 1.0 / double(sym__.num_mag_sym());
463 
464  std::vector<mdarray<double, 3>*> vrlm({&vx_rlm__, &vy_rlm__, &vz_rlm__});
465 
466  for (int i = 0; i < sym__.num_mag_sym(); i++) {
467  /* full space-group symmetry operation is {R|t} */
468  int pr = sym__.magnetic_group_symmetry(i).spg_op.proper;
469  auto eang = sym__.magnetic_group_symmetry(i).spg_op.euler_angles;
470  int isym = sym__.magnetic_group_symmetry(i).isym;
471  auto S = sym__.magnetic_group_symmetry(i).spin_rotation;
472  SHT::rotation_matrix(lmax, eang, pr, rotm);
473 
474  for (int ia = 0; ia < sym__.num_atoms(); ia++) {
475  int ja = sym__.sym_table(ia, isym);
476  auto location = spl_atoms.location(ja);
477  if (location.rank == comm__.rank()) {
478  for (int k: {0, 1, 2}) {
479  linalg(linalg_t::blas).gemm('N', 'N', lmmax, nrmax, lmmax, &alpha, rotm.at(memory_t::host), rotm.ld(),
480  vrlm[k]->at(memory_t::host, 0, 0, ia), vrlm[k]->ld(),
481  &linalg_const<double>::zero(), vtmp.at(memory_t::host, 0, 0, k), vtmp.ld());
482  }
483  #pragma omp parallel
484  for (int k: {0, 1, 2}) {
485  for (int j: {0, 1, 2}) {
486  #pragma omp for
487  for (int ir = 0; ir < nrmax; ir++) {
488  for (int lm = 0; lm < lmmax; lm++) {
489  v_sym(lm, ir, location.local_index, k) += S(k, j) * vtmp(lm, ir, j);
490  }
491  }
492  }
493  }
494  }
495  }
496  }
497

561 /// Symmetrize density or occupancy matrix according to a given list of basis functions.
562 /** Density matrix arises in LAPW or PW methods. In PW it is computed in the basis of beta-projectors. Occupancy
563  * matrix is computed for the Hubbard-U correction. In both cases the matrix has the same structure and is
564  * symmetrized in the same way The symmetrization does depend explicitly on the beta or wfc. The last
565  * parameter is on when the atom has spin-orbit coupling and hubbard correction in
566  * that case, we must skip half of the indices because of the averaging of the
567  * radial integrals over the total angular momentum
568  */
569 inline void symmetrize(const mdarray<double_complex, 4> &ns_,
570  const basis_functions_index &indexb,
571  const int ia,
572  const int ja,
573  const int ndm,
574  const mdarray<double, 2> &rotm,
575  const mdarray<double_complex, 2> &spin_rot_su2,
576  mdarray<double_complex, 4> &dm_,
577  const bool hubbard_)
578 {
579  for (int xi1 = 0; xi1 < indexb.size(); xi1++) {
580  int l1 = indexb[xi1].l;
581  int lm1 = indexb[xi1].lm;
582  int o1 = indexb[xi1].order;
583 
584  if ((hubbard_)&&(xi1 >= (2 * l1 + 1))) {
585  break;
586  }
587 
588  for (int xi2 = 0; xi2 < indexb.size(); xi2++) {
589  int l2 = indexb[xi2].l;
590  int lm2 = indexb[xi2].lm;
591  int o2 = indexb[xi2].order;
592  std::array<double_complex, 3> dm_rot_spatial = {0, 0, 0};
593 
594  //} the hubbard treatment when spin orbit coupling is present is
595  // foundamentally wrong since we consider the full hubbard
596  // correction with a averaged wave function (meaning we neglect the
597  // L.S correction within hubbard). A better option (although still
598  // wrong from physics pov) would be to consider a multi orbital case.
599 
600  if ((hubbard_)&&(xi2 >= (2 * l2 + 1))) {
601  break;
602  }
603 
604  // if (l1 == l2) {
605  // the rotation matrix of the angular momentum is block
606  // diagonal and does not couple different l.
607  for (int j = 0; j < ndm; j++) {
608  for (int m3 = -l1; m3 <= l1; m3++) {
609  int lm3 = utils::lm(l1, m3);
610  int xi3 = indexb.index_by_lm_order(lm3, o1);
611  for (int m4 = -l2; m4 <= l2; m4++) {
612  int lm4 = utils::lm(l2, m4);
613  int xi4 = indexb.index_by_lm_order(lm4, o2);
614  dm_rot_spatial[j] += ns_(xi3, xi4, j, ja) *
615  rotm(lm1, lm3) * rotm(lm2, lm4);
616  }
617  }
618  }
619 
620  /* magnetic symmetrization */
621  if (ndm == 1) {
622  dm_(xi1, xi2, 0, ia) += dm_rot_spatial[0];
623  } else {
624  double_complex spin_dm[2][2] = {
625  {dm_rot_spatial[0], dm_rot_spatial[2]},
626  {std::conj(dm_rot_spatial[2]), dm_rot_spatial[1]}};
627 
628  /* spin blocks of density matrix are: uu, dd, ud
629  the mapping from linear index (0, 1, 2) of density matrix components is:
630  for the first spin index: k & 1, i.e. (0, 1, 2) -> (0, 1, 0)
631  for the second spin index: min(k, 1), i.e. (0, 1, 2) -> (0, 1, 1)
632  */
633  for (int k = 0; k < ndm; k++) {
634  for (int is = 0; is < 2; is++) {
635  for (int js = 0; js < 2; js++) {
636  dm_(xi1, xi2, k, ia) += spin_rot_su2(k & 1, is) * spin_dm[is][js] * std::conj(spin_rot_su2(std::min(k, 1), js));
637  }
638  }
639  }
640  }
641  }
642  }
643 }
