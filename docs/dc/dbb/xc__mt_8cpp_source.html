35 void xc_mt_nonmagnetic(Radial_grid<double> const& rgrid__, SHT const& sht__, std::vector<XC_functional*> xc_func__,
36  Flm const& rho_lm__, Ftp& rho_tp__, Flm& vxc_lm__, Flm& exc_lm__)
37 {
38  bool is_gga{false};
39  for (auto& ixc : xc_func__) {
40  if (ixc->is_gga() || ixc->is_vdw()) {
41  is_gga = true;
42  }
43  }
44 
45  Ftp exc_tp(sht__.num_points(), rgrid__);
46  Ftp vxc_tp(sht__.num_points(), rgrid__);
47 
48  assert(rho_tp__.size() == vxc_tp.size());
49  assert(rho_tp__.size() == exc_tp.size());
50 
51  Ftp grad_rho_grad_rho_tp;
52  Ftp vsigma_tp;
53  Ftp lapl_rho_tp;
54  Spheric_vector_function<function_domain_t::spatial, double> grad_rho_tp;
55 
56  /* use Laplacian (true) or divergence of gradient (false) */
57  bool use_lapl{false};
58 
59  if (is_gga) {
60  /* compute gradient in Rlm spherical harmonics */
61  auto grad_rho_lm = gradient(rho_lm__);
62  grad_rho_tp = Spheric_vector_function<function_domain_t::spatial, double>(sht__.num_points(), rgrid__);
63  /* backward transform gradient from Rlm to (theta, phi) */
64  for (int x = 0; x < 3; x++) {
65  transform(sht__, grad_rho_lm[x], grad_rho_tp[x]);
66  }
67  /* compute density gradient product */
68  grad_rho_grad_rho_tp = grad_rho_tp * grad_rho_tp;
69  assert(rho_tp__.size() == grad_rho_grad_rho_tp.size());
70 
71  vsigma_tp = Ftp(sht__.num_points(), rgrid__);
72  assert(rho_tp__.size() == vsigma_tp.size());
73  if (use_lapl) {
74  /* backward transform Laplacian from Rlm to (theta, phi) */
75  lapl_rho_tp = transform(sht__, laplacian(rho_lm__));
76  assert(lapl_rho_tp.size() == rho_tp__.size());
77  }
78  }
79 
80  for (auto& ixc: xc_func__) {
81  /* if this is an LDA functional */
82  if (ixc->is_lda()) {
83  ixc->get_lda(sht__.num_points() * rgrid__.num_points(), rho_tp__.at(memory_t::host),
84  vxc_tp.at(memory_t::host), exc_tp.at(memory_t::host));
85  }
86  /* if this is a GGA functional */
87  if (ixc->is_gga()) {
88 
89  /* compute vrho and vsigma */
90  ixc->get_gga(sht__.num_points() * rgrid__.num_points(), rho_tp__.at(memory_t::host),
91  grad_rho_grad_rho_tp.at(memory_t::host), vxc_tp.at(memory_t::host), vsigma_tp.at(memory_t::host),
92  exc_tp.at(memory_t::host));
93 
94  if (use_lapl) {
95  vxc_tp -= 2.0 * vsigma_tp * lapl_rho_tp;
96 
97  /* compute gradient of vsgima in spherical harmonics */
98  auto grad_vsigma_lm = gradient(transform(sht__, vsigma_tp));
99 
100  /* backward transform gradient from Rlm to (theta, phi) */
101  Spheric_vector_function<function_domain_t::spatial, double> grad_vsigma_tp(sht__.num_points(), rgrid__);
102  for (int x = 0; x < 3; x++) {
103  transform(sht__, grad_vsigma_lm[x], grad_vsigma_tp[x]);
104  }
105 
106  /* compute scalar product of two gradients */
107  auto grad_vsigma_grad_rho_tp = grad_vsigma_tp * grad_rho_tp;
108  /* add remaining term to Vxc */
109  vxc_tp -= 2.0 * grad_vsigma_grad_rho_tp;
110  } else {
111  Spheric_vector_function<function_domain_t::spectral, double> vsigma_grad_rho_lm(sht__.lmmax(), rgrid__);
112  for (int x: {0, 1, 2}) {
113  auto vsigma_grad_rho_tp = vsigma_tp * grad_rho_tp[x];
114  transform(sht__, vsigma_grad_rho_tp, vsigma_grad_rho_lm[x]);
115  }
116  auto div_vsigma_grad_rho_tp = transform(sht__, divergence(vsigma_grad_rho_lm));
117  /* add remaining term to Vxc */
118  vxc_tp -= 2.0 * div_vsigma_grad_rho_tp;
119  }
120  }
121  exc_lm__ += transform(sht__, exc_tp);
122  vxc_lm__ += transform(sht__, vxc_tp);
123  } //ixc
124 }

126 void xc_mt_magnetic(Radial_grid<double> const& rgrid__, SHT const& sht__, int num_mag_dims__,
127  std::vector<XC_functional*> xc_func__, std::vector<Ftp> const& rho_tp__,
128  std::vector<Flm*> vxc__, Flm& exc__)
129 {
130  bool is_gga{false};
131  for (auto& ixc : xc_func__) {
132  if (ixc->is_gga() || ixc->is_vdw()) {
133  is_gga = true;
134  }
135  }
136 
137  Ftp exc_tp(sht__.num_points(), rgrid__);
138  Ftp vxc_tp(sht__.num_points(), rgrid__);
139 
140  /* convert to rho_up, rho_dn */
141  Ftp rho_dn_tp(sht__.num_points(), rgrid__);
142  Ftp rho_up_tp(sht__.num_points(), rgrid__);
143  /* loop over radial grid points */
144  for (int ir = 0; ir < rgrid__.num_points(); ir++) {
145  /* loop over points on the sphere */
146  for (int itp = 0; itp < sht__.num_points(); itp++) {
147  vector3d<double> m;
148  for (int j = 0; j < num_mag_dims__; j++) {
149  m[j] = rho_tp__[1 + j](itp, ir);
150  }
151  auto rud = get_rho_up_dn(num_mag_dims__, rho_tp__[0](itp, ir), m);
152 
153  /* compute "up" and "dn" components */
154  rho_up_tp(itp, ir) = rud.first;
155  rho_dn_tp(itp, ir) = rud.second;
156  }
157  }
158  /* transform from (theta, phi) to Rlm */
159  auto rho_up_lm = transform(sht__, rho_up_tp);
160  auto rho_dn_lm = transform(sht__, rho_dn_tp);
161 
162  std::vector<Ftp> bxc_tp(num_mag_dims__);
163 
164  Ftp vxc_up_tp(sht__.num_points(), rgrid__);
165  Ftp vxc_dn_tp(sht__.num_points(), rgrid__);
166  for (int j = 0; j < num_mag_dims__; j++) {
167  bxc_tp[j] = Ftp(sht__.num_points(), rgrid__);
168  }
169 
170  Ftp grad_rho_up_grad_rho_up_tp;
171  Ftp grad_rho_up_grad_rho_dn_tp;
172  Ftp grad_rho_dn_grad_rho_dn_tp;
173  Ftp vsigma_uu_tp;
174  Ftp vsigma_ud_tp;
175  Ftp vsigma_dd_tp;
176  Ftp lapl_rho_up_tp;
177  Ftp lapl_rho_dn_tp;
178  Spheric_vector_function<function_domain_t::spatial, double> grad_rho_up_tp;
179  Spheric_vector_function<function_domain_t::spatial, double> grad_rho_dn_tp;
180 
181  if (is_gga) {
182  /* compute gradient in Rlm spherical harmonics */
183  auto grad_rho_up_lm = gradient(rho_up_lm);
184  auto grad_rho_dn_lm = gradient(rho_dn_lm);
185  grad_rho_up_tp = Spheric_vector_function<function_domain_t::spatial, double>(sht__.num_points(), rgrid__);
186  grad_rho_dn_tp = Spheric_vector_function<function_domain_t::spatial, double>(sht__.num_points(), rgrid__);
187  /* backward transform gradient from Rlm to (theta, phi) */
188  for (int x = 0; x < 3; x++) {
189  transform(sht__, grad_rho_up_lm[x], grad_rho_up_tp[x]);
190  transform(sht__, grad_rho_dn_lm[x], grad_rho_dn_tp[x]);
191  }
192  /* compute density gradient products */
193  grad_rho_up_grad_rho_up_tp = grad_rho_up_tp * grad_rho_up_tp;
194  grad_rho_up_grad_rho_dn_tp = grad_rho_up_tp * grad_rho_dn_tp;
195  grad_rho_dn_grad_rho_dn_tp = grad_rho_dn_tp * grad_rho_dn_tp;
196 
197  vsigma_uu_tp = Ftp(sht__.num_points(), rgrid__);
198  vsigma_ud_tp = Ftp(sht__.num_points(), rgrid__);
199  vsigma_dd_tp = Ftp(sht__.num_points(), rgrid__);
200 
201  /* backward transform Laplacians from Rlm to (theta, phi) */
202  lapl_rho_up_tp = transform(sht__, laplacian(rho_up_lm));
203  lapl_rho_dn_tp = transform(sht__, laplacian(rho_dn_lm));
204  }
205 
206  for (auto& ixc: xc_func__) {
207  if (ixc->is_lda()) {
208  ixc->get_lda(sht__.num_points() * rgrid__.num_points(), rho_up_tp.at(memory_t::host),
209  rho_dn_tp.at(memory_t::host), vxc_up_tp.at(memory_t::host), vxc_dn_tp.at(memory_t::host),
210  exc_tp.at(memory_t::host));
211  }
212  if (ixc->is_gga()) {
213  /* get the vrho and vsigma */
214  ixc->get_gga(sht__.num_points() * rgrid__.num_points(), rho_up_tp.at(memory_t::host),
215  rho_dn_tp.at(memory_t::host), grad_rho_up_grad_rho_up_tp.at(memory_t::host),
216  grad_rho_up_grad_rho_dn_tp.at(memory_t::host), grad_rho_dn_grad_rho_dn_tp.at(memory_t::host),
217  vxc_up_tp.at(memory_t::host), vxc_dn_tp.at(memory_t::host), vsigma_uu_tp.at(memory_t::host),
218  vsigma_ud_tp.at(memory_t::host), vsigma_dd_tp.at(memory_t::host), exc_tp.at(memory_t::host));
219 
220  /* directly add to Vxc available contributions */
221  vxc_up_tp -= (2.0 * vsigma_uu_tp * lapl_rho_up_tp + vsigma_ud_tp * lapl_rho_dn_tp);
222  vxc_dn_tp -= (2.0 * vsigma_dd_tp * lapl_rho_dn_tp + vsigma_ud_tp * lapl_rho_up_tp);
223 
224  /* forward transform vsigma to Rlm */
225  auto vsigma_uu_lm = transform(sht__, vsigma_uu_tp);
226  auto vsigma_ud_lm = transform(sht__, vsigma_ud_tp);
227  auto vsigma_dd_lm = transform(sht__, vsigma_dd_tp);
228 
229  /* compute gradient of vsgima in spherical harmonics */
230  auto grad_vsigma_uu_lm = gradient(vsigma_uu_lm);
231  auto grad_vsigma_ud_lm = gradient(vsigma_ud_lm);
232  auto grad_vsigma_dd_lm = gradient(vsigma_dd_lm);
233 
234  /* backward transform gradient from Rlm to (theta, phi) */
235  Spheric_vector_function<function_domain_t::spatial, double> grad_vsigma_uu_tp(sht__.num_points(), rgrid__);
236  Spheric_vector_function<function_domain_t::spatial, double> grad_vsigma_ud_tp(sht__.num_points(), rgrid__);
237  Spheric_vector_function<function_domain_t::spatial, double> grad_vsigma_dd_tp(sht__.num_points(), rgrid__);
238  for (int x = 0; x < 3; x++) {
239  grad_vsigma_uu_tp[x] = transform(sht__, grad_vsigma_uu_lm[x]);
240  grad_vsigma_ud_tp[x] = transform(sht__, grad_vsigma_ud_lm[x]);
241  grad_vsigma_dd_tp[x] = transform(sht__, grad_vsigma_dd_lm[x]);
242  }
243 
244  /* compute scalar product of two gradients */
245  auto grad_vsigma_uu_grad_rho_up_tp = grad_vsigma_uu_tp * grad_rho_up_tp;
246  auto grad_vsigma_dd_grad_rho_dn_tp = grad_vsigma_dd_tp * grad_rho_dn_tp;
247  auto grad_vsigma_ud_grad_rho_up_tp = grad_vsigma_ud_tp * grad_rho_up_tp;
248  auto grad_vsigma_ud_grad_rho_dn_tp = grad_vsigma_ud_tp * grad_rho_dn_tp;
249 
250  /* add remaining terms to Vxc */
251  vxc_up_tp -= (2.0 * grad_vsigma_uu_grad_rho_up_tp + grad_vsigma_ud_grad_rho_dn_tp);
252  vxc_dn_tp -= (2.0 * grad_vsigma_dd_grad_rho_dn_tp + grad_vsigma_ud_grad_rho_up_tp);
253  }
254  /* genertate magnetic filed and effective potential inside MT sphere */
255  for (int ir = 0; ir < rgrid__.num_points(); ir++) {
256  for (int itp = 0; itp < sht__.num_points(); itp++) {
257  /* Vxc = 0.5 * (V_up + V_dn) */
258  vxc_tp(itp, ir) = 0.5 * (vxc_up_tp(itp, ir) + vxc_dn_tp(itp, ir));
259  /* Bxc = 0.5 * (V_up - V_dn) */
260  double bxc = 0.5 * (vxc_up_tp(itp, ir) - vxc_dn_tp(itp, ir));
261  /* get the sign between mag and B */
262  auto s = utils::sign((rho_up_tp(itp, ir) - rho_dn_tp(itp, ir)) * bxc);
263 
264  vector3d<double> m;
265  for (int j = 0; j < num_mag_dims__; j++) {
266  m[j] = rho_tp__[1 + j](itp, ir);
267  }
268  auto m_len = m.length();
269  if (m_len > 1e-8) {
270  for (int j = 0; j < num_mag_dims__; j++) {
271  bxc_tp[j](itp, ir) = std::abs(bxc) * s * m[j] / m_len;
272  }
273  } else {
274  for (int j = 0; j < num_mag_dims__; j++) {
275  bxc_tp[j](itp, ir) = 0.0;
276  }
277  }
278  }
279  }
280  /* convert magnetic field back to Rlm */
281  for (int j = 0; j < num_mag_dims__; j++) {
282  *vxc__[j + 1] += transform(sht__, bxc_tp[j]);
283  }
284  /* forward transform from (theta, phi) to Rlm */
285  *vxc__[0] += transform(sht__, vxc_tp);
286  exc__ += transform(sht__, exc_tp);
287  } // ixc
288 }

290 void xc_mt(Radial_grid<double> const& rgrid__, SHT const& sht__, std::vector<XC_functional*> xc_func__,
291  int num_mag_dims__, std::vector<Flm const*> rho__, std::vector<Flm*> vxc__, Flm* exc__)
292 {
293  /* zero the fields */
294  exc__->zero();
295  for (int j = 0; j < num_mag_dims__ + 1; j++) {
296  vxc__[j]->zero();
297  }
298 
299  std::vector<Ftp> rho_tp(num_mag_dims__ + 1);
300  for (int j = 0; j < num_mag_dims__ + 1; j++) {
301  /* convert density and magnetization to theta, phi */
302  rho_tp[j] = transform(sht__, *rho__[j]);
303  }
304 
305  /* check if density has negative values */
306  double rhomin{0};
307  for (int ir = 0; ir < rgrid__.num_points(); ir++) {
308  for (int itp = 0; itp < sht__.num_points(); itp++) {
309  rhomin = std::min(rhomin, rho_tp[0](itp, ir));
310  /* fix negative density */
311  if (rho_tp[0](itp, ir) < 0.0) {
312  rho_tp[0](itp, ir) = 0.0;
313  }
314  }
315  }
316 
317  if (rhomin < 0.0) {
318  std::stringstream s;
319  s << "[xc_mt] negative charge density: " << rhomin << std::endl
320  << " current Rlm expansion of the charge density may be not sufficient, try to increase lmax";
321  WARNING(s);
322  }
323 
324  if (num_mag_dims__ == 0) {
325  xc_mt_nonmagnetic(rgrid__, sht__, xc_func__, *rho__[0], rho_tp[0], *vxc__[0], *exc__);
326  } else {
327  xc_mt_magnetic(rgrid__, sht__, num_mag_dims__, xc_func__, rho_tp, vxc__, *exc__);
328  }

331 void Potential::xc_mt(Density const& density__)
332 {
333  PROFILE("sirius::Potential::xc_mt");
334 
335  #pragma omp parallel for
336  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
337  int ia = unit_cell_.spl_num_atoms(ialoc);
338  auto& rgrid = unit_cell_.atom(ia).radial_grid();
339  std::vector<Flm const*> rho(ctx_.num_mag_dims() + 1);

342  vxc[0] = &xc_potential_->f_mt(ialoc);
343  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
344  rho[j + 1] = &density__.magnetization(j).f_mt(ialoc);
345  vxc[j + 1] = &effective_magnetic_field(j).f_mt(ialoc);
346  }
347  sirius::xc_mt(rgrid, *sht_, xc_func_, ctx_.num_mag_dims(), rho, vxc, &xc_energy_density_->f_mt(ialoc));
348 
349  /* z, x, y order */
350  std::array<int, 3> comp_map = {2, 0, 1};
351  /* add auxiliary magnetic field antiparallel to starting magnetization */
352  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
353  for (int ir = 0; ir < rgrid.num_points(); ir++) {
354  effective_magnetic_field(j).f_mt<index_domain_t::local>(0, ir, ialoc) -=
355  aux_bf_(j, ia) * ctx_.unit_cell().atom(ia).vector_field()[comp_map[j]];
356  }
357  }
