25 inline void K_point::generate_spinor_wave_functions()
26 {
27  PROFILE("sirius::K_point::generate_spinor_wave_functions");
28 
29  if (ctx_.control().use_second_variation_) {
30  int nfv = ctx_.num_fv_states();
31 
32  if (!ctx_.need_sv()) {
33  /* copy eigen-states and exit */
34  spinor_wave_functions().copy_from(device_t::CPU, ctx_.num_fv_states(), fv_states(), 0, 0, 0, 0);
35  return;
36  }
37 
38  int nbnd = (ctx_.num_mag_dims() == 3) ? ctx_.num_bands() : nfv;
39 
40  if (ctx_.processing_unit() == device_t::GPU) {
41  fv_states().allocate(spin_idx(0), memory_t::device);
42  fv_states().copy_to(0, memory_t::device, 0, nfv);
43  sv_eigen_vectors_[0].allocate(memory_t::device).copy_to(memory_t::device);
44  if (ctx_.num_mag_dims() == 1) {
45  sv_eigen_vectors_[1].allocate(memory_t::device).copy_to(memory_t::device);
46  }
47  if (is_device_memory(ctx_.preferred_memory_t())) {
48  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
49  spinor_wave_functions().allocate(spin_idx(ispn), memory_t::device);
50  spinor_wave_functions().copy_to(ispn, memory_t::device, 0, nbnd);
51  }
52  }
53  }
54 
55  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
56  int s, o;
57 
58  if (ctx_.num_mag_dims() == 3) {
59  /* in case of non-collinear magnetism sv_eigen_vectors is a single 2Nx2N matrix */
60  s = 0;
61  o = ispn * nfv; /* offset for spin up is 0, for spin dn is nfv */
62  } else {
63  /* sv_eigen_vectors is composed of two NxN matrices */
64  s = ispn;
65  o = 0;
66  }
67  /* multiply consecutively up and dn blocks */
68  transform(ctx_.processing_unit(), ispn, fv_states(), 0, nfv, sv_eigen_vectors_[s], o, 0, spinor_wave_functions(), 0, nbnd);
69  }
70 
71  if (ctx_.processing_unit() == device_t::GPU) {
72  fv_states().deallocate(0, memory_t::device);
73  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
74  spinor_wave_functions().copy_to(ispn, memory_t::host, 0, nbnd);
75  }
76  sv_eigen_vectors_[0].deallocate(memory_t::device);
77  if (ctx_.num_mag_dims() == 3) {
78  sv_eigen_vectors_[1].deallocate(memory_t::device);
79  }
80  }
81  } else {
82  TERMINATE_NOT_IMPLEMENTED;
83  }
84 }
