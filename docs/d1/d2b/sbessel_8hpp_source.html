32 /// Spherical Bessel functions \f$ j_{\ell}(q x) \f$ up to lmax.
33 class Spherical_Bessel_functions

49  Spherical_Bessel_functions(int lmax__, Radial_grid<double> const& rgrid__, double q__)
50  : lmax_(lmax__)
51  , q_(q__)
52  , rgrid_(&rgrid__)
53  {
54  assert(q_ >= 0);
55 
56  sbessel_ = std::vector<Spline<double>>(lmax__ + 2);
57  for (int l = 0; l <= lmax__ + 1; l++) {
58  sbessel_[l] = Spline<double>(rgrid__);
59  }
60 
61  std::vector<double> jl(lmax__ + 2);
62  for (int ir = 0; ir < rgrid__.num_points(); ir++) {
63  double t = rgrid__[ir] * q__;
64  gsl_sf_bessel_jl_array(lmax__ + 1, t, &jl[0]);
65  for (int l = 0; l <= lmax__ + 1; l++) {
66  sbessel_[l](ir) = jl[l];
67  }
68  }
69 
70  for (int l = 0; l <= lmax__ + 1; l++) {
71  sbessel_[l].interpolate();
72  }
73  }

80  static void sbessel_deriv_q(int lmax__, double q__, double x__, double* jl_dq__)
81  {
82  std::vector<double> jl(lmax__ + 2);
83  sbessel(lmax__ + 1, x__ * q__, &jl[0]);
84 
85  for (int l = 0; l <= lmax__; l++) {
86  if (q__ != 0) {
87  jl_dq__[l] = (l / q__) * jl[l] - x__ * jl[l + 1];
88  } else {
89  if (l == 1) {
90  jl_dq__[l] = x__ / 3;
91  } else {
92  jl_dq__[l] = 0;
93  }
94  }
95  }
96  }

104  /// Derivative of Bessel function with respect to q.
105  /** \f[
106  * \frac{\partial j_{\ell}(q x)}{\partial q} = \frac{\ell}{q} j_{\ell}(q x) - x j_{\ell+1}(q x)
107  * \f]
108  */
109  Spline<double> deriv_q(int l__)
110  {
111  assert(l__ <= lmax_);
112  assert(q_ >= 0);
113  Spline<double> s(*rgrid_);
114  if (q_ != 0) {
115  for (int ir = 0; ir < rgrid_->num_points(); ir++) {
116  s(ir) = (l__ / q_) * sbessel_[l__](ir) - (*rgrid_)[ir] * sbessel_[l__ + 1](ir);
117  }
118  } else {
119  if (l__ == 1) {
120  for (int ir = 0; ir < rgrid_->num_points(); ir++) {
121  s(ir) = (*rgrid_)[ir] / 3;
122  }
123  }
124  }
125  s.interpolate();
126  return std::move(s);
127  }
