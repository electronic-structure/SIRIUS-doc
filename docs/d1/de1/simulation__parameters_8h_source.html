72  /// Import parameters from a file or a serialized json string.
73  void import(std::string const& str__)
74  {
75  PROFILE("sirius::Simulation_parameters::import");
76 
77  if (str__.size() == 0) {
78  return;
79  }
80 
81  json dict = utils::read_json_from_file_or_string(str__);
82 
83  /* read unit cell */
84  unit_cell_input_.read(dict);
85  /* read parameters of mixer */
86  mixer_input_.read(dict);
87  /* read parameters of iterative solver */
88  iterative_solver_input_.read(dict);
89  /* read controls */
90  control_input_.read(dict);
91  /* read parameters */
92  parameters_input_.read(dict);
93  /* read settings */
94  settings_input_.read(dict);
95  /* read hubbard parameters */
96  hubbard_input_.read(dict);
97  }

320  /// Number of spin components.
321  /** This parameter can take only two values: 1 -- non-magnetic calcaulation and wave-functions,

325  return (num_mag_dims() == 0) ? 1 : 2;
