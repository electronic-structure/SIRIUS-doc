272 /// Unit step function is defined to be 1 in the interstitial and 0 inside muffin-tins.
273 /** Unit step function is constructed from it's plane-wave expansion coefficients which are computed
274 * analytically:
275 * \f[
276 * \Theta({\bf r}) = \sum_{\bf G} \Theta({\bf G}) e^{i{\bf Gr}},
277 * \f]
278 * where
279 * \f[
280 * \Theta({\bf G}) = \frac{1}{\Omega} \int \Theta({\bf r}) e^{-i{\bf Gr}} d{\bf r} =
281 * \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} d{\bf r} - \frac{1}{\Omega} \int_{MT} e^{-i{\bf Gr}}
282 * d{\bf r} = \delta_{\bf G, 0} - \sum_{\alpha} \frac{1}{\Omega} \int_{MT_{\alpha}} e^{-i{\bf Gr}}
283 * d{\bf r}
284 * \f]
285 * Integralof a plane-wave over the muffin-tin volume is taken using the spherical expansion of the
286 * plane-wave around central point \f$ \tau_{\alpha} \f$:
287 * \f[ \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = e^{-i{\bf G\tau_{\alpha}}}
288 * \int_{MT_{\alpha}} 4\pi \sum_{\ell m} (-i)^{\ell} j_{\ell}(Gr) Y_{\ell m}(\hat {\bf G}) Y_{\ell m}^{*}(\hat
289 * {\bf r}) r^2 \sin \theta dr d\phi d\theta
290 * \f]
291 * In the above integral only \f$ \ell=m=0 \f$ term survives. So we have:
292 * \f[
293 * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = 4\pi e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
294 * \f]
295 * where
296 * \f[
297 * \Theta(\alpha, G) = \int_{0}^{R_{\alpha}} \frac{\sin(Gr)}{Gr} r^2 dr =
298 * \left\{ \begin{array}{ll} \displaystyle R_{\alpha}^3 / 3 & G=0 \\
299 * \Big( \sin(GR_{\alpha}) - GR_{\alpha}\cos(GR_{\alpha}) \Big) / G^3 & G \ne 0 \end{array} \right.
300 * \f]
301 * are the so-called step function form factors. With this we have a final expression for the plane-wave
302 * coefficients of the unit step function:
303 * \f[ \Theta({\bf G}) = \delta_{\bf G, 0} - \sum_{\alpha}
304 * \frac{4\pi}{\Omega} e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
305 * \f]
306 */
307 void init_step_function();

555 /// Make periodic function out of form factors.
556 /** Return vector of plane-wave coefficients */ // TODO: return mdarray
557 template <index_domain_t index_domain, typename F>
558 inline std::vector<double_complex>
559 make_periodic_function(F&& form_factors__) const
560 {
561 PROFILE("sirius::Simulation_context::make_periodic_function");
562
563 double fourpi_omega = fourpi / unit_cell().omega();
564
565 int ngv = (index_domain == index_domain_t::local) ? gvec().count() : gvec().num_gvec();
566 std::vector<double_complex> f_pw(ngv, double_complex(0, 0));
567
568 #pragma omp parallel for schedule(static)
569 for (int igloc = 0; igloc < gvec().count(); igloc++) {
570 /* global index of G-vector */
571 int ig = gvec().offset() + igloc;
572 double g = gvec().gvec_len(ig);
573
574 int j = (index_domain == index_domain_t::local) ? igloc : ig;
575 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
576 f_pw[j] += fourpi_omega * std::conj(phase_factors_t_(igloc, iat)) * form_factors__(iat, g);
577 }
578 }
579
580 if (index_domain == index_domain_t::global) {
581 comm_.allgather(&f_pw[0], gvec().count(), gvec().offset());
582 }
583
584 return f_pw;
585 }

587 /// Make periodic out of form factors computed for G-shells.
588 template <index_domain_t index_domain>
589 inline std::vector<double_complex>
590 make_periodic_function(sddk::mdarray<double, 2>& form_factors__) const
591 {
592 PROFILE("sirius::Simulation_context::make_periodic_function");
593
594 double fourpi_omega = fourpi / unit_cell().omega();
595
596 int ngv = (index_domain == index_domain_t::local) ? gvec().count() : gvec().num_gvec();
597 std::vector<double_complex> f_pw(ngv, double_complex(0, 0));
598
599 #pragma omp parallel for schedule(static)
600 for (int igloc = 0; igloc < gvec().count(); igloc++) {
601 /* global index of G-vector */
602 int ig = gvec().offset() + igloc;
603 int igsh = gvec().shell(ig);
604
605 int j = (index_domain == index_domain_t::local) ? igloc : ig;
606 for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
607 f_pw[j] += fourpi_omega * std::conj(phase_factors_t_(igloc, iat)) * form_factors__(igsh, iat);
608 }
609 }
610
611 if (index_domain == index_domain_t::global) {
612 comm_.allgather(&f_pw[0], gvec().count(), gvec().offset());
613 }
614
615 return f_pw;
616 }
