261  /// Unit step function is defined to be 1 in the interstitial and 0 inside muffin-tins.
262  /** Unit step function is constructed from it's plane-wave expansion coefficients which are computed
263  * analytically:
264  * \f[
265  * \Theta({\bf r}) = \sum_{\bf G} \Theta({\bf G}) e^{i{\bf Gr}},
266  * \f]
267  * where
268  * \f[
269  * \Theta({\bf G}) = \frac{1}{\Omega} \int \Theta({\bf r}) e^{-i{\bf Gr}} d{\bf r} =
270  * \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} d{\bf r} - \frac{1}{\Omega} \int_{MT} e^{-i{\bf Gr}}
271  * d{\bf r} = \delta_{\bf G, 0} - \sum_{\alpha} \frac{1}{\Omega} \int_{MT_{\alpha}} e^{-i{\bf Gr}}
272  * d{\bf r}
273  * \f]
274  * Integralof a plane-wave over the muffin-tin volume is taken using the spherical expansion of the
275  * plane-wave around central point \f$ \tau_{\alpha} \f$:
276  * \f[ \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = e^{-i{\bf G\tau_{\alpha}}}
277  * \int_{MT_{\alpha}} 4\pi \sum_{\ell m} (-i)^{\ell} j_{\ell}(Gr) Y_{\ell m}(\hat {\bf G}) Y_{\ell m}^{*}(\hat
278  * {\bf r}) r^2 \sin \theta dr d\phi d\theta
279  * \f]
280  * In the above integral only \f$ \ell=m=0 \f$ term survives. So we have:
281  * \f[
282  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = 4\pi e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
283  * \f]
284  * where
285  * \f[
286  * \Theta(\alpha, G) = \int_{0}^{R_{\alpha}} \frac{\sin(Gr)}{Gr} r^2 dr =
287  * \left\{ \begin{array}{ll} \displaystyle R_{\alpha}^3 / 3 & G=0 \\
288  * \Big( \sin(GR_{\alpha}) - GR_{\alpha}\cos(GR_{\alpha}) \Big) / G^3 & G \ne 0 \end{array} \right.
289  * \f]
290  * are the so-called step function form factors. With this we have a final expression for the plane-wave
291  * coefficients of the unit step function:
292  * \f[ \Theta({\bf G}) = \delta_{\bf G, 0} - \sum_{\alpha}
293  * \frac{4\pi}{\Omega} e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
294  * \f]
295  */
296  void init_step_function();

526  /// Make periodic function out of form factors.
527  /** Return vector of plane-wave coefficients */ // TODO: return mdarray
528  template <index_domain_t index_domain, typename F>
529  inline std::vector<double_complex>
530  make_periodic_function(F&& form_factors__) const
531  {
532  PROFILE("sirius::Simulation_context::make_periodic_function");
533 
534  double fourpi_omega = fourpi / unit_cell().omega();
535 
536  int ngv = (index_domain == index_domain_t::local) ? gvec().count() : gvec().num_gvec();
537  std::vector<double_complex> f_pw(ngv, double_complex(0, 0));
538 
539  #pragma omp parallel for schedule(static)
540  for (int igloc = 0; igloc < gvec().count(); igloc++) {
541  /* global index of G-vector */
542  int ig = gvec().offset() + igloc;
543  double g = gvec().gvec_len(ig);
544 
545  int j = (index_domain == index_domain_t::local) ? igloc : ig;
546  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
547  f_pw[j] += fourpi_omega * std::conj(phase_factors_t_(igloc, iat)) * form_factors__(iat, g);
548  }
549  }
550 
551  if (index_domain == index_domain_t::global) {
552  comm_.allgather(&f_pw[0], gvec().count(), gvec().offset());
553  }
554 
555  return f_pw;
556  }
