143 void Potential::update()
144 {
145  PROFILE("sirius::Potential::update");
146 
147  if (!ctx_.full_potential()) {
148  local_potential_->zero();
149  generate_local_potential();
150  } else {
151  gvec_ylm_ = ctx_.generate_gvec_ylm(ctx_.lmax_pot());
152  sbessel_mt_ = ctx_.generate_sbessel_mt(lmax_ + pseudo_density_order_ + 1);
153 
154  /* compute moments of spherical Bessel functions
155  *
156  * In[]:= Integrate[SphericalBesselJ[l,G*x]*x^(2+l),{x,0,R},Assumptions->{R>0,G>0,l>=0}]
157  * Out[]= (Sqrt[\[Pi]/2] R^(3/2+l) BesselJ[3/2+l,G R])/G^(3/2)
158  *
159  * and use relation between Bessel and spherical Bessel functions:
160  * Subscript[j, n](z)=Sqrt[\[Pi]/2]/Sqrt[z]Subscript[J, n+1/2](z) */
161  sbessel_mom_ = mdarray<double, 3>(ctx_.lmax_rho() + 1,
162  ctx_.gvec().count(),
163  unit_cell_.num_atom_types(),
164  memory_t::host, "sbessel_mom_");
165  sbessel_mom_.zero();
166  int ig0{0};
167  if (ctx_.comm().rank() == 0) {
168  /* for |G| = 0 */
169  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
170  sbessel_mom_(0, 0, iat) = std::pow(unit_cell_.atom_type(iat).mt_radius(), 3) / 3.0;
171  }
172  ig0 = 1;
173  }
174  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
175  #pragma omp parallel for schedule(static)
176  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
177  auto len = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc).length();
178  for (int l = 0; l <= ctx_.lmax_rho(); l++) {
179  sbessel_mom_(l, igloc, iat) = std::pow(unit_cell_.atom_type(iat).mt_radius(), l + 2) *
180  sbessel_mt_(l + 1, igloc, iat) / len;
181  }
182  }
183  }
184 
185  /* compute Gamma[5/2 + n + l] / Gamma[3/2 + l] / R^l
186  *
187  * use Gamma[1/2 + p] = (2p - 1)!!/2^p Sqrt[Pi] */
188  gamma_factors_R_ = mdarray<double, 2>(ctx_.lmax_rho() + 1, unit_cell_.num_atom_types(), memory_t::host, "gamma_factors_R_");
189  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
190  for (int l = 0; l <= ctx_.lmax_rho(); l++) {
191  long double Rl = std::pow(unit_cell_.atom_type(iat).mt_radius(), l);
192 
193  int n_min = (2 * l + 3);
194  int n_max = (2 * l + 1) + (2 * pseudo_density_order_ + 2);
195  /* split factorial product into two parts to avoid overflow */
196  long double f1 = 1.0;
197  long double f2 = 1.0;
198  for (int n = n_min; n <= n_max; n += 2) {
199  if (f1 < Rl) {
200  f1 *= (n / 2.0);
201  } else {
202  f2 *= (n / 2.0);
203  }
204  }
205  gamma_factors_R_(l, iat) = static_cast<double>((f1 / Rl) * f2);
206  }
207  }
208  }
209 
210  // VDWXC depends on unit cell, which might have changed.
211  for (auto& xc : xc_func_) {
212  if (xc->is_vdw()) {
213  xc->vdw_update_unit_cell(ctx_.spfft(), ctx_.unit_cell().lattice_vectors());
214  }
215  }

238 void Potential::generate(Density const& density__, bool use_symmetry__, bool transform_to_rg__)
239 {
240  PROFILE("sirius::Potential::generate");
241 
242  if (!ctx_.full_potential()) {
243  /* save current effective potential */
244  for (size_t ig = 0; ig < effective_potential().f_pw_local().size(); ig++) {
245  dveff_->f_pw_local(ig) = effective_potential().f_pw_local(ig);
246  }
247  }
248 
249  /* zero effective potential and magnetic field */
250  zero();
251 
252  auto veff_callback = ctx_.veff_callback();
253  if (veff_callback) {
254  veff_callback();
255  //if (!ctx_.full_potential()) {
256  // /* add local ionic potential to the effective potential */
257  // effective_potential().add(local_potential());
258  //}
259  /* transform to real space */
260  //fft_transform(1);
261  } else {
262  /* solve Poisson equation */
263  poisson(density__.rho());
264 
265  /* add Hartree potential to the total potential */
266  effective_potential().add(hartree_potential());
267 
268  if (ctx_.cfg().control().print_hash()) {
269  auto h = effective_potential().hash_f_rg();
270  if (ctx_.comm().rank() == 0) {
271  utils::print_hash("Vha", h);
272  }
273  }
274 
275  if (ctx_.full_potential()) {
276  xc(density__);
277  } else {
278  /* add local ionic potential to the effective potential */
279  effective_potential().add(local_potential());
280  /* construct XC potentials from rho + rho_core */
281  xc<true>(density__);
282  }
283  /* add XC potential to the effective potential */
284  effective_potential().add(xc_potential());
285 
286  if (ctx_.cfg().control().print_hash()) {
287  auto h = effective_potential().hash_f_rg();
288  if (ctx_.comm().rank() == 0) {
289  utils::print_hash("Vha+Vxc", h);
290  }
291  }
292 
293  if (ctx_.full_potential()) {
294  effective_potential().sync_mt();
295  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
296  effective_magnetic_field(j).sync_mt();
297  }
298  }
299 
300  /* get plane-wave coefficients of effective potential;
301  * they will be used in three places:
302  * 1) compute D-matrix
303  * 2) establish a mapping between fine and coarse FFT grid for the Hloc operator
304  * 3) symmetrize effective potential */
305  fft_transform(-1);
306  }
307 
308  if (use_symmetry__) {
309  /* symmetrize potential and effective magnetic field */
310  this->symmetrize();
311  if (transform_to_rg__) {
312  /* transform potential to real space after symmetrization */
313  this->fft_transform(1);
314  }
315  }
316 
317  if (!ctx_.full_potential()) {
318  /* this is needed later to compute scf correction to forces */
319  for (size_t ig = 0; ig < effective_potential().f_pw_local().size(); ig++) {
320  dveff_->f_pw_local(ig) = effective_potential().f_pw_local(ig) - dveff_->f_pw_local(ig);
321  }
322  }
323 
324  if (ctx_.cfg().control().print_hash()) {
325  auto h = effective_potential().hash_f_pw();
326  if (ctx_.comm().rank() == 0) {
327  utils::print_hash("V(G)", h);
328  }
329  }
330 
331  if (!ctx_.full_potential()) {
332  generate_D_operator_matrix();
333  generate_PAW_effective_potential(density__);
334  }
335 
336  if (ctx_.hubbard_correction()) {
337  this->U().generate_potential(density__.occupation_matrix());
338  }
339 
340  if (ctx_.cfg().parameters().reduce_aux_bf() > 0 && ctx_.cfg().parameters().reduce_aux_bf() < 1) {
341  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
342  for (int x : {0, 1, 2}) {
343  aux_bf_(x, ia) *= ctx_.cfg().parameters().reduce_aux_bf();
344  }
345  }
346  }
347 }
