30Potential::Potential(Simulation_context& ctx__)
31 : Field4D(ctx__, ctx__.lmmax_pot())
32 , unit_cell_(ctx__.unit_cell())
33 , comm_(ctx__.comm())
34 , hubbard_potential_(ctx__)
35{
36 PROFILE("sirius::Potential");
37
38 if (!ctx_.initialized()) {
39 TERMINATE("Simulation_context is not initialized");
40 }
41
42 lmax_ = std::max(ctx_.lmax_rho(), ctx_.lmax_pot());
43
44 if (lmax_ >= 0) {
45 sht_ = std::unique_ptr<SHT>(new SHT(ctx_.processing_unit(), lmax_, ctx_.cfg().settings().sht_coverage()));
46 if (ctx_.cfg().control().verification() >= 1) {
47 sht_->check();
48 }
49 l_by_lm_ = utils::l_by_lm(lmax_);
50
51 /* precompute i^l */
52 zil_.resize(lmax_ + 1);
53 for (int l = 0; l <= lmax_; l++) {
54 zil_[l] = std::pow(double_complex(0, 1), l);
55 }
56
57 zilm_.resize(utils::lmmax(lmax_));
58 for (int l = 0, lm = 0; l <= lmax_; l++) {
59 for (int m = -l; m <= l; m++, lm++) {
60 zilm_[lm] = zil_[l];
61 }
62 }
63 }
64
65 if (ctx_.full_potential()) {
66 using gc_z = Gaunt_coefficients<double_complex>;
67 gaunt_coefs_ = std::unique_ptr<gc_z>(new gc_z(ctx_.lmax_apw(), ctx_.lmax_pot(), ctx_.lmax_apw(), SHT::gaunt_hybrid));
68 }
69
70 /* create list of XC functionals */
71 for (auto& xc_label : ctx_.xc_functionals()) {
72 xc_func_.emplace_back(XC_functional(ctx_.spfft<double>(), ctx_.unit_cell().lattice_vectors(), xc_label, ctx_.num_spins()));
73 if (ctx_.cfg().parameters().xc_dens_tre() > 0) {
74 xc_func_.back().set_dens_threshold(ctx_.cfg().parameters().xc_dens_tre());
75 }
76 }
77
78 using pf = Periodic_function<double>;
79 using spf = Smooth_periodic_function<double>;
80
81 hartree_potential_ = std::unique_ptr<pf>(new pf(ctx_, ctx_.lmmax_pot()));
82 hartree_potential_->allocate_mt(false);
83
84 xc_potential_ = std::unique_ptr<pf>(new pf(ctx_, ctx_.lmmax_pot()));
85 xc_potential_->allocate_mt(false);
86
87 xc_energy_density_ = std::unique_ptr<pf>(new pf(ctx_, ctx_.lmmax_pot()));
88 xc_energy_density_->allocate_mt(false);
89
90 if (this->is_gradient_correction()) {
91 int nsigma = (ctx_.num_spins() == 1) ? 1 : 3;
92 for (int i = 0; i < nsigma ; i++) {
93 vsigma_[i] = std::unique_ptr<spf>(new spf(ctx_.spfft<double>(), ctx_.gvec_partition()));
94 }
95 }
96
97 if (!ctx_.full_potential()) {
98 local_potential_ = std::unique_ptr<spf>(new spf(ctx_.spfft<double>(), ctx_.gvec_partition()));
99 dveff_ = std::unique_ptr<spf>(new spf(ctx_.spfft<double>(), ctx_.gvec_partition()));
100 dveff_->zero();
101 }
102
103 vh_el_ = mdarray<double, 1>(unit_cell_.num_atoms());
104
105 if (ctx_.full_potential()) {
106 gvec_ylm_ = mdarray<double_complex, 2>(ctx_.lmmax_pot(), ctx_.gvec().count(), memory_t::host, "gvec_ylm_");
107
108 switch (ctx_.valence_relativity()) {
109 case relativity_t::iora: {
110 rm2_inv_pw_ = mdarray<double_complex, 1>(ctx_.gvec().num_gvec());
111 }
112 case relativity_t::zora: {
113 rm_inv_pw_ = mdarray<double_complex, 1>(ctx_.gvec().num_gvec());
114 }
115 default: {
116 veff_pw_ = mdarray<double_complex, 1>(ctx_.gvec().num_gvec());
117 }
118 }
119 }
120
121 aux_bf_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
122 aux_bf_.zero();
123
124 if (ctx_.cfg().parameters().reduce_aux_bf() > 0 && ctx_.cfg().parameters().reduce_aux_bf() < 1) {
125 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
126 for (int x : {0, 1, 2}) {
127 aux_bf_(x, ia) = 1;
128 }
129 }
130 }
131
132 /* in case of PAW */
133 init_PAW();
134
135 if (ctx_.hubbard_correction()) {
136 U_ = std::unique_ptr<Hubbard>(new Hubbard(ctx_));
137 }
138
139 update();
140}

142void Potential::update()
143{
144 PROFILE("sirius::Potential::update");
145
146 if (!ctx_.full_potential()) {
147 local_potential_->zero();
148 generate_local_potential();
149 } else {
150 gvec_ylm_ = ctx_.generate_gvec_ylm(ctx_.lmax_pot());
151 sbessel_mt_ = ctx_.generate_sbessel_mt(lmax_ + pseudo_density_order_ + 1);
152
153 /* compute moments of spherical Bessel functions
154 *
155 * In[]:= Integrate[SphericalBesselJ[l,G*x]*x^(2+l),{x,0,R},Assumptions->{R>0,G>0,l>=0}]
156 * Out[]= (Sqrt[\[Pi]/2] R^(3/2+l) BesselJ[3/2+l,G R])/G^(3/2)
157 *
158 * and use relation between Bessel and spherical Bessel functions:
159 * Subscript[j, n](z)=Sqrt[\[Pi]/2]/Sqrt[z]Subscript[J, n+1/2](z) */
160 sbessel_mom_ = mdarray<double, 3>(ctx_.lmax_rho() + 1,
161 ctx_.gvec().count(),
162 unit_cell_.num_atom_types(),
163 memory_t::host, "sbessel_mom_");
164 sbessel_mom_.zero();
165 int ig0{0};
166 if (ctx_.comm().rank() == 0) {
167 /* for |G| = 0 */
168 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
169 sbessel_mom_(0, 0, iat) = std::pow(unit_cell_.atom_type(iat).mt_radius(), 3) / 3.0;
170 }
171 ig0 = 1;
172 }
173 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
174 #pragma omp parallel for schedule(static)
175 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
176 auto len = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc).length();
177 for (int l = 0; l <= ctx_.lmax_rho(); l++) {
178 sbessel_mom_(l, igloc, iat) = std::pow(unit_cell_.atom_type(iat).mt_radius(), l + 2) *
179 sbessel_mt_(l + 1, igloc, iat) / len;
180 }
181 }
182 }
183
184 /* compute Gamma[5/2 + n + l] / Gamma[3/2 + l] / R^l
185 *
186 * use Gamma[1/2 + p] = (2p - 1)!!/2^p Sqrt[Pi] */
187 gamma_factors_R_ = mdarray<double, 2>(ctx_.lmax_rho() + 1, unit_cell_.num_atom_types(), memory_t::host, "gamma_factors_R_");
188 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
189 for (int l = 0; l <= ctx_.lmax_rho(); l++) {
190 long double Rl = std::pow(unit_cell_.atom_type(iat).mt_radius(), l);
191
192 int n_min = (2 * l + 3);
193 int n_max = (2 * l + 1) + (2 * pseudo_density_order_ + 2);
194 /* split factorial product into two parts to avoid overflow */
195 long double f1 = 1.0;
196 long double f2 = 1.0;
197 for (int n = n_min; n <= n_max; n += 2) {
198 if (f1 < Rl) {
199 f1 *= (n / 2.0);
200 } else {
201 f2 *= (n / 2.0);
202 }
203 }
204 gamma_factors_R_(l, iat) = static_cast<double>((f1 / Rl) * f2);
205 }
206 }
207 }
208
209 // VDWXC depends on unit cell, which might have changed.
210 for (auto& xc : xc_func_) {
211 if (xc.is_vdw()) {
212 xc.vdw_update_unit_cell(ctx_.spfft<double>(), ctx_.unit_cell().lattice_vectors());
213 }
214 }

228void Potential::generate(Density const& density__, bool use_symmetry__, bool transform_to_rg__)
229{
230 PROFILE("sirius::Potential::generate");
231
232 if (!ctx_.full_potential()) {
233 /* save current effective potential */
234 for (size_t ig = 0; ig < effective_potential().f_pw_local().size(); ig++) {
235 dveff_->f_pw_local(ig) = effective_potential().f_pw_local(ig);
236 }
237 }
238
239 /* zero effective potential and magnetic field */
240 zero();
241
242 auto veff_callback = ctx_.veff_callback();
243 if (veff_callback) {
244 veff_callback();
245 //if (!ctx_.full_potential()) {
246 // /* add local ionic potential to the effective potential */
247 // effective_potential().add(local_potential());
248 //}
249 /* transform to real space */
250 //fft_transform(1);
251 } else {
252 /* solve Poisson equation */
253 poisson(density__.rho());
254
255 /* add Hartree potential to the total potential */
256 effective_potential().add(hartree_potential());
257
258 if (ctx_.cfg().control().print_hash()) {
259 auto h = effective_potential().hash_f_rg();
260 if (ctx_.comm().rank() == 0) {
261 utils::print_hash("Vha", h);
262 }
263 }
264
265 if (ctx_.full_potential()) {
266 xc(density__);
267 } else {
268 /* add local ionic potential to the effective potential */
269 effective_potential().add(local_potential());
270 /* construct XC potentials from rho + rho_core */
271 xc<true>(density__);
272 }
273 /* add XC potential to the effective potential */
274 effective_potential().add(xc_potential());
275
276 if (ctx_.cfg().control().print_hash()) {
277 auto h = effective_potential().hash_f_rg();
278 if (ctx_.comm().rank() == 0) {
279 utils::print_hash("Vha+Vxc", h);
280 }
281 }
282
283 if (ctx_.full_potential()) {
284 effective_potential().sync_mt();
285 for (int j = 0; j < ctx_.num_mag_dims(); j++) {
286 effective_magnetic_field(j).sync_mt();
287 }
288 }
289
290 /* get plane-wave coefficients of effective potential;
291 * they will be used in three places:
292 * 1) compute D-matrix
293 * 2) establish a mapping between fine and coarse FFT grid for the Hloc operator
294 * 3) symmetrize effective potential */
295 fft_transform(-1);
296 }
297
298 if (use_symmetry__) {
299 /* symmetrize potential and effective magnetic field */
300 this->symmetrize();
301 if (transform_to_rg__) {
302 /* transform potential to real space after symmetrization */
303 this->fft_transform(1);
304 }
305 }
306
307 if (!ctx_.full_potential()) {
308 /* this is needed later to compute scf correction to forces */
309 for (size_t ig = 0; ig < effective_potential().f_pw_local().size(); ig++) {
310 dveff_->f_pw_local(ig) = effective_potential().f_pw_local(ig) - dveff_->f_pw_local(ig);
311 }
312 }
313
314 if (ctx_.cfg().control().print_hash()) {
315 auto h = effective_potential().hash_f_pw();
316 if (ctx_.comm().rank() == 0) {
317 utils::print_hash("V(G)", h);
318 }
319 }
320
321 if (!ctx_.full_potential()) {
322 generate_D_operator_matrix();
323 generate_PAW_effective_potential(density__);
324 }
325
326 if (ctx_.hubbard_correction()) {
327 ::sirius::generate_potential(density__.occupation_matrix(), this->hubbard_potential());
328 }
329
330 if (ctx_.cfg().parameters().reduce_aux_bf() > 0 && ctx_.cfg().parameters().reduce_aux_bf() < 1) {
331 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
332 for (int x : {0, 1, 2}) {
333 aux_bf_(x, ia) *= ctx_.cfg().parameters().reduce_aux_bf();
334 }
335 }
336 }
337}

339void Potential::save()
340{
341 effective_potential().hdf5_write(storage_file_name, "effective_potential");
342 for (int j = 0; j < ctx_.num_mag_dims(); j++) {
343 std::stringstream s;
344 s << "effective_magnetic_field/" << j;
345 effective_magnetic_field(j).hdf5_write(storage_file_name, s.str());
346 }
347 if (ctx_.comm().rank() == 0 && !ctx_.full_potential()) {
348 HDF5_tree fout(storage_file_name, hdf5_access_t::read_write);
349 for (int j = 0; j < ctx_.unit_cell().num_atoms(); j++) {
350 if (ctx_.unit_cell().atom(j).mt_basis_size() != 0) {
351 fout["unit_cell"]["atoms"][j].write("D_operator", ctx_.unit_cell().atom(j).d_mtrx());
352 }
353 }
354 }
355 comm_.barrier();
356}

358void Potential::load()
359{
360 HDF5_tree fin(storage_file_name, hdf5_access_t::read_only);
361
362 int ngv;
363 fin.read("/parameters/num_gvec", &ngv, 1);
364 if (ngv != ctx_.gvec().num_gvec()) {
365 TERMINATE("wrong number of G-vectors");
366 }
367 mdarray<int, 2> gv(3, ngv);
368 fin.read("/parameters/gvec", gv);
369
370 effective_potential().hdf5_read(fin["effective_potential"], gv);
371
372 for (int j = 0; j < ctx_.num_mag_dims(); j++) {
373 effective_magnetic_field(j).hdf5_read(fin["effective_magnetic_field"][j], gv);
374 }
375
376 if (ctx_.full_potential()) {
377 update_atomic_potential();
378 }
379
380 if (!ctx_.full_potential()) {
381 HDF5_tree fout(storage_file_name, hdf5_access_t::read_only);
382 for (int j = 0; j < ctx_.unit_cell().num_atoms(); j++) {
383 fout["unit_cell"]["atoms"][j].read("D_operator", ctx_.unit_cell().atom(j).d_mtrx());
384 }
385 }

388void Potential::update_atomic_potential()
389{
390 for (int ic = 0; ic < unit_cell_.num_atom_symmetry_classes(); ic++) {
391 int ia = unit_cell_.atom_symmetry_class(ic).atom_id(0);
392 int nmtp = unit_cell_.atom(ia).num_mt_points();
393
394 std::vector<double> veff(nmtp);
395
396 for (int ir = 0; ir < nmtp; ir++) {
397 veff[ir] = y00 * effective_potential().f_mt<index_domain_t::global>(0, ir, ia);
398 }
399
400 unit_cell_.atom_symmetry_class(ic).set_spherical_potential(veff);
401 }
402
403 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
404 double* veff = &effective_potential().f_mt<index_domain_t::global>(0, 0, ia);
405
406 double* beff[] = {nullptr, nullptr, nullptr};
407 for (int i = 0; i < ctx_.num_mag_dims(); i++) {
408 beff[i] = &effective_magnetic_field(i).f_mt<index_domain_t::global>(0, 0, ia);
409 }
410
411 unit_cell_.atom(ia).set_nonspherical_potential(veff, beff);
412 }
413}
