34 /// MPI grid interface
35 /** The following terminology is used. Suppose we have a 4x5 grid of MPI ranks. We say it's a two-\em dimensional
36  * grid with the first dimension of the size 4 and the second dimensoion of the size 5. The \em actual number of
37  * grid dimensions is two, however we may also consider the grid as being a D-dimensional (D >= 2) with implicit
38  * dimension sizes equal to one, e.g. 4x5 := 4x5x1x1x1... The communication happens along single or multiple
39  * \em directions along the grid dimensions. We specify directions wth bits, eg. directions=00000101 reads as
40  * "communication along 1-st and 3-rd dimensions".
41  * \image html mpi_grid_comm.png "Communication along dimension d0 (between ranks of d0)."
42  * In the provided example the corresponding communicator is MPI_grid::communicator(1 << d0), where d0 is the integer
43  * index of dimension.
44  */
45 class MPI_grid
46 {
47  private:
48  /// Dimensions of the grid.
49  std::vector<int> dimensions_;
50 
51  /// Parent communicator
52  Communicator const& parent_communicator_;
53 
54  /// Grid communicator of the enrire grid returned by MPI_Cart_create
55  Communicator base_grid_communicator_;
56 
57  /// Grid communicators
58  /** Grid comminicators are built for all possible combinations of directions, i.e. 001, 010, 011, etc.
59  * First communicator is the trivial "self" communicator; the last communicator handles the entire grid. */
60  std::vector<Communicator> communicators_;
61 
62  /// Return valid directions for the current grid dimensionality.
63  inline int valid_directions(int directions__) const
64  {
65  return (directions__ & ((1 << dimensions_.size()) - 1));
66  }
67 
68  /// Initialize the grid.
69  void initialize()
70  {
71  if (dimensions_.size() == 0) {
72  TERMINATE("no dimensions provided for the MPI grid");
73  }
74 
75  int sz{1};
76  for (size_t i = 0; i < dimensions_.size(); i++) {
77  sz *= dimensions_[i];
78  }
79 
80  if (parent_communicator_.size() != sz) {
81  std::stringstream s;
82  s << "Number of MPI ranks doesn't match the size of the grid." << std::endl << " grid dimensions :";
83  for (auto& e : dimensions_) {
84  s << " " << e;
85  }
86  s << std::endl << " available number of MPI ranks : " << parent_communicator_.size();
87 
88  TERMINATE(s);
89  }
90 
91  /* communicator of the entire grid */
92  std::vector<int> periods(dimensions_.size(), 0);
93  base_grid_communicator_ = parent_communicator_.cart_create((int)dimensions_.size(), dimensions_.data(),
94  periods.data());
95 
96  /* total number of communicators inside the grid */
97  int num_comm = 1 << dimensions_.size();
98 
99  communicators_ = std::vector<Communicator>(num_comm);
100 
101  /* get all possible communicators */
102  for (int i = 1; i < num_comm; i++) {
103  //bool is_root = true;
104  //int comm_size = 1;
105  std::vector<int> flg(dimensions_.size(), 0);
106 
107  /* each bit represents a directions */
108  for (int j = 0; j < (int)dimensions_.size(); j++) {
109  if (i & (1 << j)) {
110  flg[j] = 1;
111  }
112  }
113  /* subcommunicators */
114  communicators_[i] = base_grid_communicator_.cart_sub(flg.data());
115  }
116 
117  /* expicitly set the "self" communicator */
118  communicators_[0] = Communicator(MPI_COMM_SELF);
119  }
120 
121  /* forbid copy constructor */
122  MPI_grid(MPI_grid const& src) = delete;
123  /* forbid assigment operator */
124  MPI_grid& operator=(MPI_grid const& src) = delete;
125 
126  public:
127  MPI_grid(std::vector<int> dimensions__, Communicator const& parent_communicator__)
128  : dimensions_(dimensions__)
129  , parent_communicator_(parent_communicator__)
130  {
131  initialize();
132  }
133 
134  /// Actual number of grid dimensions
135  inline int num_dimensions() const
136  {
137  return static_cast<int>(dimensions_.size());
138  }
139 
140  inline Communicator const& communicator(int directions__ = 0xFF) const
141  {
142  assert(communicators_.size() != 0);
143  return communicators_[valid_directions(directions__)];
144  }
145 };
