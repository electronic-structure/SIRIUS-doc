102/** Derivative of the occupation function \f$f(x,w)\f$.
103 * \f[
104 * \frac{\partial f(x,w)}{\partial x} = \frac{e^{-y^2} \left(-\sqrt{2} y+1\right)}{\sqrt{\pi } w}, \qquad
105 * y=\frac{x}{w} - \frac{1}{\sqrt{2}} \f]
106 */

116/** Second derivative of the occupation function \f$f(x,w)\f$.
117 * \f[
118 * \frac{\partial^2 f(x,w)}{\partial x^2} = \frac{e^{-y^2} \left(2 \sqrt{2} y^2-2 y-\sqrt{2}\right)}{\sqrt{\pi }
119 * w^2}, \qquad y=\frac{x}{w} - \frac{1}{\sqrt{2}} \f]

193double
194methfessel_paxton::entropy(double x__, double width__, int n__)
195{
196 // see Moodules/w1gauss.f90:74 in function w1gauss (QE code)
197 double x = x__ / width__;
198 double arg = std::min(200.0, x * x);
199 double S = -0.5 * std::exp(-arg) / std::sqrt(pi);
200 if (n__ == 0)
201 return S;
202 double hd{0};
203 double hp = std::exp(-arg);
204 int ni = 0;
205 double a = 1 / std::sqrt(pi);
206 for (int i = 1; i <= n__; ++i) {
207 hd = 2 * x * hp - 2 * ni * hd;
208 ni += 1;
209 double hpm1 = hp;
210 hp = 2 * x * hd - 2 * ni * hp;
211 ni += 1;
212 a = -a / (i + 4.0);
213 S = S - a * (0.5 * hp + ni * hpm1);
214 }
215 return S;
216}
