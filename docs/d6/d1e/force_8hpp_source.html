97  inline void symmetrize(mdarray<double, 2>& forces__) const
98  {
99  if (!ctx_.use_symmetry()) {
100  return;
101  }
102 
103  mdarray<double, 2> sym_forces(3, ctx_.unit_cell().num_atoms());
104  sym_forces.zero();
105 
106  auto& lattice_vectors = ctx_.unit_cell().symmetry().lattice_vectors();
107  auto& inverse_lattice_vectors = ctx_.unit_cell().symmetry().inverse_lattice_vectors();
108 
109  sym_forces.zero();
110 
111  #pragma omp parallel for
112  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
113  vector3d<double> cart_force(&forces__(0, ia));
114  vector3d<double> lat_force =
115  inverse_lattice_vectors * (cart_force / (double)ctx_.unit_cell().symmetry().num_mag_sym());
116 
117  for (int isym = 0; isym < ctx_.unit_cell().symmetry().num_mag_sym(); isym++) {
118  int ja = ctx_.unit_cell().symmetry().sym_table(ia, isym);
119  auto& R = ctx_.unit_cell().symmetry().magnetic_group_symmetry(isym).spg_op.R;
120  vector3d<double> rot_force = lattice_vectors * (R * lat_force);
121 
122  #pragma omp atomic update
123  sym_forces(0, ja) += rot_force[0];
124 
125  #pragma omp atomic update
126  sym_forces(1, ja) += rot_force[1];
127 
128  #pragma omp atomic update
129  sym_forces(2, ja) += rot_force[2];
130  }
131  }
132  sym_forces >> forces__;
133  }

207  void hubbard_force_add_k_contribution_colinear(K_point &kp__, mdarray<double, 2>& forceh_)
208  {
209  mdarray<double_complex, 6> dn_(2 * hamiltonian_.U().hubbard_lmax() + 1,
210  2 * hamiltonian_.U().hubbard_lmax() + 1,
211  2,
212  ctx_.unit_cell().num_atoms(),
213  3,
214  ctx_.unit_cell().num_atoms());
215 
216  hamiltonian_.U().compute_occupancies_derivatives(kp__,
217  hamiltonian_.Q<double_complex>(),
218  dn_);
219 
220  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
221  // compute the derivative of the occupancies numbers
222  for (int dir = 0; dir < 3; dir++) {
223  for (int ia1 = 0; ia1 < ctx_.unit_cell().num_atoms(); ia1++) {
224  const auto& atom = ctx_.unit_cell().atom(ia1);
225  if (atom.type().hubbard_correction()) {
226  const int lmax_at = 2 * atom.type().hubbard_orbital(0).hubbard_l() + 1;
227  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
228  for (int m1 = 0; m1 < lmax_at; m1++) {
229  for (int m2 = 0; m2 < lmax_at; m2++) {
230  forceh_(dir, ia) -= (hamiltonian_.U().U(m2, m1, ispn, ia1) *
231  dn_(m1, m2, ispn, ia1, dir, ia)).real();
232  }
233  }
234  }
235  }
236  }

242  void ibs_force(K_point* kp__,
243  mdarray<double, 2>& ffac__,
244  mdarray<double, 2>& forcek__)
245  {
246 
247  PROFILE("sirius::Force::ibs_force");
248 
249  auto& uc = ctx_.unit_cell();
250  //auto rl = ctx__.reciprocal_lattice();
251 
252  forcek__.zero();
253 
254  dmatrix<double_complex> dm(ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.blacs_grid(),
255  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
256  compute_dmat(kp__, dm);
257 
258  auto& fv_evec = kp__->fv_eigen_vectors();
259 
260  dmatrix<double_complex> h(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx_.blacs_grid(),
261  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
262  dmatrix<double_complex> o(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx_.blacs_grid(),
263  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
264 
265  dmatrix<double_complex> h1(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx_.blacs_grid(),
266  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
267  dmatrix<double_complex> o1(kp__->gklo_basis_size(), kp__->gklo_basis_size(), ctx_.blacs_grid(),
268  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
269 
270  dmatrix<double_complex> zm1(kp__->gklo_basis_size(), ctx_.num_fv_states(), ctx_.blacs_grid(),
271  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
272  dmatrix<double_complex> zf(ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.blacs_grid(),
273  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
274 
275  mdarray<double_complex, 2> alm_row(kp__->num_gkvec_row(), uc.max_mt_aw_basis_size());
276  mdarray<double_complex, 2> alm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
277  mdarray<double_complex, 2> halm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
278 
279  for (int ia = 0; ia < uc.num_atoms(); ia++) {
280  h.zero();
281  o.zero();
282 
283  auto& atom = uc.atom(ia);
284  auto& type = atom.type();
285 
286  /* generate matching coefficients for current atom */
287  kp__->alm_coeffs_row().generate(ia, alm_row);
288  kp__->alm_coeffs_col().generate(ia, alm_col);
289 
290  /* setup apw-lo and lo-apw blocks */
291  hamiltonian_.set_fv_h_o_apw_lo(kp__, type, atom, ia, alm_row, alm_col, h, o);
292 
293  /* apply MT Hamiltonian to column coefficients */
294  hamiltonian_.apply_hmt_to_apw<spin_block_t::nm>(atom, kp__->num_gkvec_col(), alm_col, halm_col);
295 
296  /* conjugate row (<bra|) matching coefficients */
297  for (int i = 0; i < type.mt_aw_basis_size(); i++) {
298  for (int igk = 0; igk < kp__->num_gkvec_row(); igk++) {
299  alm_row(igk, i) = std::conj(alm_row(igk, i));
300  }
301  }
302 
303  /* apw-apw block of the overlap matrix */
304  linalg<CPU>::gemm(0, 1, kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
305  alm_row.at<CPU>(), alm_row.ld(), alm_col.at<CPU>(), alm_col.ld(), o.at<CPU>(), o.ld());
306 
307  /* apw-apw block of the Hamiltonian matrix */
308  linalg<CPU>::gemm(0, 1, kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
309  alm_row.at<CPU>(), alm_row.ld(), halm_col.at<CPU>(), halm_col.ld(), h.at<CPU>(), h.ld());
310 
311  int iat = type.id();
312 
313  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) { // loop over columns
314  auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
315  auto gkvec_col_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_col(igk_col));
316  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) { // for each column loop over rows
317  auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
318  auto gkvec_row_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_row(igk_row));
319 
320  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
321 
322  int igs = ctx_.gvec().shell(ig12);
323 
324  double_complex zt = std::conj(ctx_.gvec_phase_factor(ig12, ia)) * ffac__(iat, igs) * fourpi / uc.omega();
325 
326  double t1 = 0.5 * dot(gkvec_row_cart, gkvec_col_cart);
327 
328  h(igk_row, igk_col) -= t1 * zt;
329  o(igk_row, igk_col) -= zt;
330  }
331  }
332 
333  for (int x = 0; x < 3; x++) {
334  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) { // loop over columns
335  auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
336  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) { // for each column loop over rows
337  auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
338  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
339 
340  vector3d<double> vg = ctx_.gvec().gvec_cart<index_domain_t::global>(ig12);
341  h1(igk_row, igk_col) = double_complex(0.0, vg[x]) * h(igk_row, igk_col);
342  o1(igk_row, igk_col) = double_complex(0.0, vg[x]) * o(igk_row, igk_col);
343  }
344  }
345 
346  for (int icol = 0; icol < kp__->num_lo_col(); icol++) {
347  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) {
348  auto gkvec_row_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_row(igk_row));
349  h1(igk_row, icol + kp__->num_gkvec_col()) = double_complex(0.0, gkvec_row_cart[x]) * h(igk_row, icol + kp__->num_gkvec_col());
350  o1(igk_row, icol + kp__->num_gkvec_col()) = double_complex(0.0, gkvec_row_cart[x]) * o(igk_row, icol + kp__->num_gkvec_col());
351  }
352  }
353 
354  for (int irow = 0; irow < kp__->num_lo_row(); irow++) {
355  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) {
356  auto gkvec_col_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_col(igk_col));
357  h1(irow + kp__->num_gkvec_row(), igk_col) = double_complex(0.0, -gkvec_col_cart[x]) * h(irow + kp__->num_gkvec_row(), igk_col);
358  o1(irow + kp__->num_gkvec_row(), igk_col) = double_complex(0.0, -gkvec_col_cart[x]) * o(irow + kp__->num_gkvec_row(), igk_col);
359  }
360  }
361 
362  /* zm1 = H * V */
363  linalg<CPU>::gemm(0, 0, kp__->gklo_basis_size(), ctx_.num_fv_states(), kp__->gklo_basis_size(),
364  linalg_const<double_complex>::one(), h1, fv_evec, linalg_const<double_complex>::zero(), zm1);
365 
366  /* F = V^{+} * zm1 = V^{+} * H * V */
367  linalg<CPU>::gemm(2, 0, ctx_.num_fv_states(), ctx_.num_fv_states(), kp__->gklo_basis_size(),
368  linalg_const<double_complex>::one(), fv_evec, zm1, linalg_const<double_complex>::zero(), zf);
369 
370  /* zm1 = O * V */
371  linalg<CPU>::gemm(0, 0, kp__->gklo_basis_size(), ctx_.num_fv_states(), kp__->gklo_basis_size(),
372  linalg_const<double_complex>::one(), o1, fv_evec, linalg_const<double_complex>::zero(), zm1);
373 
374  STOP();
375  ///* multiply by energy */
376  //for (int i = 0; i < (int)kp__->spl_fv_states().local_size(); i++)
377  //{
378  // int ist = kp__->spl_fv_states(i);
379  // for (int j = 0; j < kp__->gklo_basis_size_row(); j++) zm1(j, i) = zm1(j, i) * kp__->fv_eigen_value(ist);
380  //}
381 
382  /* F = F - V^{+} * zm1 = F - V^{+} * O * (E*V) */
383  linalg<CPU>::gemm(2, 0, ctx_.num_fv_states(), ctx_.num_fv_states(), kp__->gklo_basis_size(),
384  double_complex(-1, 0), fv_evec, zm1, double_complex(1, 0), zf);
385 
386  for (int i = 0; i < dm.num_cols_local(); i++) {
387  for (int j = 0; j < dm.num_rows_local(); j++) {
388  forcek__(x, ia) += kp__->weight() * real(dm(j, i) * zf(j, i));
389  }
390  }
391  }
392  } //ia
393  }

405  inline mdarray<double, 2> const& calc_forces_vloc()
406  {
407  PROFILE("sirius::Force::calc_forces_vloc");
408 
409  forces_vloc_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
410  forces_vloc_.zero();
411 
412  auto& valence_rho = density_.rho();
413 
414  auto& ri = ctx_.vloc_ri();
415 
416  Unit_cell& unit_cell = ctx_.unit_cell();
417 
418  Gvec const& gvecs = ctx_.gvec();
419 
420  int gvec_count = gvecs.gvec_count(ctx_.comm().rank());
421  int gvec_offset = gvecs.gvec_offset(ctx_.comm().rank());
422 
423  double fact = valence_rho.gvec().reduced() ? 2.0 : 1.0;
424 
425  /* here the calculations are in lattice vectors space */
426  #pragma omp parallel for
427  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
428  Atom& atom = unit_cell.atom(ia);
429 
430  int iat = atom.type_id();
431 
432  for (int igloc = 0; igloc < gvec_count; igloc++) {
433  int ig = gvec_offset + igloc;
434 
435  /* cartesian form for getting cartesian force components */
436  vector3d<double> gvec_cart = gvecs.gvec_cart<index_domain_t::local>(igloc);
437 
438  /* scalar part of a force without multiplying by G-vector */
439  double_complex z = fact * fourpi * ri.value(iat, gvecs.gvec_len(ig)) *
440  std::conj(valence_rho.f_pw_local(igloc)) *
441  std::conj(ctx_.gvec_phase_factor(ig, ia));
442 
443  /* get force components multiplying by cartesian G-vector */
444  for (int x: {0, 1, 2}) {
445  forces_vloc_(x, ia) -= (gvec_cart[x] * z).imag();
446  }
447  }
448  }
449 
450  ctx_.comm().allreduce(&forces_vloc_(0, 0), 3 * ctx_.unit_cell().num_atoms());
451 
452  return forces_vloc_;
453  }

607  inline mdarray<double, 2> const& calc_forces_us()
608  {
609  PROFILE("sirius::Force::calc_forces_us");
610 
611  forces_us_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
612  forces_us_.zero();
613 
614  potential_.fft_transform(-1);
615 
616  Unit_cell& unit_cell = ctx_.unit_cell();
617 
618  double reduce_g_fact = ctx_.gvec().reduced() ? 2.0 : 1.0;
619 
620  /* over atom types */
621  for (int iat = 0; iat < unit_cell.num_atom_types(); iat++){
622  auto& atom_type = unit_cell.atom_type(iat);
623 
624  if (!atom_type.augment() || atom_type.num_atoms() == 0) {
625  continue;
626  }
627 
628  const Augmentation_operator& aug_op = ctx_.augmentation_op(iat);
629 
630  int nbf = atom_type.mt_basis_size();
631 
632  /* get auxiliary density matrix */
633  auto dm = density_.density_matrix_aux(iat);
634 
635  mdarray<double, 2> v_tmp(atom_type.num_atoms(), ctx_.gvec().count() * 2);
636  mdarray<double, 2> tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms());
637 
638  /* over spin components, can be from 1 to 4*/
639  for (int ispin = 0; ispin < ctx_.num_mag_dims() + 1; ispin++ ){
640  /* over 3 components of the force/G - vectors */
641  for (int ivec = 0; ivec < 3; ivec++ ){
642  /* over local rank G vectors */
643  #pragma omp parallel for schedule(static)
644  for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
645  int ig = ctx_.gvec().offset() + igloc;
646  auto gvc = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
647  for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
648  /* here we write in v_tmp -i * G * exp[ iGRn] Veff(G)
649  * but in formula we have i * G * exp[-iGRn] Veff*(G)
650  * the differences because we unfold complex array in the real one
651  * and need negative imagine part due to a multiplication law of complex numbers */
652  auto z = double_complex(0, -gvc[ivec]) * ctx_.gvec_phase_factor(ig, atom_type.atom_id(ia)) *
653  potential_.component(ispin).f_pw_local(igloc);
654  v_tmp(ia, 2 * igloc) = z.real();
655  v_tmp(ia, 2 * igloc + 1) = z.imag();
656  }
657  }
658 
659  /* multiply tmp matrices, or sum over G*/
660  linalg<CPU>::gemm(0, 1, nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
661  aug_op.q_pw(), v_tmp, tmp);
662 
663 
664  #pragma omp parallel for
665  for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
666  for (int i = 0; i < nbf * (nbf + 1) / 2; i++) {
667  forces_us_(ivec, atom_type.atom_id(ia)) += ctx_.unit_cell().omega() * reduce_g_fact *
668  dm(i, ia, ispin) * aug_op.sym_weight(i) * tmp(i, ia);
669  }
670  }
671  }
672  }
673  }
674 
675  ctx_.comm().allreduce(&forces_us_(0, 0), 3 * ctx_.unit_cell().num_atoms());
676 
677  return forces_us_;
678  }

685  inline mdarray<double, 2> const& calc_forces_ewald()
686  {
687  PROFILE("sirius::Force::calc_forces_ewald");
688 
689  forces_ewald_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
690  forces_ewald_.zero();
691 
692  Unit_cell& unit_cell = ctx_.unit_cell();
693 
694  double alpha = ctx_.ewald_lambda();
695 
696  double prefac = (ctx_.gvec().reduced() ? 4.0 : 2.0) * (twopi / unit_cell.omega());
697 
698  int ig0{0};
699  if (ctx_.comm().rank() == 0) {
700  ig0 = 1;
701  }
702 
703  mdarray<double_complex, 1> rho_tmp(ctx_.gvec().count());
704  rho_tmp.zero();
705  #pragma omp parallel for schedule(static)
706  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
707  int ig = ctx_.gvec().offset() + igloc;
708 
709  double_complex rho(0, 0);
710 
711  for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
712  rho += ctx_.gvec_phase_factor(ig, ja) * static_cast<double>(unit_cell.atom(ja).zn());
713  }
714 
715  rho_tmp[igloc] = std::conj(rho);
716  }
717 
718  #pragma omp parallel for
719  for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
720  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
721  int ig = ctx_.gvec().offset() + igloc;
722 
723  double g2 = std::pow(ctx_.gvec().gvec_len(ig), 2);
724 
725  /* cartesian form for getting cartesian force components */
726  vector3d<double> gvec_cart = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
727  double_complex rho(0, 0);
728 
729  double scalar_part = prefac * (rho_tmp[igloc] * ctx_.gvec_phase_factor(ig, ja)).imag() *
730  static_cast<double>(unit_cell.atom(ja).zn()) * std::exp(-g2 / (4 * alpha)) / g2;
731 
732  for (int x : {0, 1, 2}) {
733  forces_ewald_(x, ja) += scalar_part * gvec_cart[x];
734  }
735  }
736  }
737 
738  ctx_.comm().allreduce(&forces_ewald_(0, 0), 3 * ctx_.unit_cell().num_atoms());
739 
740  double invpi = 1. / pi;
741 
742  #pragma omp parallel for
743  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
744  for (int i = 1; i < unit_cell.num_nearest_neighbours(ia); i++) {
745  int ja = unit_cell.nearest_neighbour(i, ia).atom_id;
746 
747  double d = unit_cell.nearest_neighbour(i, ia).distance;
748  double d2 = d * d;
749 
750  vector3d<double> t = unit_cell.lattice_vectors() * vector3d<int>(unit_cell.nearest_neighbour(i, ia).translation);
751 
752  double scalar_part =
753  static_cast<double>(unit_cell.atom(ia).zn() * unit_cell.atom(ja).zn()) / d2 *
754  (std::erfc(std::sqrt(alpha) * d) / d + 2.0 * std::sqrt(alpha * invpi) * std::exp(-d2 * alpha));
755 
756  for (int x : {0, 1, 2}) {
757  forces_ewald_(x, ia) += scalar_part * t[x];
758  }
759  }
760  }
761 
762  return forces_ewald_;
763  }
