82 /** In the second-variational approach we need to compute the following expression for the k-dependent
83 * contribution to the forces:
84 * \f[
85 * {\bf F}_{\rm IBS}^{\alpha}=\sum_{\bf k}w_{\bf k}\sum_{l\sigma}n_{l{\bf k}}
86 * \sum_{ij}c_{\sigma i}^{l{\bf k}*}c_{\sigma j}^{l{\bf k}}
87 * {\bf F}_{ij}^{\alpha{\bf k}}
88 * \f]
89 * This function sums over band and spin indices to get the "density matrix":
90 * \f[
91 * q_{ij} = \sum_{l\sigma}n_{l{\bf k}} c_{\sigma i}^{l{\bf k}*}c_{\sigma j}^{l{\bf k}}
92 * \f]
93 */
94 void compute_dmat(K_point<double>* kp__, sddk::dmatrix<double_complex>& dm__) const;
