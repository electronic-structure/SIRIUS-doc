29 void Atom_type::init(int offset_lo__)
30 {
31  PROFILE("sirius::Atom_type::init");
32 
33  /* check if the class instance was already initialized */
34  if (initialized_) {
35  TERMINATE("can't initialize twice");
36  }
37 
38  offset_lo_ = offset_lo__;
39 
40  /* read data from file if it exists */
41  read_input(file_name_);
42 
43  /* check the nuclear charge */
44  if (zn_ == 0) {
45  TERMINATE("zero atom charge");
46  }
47 
48  if (parameters_.full_potential()) {
49  /* add valence levels to the list of atom's levels */
50  for (auto& e : atomic_conf[zn_ - 1]) {
51  /* check if this level is already in the list */
52  bool in_list{false};
53  for (auto& c : atomic_levels_) {
54  if (c.n == e.n && c.l == e.l && c.k == e.k) {
55  in_list = true;
56  break;
57  }
58  }
59  if (!in_list) {
60  auto level = e;
61  level.core = false;
62  atomic_levels_.push_back(level);
63  }
64  }
65  /* get the number of core electrons */
66  for (auto& e : atomic_levels_) {
67  if (e.core) {
68  num_core_electrons_ += e.occupancy;
69  }
70  }
71 
72  /* initialize aw descriptors if they were not set manually */
73  if (aw_descriptors_.size() == 0) {
74  init_aw_descriptors(parameters_.lmax_apw());
75  }
76 
77  if (static_cast<int>(aw_descriptors_.size()) != (parameters_.lmax_apw() + 1)) {
78  TERMINATE("wrong size of augmented wave descriptors");
79  }
80 
81  max_aw_order_ = 0;
82  for (int l = 0; l <= parameters_.lmax_apw(); l++) {
83  max_aw_order_ = std::max(max_aw_order_, (int)aw_descriptors_[l].size());
84  }
85 
86  if (max_aw_order_ > 3) {
87  TERMINATE("maximum aw order > 3");
88  }
89  }
90 
91  /* initialize index of radial functions */
92  indexr_.init(aw_descriptors_, lo_descriptors_);
93 
94  /* initialize index of muffin-tin basis functions */
95  indexb_.init(indexr_);
96 
97  /* initialize index for wave functions */
98  if (ps_atomic_wfs_.size()) {
99  for (auto& e: ps_atomic_wfs_) {
100  /* add angular quantum number to the list */
101  indexr_wfs_.add(e.aqn);
102  }
103  indexb_wfs_ = sirius::experimental::basis_functions_index(indexr_wfs_, false);
104  if (ps_atomic_wfs_.size() != indexr_wfs_.size()) {
105  TERMINATE("[sirius::Atom_type::init] wrong size of atomic orbital list");
106  }
107  }
108 
109  if (hubbard_correction_) {
110  /* circus for the hubbard orbitals */
111  //hubbard_indexr_.init(hubbard_lo_descriptors_);
112  //hubbard_indexb_.init(hubbard_indexr_);
113 
114  indexb_hub_ = sirius::experimental::basis_functions_index(indexr_hub_, false);
115  }
116 
117  if (!parameters_.full_potential()) {
118  assert(mt_radial_basis_size() == num_beta_radial_functions());
119  assert(lmax_beta() == indexr().lmax());
120  }
121 
122  /* get number of valence electrons */
123  num_valence_electrons_ = zn_ - num_core_electrons_;
124 
125  int lmmax_pot = utils::lmmax(parameters_.lmax_pot());
126 
127  if (parameters_.full_potential()) {
128  auto l_by_lm = utils::l_by_lm(parameters_.lmax_pot());
129 
130  /* index the non-zero radial integrals */
131  std::vector<std::pair<int, int>> non_zero_elements;
132 
133  for (int lm = 0; lm < lmmax_pot; lm++) {
134  int l = l_by_lm[lm];
135 
136  for (int i2 = 0; i2 < indexr().size(); i2++) {
137  int l2 = indexr(i2).l;
138  for (int i1 = 0; i1 <= i2; i1++) {
139  int l1 = indexr(i1).l;
140  if ((l + l1 + l2) % 2 == 0) {
141  if (lm) {
142  non_zero_elements.push_back(std::pair<int, int>(i2, lm + lmmax_pot * i1));
143  }
144  for (int j = 0; j < parameters_.num_mag_dims(); j++) {
145  int offs = (j + 1) * lmmax_pot * indexr().size();
146  non_zero_elements.push_back(std::pair<int, int>(i2, lm + lmmax_pot * i1 + offs));
147  }
148  }
149  }
150  }
151  }
152  idx_radial_integrals_ = mdarray<int, 2>(2, non_zero_elements.size());
153  for (int j = 0; j < (int)non_zero_elements.size(); j++) {
154  idx_radial_integrals_(0, j) = non_zero_elements[j].first;
155  idx_radial_integrals_(1, j) = non_zero_elements[j].second;
156  }
157  }
158 
159  if (parameters_.processing_unit() == device_t::GPU && parameters_.full_potential()) {
160  idx_radial_integrals_.allocate(memory_t::device).copy_to(memory_t::device);
161  rf_coef_ = mdarray<double, 3>(num_mt_points(), 4, indexr().size(), memory_t::host_pinned, "Atom_type::rf_coef_");
162  vrf_coef_ = mdarray<double, 3>(num_mt_points(), 4, lmmax_pot * indexr().size() * (parameters_.num_mag_dims() + 1),
163  memory_t::host_pinned, "Atom_type::vrf_coef_");
164  rf_coef_.allocate(memory_t::device);
165  vrf_coef_.allocate(memory_t::device);
166  }
167 
168  if (this->spin_orbit_coupling()) {
169  this->generate_f_coefficients();
170  }
171 
172  if (is_paw()) {
173  if (num_beta_radial_functions() != num_ps_paw_wf()) {
174  TERMINATE("wrong number of pseudo wave-functions for PAW");
175  }
176  if (num_beta_radial_functions() != num_ae_paw_wf()) {
177  TERMINATE("wrong number of all-electron wave-functions for PAW");
178  }
179  ae_paw_wfs_array_ = mdarray<double, 2>(num_mt_points(), num_beta_radial_functions());
180  ae_paw_wfs_array_.zero();
181  ps_paw_wfs_array_ = mdarray<double, 2>(num_mt_points(), num_beta_radial_functions());
182  ps_paw_wfs_array_.zero();
183 
184  for (int i = 0; i < num_beta_radial_functions(); i++) {
185  std::copy(ae_paw_wf(i).begin(), ae_paw_wf(i).end(), &ae_paw_wfs_array_(0, i));
186  std::copy(ps_paw_wf(i).begin(), ps_paw_wf(i).end(), &ps_paw_wfs_array_(0, i));
187  }
188  }
189 
190  initialized_ = true;
191 }

313 void Atom_type::read_input_core(json const& parser)
314 {
315  std::string core_str = std::string(parser["core"]);
316  if (int size = (int)core_str.size()) {
317  if (size % 2) {
318  std::stringstream s;
319  s << "wrong core configuration string : " << core_str;
320  TERMINATE(s);
321  }
322  int j = 0;
323  while (j < size) {
324  char c1 = core_str[j++];
325  char c2 = core_str[j++];
326 
327  int n = -1;
328  int l = -1;
329 
330  std::istringstream iss(std::string(1, c1));
331  iss >> n;
332 
333  if (n <= 0 || iss.fail()) {
334  std::stringstream s;
335  s << "wrong principal quantum number : " << std::string(1, c1);
336  TERMINATE(s);
337  }
338 
339  switch (c2) {
340  case 's': {
341  l = 0;
342  break;
343  }
344  case 'p': {
345  l = 1;
346  break;
347  }
348  case 'd': {
349  l = 2;
350  break;
351  }
352  case 'f': {
353  l = 3;
354  break;
355  }
356  default: {
357  std::stringstream s;
358  s << "wrong angular momentum label : " << std::string(1, c2);
359  TERMINATE(s);
360  }
361  }
362 
363  for (auto& e: atomic_conf[zn_ - 1]) {
364  if (e.n == n && e.l == l) {
365  auto level = e;
366  level.core = true;
367  atomic_levels_.push_back(level);
368  }
369  }
370  }
371  }

433 void Atom_type::read_pseudo_uspp(json const& parser)
434 {
435  symbol_ = parser["pseudo_potential"]["header"]["element"].get<std::string>();
436 
437  double zp;
438  zp = parser["pseudo_potential"]["header"]["z_valence"].get<double>();
439  zn_ = int(zp + 1e-10);
440 
441  int nmtp = parser["pseudo_potential"]["header"]["mesh_size"].get<int>();
442 
443  auto rgrid = parser["pseudo_potential"]["radial_grid"].get<std::vector<double>>();
444  if (static_cast<int>(rgrid.size()) != nmtp) {
445  TERMINATE("wrong mesh size");
446  }
447  /* set the radial grid */
448  set_radial_grid(nmtp, rgrid.data());
449 
450  local_potential(parser["pseudo_potential"]["local_potential"].get<std::vector<double>>());
451 
452  ps_core_charge_density(parser["pseudo_potential"].value("core_charge_density", std::vector<double>(rgrid.size(), 0)));
453 
454  ps_total_charge_density(parser["pseudo_potential"]["total_charge_density"].get<std::vector<double>>());
455 
456  if (local_potential().size() != rgrid.size() || ps_core_charge_density().size() != rgrid.size() ||
457  ps_total_charge_density().size() != rgrid.size()) {
458  std::cout << local_potential().size() << " " << ps_core_charge_density().size() << " "
459  << ps_total_charge_density().size() << std::endl;
460  TERMINATE("wrong array size");
461  }
462 
463  if (parser["pseudo_potential"]["header"].count("spin_orbit")) {
464  spin_orbit_coupling_ = parser["pseudo_potential"]["header"].value("spin_orbit", spin_orbit_coupling_);
465  }
466 
467  int nbf = parser["pseudo_potential"]["header"]["number_of_proj"].get<int>();
468 
469  for (int i = 0; i < nbf; i++) {
470  auto beta = parser["pseudo_potential"]["beta_projectors"][i]["radial_function"].get<std::vector<double>>();
471  if (static_cast<int>(beta.size()) > num_mt_points()) {
472  std::stringstream s;
473  s << "wrong size of beta functions for atom type " << symbol_ << " (label: " << label_ << ")" << std::endl
474  << "size of beta radial functions in the file: " << beta.size() << std::endl
475  << "radial grid size: " << num_mt_points();
476  TERMINATE(s);
477  }
478  int l = parser["pseudo_potential"]["beta_projectors"][i]["angular_momentum"].get<int>();
479  if (spin_orbit_coupling_) {
480  // we encode the fact that the total angular momentum j = l
481  // -1/2 or l + 1/2 by changing the sign of l
482 
483  double j = parser["pseudo_potential"]["beta_projectors"][i]["total_angular_momentum"].get<double>();
484  if (j < (double)l) {
485  l *= -1;
486  }
487  }
488  add_beta_radial_function(l, beta);
489  }
490 
491  mdarray<double, 2> d_mtrx(nbf, nbf);
492  d_mtrx.zero();
493  auto v = parser["pseudo_potential"]["D_ion"].get<std::vector<double>>();
494 
495  for (int i = 0; i < nbf; i++) {
496  for (int j = 0; j < nbf; j++) {
497  d_mtrx(i, j) = v[j * nbf + i];
498  }
499  }
500  d_mtrx_ion(d_mtrx);
501 
502  if (parser["pseudo_potential"].count("augmentation")) {
503  for (size_t k = 0; k < parser["pseudo_potential"]["augmentation"].size(); k++) {
504  int i = parser["pseudo_potential"]["augmentation"][k]["i"].get<int>();
505  int j = parser["pseudo_potential"]["augmentation"][k]["j"].get<int>();
506  //int idx = j * (j + 1) / 2 + i;
507  int l = parser["pseudo_potential"]["augmentation"][k]["angular_momentum"].get<int>();
508  auto qij = parser["pseudo_potential"]["augmentation"][k]["radial_function"].get<std::vector<double>>();
509  if ((int)qij.size() != num_mt_points()) {
510  TERMINATE("wrong size of qij");
511  }
512  add_q_radial_function(i, j, l, qij);
513  }
514  }
515 
516  /* read starting wave functions ( UPF CHI ) */
517  if (parser["pseudo_potential"].count("atomic_wave_functions")) {
518  /* total number of pseudo atomic wave-functions */
519  size_t nwf = parser["pseudo_potential"]["atomic_wave_functions"].size();
520  /* loop over wave-functions */
521  for (size_t k = 0; k < nwf; k++) {
522  auto v = parser["pseudo_potential"]["atomic_wave_functions"][k]["radial_function"].get<std::vector<double>>();
523 
524  if ((int)v.size() != num_mt_points()) {
525  std::stringstream s;
526  s << "wrong size of atomic functions for atom type " << symbol_ << " (label: " << label_ << ")"
527  << std::endl
528  << "size of atomic radial functions in the file: " << v.size() << std::endl
529  << "radial grid size: " << num_mt_points();
530  TERMINATE(s);

662 void Atom_type::generate_f_coefficients()
663 {
664  // we consider Pseudo potentials with spin orbit couplings
665 
666  // First thing, we need to compute the
667  // \f[f^{\sigma\sigma^\prime}_{l,j,m;l\prime,j\prime,m\prime}\f]
668  // They are defined by Eq.9 of doi:10.1103/PhysRevB.71.115106
669  // and correspond to transformations of the
670  // spherical harmonics
671  if (!this->spin_orbit_coupling()) {
672  return;
673  }
674 
675  // number of beta projectors
676  int nbf = this->mt_basis_size();
677  f_coefficients_ = mdarray<double_complex, 4>(nbf, nbf, 2, 2);
678  f_coefficients_.zero();
679 
680  for (int xi2 = 0; xi2 < nbf; xi2++) {
681  const int l2 = this->indexb(xi2).l;
682  const double j2 = this->indexb(xi2).j;
683  const int m2 = this->indexb(xi2).m;
684  for (int xi1 = 0; xi1 < nbf; xi1++) {
685  const int l1 = this->indexb(xi1).l;
686  const double j1 = this->indexb(xi1).j;
687  const int m1 = this->indexb(xi1).m;
688 
689  if ((l2 == l1) && (std::abs(j1 - j2) < 1e-8)) {
690  // take beta projectors with same l and j
691  for (auto sigma2 = 0; sigma2 < 2; sigma2++) {
692  for (auto sigma1 = 0; sigma1 < 2; sigma1++) {
693  double_complex coef = {0.0, 0.0};
694 
695  // yes dirty but loop over double is worst.
696  // since mj is only important for the rotation
697  // of the spherical harmonics the code takes
698  // into account this odd convention.
699 
700  int jj1 = static_cast<int>(2.0 * j1 + 1e-8);
701  for (int mj = -jj1; mj <= jj1; mj += 2) {
702  coef += SHT::calculate_U_sigma_m(l1, j1, mj, m1, sigma1) *
703  SHT::ClebschGordan(l1, j1, mj / 2.0, sigma1) *
704  std::conj(SHT::calculate_U_sigma_m(l2, j2, mj, m2, sigma2)) *
705  SHT::ClebschGordan(l2, j2, mj / 2.0, sigma2);
706  }
707  f_coefficients_(xi1, xi2, sigma1, sigma2) = coef;
708  }
709  }
710  }
711  }
712  }

733 void Atom_type::add_hubbard_orbital(int n__, int l__, double occ__, double U, double J, const double *hub_coef__,
734  double alpha__, double beta__, double J0__, std::vector<double> initial_occupancy__)
735 {
736  // TODO: pass radial function for l or for j=l+1/2 j=l-1/2 and don't rely on the list of pseudoatomic wfs
737 
738  if (n__ <= 0) {
739  TERMINATE("negative principal quantum number");
740  }
741 
742  /* we have to find one (or two in case of spin-orbit) atomic functions and construct hubbard orbital */
743  std::vector<int> idx_rf;
744  for (int s = 0; s < (int)ps_atomic_wfs_.size(); s++) {
745  auto& e = ps_atomic_wfs_[s];
746  int n = e.n;
747  auto aqn = e.aqn;
748  if (n == n__ && aqn.l() == l__) {
749  idx_rf.push_back(s);
750  /* in spin orbit case we need to find the second radial function, otherwise we break */
751  if (!(aqn.s() && aqn.l() > 0)) {
752  break;
753  }
754  }
755  }
756  if (idx_rf.size() == 0) {
757  std::stringstream s;
758  s << "[add_hubbard_orbital] atomic radial function is not found for atom type " << label_ << std::endl
759  << " the following atomic wave-functions are set: " << std::endl;
760  for (int k = 0; k < (int)ps_atomic_wfs_.size(); k++) {
761  auto& e = ps_atomic_wfs_[k];
762  int n = e.n;
763  auto aqn = e.aqn;
764  s << " n=" << n << " l=" << aqn.l() << " j=" << aqn.j() << std::endl;
765  }
766  s << " the following atomic orbital is requested for U-correction: n=" << n__ << " l=" << l__;
767  TERMINATE(s);
768  }
769  if (idx_rf.size() > 2) {
770  std::stringstream s;
771  s << "[add_hubbard_orbital] number of atomic functions > 2";
772  TERMINATE(s);
773  }
774 
775  /* create a scalar hubbard wave-function from one or two atomic radial functions */
776  Spline<double> s(radial_grid_);
777  double f = 1.0 / static_cast<double>(idx_rf.size());
778  for (int i: idx_rf) {
779  auto& rwf = ps_atomic_wfs_[i].f;
780  for (int ir = 0; ir < s.num_points(); ir++) {
781  s(ir) += f * rwf(ir);
782  }
783  }
784 
785  /* add a record in radial function index */
786  indexr_hub_.add(sirius::experimental::angular_momentum_quantum_number(l__));
787 
788  /* add radial function of Hubbard orbital to a list */
789  hubbard_radial_functions_.push_back(std::move(s.interpolate()));
790 
791  /* add Hubbard orbital descriptor to a list */
792  lo_descriptors_hub_.emplace_back(n__, l__, -1, occ__, J, U, hub_coef__, alpha__, beta__, J0__, initial_occupancy__);
793 }
