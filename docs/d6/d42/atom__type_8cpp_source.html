29void Atom_type::init(int offset_lo__)
30{
31 PROFILE("sirius::Atom_type::init");
32
33 /* check if the class instance was already initialized */
34 if (initialized_) {
35 RTE_THROW("can't initialize twice");
36 }
37
38 offset_lo_ = offset_lo__;
39
40 /* read data from file if it exists */
41 read_input(file_name_);
42
43 /* check the nuclear charge */
44 if (zn_ == 0) {
45 RTE_THROW("zero atom charge");
46 }
47
48 if (parameters_.full_potential()) {
49 /* add valence levels to the list of atom's levels */
50 for (auto& e : atomic_conf[zn_ - 1]) {
51 /* check if this level is already in the list */
52 bool in_list{false};
53 for (auto& c : atomic_levels_) {
54 if (c.n == e.n && c.l == e.l && c.k == e.k) {
55 in_list = true;
56 break;
57 }
58 }
59 if (!in_list) {
60 auto level = e;
61 level.core = false;
62 atomic_levels_.push_back(level);
63 }
64 }
65 /* get the number of core electrons */
66 for (auto& e : atomic_levels_) {
67 if (e.core) {
68 num_core_electrons_ += e.occupancy;
69 }
70 }
71
72 /* initialize aw descriptors if they were not set manually */
73 if (aw_descriptors_.size() == 0) {
74 init_aw_descriptors(parameters_.lmax_apw());
75 }
76
77 if (static_cast<int>(aw_descriptors_.size()) != (parameters_.lmax_apw() + 1)) {
78 RTE_THROW("wrong size of augmented wave descriptors");
79 }
80
81 max_aw_order_ = 0;
82 for (int l = 0; l <= parameters_.lmax_apw(); l++) {
83 max_aw_order_ = std::max(max_aw_order_, (int)aw_descriptors_[l].size());
84 }
85
86 if (max_aw_order_ > 3) {
87 RTE_THROW("maximum aw order > 3");
88 }
89 }
90
91 /* initialize index of radial functions */
92 indexr_.init(aw_descriptors_, lo_descriptors_);
93
94 /* initialize index of muffin-tin basis functions */
95 indexb_.init(indexr_);
96
97 /* initialize index for wave functions */
98 if (ps_atomic_wfs_.size()) {
99 for (size_t i = 0; i < ps_atomic_wfs_.size(); i++) {
100 if (ps_atomic_wfs_[i].am.s()) {
101 if (ps_atomic_wfs_[i].am.l() == 0) {
102 indexr_wfs_.add(ps_atomic_wfs_[i].am);
103 } else {
104 indexr_wfs_.add(ps_atomic_wfs_[i].am, ps_atomic_wfs_[i + 1].am);
105 i += 1;
106 }
107 } else {
108 indexr_wfs_.add(ps_atomic_wfs_[i].am);
109 }
110 }
111 indexb_wfs_ = sirius::experimental::basis_functions_index(indexr_wfs_, false);
112 if (static_cast<int>(ps_atomic_wfs_.size()) != indexr_wfs_.size()) {
113 RTE_THROW("wrong size of atomic orbital list");
114 }
115 }
116
117 if (hubbard_correction_) {
118 indexb_hub_ = sirius::experimental::basis_functions_index(indexr_hub_, false);
119 }
120
121 if (!parameters_.full_potential()) {
122 assert(mt_radial_basis_size() == num_beta_radial_functions());
123 assert(lmax_beta() == indexr().lmax());
124 }
125
126 /* get number of valence electrons */
127 num_valence_electrons_ = zn_ - num_core_electrons_;
128
129 int lmmax_pot = utils::lmmax(parameters_.lmax_pot());
130
131 if (parameters_.full_potential()) {
132 auto l_by_lm = utils::l_by_lm(parameters_.lmax_pot());
133
134 /* index the non-zero radial integrals */
135 std::vector<std::pair<int, int>> non_zero_elements;
136
137 for (int lm = 0; lm < lmmax_pot; lm++) {
138 int l = l_by_lm[lm];
139
140 for (int i2 = 0; i2 < indexr().size(); i2++) {
141 int l2 = indexr(i2).l;
142 for (int i1 = 0; i1 <= i2; i1++) {
143 int l1 = indexr(i1).l;
144 if ((l + l1 + l2) % 2 == 0) {
145 if (lm) {
146 non_zero_elements.push_back(std::pair<int, int>(i2, lm + lmmax_pot * i1));
147 }
148 for (int j = 0; j < parameters_.num_mag_dims(); j++) {
149 int offs = (j + 1) * lmmax_pot * indexr().size();
150 non_zero_elements.push_back(std::pair<int, int>(i2, lm + lmmax_pot * i1 + offs));
151 }
152 }
153 }
154 }
155 }
156 idx_radial_integrals_ = mdarray<int, 2>(2, non_zero_elements.size());
157 for (int j = 0; j < (int)non_zero_elements.size(); j++) {
158 idx_radial_integrals_(0, j) = non_zero_elements[j].first;
159 idx_radial_integrals_(1, j) = non_zero_elements[j].second;
160 }
161 }
162
163 if (parameters_.processing_unit() == device_t::GPU && parameters_.full_potential()) {
164 idx_radial_integrals_.allocate(memory_t::device).copy_to(memory_t::device);
165 rf_coef_ = mdarray<double, 3>(num_mt_points(), 4, indexr().size(), memory_t::host_pinned, "Atom_type::rf_coef_");
166 vrf_coef_ = mdarray<double, 3>(num_mt_points(), 4, lmmax_pot * indexr().size() * (parameters_.num_mag_dims() + 1),
167 memory_t::host_pinned, "Atom_type::vrf_coef_");
168 rf_coef_.allocate(memory_t::device);
169 vrf_coef_.allocate(memory_t::device);
170 }
171 if (parameters_.processing_unit() == device_t::GPU) {
172 radial_grid_.copy_to_device();
173 }
174
175 if (this->spin_orbit_coupling()) {
176 this->generate_f_coefficients();
177 }
178
179 if (is_paw()) {
180 if (num_beta_radial_functions() != num_ps_paw_wf()) {
181 RTE_THROW("wrong number of pseudo wave-functions for PAW");
182 }
183 if (num_beta_radial_functions() != num_ae_paw_wf()) {
184 RTE_THROW("wrong number of all-electron wave-functions for PAW");
185 }
186 ae_paw_wfs_array_ = mdarray<double, 2>(num_mt_points(), num_beta_radial_functions());
187 ae_paw_wfs_array_.zero();
188 ps_paw_wfs_array_ = mdarray<double, 2>(num_mt_points(), num_beta_radial_functions());
189 ps_paw_wfs_array_.zero();
190
191 for (int i = 0; i < num_beta_radial_functions(); i++) {
192 std::copy(ae_paw_wf(i).begin(), ae_paw_wf(i).end(), &ae_paw_wfs_array_(0, i));
193 std::copy(ps_paw_wf(i).begin(), ps_paw_wf(i).end(), &ps_paw_wfs_array_(0, i));
194 }
195 }
196
197 if (free_atom_radial_grid_.num_points() == 0) {
198 free_atom_radial_grid_ = Radial_grid_factory<double>(radial_grid_t::power, 3000, radial_grid_.first(), 10.0, 2);
199 free_atom_density_.resize(free_atom_radial_grid_.num_points());
200 for (int i = 0; i < free_atom_radial_grid_.num_points(); i++) {
201 auto x = free_atom_radial_grid_.x(i);
202 free_atom_density_[i] = std::exp(-x) * zn_ / 8 / pi;
203 }
204 }
205 initialized_ = true;
206}

208void Atom_type::init_free_atom_density(bool smooth)
209{
210 free_atom_density_spline_ = Spline<double>(free_atom_radial_grid_);
211 /* smooth free atom density inside the muffin-tin sphere */
212 if (smooth) {
213 /* find point on the grid close to the muffin-tin radius */
214 int irmt = free_atom_radial_grid_.index_of(mt_radius());
215 /* interpolate at this point near MT radius */
216 double R = free_atom_radial_grid_[irmt];
217
218 /* make smooth free atom density inside muffin-tin */
219 for (int i = 0; i <= irmt; i++) {
220 double x = free_atom_radial_grid(i);
221 //free_atom_density_spline_(i) = b(0) * std::pow(free_atom_radial_grid(i), 2) + b(1) * std::pow(free_atom_radial_grid(i), 3);
222 free_atom_density_spline_(i) = free_atom_density_[i] * 0.5 * (1 + std::erf((x / R - 0.5) * 10));
223 }
224
225 ///* write smoothed density */
226 //sstr.str("");
227 //sstr << "free_density_modified_" << id_ << ".dat";
228 //fout = fopen(sstr.str().c_str(), "w");
229
230 //for (int ir = 0; ir < free_atom_radial_grid().num_points(); ir++) {
231 // fprintf(fout, "%18.12f %18.12f \n", free_atom_radial_grid(ir), free_atom_density(ir));
232 //}
233 //fclose(fout);
234 } else {
235 for (int i = 0; i < free_atom_radial_grid_.num_points(); i++) {
236 free_atom_density_spline_(i) = free_atom_density_[i];
237 }
238 }
239 free_atom_density_spline_.interpolate();
240}

325void Atom_type::read_input_core(nlohmann::json const& parser)
326{
327 std::string core_str = std::string(parser["core"]);
328 if (int size = (int)core_str.size()) {
329 if (size % 2) {
330 std::stringstream s;
331 s << "wrong core configuration string : " << core_str;
332 RTE_THROW(s);
333 }
334 int j = 0;
335 while (j < size) {
336 char c1 = core_str[j++];
337 char c2 = core_str[j++];
338
339 int n = -1;
340 int l = -1;
341
342 std::istringstream iss(std::string(1, c1));
343 iss >> n;
344
345 if (n <= 0 || iss.fail()) {
346 std::stringstream s;
347 s << "wrong principal quantum number : " << std::string(1, c1);
348 TERMINATE(s);
349 }
350
351 switch (c2) {
352 case 's': {
353 l = 0;
354 break;
355 }
356 case 'p': {
357 l = 1;
358 break;
359 }
360 case 'd': {
361 l = 2;
362 break;
363 }
364 case 'f': {
365 l = 3;
366 break;
367 }
368 default: {
369 std::stringstream s;
370 s << "wrong angular momentum label : " << std::string(1, c2);
371 TERMINATE(s);
372 }
373 }
374
375 for (auto& e: atomic_conf[zn_ - 1]) {
376 if (e.n == n && e.l == l) {
377 auto level = e;
378 level.core = true;
379 atomic_levels_.push_back(level);
380 }
381 }
382 }
383 }
384}

386void Atom_type::read_input_aw(nlohmann::json const& parser)
387{
388 radial_solution_descriptor rsd;
389 radial_solution_descriptor_set rsd_set;
390
391 /* default augmented wave basis */
392 rsd.n = -1;
393 rsd.l = -1;
394 for (size_t order = 0; order < parser["valence"][0]["basis"].size(); order++) {
395 rsd.enu = parser["valence"][0]["basis"][order]["enu"].get<double>();
396 rsd.dme = parser["valence"][0]["basis"][order]["dme"].get<int>();
397 rsd.auto_enu = parser["valence"][0]["basis"][order]["auto"].get<int>();
398 aw_default_l_.push_back(rsd);
399 }
400
401 for (size_t j = 1; j < parser["valence"].size(); j++) {
402 rsd.l = parser["valence"][j]["l"].get<int>();
403 rsd.n = parser["valence"][j]["n"].get<int>();
404 rsd_set.clear();
405 for (size_t order = 0; order < parser["valence"][j]["basis"].size(); order++) {
406 rsd.enu = parser["valence"][j]["basis"][order]["enu"].get<double>();
407 rsd.dme = parser["valence"][j]["basis"][order]["dme"].get<int>();
408 rsd.auto_enu = parser["valence"][j]["basis"][order]["auto"].get<int>();
409 rsd_set.push_back(rsd);
410 }
411 aw_specific_l_.push_back(rsd_set);
412 }
413}

415void Atom_type::read_input_lo(nlohmann::json const& parser)
416{
417 radial_solution_descriptor rsd;
418 radial_solution_descriptor_set rsd_set;
419
420 if (!parser.count("lo")) {
421 return;
422 }
423
424 int l;
425 for (size_t j = 0; j < parser["lo"].size(); j++) {
426 l = parser["lo"][j]["l"].get<int>();
427
428 local_orbital_descriptor lod;
429 lod.l = l;
430 rsd.l = l;
431 rsd_set.clear();
432 for (size_t order = 0; order < parser["lo"][j]["basis"].size(); order++) {
433 rsd.n = parser["lo"][j]["basis"][order]["n"].get<int>();
434 rsd.enu = parser["lo"][j]["basis"][order]["enu"].get<double>();
435 rsd.dme = parser["lo"][j]["basis"][order]["dme"].get<int>();
436 rsd.auto_enu = parser["lo"][j]["basis"][order]["auto"].get<int>();
437 rsd_set.push_back(rsd);
438 }
439 lod.rsd_set = rsd_set;
440 lo_descriptors_.push_back(lod);
441 }
442}

445void Atom_type::read_pseudo_uspp(nlohmann::json const& parser)
446{
447 symbol_ = parser["pseudo_potential"]["header"]["element"].get<std::string>();
448
449 double zp;
450 zp = parser["pseudo_potential"]["header"]["z_valence"].get<double>();
451 zn_ = int(zp + 1e-10);
452
453 int nmtp = parser["pseudo_potential"]["header"]["mesh_size"].get<int>();
454
455 auto rgrid = parser["pseudo_potential"]["radial_grid"].get<std::vector<double>>();
456 if (static_cast<int>(rgrid.size()) != nmtp) {
457 TERMINATE("wrong mesh size");
458 }
459 /* set the radial grid */
460 set_radial_grid(nmtp, rgrid.data());
461
462 local_potential(parser["pseudo_potential"]["local_potential"].get<std::vector<double>>());
463
464 ps_core_charge_density(parser["pseudo_potential"].value("core_charge_density", std::vector<double>(rgrid.size(), 0)));
465
466 ps_total_charge_density(parser["pseudo_potential"]["total_charge_density"].get<std::vector<double>>());
467
468 if (local_potential().size() != rgrid.size() || ps_core_charge_density().size() != rgrid.size() ||
469 ps_total_charge_density().size() != rgrid.size()) {
470 std::cout << local_potential().size() << " " << ps_core_charge_density().size() << " "
471 << ps_total_charge_density().size() << std::endl;
472 TERMINATE("wrong array size");
473 }
474
475 if (parser["pseudo_potential"]["header"].count("spin_orbit")) {
476 spin_orbit_coupling_ = parser["pseudo_potential"]["header"].value("spin_orbit", spin_orbit_coupling_);
477 }
478
479 int nbf = parser["pseudo_potential"]["header"]["number_of_proj"].get<int>();
480
481 for (int i = 0; i < nbf; i++) {
482 auto beta = parser["pseudo_potential"]["beta_projectors"][i]["radial_function"].get<std::vector<double>>();
483 if (static_cast<int>(beta.size()) > num_mt_points()) {
484 std::stringstream s;
485 s << "wrong size of beta functions for atom type " << symbol_ << " (label: " << label_ << ")" << std::endl
486 << "size of beta radial functions in the file: " << beta.size() << std::endl
487 << "radial grid size: " << num_mt_points();
488 TERMINATE(s);
489 }
490 int l = parser["pseudo_potential"]["beta_projectors"][i]["angular_momentum"].get<int>();
491 if (spin_orbit_coupling_) {
492 // we encode the fact that the total angular momentum j = l
493 // -1/2 or l + 1/2 by changing the sign of l
494
495 double j = parser["pseudo_potential"]["beta_projectors"][i]["total_angular_momentum"].get<double>();
496 if (j < (double)l) {
497 l *= -1;
498 }
499 }
500 add_beta_radial_function(l, beta);
501 }
502
503 mdarray<double, 2> d_mtrx(nbf, nbf);
504 d_mtrx.zero();
505 auto v = parser["pseudo_potential"]["D_ion"].get<std::vector<double>>();
506
507 for (int i = 0; i < nbf; i++) {
508 for (int j = 0; j < nbf; j++) {
509 d_mtrx(i, j) = v[j * nbf + i];
510 }
511 }
512 d_mtrx_ion(d_mtrx);
513
514 if (parser["pseudo_potential"].count("augmentation")) {
515 for (size_t k = 0; k < parser["pseudo_potential"]["augmentation"].size(); k++) {
516 int i = parser["pseudo_potential"]["augmentation"][k]["i"].get<int>();
517 int j = parser["pseudo_potential"]["augmentation"][k]["j"].get<int>();
518 //int idx = j * (j + 1) / 2 + i;
519 int l = parser["pseudo_potential"]["augmentation"][k]["angular_momentum"].get<int>();
520 auto qij = parser["pseudo_potential"]["augmentation"][k]["radial_function"].get<std::vector<double>>();
521 if ((int)qij.size() != num_mt_points()) {
522 TERMINATE("wrong size of qij");
523 }
524 add_q_radial_function(i, j, l, qij);
525 }
526 }
527
528 /* read starting wave functions ( UPF CHI ) */
529 if (parser["pseudo_potential"].count("atomic_wave_functions")) {
530 /* total number of pseudo atomic wave-functions */
531 size_t nwf = parser["pseudo_potential"]["atomic_wave_functions"].size();
532 /* loop over wave-functions */
533 for (size_t k = 0; k < nwf; k++) {
534 auto v = parser["pseudo_potential"]["atomic_wave_functions"][k]["radial_function"].get<std::vector<double>>();
535
536 if ((int)v.size() != num_mt_points()) {
537 std::stringstream s;
538 s << "wrong size of atomic functions for atom type " << symbol_ << " (label: " << label_ << ")"
539 << std::endl
540 << "size of atomic radial functions in the file: " << v.size() << std::endl
541 << "radial grid size: " << num_mt_points();
542 TERMINATE(s);
543 }
544
545 int l = parser["pseudo_potential"]["atomic_wave_functions"][k]["angular_momentum"].get<int>();
546 int n = -1;
547 double occ{0};
548 if (parser["pseudo_potential"]["atomic_wave_functions"][k].count("occupation")) {
549 occ = parser["pseudo_potential"]["atomic_wave_functions"][k]["occupation"].get<double>();
550 }
551
552 if (parser["pseudo_potential"]["atomic_wave_functions"][k].count("label")) {

674void Atom_type::generate_f_coefficients()
675{
676 // we consider Pseudo potentials with spin orbit couplings
677
678 // First thing, we need to compute the
679 // \f[f^{\sigma\sigma^\prime}_{l,j,m;l\prime,j\prime,m\prime}\f]
680 // They are defined by Eq.9 of doi:10.1103/PhysRevB.71.115106
681 // and correspond to transformations of the
682 // spherical harmonics
683 if (!this->spin_orbit_coupling()) {
684 return;
685 }
686
687 // number of beta projectors
688 int nbf = this->mt_basis_size();
689 f_coefficients_ = mdarray<double_complex, 4>(nbf, nbf, 2, 2);
690 f_coefficients_.zero();
691
692 for (int xi2 = 0; xi2 < nbf; xi2++) {
693 const int l2 = this->indexb(xi2).l;
694 const double j2 = this->indexb(xi2).j;
695 const int m2 = this->indexb(xi2).m;
696 for (int xi1 = 0; xi1 < nbf; xi1++) {
697 const int l1 = this->indexb(xi1).l;
698 const double j1 = this->indexb(xi1).j;
699 const int m1 = this->indexb(xi1).m;
700
701 if ((l2 == l1) && (std::abs(j1 - j2) < 1e-8)) {
702 // take beta projectors with same l and j
703 for (auto sigma2 = 0; sigma2 < 2; sigma2++) {
704 for (auto sigma1 = 0; sigma1 < 2; sigma1++) {
705 double_complex coef = {0.0, 0.0};
706
707 // yes dirty but loop over double is worst.
708 // since mj is only important for the rotation
709 // of the spherical harmonics the code takes
710 // into account this odd convention.
711
712 int jj1 = static_cast<int>(2.0 * j1 + 1e-8);
713 for (int mj = -jj1; mj <= jj1; mj += 2) {
714 coef += sht::calculate_U_sigma_m(l1, j1, mj, m1, sigma1) *
715 sht::ClebschGordan(l1, j1, mj / 2.0, sigma1) *
716 std::conj(sht::calculate_U_sigma_m(l2, j2, mj, m2, sigma2)) *
717 sht::ClebschGordan(l2, j2, mj / 2.0, sigma2);
718 }
719 f_coefficients_(xi1, xi2, sigma1, sigma2) = coef;
720 }
721 }
722 }
723 }
724 }

727void Atom_type::read_hubbard_input()
728{
729 if (!parameters_.cfg().parameters().hubbard_correction()) {
730 return;
731 }
732
733 this->hubbard_correction_ = false;
734
735 for (int i = 0; i < parameters_.cfg().hubbard().local().size(); i++) {
736 auto ho = parameters_.cfg().hubbard().local(i);
737 if (ho.atom_type() == this->label()) {
738 std::array<double, 6> coeff{0, 0, 0, 0, 0, 0};
739 if (ho.contains("U")) {
740 coeff[0] = ho.U();
741 }
742 if (ho.contains("J")) {
743 coeff[1] = ho.J();
744 }
745 if (ho.contains("BE2")) {
746 coeff[2] = ho.BE2();
747 }
748 if (ho.contains("E3")) {
749 coeff[3] = ho.E3();
750 }
751 if (ho.contains("alpha")) {
752 coeff[4] = ho.alpha();
753 }
754 if (ho.contains("beta")) {
755 coeff[5] = ho.beta();
756 }
757 /* now convert eV in Ha */
758 for (int s = 0; s < 6; s++) {
759 coeff[s] /= ha2ev;
760 }
761 std::vector<double> initial_occupancy;
762 if (ho.contains("initial_occupancy")) {
763 initial_occupancy = ho.initial_occupancy();
764
765 int sz = static_cast<int>(initial_occupancy.size());
766 int lmmax = 2 * ho.l() + 1;
767 if (!(sz == 0 || sz == lmmax || sz == 2 * lmmax)) {
768 std::stringstream s;
769 s << "wrong size of initial occupacies vector (" << sz << ") for l = " << ho.l();
770 RTE_THROW(s);
771 }
772 }
773
774 add_hubbard_orbital(ho.n(), ho.l(), ho.total_initial_occupancy(), coeff[0], coeff[1], &coeff[0],
775 coeff[4], coeff[5], 0.0, initial_occupancy);
776
777 this->hubbard_correction_ = true;
778 }
779 }

782void Atom_type::add_hubbard_orbital(int n__, int l__, double occ__, double U, double J, const double *hub_coef__,
783 double alpha__, double beta__, double J0__, std::vector<double> initial_occupancy__)
784{
785 // TODO: pass radial function for l or for j=l+1/2 j=l-1/2 and don't rely on the list of pseudoatomic wfs
786
787 if (n__ <= 0) {
788 RTE_THROW("negative principal quantum number");
789 }
790
791 /* we have to find one (or two in case of spin-orbit) atomic functions and construct hubbard orbital */
792 std::vector<int> idx_rf;
793 for (int s = 0; s < (int)ps_atomic_wfs_.size(); s++) {
794 auto& e = ps_atomic_wfs_[s];
795 int n = e.n;
796 auto aqn = e.am;
797 if (n == n__ && aqn.l() == l__) {
798 idx_rf.push_back(s);
799 /* in spin orbit case we need to find the second radial function, otherwise we break */
800 if (!(aqn.s() && aqn.l() > 0)) {
801 break;
802 }
803 }
804 }
805 if (idx_rf.size() == 0) {
806 std::stringstream s;
807 s << "atomic radial function is not found for atom type " << label_ << std::endl
808 << " the following atomic wave-functions are set: " << std::endl;
809 for (int k = 0; k < (int)ps_atomic_wfs_.size(); k++) {
810 auto& e = ps_atomic_wfs_[k];
811 int n = e.n;
812 auto aqn = e.am;
813 s << " n=" << n << " l=" << aqn.l() << " j=" << aqn.j() << std::endl;
814 }
815 s << " the following atomic orbital is requested for U-correction: n=" << n__ << " l=" << l__;
816 RTE_THROW(s);
817 }
818 if (idx_rf.size() > 2) {
819 std::stringstream s;
820 s << "number of atomic functions > 2";
821 RTE_THROW(s);
822 }
823
824 /* create a scalar hubbard wave-function from one or two atomic radial functions */
825 Spline<double> s(radial_grid_);
826 double f = 1.0 / static_cast<double>(idx_rf.size());
827 for (int i: idx_rf) {
828 auto& rwf = ps_atomic_wfs_[i].f;
829 for (int ir = 0; ir < s.num_points(); ir++) {
830 s(ir) += f * rwf(ir);
831 }
832 }
833
834 /* add a record in radial function index */
835 indexr_hub_.add(sirius::experimental::angular_momentum(l__));
836
837 /* add Hubbard orbital descriptor to a list */
838 lo_descriptors_hub_.emplace_back(n__, l__, -1, occ__, J, U, hub_coef__, alpha__, beta__, J0__, initial_occupancy__, std::move(s.interpolate()));
839}
