37 /// Select and create a new mixer.
38 /** \param [in] mix_cfg Parameters for mixer selection and creation.
39  * \param [in] comm Communicator passed to the mixer.
40  */
41 template <typename... FUNCS>
42 inline std::unique_ptr<Mixer<FUNCS...>> Mixer_factory(Mixer_input mix_cfg)
43 {
44  std::unique_ptr<Mixer<FUNCS...>> mixer;
45 
46  if (mix_cfg.type_ == "linear") {
47  mixer.reset(new Linear<FUNCS...>(mix_cfg.beta_));
48  } else if (mix_cfg.type_ == "broyden1") {
49  mixer.reset(new Broyden1<FUNCS...>(mix_cfg.max_history_, mix_cfg.beta_, mix_cfg.beta0_,
50  mix_cfg.beta_scaling_factor_));
51  } else if (mix_cfg.type_ == "broyden2") {
52  mixer.reset(new Broyden2<FUNCS...>(mix_cfg.max_history_, mix_cfg.beta_, mix_cfg.beta0_,
53  mix_cfg.beta_scaling_factor_, mix_cfg.linear_mix_rms_tol_));
54  } else {
55  TERMINATE("wrong type of mixer");
56  }
57  return mixer;
58 }
