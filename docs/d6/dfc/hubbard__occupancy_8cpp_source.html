25 /** Compute the occupation numbers associated to the hubbard wavefunctions (locally centered orbitals, wannier
26  * functions, etc) that are relevant for the hubbard correction.
27  *
28  * These quantities are defined by
29  * \f[
30  * n_{m,m'}^I \sigma = \sum_{kv} f(\varepsilon_{kv}) |<\psi_{kv}| phi_I_m>|^2
31  * \f]
32  * where \f[m=-l\cdot l$ (same for m')\f], I is the atom.
33  *
34  * Requires symmetrization. */

40 void Hubbard::symmetrize_occupancy_matrix(sddk::mdarray<double_complex, 4>& om__)
41 {
42  auto& sym = unit_cell_.symmetry();
43 
44  // check if we have some symmetries
45  if (sym.num_mag_sym()) {
46  int lmax = unit_cell_.lmax();
47  int lmmax = utils::lmmax(lmax);
48 
49  mdarray<double, 2> rotm(lmmax, lmmax);
50  mdarray<double_complex, 4> dm(om__.size(0), om__.size(1), om__.size(2), unit_cell_.num_atoms());
51  double alpha = 1.0 / static_cast<double>(sym.num_mag_sym());
52 
53  dm.zero();
54 
55  for (int i = 0; i < sym.num_mag_sym(); i++) {
56  int pr = sym.magnetic_group_symmetry(i).spg_op.proper;
57  auto eang = sym.magnetic_group_symmetry(i).spg_op.euler_angles;
58  int isym = sym.magnetic_group_symmetry(i).isym;
59  SHT::rotation_matrix(lmax, eang, pr, rotm);
60  auto spin_rot_su2 = rotation_matrix_su2(sym.magnetic_group_symmetry(i).spin_rotation);
61 
62  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
63  auto& atom_type = unit_cell_.atom(ia).type();
64  int ja = sym.sym_table(ia, isym);
65  if (atom_type.hubbard_correction()) {
66  sirius::symmetrize(om__, unit_cell_.atom(ia).type().indexb_hub(), ia, ja,
67  ctx_.num_mag_comp(), rotm, spin_rot_su2, dm, true);
68  }
69  }
70  }
71 
72  for (auto d3 = 0u; d3 < dm.size(3); d3++) {
73  for(auto d1 = 0u; d1 < dm.size(1); d1++) {
74  for(auto d0 = 0u; d0 < dm.size(0); d0++) {
75  dm(d0, d1, 0, d3) = dm(d0, d1, 0, d3) * alpha;
76  dm(d0, d1, 1, d3) = dm(d0, d1, 1, d3) * alpha;
77  dm(d0, d1, 2, d3) = std::conj(dm(d0, d1, 2, d3)) * alpha;
78  dm(d0, d1, 3, d3) = dm(d0, d1, 2, d3) * alpha;
79  }
80  }
81  }
82  dm >> om__;
83  }
84 }
85 
86 }
