25 /** Compute the occupation numbers associated to the hubbard wavefunctions (locally centered orbitals, wannier
26  * functions, etc) that are relevant for the hubbard correction.
27  *
28  * These quantities are defined by
29  * \f[
30  * n_{m,m'}^I \sigma = \sum_{kv} f(\varepsilon_{kv}) |<\psi_{kv}| phi_I_m>|^2
31  * \f]
32  * where \f[m=-l\cdot l$ (same for m')\f], I is the atom.
33  *
34  * Requires symmetrization. */
35 
36 #include "hubbard.hpp"
37 #include "symmetry/symmetrize.hpp"

39 namespace sirius {
40 void Hubbard::hubbard_compute_occupation_numbers(K_point_set& kset_)
41 {
42  PROFILE("sirius::Hubbard::compute_occupancy");
43 
44  if (!ctx_.hubbard_correction()) {
45  return;
46  }
47 
48  this->occupancy_number_.zero();
49 
50  int HowManyBands = -1;
51  /* if we are doing calculations for non colinear magnetism or
52  simple LDA then do not change the number of bands. the factor
53  two is important for colinear magnetism since the up-up and
54  down-down blocks are decoupled but the wave-functions are up
55  and down are still stored as a spinor to conserve space. */
56  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
57  int ik = kset_.spl_num_kpoints(ikloc);
58  auto kp = kset_[ik];
59 
60  HowManyBands = std::max(kp->num_occupied_bands(0), HowManyBands);
61  if (ctx_.num_mag_dims() == 1) {
62  HowManyBands = std::max(kp->num_occupied_bands(1), HowManyBands);
63  }
64  }
65 
66  /* now for each spin components and each atom we need to calculate
67  <psi_{nk}|phi^I_m'><phi^I_m|psi_{nk}> */
68  int Ncf = 1;
69  if (ctx_.num_mag_dims() == 1) {
70  /* !!! colinear magnetism. We need to have 2 times more space. */
71  Ncf = 2;
72  }
73 
74  dmatrix<double_complex> dm(HowManyBands, this->number_of_hubbard_orbitals() * Ncf);
75  matrix<double_complex> dm1(HowManyBands, this->number_of_hubbard_orbitals() * Ncf);
76  matrix<double_complex> Op(this->number_of_hubbard_orbitals() * Ncf, this->number_of_hubbard_orbitals() * Ncf);
77 
78  dm.zero();
79 
80  if (ctx_.processing_unit() == device_t::GPU) {
81  /* the communicator is always of size 1. I need to allocate memory
82  on the device manually */
83  dm.allocate(memory_t::device);
84  }
85  memory_t mem{memory_t::host};
86  linalg_t la{linalg_t::blas};
87  if (ctx_.processing_unit() == device_t::GPU) {
88  mem = memory_t::device;
89  la = linalg_t::gpublas;
90  }
91 
92  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
93  int ik = kset_.spl_num_kpoints(ikloc);
94  auto kp = kset_[ik];
95  if (ctx_.processing_unit() == device_t::GPU) {
96  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
97  /* allocate GPU memory */
98  kp->spinor_wave_functions().pw_coeffs(ispn).prime().allocate(memory_t::device);
99  kp->spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, kp->num_occupied_bands(ispn));
100  }
101 
102  for (int ispn = 0; ispn < kp->hubbard_wave_functions().num_sc(); ispn++) {
103 
104  if (!kp->hubbard_wave_functions().pw_coeffs(ispn).prime().on_device()) {
105  kp->hubbard_wave_functions().pw_coeffs(ispn).prime().allocate(memory_t::device);
106  }
107 
108  kp->hubbard_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, this->number_of_hubbard_orbitals());
109  }
110  }
111 
112  dm.zero();
113 
114  if (ctx_.num_mag_dims() == 3) {
115  /* full non colinear magnetism */
116  inner(mem, la, 2, kp->spinor_wave_functions(), 0, kp->num_occupied_bands(), kp->hubbard_wave_functions(), 0,
117  this->number_of_hubbard_orbitals(), dm, 0, 0);
118  } else {
119  /* SLDA + U, we need to do the explicit calculation. The hubbard
120  orbitals only have one component while the bloch wave functions
121  have two. The inner product takes care of this case internally. */
122 
123  for (int ispn_ = 0; ispn_ < ctx_.num_spins(); ispn_++) {
124  inner(mem, la, ispn_, kp->spinor_wave_functions(), 0, kp->num_occupied_bands(ispn_),
125  kp->hubbard_wave_functions(), 0, this->number_of_hubbard_orbitals(), dm, 0,
126  ispn_ * this->number_of_hubbard_orbitals());
127  }
128  }
129 
130  if (ctx_.processing_unit() == device_t::GPU) {
131  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
132  /* deallocate GPU memory */
133  kp->spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
134  }
135 
136  for (int ispn = 0; ispn < kp->hubbard_wave_functions().num_sc(); ispn++) {
137  kp->hubbard_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
138  }
139 
140  dm.copy_to(memory_t::host);
141  }
142 
143  // compute O'_{nk,j} = O_{nk,j} * f_{nk}
144  // NO summation over band yet
145 
146  dm1.zero(); // O'
147 
148  if (ctx_.num_mag_dims() == 3) {
149 #pragma omp parallel for
150  for (int m = 0; m < this->number_of_hubbard_orbitals(); m++) {
151  for (int nband = 0; nband < kp->num_occupied_bands(0); nband++) {
152  dm1(nband, m) = dm(nband, m) * kp->band_occupancy(nband, 0);
153  }
154  }
155  } else {
156 #pragma omp parallel for
157  for (int m = 0; m < this->number_of_hubbard_orbitals(); m++) {
158  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
159  for (int nband = 0; nband < kp->num_occupied_bands(ispn); nband++) {
160  dm1(nband, ispn * this->number_of_hubbard_orbitals() + m) =
161  dm(nband, ispn * this->number_of_hubbard_orbitals() + m) * kp->band_occupancy(nband, ispn);
162  }
163  }
164  }
165  }
166 
167  // now compute O_{ij}^{sigma,sigma'} = \sum_{nk} <psi_nk|phi_{i,sigma}><phi_{j,sigma^'}|psi_nk> f_{nk}
168  const double scal = (ctx_.num_mag_dims() == 0) ? 0.5 : 1.0;
169  auto alpha = double_complex(kp->weight() * scal, 0.0);
170  linalg(linalg_t::blas).gemm('C', 'N', this->number_of_hubbard_orbitals() * Ncf,
171  this->number_of_hubbard_orbitals() * Ncf, HowManyBands, &alpha, dm.at(memory_t::host), dm.ld(),
172  dm1.at(memory_t::host), dm1.ld(), &linalg_const<double_complex>::zero(), Op.at(memory_t::host), Op.ld());
173 
174  if (ctx_.num_mag_dims() == 3) {
175  // there must be a way to do that with matrix multiplication
176  #pragma omp parallel for schedule(static)
177  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
178  const auto& atom = unit_cell_.atom(ia);
179  if (atom.type().hubbard_correction()) {
180 
181  /* loop over the different channels */
182  /* note that for atom with SO interactions, we need to jump
183  by 2 instead of 1. This is due to the fact that the
184  relativistic wave functions have different total angular
185  momentum for the same n */
186 
187  for (int orb = 0; orb < atom.type().num_hubbard_orbitals(); orb += (atom.type().spin_orbit_coupling() ? 2 : 1)) {
188  /*
189  I know that the index of the hubbard wave functions (indexb_....) is
190  consistent with the index of the hubbard orbitals
191  */
192  const int lmax_at = 2 * atom.type().hubbard_orbital(0).l + 1;
193  for (int s1 = 0; s1 < ctx_.num_spins(); s1++) {
194  for (int s2 = 0; s2 < ctx_.num_spins(); s2++) {
195  int s = (s1 == s2) * s1 + (s1 != s2) * (1 + 2 * s2 + s1);
196  for (int mp = 0; mp < lmax_at; mp++) {
197  for (int m = 0; m < lmax_at; m++) {
198  this->occupancy_number_(m, mp, s, ia) +=
199  Op(this->offset_[ia] + m + s1 * lmax_at, this->offset_[ia] + mp + s2 * lmax_at);
200  }
201  }
202  }
203  }
204  }
205  }
206  }
207  } else {
208  // Well we need to apply a factor 1/2 (the constant scal
209  // above) when we compute the occupancies for the boring LDA
210  // + U. It is because the calculations of E and U consider
211  // occupancies <= 1. Sirius for the boring lda+U has a
212  // factor 2 in the kp band occupancies. We need to
213  // compensate for it because it is taken into account in the
214  // calculation of the hubbard potential
215 
216  #pragma omp parallel for schedule(static)
217  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
218  const auto& atom = unit_cell_.atom(ia);
219  if (atom.type().hubbard_correction()) {
220  for (int orb = 0; orb < atom.type().num_hubbard_orbitals(); orb++) {
221  const int lmax_at = 2 * atom.type().hubbard_orbital(0).l + 1;
222  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
223  for (int mp = 0; mp < lmax_at; mp++) {
224  const int mmp = this->offset_[ia] + mp + ispn * this->number_of_hubbard_orbitals();
225  for (int m = 0; m < lmax_at; m++) {
226  const int mm = this->offset_[ia] + m + ispn * this->number_of_hubbard_orbitals();
227  this->occupancy_number_(m, mp, ispn, ia) += Op(mm, mmp);
228  }
229  }
230  }
231  }
232  }
233  }
234  }
235  }
236 
237  /* global reduction over k points */
238  ctx_.comm_k().allreduce<double_complex, mpi_op_t::sum>(this->occupancy_number_.at(memory_t::host),
239  static_cast<int>(this->occupancy_number_.size()));
240 
241  // Now symmetrization procedure. We need to review that
242  //symmetrize_occupancy_matrix();
243 
244  print_occupancies();
245 }

247 // The initial occupancy is calculated following Hund rules. We first
248 // fill the d (f) states according to the hund's rules and with majority
249 // spin first and the remaining electrons distributed among the minority
250 // states.
251 void
252 Hubbard::calculate_initial_occupation_numbers()
253 {
254  this->occupancy_number_.zero();
255 #pragma omp parallel for schedule(static)
256  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
257  const auto& atom = unit_cell_.atom(ia);
258  if (atom.type().hubbard_correction()) {
259  const int lmax_at = 2 * atom.type().hubbard_orbital(0).l + 1;
260  // compute the total charge for the hubbard orbitals
261  double charge = atom.type().hubbard_orbital(0).occupancy();
262  bool nm = true; // true if the atom is non magnetic
263  int majs, mins;
264  if (ctx_.num_spins() != 1) {
265  if (atom.vector_field()[2] > 0.0) {
266  nm = false;
267  majs = 0;
268  mins = 1;
269  } else if (atom.vector_field()[2] < 0.0) {
270  nm = false;
271  majs = 1;
272  mins = 0;
273  }
274  }
275 
276  if (!nm) {
277  if (ctx_.num_mag_dims() != 3) {
278  // colinear case
279 
280  if (charge > (lmax_at)) {
281  for (int m = 0; m < lmax_at; m++) {
282  this->occupancy_number_(m, m, majs, ia) = 1.0;
283  this->occupancy_number_(m, m, mins, ia) =
284  (charge - static_cast<double>(lmax_at)) / static_cast<double>(lmax_at);
285  }
286  } else {
287  for (int m = 0; m < lmax_at; m++) {
288  this->occupancy_number_(m, m, majs, ia) = charge / static_cast<double>(lmax_at);
289  }
290  }
291  } else {
292  // double c1, s1;
293  // sincos(atom.type().starting_magnetization_theta(), &s1, &c1);
294  double c1 = atom.vector_field()[2];
295  double_complex cs = double_complex(atom.vector_field()[0], atom.vector_field()[1]) / sqrt(1.0 - c1 * c1);
296  double_complex ns[4];
297 
298  if (charge > (lmax_at)) {
299  ns[majs] = 1.0;
300  ns[mins] = (charge - static_cast<double>(lmax_at)) / static_cast<double>(lmax_at);
301  } else {
302  ns[majs] = charge / static_cast<double>(lmax_at);
303  ns[mins] = 0.0;
304  }
305 
306  // charge and moment
307  double nc = ns[majs].real() + ns[mins].real();
308  double mag = ns[majs].real() - ns[mins].real();
309 
310  // rotate the occ matrix
311  ns[0] = (nc + mag * c1) * 0.5;
312  ns[1] = (nc - mag * c1) * 0.5;
313  ns[2] = mag * std::conj(cs) * 0.5;
314  ns[3] = mag * cs * 0.5;
315 
316  for (int m = 0; m < lmax_at; m++) {
317  this->occupancy_number_(m, m, 0, ia) = ns[0];
318  this->occupancy_number_(m, m, 1, ia) = ns[1];
319  this->occupancy_number_(m, m, 2, ia) = ns[2];
320  this->occupancy_number_(m, m, 3, ia) = ns[3];
321  }
322  }
323  } else {
324  for (int s = 0; s < ctx_.num_spins(); s++) {
325  for (int m = 0; m < lmax_at; m++) {
326  this->occupancy_number_(m, m, s, ia) = charge * 0.5 / static_cast<double>(lmax_at);
327  }
328  }
329  }
330  }
331  }
332 
333  print_occupancies();
334 }

336 void Hubbard::print_occupancies()
337 {

345  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
346  std::printf("Atom : %d\n", ia);
347  std::printf("Mag Dim : %d\n", ctx_.num_mag_dims());
348  const auto& atom = unit_cell_.atom(ia);
349 
350  if (atom.type().hubbard_correction()) {
351  const int lmax_at = 2 * atom.type().hubbard_orbital(0).l + 1;
352  for (int m1 = 0; m1 < lmax_at; m1++) {

361  }

363  std::printf("\n");
364  }
365 
366  if (ctx_.num_spins() == 2) {
367  for (int m1 = 0; m1 < lmax_at; m1++) {
368  if (ctx_.num_mag_dims() == 3) {
369  for (int m2 = 0; m2 < lmax_at; m2++) {
370  std::printf("%.3lf ", std::abs(this->occupancy_number_(m1, m2, 3, ia)));
371  }
372  std::printf(" ");
373  }
374  for (int m2 = 0; m2 < lmax_at; m2++) {

392  }
393  std::printf("\n");
394  n_total = n_up + n_down;
395  if (ctx_.num_spins() == 2) {
396  std::printf("Atom charge (total) %.5lf (n_up) %.5lf (n_down) %.5lf (mz) %.5lf\n", n_total, n_up, n_down, n_up - n_down);
397  } else {
398  std::printf("Atom charge (total) %.5lf\n", 2.0 * n_total);
399  }
400 
401  std::printf("\n");
402  for (int ci = 0; ci < 10; ci++) {
403  std::printf("--------");
404  }
405  std::printf("\n");
406  }
407  }

411 void Hubbard::symmetrize_occupancy_matrix()
412 {
413  auto& sym = unit_cell_.symmetry();
414 
415  // check if we have some symmetries
416  if (sym.num_mag_sym()) {
417  int lmax = unit_cell_.lmax();
418  int lmmax = utils::lmmax(lmax);
419 
420  mdarray<double, 2> rotm(lmmax, lmmax);
421  mdarray<double_complex, 4> dm_(occupancy_number_.size(0),
422  occupancy_number_.size(1),
423  occupancy_number_.size(2),
424  unit_cell_.num_atoms());
425  double alpha = 1.0 / static_cast<double>(sym.num_mag_sym());
426 
427  dm_.zero();
428 
429  for (int i = 0; i < sym.num_mag_sym(); i++) {
430  int pr = sym.magnetic_group_symmetry(i).spg_op.proper;
431  auto eang = sym.magnetic_group_symmetry(i).spg_op.euler_angles;
432  int isym = sym.magnetic_group_symmetry(i).isym;
433  SHT::rotation_matrix(lmax, eang, pr, rotm);
434  auto spin_rot_su2 = rotation_matrix_su2(sym.magnetic_group_symmetry(i).spin_rotation);
435 
436  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
437  auto& atom_type = unit_cell_.atom(ia).type();
438  int ja = sym.sym_table(ia, isym);
439  if (atom_type.hubbard_correction()) {
440  sirius::symmetrize(occupancy_number_, unit_cell_.atom(ia).type().hubbard_indexb_wfc(), ia, ja,
441  ctx_.num_mag_comp(), rotm, spin_rot_su2, dm_, true);
442  }
443  }
444  }
445 
446  for (auto d3 = 0u; d3 < dm_.size(3); d3++) {
447  for(auto d1 = 0u; d1 < dm_.size(1); d1++) {
448  for(auto d0 = 0u; d0 < dm_.size(0); d0++) {
449  dm_(d0, d1, 0, d3) = dm_(d0, d1, 0, d3) * alpha;
450  dm_(d0, d1, 1, d3) = dm_(d0, d1, 1, d3) * alpha;
451  dm_(d0, d1, 2, d3) = std::conj(dm_(d0, d1, 2, d3)) * alpha;
452  dm_(d0, d1, 3, d3) = dm_(d0, d1, 2, d3) * alpha;
453  }
454  }
455  }
456  dm_ >> occupancy_number_;
457  }

461 /**
462  * retrieve or initialize the hubbard occupancies
463  *
464  * this functions helps retrieving or setting up the hubbard occupancy
465  * tensors from an external tensor. Retrieving it is done by specifying
466  * "get" in the first argument of the method while setting it is done
467  * with the parameter set up to "set". The second parameter is the
468  * output pointer and the last parameter is the leading dimension of the
469  * tensor.
470  *
471  * The returned result has the same layout than SIRIUS layout, * i.e.,
472  * the harmonic orbitals are stored from m_z = -l..l. The occupancy
473  * matrix can also be accessed through the method occupation_matrix()
474  *
475  *
476  * @param what__ string to set to "set" for initializing sirius
477  * occupancy tensor and "get" for retrieving it
478  * @param pointer to external occupancy tensor
479  * @param leading dimension of the outside tensor
480  * @return
481  * return the occupancy matrix if the first parameter is set to "get"
482  */
483 
484 void
485 Hubbard::access_hubbard_occupancies(char const* what__,
486  double_complex* occ__,
487  int const* ld__)
488 {
489  std::string what(what__);
490 
491  if (!(what == "get" || what == "set")) {
492  std::stringstream s;
493  s << "wrong access label: " << what;
494  TERMINATE(s);
495  }
496 
497  mdarray<double_complex, 4> occ_mtrx;
498  /* in non-collinear case the occupancy matrix is complex */
499  if (ctx_.num_mag_dims() == 3) {
500  occ_mtrx = mdarray<double_complex, 4>(reinterpret_cast<double_complex*>(occ__), *ld__, *ld__, 4, ctx_.unit_cell().num_atoms());
501  } else {
502  occ_mtrx = mdarray<double_complex, 4>(reinterpret_cast<double_complex*>(occ__), *ld__, *ld__, ctx_.num_spins(), ctx_.unit_cell().num_atoms());
503  }
504  if (what == "get") {
505  occ_mtrx.zero();
506  }
507 
508  auto& occupation_matrix = this->occupation_matrix();
509 
510  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
511  auto& atom = ctx_.unit_cell().atom(ia);
512  if (atom.type().hubbard_correction()) {
513  const int l = ctx_.unit_cell().atom(ia).type().hubbard_orbital(0).l;
514  for (int m1 = -l; m1 <= l; m1++) {
515  for (int m2 = -l; m2 <= l; m2++) {
516  if (what == "get") {
517  for (int j = 0; j < ((ctx_.num_mag_dims() == 3) ? 4 : ctx_.num_spins()); j++) {
518  occ_mtrx(l + m1, l + m2, j, ia) = occupation_matrix(l + m1, l + m2, j, ia);
519  }
520  } else {
521  for (int j = 0; j < ((ctx_.num_mag_dims() == 3) ? 4 : ctx_.num_spins()); j++) {
522  occupation_matrix(l + m1, l + m2, j, ia) = occ_mtrx(l + m1, l + m2, j, ia);
523  }
524  }
525  }
526  }
527  }
