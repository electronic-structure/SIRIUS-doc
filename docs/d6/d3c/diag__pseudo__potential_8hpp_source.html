35 template <typename T>
36 int Band::diag_pseudo_potential(K_point* kp__, Hamiltonian& H__) const
37 {
38  PROFILE("sirius::Band::diag_pseudo_potential");
39 
40  H__.local_op().prepare(kp__->gkvec_partition());
41  ctx_.fft_coarse().prepare(kp__->gkvec_partition());
42 
43  int niter{0};
44 
45  auto& itso = ctx_.iterative_solver_input();
46  if (itso.type_ == "exact") {
47  if (ctx_.num_mag_dims() != 3) {
48  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
49  diag_pseudo_potential_exact<double_complex>(kp__, ispn, H__);
50  }
51  } else {
52  STOP();
53  }
54  } else if (itso.type_ == "davidson") {
55  niter = diag_pseudo_potential_davidson<T>(kp__, H__);
56  } else if (itso.type_ == "rmm-diis") {
57  if (ctx_.num_mag_dims() != 3) {
58  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
59  diag_pseudo_potential_rmm_diis<T>(kp__, ispn, H__);
60  }
61  } else {
62  STOP();
63  }
64  } else if (itso.type_ == "chebyshev") {
65  P_operator<T> p_op(ctx_, kp__->p_mtrx());
66  if (ctx_.num_mag_dims() != 3) {
67  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
68  diag_pseudo_potential_chebyshev<T>(kp__, ispn, H__, p_op);
69  }
70  } else {
71  STOP();
72  }
73  } else {
74  TERMINATE("unknown iterative solver type");
75  }
76 
77  /* check residuals */
78  if (ctx_.control().verification_ >= 1) {
79  check_residuals<T>(kp__, H__);
80  }
81 
82  ctx_.fft_coarse().dismiss();
83  return niter;
84 }

86 template <typename T>
87 inline void Band::diag_pseudo_potential_exact(K_point* kp__,
88  int ispn__,
89  Hamiltonian& H__) const
90 {
91  PROFILE("sirius::Band::diag_pseudo_potential_exact");
92 
93  const int bs = ctx_.cyclic_block_size();
94  dmatrix<T> hmlt(kp__->num_gkvec(), kp__->num_gkvec(), ctx_.blacs_grid(), bs, bs);
95  dmatrix<T> ovlp(kp__->num_gkvec(), kp__->num_gkvec(), ctx_.blacs_grid(), bs, bs);
96  dmatrix<T> evec(kp__->num_gkvec(), kp__->num_gkvec(), ctx_.blacs_grid(), bs, bs);
97  std::vector<double> eval(kp__->num_gkvec());
98 
99  hmlt.zero();
100  ovlp.zero();
101 
102  auto gen_solver = ctx_.gen_evp_solver<T>();
103 
104  for (int ig = 0; ig < kp__->num_gkvec(); ig++) {
105  hmlt.set(ig, ig, 0.5 * std::pow(kp__->gkvec().gkvec_cart<index_domain_t::global>(ig).length(), 2));
106  ovlp.set(ig, ig, 1);
107  }
108 
109  auto veff = H__.potential().effective_potential().gather_f_pw();
110  std::vector<double_complex> beff;
111  if (ctx_.num_mag_dims() == 1) {
112  beff = H__.potential().effective_magnetic_field(0).gather_f_pw();
113  for (int ig = 0; ig < ctx_.gvec().num_gvec(); ig++) {
114  auto z1 = veff[ig];
115  auto z2 = beff[ig];
116  veff[ig] = z1 + z2;
117  beff[ig] = z1 - z2;
118  }
119  }
120 
121  #pragma omp parallel for schedule(static)
122  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) {
123  int ig_col = kp__->igk_col(igk_col);
124  auto gvec_col = kp__->gkvec().gvec(ig_col);
125  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) {
126  int ig_row = kp__->igk_row(igk_row);
127  auto gvec_row = kp__->gkvec().gvec(ig_row);
128  auto ig12 = ctx_.gvec().index_g12_safe(gvec_row, gvec_col);
129 
130  if (ispn__ == 0) {
131  if (ig12.second) {
132  hmlt(igk_row, igk_col) += std::conj(veff[ig12.first]);
133  } else {
134  hmlt(igk_row, igk_col) += veff[ig12.first];
135  }
136  } else {
137  if (ig12.second) {
138  hmlt(igk_row, igk_col) += std::conj(beff[ig12.first]);
139  } else {
140  hmlt(igk_row, igk_col) += beff[ig12.first];
141  }
142  }
143  }
144  }
145 
146  mdarray<double_complex, 2> dop(ctx_.unit_cell().max_mt_basis_size(), ctx_.unit_cell().max_mt_basis_size());
147  mdarray<double_complex, 2> qop(ctx_.unit_cell().max_mt_basis_size(), ctx_.unit_cell().max_mt_basis_size());
148 
149  mdarray<double_complex, 2> btmp(kp__->num_gkvec_row(), ctx_.unit_cell().max_mt_basis_size());
150 
151  kp__->beta_projectors_row().prepare();
152  kp__->beta_projectors_col().prepare();
153  for (int ichunk = 0; ichunk < kp__->beta_projectors_row().num_chunks(); ichunk++) {
154  /* generate beta-projectors for a block of atoms */
155  kp__->beta_projectors_row().generate(ichunk);
156  kp__->beta_projectors_col().generate(ichunk);
157 
158  auto& beta_row = kp__->beta_projectors_row().pw_coeffs_a();
159  auto& beta_col = kp__->beta_projectors_col().pw_coeffs_a();
160 
161  for (int i = 0; i < kp__->beta_projectors_row().chunk(ichunk).num_atoms_; i++) {
162  /* number of beta functions for a given atom */
163  int nbf = kp__->beta_projectors_row().chunk(ichunk).desc_(beta_desc_idx::nbf, i);
164  int offs = kp__->beta_projectors_row().chunk(ichunk).desc_(beta_desc_idx::offset, i);
165  int ia = kp__->beta_projectors_row().chunk(ichunk).desc_(beta_desc_idx::ia, i);
166 
167  const auto& augment_op = ctx_.augmentation_op(ctx_.unit_cell().atom(ia).type_id());
168 
169  for (int xi1 = 0; xi1 < nbf; xi1++) {
170  for (int xi2 = 0; xi2 < nbf; xi2++) {
171  if (ctx_.num_mag_dims() == 1) {
172  if (ispn__ == 0) {
173  dop(xi1, xi2) = ctx_.unit_cell().atom(ia).d_mtrx(xi1, xi2, 0) + ctx_.unit_cell().atom(ia).d_mtrx(xi1, xi2, 1);
174  } else {
175  dop(xi1, xi2) = ctx_.unit_cell().atom(ia).d_mtrx(xi1, xi2, 0) - ctx_.unit_cell().atom(ia).d_mtrx(xi1, xi2, 1);
176  }
177  } else {
178  dop(xi1, xi2) = ctx_.unit_cell().atom(ia).d_mtrx(xi1, xi2, 0);
179  }
180  if(augment_op.atom_type().augment()) {
181  qop(xi1, xi2) = augment_op.q_mtrx(xi1, xi2);
182  }
183  }
184  }
185  linalg<CPU>::gemm(0, 0, kp__->num_gkvec_row(), nbf, nbf,
186  &beta_row(0, offs), beta_row.ld(),
187  &dop(0, 0), dop.ld(),
188  &btmp(0, 0), btmp.ld());
189  linalg<CPU>::gemm(0, 2, kp__->num_gkvec_row(), kp__->num_gkvec_col(), nbf,
190  linalg_const<double_complex>::one(),
191  &btmp(0, 0), btmp.ld(),
192  &beta_col(0, offs), beta_col.ld(),
193  linalg_const<double_complex>::one(),
194  &hmlt(0, 0), hmlt.ld());
195  if(augment_op.atom_type().augment()) {
196  linalg<CPU>::gemm(0, 0, kp__->num_gkvec_row(), nbf, nbf,
197  &beta_row(0, offs), beta_row.ld(),
198  &qop(0, 0), qop.ld(),
199  &btmp(0, 0), btmp.ld());
200  }
201  linalg<CPU>::gemm(0, 2, kp__->num_gkvec_row(), kp__->num_gkvec_col(), nbf,
202  linalg_const<double_complex>::one(),
203  &btmp(0, 0), btmp.ld(),
204  &beta_col(0, offs), beta_col.ld(),
205  linalg_const<double_complex>::one(),
206  &ovlp(0, 0), ovlp.ld());
207  }
208  }
209  kp__->beta_projectors_row().dismiss();
210  kp__->beta_projectors_col().dismiss();
211 
212  if (gen_solver->solve(kp__->num_gkvec(), ctx_.num_bands(), hmlt, ovlp, eval.data(), evec)) {
213  std::stringstream s;
214  s << "error in full diagonalziation";
215  TERMINATE(s);
216  }
217 
218  for (int j = 0; j < ctx_.num_bands(); j++) {
219  kp__->band_energy(j, ispn__, eval[j]);
220  }
221 
222  kp__->spinor_wave_functions().pw_coeffs(ispn__).remap_from(evec, 0);
223 }

225 template <typename T>
226 inline int Band::diag_pseudo_potential_davidson(K_point* kp__,
227  Hamiltonian& H__) const
228 {
229  PROFILE("sirius::Band::diag_pseudo_potential_davidson");
230 
231  ctx_.print_memory_usage(__FILE__, __LINE__);
232 
233  auto& itso = ctx_.iterative_solver_input();
234 
235  bool converge_by_energy = (itso.converge_by_energy_ == 1);
236 
237  if (ctx_.control().verbosity_ >= 2 && kp__->comm().rank() == 0) {
238  printf("iterative solver tolerance: %18.12f\n", ctx_.iterative_solver_tolerance());
239  }
240 
241  /* true if this is a non-collinear case */
242  const bool nc_mag = (ctx_.num_mag_dims() == 3);
243 
244  /* number of spin components, treated simultaneously
245  * 1 - in case of non-magnetic or collinear calculation
246  * 2 - in case of non-collinear calculation
247  */
248  const int num_sc = nc_mag ? 2 : 1;
249 
250  /* short notation for number of target wave-functions */
251  const int num_bands = ctx_.num_bands();
252 
253  /* short notation for target wave-functions */
254  auto& psi = kp__->spinor_wave_functions();
255 
256  /* maximum subspace size */
257  int num_phi = itso.subspace_size_ * num_bands;
258 
259  if (num_phi > kp__->num_gkvec()) {
260  std::stringstream s;
261  s << "subspace size is too large!";
262  TERMINATE(s);
263  }
264  /* alias for memory pool */
265  auto& mp = ctx_.mem_pool(ctx_.host_memory_t());
266 
267  /* allocate wave-functions */
268  utils::timer t2("sirius::Band::diag_pseudo_potential_davidson|alloc");
269 
270  /* auxiliary wave-functions */
271  Wave_functions phi(mp, kp__->gkvec_partition(), num_phi, num_sc);
272 
273  /* Hamiltonian, applied to auxiliary wave-functions */
274  Wave_functions hphi(mp, kp__->gkvec_partition(), num_phi, num_sc);
275 
276  /* S operator, applied to auxiliary wave-functions */
277  Wave_functions sphi(mp, kp__->gkvec_partition(), num_phi, num_sc);
278 
279  /* Hamiltonain, applied to new Psi wave-functions */
280  Wave_functions hpsi(mp, kp__->gkvec_partition(), num_bands, num_sc);
281 
282  /* S operator, applied to new Psi wave-functions */
283  Wave_functions spsi(mp, kp__->gkvec_partition(), num_bands, num_sc);
284 
285  /* residuals */
286  Wave_functions res(mp, kp__->gkvec_partition(), num_bands, num_sc);
287 
288  const int bs = ctx_.cyclic_block_size();
289 
290  dmatrix<T> hmlt(mp, num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
291  dmatrix<T> ovlp(mp, num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
292  dmatrix<T> evec(mp, num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
293  dmatrix<T> hmlt_old(mp, num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
294  dmatrix<T> ovlp_old(mp, num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
295 
296  kp__->beta_projectors().prepare();
297 
298  if (is_device_memory(ctx_.aux_preferred_memory_t())) {
299  auto& mpd = ctx_.mem_pool(memory_t::device);
300  for (int i = 0; i < num_sc; i++) {
301  phi.pw_coeffs(i).allocate(mpd);
302  }
303  phi.preferred_memory_t(ctx_.aux_preferred_memory_t());
304  }
305 
306  if (is_device_memory(ctx_.preferred_memory_t())) {
307  auto& mpd = ctx_.mem_pool(memory_t::device);
308  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
309  psi.pw_coeffs(ispn).allocate(mpd);
310  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, num_bands);
311  }
312  psi.preferred_memory_t(ctx_.preferred_memory_t());
313  for (int i = 0; i < num_sc; i++) {
314  res.pw_coeffs(i).allocate(mpd);
315 
316  hphi.pw_coeffs(i).allocate(mpd);
317  sphi.pw_coeffs(i).allocate(mpd);
318 
319  hpsi.pw_coeffs(i).allocate(mpd);
320  spsi.pw_coeffs(i).allocate(mpd);
321  }
322  res.preferred_memory_t(ctx_.preferred_memory_t());
323  hphi.preferred_memory_t(ctx_.preferred_memory_t());
324  sphi.preferred_memory_t(ctx_.preferred_memory_t());
325  hpsi.preferred_memory_t(ctx_.preferred_memory_t());
326  spsi.preferred_memory_t(ctx_.preferred_memory_t());
327 
328  if (ctx_.blacs_grid().comm().size() == 1) {
329  evec.allocate(mpd);
330  ovlp.allocate(mpd);
331  hmlt.allocate(mpd);
332  }
333  }
334 
335  ctx_.print_memory_usage(__FILE__, __LINE__);
336  t2.stop();
337 
338  /* get diagonal elements for preconditioning */
339  auto h_diag = H__.get_h_diag<T>(kp__);
340  auto o_diag = H__.get_o_diag<T>(kp__);
341 
342  if (ctx_.control().print_checksum_) {
343  auto cs1 = h_diag.checksum();
344  auto cs2 = o_diag.checksum();
345  kp__->comm().allreduce(&cs1, 1);
346  kp__->comm().allreduce(&cs2, 1);
347  if (kp__->comm().rank() == 0) {
348  utils::print_checksum("h_diag", cs1);
349  utils::print_checksum("o_diag", cs2);
350  }
351  }
352 
353  auto std_solver = ctx_.std_evp_solver<T>();
354  auto gen_solver = ctx_.gen_evp_solver<T>();
355 
356  if (ctx_.control().print_checksum_) {
357  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
358  auto cs = psi.checksum_pw(get_device_t(psi.preferred_memory_t()), ispn, 0, num_bands);
359  std::stringstream s;
360  s << "input spinor_wave_functions_" << ispn;
361  if (kp__->comm().rank() == 0) {
362  utils::print_checksum(s.str(), cs);
363  }
364  }
365  }
366 
367  int niter{0};
368 
369  utils::timer t3("sirius::Band::diag_pseudo_potential_davidson|iter");
370  for (int ispin_step = 0; ispin_step < ctx_.num_spin_dims(); ispin_step++) {
371 
372  std::vector<double> eval(num_bands);
373  std::vector<double> eval_old(num_bands, 1e100);
374 
375  if (itso.init_eval_old_) {
376  for (int j = 0; j < num_bands; j++) {
377  eval_old[j] = kp__->band_energy(j, ispin_step);
378  }
379  }
380 
381  /* trial basis functions */
382  for (int ispn = 0; ispn < num_sc; ispn++) {
383  phi.copy_from(psi, num_bands, nc_mag ? ispn : ispin_step, 0, ispn, 0);
384  }
385  if (ctx_.control().print_checksum_) {
386  for (int ispn = 0; ispn < num_sc; ispn++) {
387  auto cs = phi.checksum_pw(get_device_t(phi.preferred_memory_t()), ispn, 0, num_bands);
388  std::stringstream s;
389  s << "input phi" << ispn;
390  if (kp__->comm().rank() == 0) {
391  utils::print_checksum(s.str(), cs);
392  }
393  }
394  }
395 
396  /* fisrt phase: setup and diagonalize reduced Hamiltonian and get eigen-values;
397  * this is done before the main itertive loop */
398 
399  /* apply Hamiltonian and S operators to the basis functions */
400  H__.apply_h_s<T>(kp__, nc_mag ? 2 : ispin_step, 0, num_bands, phi, &hphi, &sphi);
401 
402  /* setup eigen-value problem
403  * N is the number of previous basis functions
404  * n is the number of new basis functions */
405  set_subspace_mtrx(0, num_bands, phi, hphi, hmlt, hmlt_old);
406  /* setup overlap matrix */
407  set_subspace_mtrx(0, num_bands, phi, sphi, ovlp, ovlp_old);
408 
409  if (ctx_.control().verification_ >= 1) {
410  double max_diff = check_hermitian(hmlt, num_bands);
411  if (max_diff > 1e-12) {
412  std::stringstream s;
413  s << "H matrix is not hermitian, max_err = " << max_diff;
414  WARNING(s);
415  }
416  max_diff = check_hermitian(ovlp, num_bands);
417  if (max_diff > 1e-12) {
418  std::stringstream s;
419  s << "S matrix is not hermitian, max_err = " << max_diff;
420  WARNING(s);
421  }
422 
423  if (ctx_.control().verification_ >= 2) {
424  hmlt.serialize("H matrix", num_bands);
425  }
426  if (ctx_.control().verification_ >= 2) {
427  ovlp.serialize("S matrix", num_bands);
428  }
429  }
430 
431  /* current subspace size */
432  int N = num_bands;
433 
434  utils::timer t1("sirius::Band::diag_pseudo_potential_davidson|evp");
435  /* solve generalized eigen-value problem with the size N and get lowest num_bands eigen-vectors */
436  if (gen_solver->solve(N, num_bands, hmlt, ovlp, eval.data(), evec)) {
437  std::stringstream s;
438  s << "error in diagonalziation";
439  TERMINATE(s);
440  }
441  t1.stop();
442 
443  evp_work_count() += 1;
444 
445  if (ctx_.control().verbosity_ >= 4 && kp__->comm().rank() == 0) {
446  for (int i = 0; i < num_bands; i++) {
447  printf("eval[%i]=%20.16f\n", i, eval[i]);
448  }
449  }
450 
451  /* number of newly added basis functions */
452  int n{0};
453 
454  /* second phase: start iterative diagonalization */
455  for (int k = 0; k < itso.num_steps_; k++) {
456 
457  /* don't compute residuals on last iteration */
458  if (k != itso.num_steps_ - 1) {
459  /* get new preconditionined residuals, and also hpsi and opsi as a by-product */
460  n = residuals<T>(kp__, nc_mag ? 2 : ispin_step, N, num_bands, eval, eval_old, evec, hphi,
461  sphi, hpsi, spsi, res, h_diag, o_diag);
462  }
463 
464  /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
465  if (N + n > num_phi || n <= itso.min_num_res_ || k == (itso.num_steps_ - 1)) {
466  utils::timer t1("sirius::Band::diag_pseudo_potential_davidson|update_phi");
467  /* recompute wave-functions */
468  /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
469  if (ctx_.settings().always_update_wf_ || k + n > 0) {
470  /* in case of non-collinear magnetism transform two components */
471  transform<T>(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), nc_mag ? 2 : ispin_step, {&phi}, 0, N, evec, 0, 0,
472  {&psi}, 0, num_bands);
473  /* update eigen-values */
474  for (int j = 0; j < num_bands; j++) {
475  kp__->band_energy(j, ispin_step, eval[j]);
476  }
477  } else {
478  if (ctx_.control().verbosity_ >= 2 && kp__->comm().rank() == 0) {
479  printf("wave-functions are not recomputed\n");
480  }
481  }
482 
483  /* exit the loop if the eigen-vectors are converged or this is a last iteration */
484  if (n <= itso.min_num_res_ || k == (itso.num_steps_ - 1)) {
485  break;
486  } else { /* otherwise, set Psi as a new trial basis */
487  if (ctx_.control().verbosity_ >= 3 && kp__->comm().rank() == 0) {
488  printf("subspace size limit reached\n");
489  }
490  hmlt_old.zero();
491  for (int i = 0; i < num_bands; i++) {
492  hmlt_old.set(i, i, eval[i]);
493  }
494  if (!itso.orthogonalize_) {
495  ovlp_old.zero();
496  for (int i = 0; i < num_bands; i++) {
497  ovlp_old.set(i, i, 1);
498  }
499  }
500 
501  /* need to compute all hpsi and opsi states (not only unconverged) */
502  if (converge_by_energy) {
503  transform<T>(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), nc_mag ? 2 : ispin_step, 1.0,
504  std::vector<Wave_functions*>({&hphi, &sphi}), 0, N, evec, 0, 0, 0.0,
505  {&hpsi, &spsi}, 0, num_bands);
506  }
507 
508  /* update basis functions, hphi and ophi */
509  for (int ispn = 0; ispn < num_sc; ispn++) {
510  phi.copy_from(psi, num_bands, nc_mag ? ispn : ispin_step, 0, nc_mag ? ispn : 0, 0);
511  hphi.copy_from(hpsi, num_bands, ispn, 0, ispn, 0);
512  sphi.copy_from(spsi, num_bands, ispn, 0, ispn, 0);
513  }
514  /* number of basis functions that we already have */
