336 inline void Atom_symmetry_class::generate_aw_radial_functions(relativity_t rel__)
337 {
338  int nmtp = atom_type_.num_mt_points();
339 
340  Radial_solver solver(atom_type_.zn(), spherical_potential_, atom_type_.radial_grid());
341 
342  #pragma omp parallel default(shared)
343  {
344  Spline<double> s(atom_type_.radial_grid());
345 
346  std::vector<double> p;
347  std::vector<double> rdudr;
348  std::array<double, 2> uderiv;
349 
350  #pragma omp for schedule(dynamic, 1)
351  for (int l = 0; l < num_aw_descriptors(); l++) {
352  for (int order = 0; order < (int)aw_descriptor(l).size(); order++) {
353  auto rsd = aw_descriptor(l)[order];
354 
355  int idxrf = atom_type_.indexr().index_by_l_order(l, order);
356 
357  solver.solve(rel__, rsd.dme, rsd.l, rsd.enu, p, rdudr, uderiv);
358 
359  /* normalize */
360  for (int ir = 0; ir < nmtp; ir++) {
361  s(ir) = std::pow(p[ir], 2);
362  }
363  double norm = 1.0 / std::sqrt(s.interpolate().integrate(0));
364 
365  for (int ir = 0; ir < nmtp; ir++) {
366  radial_functions_(ir, idxrf, 0) = p[ir] * norm;
367  radial_functions_(ir, idxrf, 1) = rdudr[ir] * norm;
368  }
369  aw_surface_derivatives_(order, l, 0) = norm * p.back() / atom_type_.mt_radius();
370  for (int i : {0, 1}) {
371  aw_surface_derivatives_(order, l, i + 1) = uderiv[i] * norm;
372  }
373 
374  /* orthogonalize to previous radial functions */
375  for (int order1 = 0; order1 < order; order1++) {
376  int idxrf1 = atom_type_.indexr().index_by_l_order(l, order1);
377 
378  for (int ir = 0; ir < nmtp; ir++) {
379  s(ir) = radial_functions_(ir, idxrf, 0) * radial_functions_(ir, idxrf1, 0);
380  }
381 
382  /* <u_{\nu'}|u_{\nu}> */
383  double ovlp = s.interpolate().integrate(0);
384 
385  for (int ir = 0; ir < nmtp; ir++) {
386  radial_functions_(ir, idxrf, 0) -= radial_functions_(ir, idxrf1, 0) * ovlp;
387  radial_functions_(ir, idxrf, 1) -= radial_functions_(ir, idxrf1, 1) * ovlp;
388  }
389  for (int i : {0, 1, 2}) {
390  aw_surface_derivatives_(order, l, i) -= aw_surface_derivatives_(order1, l, i) * ovlp;
391  }
392  }
393 
394  /* normalize again */
395  for (int ir = 0; ir < nmtp; ir++) {
396  s(ir) = std::pow(radial_functions_(ir, idxrf, 0), 2);
397  }
398  norm = s.interpolate().integrate(0);
399 
400  if (std::abs(norm) < 1e-10) {
401  std::stringstream s;
402  s << "AW radial function for atom " << atom_type_.label() << " is linearly dependent" << std::endl
403  << " order: " << order << std::endl
404  << " l: " << l << std::endl
405  << " dme: " << rsd.dme << std::endl
406  << " enu: " << rsd.enu;
407  TERMINATE(s);
408  }
409 
410  norm = 1.0 / std::sqrt(norm);
411 
412  for (int ir = 0; ir < nmtp; ir++) {
413  radial_functions_(ir, idxrf, 0) *= norm;
414  radial_functions_(ir, idxrf, 1) *= norm;
415  }
416  for (int i : {0, 1, 2}) {
417  aw_surface_derivatives_(order, l, i) *= norm;
418  }
419  }
420  /* divide by r */
421  for (int order = 0; order < (int)aw_descriptor(l).size(); order++) {
422  int idxrf = atom_type_.indexr().index_by_l_order(l, order);
423  for (int ir = 0; ir < nmtp; ir++) {
424  radial_functions_(ir, idxrf, 0) *= atom_type_.radial_grid().x_inv(ir);
425  }
426  }
427  }
428  }
429 }

431 inline void Atom_symmetry_class::generate_lo_radial_functions(relativity_t rel__)
432 {
433  int nmtp = atom_type_.num_mt_points();
434 
435  Radial_solver solver(atom_type_.zn(), spherical_potential_, atom_type_.radial_grid());
436 
437  #pragma omp parallel default(shared)
438  {
439  Spline<double> s(atom_type_.radial_grid());
440 
441  double a[3][3];
442 
443  #pragma omp for schedule(dynamic, 1)
444  for (int idxlo = 0; idxlo < num_lo_descriptors(); idxlo++) {
445  /* number of radial solutions */
446  int num_rs = static_cast<int>(lo_descriptor(idxlo).rsd_set.size());
447  assert(num_rs <= 3);
448 
449  std::vector<std::vector<double>> p(num_rs);
450  std::vector<std::vector<double>> rdudr(num_rs);
451  std::array<double, 2> uderiv;
452 
453  for (int order = 0; order < num_rs; order++) {
454  auto rsd = lo_descriptor(idxlo).rsd_set[order];
455 
456  solver.solve(rel__, rsd.dme, rsd.l, rsd.enu, p[order], rdudr[order], uderiv);
457 
458  /* find norm of the radial solution */
459  for (int ir = 0; ir < nmtp; ir++) {
460  s(ir) = std::pow(p[order][ir], 2);
461  }
462  double norm = 1.0 / std::sqrt(s.interpolate().integrate(0));
463 
464  /* normalize radial solution and divide by r */
465  for (int ir = 0; ir < nmtp; ir++) {
466  p[order][ir] *= (norm * atom_type_.radial_grid().x_inv(ir));
467  /* don't divide rdudr by r */
468  rdudr[order][ir] *= norm;
469  }
470  uderiv[0] *= norm;
471  uderiv[1] *= norm;
472 
473  /* matrix of derivatives */
474  a[order][0] = p[order].back();
475  a[order][1] = uderiv[0];
476  a[order][2] = uderiv[1];
477  }
478 
479  double b[] = {0, 0, 0};
480  b[num_rs - 1] = 1.0;
481 
482  int info = linalg<CPU>::gesv(num_rs, 1, &a[0][0], 3, b, 3);
483 
484  if (info) {
485  std::stringstream s;
486  s << "a[i][j] = ";
487  for (int i = 0; i < num_rs; i++) {
488  for (int j = 0; j < num_rs; j++) {
489  s << a[i][j] << " ";
490  }
491  }
492  s << std::endl;
493  s << "atom: " << atom_type_.label() << std::endl
494  << "zn: " << atom_type_.zn() << std::endl
495  << "l: " << lo_descriptor(idxlo).l << std::endl;
496  s << "gesv returned " << info;
497  TERMINATE(s);
498  }
499 
500  /* index of local orbital radial function */
501  int idxrf = atom_type_.indexr().index_by_idxlo(idxlo);
502  /* take linear combination of radial solutions */
503  for (int order = 0; order < num_rs; order++) {
504  for (int ir = 0; ir < nmtp; ir++) {
505  radial_functions_(ir, idxrf, 0) += b[order] * p[order][ir];
506  radial_functions_(ir, idxrf, 1) += b[order] * rdudr[order][ir];
507  }
508  }
509 
510  /* find norm of constructed local orbital */
511  for (int ir = 0; ir < nmtp; ir++) {
512  s(ir) = std::pow(radial_functions_(ir, idxrf, 0), 2);
513  }
514  double norm = 1.0 / std::sqrt(s.interpolate().integrate(2));
515 
516  /* normalize */
517  for (int ir = 0; ir < nmtp; ir++) {
518  radial_functions_(ir, idxrf, 0) *= norm;
519  radial_functions_(ir, idxrf, 1) *= norm;
520  }
521 
522  if (std::abs(radial_functions_(nmtp - 1, idxrf, 0)) > 1e-10) {
523  std::stringstream s;
524  s << "local orbital " << idxlo << " is not zero at MT boundary" << std::endl
525  << " atom symmetry class id : " << id() << " (" << atom_type().symbol() << ")" << std::endl
526  << " value : " << radial_functions_(nmtp - 1, idxrf, 0) << std::endl
527  << " number of MT points: " << nmtp << std::endl
528  << " MT radius: " << atom_type_.radial_grid().last() << std::endl
529  << " b_coeffs: ";
530  for (int j = 0; j < num_rs; j++) {
531  s << b[j] << " ";
532  }
533  WARNING(s);
534  }
535  }
536  }
537 
538  if (atom_type_.parameters().control().verification_ > 0 && num_lo_descriptors() > 0) {
539  check_lo_linear_independence(0.0001);
540  }
541 }

543 inline std::vector<int> Atom_symmetry_class::check_lo_linear_independence(double tol__)
544 {
545  int nmtp = atom_type_.num_mt_points();
546 
547  Spline<double> s(atom_type_.radial_grid());
548  dmatrix<double> loprod(num_lo_descriptors(), num_lo_descriptors());
549  loprod.zero();
550  for (int idxlo1 = 0; idxlo1 < num_lo_descriptors(); idxlo1++) {
551 
552  int idxrf1 = atom_type_.indexr().index_by_idxlo(idxlo1);
553 
554  for (int idxlo2 = 0; idxlo2 < num_lo_descriptors(); idxlo2++) {
555 
556  int idxrf2 = atom_type_.indexr().index_by_idxlo(idxlo2);
557 
558  if (lo_descriptor(idxlo1).l == lo_descriptor(idxlo2).l) {
559 
560  for (int ir = 0; ir < nmtp; ir++) {
561  s(ir) = radial_functions_(ir, idxrf1, 0) * radial_functions_(ir, idxrf2, 0);
562  }
563  loprod(idxlo1, idxlo2) = s.interpolate().integrate(2);
564  }
565  }
566  }
567 
568  mdarray<double, 2> ovlp(num_lo_descriptors(), num_lo_descriptors());
569  loprod >> ovlp;
570 
571  Eigensolver_lapack<double> stdevp;
572 
573  std::vector<double> loprod_eval(num_lo_descriptors());
574  dmatrix<double> loprod_evec(num_lo_descriptors(), num_lo_descriptors());
575 
576  stdevp.solve(num_lo_descriptors(), loprod, &loprod_eval[0], loprod_evec);
577 
578  if (std::abs(loprod_eval[0]) < tol__) {
579  printf("\n");
580  printf("local orbitals for atom symmetry class %i are almost linearly dependent\n", id_);
581  printf("local orbitals overlap matrix:\n");
582  for (int i = 0; i < num_lo_descriptors(); i++) {
583  for (int j = 0; j < num_lo_descriptors(); j++) {
584  printf("%12.6f", ovlp(i, j));
585  }
586  printf("\n");
587  }
588  printf("overlap matrix eigen-values:\n");
589  for (int i = 0; i < num_lo_descriptors(); i++) {
590  printf("%12.6f", loprod_eval[i]);
591  }
592  printf("\n");
593  printf("smallest eigenvalue: %20.16f\n", loprod_eval[0]);
594  }
595 
596  std::vector<int> inc(num_lo_descriptors(), 0);
597 
598  /* try all local orbitals */
599  for (int i = 0; i < num_lo_descriptors(); i++) {
600  inc[i] = 1;
601 
602  std::vector<int> ilo;
603  for (int j = 0; j < num_lo_descriptors(); j++) {
604  if (inc[j] == 1) {
605  ilo.push_back(j);
606  }
607  }
608 
609  std::vector<double> eval(ilo.size());
610  dmatrix<double> evec(static_cast<int>(ilo.size()), static_cast<int>(ilo.size()));
611  dmatrix<double> tmp(static_cast<int>(ilo.size()), static_cast<int>(ilo.size()));
612  for (size_t j1 = 0; j1 < ilo.size(); j1++) {

790 inline void Atom_symmetry_class::generate_radial_integrals(relativity_t rel__)
791 {
792  PROFILE("sirius::Atom_symmetry_class::generate_radial_integrals");
793 
794  int nmtp = atom_type_.num_mt_points();
795 
796  double sq_alpha_half = 0.5 * std::pow(speed_of_light, -2);
797  if (rel__ == relativity_t::none) {
798  sq_alpha_half = 0;
799  }
800 
801  h_spherical_integrals_.zero();
802  #pragma omp parallel default(shared)
803  {
804  Spline<double> s(atom_type_.radial_grid());
805  #pragma omp for
806  for (int i1 = 0; i1 < atom_type_.mt_radial_basis_size(); i1++) {
807  for (int i2 = 0; i2 < atom_type_.mt_radial_basis_size(); i2++) {
808  /* for spherical part of potential integrals are diagonal in l */
809  if (atom_type_.indexr(i1).l == atom_type_.indexr(i2).l) {
810  int ll = atom_type_.indexr(i1).l * (atom_type_.indexr(i1).l + 1);
811  for (int ir = 0; ir < nmtp; ir++) {
812  double Minv = 1.0 / (1 - spherical_potential_[ir] * sq_alpha_half);
813  /* u_1(r) * u_2(r) */
814  double t0 = radial_functions_(ir, i1, 0) * radial_functions_(ir, i2, 0);
815  /* r*u'_1(r) * r*u'_2(r) */
816  double t1 = radial_functions_(ir, i1, 1) * radial_functions_(ir, i2, 1);
817  s(ir) = 0.5 * t1 * Minv + t0 * (0.5 * ll * Minv + spherical_potential_[ir] * std::pow(atom_type_.radial_grid(ir), 2));
818  }
819  h_spherical_integrals_(i1, i2) = s.interpolate().integrate(0) / y00;
820  }
821  }
822  }
823  }
824 
825  o_radial_integrals_.zero();
826  #pragma omp parallel default(shared)
827  {
828  Spline<double> s(atom_type_.radial_grid());
829  #pragma omp for
830  for (int l = 0; l <= atom_type_.indexr().lmax(); l++) {
831  int nrf = atom_type_.indexr().num_rf(l);
832 
833  for (int order1 = 0; order1 < nrf; order1++) {
834  int idxrf1 = atom_type_.indexr().index_by_l_order(l, order1);
835  for (int order2 = 0; order2 < nrf; order2++) {
836  int idxrf2 = atom_type_.indexr().index_by_l_order(l, order2);
837  if (order1 == order2) {
838  o_radial_integrals_(l, order1, order2) = 1.0;
839  } else {
840  for (int ir = 0; ir < nmtp; ir++) {
841  s(ir) = radial_functions_(ir, idxrf1, 0) * radial_functions_(ir, idxrf2, 0);
842  }
843  o_radial_integrals_(l, order1, order2) = s.interpolate().integrate(2);
844  }
845  }
846  }
847  }
848  }
849  if (atom_type_.parameters().valence_relativity() == relativity_t::iora) {
850  o1_radial_integrals_.zero();
851  #pragma omp parallel default(shared)
852  {
853  Spline<double> s(atom_type_.radial_grid());
854  #pragma omp for
855  for (int i1 = 0; i1 < atom_type_.mt_radial_basis_size(); i1++) {
856  for (int i2 = 0; i2 < atom_type_.mt_radial_basis_size(); i2++) {
857  /* for spherical part of potential integrals are diagonal in l */
858  if (atom_type_.indexr(i1).l == atom_type_.indexr(i2).l) {
859  int ll = atom_type_.indexr(i1).l * (atom_type_.indexr(i1).l + 1);
860  for (int ir = 0; ir < nmtp; ir++) {
861  double Minv = std::pow(1 - spherical_potential_[ir] * sq_alpha_half, -2);
862  /* u_1(r) * u_2(r) */
863  double t0 = radial_functions_(ir, i1, 0) * radial_functions_(ir, i2, 0);
864  /* r*u'_1(r) * r*u'_2(r) */
865  double t1 = radial_functions_(ir, i1, 1) * radial_functions_(ir, i2, 1);
866  s(ir) = sq_alpha_half * 0.5 * Minv * (t1 + t0 * 0.5 * ll);
867  }
868  o1_radial_integrals_(i1, i2) = s.interpolate().integrate(0);
869  }
870  }
871  }
872  }
873  }
874 
875  if (false) // TODO: if it's slow, compute only when spin-orbit is turned on
876  {
877  double soc = std::pow(2 * speed_of_light, -2);
878 
879  Spline<double> s(atom_type_.radial_grid());
880  Spline<double> s1(atom_type_.radial_grid());
881  Spline<double> ve(atom_type_.radial_grid());
882 
883  for (int i = 0; i < nmtp; i++) {
884  ve(i) = spherical_potential_[i] + atom_type_.zn() / atom_type_.radial_grid(i);
885  }
886  ve.interpolate();
887 
888  so_radial_integrals_.zero();
889  for (int l = 0; l <= atom_type_.indexr().lmax(); l++) {
890  int nrf = atom_type_.indexr().num_rf(l);
891 
892  for (int order1 = 0; order1 < nrf; order1++) {
893  int idxrf1 = atom_type_.indexr().index_by_l_order(l, order1);
894  for (int order2 = 0; order2 < nrf; order2++) {
895  int idxrf2 = atom_type_.indexr().index_by_l_order(l, order2);
896 
897  for (int ir = 0; ir < nmtp; ir++) {
898  double M = 1.0 - 2 * soc * spherical_potential_[ir];
899  /* first part <f| dVe / dr |f'> */
900  s(ir) = radial_functions_(ir, idxrf1, 0) * radial_functions_(ir, idxrf2, 0) *
901  soc * ve.deriv(1, ir) / pow(M, 2);
902 
903  /* second part <f| d(z/r) / dr |f'> */
904  s1(ir) = radial_functions_(ir, idxrf1, 0) * radial_functions_(ir, idxrf2, 0) *
905  soc * atom_type_.zn() / pow(M, 2);
906  }
907  s.interpolate();
908  s1.interpolate();
909 
910  so_radial_integrals_(l, order1, order2) = s.integrate(1) + s1.integrate(-1);
911  }
912  }
913  }
914  }
915 }

952 inline void Atom_symmetry_class::generate_core_charge_density(relativity_t core_rel__)
953 {
954  PROFILE("sirius::Atom_symmetry_class::generate_core_charge_density");
955 
956  /* nothing to do */
957  if (atom_type_.num_core_electrons() == 0.0) {
958  return;
959  }
960 
961  int nmtp = atom_type_.num_mt_points();
962 
963  std::vector<double> free_atom_grid(nmtp);
964  for (int i = 0; i < nmtp; i++) {
965  free_atom_grid[i] = atom_type_.radial_grid(i);
966  }
967 
968  /* extend radial grid */
969  double x = atom_type_.radial_grid(nmtp - 1);
970  double dx = atom_type_.radial_grid().dx(nmtp - 2);
971  while (x < 30.0 + atom_type_.zn() / 4.0) {
972  x += dx;
973  free_atom_grid.push_back(x);
974  dx *= 1.025;
975  }
976  Radial_grid_ext<double> rgrid(static_cast<int>(free_atom_grid.size()), free_atom_grid.data());
977 
978  /* interpolate spherical potential inside muffin-tin */
979  Spline<double> svmt(atom_type_.radial_grid());
980  /* remove nucleus contribution from Vmt */
981  for (int ir = 0; ir < nmtp; ir++) {
982  svmt(ir) = spherical_potential_[ir] + atom_type_.zn() * atom_type_.radial_grid().x_inv(ir);
983  }
984  svmt.interpolate();
985  /* fit tail to alpha/r + beta */
986  double alpha = -(std::pow(atom_type_.mt_radius(), 2) * svmt.deriv(1, nmtp - 1) + atom_type_.zn());
987  double beta = svmt(nmtp - 1) - (atom_type_.zn() + alpha) / atom_type_.mt_radius();
988 
989  /* cook an effective potential from muffin-tin part and a tail */
990  std::vector<double> veff(rgrid.num_points());
991  for (int ir = 0; ir < nmtp; ir++) {
992  veff[ir] = spherical_potential_[ir];
993  }
994  /* simple tail alpha/r + beta */
995  for (int ir = nmtp; ir < rgrid.num_points(); ir++) {
996  veff[ir] = alpha * rgrid.x_inv(ir) + beta;
997  }
998 
999  //== /* write spherical potential */
1000  //== std::stringstream sstr;
1001  //== sstr << "spheric_potential_" << id_ << ".dat";
1002  //== FILE* fout = fopen(sstr.str().c_str(), "w");
1003 
1004  //== for (int ir = 0; ir < rgrid.num_points(); ir++)
1005  //== {
1006  //== fprintf(fout, "%18.10f %18.10f\n", rgrid[ir], veff[ir]);
1007  //== }
1008  //== fclose(fout);
1009  //== STOP();
1010 
1011  /* charge density */
1012  Spline<double> rho(rgrid);
1013 
1014  /* atomic level energies */
1015  std::vector<double> level_energy(atom_type_.num_atomic_levels());
1016 
1017  for (int ist = 0; ist < atom_type_.num_atomic_levels(); ist++) {
1018  level_energy[ist] = -1.0 * atom_type_.zn() / 2 / std::pow(double(atom_type_.atomic_level(ist).n), 2);
1019  }
1020 
1021  #pragma omp parallel default(shared)
1022  {
1023  std::vector<double> rho_t(rho.num_points());
1024  std::memset(&rho_t[0], 0, rho.num_points() * sizeof(double));
1025 
1026  #pragma omp for
1027  for (int ist = 0; ist < atom_type_.num_atomic_levels(); ist++) {
1028  if (atom_type_.atomic_level(ist).core) {
1029  Bound_state bs(core_rel__, atom_type_.zn(), atom_type_.atomic_level(ist).n, atom_type_.atomic_level(ist).l,
1030  atom_type_.atomic_level(ist).k, rgrid, veff, level_energy[ist]);
1031 
1032  auto& rho = bs.rho();
1033  for (int i = 0; i < rgrid.num_points(); i++) {
1034  rho_t[i] += atom_type_.atomic_level(ist).occupancy * rho(i) / fourpi;
1035  }
1036 
1037  level_energy[ist] = bs.enu();
1038  }
1039  }
1040 
1041  #pragma omp critical
1042  for (int i = 0; i < rho.num_points(); i++) {
1043  rho(i) += rho_t[i];
1044  }
1045  }
1046 
1047  for (int ir = 0; ir < atom_type_.num_mt_points(); ir++) {
1048  ae_core_charge_density_[ir] = rho(ir);
1049  }
1050 
1051  /* interpolate muffin-tin part of core density */
1052  Spline<double> rho_mt(atom_type_.radial_grid(), ae_core_charge_density_);
1053 
1054  /* compute core leakage */
1055  core_leakage_ = fourpi * (rho.interpolate().integrate(2) - rho_mt.integrate(2));
1056 
1057  /* compute eigen-value sum of core states */
1058  core_eval_sum_ = 0.0;
1059  for (int ist = 0; ist < atom_type_.num_atomic_levels(); ist++) {
1060  if (atom_type_.atomic_level(ist).core) {
1061  core_eval_sum_ += level_energy[ist] * atom_type_.atomic_level(ist).occupancy;
1062  }
1063  }
1064 }
