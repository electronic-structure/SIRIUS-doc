25 inline void K_point::generate_fv_states()
26 {
27  PROFILE("sirius::K_point::generate_fv_states");
28 
29  if (!ctx_.full_potential()) {
30  return;
31  }
32 
33  mdarray<double_complex, 2> alm(num_gkvec_loc(), unit_cell_.max_mt_aw_basis_size(), memory_t::host);
34  mdarray<double_complex, 2> tmp(unit_cell_.max_mt_aw_basis_size(), ctx_.num_fv_states());
35 
36  if (ctx_.processing_unit() == device_t::GPU) {
37  fv_eigen_vectors_slab().pw_coeffs(0).allocate(memory_t::device);
38  fv_eigen_vectors_slab().pw_coeffs(0).copy_to(memory_t::device, 0, ctx_.num_fv_states());
39  alm.allocate(memory_t::device);
40  tmp.allocate(memory_t::device);
41  }
42 
43  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
44  auto location = fv_eigen_vectors_slab().spl_num_atoms().location(ia);
45  /* number of alm coefficients for atom */
46  int mt_aw_size = unit_cell_.atom(ia).mt_aw_basis_size();
47  int mt_lo_size = unit_cell_.atom(ia).mt_lo_basis_size();
48  /* generate matching coefficients for all G-vectors */
49  alm_coeffs_loc_->generate(ia, alm);
50 
51  double_complex* tmp_ptr_gpu = (ctx_.processing_unit() == GPU) ? tmp.at(memory_t::device) : nullptr;
52  mdarray<double_complex, 2> tmp1(tmp.at(memory_t::host), tmp_ptr_gpu, mt_aw_size, ctx_.num_fv_states());
53 
54  /* compute F(lm, i) = A(lm, G)^{T} * evec(G, i) for a single atom */
55  if (ctx_.processing_unit() == CPU) {
56  linalg<CPU>::gemm(1, 0, mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(),
57  alm.at(memory_t::host), alm.ld(),
58  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::host),
59  fv_eigen_vectors_slab().pw_coeffs(0).prime().ld(),
60  tmp1.at(memory_t::host), tmp1.ld());
61  }
62  #ifdef __GPU
63  if (ctx_.processing_unit() == GPU) {
64  alm.copy_to(memory_t::device, 0, mt_aw_size * num_gkvec_loc());
65  linalg<GPU>::gemm(1, 0, mt_aw_size, ctx_.num_fv_states(), num_gkvec_loc(),
66  alm.at(memory_t::device), alm.ld(),
67  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::device),
68  fv_eigen_vectors_slab().pw_coeffs(0).prime().ld(),
69  tmp1.at(memory_t::device), tmp1.ld());
70  tmp1.copy_to(memory_t::host);
71  }
72  #endif
73 
74  comm_.reduce(tmp1.at(memory_t::host), static_cast<int>(tmp1.size()), location.rank);
75 
76  #ifdef __PRINT_OBJECT_CHECKSUM
77  auto z1 = tmp1.checksum();
78  DUMP("checksum(tmp1): %18.10f %18.10f", std::real(z1), std::imag(z1));
79  #endif
80 
81  if (location.rank == comm_.rank()) {
82  int offset1 = fv_states().offset_mt_coeffs(location.local_index);
83  int offset2 = fv_eigen_vectors_slab().offset_mt_coeffs(location.local_index);
84  for (int i = 0; i < ctx_.num_fv_states(); i++) {
85  /* aw block */
86  std::memcpy(fv_states().mt_coeffs(0).prime().at(memory_t::host, offset1, i),
87  tmp1.at(memory_t::host, 0, i),
88  mt_aw_size * sizeof(double_complex));
89  /* lo block */
90  if (mt_lo_size) {
91  std::memcpy(fv_states().mt_coeffs(0).prime().at(memory_t::host, offset1 + mt_aw_size, i),
92  fv_eigen_vectors_slab().mt_coeffs(0).prime().at(memory_t::host, offset2, i),
93  mt_lo_size * sizeof(double_complex));
94  }
95  }
96  }
97  }
98 
99  #pragma omp parallel for
100  for (int i = 0; i < ctx_.num_fv_states(); i++) {
101  /* G+k block */
102  std::memcpy(fv_states().pw_coeffs(0).prime().at(memory_t::host, 0, i),
103  fv_eigen_vectors_slab().pw_coeffs(0).prime().at(memory_t::host, 0, i),
104  num_gkvec_loc() * sizeof(double_complex));
105  }
106 
107  if (ctx_.processing_unit() == device_t::GPU) {
108  fv_eigen_vectors_slab().pw_coeffs(0).deallocate(memory_t::device);
109  }
110 }
