58  /// Compute the ion-ion electrostatic energy using Ewald method.
59  /** The following contribution (per unit cell) to the total energy has to be computed:
60  * \f[
61  * E^{ion-ion} = \frac{1}{N} \frac{1}{2} \sum_{i \neq j} \frac{Z_i Z_j}{|{\bf r}_i - {\bf r}_j|} =
62  * \frac{1}{2} \sideset{}{'} \sum_{\alpha \beta {\bf T}} \frac{Z_{\alpha} Z_{\beta}}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|}
63  * \f]
64  * where \f$ N \f$ is the number of unit cells in the crystal.
65  * Following the idea of Ewald the Coulomb interaction is split into two terms:
66  * \f[
67  * \frac{1}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} =
68  * \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} +
69  * \frac{{\rm erfc}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|}
70  * \f]
71  * Second term is computed directly. First term is computed in the reciprocal space. Remembering that
72  * \f[
73  * \frac{1}{\Omega} \sum_{\bf G} e^{i{\bf Gr}} = \sum_{\bf T} \delta({\bf r - T})
74  * \f]
75  * we rewrite the first term as
76  * \f[
77  * \frac{1}{2} \sideset{}{'} \sum_{\alpha \beta {\bf T}} Z_{\alpha} Z_{\beta}
78  * \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} =
79  * \frac{1}{2} \sum_{\alpha \beta {\bf T}} Z_{\alpha} Z_{\beta} \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} -
80  * \frac{1}{2} \sum_{\alpha} Z_{\alpha}^2 2 \sqrt{\frac{\lambda}{\pi}} = \\
81  * \frac{1}{2} \sum_{\alpha \beta} Z_{\alpha} Z_{\beta} \frac{1}{\Omega} \sum_{\bf G} \int e^{i{\bf Gr}} \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha\beta} + {\bf r}|)}{|{\bf r}_{\alpha\beta} + {\bf r}|} d{\bf r} -
82  * \sum_{\alpha} Z_{\alpha}^2 \sqrt{\frac{\lambda}{\pi}}
83  * \f]
84  * The integral is computed using the \f$ \ell=0 \f$ term of the spherical expansion of the plane-wave:
85  * \f[
86  * \int e^{i{\bf Gr}} \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}_{\alpha\beta} + {\bf r}|)}{|{\bf r}_{\alpha\beta} + {\bf r}|} d{\bf r} =
87  * \int e^{-i{\bf r}_{\alpha \beta}{\bf G}} e^{i{\bf Gr}} \frac{{\rm erf}(\sqrt{\lambda} |{\bf r}|)}{|{\bf r}|} d{\bf r} =
88  * e^{-i{\bf r}_{\alpha \beta}{\bf G}} 4 \pi \int_0^{\infty} \frac{\sin({G r})}{G} {\rm erf}(\sqrt{\lambda} r ) dr
89  * \f]
90  * We will split integral in two parts:
91  * \f[
92  * \int_0^{\infty} \sin({G r}) {\rm erf}(\sqrt{\lambda} r ) dr = \int_0^{b} \sin({G r}) {\rm erf}(\sqrt{\lambda} r ) dr +
93  * \int_b^{\infty} \sin({G r}) dr = \frac{1}{G} e^{-\frac{G^2}{4 \lambda}}
94  * \f]
95  * where \f$ b \f$ is sufficiently large. To reproduce in Mathrmatica:
96  \verbatim
97  Limit[Limit[
98  Integrate[Sin[g*x]*Erf[Sqrt[nu] * x], {x, 0, b},
99  Assumptions -> {nu > 0, g >= 0, b > 0}] +
100  Integrate[Sin[g*(x + I*a)], {x, b, \[Infinity]},
101  Assumptions -> {a > 0, nu > 0, g >= 0, b > 0}], a -> 0],
102  b -> \[Infinity], Assumptions -> {nu > 0, g >= 0}]
103  \endverbatim
104  * The first term of the Ewald sum thus becomes:
105  * \f[
106  * \frac{2 \pi}{\Omega} \sum_{{\bf G}} \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \Big| \sum_{\alpha} Z_{\alpha} e^{-i{\bf r}_{\alpha}{\bf G}} \Big|^2 -
107  * \sum_{\alpha} Z_{\alpha}^2 \sqrt{\frac{\lambda}{\pi}}
108  * \f]
109  * For \f$ G=0 \f$ the following is done:
110  * \f[
111  * \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \approx \frac{1}{G^2}-\frac{1}{4 \lambda }
112  * \f]
113  * The term \f$ \frac{1}{G^2} \f$ is compensated together with the corresponding Hartree terms in electron-electron and
114  * electron-ion interactions (cell should be neutral) and we are left with the following conribution:
115  * \f[
116  * -\frac{2\pi}{\Omega}\frac{N_{el}^2}{4 \lambda}
117  * \f]
118  * Final expression for the Ewald energy:
119  * \f[
120  * E^{ion-ion} = \frac{1}{2} \sideset{}{'} \sum_{\alpha \beta {\bf T}} Z_{\alpha} Z_{\beta}
121  * \frac{{\rm erfc}(\sqrt{\lambda} |{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|)}{|{\bf r}_{\alpha} - {\bf r}_{\beta} + {\bf T}|} +
122  * \frac{2 \pi}{\Omega} \sum_{{\bf G}\neq 0} \frac{e^{-\frac{G^2}{4 \lambda}}}{G^2} \Big| \sum_{\alpha} Z_{\alpha} e^{-i{\bf r}_{\alpha}{\bf G}} \Big|^2 -
123  * \sum_{\alpha} Z_{\alpha}^2 \sqrt{\frac{\lambda}{\pi}} - \frac{2\pi}{\Omega}\frac{N_{el}^2}{4 \lambda}
124  * \f]
125  */
126  double ewald_energy();

128  mdarray<double, 2> compute_atomic_mag_mom() const
129  {
130  mdarray<double, 2> mmom(3, unit_cell_.num_atoms());
131  mmom.zero();
132 
133  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
134  for (int j0 = 0; j0 < ctx_.fft().grid().size(0); j0++) {
135  for (int j1 = 0; j1 < ctx_.fft().grid().size(1); j1++) {
136  for (int j2 = 0; j2 < ctx_.fft().local_size_z(); j2++) {
137  /* get real space fractional coordinate */
138  auto v0 = vector3d<double>(double(j0) / ctx_.fft().grid().size(0),
139  double(j1) / ctx_.fft().grid().size(1),
140  double(ctx_.fft().offset_z() + j2) / ctx_.fft().grid().size(2));
141  /* index of real space point */
142  int ir = ctx_.fft().grid().index_by_coord(j0, j1, j2);
143 
144  for (int t0 = -1; t0 <= 1; t0++) {
145  for (int t1 = -1; t1 <= 1; t1++) {
146  for (int t2 = -1; t2 <= 1; t2++) {
147  vector3d<double> v1 = v0 - (unit_cell_.atom(ia).position() + vector3d<double>(t0, t1, t2));
148  auto r = unit_cell_.get_cartesian_coordinates(v1);
149  auto a = r.length();
150 
151  if (a <= 2.0) {
152  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
153  mmom(j, ia) += density_.magnetization(j)->f_rg(ir);
154  }
155  }
156  }
157  }
158  }
159  }
160  }
161  }
162  for (int j: {0, 1, 2}) {
163  mmom(j, ia) *= (unit_cell_.omega() / ctx_.fft().size());
164  }
165  }
166  ctx_.fft().comm().allreduce(&mmom(0, 0), static_cast<int>(mmom.size()));
167  return std::move(mmom);
168  }

277  /// Total energy of the electronic subsystem.
278  /** From the definition of the density functional we have:
279  * \f[
280  * E[\rho] = T[\rho] + E^{H}[\rho] + E^{XC}[\rho] + E^{ext}[\rho]
281  * \f]
282  * where \f$ T[\rho] \f$ is the kinetic energy, \f$ E^{H}[\rho] \f$ - electrostatic energy of
283  * electron-electron density interaction, \f$ E^{XC}[\rho] \f$ - exchange-correlation energy
284  * and \f$ E^{ext}[\rho] \f$ - energy in the external field of nuclei.
285  *
286  * Electrostatic and external field energies are grouped in the following way:
287  * \f[
288  * \frac{1}{2} \int \int \frac{\rho({\bf r})\rho({\bf r'}) d{\bf r} d{\bf r'}}{|{\bf r} - {\bf r'}|} +
289  * \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r} = \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} +
290  * \frac{1}{2} \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r}
291  * \f]
292  * Here \f$ V^{H}({\bf r}) \f$ is the total (electron + nuclei) electrostatic potential returned by the
293  * poisson solver. Next we transform the remaining term:
294  * \f[
295  * \frac{1}{2} \int \rho({\bf r}) V^{nuc}({\bf r}) d{\bf r} =
296  * \frac{1}{2} \int \int \frac{\rho({\bf r})\rho^{nuc}({\bf r'}) d{\bf r} d{\bf r'}}{|{\bf r} - {\bf r'}|} =
297  * \frac{1}{2} \int V^{H,el}({\bf r}) \rho^{nuc}({\bf r}) d{\bf r}
298  * \f]
299  *
300  * Total energy in PW-PP method has the following expression:
301  * \f[
302  * E_{tot} = \sum_{i} f_i \langle \psi_i | \Big( \hat T + \sum_{\xi \xi'} |\beta_{\xi} \rangle D_{\xi \xi'}^{ion} \langle \beta_{\xi'} |\Big) | \psi_i \rangle +
303  * \int V^{ion}({\bf r})\rho({\bf r})d{\bf r} + \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} + E^{XC}[\rho + \rho_{core}]
304  * \f]
305  * The following rearrangement is performed:
306  * \f[
307  * \int V^{ion}({\bf r})\rho({\bf r})d{\bf r} + \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} = \\
308  * \int V^{ion}({\bf r})\rho({\bf r})d{\bf r} + \int V^{H}({\bf r})\rho({\bf r})d{\bf r} + \int V^{XC}({\bf r})\rho({\bf r})d{\bf r} -
309  * \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r})d{\bf r} = \\
310  * \int V^{eff}({\bf r})\rho({\bf r})d{\bf r} - \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r})d{\bf r} = \\
311  * \int V^{eff}({\bf r})\rho^{ps}({\bf r})d{\bf r} + \int V^{eff}({\bf r})\rho^{aug}({\bf r})d{\bf r} - \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r})d{\bf r}
312  * \f]
313  * Where
314  * \f[
315  * V^{eff}({\bf r}) = V^{H}({\bf r}) + V^{XC}({\bf r}) + V^{ion}({\bf r})
316  * \f]
317  * Next, we use the definition of \f$ \rho^{aug} \f$:
318  * \f[
319  * \int V^{eff}({\bf r})\rho^{aug}({\bf r})d{\bf r} = \int V^{eff}({\bf r}) \sum_{i}\sum_{\xi \xi'} f_i \langle \psi_i | \beta_{\xi} \rangle Q_{\xi \xi'}({\bf r}) \langle \beta_{\xi'} | \psi_i \rangle d{\bf r} =
320  * \sum_{i}\sum_{\xi \xi'} f_i \langle \psi_i | \beta_{\xi} \rangle D_{\xi \xi'}^{aug} \langle \beta_{\xi'} | \psi_i \rangle
321  * \f]
322  * Now we can rewrite the total energy expression:
323  * \f[
324  * E_{tot} = \sum_{i} f_i \langle \psi_i | \Big( \hat T + \sum_{\xi \xi'} |\beta_{\xi} \rangle D_{\xi \xi'}^{ion} + D_{\xi \xi'}^{aug} \langle \beta_{\xi'} |\Big) | \psi_i \rangle +
325  * \int V^{eff}({\bf r})\rho^{ps}({\bf r})d{\bf r} - \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r}) d{\bf r} + E^{XC}[\rho + \rho_{core}]
326  * \f]
327  * From the Kohn-Sham equations
328  * \f[
329  * \Big( \hat T + \sum_{\xi \xi'} |\beta_{\xi} \rangle D_{\xi \xi'}^{ion} + D_{\xi \xi'}^{aug} \langle \beta_{\xi'}| + \hat V^{eff} \Big) | \psi_i \rangle = \varepsilon_i \Big( 1+\hat S \Big) |\psi_i \rangle
330  * \f]
331  * we can extract the sum
332  * \f[
333  * \sum_{i} f_i \langle \psi_i | \Big( \hat T + \sum_{\xi \xi'} |\beta_{\xi} \rangle D_{\xi \xi'}^{ion} + D_{\xi \xi'}^{aug} \langle \beta_{\xi'} |\Big) | \psi_i \rangle =
334  * \sum_{i} f_i \varepsilon_i - \int V^{eff}({\bf r}) \rho^{ps}({\bf r}) d{\bf r}
335  * \f]
336  * Combining all together we obtain the following expression for total energy:
337  * \f[
338  * E_{tot} = \sum_{i} f_i \varepsilon_i - \frac{1}{2} \int V^{H}({\bf r})\rho({\bf r})d{\bf r} - \int V^{XC}({\bf r})\rho({\bf r}) d{\bf r} + E^{XC}[\rho + \rho_{core}]
339  * \f]
340  */
341  double total_energy()
342  {
343  double tot_en{0};
344 
345  switch (ctx_.esm_type()) {
346  case electronic_structure_method_t::full_potential_lapwlo: {
347  tot_en = (energy_kin() + energy_exc() + 0.5 * energy_vha() + energy_enuc());
348  break;
349  }
350 
351  case electronic_structure_method_t::pseudopotential: {
352  //tot_en = (kset_.valence_eval_sum() - energy_veff() + energy_vloc() - potential_.PAW_one_elec_energy()) +
353  // 0.5 * energy_vha() + energy_exc() + potential_.PAW_total_energy() + ewald_energy_;
354  tot_en = (kset_.valence_eval_sum() - energy_vxc() - potential_.PAW_one_elec_energy()) -
355  0.5 * energy_vha() + energy_exc() + potential_.PAW_total_energy() + ewald_energy_;
356  break;
357  }
358  }
359 
360  return tot_en;
361  }

363  void symmetrize(Periodic_function<double>* f__,
364  Periodic_function<double>* gz__,
365  Periodic_function<double>* gx__,
366  Periodic_function<double>* gy__)

369 
370  auto& comm = ctx_.comm();
371 
372  /* symmetrize PW components */
373  auto v = f__->gather_f_pw();
374  unit_cell_.symmetry().symmetrize_function(&v[0], ctx_.gvec(), comm);
375  f__->scatter_f_pw(v);
376  switch (ctx_.num_mag_dims()) {
377  case 1: {
378  auto vz = gz__->gather_f_pw();
379  unit_cell_.symmetry().symmetrize_vector_function(&vz[0], ctx_.gvec(), comm);
380  gz__->scatter_f_pw(vz);
381  break;
382  }
383  case 3: {
384  auto vx = gx__->gather_f_pw();
385  auto vy = gy__->gather_f_pw();
386  auto vz = gz__->gather_f_pw();
387  unit_cell_.symmetry().symmetrize_vector_function(&vx[0], &vy[0], &vz[0], ctx_.gvec(), comm);
388  gx__->scatter_f_pw(vx);
389  gy__->scatter_f_pw(vy);
390  gz__->scatter_f_pw(vz);
391  break;
392  }
393  }
394 
395  if (ctx_.full_potential()) {
396  /* symmetrize MT components */
397  unit_cell_.symmetry().symmetrize_function(f__->f_mt(), comm);
398  switch (ctx_.num_mag_dims()) {
399  case 1: {
400  unit_cell_.symmetry().symmetrize_vector_function(gz__->f_mt(), comm);
401  break;
402  }
403  case 3: {
404  unit_cell_.symmetry().symmetrize_vector_function(gx__->f_mt(), gy__->f_mt(), gz__->f_mt(), comm);
405  break;
406  }
407  }
408  }

460 inline double DFT_ground_state::ewald_energy()
461 {
462  PROFILE("sirius::DFT_ground_state::ewald_energy");
463 
464  double alpha = 1.5;
465 
466  double ewald_g = 0;
467 
468  #pragma omp parallel
469  {
470  double ewald_g_pt = 0;
471 
472  #pragma omp for
473  for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
474  int ig = ctx_.gvec().offset() + igloc;
475  if (!ig) {
476  continue;
477  }
478 
479  double g2 = std::pow(ctx_.gvec().gvec_len(ig), 2);
480 
481  double_complex rho(0, 0);
482 
483  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
484  rho += ctx_.gvec_phase_factor(ig, ia) * static_cast<double>(unit_cell_.atom(ia).zn());
485  }
486 
487  ewald_g_pt += std::pow(std::abs(rho), 2) * std::exp(-g2 / 4 / alpha) / g2;
488  }
489 
490  #pragma omp critical
491  ewald_g += ewald_g_pt;
492  }
493  ctx_.comm().allreduce(&ewald_g, 1);
494  if (ctx_.gvec().reduced()) {
495  ewald_g *= 2;
496  }
497  /* remaining G=0 contribution */
498  ewald_g -= std::pow(unit_cell_.num_electrons(), 2) / alpha / 4;
499  ewald_g *= (twopi / unit_cell_.omega());
500 
501  /* remove self-interaction */
502  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
503  ewald_g -= std::sqrt(alpha / pi) * std::pow(unit_cell_.atom(ia).zn(), 2);
504  }
505 
506  double ewald_r = 0;
507  #pragma omp parallel
508  {
509  double ewald_r_pt = 0;
510 
511  #pragma omp for
512  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
513  for (int i = 1; i < unit_cell_.num_nearest_neighbours(ia); i++) {
514  int ja = unit_cell_.nearest_neighbour(i, ia).atom_id;
515  double d = unit_cell_.nearest_neighbour(i, ia).distance;
516  ewald_r_pt += 0.5 * unit_cell_.atom(ia).zn() * unit_cell_.atom(ja).zn() *
517  gsl_sf_erfc(std::sqrt(alpha) * d) / d;
518  }
519  }
520 
521  #pragma omp critical
522  ewald_r += ewald_r_pt;
523  }
524 
525  return (ewald_g + ewald_r);

787 /** \page DFT Spin-polarized DFT
788  * \section section1 Preliminary notes
789  *
790  * \note Here and below sybol \f$ {\boldsymbol \sigma} \f$ is reserved for the vector of Pauli matrices. Spin components
791  * are labeled with \f$ \alpha \f$ or \f$ \beta \f$.
792  *
793  * Wave-function of spin-1/2 particle is a two-component spinor:
794  * \f[
795  * {\bf \Psi}({\bf r}) = \left( \begin{array}{c} \psi^{\uparrow}({\bf r}) \\ \psi^{\downarrow}({\bf r}) \end{array} \right)
796  * \f]
797  * Operator of spin:
798  * \f[
799  * {\bf \hat S}=\frac{\hbar}{2}{\boldsymbol \sigma},
800  * \f]
801  * Pauli matrices:
802  * \f[
803  * \sigma_x=\left( \begin{array}{cc}
804  * 0 & 1 \\
805  * 1 & 0 \\ \end{array} \right) \,
806  * \sigma_y=\left( \begin{array}{cc}
807  * 0 & -i \\
808  * i & 0 \\ \end{array} \right) \,
809  * \sigma_z=\left( \begin{array}{cc}
810  * 1 & 0 \\
811  * 0 & -1 \\ \end{array} \right)
812  * \f]
813  *
814  * Spin moment of an electron in quantum state \f$ | \Psi \rangle \f$:
815  * \f[
816  * {\bf S}=\langle \Psi | {\bf \hat S} | \Psi \rangle = \frac{\hbar}{2} \langle \Psi | {\boldsymbol \sigma} | \Psi \rangle
817  * \f]
818  *
819  * Spin magnetic moment of electron:
820  * \f[
821  * {\bf \mu}_e=\gamma_e {\bf S},
822  * \f]
823  * where \f$ \gamma_e \f$ is the gyromagnetic ratio for the electron.
824  * \f[
825  * \gamma_e=-\frac{g_e \mu_B}{\hbar} \;\;\; \mu_B=\frac{e\hbar}{2m_ec}
826  * \f]
827  * Here \f$ g_e \f$ is a g-factor for electron which is ~2, and \f$ \mu_B \f$ - Bohr magneton (defined as positive constant).
828  * Finally, magnetic moment of electron:
829  * \f[
830  * {\bf \mu}_e=-{\bf \mu}_B \langle \Psi | {\boldsymbol \sigma} | \Psi \rangle
831  * \f]
832  * Potential energy of magnetic dipole in magnetic field:
833  * \f[
834  * U=-{\bf B}{\bf \mu}={\bf \mu}_B {\bf B} \langle \Psi | {\boldsymbol \sigma} | \Psi \rangle
835  * \f]
836  *
837  * \section section2 Density and magnetization
838  * In magnetic calculations we have charge density \f$ \rho({\bf r}) \f$ (scalar function) and magnetization density
839  * \f$ {\bf m}({\bf r}) \f$ (vector function). Density is defined as:
840  * \f[
841  * \rho({\bf r}) = \sum_{j}^{occ} \Psi_{j}^{*}({\bf r}){\bf I} \Psi_{j}({\bf r}) =
842  * \sum_{j}^{occ} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) +
843  * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r})
844  * \f]
845  * Magnetization is defined as:
846  * \f[
847  * {\bf m}({\bf r}) = \sum_{j}^{occ} \Psi_{j}^{*}({\bf r}) {\boldsymbol \sigma} \Psi_{j}({\bf r})
848  * \f]
849  * \f[
850  * m_x({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) +
851  * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r})
852  * \f]
853  * \f[
854  * m_y({\bf r}) = \sum_{j}^{occ} -i \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) +
855  * i \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r})
856  * \f]
857  * \f[
858  * m_z({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) -
859  * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r})
860  * \f]
861  * Density and magnetization can be grouped into a \f$ 2 \times 2 \f$ density matrix, which is defined as:
862  * \f[
863  * {\boldsymbol \rho}({\bf r}) = \frac{1}{2} \Big( {\bf I}\rho({\bf r}) + {\boldsymbol \sigma} {\bf m}({\bf r})\Big) =
864  * \frac{1}{2} \left( \begin{array}{cc} \rho({\bf r}) + m_z({\bf r}) & m_x({\bf r}) - i m_y({\bf r}) \\
865  * m_x({\bf r}) + i m_y({\bf r}) & \rho({\bf r}) - m_z({\bf r}) \end{array} \right) =
866  * \sum_{j}^{occ} \left( \begin{array}{cc} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) &
867  * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) \\
868  * \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) &
869  * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) \end{array} \right)
870  * \f]
871  * or simply
872  * \f[
873  * \rho_{\alpha \beta}({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\beta *}({\bf r})\psi_{j}^{\alpha}({\bf r})
874  * \f]
875  * Pay attention to the order of spin indices in the \f$ 2 \times 2 \f$ density matrix.
876  * External potential \f$ v^{ext}({\bf r}) \f$ and external magnetic field \f$ {\bf B}^{ext}({\bf r}) \f$ can
877  * also be grouped into a \f$ 2 \times 2 \f$ matrix:
878  * \f[
879  * V_{\alpha\beta}^{ext}({\bf r})=\Big({\bf I}v^{ext}({\bf r})+\mu_{B}{\boldsymbol \sigma}{\bf B}^{ext}({\bf r}) \Big) =
880  * \left( \begin{array}{cc} v^{ext}({\bf r})+\mu_{B}B_z^{ext}({\bf r}) & \mu_{B} \Big( B_x^{ext}({\bf r})-iB_y^{exp}({\bf r}) \Big) \\
881  * \mu_{B} \Big( B_x^{ext}({\bf r})+iB_y^{ext}({\bf r}) \Big) & v^{ext}({\bf r})-\mu_{B}B_z^{ext}({\bf r}) \end{array} \right)
882  * \f]
883  * Let's check that potential energy in external fields can be written in the following way:
884  * \f[
885  * E^{ext}=\int Tr \Big( {\boldsymbol \rho}({\bf r}) {\bf V}^{ext}({\bf r}) \Big) d{\bf r} =
886  * \sum_{\alpha\beta} \int \rho_{\alpha\beta}({\bf r})V_{\beta\alpha}^{ext}({\bf r}) d{\bf r}
887  * \f]
888  * (below \f$ {\bf r} \f$, \f$ ext \f$ and \f$ \int d{\bf r} \f$ are dropped for simplicity)
889  * \f[
890  * \begin{eqnarray}
891  * \rho_{11}V_{11} &= \frac{1}{2}(\rho+m_z)(v+\mu_{B}B_z) = \frac{1}{2}(\rho v +\mu_{B} \rho B_z + m_z v + \mu_{B}m_zB_z) \\
892  * \rho_{22}V_{22} &= \frac{1}{2}(\rho-m_z)(v-\mu_{B}B_z) = \frac{1}{2}(\rho v -\mu_{B} \rho B_z - m_z v + \mu_{B}m_zB_z) \\
893  * \rho_{12}V_{21} &= \frac{1}{2}(m_x-im_y)\Big( \mu_{B}( B_x+iB_y) \Big) = \frac{\mu_B}{2}(m_xB_x+im_xB_y-im_yB_x+m_yB_y) \\
894  * \rho_{21}V_{12} &= \frac{1}{2}(m_x+im_y)\Big( \mu_{B}( B_x-iB_y) \Big) = \frac{\mu_B}{2}(m_xB_x-im_xB_y+im_yB_x+m_yB_y)
895  * \end{eqnarray}
896  * \f]
897  * The sum of this four terms will give exactly \f$ \int \rho({\bf r}) v^{ext}({\bf r}) + \mu_{B}{\bf m}({\bf r}) {\bf B}^{ext}({\bf r}) d{\bf r}\f$
898  *
899  * \section section3 Total energy variation
900  *
901  * To derive Kohn-Sham equations we need to write total energy functional of density matrix \f$ \rho_{\alpha\beta}({\bf r}) \f$:
902  * \f[
903  * E^{tot}[\rho_{\alpha\beta}] = E^{kin}[\rho_{\alpha\beta}] + E^{H}[\rho_{\alpha\beta}] + E^{ext}[\rho_{\alpha\beta}] +
904  * E^{XC}[\rho_{\alpha\beta}]
905  * \f]
906  * Kinetic energy of non-interacting electrons is written in the following way:
907  * \f[
908  * E^{kin}[\rho_{\alpha\beta}] \equiv E^{kin}[\Psi[\rho_{\alpha\beta}]] =
909  * -\frac{1}{2} \sum_{i}^{occ}\sum_{\alpha} \int \psi_{i}^{\alpha*}({\bf r}) \nabla^{2} \psi_{i}^{\alpha}({\bf r})d^3{\bf r}
910  * \f]
911  * Hartree energy:
912  * \f[
913  * E^{H}[\rho_{\alpha\beta}]= \frac{1}{2} \iint \frac{\rho({\bf r})\rho({\bf r'})}{|{\bf r}-{\bf r'}|} d{\bf r} d{\bf r'} =
914  * \frac{1}{2} \iint \sum_{\alpha\beta}\delta_{\alpha\beta} \frac{\rho_{\alpha\beta}({\bf r}) \rho({\bf r'})}{|{\bf r}-{\bf r'}|} d{\bf r} d{\bf r'}
915  * \f]
916  * where \f$ \rho({\bf r}) = Tr \rho_{\alpha\beta}({\bf r}) \f$.
917  *
918  * Exchange-correlation energy:
919  * \f[
920  * E^{XC}[\rho_{\alpha\beta}({\bf r})] \equiv E^{XC}[\rho({\bf r}),|{\bf m}({\bf r})|] =
921  * \int \rho({\bf r}) \eta_{XC}(\rho({\bf r}), m({\bf r})) d{\bf r} =
922  * \int \rho({\bf r}) \eta_{XC}(\rho^{\uparrow}({\bf r}), \rho_{\downarrow}({\bf r})) d{\bf r}
923  * \f]
924  * Now we can write the total energy variation over auxiliary orbitals with constrain of orbital normalization:
925  * \f[
926  * \frac{\delta \Big( E^{tot}+\varepsilon_i \big( 1-\sum_{\alpha} \int \psi^{\alpha*}_{i}({\bf r})\psi^{\alpha}_{i}({\bf r})d{\bf r} \big) \Big)}
927  * {\delta \psi_{i}^{\gamma*}({\bf r})} = 0
928  * \f]
929  * We will use the following chain rule:
930  * \f[
931  * \frac{\delta F[\rho_{\alpha\beta}]}{\delta \psi_{i}^{\gamma *}({\bf r})} =
932  * \sum_{\alpha' \beta'} \frac{\delta F[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\beta'}({\bf r})}
933  * \frac{\delta \rho_{\alpha'\beta'}({\bf r})}{\delta \psi_{i}^{\gamma *}({\bf r})} =
934  * \sum_{\alpha'\beta'}\frac{\delta F[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\beta'}({\bf r})}
935  * \psi_{i}^{\alpha'}({\bf r}) \delta_{\beta'\gamma} =
936  * \sum_{\alpha'}\frac{\delta F[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\gamma}({\bf r})}\psi_{i}^{\alpha'}({\bf r})
937  * \f]
938  * Variation of the normalization integral:
939  * \f[
940  * \frac{\delta \sum_{\alpha} \int \psi_{i}^{\alpha*}({\bf r}) \psi_{i}^{\alpha}({\bf r}) d {\bf r} }{\delta \psi_{i}^{\gamma *}({\bf r})} =
941  * \psi_{i}^{\gamma}({\bf r})
942  * \f]
943  * Variation of the kinetic energy functional:
944  * \f[
945  * \frac{\delta E^{kin}}{\delta \psi_{i}^{\gamma*}({\bf r})} = -\frac{1}{2} \sum_{\alpha} \nabla^{2} \psi_{i}^{\alpha}({\bf r})
946  * \delta_{\alpha\gamma} = -\frac{1}{2}\nabla^{2}\psi_{i}^{\gamma}({\bf r})
