45 template <typename T>
46 void
47 Band::set_subspace_mtrx(int N__, int n__, int num_locked, Wave_functions& phi__, Wave_functions& op_phi__, dmatrix<T>& mtrx__,
48  dmatrix<T>* mtrx_old__) const
49 {
50  PROFILE("sirius::Band::set_subspace_mtrx");
51 
52  assert(n__ != 0);
53  if (mtrx_old__ && mtrx_old__->size()) {
54  assert(&mtrx__.blacs_grid() == &mtrx_old__->blacs_grid());
55  }
56 
57  /* copy old N - num_locked x N - num_locked distributed matrix */
58  if (N__ > 0) {
59  splindex<splindex_t::block_cyclic> spl_row(N__ - num_locked, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(),
60  mtrx__.bs_row());
61  splindex<splindex_t::block_cyclic> spl_col(N__ - num_locked, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(),
62  mtrx__.bs_col());
63 
64  if (mtrx_old__) {
65  #pragma omp parallel for schedule(static)
66  for (int i = 0; i < spl_col.local_size(); i++) {
67  std::copy(&(*mtrx_old__)(0, i), &(*mtrx_old__)(0, i) + spl_row.local_size(), &mtrx__(0, i));
68  }
69  }
70 
71  if (ctx_.control().print_checksum_) {
72  double_complex cs(0, 0);
73  for (int i = 0; i < spl_col.local_size(); i++) {
74  for (int j = 0; j < spl_row.local_size(); j++) {
75  cs += mtrx__(j, i);
76  }
77  }
78  mtrx__.blacs_grid().comm().allreduce(&cs, 1);
79  if (ctx_.comm_band().rank() == 0) {
80  utils::print_checksum("subspace_mtrx_old", cs);
81  }
82  }
83  }
84 
85  /* <{phi,phi_new}|Op|phi_new> */
86  inner(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), (ctx_.num_mag_dims() == 3) ? 2 : 0, phi__, num_locked, N__ + n__ - num_locked,
87  op_phi__, N__, n__, mtrx__, 0, N__ - num_locked);
88 
89  /* restore lower part */
90  if (N__ > 0) {
91  if (mtrx__.blacs_grid().comm().size() == 1) {
92  #pragma omp parallel for
93  for (int i = 0; i < N__ - num_locked; i++) {
94  for (int j = N__ - num_locked; j < N__ + n__ - num_locked; j++) {
95  mtrx__(j, i) = utils::conj(mtrx__(i, j));
96  }
97  }
98  } else {
99  linalg(linalg_t::scalapack).tranc(n__, N__ - num_locked, mtrx__, 0, N__ - num_locked, mtrx__, N__ - num_locked, 0);
100  }
101  }
102 
103  if (ctx_.control().print_checksum_) {
104  splindex<splindex_t::block_cyclic> spl_row(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_row(),
105  mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
106  splindex<splindex_t::block_cyclic> spl_col(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_col(),
107  mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
108  double_complex cs(0, 0);
109  for (int i = 0; i < spl_col.local_size(); i++) {
110  for (int j = 0; j < spl_row.local_size(); j++) {
111  cs += mtrx__(j, i);
112  }
113  }
114  mtrx__.blacs_grid().comm().allreduce(&cs, 1);
115  if (ctx_.comm_band().rank() == 0) {
116  utils::print_checksum("subspace_mtrx", cs);
117  }
118  }
119 
120  /* kill any numerical noise */
121  mtrx__.make_real_diag(N__ + n__ - num_locked);
122 
123  /* save new matrix */
124  if (mtrx_old__) {
125  splindex<splindex_t::block_cyclic> spl_row(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_row(),
126  mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
127  splindex<splindex_t::block_cyclic> spl_col(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_col(),
128  mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
129 
130  #pragma omp parallel for schedule(static)
131  for (int i = 0; i < spl_col.local_size(); i++) {
132  std::copy(&mtrx__(0, i), &mtrx__(0, i) + spl_row.local_size(), &(*mtrx_old__)(0, i));
133  }
134  }

137 void
138 Band::initialize_subspace(K_point_set& kset__, Hamiltonian0& H0__) const
139 {
140  PROFILE("sirius::Band::initialize_subspace");
141 
142  int N{0};
143 
144  if (ctx_.iterative_solver_input().init_subspace_ == "lcao") {
145  /* get the total number of atomic-centered orbitals */
146  N = unit_cell_.num_ps_atomic_wf();
147  }
148 
149  for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
150  int ik = kset__.spl_num_kpoints(ikloc);
151  auto kp = kset__[ik];
152  auto Hk = H0__(*kp);
153  if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
154  initialize_subspace<double>(Hk, N);
155  } else {
156  initialize_subspace<double_complex>(Hk, N);
157  }
158  }
159 
160  /* reset the energies for the iterative solver to do at least two steps */
161  for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
162  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
163  for (int i = 0; i < ctx_.num_bands(); i++) {
164  kset__[ik]->band_energy(i, ispn, 0);
165  kset__[ik]->band_occupancy(i, ispn, ctx_.max_occupancy());
166  }
167  }
168  }

171 template <typename T>
172 void Band::initialize_subspace(Hamiltonian_k& Hk__, int num_ao__) const
173 {
174  PROFILE("sirius::Band::initialize_subspace|kp");
175 
176  if (ctx_.control().verification_ >= 1) {
177  auto eval = diag_S_davidson<T>(Hk__);
178  if (eval[0] <= 0) {
179  std::stringstream s;
180  s << "[sirius::Band::initialize_subspace] S-operator matrix is not positive definite\n"
181  << " lowest eigen-value: " << eval[0];
182  TERMINATE(s);
183  }
184  }
185 
186  /* number of non-zero spin components */
187  const int num_sc = (ctx_.num_mag_dims() == 3) ? 2 : 1;
188 
189  /* short notation for number of target wave-functions */
190  int num_bands = ctx_.num_bands();
191 
192  /* number of basis functions */
193  int num_phi = std::max(num_ao__, num_bands / num_sc);
194 
195  int num_phi_tot = num_phi * num_sc;
196 
197  auto& mp = ctx_.mem_pool(ctx_.host_memory_t());
198 
199  ctx_.print_memory_usage(__FILE__, __LINE__);
200 
201  /* initial basis functions */
202  Wave_functions phi(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
203  for (int ispn = 0; ispn < num_sc; ispn++) {
204  phi.pw_coeffs(ispn).prime().zero();
205  }
206 
207  /* generate the initial atomic wavefunctions */
208  std::vector<int> atoms(ctx_.unit_cell().num_atoms());
209  std::iota(atoms.begin(), atoms.end(), 0);
210  Hk__.kp().generate_atomic_wave_functions(atoms, [&](int iat){return &ctx_.unit_cell().atom_type(iat).indexb_wfs();},
211  ctx_.atomic_wf_ri(), phi);
212 
213  /* generate some random noise */
214  std::vector<double> tmp(4096);
215  for (int i = 0; i < 4096; i++) {
216  tmp[i] = 1e-5 * utils::random<double>();
217  }
218  PROFILE_START("sirius::Band::initialize_subspace|kp|wf");
219  /* fill remaining wave-functions with pseudo-random guess */
220  assert(Hk__.kp().num_gkvec() > num_phi + 10);
221  #pragma omp parallel
222  {
223  for (int i = 0; i < num_phi - num_ao__; i++) {
224  #pragma omp for schedule(static) nowait
225  for (int igk_loc = 0; igk_loc < Hk__.kp().num_gkvec_loc(); igk_loc++) {
226  /* global index of G+k vector */
227  int igk = Hk__.kp().idxgk(igk_loc);
228  if (igk == i + 1) {
229  phi.pw_coeffs(0).prime(igk_loc, num_ao__ + i) = 1.0;
230  }
231  if (igk == i + 2) {
232  phi.pw_coeffs(0).prime(igk_loc, num_ao__ + i) = 0.5;
233  }
234  if (igk == i + 3) {
235  phi.pw_coeffs(0).prime(igk_loc, num_ao__ + i) = 0.25;
236  }
237  }
238  }
239  /* add random noise */
240  for (int i = 0; i < num_phi; i++) {
241  #pragma omp for schedule(static) nowait
242  for (int igk_loc = Hk__.kp().gkvec().skip_g0(); igk_loc < Hk__.kp().num_gkvec_loc(); igk_loc++) {
243  /* global index of G+k vector */
244  int igk = Hk__.kp().idxgk(igk_loc);
245  phi.pw_coeffs(0).prime(igk_loc, i) += tmp[igk & 0xFFF];
246  }
247  }
248  }
249 
250  if (ctx_.num_mag_dims() == 3) {
251  /* make pure spinor up- and dn- wave functions */
252  phi.copy_from(device_t::CPU, num_phi, phi, 0, 0, 1, num_phi);
253  }
254  PROFILE_STOP("sirius::Band::initialize_subspace|kp|wf");
255 
256  /* allocate wave-functions */
257  Wave_functions hphi(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
258  Wave_functions ophi(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
259  /* temporary wave-functions required as a storage during orthogonalization */
260  Wave_functions wf_tmp(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
261 
262  int bs = ctx_.cyclic_block_size();
263 
264  auto& gen_solver = ctx_.gen_evp_solver();
265 
266  sddk::dmatrix<T> hmlt(num_phi_tot, num_phi_tot, ctx_.blacs_grid(), bs, bs, mp);
267  sddk::dmatrix<T> ovlp(num_phi_tot, num_phi_tot, ctx_.blacs_grid(), bs, bs, mp);
268  sddk::dmatrix<T> evec(num_phi_tot, num_phi_tot, ctx_.blacs_grid(), bs, bs, mp);
269 
270  std::vector<double> eval(num_bands);
271 
272  ctx_.print_memory_usage(__FILE__, __LINE__);
273 
274  if (is_device_memory(ctx_.preferred_memory_t())) {
275  auto& mpd = ctx_.mem_pool(memory_t::device);
276 
277  for (int ispn = 0; ispn < num_sc; ispn++) {
278  phi.pw_coeffs(ispn).allocate(mpd);
279  phi.pw_coeffs(ispn).copy_to(memory_t::device, 0, num_phi_tot);
280  }
281 
282  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
283  Hk__.kp().spinor_wave_functions().pw_coeffs(ispn).allocate(mpd);
284  }
285 
286  for (int ispn = 0; ispn < num_sc; ispn++) {
287  hphi.pw_coeffs(ispn).allocate(mpd);
288  ophi.pw_coeffs(ispn).allocate(mpd);
289  wf_tmp.pw_coeffs(ispn).allocate(mpd);
290  }
291  evec.allocate(mpd);
292  hmlt.allocate(mpd);
293  ovlp.allocate(mpd);
294  }
295 
296  ctx_.print_memory_usage(__FILE__, __LINE__);
297 
298  if (ctx_.control().print_checksum_) {
299  for (int ispn = 0; ispn < num_sc; ispn++) {
300  auto cs = phi.checksum_pw(get_device_t(ctx_.preferred_memory_t()), ispn, 0, num_phi_tot);
301  if (Hk__.kp().comm().rank() == 0) {
302  std::stringstream s;
303  s << "initial_phi" << ispn;
304  utils::print_checksum(s.str(), cs);
305  }
306  }
307  }
308 
309  Hk__.kp().copy_hubbard_orbitals_on_device();
310 
311  for (int ispn_step = 0; ispn_step < ctx_.num_spin_dims(); ispn_step++) {
312  /* apply Hamiltonian and overlap operators to the new basis functions */
313  Hk__.apply_h_s<T>(spin_range((ctx_.num_mag_dims() == 3) ? 2 : ispn_step), 0, num_phi_tot, phi, &hphi, &ophi);
314 
315  /* do some checks */
316  if (ctx_.control().verification_ >= 1) {
317 
318  set_subspace_mtrx<T>(0, num_phi_tot, 0, phi, ophi, ovlp);
319  if (ctx_.control().verification_ >= 2 && ctx_.control().verbosity_ >= 2) {
320  ovlp.serialize("overlap", num_phi_tot);
321  }
322 
323  double max_diff = check_hermitian(ovlp, num_phi_tot);
324  if (max_diff > 1e-12) {
325  std::stringstream s;
326  s << "overlap matrix is not hermitian, max_err = " << max_diff;
327  TERMINATE(s);
328  }
329  std::vector<double> eo(num_phi_tot);
330  auto& std_solver = ctx_.std_evp_solver();
331  if (std_solver.solve(num_phi_tot, num_phi_tot, ovlp, eo.data(), evec)) {
332  std::stringstream s;
333  s << "error in diagonalization";
334  TERMINATE(s);
335  }
336  Hk__.kp().message(1, __function_name__, "minimum eigen-value of the overlap matrix: %18.12f\n", eo[0]);
337  if (eo[0] < 0) {
338  TERMINATE("overlap matrix is not positively defined");
339  }
340  }
341 
342  /* setup eigen-value problem */
343  set_subspace_mtrx<T>(0, num_phi_tot, 0, phi, hphi, hmlt);
344  set_subspace_mtrx<T>(0, num_phi_tot, 0, phi, ophi, ovlp);
345 
346  if (ctx_.control().verification_ >= 2 && ctx_.control().verbosity_ >= 2) {
347  hmlt.serialize("hmlt", num_phi_tot);
348  ovlp.serialize("ovlp", num_phi_tot);
349  }
350 
351  /* solve generalized eigen-value problem with the size N and get lowest num_bands eigen-vectors */
352  if (gen_solver.solve(num_phi_tot, num_bands, hmlt, ovlp, eval.data(), evec)) {
353  std::stringstream s;
354  s << "[sirius::Band::initialize_subspace] error in diagonalziation";
355  TERMINATE(s);
356  }
357 
358  if (ctx_.control().print_checksum_) {
359  auto cs = evec.checksum();
360  evec.blacs_grid().comm().allreduce(&cs, 1);
361  double cs1{0};
362  for (int i = 0; i < num_bands; i++) {
363  cs1 += eval[i];
364  }
365  if (Hk__.kp().comm().rank() == 0) {
366  utils::print_checksum("evec", cs);
367  utils::print_checksum("eval", cs1);
368  }
369  }
370  for (int i = 0; i < num_bands; i++) {
371  Hk__.kp().message(3, __function_name__, "eval[%i]=%20.16f\n", i, eval[i]);
372  }
373 
374  /* compute wave-functions */
375  /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
376  transform<T>(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), (ctx_.num_mag_dims() == 3) ? 2 : ispn_step,
377  {&phi}, 0, num_phi_tot, evec, 0, 0, {&Hk__.kp().spinor_wave_functions()}, 0, num_bands);
378 
379  for (int j = 0; j < num_bands; j++) {
380  Hk__.kp().band_energy(j, ispn_step, eval[j]);
381  }
382  }
383 
384  if (ctx_.control().print_checksum_) {
385  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
386  auto cs = Hk__.kp().spinor_wave_functions().checksum_pw(get_device_t(ctx_.preferred_memory_t()), ispn, 0, num_bands);
387  std::stringstream s;
388  s << "initial_spinor_wave_functions_" << ispn;
389  if (Hk__.kp().comm().rank() == 0) {
390  utils::print_checksum(s.str(), cs);
391  }
392  }
393  }
394 
395  if (is_device_memory(ctx_.preferred_memory_t())) {
396  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
397  Hk__.kp().spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::host, 0, num_bands);
398  Hk__.kp().spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
399  }
400  }
401 
402  Hk__.kp().release_hubbard_orbitals_on_device();
403 
404  if (ctx_.control().print_checksum_) {
405  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
406  auto cs = Hk__.kp().spinor_wave_functions().checksum_pw(device_t::CPU, ispn, 0, num_bands);
407  std::stringstream s;
408  s << "initial_spinor_wave_functions_" << ispn;
409  if (Hk__.kp().comm().rank() == 0) {
410  utils::print_checksum(s.str(), cs);
411  }
412  }
413  }
414 
415  /* check residuals */
416  if (ctx_.control().verification_ >= 2) {
417  check_residuals<T>(Hk__);
418  check_wave_functions<T>(Hk__);
419  }
420 
421  ctx_.print_memory_usage(__FILE__, __LINE__);
422 }

424 template <typename T>
425 void Band::check_residuals(Hamiltonian_k& Hk__) const
426 {
427  auto& kp = Hk__.kp();
428  kp.message(1, __function_name__, "%s", "checking residuals\n");
429 
430  const bool nc_mag = (ctx_.num_mag_dims() == 3);
431  const int num_sc = nc_mag ? 2 : 1;
432 
433  auto& psi = kp.spinor_wave_functions();
434  Wave_functions hpsi(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
435  Wave_functions spsi(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
436  Wave_functions res(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
437 
438  if (is_device_memory(ctx_.preferred_memory_t())) {
439  auto& mpd = ctx_.mem_pool(memory_t::device);
440  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
441  psi.pw_coeffs(ispn).allocate(mpd);
442  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
443  }
444  for (int i = 0; i < num_sc; i++) {
445  res.pw_coeffs(i).allocate(mpd);
446  hpsi.pw_coeffs(i).allocate(mpd);
447  spsi.pw_coeffs(i).allocate(mpd);
448  }
449 
450  kp.copy_hubbard_orbitals_on_device();
451  }
452  /* compute residuals */
453  for (int ispin_step = 0; ispin_step < ctx_.num_spin_dims(); ispin_step++) {
454  /* apply Hamiltonian and S operators to the wave-functions */
455  Hk__.apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), 0, ctx_.num_bands(), psi, &hpsi, &spsi);
456 
457  for (int ispn = 0; ispn < num_sc; ispn++) {
458  if (is_device_memory(ctx_.preferred_memory_t())) {
459  hpsi.copy_to(spin_range(ispn), memory_t::host, 0, ctx_.num_bands());
460  spsi.copy_to(spin_range(ispn), memory_t::host, 0, ctx_.num_bands());
461  }
462  #pragma omp parallel for schedule(static)
463  for (int j = 0; j < ctx_.num_bands(); j++) {
464  for (int ig = 0; ig < kp.num_gkvec_loc(); ig++) {
465  res.pw_coeffs(ispn).prime(ig, j) = hpsi.pw_coeffs(ispn).prime(ig, j) -
466  spsi.pw_coeffs(ispn).prime(ig, j) *
467  kp.band_energy(j, ispin_step);
468  }
469  }
470  }
471  /* get the norm */
472  auto l2norm = res.l2norm(device_t::CPU, nc_mag ? spin_range(2) : spin_range(0), ctx_.num_bands());
473 
474  for (int j = 0; j < ctx_.num_bands(); j++) {
475  Hk__.kp().message(1, __function_name__, "band: %3i, residual l2norm: %18.12f\n", j, l2norm[j]);
476  }
477  }
478  if (is_device_memory(ctx_.preferred_memory_t())) {
479  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
480  psi.pw_coeffs(ispn).deallocate(memory_t::device);
481  }
482  }
483 
484  kp.release_hubbard_orbitals_on_device();
485 }

487 /// Check wave-functions for orthonormalization.
488 template <typename T>
489 void Band::check_wave_functions(Hamiltonian_k& Hk__) const
490 {
491  auto& kp = Hk__.kp();
492  kp.message(1, __function_name__, "%s", "checking wave-functions\n");
493 
494  if (!ctx_.full_potential()) {
495 
496  dmatrix<T> ovlp(ctx_.num_bands(), ctx_.num_bands(), ctx_.blacs_grid(), ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
497 
498  const bool nc_mag = (ctx_.num_mag_dims() == 3);
499  const int num_sc = nc_mag ? 2 : 1;
500 
501  auto& psi = kp.spinor_wave_functions();
502  Wave_functions spsi(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
503 
504  if (is_device_memory(ctx_.preferred_memory_t())) {
505  auto& mpd = ctx_.mem_pool(memory_t::device);
506  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
507  psi.pw_coeffs(ispn).allocate(mpd);
508  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
509  }
510  for (int i = 0; i < num_sc; i++) {
511  spsi.pw_coeffs(i).allocate(mpd);
512  }

514  }

517 
518  /* compute residuals */
519  for (int ispin_step = 0; ispin_step < ctx_.num_spin_dims(); ispin_step++) {
520  /* apply Hamiltonian and S operators to the wave-functions */
521  Hk__.apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), 0, ctx_.num_bands(), psi, nullptr, &spsi);
522  inner(ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), nc_mag ? 2 : ispin_step, psi, 0, ctx_.num_bands(),
523  spsi, 0, ctx_.num_bands(), ovlp, 0, 0);
524 
525  double diff = check_identity(ovlp, ctx_.num_bands());
526 
527  if (diff > 1e-12) {
528  kp.message(1, __function_name__, "overlap matrix is not identity, maximum error : %20.12f\n", diff);
529  } else {
530  kp.message(1, __function_name__, "%s", "OK! Wave functions are orthonormal.\n");
531  }
532  }
533  if (is_device_memory(ctx_.preferred_memory_t())) {
534  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
535  psi.pw_coeffs(ispn).deallocate(memory_t::device);
536  }
537  }
538 
539  Hk__.kp().release_hubbard_orbitals_on_device();
540  }
