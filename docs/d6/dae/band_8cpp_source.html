45template <typename T, typename F>
46void
47Band::set_subspace_mtrx(int N__, int n__, int num_locked, Wave_functions<real_type<T>>& phi__,
48 Wave_functions<real_type<T>>& op_phi__, dmatrix<F>& mtrx__,
49 dmatrix<F>* mtrx_old__) const
50{
51 PROFILE("sirius::Band::set_subspace_mtrx");
52
53 assert(n__ != 0);
54 if (mtrx_old__ && mtrx_old__->size()) {
55 assert(&mtrx__.blacs_grid() == &mtrx_old__->blacs_grid());
56 }
57
58 /* copy old N - num_locked x N - num_locked distributed matrix */
59 if (N__ > 0) {
60 splindex<splindex_t::block_cyclic> spl_row(N__ - num_locked, mtrx__.blacs_grid().num_ranks_row(),
61 mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
62 splindex<splindex_t::block_cyclic> spl_col(N__ - num_locked, mtrx__.blacs_grid().num_ranks_col(),
63 mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
64
65 if (mtrx_old__) {
66 #pragma omp parallel for schedule(static)
67 for (int i = 0; i < spl_col.local_size(); i++) {
68 std::copy(&(*mtrx_old__)(0, i), &(*mtrx_old__)(0, i) + spl_row.local_size(), &mtrx__(0, i));
69 }
70 }
71
72 if (ctx_.print_checksum()) {
73 double_complex cs(0, 0);
74 for (int i = 0; i < spl_col.local_size(); i++) {
75 for (int j = 0; j < spl_row.local_size(); j++) {
76 cs += mtrx__(j, i);
77 }
78 }
79 mtrx__.blacs_grid().comm().allreduce(&cs, 1);
80 if (ctx_.comm_band().rank() == 0) {
81 utils::print_checksum("subspace_mtrx_old", cs);
82 }
83 }
84 }
85
86 /* <{phi,phi_new}|Op|phi_new> */
87 inner(ctx_.spla_context(), spin_range((ctx_.num_mag_dims() == 3) ? 2 : 0), phi__, num_locked,
88 N__ + n__ - num_locked, op_phi__, N__, n__, mtrx__, 0, N__ - num_locked);
89
90 /* restore lower part */
91 if (N__ > 0) {
92 if (mtrx__.blacs_grid().comm().size() == 1) {
93 #pragma omp parallel for
94 for (int i = 0; i < N__ - num_locked; i++) {
95 for (int j = N__ - num_locked; j < N__ + n__ - num_locked; j++) {
96 mtrx__(j, i) = utils::conj(mtrx__(i, j));
97 }
98 }
99 } else {
100 linalg(linalg_t::scalapack)
101 .tranc(n__, N__ - num_locked, mtrx__, 0, N__ - num_locked, mtrx__, N__ - num_locked, 0);
102 }
103 }
104
105 if (ctx_.print_checksum()) {
106 splindex<splindex_t::block_cyclic> spl_row(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_row(),
107 mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
108 splindex<splindex_t::block_cyclic> spl_col(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_col(),
109 mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
110 double_complex cs(0, 0);
111 for (int i = 0; i < spl_col.local_size(); i++) {
112 for (int j = 0; j < spl_row.local_size(); j++) {
113 cs += mtrx__(j, i);
114 }
115 }
116 mtrx__.blacs_grid().comm().allreduce(&cs, 1);
117 if (ctx_.comm_band().rank() == 0) {
118 utils::print_checksum("subspace_mtrx", cs);
119 }
120 }
121
122 /* kill any numerical noise */
123 mtrx__.make_real_diag(N__ + n__ - num_locked);
124
125 /* save new matrix */
126 if (mtrx_old__) {
127 splindex<splindex_t::block_cyclic> spl_row(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_row(),
128 mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
129 splindex<splindex_t::block_cyclic> spl_col(N__ + n__ - num_locked, mtrx__.blacs_grid().num_ranks_col(),
130 mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
131
132 #pragma omp parallel for schedule(static)
133 for (int i = 0; i < spl_col.local_size(); i++) {
134 std::copy(&mtrx__(0, i), &mtrx__(0, i) + spl_row.local_size(), &(*mtrx_old__)(0, i));
135 }
136 }

139template <typename T>
140void
141Band::initialize_subspace(K_point_set& kset__, Hamiltonian0<T>& H0__) const
142{
143 PROFILE("sirius::Band::initialize_subspace");
144
145 int N{0};
146
147 if (ctx_.cfg().iterative_solver().init_subspace() == "lcao") {
148 /* get the total number of atomic-centered orbitals */
149 N = unit_cell_.num_ps_atomic_wf();
150 }
151
152 for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
153 int ik = kset__.spl_num_kpoints(ikloc);
154 auto kp = kset__.get<T>(ik);
155 auto Hk = H0__(*kp);
156 if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
157 initialize_subspace<T>(Hk, N);
158 } else {
159 initialize_subspace<std::complex<T>>(Hk, N);
160 }
161 }
162
163 /* reset the energies for the iterative solver to do at least two steps */
164 for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
165 for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
166 for (int i = 0; i < ctx_.num_bands(); i++) {
167 kset__.get<T>(ik)->band_energy(i, ispn, 0);
168 kset__.get<T>(ik)->band_occupancy(i, ispn, ctx_.max_occupancy());
169 }
170 }
171 }

174template <typename T>
175void
176Band::initialize_subspace(Hamiltonian_k<real_type<T>>& Hk__, int num_ao__) const
177{
178 PROFILE("sirius::Band::initialize_subspace|kp");
179
180 if (ctx_.cfg().control().verification() >= 1) {
181 auto eval = diag_S_davidson<T>(Hk__);
182 if (eval[0] <= 0) {
183 std::stringstream s;
184 s << "S-operator matrix is not positive definite\n"
185 << " lowest eigen-value: " << eval[0];
186 WARNING(s);
187 }
188 }
189
190 /* number of non-zero spin components */
191 const int num_sc = (ctx_.num_mag_dims() == 3) ? 2 : 1;
192
193 /* short notation for number of target wave-functions */
194 int num_bands = ctx_.num_bands();
195
196 /* number of basis functions */
197 int num_phi = std::max(num_ao__, num_bands / num_sc);
198
199 int num_phi_tot = num_phi * num_sc;
200
201 auto& mp = ctx_.mem_pool(ctx_.host_memory_t());
202
203 ctx_.print_memory_usage(__FILE__, __LINE__);
204
205 /* initial basis functions */
206 Wave_functions<real_type<T>> phi(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
207 for (int ispn = 0; ispn < num_sc; ispn++) {
208 phi.pw_coeffs(ispn).prime().zero();
209 }
210
211 /* generate the initial atomic wavefunctions */
212 std::vector<int> atoms(ctx_.unit_cell().num_atoms());
213 std::iota(atoms.begin(), atoms.end(), 0);
214 Hk__.kp().generate_atomic_wave_functions(atoms, [&](int iat){return &ctx_.unit_cell().atom_type(iat).indexb_wfs();},
215 ctx_.atomic_wf_ri(), phi);
216
217 /* generate some random noise */
218 std::vector<double> tmp(4096);
219 for (int i = 0; i < 4096; i++) {
220 tmp[i] = 1e-5 * utils::random<double>();
221 }
222 PROFILE_START("sirius::Band::initialize_subspace|kp|wf");
223 /* fill remaining wave-functions with pseudo-random guess */
224 assert(Hk__.kp().num_gkvec() > num_phi + 10);
225 #pragma omp parallel
226 {
227 for (int i = 0; i < num_phi - num_ao__; i++) {
228 #pragma omp for schedule(static) nowait
229 for (int igk_loc = 0; igk_loc < Hk__.kp().num_gkvec_loc(); igk_loc++) {
230 /* global index of G+k vector */
231 int igk = Hk__.kp().idxgk(igk_loc);
232 if (igk == i + 1) {
233 phi.pw_coeffs(0).prime(igk_loc, num_ao__ + i) = 1.0;
234 }
235 if (igk == i + 2) {
236 phi.pw_coeffs(0).prime(igk_loc, num_ao__ + i) = 0.5;
237 }
238 if (igk == i + 3) {
239 phi.pw_coeffs(0).prime(igk_loc, num_ao__ + i) = 0.25;
240 }
241 }
242 }
243 /* add random noise */
244 for (int i = 0; i < num_phi; i++) {
245 #pragma omp for schedule(static) nowait
246 for (int igk_loc = Hk__.kp().gkvec().skip_g0(); igk_loc < Hk__.kp().num_gkvec_loc(); igk_loc++) {
247 /* global index of G+k vector */
248 int igk = Hk__.kp().idxgk(igk_loc);
249 phi.pw_coeffs(0).prime(igk_loc, i) += tmp[igk & 0xFFF];
250 }
251 }
252 }
253
254 if (ctx_.num_mag_dims() == 3) {
255 /* make pure spinor up- and dn- wave functions */
256 phi.copy_from(device_t::CPU, num_phi, phi, 0, 0, 1, num_phi);
257 }
258 PROFILE_STOP("sirius::Band::initialize_subspace|kp|wf");
259
260 /* allocate wave-functions */
261 Wave_functions<real_type<T>> hphi(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
262 Wave_functions<real_type<T>> ophi(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
263 /* temporary wave-functions required as a storage during orthogonalization */
264 Wave_functions<real_type<T>> wf_tmp(mp, Hk__.kp().gkvec_partition(), num_phi_tot, ctx_.preferred_memory_t(), num_sc);
265
266 int bs = ctx_.cyclic_block_size();
267
268 auto& gen_solver = ctx_.gen_evp_solver();
269
270 sddk::dmatrix<T> hmlt(num_phi_tot, num_phi_tot, ctx_.blacs_grid(), bs, bs, mp);
271 sddk::dmatrix<T> ovlp(num_phi_tot, num_phi_tot, ctx_.blacs_grid(), bs, bs, mp);
272 sddk::dmatrix<T> evec(num_phi_tot, num_phi_tot, ctx_.blacs_grid(), bs, bs, mp);
273
274 std::vector<real_type<T>> eval(num_bands);
275
276 ctx_.print_memory_usage(__FILE__, __LINE__);
277
278 if (is_device_memory(ctx_.preferred_memory_t())) {
279 auto& mpd = ctx_.mem_pool(memory_t::device);
280
281 for (int ispn = 0; ispn < num_sc; ispn++) {
282 phi.pw_coeffs(ispn).allocate(mpd);
283 phi.pw_coeffs(ispn).copy_to(memory_t::device, 0, num_phi_tot);
284 }
285
286 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
287 Hk__.kp().spinor_wave_functions().pw_coeffs(ispn).allocate(mpd);
288 }
289
290 for (int ispn = 0; ispn < num_sc; ispn++) {
291 hphi.pw_coeffs(ispn).allocate(mpd);
292 ophi.pw_coeffs(ispn).allocate(mpd);
293 wf_tmp.pw_coeffs(ispn).allocate(mpd);
294 }
295 evec.allocate(mpd);
296 hmlt.allocate(mpd);
297 ovlp.allocate(mpd);
298 }
299
300 ctx_.print_memory_usage(__FILE__, __LINE__);
301
302 if (ctx_.print_checksum()) {
303 for (int ispn = 0; ispn < num_sc; ispn++) {
304 auto cs = phi.checksum_pw(get_device_t(ctx_.preferred_memory_t()), ispn, 0, num_phi_tot);
305 if (Hk__.kp().comm().rank() == 0) {
306 std::stringstream s;
307 s << "initial_phi" << ispn;
308 utils::print_checksum(s.str(), cs);
309 }
310 }
311 }
312
313 Hk__.kp().copy_hubbard_orbitals_on_device();
314
315 for (int ispn_step = 0; ispn_step < ctx_.num_spinors(); ispn_step++) {
316 /* apply Hamiltonian and overlap operators to the new basis functions */
317 Hk__.template apply_h_s<T>(spin_range((ctx_.num_mag_dims() == 3) ? 2 : ispn_step), 0, num_phi_tot, phi, &hphi, &ophi);
318
319 /* do some checks */
320 if (ctx_.cfg().control().verification() >= 1) {
321
322 set_subspace_mtrx<T>(0, num_phi_tot, 0, phi, ophi, ovlp);
323 if (ctx_.cfg().control().verification() >= 2 && ctx_.verbosity() >= 2) {
324 ovlp.serialize("overlap", num_phi_tot);
325 }
326
327 double max_diff = check_hermitian(ovlp, num_phi_tot);
328 if (max_diff > 1e-12) {
329 std::stringstream s;
330 s << "overlap matrix is not hermitian, max_err = " << max_diff;
331 WARNING(s);
332 }
333 std::vector<real_type<T>> eo(num_phi_tot);
334 auto& std_solver = ctx_.std_evp_solver();
335 if (std_solver.solve(num_phi_tot, num_phi_tot, ovlp, eo.data(), evec)) {
336 std::stringstream s;
337 s << "error in diagonalization";
338 WARNING(s);
339 }
340 Hk__.kp().message(1, __function_name__, "minimum eigen-value of the overlap matrix: %18.12f\n", eo[0]);
341 if (eo[0] < 0) {
342 WARNING("overlap matrix is not positively defined");
343 }
344 }
345
346 /* setup eigen-value problem */
347 set_subspace_mtrx<T>(0, num_phi_tot, 0, phi, hphi, hmlt);
348 set_subspace_mtrx<T>(0, num_phi_tot, 0, phi, ophi, ovlp);
349
350 if (ctx_.cfg().control().verification() >= 2 && ctx_.verbosity() >= 2) {
351 hmlt.serialize("hmlt", num_phi_tot);
352 ovlp.serialize("ovlp", num_phi_tot);
353 }
354
355 /* solve generalized eigen-value problem with the size N and get lowest num_bands eigen-vectors */
356 if (gen_solver.solve(num_phi_tot, num_bands, hmlt, ovlp, eval.data(), evec)) {
357 RTE_THROW("error in diagonalization");
358 }
359
360 if (ctx_.print_checksum()) {
361 auto cs = evec.checksum();
362 evec.blacs_grid().comm().allreduce(&cs, 1);
363 real_type<T> cs1{0};
364 for (int i = 0; i < num_bands; i++) {
365 cs1 += eval[i];
366 }
367 if (Hk__.kp().comm().rank() == 0) {
368 utils::print_checksum("evec", cs);
369 utils::print_checksum("eval", cs1);
370 }
371 }
372 for (int i = 0; i < num_bands; i++) {
373 Hk__.kp().message(3, __function_name__, "eval[%i]=%20.16f\n", i, eval[i]);
374 }
375
376 /* compute wave-functions */
377 /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
378 transform<T, T>(ctx_.spla_context(), (ctx_.num_mag_dims() == 3) ? 2 : ispn_step, {&phi}, 0, num_phi_tot, evec, 0,
379 0, {&Hk__.kp().spinor_wave_functions()}, 0, num_bands);
380
381 for (int j = 0; j < num_bands; j++) {
382 Hk__.kp().band_energy(j, ispn_step, eval[j]);
383 }
384 }
385
386 if (ctx_.print_checksum()) {
387 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
388 auto cs = Hk__.kp().spinor_wave_functions().checksum_pw(get_device_t(ctx_.preferred_memory_t()), ispn, 0, num_bands);
389 std::stringstream s;
390 s << "initial_spinor_wave_functions_" << ispn;
391 if (Hk__.kp().comm().rank() == 0) {
392 utils::print_checksum(s.str(), cs);
393 }
394 }
395 }
396
397 if (is_device_memory(ctx_.preferred_memory_t())) {
398 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
399 Hk__.kp().spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::host, 0, num_bands);
400 Hk__.kp().spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
401 }
402 }
403
404 Hk__.kp().release_hubbard_orbitals_on_device();
405
406 if (ctx_.print_checksum()) {
407 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
408 auto cs = Hk__.kp().spinor_wave_functions().checksum_pw(device_t::CPU, ispn, 0, num_bands);
409 std::stringstream s;
410 s << "initial_spinor_wave_functions_" << ispn;
411 if (Hk__.kp().comm().rank() == 0) {
412 utils::print_checksum(s.str(), cs);
413 }
414 }
415 }
416
417 /* check residuals */
418 if (ctx_.cfg().control().verification() >= 2) {
419 check_residuals<T>(Hk__);
420 check_wave_functions<T>(Hk__);
421 }
422
423 ctx_.print_memory_usage(__FILE__, __LINE__);
424}

426template <typename T>
427void Band::check_residuals(Hamiltonian_k<real_type<T>>& Hk__) const
428{
429 auto& kp = Hk__.kp();
430 kp.message(1, __function_name__, "%s", "checking residuals\n");
431
432 const bool nc_mag = (ctx_.num_mag_dims() == 3);
433 const int num_sc = nc_mag ? 2 : 1;
434
435 auto& psi = kp.spinor_wave_functions();
436 Wave_functions<real_type<T>> hpsi(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
437 Wave_functions<real_type<T>> spsi(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
438 Wave_functions<real_type<T>> res(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
439
440 if (is_device_memory(ctx_.preferred_memory_t())) {
441 auto& mpd = ctx_.mem_pool(memory_t::device);
442 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
443 psi.pw_coeffs(ispn).allocate(mpd);
444 psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
445 }
446 for (int i = 0; i < num_sc; i++) {
447 res.pw_coeffs(i).allocate(mpd);
448 hpsi.pw_coeffs(i).allocate(mpd);
449 spsi.pw_coeffs(i).allocate(mpd);
450 }
451
452 kp.copy_hubbard_orbitals_on_device();
453 }
454 /* compute residuals */
455 for (int ispin_step = 0; ispin_step < ctx_.num_spinors(); ispin_step++) {
456 /* apply Hamiltonian and S operators to the wave-functions */
457 Hk__.template apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), 0, ctx_.num_bands(), psi, &hpsi, &spsi);
458
459 for (int ispn = 0; ispn < num_sc; ispn++) {
460 if (is_device_memory(ctx_.preferred_memory_t())) {
461 hpsi.copy_to(spin_range(ispn), memory_t::host, 0, ctx_.num_bands());
462 spsi.copy_to(spin_range(ispn), memory_t::host, 0, ctx_.num_bands());
463 }
464 #pragma omp parallel for schedule(static)
465 for (int j = 0; j < ctx_.num_bands(); j++) {
466 for (int ig = 0; ig < kp.num_gkvec_loc(); ig++) {
467 res.pw_coeffs(ispn).prime(ig, j) = hpsi.pw_coeffs(ispn).prime(ig, j) -
468 spsi.pw_coeffs(ispn).prime(ig, j) *
469 static_cast<real_type<T>>(kp.band_energy(j, ispin_step));
470 }
471 }
472 }
473 /* get the norm */
474 auto l2norm = res.l2norm(device_t::CPU, nc_mag ? spin_range(2) : spin_range(0), ctx_.num_bands());
475
476 for (int j = 0; j < ctx_.num_bands(); j++) {
477 Hk__.kp().message(1, __function_name__, "band: %3i, residual l2norm: %18.12f\n", j, l2norm[j]);
478 }
479 }
480 if (is_device_memory(ctx_.preferred_memory_t())) {
481 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
482 psi.pw_coeffs(ispn).deallocate(memory_t::device);
483 }
484 }
485
486 kp.release_hubbard_orbitals_on_device();
487}

489/// Check wave-functions for orthonormalization.
490template <typename T>
491void Band::check_wave_functions(Hamiltonian_k<real_type<T>>& Hk__) const
492{
493 auto& kp = Hk__.kp();
494 kp.message(1, __function_name__, "%s", "checking wave-functions\n");
495
496 if (!ctx_.full_potential()) {
497
498 dmatrix<T> ovlp(ctx_.num_bands(), ctx_.num_bands(), ctx_.blacs_grid(), ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
499
500 const bool nc_mag = (ctx_.num_mag_dims() == 3);
501 const int num_sc = nc_mag ? 2 : 1;
502
503 auto& psi = kp.spinor_wave_functions();
504 Wave_functions<real_type<T>> spsi(kp.gkvec_partition(), ctx_.num_bands(), ctx_.preferred_memory_t(), num_sc);
505
506 if (is_device_memory(ctx_.preferred_memory_t())) {
507 auto& mpd = ctx_.mem_pool(memory_t::device);
508 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
509 psi.pw_coeffs(ispn).allocate(mpd);
510 psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, ctx_.num_bands());
511 }
512 for (int i = 0; i < num_sc; i++) {
513 spsi.pw_coeffs(i).allocate(mpd);
514 }
515 ovlp.allocate(memory_t::device);
516 }
517
518 Hk__.kp().copy_hubbard_orbitals_on_device();
519
520 /* compute residuals */
521 for (int ispin_step = 0; ispin_step < ctx_.num_spinors(); ispin_step++) {
522 auto sr = spin_range(nc_mag ? 2 : ispin_step);
523 /* apply Hamiltonian and S operators to the wave-functions */
524 Hk__.template apply_h_s<T>(sr, 0, ctx_.num_bands(), psi, nullptr, &spsi);
525 inner(ctx_.spla_context(), sr, psi, 0, ctx_.num_bands(), spsi, 0, ctx_.num_bands(), ovlp, 0, 0);
526
527 double diff = check_identity(ovlp, ctx_.num_bands());
528
529 if (diff > 1e-12) {
