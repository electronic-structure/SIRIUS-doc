446 template <typename T>
447 inline matrix3d<T> inverse_aux(matrix3d<T> src)
448 {
449  matrix3d<T> mtrx;
450 
451  mtrx(0, 0) = (src(1, 1) * src(2, 2) - src(1, 2) * src(2, 1));
452  mtrx(0, 1) = (src(0, 2) * src(2, 1) - src(0, 1) * src(2, 2));
453  mtrx(0, 2) = (src(0, 1) * src(1, 2) - src(0, 2) * src(1, 1));
454  mtrx(1, 0) = (src(1, 2) * src(2, 0) - src(1, 0) * src(2, 2));
455  mtrx(1, 1) = (src(0, 0) * src(2, 2) - src(0, 2) * src(2, 0));
456  mtrx(1, 2) = (src(0, 2) * src(1, 0) - src(0, 0) * src(1, 2));
457  mtrx(2, 0) = (src(1, 0) * src(2, 1) - src(1, 1) * src(2, 0));
458  mtrx(2, 1) = (src(0, 1) * src(2, 0) - src(0, 0) * src(2, 1));
459  mtrx(2, 2) = (src(0, 0) * src(1, 1) - src(0, 1) * src(1, 0));
460 
461  return mtrx;
462 }

509 inline std::pair<vector3d<double>, vector3d<int>> reduce_coordinates(vector3d<double> coord)
510 {
511  const double eps{1e-9};
512 
513  std::pair<vector3d<double>, vector3d<int>> v;
514 
515  v.first = coord;
516  for (int i = 0; i < 3; i++) {
517  v.second[i] = (int)floor(v.first[i]);
518  v.first[i] -= v.second[i];
519  if (v.first[i] < -eps || v.first[i] > 1.0 + eps) {
520  std::stringstream s;
521  s << "wrong fractional coordinates" << std::endl
522  << v.first[0] << " " << v.first[1] << " " << v.first[2];
523  throw std::runtime_error(s.str());
524  }
525  if (v.first[i] < 0) {
526  v.first[i] = 0;
527  }
528  if (v.first[i] >= (1 - eps)) {
529  v.first[i] = 0;
530  v.second[i] += 1;
531  }
532  if (v.first[i] < 0 || v.first[i] >= 1) {
533  std::stringstream s;
534  s << "wrong fractional coordinates" << std::endl
535  << v.first[0] << " " << v.first[1] << " " << v.first[2];
536  throw std::runtime_error(s.str());
537  }
538  }
539  for (int x : {0, 1, 2}) {
540  if (std::abs(coord[x] - (v.first[x] + v.second[x])) > eps) {
541  std::stringstream s;
542  s << "wrong coordinate reduction" << std::endl
543  << " original coord: " << coord << std::endl
544  << " reduced coord: " << v.first << std::endl
545  << " T: " << v.second;
546  throw std::runtime_error(s.str());
547  }
548  }
549  return v;
550 }

552 /// Find supercell that circumscribes the sphere with a given radius.
553 /** Search for the translation limits (N1, N2, N3) such that the resulting supercell with the lattice
554  * vectors a1 * N1, a2 * N2, a3 * N3 fully contains the sphere with a given radius. This is done
555  * by equating the expressions for the volume of the supercell:
556  * Volume = |(A1 x A2) * A3| = N1 * N2 * N3 * |(a1 x a2) * a3|
557  * Volume = h * S = 2 * R * |a_i x a_j| * N_i * N_j */
558 inline vector3d<int> find_translations(double radius__, matrix3d<double> const& lattice_vectors__)
559 {
560  vector3d<double> a0(lattice_vectors__(0, 0), lattice_vectors__(1, 0), lattice_vectors__(2, 0));
561  vector3d<double> a1(lattice_vectors__(0, 1), lattice_vectors__(1, 1), lattice_vectors__(2, 1));
562  vector3d<double> a2(lattice_vectors__(0, 2), lattice_vectors__(1, 2), lattice_vectors__(2, 2));
563 
564  double det = std::abs(lattice_vectors__.det());
565 
566  vector3d<int> limits;
567 
568  limits[0] = static_cast<int>(2 * radius__ * cross(a1, a2).length() / det) + 1;
569  limits[1] = static_cast<int>(2 * radius__ * cross(a0, a2).length() / det) + 1;
570  limits[2] = static_cast<int>(2 * radius__ * cross(a0, a1).length() / det) + 1;
571 
572  return {limits[0], limits[1], limits[2]};
573 }

575 /// Transform Cartesian coordinates [x,y,z] to spherical coordinates [r,theta,phi]
576 inline vector3d<double> spherical_coordinates(vector3d<double> vc)
577 {
578  geometry3d::vector3d<double> vs;
579 
580  const double eps{1e-12};
581 
582  const double twopi = 6.2831853071795864769;
583 
584  vs[0] = vc.length();
585 
586  if (vs[0] <= eps) {
587  vs[1] = 0.0;
588  vs[2] = 0.0;
589  } else {
590  vs[1] = std::acos(vc[2] / vs[0]); // theta = cos^{-1}(z/r)
591 
592  if (std::abs(vc[0]) > eps || std::abs(vc[1]) > eps) {
593  vs[2] = std::atan2(vc[1], vc[0]); // phi = tan^{-1}(y/x)
594  if (vs[2] < 0.0) {
595  vs[2] += twopi;
596  }
597  } else {
598  vs[2] = 0.0;
599  }
600  }
601 
602  return vs;
603 }
