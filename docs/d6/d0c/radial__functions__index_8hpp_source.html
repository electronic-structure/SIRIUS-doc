30 /// Descriptor for the atomic radial functions.
31 /** The radial functions \f$ f_{\ell \nu}(r) \f$ are labeled by two indices: orbital quantum number \f$ \ell \f$ and
32  * an order \f$ \nu \f$ for a given \f$ \ell \f$. Radial functions can be any of augmented waves of local orbitals
33  * (in case of FP-LAPW) or bete projectors, atomic or Hubbard wave functions in case of PP-PW.
34  */
35 struct radial_function_index_descriptor
36 {
37  /// Orbital quantum number \f$ \ell \f$.
38  int l;
39 
40  /// Total angular momentum
41  double j;
42 
43  /// Order of a function for a given \f$ \ell \f$.
44  int order;
45 
46  /// If this is a local orbital radial function, idxlo is it's index in the list of local orbital descriptors.
47  int idxlo;
48 
49  /// Constructor.
50  radial_function_index_descriptor(int l, int order, int idxlo = -1)
51  : l(l)
52  , order(order)
53  , idxlo(idxlo)
54  {
55  assert(l >= 0);
56  assert(order >= 0);
57  }
58 
59  radial_function_index_descriptor(int l, double j, int order, int idxlo = -1)
60  : l(l)
61  , j(j)
62  , order(order)
63  , idxlo(idxlo)
64  {
65  assert(l >= 0);
66  assert(order >= 0);
67  }
68 };

70 /// A helper class to establish various index mappings for the atomic radial functions.
71 class radial_functions_index
72 {
73  private:
74  /// A list of radial function index descriptors.
75  /** This list establishes a mapping \f$ f_{\mu}(r) \leftrightarrow f_{\ell \nu}(r) \f$ between a
76  * composite index \f$ \mu \f$ of radial functions and
77  * corresponding \f$ \ell \nu \f$ indices, where \f$ \ell \f$ is the orbital quantum number and
78  * \f$ \nu \f$ is the order of radial function for a given \f$ \ell \f$. */
79  std::vector<radial_function_index_descriptor> radial_function_index_descriptors_;
80 
81  mdarray<int, 2> index_by_l_order_;
82 
83  mdarray<int, 1> index_by_idxlo_;
84 
85  /// Number of radial functions for each angular momentum quantum number.
86  std::vector<int> num_rf_;
87 
88  /// Number of local orbitals for each angular momentum quantum number.
89  std::vector<int> num_lo_;
90 
91  // Maximum number of radial functions across all angular momentums.
92  int max_num_rf_;
93 
94  /// Maximum orbital quantum number of augmented-wave radial functions.
95  int lmax_aw_;
96 
97  /// Maximum orbital quantum number of local orbital radial functions.
98  int lmax_lo_;
99 
100  /// Maximum orbital quantum number of radial functions.
101  int lmax_;
102 
103  public:
104  /// Initialize a list of radial functions only from the list of local orbitals.
105  void init(std::vector<local_orbital_descriptor> const& lo_descriptors__)
106  {
107  /* create an empty descriptor */
108  std::vector<radial_solution_descriptor_set> aw_descriptors;
109  this->init(aw_descriptors, lo_descriptors__);
110  }
111 
112  /// Initialize a list of radial functions from the list of APW radial functions and the list of local orbitals.
113  void init(std::vector<radial_solution_descriptor_set> const& aw_descriptors,
114  std::vector<local_orbital_descriptor> const& lo_descriptors)
115  {
116  lmax_aw_ = static_cast<int>(aw_descriptors.size()) - 1;
117  lmax_lo_ = -1;
118  for (size_t idxlo = 0; idxlo < lo_descriptors.size(); idxlo++) {
119  int l = lo_descriptors[idxlo].l;
120  lmax_lo_ = std::max(lmax_lo_, l);
121  }
122 
123  lmax_ = std::max(lmax_aw_, lmax_lo_);
124 
125  num_rf_ = std::vector<int>(lmax_ + 1, 0);
126  num_lo_ = std::vector<int>(lmax_ + 1, 0);
127 
128  max_num_rf_ = 0;
129 
130  radial_function_index_descriptors_.clear();
131 
132  for (int l = 0; l <= lmax_aw_; l++) {
133  assert(aw_descriptors[l].size() <= 3);
134 
135  for (size_t order = 0; order < aw_descriptors[l].size(); order++) {
136  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l]));
137  num_rf_[l]++;
138  }
139  }
140 
141  for (int idxlo = 0; idxlo < static_cast<int>(lo_descriptors.size()); idxlo++) {
142  int l = lo_descriptors[idxlo].l;
143  radial_function_index_descriptors_.push_back(
144  radial_function_index_descriptor(l, lo_descriptors[idxlo].total_angular_momentum, num_rf_[l], idxlo));
145  num_rf_[l]++;
146  num_lo_[l]++;
147  }
148 
149  for (int l = 0; l <= lmax_; l++) {
150  max_num_rf_ = std::max(max_num_rf_, num_rf_[l]);
151  }
152 
153  index_by_l_order_ = mdarray<int, 2>(lmax_ + 1, max_num_rf_);
154 
155  if (lo_descriptors.size()) {
156  index_by_idxlo_ = mdarray<int, 1>(lo_descriptors.size());
157  }
158 
159  for (int i = 0; i < (int)radial_function_index_descriptors_.size(); i++) {
160  int l = radial_function_index_descriptors_[i].l;
161  int order = radial_function_index_descriptors_[i].order;
162  int idxlo = radial_function_index_descriptors_[i].idxlo;
163  index_by_l_order_(l, order) = i;
164  if (idxlo >= 0) {
165  index_by_idxlo_(idxlo) = i;
166  }
167  }
168  }
169
