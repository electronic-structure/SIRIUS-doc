39 /// Angular momentum quantum number.
40 /** This class handles orbital or total angluar momentum quantum number. */
41 class angular_momentum
42 {
43  private:
44 
45  /// Orbital quantum number l.
46  int l_;
47 
48  /// Spin quantum number in the units of 1/2.
49  /** This variable can have only three values: -1,0,1. It is used to consruct the total angular
50  * momentum j = l + s/2. In case s = 0 total agular momentum j = l (no level splitting). */
51  int s_{0};
52 
53  public:
54 
55  /// Constructor.
56  explicit angular_momentum(int l__)
57  : l_(l__)
58  {
59  if (l__ < 0) {
60  THROW("l can't be negative");
61  }
62  }
63 
64  /// Constructor.
65  explicit angular_momentum(int l__, int s__)
66  : l_(l__)
67  , s_(s__)
68  {
69  if (l__ < 0) {
70  THROW("l can't be negative");
71  }
72  if (s__ != -1 && s__ != 0 && s__ != 1) {
73  THROW("wrong value of s");
74  }
75  if (l__ == 0 && s__ == -1) {
76  THROW("incompatible combination of l and s quantum numbers");
77  }
78  }
79 
80  /// Get orbital quantum number l.
81  inline auto l() const
82  {
83  return l_;
84  }
85 
86  /// Get total angular momentum j = l +/- 1/2
87  inline auto j() const
88  {
89  return l_ + s_ / 2.0;
90  }
91 
92  /// Get twice the total angular momentum 2j = 2l +/- 1
93  inline auto j2() const
94  {
95  return 2 * l_ + s_;
96  }
97 
98  /// Get spin quantum number s.
99  inline auto s() const
100  {
101  return s_;
102  }
103 };

105 class radial_functions_index
106 {
107  private:
108  int size_{0};
109  std::vector<std::vector<std::array<int, 2>>> index_by_j_order_;
110  std::vector<int> l_;
111  std::vector<int> s_;
112  std::vector<int> o_;
113  public:
114  void add(angular_momentum am__)
115  {
116  /* current l */
117  auto l = am__.l();
118  /* current s */
119  auto s = am__.s();
120 
121  if (s != 0 && l > 0) {
122  THROW("for l > 0 full-j radial functions are added in pairs");
123  }
124 
125  /* make sure that the space is available */
126  if (static_cast<int>(index_by_j_order_.size()) < l + 1) {
127  index_by_j_order_.resize(l + 1);
128  }
129 
130  std::array<int, 2> idx({-1, -1});
131  /* std::max(s, 0) maps s = -1 -> 0, s = 0 -> 0, s = 1 -> 1 */
132  idx[std::max(s, 0)] = size_;
133  /* current order */
134  auto o = static_cast<int>(index_by_j_order_[l].size());
135  index_by_j_order_[l].push_back(idx);
136  l_.push_back(l);
137  s_.push_back(s);
138  o_.push_back(o);
139  size_++;
140  }
141 
142  void add(angular_momentum am1__, angular_momentum am2__)
143  {
144  /* current l */
145  auto l1 = am1__.l();
146  /* current s */
147  auto s1 = am1__.s();
148  /* current l */
149  auto l2 = am2__.l();
150  /* current s */
151  auto s2 = am2__.s();
152 
153  if (l1 != l2) {
154  THROW("orbital quantum numbers are different");
155  }
156 
157  if (s1 == s2) {
158  THROW("spin quantum numbers are the same");
159  }
160 
161  if (s1 * s2 == 0) {
162  THROW("spin quantum numbers can't be zero in case of full orbital momentum");
163  }
164 
165  /* make sure that the space is available */
166  if (static_cast<int>(index_by_j_order_.size()) < l1 + 1) {
167  index_by_j_order_.resize(l1 + 1);
168  }
169 
170  /* current order */
171  auto o = static_cast<int>(index_by_j_order_[l1].size());
172 
173  std::array<int, 2> idx({-1, -1});
174  idx[std::max(s1, 0)] = size_;
175  idx[std::max(s2, 0)] = size_ + 1;
176 
177  l_.push_back(l1);
178  l_.push_back(l2);
179  s_.push_back(s1);
180  s_.push_back(s2);
181  o_.push_back(o);
182  o_.push_back(o);
183 
184  index_by_j_order_[l1].push_back(idx);
185  size_ += 2;
186  }
187 
188  inline auto am(int idx__) const
189  {
190  return angular_momentum(l_[idx__], s_[idx__]);
191  }
192 
193  inline auto order(int idx__) const
194  {
195  return o_[idx__];
196  }
197 
198  inline auto lmax() const
199  {
200  return static_cast<int>(index_by_j_order_.size()) - 1;
201  }
202 
203  inline auto max_order(int l__) const
204  {
205  return static_cast<int>(index_by_j_order_[l__].size());
206  }
207 
208  inline auto index_of(angular_momentum am__, int order__) const
209  {
210  return index_by_j_order_[am__.l()][order__][std::max(am__.s(), 0)];
211  }
212 
213  auto full_j(int l__, int o__) const
214  {
215  /* look at index of l + s */
216  if (index_by_j_order_[l__][o__][1] >= 0) {
217  return true;
218  } else {
219  return false;
220  }
221  }
222 
223  auto spins(int l__, int o__) const
224  {
225  if (full_j(l__, o__)) {
226  if (l__ == 0) {
227  return std::vector<int>({1});
228  } else {
229  return std::vector<int>({-1, 1});
230  }
231  } else {
232  return std::vector<int>({0});
233  }
234  }
235 
236  inline auto size() const
237  {
238  return size_;
239  }

430 /// Descriptor for the atomic radial functions.
431 /** The radial functions \f$ f_{\ell \nu}(r) \f$ are labeled by two indices: orbital quantum number \f$ \ell \f$ and
432  * an order \f$ \nu \f$ for a given \f$ \ell \f$. Radial functions can be any of augmented waves of local orbitals
433  * (in case of FP-LAPW) or bete projectors, atomic or Hubbard wave functions in case of PP-PW.
434  */
435 struct radial_function_index_descriptor
436 {
437  /// Orbital quantum number \f$ \ell \f$.
438  int l;
439 
440  /// Total angular momentum
441  double j;
442 
443  /// Order of a function for a given \f$ \ell \f$.
444  int order;
445 
446  /// If this is a local orbital radial function, idxlo is it's index in the list of local orbital descriptors.
447  int idxlo;
448 
449  /// Constructor.
450  radial_function_index_descriptor(int l, int order, int idxlo = -1)
451  : l(l)
452  , order(order)
453  , idxlo(idxlo)
454  {
455  assert(l >= 0);
456  assert(order >= 0);
457  }
458 
459  radial_function_index_descriptor(int l, double j, int order, int idxlo = -1)
460  : l(l)
461  , j(j)
462  , order(order)
463  , idxlo(idxlo)
464  {
465  assert(l >= 0);
466  assert(order >= 0);
467  }
468 };

470 /// A helper class to establish various index mappings for the atomic radial functions.
471 class radial_functions_index
472 {
473  private:
474  /// A list of radial function index descriptors.
475  /** This list establishes a mapping \f$ f_{\mu}(r) \leftrightarrow f_{\ell \nu}(r) \f$ between a
476  * composite index \f$ \mu \f$ of radial functions and
477  * corresponding \f$ \ell \nu \f$ indices, where \f$ \ell \f$ is the orbital quantum number and
478  * \f$ \nu \f$ is the order of radial function for a given \f$ \ell \f$. */
479  std::vector<radial_function_index_descriptor> radial_function_index_descriptors_;
480 
481  mdarray<int, 2> index_by_l_order_;

483  mdarray<int, 1> index_by_idxlo_;

486  std::vector<int> num_rf_;
487 
488  /// Number of local orbitals for each angular momentum quantum number.
489  std::vector<int> num_lo_;
490 
491  // Maximum number of radial functions across all angular momentums.
492  int max_num_rf_;
493 
494  /// Maximum orbital quantum number of augmented-wave radial functions.
495  int lmax_aw_;
496 
497  /// Maximum orbital quantum number of local orbital radial functions.
498  int lmax_lo_;
499 
500  /// Maximum orbital quantum number of radial functions.
501  int lmax_;
502 
503  public:
504  /// Initialize a list of radial functions from the list of local orbitals.
505  template <typename T>
506  void init(std::vector<T> const& lo_descriptors__)
507  {

513  /// Initialize a list of radial functions from the list of APW radial functions and the list of local orbitals.
514  template <typename T>
515  void init(std::vector<radial_solution_descriptor_set> const& aw_descriptors,
516  std::vector<T> const& lo_descriptors)
517  {
518  lmax_aw_ = static_cast<int>(aw_descriptors.size()) - 1;
519  lmax_lo_ = -1;
520  for (size_t idxlo = 0; idxlo < lo_descriptors.size(); idxlo++) {
521  int l = lo_descriptors[idxlo].l;
522  lmax_lo_ = std::max(lmax_lo_, l);
523  }
524 
525  lmax_ = std::max(lmax_aw_, lmax_lo_);
526 
527  num_rf_ = std::vector<int>(lmax_ + 1, 0);
528  num_lo_ = std::vector<int>(lmax_ + 1, 0);
529 
530  max_num_rf_ = 0;
531 
532  radial_function_index_descriptors_.clear();
533 
534  for (int l = 0; l <= lmax_aw_; l++) {
535  assert(aw_descriptors[l].size() <= 3);
536 
537  for (size_t order = 0; order < aw_descriptors[l].size(); order++) {
538  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l]));
539  num_rf_[l]++;
540  }
541  }
542 
543  for (int idxlo = 0; idxlo < static_cast<int>(lo_descriptors.size()); idxlo++) {
544  int l = lo_descriptors[idxlo].l;
545  radial_function_index_descriptors_.push_back(
546  radial_function_index_descriptor(l, lo_descriptors[idxlo].total_angular_momentum, num_rf_[l], idxlo));
547  num_rf_[l]++;
548  num_lo_[l]++;
549  }
550 
551  for (int l = 0; l <= lmax_; l++) {
552  max_num_rf_ = std::max(max_num_rf_, num_rf_[l]);
553  }
554 
555  index_by_l_order_ = mdarray<int, 2>(lmax_ + 1, max_num_rf_);
556 
557  if (lo_descriptors.size()) {
558  index_by_idxlo_ = mdarray<int, 1>(lo_descriptors.size());
559  }
560 
561  for (int i = 0; i < (int)radial_function_index_descriptors_.size(); i++) {
562  int l = radial_function_index_descriptors_[i].l;
563  int order = radial_function_index_descriptors_[i].order;
564  int idxlo = radial_function_index_descriptors_[i].idxlo;
565  index_by_l_order_(l, order) = i;
566  if (idxlo >= 0) {
567  index_by_idxlo_(idxlo) = i;
568  }
569  }
570  }
