82  void init(std::vector<radial_solution_descriptor_set> const& aw_descriptors,
83  std::vector<local_orbital_descriptor> const& lo_descriptors)
84  {
85  lmax_aw_ = static_cast<int>(aw_descriptors.size()) - 1;
86  lmax_lo_ = -1;
87  for (size_t idxlo = 0; idxlo < lo_descriptors.size(); idxlo++) {
88  int l = lo_descriptors[idxlo].l;
89  lmax_lo_ = std::max(lmax_lo_, l);
90  }
91 
92  lmax_ = std::max(lmax_aw_, lmax_lo_);
93 
94  num_rf_ = std::vector<int>(lmax_ + 1, 0);
95  num_lo_ = std::vector<int>(lmax_ + 1, 0);
96 
97  max_num_rf_ = 0;
98 
99  radial_function_index_descriptors_.clear();
100 
101  for (int l = 0; l <= lmax_aw_; l++) {
102  assert(aw_descriptors[l].size() <= 3);
103 
104  for (size_t order = 0; order < aw_descriptors[l].size(); order++) {
105  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l]));
106  num_rf_[l]++;
107  }
108  }
109 
110  for (int idxlo = 0; idxlo < static_cast<int>(lo_descriptors.size()); idxlo++) {
111  int l = lo_descriptors[idxlo].l;
112  radial_function_index_descriptors_.push_back(
113  radial_function_index_descriptor(l, lo_descriptors[idxlo].total_angular_momentum, num_rf_[l], idxlo));
114  num_rf_[l]++;
115  num_lo_[l]++;
116  }
117 
118  for (int l = 0; l <= lmax_; l++) {
119  max_num_rf_ = std::max(max_num_rf_, num_rf_[l]);
120  }
121 
122  index_by_l_order_ = mdarray<int, 2>(lmax_ + 1, max_num_rf_);
123 
124  if (lo_descriptors.size()) {
125  index_by_idxlo_ = mdarray<int, 1>(lo_descriptors.size());
126  }
127 
128  for (int i = 0; i < (int)radial_function_index_descriptors_.size(); i++) {
129  int l = radial_function_index_descriptors_[i].l;
130  int order = radial_function_index_descriptors_[i].order;
131  int idxlo = radial_function_index_descriptors_[i].idxlo;
132  index_by_l_order_(l, order) = i;
133  if (idxlo >= 0)
134  index_by_idxlo_(idxlo) = i;
135  }
136  }
