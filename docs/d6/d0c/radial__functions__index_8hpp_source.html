34/// Angular momentum quantum number.
35/** This class handles orbital or total angluar momentum quantum number. */
36class angular_momentum
37{
38 private:
39
40 /// Orbital quantum number l.
41 int l_;
42
43 /// Spin quantum number in the units of 1/2.
44 /** This variable can have only three values: -1,0,1. It is used to consruct the total angular
45 * momentum j = l + s/2. In case s = 0 total agular momentum j = l (no level splitting). */
46 int s_{0};
47
48 public:
49
50 /// Constructor.
51 explicit angular_momentum(int l__)
52 : l_(l__)
53 {
54 if (l__ < 0) {
55 RTE_THROW("l can't be negative");
56 }
57 }
58
59 /// Constructor.
60 explicit angular_momentum(int l__, int s__)
61 : l_(l__)
62 , s_(s__)
63 {
64 if (l__ < 0) {
65 RTE_THROW("l can't be negative");
66 }
67 if (s__ != -1 && s__ != 0 && s__ != 1) {
68 RTE_THROW("wrong value of s");
69 }
70 if (l__ == 0 && s__ == -1) {
71 RTE_THROW("incompatible combination of l and s quantum numbers");
72 }
73 }
74
75 /// Get orbital quantum number l.
76 inline auto l() const
77 {
78 return l_;
79 }
80
81 /// Get total angular momentum j = l +/- 1/2
82 inline auto j() const
83 {
84 return l_ + s_ / 2.0;
85 }
86
87 /// Get twice the total angular momentum 2j = 2l +/- 1
88 inline auto j2() const
89 {
90 return 2 * l_ + s_;
91 }
92
93 /// The size of the subshell for the angular momentum l or j.
94 /** This is the number of m_l values in the range [-l, l] or the number of
95 * m_j values in the range [-j, j] */
96 inline auto subshell_size() const
97 {
98 return j2() + 1;
99 }
100
101 /// Get spin quantum number s.
102 inline auto s() const
103 {
104 return s_;
105 }
106};

108class radial_functions_index
109{
110 private:
111 int size_{0};
112 std::vector<std::vector<std::array<int, 2>>> index_by_j_order_;
113 std::vector<int> l_;
114 std::vector<int> s_;
115 std::vector<int> o_;
116 public:
117 void add(angular_momentum am__)
118 {
119 /* current l */
120 auto l = am__.l();
121 /* current s */
122 auto s = am__.s();
123
124 if (s != 0 && l > 0) {
125 RTE_THROW("for l > 0 full-j radial functions are added in pairs");
126 }
127
128 /* make sure that the space is available */
129 if (static_cast<int>(index_by_j_order_.size()) < l + 1) {
130 index_by_j_order_.resize(l + 1);
131 }
132
133 std::array<int, 2> idx({-1, -1});
134 /* std::max(s, 0) maps s = -1 -> 0, s = 0 -> 0, s = 1 -> 1 */
135 idx[std::max(s, 0)] = size_;
136 /* current order */
137 auto o = static_cast<int>(index_by_j_order_[l].size());
138 index_by_j_order_[l].push_back(idx);
139 l_.push_back(l);
140 s_.push_back(s);
141 o_.push_back(o);
142 size_++;
143 }
144
145 void add(angular_momentum am1__, angular_momentum am2__)
146 {
147 /* current l */
148 auto l1 = am1__.l();
149 /* current s */
150 auto s1 = am1__.s();
151 /* current l */
152 auto l2 = am2__.l();
153 /* current s */
154 auto s2 = am2__.s();
155
156 if (l1 != l2) {
157 RTE_THROW("orbital quantum numbers are different");
158 }
159
160 if (s1 == s2) {
161 RTE_THROW("spin quantum numbers are the same");
162 }
163
164 if (s1 * s2 == 0) {
165 RTE_THROW("spin quantum numbers can't be zero in case of full orbital momentum");
166 }
167
168 /* make sure that the space is available */
169 if (static_cast<int>(index_by_j_order_.size()) < l1 + 1) {
170 index_by_j_order_.resize(l1 + 1);
171 }
172
173 /* current order */
174 auto o = static_cast<int>(index_by_j_order_[l1].size());
175
176 std::array<int, 2> idx({-1, -1});
177 idx[std::max(s1, 0)] = size_;
178 idx[std::max(s2, 0)] = size_ + 1;
179
180 l_.push_back(l1);
181 l_.push_back(l2);
182 s_.push_back(s1);
183 s_.push_back(s2);
184 o_.push_back(o);
185 o_.push_back(o);
186
187 index_by_j_order_[l1].push_back(idx);
188 size_ += 2;
189 }
190
191 inline auto am(int idx__) const
192 {
193 return angular_momentum(l_[idx__], s_[idx__]);
194 }
195
196 inline auto order(int idx__) const
197 {
198 return o_[idx__];
199 }
200
201 inline auto lmax() const
202 {
203 return static_cast<int>(index_by_j_order_.size()) - 1;
204 }
205
206 inline auto max_order(int l__) const
207 {
208 return static_cast<int>(index_by_j_order_[l__].size());
209 }
210
211 inline auto index_of(angular_momentum am__, int order__) const
212 {
213 return index_by_j_order_[am__.l()][order__][std::max(am__.s(), 0)];
214 }
215
216 auto full_j(int l__, int o__) const
217 {
218 /* look at index of l + s */
219 if (index_by_j_order_[l__][o__][1] >= 0) {
220 return true;
221 } else {
222 return false;
223 }
224 }
225
226 auto subshell(int l__, int o__) const
227 {
228 if (full_j(l__, o__)) {
229 if (l__ == 0) {
230 return std::vector<angular_momentum>({angular_momentum(l__, 1)});
231 } else {
232 return std::vector<angular_momentum>({angular_momentum(l__, -1), angular_momentum(l__, 1)});
233 }
234 } else {
235 return std::vector<angular_momentum>({angular_momentum(l__)});
236 }
237 }
238
239 inline auto size() const
240 {
241 return size_;
242 }
243
244 inline auto subshell_size(int l__, int o__) const
245 {
246 int size{0};
247 for (auto j: subshell(l__, o__)) {
248 size += j.subshell_size();
249 }
250 return size;
251 }

451/// Descriptor for the atomic radial functions.
452/** The radial functions \f$ f_{\ell \nu}(r) \f$ are labeled by two indices: orbital quantum number \f$ \ell \f$ and
453 * an order \f$ \nu \f$ for a given \f$ \ell \f$. Radial functions can be any of augmented waves of local orbitals
454 * (in case of FP-LAPW) or bete projectors, atomic or Hubbard wave functions in case of PP-PW.
455 */
456struct radial_function_index_descriptor
457{
458 /// Orbital quantum number \f$ \ell \f$.
459 int l;
460
461 /// Total angular momentum
462 double j;
463
464 /// Order of a function for a given \f$ \ell \f$.
465 int order;
466
467 /// If this is a local orbital radial function, idxlo is it's index in the list of local orbital descriptors.
468 int idxlo;
469
470 /// Constructor.
471 radial_function_index_descriptor(int l, int order, int idxlo = -1)
472 : l(l)
473 , order(order)
474 , idxlo(idxlo)
475 {
476 assert(l >= 0);
477 assert(order >= 0);
478 }
479
480 radial_function_index_descriptor(int l, double j, int order, int idxlo = -1)
481 : l(l)
482 , j(j)
483 , order(order)
484 , idxlo(idxlo)
485 {
486 assert(l >= 0);
487 assert(order >= 0);
488 }
489};

491/// A helper class to establish various index mappings for the atomic radial functions.
492class radial_functions_index
493{
494 private:
495 /// A list of radial function index descriptors.
496 /** This list establishes a mapping \f$ f_{\mu}(r) \leftrightarrow f_{\ell \nu}(r) \f$ between a
497 * composite index \f$ \mu \f$ of radial functions and
498 * corresponding \f$ \ell \nu \f$ indices, where \f$ \ell \f$ is the orbital quantum number and
499 * \f$ \nu \f$ is the order of radial function for a given \f$ \ell \f$. */
500 std::vector<radial_function_index_descriptor> radial_function_index_descriptors_;
501
502 mdarray<int, 2> index_by_l_order_;
503
504 mdarray<int, 1> index_by_idxlo_;
505
506 /// Number of radial functions for each angular momentum quantum number.
507 std::vector<int> num_rf_;
508
509 /// Number of local orbitals for each angular momentum quantum number.
510 std::vector<int> num_lo_;
511
512 // Maximum number of radial functions across all angular momentums.
513 int max_num_rf_;
514
515 /// Maximum orbital quantum number of augmented-wave radial functions.
516 int lmax_aw_;
517
518 /// Maximum orbital quantum number of local orbital radial functions.
519 int lmax_lo_;
520
521 /// Maximum orbital quantum number of radial functions.
522 int lmax_;
523
524 public:
525 /// Initialize a list of radial functions from the list of local orbitals.
526 template <typename T>
527 void init(std::vector<T> const& lo_descriptors__)
528 {
529 /* create an empty descriptor */
530 std::vector<radial_solution_descriptor_set> aw_descriptors;
531 this->init(aw_descriptors, lo_descriptors__);
532 }
533
534 /// Initialize a list of radial functions from the list of APW radial functions and the list of local orbitals.
535 template <typename T>
536 void init(std::vector<radial_solution_descriptor_set> const& aw_descriptors,
537 std::vector<T> const& lo_descriptors)
538 {
539 lmax_aw_ = static_cast<int>(aw_descriptors.size()) - 1;
540 lmax_lo_ = -1;
541 for (size_t idxlo = 0; idxlo < lo_descriptors.size(); idxlo++) {
542 int l = lo_descriptors[idxlo].l;
543 lmax_lo_ = std::max(lmax_lo_, l);
544 }
545
546 lmax_ = std::max(lmax_aw_, lmax_lo_);
547
548 num_rf_ = std::vector<int>(lmax_ + 1, 0);
549 num_lo_ = std::vector<int>(lmax_ + 1, 0);
550
551 max_num_rf_ = 0;
552
553 radial_function_index_descriptors_.clear();
554
555 for (int l = 0; l <= lmax_aw_; l++) {
556 assert(aw_descriptors[l].size() <= 3);
557
558 for (size_t order = 0; order < aw_descriptors[l].size(); order++) {
559 radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l]));
560 num_rf_[l]++;
561 }
562 }
563
564 for (int idxlo = 0; idxlo < static_cast<int>(lo_descriptors.size()); idxlo++) {
565 int l = lo_descriptors[idxlo].l;
566 radial_function_index_descriptors_.push_back(
567 radial_function_index_descriptor(l, lo_descriptors[idxlo].total_angular_momentum, num_rf_[l], idxlo));
568 num_rf_[l]++;
569 num_lo_[l]++;
570 }
571
572 for (int l = 0; l <= lmax_; l++) {
573 max_num_rf_ = std::max(max_num_rf_, num_rf_[l]);
574 }
575
576 index_by_l_order_ = mdarray<int, 2>(lmax_ + 1, max_num_rf_);
577
578 if (lo_descriptors.size()) {
579 index_by_idxlo_ = mdarray<int, 1>(lo_descriptors.size());
580 }
581
582 for (int i = 0; i < (int)radial_function_index_descriptors_.size(); i++) {
583 int l = radial_function_index_descriptors_[i].l;
584 int order = radial_function_index_descriptors_[i].order;
585 int idxlo = radial_function_index_descriptors_[i].idxlo;
586 index_by_l_order_(l, order) = i;
587 if (idxlo >= 0) {
588 index_by_idxlo_(idxlo) = i;
589 }
590 }
591 }

626 }

632 }

637 }

641 return lmax_lo_;
642 }
