25 inline void Potential::init_PAW()
26 {
27  paw_potential_data_.clear();
28  if (!unit_cell_.num_paw_atoms()) {
29  return;
30  }
31 
32  for (int i = 0; i < unit_cell_.spl_num_paw_atoms().local_size(); i++) {
33  int ia_paw = unit_cell_.spl_num_paw_atoms(i);
34  int ia = unit_cell_.paw_atom_index(ia_paw);
35 
36  auto& atom = unit_cell_.atom(ia);
37 
38  auto& atom_type = atom.type();
39 
40  int l_max = 2 * atom_type.indexr().lmax_lo();
41  int lm_max_rho = utils::lmmax(l_max);
42 
43  paw_potential_data_t ppd;
44 
45  ppd.atom_ = &atom;
46 
47  ppd.ia = ia;
48 
49  ppd.ia_paw = ia_paw;
50 
51  /* allocate potential arrays */
52  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
53  ppd.ae_potential_.push_back(Spheric_function<spectral, double>(lm_max_rho, ppd.atom_->radial_grid()));
54  ppd.ps_potential_.push_back(Spheric_function<spectral, double>(lm_max_rho, ppd.atom_->radial_grid()));
55  }
56 
57  ppd.core_energy_ = atom_type.paw_core_energy();
58 
59  paw_potential_data_.push_back(std::move(ppd));
60  }
61 
62  for (int i = 0; i < unit_cell_.num_paw_atoms(); i++) {
63  int ia = unit_cell_.paw_atom_index(i);
64  int bs = unit_cell_.atom(ia).mt_basis_size();
65  max_paw_basis_size_ = std::max(max_paw_basis_size_, bs);
66  }
67 
68  /* initialize dij matrix */
69  paw_dij_ = mdarray<double, 4>(max_paw_basis_size_, max_paw_basis_size_, ctx_.num_mag_dims() + 1, unit_cell_.num_paw_atoms(),
70  memory_t::host, "paw_dij_");
71 
72  /* allocate PAW energy array */
73  paw_hartree_energies_.resize(unit_cell_.num_paw_atoms());
74  paw_xc_energies_.resize(unit_cell_.num_paw_atoms());
75  paw_core_energies_.resize(unit_cell_.num_paw_atoms());
76  paw_one_elec_energies_.resize(unit_cell_.num_paw_atoms());
77 }

79 inline void Potential::generate_PAW_effective_potential(Density const& density)
80 {
81  PROFILE("sirius::Potential::generate_PAW_effective_potential");
82 
83  if (!unit_cell_.num_paw_atoms()) {
84  return;
85  }
86 
87  /* zero PAW arrays */
88  std::fill(paw_one_elec_energies_.begin(), paw_one_elec_energies_.end(), 0.0);
89  std::fill(paw_hartree_energies_.begin(), paw_hartree_energies_.end(), 0.0);
90  std::fill(paw_xc_energies_.begin(), paw_xc_energies_.end(), 0.0);
91 
92  /* zero Dij */
93  paw_dij_.zero();
94 
95  /* calculate xc and hartree for atoms */
96  for(int i = 0; i < unit_cell_.spl_num_paw_atoms().local_size(); i++) {
97  calc_PAW_local_potential(paw_potential_data_[i],
98  density.ae_paw_atom_density(i),
99  density.ps_paw_atom_density(i));
100  }
101 
102 
103  /* calculate PAW Dij matrix */
104  #pragma omp parallel for
105  for(int i = 0; i < unit_cell_.spl_num_paw_atoms().local_size(); i++) {
106  calc_PAW_local_Dij(paw_potential_data_[i], paw_dij_);
107 
108  calc_PAW_one_elec_energy(paw_potential_data_[i], density.density_matrix(), paw_dij_);
109  }
110 
111  // collect Dij and add to atom d_mtrx
112  comm_.allreduce(&paw_dij_(0, 0, 0, 0), static_cast<int>(paw_dij_.size()));
113 
114  if (ctx_.control().print_checksum_ && comm_.rank() == 0) {
115  auto cs = paw_dij_.checksum();
116  utils::print_checksum("paw_dij", cs);
117  }
118 
119  // add paw Dij to uspp Dij
120  add_paw_Dij_to_atom_Dmtrx();
121 
122  // calc total energy
123  double energies[] = {0.0, 0.0, 0.0, 0.0};
124 
125  for(int ia = 0; ia < unit_cell_.spl_num_paw_atoms().local_size(); ia++) {
126  energies[0] += paw_potential_data_[ia].hartree_energy_;
127  energies[1] += paw_potential_data_[ia].xc_energy_;
128  energies[2] += paw_potential_data_[ia].one_elec_energy_;
129  energies[3] += paw_potential_data_[ia].core_energy_; // it is light operation
130  }
131 
132  comm_.allreduce(&energies[0], 4);
133 
134  paw_hartree_total_energy_ = energies[0];
135  paw_xc_total_energy_ = energies[1];
136  paw_one_elec_energy_ = energies[2];
137  paw_total_core_energy_ = energies[3];
138 }

140 inline double Potential::xc_mt_PAW_nonmagnetic(Spheric_function<spectral, double>& full_potential,
141  Spheric_function<spectral, double> const& full_density,
142  std::vector<double> const& rho_core)
143 {
144  int lmmax = static_cast<int>(full_density.size(0));
145 
146  Radial_grid<double> const& rgrid = full_density.radial_grid();
147 
148  /* new array to store core and valence densities */
149  Spheric_function<spectral, double> full_rho_lm_sf_new(lmmax, rgrid);
150 
151  full_rho_lm_sf_new.zero();
152  full_rho_lm_sf_new += full_density;
153 
154  double invY00 = 1.0 / y00 ;
155 
156  /* adding core part */
157  for (int ir = 0; ir < rgrid.num_points(); ir++) {
158  full_rho_lm_sf_new(0,ir) += invY00 * rho_core[ir];
159  }
160 
161  Spheric_function<spatial,double> full_rho_tp_sf = transform(sht_.get(), full_rho_lm_sf_new);
162 
163  /* create potential in theta phi */
164  Spheric_function<spatial,double> vxc_tp_sf(sht_->num_points(), rgrid);
165 
166  /* create energy in theta phi */
167  Spheric_function<spatial,double> exc_tp_sf(sht_->num_points(), rgrid);
168 
169  xc_mt_nonmagnetic(rgrid, xc_func_, full_rho_lm_sf_new, full_rho_tp_sf, vxc_tp_sf, exc_tp_sf);
170 
171  full_potential += transform(sht_.get(), vxc_tp_sf);
172 
173  /* calculate energy */
174  Spheric_function<spectral,double> exc_lm_sf = transform(sht_.get(), exc_tp_sf );
175 
176  return inner(exc_lm_sf, full_rho_lm_sf_new);
177 }

179 inline double Potential::xc_mt_PAW_collinear(std::vector<Spheric_function<spectral, double>>& potential,
180  std::vector<Spheric_function<spectral, double>> const& density,
181  std::vector<double> const& rho_core)
182 {
183  int lmsize_rho = static_cast<int>(density[0].size(0));
184 
185  Radial_grid<double> const& rgrid = density[0].radial_grid();
186 
187  /* new array to store core and valence densities */
188  Spheric_function<spectral, double> full_rho_lm_sf_new(lmsize_rho, rgrid);
189 
190  full_rho_lm_sf_new.zero();
191  full_rho_lm_sf_new += density[0];
192 
193  double invY00 = 1 / y00;
194 
195  /* adding core part */
196  for (int ir = 0; ir < rgrid.num_points(); ir++ ) {
197  full_rho_lm_sf_new(0,ir) += invY00 * rho_core[ir];
198  }
199 
200  // calculate spin up spin down density components in lm components
201  // up = 1/2 ( rho + magn ); down = 1/2 ( rho - magn )
202  Spheric_function<spectral, double> rho_u_lm_sf = 0.5 * (full_rho_lm_sf_new + density[1]);
203  Spheric_function<spectral, double> rho_d_lm_sf = 0.5 * (full_rho_lm_sf_new - density[1]);
204 
205  // transform density to theta phi components
206  Spheric_function<spatial, double> rho_u_tp_sf = transform(sht_.get(), rho_u_lm_sf );
207  Spheric_function<spatial, double> rho_d_tp_sf = transform(sht_.get(), rho_d_lm_sf );
208 
209  // create potential in theta phi
210  Spheric_function<spatial, double> vxc_u_tp_sf(sht_->num_points(), rgrid);
211  Spheric_function<spatial, double> vxc_d_tp_sf(sht_->num_points(), rgrid);
212 
213  // create energy in theta phi
214  Spheric_function<spatial, double> exc_tp_sf(sht_->num_points(), rgrid);
215 
216  // calculate XC
217  xc_mt_magnetic(rgrid, xc_func_,
218  rho_u_lm_sf, rho_u_tp_sf,
219  rho_d_lm_sf, rho_d_tp_sf,
220  vxc_u_tp_sf, vxc_d_tp_sf,
221  exc_tp_sf);
222 
223  // transform back in lm
224  potential[0] += transform(sht_.get(), 0.5 * (vxc_u_tp_sf + vxc_d_tp_sf) );
225  potential[1] += transform(sht_.get(), 0.5 * (vxc_u_tp_sf - vxc_d_tp_sf));
226 
227  //------------------------
228  //--- calculate energy ---
229  //------------------------
230  Spheric_function<spectral,double> exc_lm_sf = transform(sht_.get(), exc_tp_sf );
231 
232  return inner(exc_lm_sf, full_rho_lm_sf_new);
233 }

236 inline double Potential::xc_mt_PAW_noncollinear(std::vector<Spheric_function<spectral, double>>& potential,
237  std::vector<Spheric_function<spectral, double>> const& density,
238  std::vector<double> const& rho_core)
239 {
240  if (density.size() != 4 || potential.size() != 4){
241  TERMINATE("xc_mt_PAW_noncollinear FATAL ERROR!")
242  }
243 
244  Radial_grid<double> const& rgrid = density[0].radial_grid();
245 
246  /* transform density to theta phi components */
247  std::vector<Spheric_function<spatial, double>> rho_tp(density.size());
248 
249  for (size_t i = 0; i < density.size(); i++){
250  rho_tp[i] = transform(sht_.get(), density[i]);
251  }
252 
253  /* transform 4D magnetization to spin-up, spin-down form (correct for LSDA) rho Â± |magn| */
254  Spheric_function<spatial, double> rho_u_tp(sht_->num_points(), rgrid);
255  Spheric_function<spatial, double> rho_d_tp(sht_->num_points(), rgrid);
256 
257  //#pragma omp parallel for
258  for (int ir = 0; ir < rgrid.num_points(); ir++ ) {
259  for (int itp = 0; itp < sht_->num_points(); itp++ ) {
260  vector3d<double> magn({rho_tp[1](itp, ir), rho_tp[2](itp, ir), rho_tp[3](itp, ir)});
261  double norm = magn.length();
262 
263  rho_u_tp(itp, ir) = 0.5 * (rho_tp[0](itp, ir) + rho_core[ir] + norm);
264  rho_d_tp(itp, ir) = 0.5 * (rho_tp[0](itp, ir) + rho_core[ir] - norm);
265  }
266  }
267 
268  /* in lm representation */
269  Spheric_function<spectral, double> rho_u_lm = transform(sht_.get(), rho_u_tp);
270  Spheric_function<spectral, double> rho_d_lm = transform(sht_.get(), rho_d_tp);
271 
272  // allocate potential in theta phi
273  Spheric_function<spatial, double> vxc_u_tp(sht_->num_points(), rgrid);
274  Spheric_function<spatial, double> vxc_d_tp(sht_->num_points(), rgrid);
275 
276  // allocate energy in theta phi
277  Spheric_function<spatial, double> exc_tp(sht_->num_points(), rgrid);
278 
279  // calculate XC
280  xc_mt_magnetic(rgrid, xc_func_,
281  rho_u_lm, rho_u_tp,
282  rho_d_lm, rho_d_tp,
283  vxc_u_tp, vxc_d_tp,
284  exc_tp);
285 
286  /* allocate 4D potential in theta phi components */
287  std::vector<Spheric_function<spatial, double>> vxc_tp;
288 
289  /* allocate the rest */
290  for (size_t i = 0; i < potential.size(); i++){
291  vxc_tp.push_back(Spheric_function<spatial, double>(sht_->num_points(), rgrid));
292  }
293 
294  /* transform back potential from up/down to 4D form*/
295  //#pragma omp parallel for
296  for (int ir = 0; ir < rgrid.num_points(); ir++ ) {
297  for (int itp = 0; itp < sht_->num_points(); itp++ ) {
298  /* get total potential and field abs value*/
299  double pot = 0.5 * (vxc_u_tp(itp, ir) + vxc_d_tp(itp, ir));
300  double field = 0.5 * (vxc_u_tp(itp, ir) - vxc_d_tp(itp, ir));
301 
302  /* get unit magnetization vector*/
303  vector3d<double> magn({rho_tp[1](itp, ir), rho_tp[2](itp, ir), rho_tp[3](itp, ir)});
304  double norm = magn.length();
305  magn = magn * (norm > 0.0 ? field / norm : 0.0) ;
306 
307  /* add total potential and effective field values at current point */
308  vxc_tp[0](itp, ir) = pot;
309  for (int i: {0,1,2}) {
310  vxc_tp[i+1](itp, ir) = magn[i];
311  }
312  }
313  }
314 
315  /* transform back to lm- domain */
316  for(size_t i = 0; i < density.size(); i++){
317  potential[i] += transform(sht_.get(), vxc_tp[i]);
318  }
319 
320  /* transform to lm- domain */
321  Spheric_function<spectral, double> exc_lm = transform(sht_.get(), exc_tp);
322 
323  return inner(exc_lm, rho_u_lm + rho_d_lm);
324 }

327 inline double Potential::calc_PAW_hartree_potential(Atom& atom,
328  Spheric_function<spectral, double> const& full_density,
329  Spheric_function<spectral, double>& full_potential)
330 {
331  int lmsize_rho = static_cast<int>(full_density.size(0));
332 
333  const Radial_grid<double>& grid = full_density.radial_grid();
334 
335  /* array passed to poisson solver */
336  Spheric_function<spectral, double> atom_pot_sf(lmsize_rho, grid);
337  atom_pot_sf.zero();
338 
339  poisson_vmt<true>(atom, full_density, atom_pot_sf);
340 
341  /* add hartree contribution */
342  full_potential += atom_pot_sf;
343 
344  /* calculate energy */
345 
346  auto l_by_lm = utils::l_by_lm(utils::lmax(lmsize_rho));
347 
348  /* create array for integration */
349  std::vector<double> intdata(grid.num_points(),0);
350 
351  double hartree_energy{0};
352 
353  for (int lm=0; lm < lmsize_rho; lm++){
354  /* fill data to integrate */
355  for (int irad = 0; irad < grid.num_points(); irad++){
356  intdata[irad] = full_density(lm,irad) * full_potential(lm, irad) * grid[irad] * grid[irad];
357  }
358 
359  /* create spline from the data */
360  Spline<double> h_spl(grid,intdata);
361  hartree_energy += 0.5 * h_spl.integrate(0);
362  }
363 
364  return hartree_energy;
365 }

367 inline void Potential::calc_PAW_local_potential(paw_potential_data_t &ppd,
368  std::vector<Spheric_function<spectral, double>> const& ae_density,
369  std::vector<Spheric_function<spectral, double>> const& ps_density)
370 {
371  /* calculation of Hartree potential */
372  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
373  ppd.ae_potential_[i].zero();
374  ppd.ps_potential_[i].zero();
375  }
376 
377  double ae_hartree_energy = calc_PAW_hartree_potential(*ppd.atom_,
378  ae_density[0],
379  ppd.ae_potential_[0]);
380 
381  double ps_hartree_energy = calc_PAW_hartree_potential(*ppd.atom_,
382  ps_density[0],
383  ppd.ps_potential_[0]);
384 
385  ppd.hartree_energy_ = ae_hartree_energy - ps_hartree_energy;
386 
387  /* calculation of XC potential */
388  auto& ps_core = ppd.atom_->type().ps_core_charge_density();
389  auto& ae_core = ppd.atom_->type().paw_ae_core_charge_density();
390 
391  double ae_xc_energy = 0.0;
392  double ps_xc_energy = 0.0;
393 
394  switch (ctx_.num_mag_dims()) {
395  case 0: {
396  ae_xc_energy = xc_mt_PAW_nonmagnetic(ppd.ae_potential_[0], ae_density[0], ae_core);
397  ps_xc_energy = xc_mt_PAW_nonmagnetic(ppd.ps_potential_[0], ps_density[0], ps_core);
398  break;
399  }
400 
401  case 1: {
402  ae_xc_energy = xc_mt_PAW_collinear(ppd.ae_potential_, ae_density, ae_core);
403  ps_xc_energy = xc_mt_PAW_collinear(ppd.ps_potential_, ps_density, ps_core);
404  break;
405  }
406 
407  case 3:{
408  ae_xc_energy = xc_mt_PAW_noncollinear(ppd.ae_potential_, ae_density, ae_core);
409  ps_xc_energy = xc_mt_PAW_noncollinear(ppd.ps_potential_, ps_density, ps_core);
410  break;
411  }
412 
413  default:{
414  TERMINATE("PAW local potential error! Wrong number of spins!")
415  }
416  }
417 
418  /* save xc energy in pdd structure */
419  ppd.xc_energy_ = ae_xc_energy - ps_xc_energy;
420 }

422 inline void Potential::calc_PAW_local_Dij(paw_potential_data_t& pdd, mdarray<double, 4>& paw_dij)
423 {
424  int paw_ind = pdd.ia_paw;
425 
426  auto& atom_type = pdd.atom_->type();
427 
428  auto& paw_ae_wfs = atom_type.ae_paw_wfs_array();
429  auto& paw_ps_wfs = atom_type.ps_paw_wfs_array();
430 
431  /* get lm size for density */
432  int lmax = atom_type.indexr().lmax_lo();
433  int lmsize_rho = utils::lmmax(2 * lmax);
434 
435  auto l_by_lm = utils::l_by_lm(2 * lmax);
436 
437  Gaunt_coefficients<double> GC(lmax, 2 * lmax, lmax, SHT::gaunt_rlm);
438 
439  /* store integrals here */
440  mdarray<double, 3> integrals(lmsize_rho, atom_type.num_beta_radial_functions() * (atom_type.num_beta_radial_functions() + 1) / 2,
441  ctx_.num_mag_dims() + 1);
442 
443  auto& rgrid = atom_type.radial_grid();
444 
445  for(int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++ ){
446  auto &ae_atom_pot = pdd.ae_potential_[imagn];
447  auto &ps_atom_pot = pdd.ps_potential_[imagn];
448 
449  for (int irb2 = 0; irb2 < atom_type.num_beta_radial_functions(); irb2++){
450  for (int irb1 = 0; irb1 <= irb2; irb1++){
451  int iqij = (irb2 * (irb2 + 1)) / 2 + irb1;
452 
453  /* create array for integration */
454  std::vector<double> intdata(rgrid.num_points());
455 
456  for (int lm3 = 0; lm3 < lmsize_rho; lm3++) {
457  /* fill array */
458  for (int irad = 0; irad < rgrid.num_points(); irad++) {
459  double ae_part = paw_ae_wfs(irad, irb1) * paw_ae_wfs(irad, irb2);
460  double ps_part = paw_ps_wfs(irad, irb1) * paw_ps_wfs(irad, irb2) +
461  atom_type.q_radial_function(irb1, irb2, l_by_lm[lm3])(irad);
462 
463  intdata[irad] = ae_atom_pot(lm3, irad) * ae_part - ps_atom_pot(lm3, irad) * ps_part;
464  }
465 
466  /* create spline from data arrays */
467  Spline<double> dij_spl(rgrid, intdata);
468 
469  /* integrate */
470  integrals(lm3, iqij, imagn) = dij_spl.integrate(0);
471  }
472  }
473  }
474  }
475 
476  /* calculate Dij */
477  for (int ib2 = 0; ib2 < atom_type.mt_lo_basis_size(); ib2++) {
478  for (int ib1 = 0; ib1 <= ib2; ib1++) {
479 
480  /* get lm quantum numbers (lm index) of the basis functions */
481  int lm1 = atom_type.indexb(ib1).lm;
482  int lm2 = atom_type.indexb(ib2).lm;
483 
484  /* get radial basis functions indices */
485  int irb1 = atom_type.indexb(ib1).idxrf;
486  int irb2 = atom_type.indexb(ib2).idxrf;
487 
488  /* common index */
489  int iqij = (irb2 * (irb2 + 1)) / 2 + irb1;
490 
491  /* get num of non-zero GC */
492  int num_non_zero_gk = GC.num_gaunt(lm1,lm2);
493 
494  for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++ ){
495  /* add nonzero coefficients */
496  for (int inz = 0; inz < num_non_zero_gk; inz++) {
497  auto& lm3coef = GC.gaunt(lm1, lm2, inz);
498 
499  /* add to atom Dij an integral of dij array */
500  paw_dij(ib1, ib2, imagn, paw_ind) += lm3coef.coef * integrals(lm3coef.lm3, iqij, imagn);
501  }
502 
503  if (ib1 != ib2) {
504  paw_dij(ib2, ib1, imagn, paw_ind) = paw_dij(ib1, ib2, imagn, paw_ind);
505  }
506  }
507  }
508  }
509 }

511 inline double Potential::calc_PAW_one_elec_energy(paw_potential_data_t& pdd,
512  const mdarray<double_complex, 4>& density_matrix,
513  const mdarray<double, 4>& paw_dij)
514 {
515  int ia = pdd.ia;
516  int paw_ind = pdd.ia_paw;
517 
518  double_complex energy = 0.0;
519 
520  for (int ib2 = 0; ib2 < pdd.atom_->mt_lo_basis_size(); ib2++ ) {
521  for (int ib1 = 0; ib1 < pdd.atom_->mt_lo_basis_size(); ib1++ ) {
522  double dm[4]={0,0,0,0};
523  switch (ctx_.num_mag_dims()) {
524  case 3: {
525  dm[2] = 2 * std::real(density_matrix(ib1, ib2, 2, ia));
