31 /// Representation of a smooth (Fourier-transformable) periodic function.
32 /** The class is designed to handle periodic functions such as density or potential, defined on a regular FFT grid.
33  * The following functionality is expected:
34  * - access to real-space values
35  * - access to plane-wave coefficients
36  * - distribution of plane-wave coefficients over entire communicator
37  * - Fourier transformation using FFT communicator
38  * - gather PW coefficients into global array
39  */

161  void fft_transform(int direction__)
162  {
163  PROFILE("sirius::Smooth_periodic_function::fft_transform");
164 
165  assert(gvecp_ != nullptr);
166 
167  switch (direction__) {
168  case 1: {
169  gather_f_pw_fft();
170  fft_->transform<1>(f_pw_fft_.at(memory_t::host));
171  fft_->output(f_rg_.at(memory_t::host));
172  break;
173  }
174  case -1: {
175  fft_->input(f_rg_.at(memory_t::host));
176  fft_->transform<-1>(f_pw_fft_.at(memory_t::host));
177  int count = gvecp_->gvec_fft_slab().counts[gvecp_->comm_ortho_fft().rank()];
178  int offset = gvecp_->gvec_fft_slab().offsets[gvecp_->comm_ortho_fft().rank()];
179  std::memcpy(f_pw_local_.at(memory_t::host), f_pw_fft_.at(memory_t::host, offset),
180  count * sizeof(double_complex));
181  break;
182  }
183  default: {
184  TERMINATE("wrong fft direction");
185  }
186  }
187  }

290 /// Gradient of the function in the plane-wave domain.
291 /** Input functions is expected in the plane wave domain, output function is also in the plane-wave domain */
292 inline Smooth_periodic_vector_function<double> gradient(Smooth_periodic_function<double>& f__)
293 {
294  utils::timer t1("sirius::gradient");
295 
296  Smooth_periodic_vector_function<double> g(f__.fft(), f__.gvec_partition());
297 
298  #pragma omp parallel for schedule(static)
299  for (int igloc = 0; igloc < f__.gvec().count(); igloc++) {
300  auto G = f__.gvec().gvec_cart<index_domain_t::local>(igloc);
301  for (int x : {0, 1, 2}) {
302  g[x].f_pw_local(igloc) = f__.f_pw_local(igloc) * double_complex(0, G[x]);
303  }
304  }
305  return std::move(g);
306 }

309 /// Divergence of the vecor function.
310 /** Input is expected in the plane-wave domain, output is in real-space domain. */
311 inline Smooth_periodic_function<double> divergence(Smooth_periodic_vector_function<double>& g__)
312 {
313  utils::timer t1("sirius::divergence");
314 
315  Smooth_periodic_function<double> f(g__.fft(), g__.gvec_partition());
316  f.zero();
317  Smooth_periodic_function<double> g_tmp(g__.fft(), g__.gvec_partition());
318  for (int x : {0, 1, 2}) {
319  for (int igloc = 0; igloc < f.gvec().count(); igloc++) {
320  auto G = f.gvec().gvec_cart<index_domain_t::local>(igloc);
321  g_tmp.f_pw_local(igloc) = g__[x].f_pw_local(igloc) * double_complex(0, G[x]);
322  }
323  g_tmp.fft_transform(1);
324  for (int ir = 0; ir < f.fft().local_size(); ir++) {
325  f.f_rg(ir) += g_tmp.f_rg(ir);
326  }
327  }
328  return std::move(f);
329 }

331 /// Laplacian of the function in the plane-wave domain.
332 inline Smooth_periodic_function<double> laplacian(Smooth_periodic_function<double>& f__)
333 {
334  utils::timer t1("sirius::laplacian");
335 
336  Smooth_periodic_function<double> g(f__.fft(), f__.gvec_partition());
337 
338  #pragma omp parallel for schedule(static)
339  for (int igloc = 0; igloc < f__.gvec().count(); igloc++) {
340  auto G = f__.gvec().gvec_cart<index_domain_t::local>(igloc);
341  g.f_pw_local(igloc) = f__.f_pw_local(igloc) * double_complex(-std::pow(G.length(), 2), 0);
342  }
343 
344  return std::move(g);
345 }

347 template <typename T>
348 inline Smooth_periodic_function<T> dot(Smooth_periodic_vector_function<T>& vf__,
349  Smooth_periodic_vector_function<T>& vg__)
350 
351 {
352  utils::timer t1("sirius::dot");
353 
354  Smooth_periodic_function<T> result(vf__.fft(), vf__.gvec_partition());
355 
356  #pragma omp parallel for schedule(static)
357  for (int ir = 0; ir < vf__.fft().local_size(); ir++) {
358  double d{0};
359  for (int x : {0, 1, 2}) {
360  d += vf__[x].f_rg(ir) * vg__[x].f_rg(ir);
361  }
362  result.f_rg(ir) = d;
363  }
364 
365  return std::move(result);
366 }

368 /// Compute inner product <f|g>
369 template <typename T>
370 T inner(Smooth_periodic_function<T> const& f__, Smooth_periodic_function<T> const& g__)
371 {
372  utils::timer t1("sirius::Smooth_periodic_function|inner");
373 
374  assert(&f__.fft() == &g__.fft());
375 
376  T result_rg{0};
377 
378  #pragma omp parallel for schedule(static) reduction(+:result_rg)
379  for (int irloc = 0; irloc < f__.fft().local_size(); irloc++) {
380  // result_rg += type_wrapper<T>::bypass(std::conj(f__.f_rg(irloc)) * g__.f_rg(irloc));
381  result_rg += utils::conj(f__.f_rg(irloc)) * g__.f_rg(irloc);
382  }
383 
384  result_rg *= (f__.gvec().omega() / f__.fft().size());
385 
386  f__.fft().comm().allreduce(&result_rg, 1);
387 
388  return result_rg;
389 }
