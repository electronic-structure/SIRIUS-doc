38 /// Representation of a smooth (Fourier-transformable) periodic function.
39 /** The class is designed to handle periodic functions such as density or potential, defined on a regular FFT grid.
40  * The following functionality is provided:
41  * - access to real-space values
42  * - access to plane-wave coefficients
43  * - distribution of plane-wave coefficients over entire communicator
44  * - Fourier transformation using FFT communicator
45  * - gather PW coefficients into global array
46  */

206  void fft_transform(int direction__)
207  {
208  PROFILE("sirius::Smooth_periodic_function::fft_transform");
209 
210  assert(gvecp_ != nullptr);
211 
212  auto frg_ptr = (spfft_->local_slice_size() == 0) ? nullptr : &f_rg_[0];
213 
214  switch (direction__) {
215  case 1: {
216  if (gvecp_->comm_ortho_fft().size() != 1) {
217  gather_f_pw_fft();
218  }
219  spfft_->backward(reinterpret_cast<double const*>(f_pw_fft_.at(sddk::memory_t::host)), SPFFT_PU_HOST);
220  spfft_output(*spfft_, frg_ptr);
221  break;
222  }
223  case -1: {
224  spfft_input(*spfft_, frg_ptr);
225  spfft_->forward(SPFFT_PU_HOST, reinterpret_cast<double*>(f_pw_fft_.at(sddk::memory_t::host)),
226  SPFFT_FULL_SCALING);
227  if (gvecp_->comm_ortho_fft().size() != 1) {
228  int count = gvecp_->gvec_fft_slab().counts[gvecp_->comm_ortho_fft().rank()];
229  int offset = gvecp_->gvec_fft_slab().offsets[gvecp_->comm_ortho_fft().rank()];
230  std::memcpy(f_pw_local_.at(sddk::memory_t::host), f_pw_fft_.at(sddk::memory_t::host, offset),
231  count * sizeof(double_complex));
232  }
233  break;
234  }
235  default: {
236  throw std::runtime_error("wrong FFT direction");
237  }
238  }
239  }

353 /// Gradient of the function in the plane-wave domain.
354 /** Input functions is expected in the plane wave domain, output function is also in the plane-wave domain */
355 inline Smooth_periodic_vector_function<double> gradient(Smooth_periodic_function<double>& f__)
356 {
357  PROFILE("sirius::gradient");
358 
359  Smooth_periodic_vector_function<double> g(f__.spfft(), f__.gvec_partition());
360 
361  #pragma omp parallel for schedule(static)
362  for (int igloc = 0; igloc < f__.gvec().count(); igloc++) {
363  auto G = f__.gvec().gvec_cart<sddk::index_domain_t::local>(igloc);
364  for (int x : {0, 1, 2}) {
365  g[x].f_pw_local(igloc) = f__.f_pw_local(igloc) * double_complex(0, G[x]);
366  }
367  }
368  return g;
369 }

371 /// Divergence of the vecor function.
372 /** Input and output functions are in plane-wave domain */
373 inline Smooth_periodic_function<double> divergence(Smooth_periodic_vector_function<double>& g__)
374 {
375  PROFILE("sirius::divergence");
376 
377  /* resulting scalar function */
378  Smooth_periodic_function<double> f(g__.spfft(), g__.gvec_partition());
379  f.zero();
380  for (int x : {0, 1, 2}) {
381  for (int igloc = 0; igloc < f.gvec().count(); igloc++) {
382  auto G = f.gvec().gvec_cart<sddk::index_domain_t::local>(igloc);
383  f.f_pw_local(igloc) += g__[x].f_pw_local(igloc) * double_complex(0, G[x]);
384  }
385  }
386 
387  return f;
388 }

390 /// Laplacian of the function in the plane-wave domain.
391 inline Smooth_periodic_function<double> laplacian(Smooth_periodic_function<double>& f__)
392 {
393  PROFILE("sirius::laplacian");
394 
395  Smooth_periodic_function<double> g(f__.spfft(), f__.gvec_partition());
396 
397  #pragma omp parallel for schedule(static)
398  for (int igloc = 0; igloc < f__.gvec().count(); igloc++) {
399  auto G = f__.gvec().gvec_cart<sddk::index_domain_t::local>(igloc);
400  g.f_pw_local(igloc) = f__.f_pw_local(igloc) * double_complex(-std::pow(G.length(), 2), 0);
401  }
402 
403  return g;
404 }

406 template <typename T>
407 inline Smooth_periodic_function<T>
408 dot(Smooth_periodic_vector_function<T>& vf__, Smooth_periodic_vector_function<T>& vg__)
409 
410 {
411  PROFILE("sirius::dot");
412 
413  Smooth_periodic_function<T> result(vf__.spfft(), vf__.gvec_partition());
414 
415  #pragma omp parallel for schedule(static)
416  for (int ir = 0; ir < vf__.spfft().local_slice_size(); ir++) {
417  double d{0};
418  for (int x : {0, 1, 2}) {
419  d += vf__[x].f_rg(ir) * vg__[x].f_rg(ir);
420  }
421  result.f_rg(ir) = d;
422  }
423 
424  return result;
425 }

427 /// Compute inner product <f|g>
428 template <typename T, typename F>
429 inline T
430 inner_local(Smooth_periodic_function<T> const& f__, Smooth_periodic_function<T> const& g__, F&& theta__)
431 {
432  PROFILE("sirius::Smooth_periodic_function|inner_local");
433 
434  assert(&f__.spfft() == &g__.spfft());
435 
436  T result_rg{0};
437 
438  //#pragma omp parallel for schedule(static) reduction(+:result_rg)
439  for (int irloc = 0; irloc < f__.spfft().local_slice_size(); irloc++) {
440  result_rg += utils::conj(f__.f_rg(irloc)) * g__.f_rg(irloc) * theta__(irloc);
441  }
442 
443  result_rg *= (f__.gvec().omega() / spfft_grid_size(f__.spfft()));
444 
445  return result_rg;
446 }
