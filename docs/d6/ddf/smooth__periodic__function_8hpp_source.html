38/// Representation of a smooth (Fourier-transformable) periodic function.
39/** The class is designed to handle periodic functions such as density or potential, defined on a regular FFT grid.
40 * The following functionality is provided:
41 * - access to real-space values
42 * - access to plane-wave coefficients
43 * - distribution of plane-wave coefficients over entire communicator
44 * - Fourier transformation using FFT communicator
45 * - gather PW coefficients into global array
46 */

108 "Smooth_periodic_function.f_pw_fft_");
109 }
110 f_pw_fft_.zero();
111 } else {
112 /* alias to f_pw_local array */
113 f_pw_fft_ = sddk::mdarray<complex_type<T>, 1>(&f_pw_local_[0], gvecp_->gvec().count());

204 void fft_transform(int direction__)
205 {
206 PROFILE("sirius::Smooth_periodic_function::fft_transform");
207
208 assert(gvecp_ != nullptr);
209
210 auto frg_ptr = (spfft_->local_slice_size() == 0) ? nullptr : &f_rg_[0];
211
212#if defined(USE_FP32)
213 using precision_type = typename std::conditional<std::is_same<real_type<T>, double>::value, double, float>::type;
214#else
215 using precision_type = double;
216#endif
217 switch (direction__) {
218 case 1: {
219 if (gvecp_->comm_ortho_fft().size() != 1) {
220 gather_f_pw_fft();
221 }
222 spfft_->backward(reinterpret_cast<precision_type const*>(f_pw_fft_.at(sddk::memory_t::host)), SPFFT_PU_HOST);
223 spfft_output(*spfft_, frg_ptr);
224 break;
225 }
226 case -1: {
227 spfft_input(*spfft_, frg_ptr);
228 spfft_->forward(SPFFT_PU_HOST, reinterpret_cast<precision_type*>(f_pw_fft_.at(sddk::memory_t::host)),
229 SPFFT_FULL_SCALING);
230 if (gvecp_->comm_ortho_fft().size() != 1) {
231 int count = gvecp_->gvec_fft_slab().counts[gvecp_->comm_ortho_fft().rank()];
232 int offset = gvecp_->gvec_fft_slab().offsets[gvecp_->comm_ortho_fft().rank()];
233 std::memcpy(f_pw_local_.at(sddk::memory_t::host), f_pw_fft_.at(sddk::memory_t::host, offset),
234 count * sizeof(complex_type<T>));
235 }
236 break;
237 }
238 default: {
239 throw std::runtime_error("wrong FFT direction");
240 }
241 }
242 }

244 inline std::vector<complex_type<T>> gather_f_pw()
245 {
246 PROFILE("sirius::Smooth_periodic_function::gather_f_pw");
247
248 std::vector<complex_type<T>> fpw(gvecp_->gvec().num_gvec());
249 gvec().comm().allgather(&f_pw_local_[0], fpw.data(), gvec().count(), gvec().offset());

260 void add(Smooth_periodic_function<T> const& g__)
261 {
262 #pragma omp parallel for schedule(static)
263 for (int irloc = 0; irloc < this->spfft_->local_slice_size(); irloc++) {
264 this->f_rg_(irloc) += g__.f_rg(irloc);
265 }
266 }

272 return cs;

282 inline uint64_t hash_f_pw() const
283 {
284 auto h = f_pw_local_.hash();
285 gvecp_->gvec().comm().bcast(&h, 1, 0);
286
287 for (int r = 1; r < gvecp_->gvec().comm().size(); r++) {
288 h = f_pw_local_.hash(h);
289 gvecp_->gvec().comm().bcast(&h, 1, r);
290 }
291 return h;
292 }
293
294 inline uint64_t hash_f_rg() const
295 {
296 auto comm = sddk::Communicator(spfft_->communicator());
297
298 uint64_t h;
299 for (int r = 0; r < comm.size(); r++) {
300 if (r == 0) {
301 h = f_rg_.hash();
302 } else {
303 h = f_rg_.hash(h);
304 }
305 comm.bcast(&h, 1, r);
306 }
307 return h;
308 }

357/// Gradient of the function in the plane-wave domain.
358/** Input functions is expected in the plane wave domain, output function is also in the plane-wave domain */
359template <typename T>
360inline Smooth_periodic_vector_function<T> gradient(Smooth_periodic_function<T>& f__)
361{
362 PROFILE("sirius::gradient");
363
364 Smooth_periodic_vector_function<T> g(f__.spfft(), f__.gvec_partition());
365
366 #pragma omp parallel for schedule(static)
367 for (int igloc = 0; igloc < f__.gvec().count(); igloc++) {
368 auto G = f__.gvec().template gvec_cart<sddk::index_domain_t::local>(igloc);
369 for (int x : {0, 1, 2}) {
370 g[x].f_pw_local(igloc) = f__.f_pw_local(igloc) * std::complex<T>(0, G[x]);
371 }
372 }
373 return g;
374}

376/// Divergence of the vecor function.
377/** Input and output functions are in plane-wave domain */
378template <typename T>
379inline Smooth_periodic_function<T> divergence(Smooth_periodic_vector_function<T>& g__)
380{
381 PROFILE("sirius::divergence");
382
383 /* resulting scalar function */
384 Smooth_periodic_function<T> f(g__.spfft(), g__.gvec_partition());
385 f.zero();
386 for (int x : {0, 1, 2}) {
387 for (int igloc = 0; igloc < f.gvec().count(); igloc++) {
388 auto G = f.gvec().template gvec_cart<sddk::index_domain_t::local>(igloc);
389 f.f_pw_local(igloc) += g__[x].f_pw_local(igloc) * std::complex<T>(0, G[x]);
390 }
391 }
392
393 return f;
394}

396/// Laplacian of the function in the plane-wave domain.
397template <typename T>
398inline Smooth_periodic_function<T> laplacian(Smooth_periodic_function<T>& f__)
399{
400 PROFILE("sirius::laplacian");
401
402 Smooth_periodic_function<T> g(f__.spfft(), f__.gvec_partition());
403
404 #pragma omp parallel for schedule(static)
405 for (int igloc = 0; igloc < f__.gvec().count(); igloc++) {
406 auto G = f__.gvec().template gvec_cart<sddk::index_domain_t::local>(igloc);
407 g.f_pw_local(igloc) = f__.f_pw_local(igloc) * std::complex<T>(-std::pow(G.length(), 2), 0);
408 }
409
410 return g;
411}

413template <typename T>
414inline Smooth_periodic_function<T>
415dot(Smooth_periodic_vector_function<T>& vf__, Smooth_periodic_vector_function<T>& vg__)
416
417{
418 PROFILE("sirius::dot");
419
420 Smooth_periodic_function<T> result(vf__.spfft(), vf__.gvec_partition());
421
422 #pragma omp parallel for schedule(static)
423 for (int ir = 0; ir < vf__.spfft().local_slice_size(); ir++) {
424 T d{0};
425 for (int x : {0, 1, 2}) {
426 d += vf__[x].f_rg(ir) * vg__[x].f_rg(ir);
427 }
428 result.f_rg(ir) = d;
429 }
430
431 return result;
432}

434/// Compute inner product <f|g>
435template <typename T, typename F>
436inline T
437inner_local(Smooth_periodic_function<T> const& f__, Smooth_periodic_function<T> const& g__, F&& theta__)
438{
439 assert(&f__.spfft() == &g__.spfft());
440
441 T result_rg{0};
442
443 //#pragma omp parallel for schedule(static) reduction(+:result_rg)
444 for (int irloc = 0; irloc < f__.spfft().local_slice_size(); irloc++) {
445 result_rg += utils::conj(f__.f_rg(irloc)) * g__.f_rg(irloc) * theta__(irloc);
446 }
447
448 result_rg *= (f__.gvec().omega() / spfft_grid_size(f__.spfft()));
449
450 return result_rg;
451}
452template <typename T, typename F>
453inline T
454inner(Smooth_periodic_function<T> const& f__, Smooth_periodic_function<T> const& g__, F&& theta__)
455{
456 PROFILE("sirius::inner");
457
458 T result_rg = inner_local(f__, g__, std::forward<F>(theta__));
459 sddk::Communicator(f__.spfft().communicator()).allreduce(&result_rg, 1);
460
461 return result_rg;
462}
