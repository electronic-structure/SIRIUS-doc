254 template <bool add_pseudo_core__>
255 void Potential::xc_rg_magnetic(Density const& density__)
256 {
257  PROFILE("sirius::Potential::xc_rg_magnetic");
258 
259  bool is_gga = is_gradient_correction();
260 
261  int num_points = ctx_.spfft().local_slice_size();
262 
263  auto result = get_rho_up_dn<add_pseudo_core__>(density__, add_delta_rho_xc_, add_delta_mag_xc_);
264 
265  auto& rho_up = *result[0];
266  auto& rho_dn = *result[1];
267 
268  if (ctx_.cfg().control().print_hash()) {
269  auto h1 = rho_up.hash_f_rg();
270  auto h2 = rho_dn.hash_f_rg();
271  if (ctx_.comm().rank() == 0) {
272  utils::print_hash("rho_up", h1);
273  utils::print_hash("rho_dn", h2);
274  }
275  }
276 
277  Smooth_periodic_vector_function<double> grad_rho_up;
278  Smooth_periodic_vector_function<double> grad_rho_dn;
279  Smooth_periodic_function<double> grad_rho_up_grad_rho_up;
280  Smooth_periodic_function<double> grad_rho_up_grad_rho_dn;
281  Smooth_periodic_function<double> grad_rho_dn_grad_rho_dn;
282 
283  if (is_gga) {
284  PROFILE("sirius::Potential::xc_rg_magnetic|grad1");
285  /* get plane-wave coefficients of densities */
286  rho_up.fft_transform(-1);
287  rho_dn.fft_transform(-1);
288 
289  /* generate pw coeffs of the gradient and laplacian */
290  grad_rho_up = gradient(rho_up);
291  grad_rho_dn = gradient(rho_dn);
292 
293  /* gradient in real space */
294  for (int x: {0, 1, 2}) {
295  grad_rho_up[x].fft_transform(1);
296  grad_rho_dn[x].fft_transform(1);
297  }
298 
299  /* product of gradients */
300  grad_rho_up_grad_rho_up = dot(grad_rho_up, grad_rho_up);
301  grad_rho_up_grad_rho_dn = dot(grad_rho_up, grad_rho_dn);
302  grad_rho_dn_grad_rho_dn = dot(grad_rho_dn, grad_rho_dn);
303 
304  if (ctx_.cfg().control().print_hash()) {
305  auto h3 = grad_rho_up_grad_rho_up.hash_f_rg();
306  auto h4 = grad_rho_up_grad_rho_dn.hash_f_rg();
307  auto h5 = grad_rho_dn_grad_rho_dn.hash_f_rg();
308 
309  if (ctx_.comm().rank() == 0) {
310  utils::print_hash("grad_rho_up_grad_rho_up", h3);
311  utils::print_hash("grad_rho_up_grad_rho_dn", h4);
312  utils::print_hash("grad_rho_dn_grad_rho_dn", h5);
313  }
314  }
315  }
316 
317  /* vsigma_uu: dϵ/dσ↑↑ */
318  Smooth_periodic_function<double> vsigma_uu;
319  /* vsigma_ud: dϵ/dσ↑↓ */
320  Smooth_periodic_function<double> vsigma_ud;
321  /* vsigma_dd: dϵ/dσ↓↓ */
322  Smooth_periodic_function<double> vsigma_dd;
323 
324  if (is_gga) {
325  vsigma_uu = Smooth_periodic_function<double>(ctx_.spfft(), ctx_.gvec_partition());
326  vsigma_ud = Smooth_periodic_function<double>(ctx_.spfft(), ctx_.gvec_partition());
327  vsigma_dd = Smooth_periodic_function<double>(ctx_.spfft(), ctx_.gvec_partition());
328  for (int i = 0; i < 3; i++) {
329  vsigma_[i]->zero();
330  }
331  }
332 
333  sddk::mdarray<double, 1> exc(num_points, memory_t::host, "exc_tmp");
334  sddk::mdarray<double, 1> vxc_up(num_points, memory_t::host, "vxc_up_tmp");
335  sddk::mdarray<double, 1> vxc_dn(num_points, memory_t::host, "vxc_dn_dmp");
336 
337  /* loop over XC functionals */
338  for (auto& ixc: xc_func_) {
339  PROFILE_START("sirius::Potential::xc_rg_magnetic|libxc");
340  if (ixc->is_vdw()) {
341 #if defined(SIRIUS_USE_VDWXC)
342  /* all ranks should make a call because VdW uses FFT internaly */
343  if (num_points) {
344  ixc->get_vdw(&rho_up.f_rg(0), &rho_dn.f_rg(0), &grad_rho_up_grad_rho_up.f_rg(0),
345  &grad_rho_dn_grad_rho_dn.f_rg(0), vxc_up.at(memory_t::host), vxc_dn.at(memory_t::host),
346  &vsigma_uu.f_rg(0), &vsigma_dd.f_rg(0), exc.at(memory_t::host));
347  } else {
348  ixc->get_vdw(nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr);
349  }
350 #else
351  TERMINATE("You should not be there since sirius is not compiled with libVDWXC\n");
352 #endif
353  } else {
354  if (num_points) {
355  #pragma omp parallel
356  {
357  /* split local size between threads */
358  splindex<splindex_t::block> spl_t(num_points, omp_get_num_threads(), omp_get_thread_num());
359  /* if this is an LDA functional */
360  if (ixc->is_lda()) {
361  ixc->get_lda(spl_t.local_size(), &rho_up.f_rg(spl_t.global_offset()),
362  &rho_dn.f_rg(spl_t.global_offset()),
363  vxc_up.at(memory_t::host, spl_t.global_offset()),
364  vxc_dn.at(memory_t::host, spl_t.global_offset()),
365  exc.at(memory_t::host, spl_t.global_offset()));
366  }
367  /* if this is a GGA functional */
368  if (ixc->is_gga()) {
369  ixc->get_gga(spl_t.local_size(), &rho_up.f_rg(spl_t.global_offset()),
370  &rho_dn.f_rg(spl_t.global_offset()),
371  &grad_rho_up_grad_rho_up.f_rg(spl_t.global_offset()),
372  &grad_rho_up_grad_rho_dn.f_rg(spl_t.global_offset()),
373  &grad_rho_dn_grad_rho_dn.f_rg(spl_t.global_offset()),
374  vxc_up.at(memory_t::host, spl_t.global_offset()),
375  vxc_dn.at(memory_t::host, spl_t.global_offset()),
376  &vsigma_uu.f_rg(spl_t.global_offset()),
377  &vsigma_ud.f_rg(spl_t.global_offset()),
378  &vsigma_dd.f_rg(spl_t.global_offset()),
379  exc.at(memory_t::host, spl_t.global_offset()));
380  }
381  } // omp parallel region
382  } // num_points != 0
383  }
384  PROFILE_STOP("sirius::Potential::xc_rg_magnetic|libxc");
385  if (ixc->is_gga()) {
386  #pragma omp parallel for
387  for (int ir = 0; ir < num_points; ir++) {
388  /* save for future reuse in XC stress calculation */
389  vsigma_[0]->f_rg(ir) += vsigma_uu.f_rg(ir);
390  vsigma_[1]->f_rg(ir) += vsigma_ud.f_rg(ir);
391  vsigma_[2]->f_rg(ir) += vsigma_dd.f_rg(ir);
392  }
393 
394  Smooth_periodic_vector_function<double> up_gradrho_vsigma(ctx_.spfft(), ctx_.gvec_partition());
395  Smooth_periodic_vector_function<double> dn_gradrho_vsigma(ctx_.spfft(), ctx_.gvec_partition());
396  for (int x: {0, 1, 2}) {
397  for(int ir = 0; ir < num_points; ir++) {
398  up_gradrho_vsigma[x].f_rg(ir) = 2 * grad_rho_up[x].f_rg(ir) * vsigma_uu.f_rg(ir) + grad_rho_dn[x].f_rg(ir) * vsigma_ud.f_rg(ir);
399  dn_gradrho_vsigma[x].f_rg(ir) = 2 * grad_rho_dn[x].f_rg(ir) * vsigma_dd.f_rg(ir) + grad_rho_up[x].f_rg(ir) * vsigma_ud.f_rg(ir);
400  }
401  /* transform to plane wave domain */
402  up_gradrho_vsigma[x].fft_transform(-1);
403  dn_gradrho_vsigma[x].fft_transform(-1);
404  }
405 
406  auto div_up_gradrho_vsigma = divergence(up_gradrho_vsigma);
407  div_up_gradrho_vsigma.fft_transform(1);
408  auto div_dn_gradrho_vsigma = divergence(dn_gradrho_vsigma);
409  div_dn_gradrho_vsigma.fft_transform(1);
410 
411  /* add remaining term to Vxc */
412  #pragma omp parallel for
413  for (int ir = 0; ir < num_points; ir++) {
414  vxc_up(ir) -= div_up_gradrho_vsigma.f_rg(ir);
415  vxc_dn(ir) -= div_dn_gradrho_vsigma.f_rg(ir);
416  }
417  }
418 
419  #pragma omp parallel for
420  for (int irloc = 0; irloc < num_points; irloc++) {
421  /* add XC energy density */
422  xc_energy_density_->f_rg(irloc) += exc(irloc);
423  /* add XC potential */
424  xc_potential_->f_rg(irloc) += 0.5 * (vxc_up(irloc) + vxc_dn(irloc));
425 
426  double bxc = 0.5 * (vxc_up(irloc) - vxc_dn(irloc));
427 
428  /* get the sign between mag and B */
429  auto s = utils::sign((rho_up.f_rg(irloc) - rho_dn.f_rg(irloc)) * bxc);
430 
431  vector3d<double> m;
432  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
433  m[j] = density__.magnetization(j).f_rg(irloc);
434  }
435  auto m_len = m.length();
436 
437  if (m_len > 1e-8) {
438  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
439  effective_magnetic_field(j).f_rg(irloc) += std::abs(bxc) * s * m[j] / m_len;
440  }
441  }
442  }
443  } // for loop over XC functionals
444 }

446 template <bool add_pseudo_core__>
447 void Potential::xc(Density const& density__)
448 {
449  PROFILE("sirius::Potential::xc");
450 
451  /* zero all fields */
452  xc_potential_->zero();
453  xc_energy_density_->zero();
454  for (int i = 0; i < ctx_.num_mag_dims(); i++) {
455  effective_magnetic_field(i).zero();
456  }
457  /* quick return */
458  if (xc_func_.size() == 0) {
459  return;
460  }
461 
462  if (ctx_.full_potential()) {
463  xc_mt(density__);
464  }
465 
466  if (ctx_.num_spins() == 1) {
467  xc_rg_nonmagnetic<add_pseudo_core__>(density__);
468  } else {
469  xc_rg_magnetic<add_pseudo_core__>(density__);
470  }
471 
472  if (ctx_.cfg().control().print_hash()) {
473  auto h = xc_energy_density_->hash_f_rg();
474  if (ctx_.comm().rank() == 0) {
475  utils::print_hash("Exc", h);
476  }
477  }
