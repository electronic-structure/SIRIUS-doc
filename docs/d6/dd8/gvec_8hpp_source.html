48 /// Descriptor of the z-column (x,y fixed, z varying) of the G-vectors.

51 {
52  /// X-coordinate (can be negative and positive).
53  int x;
54  /// Y-coordinate (can be negative and positive).

60  : x(x__)
61  , y(y__)
62  , z(z__)
63  {
64  }
65  /// Default constructor.
66  z_column_descriptor()
67  {
68  }
69 };

410  /// Return G vector in Cartesian coordinates.
411  template <index_domain_t idx_t>
412  inline vector3d<double> gvec_cart(int ig__) const
413  {
414  switch (idx_t) {
415  case index_domain_t::local: {
416  return vector3d<double>(gvec_cart_(0, ig__), gvec_cart_(1, ig__), gvec_cart_(2, ig__));
417  }
418  case index_domain_t::global: {
419  auto G = gvec_by_full_index(gvec_full_index_(ig__));
420  return lattice_vectors_ * vector3d<double>(G[0], G[1], G[2]);
421  }
422  }
423  }
424 
425  /// Return G+k vector in Cartesian coordinates.
426  template <index_domain_t idx_t>
427  inline vector3d<double> gkvec_cart(int ig__) const
428  {
429  switch (idx_t) {
430  case index_domain_t::local: {
431  return vector3d<double>(gkvec_cart_(0, ig__), gkvec_cart_(1, ig__), gkvec_cart_(2, ig__));
432  }
433  case index_domain_t::global: {
434  auto G = gvec_by_full_index(gvec_full_index_(ig__));
435  return lattice_vectors_ * (vector3d<double>(G[0], G[1], G[2]) + vk_);
436  }
437  }
