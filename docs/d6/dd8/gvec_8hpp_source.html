185  /// Return corresponding G-vector for an index in the range [0, num_gvec).
186  inline vector3d<int> gvec_by_full_index(uint32_t idx__) const
187  {
188  /* index of the z coordinate of G-vector: first 12 bits */
189  uint32_t j = idx__ & 0xFFF;
190  /* index of z-column: last 20 bits */
191  uint32_t i = idx__ >> 12;
192  assert(i < (uint32_t)z_columns_.size());
193  assert(j < (uint32_t)z_columns_[i].z.size());
194  int x = z_columns_[i].x;
195  int y = z_columns_[i].y;
196  int z = z_columns_[i].z[j];
197  return vector3d<int>(x, y, z);
198  }

200  /// Find z-columns of G-vectors inside a sphere with Gmax radius.
201  /** This function also computes the total number of G-vectors. */
202  inline void find_z_columns(double Gmax__, FFT3D_grid const& fft_box__)
203  {
204  mdarray<int, 2> non_zero_columns(fft_box__.limits(0), fft_box__.limits(1));
205  non_zero_columns.zero();
206 
207  num_gvec_ = 0;
208 
209  auto add_new_column = [&](int i, int j)
210  {
211  std::vector<int> zcol;
212 
213  /* in general case take z in [0, Nz) */
214  int zmax = fft_box__.size(2) - 1;
215  /* in case of G-vector reduction take z in [0, Nz/2] for {x=0,y=0} stick */
216  if (reduce_gvec_ && !i && !j) {
217  zmax = fft_box__.limits(2).second;
218  }
219  /* loop over z-coordinates of FFT grid */
220  for (int iz = 0; iz <= zmax; iz++) {
221  /* get z-coordinate of G-vector */
222  int k = fft_box__.freq_by_coord<2>(iz);
223  /* take G+k */
224  auto vgk = lattice_vectors_ * (vector3d<double>(i, j, k) + vk_);
225  /* add z-coordinate of G-vector to the list */
226  if (vgk.length() <= Gmax__) {
227  zcol.push_back(k);
228  }
229  }
230 
231  /* add column to the list */
232  if (zcol.size() && !non_zero_columns(i, j)) {
233  z_columns_.push_back(z_column_descriptor(i, j, zcol));
234  num_gvec_ += static_cast<int>(zcol.size());
235 
236  non_zero_columns(i, j) = 1;
237  if (reduce_gvec_) {
238  int mi = -i;
239  int mj = -j;
240  if (mi >= fft_box__.limits(0).first && mi <= fft_box__.limits(0).second &&
241  mj >= fft_box__.limits(1).first && mj <= fft_box__.limits(1).second) {
242  non_zero_columns(mi, mj) = 1;
243  }
244  }
245  }
246  };
247 
248  /* copy column order from previous G-vector set */
249  if (gvec_base_) {
250  for (int icol = 0; icol < gvec_base_->num_zcol(); icol++) {
251  int i = gvec_base_->zcol(icol).x;
252  int j = gvec_base_->zcol(icol).y;
253  add_new_column(i, j);
254  }
255  }
256 
257  for (int i = fft_box__.limits(0).first; i <= fft_box__.limits(0).second; i++) {
258  for (int j = fft_box__.limits(1).first; j <= fft_box__.limits(1).second; j++) {
259  add_new_column(i, j);
260  }
261  }
262 
263  if (!gvec_base_) {
264  /* put column with {x, y} = {0, 0} to the beginning */
265  for (size_t i = 0; i < z_columns_.size(); i++) {
266  if (z_columns_[i].x == 0 && z_columns_[i].y == 0) {
267  std::swap(z_columns_[i], z_columns_[0]);
268  break;
269  }
270  }
271  }
272 
273  /* sort z-columns starting from the second or skip num_zcol of base distribution */
274  int n = (gvec_base_) ? gvec_base_->num_zcol() : 1;
275  std::sort(z_columns_.begin() + n, z_columns_.end(),
276  [](z_column_descriptor const& a, z_column_descriptor const& b) { return a.z.size() > b.z.size(); });
277  }

279  /// Distribute z-columns between MPI ranks.
280  inline void distribute_z_columns()
281  {
282  gvec_distr_ = block_data_descriptor(comm().size());
283  zcol_distr_ = block_data_descriptor(comm().size());
284  /* local number of z-columns for each rank */
285  std::vector<std::vector<z_column_descriptor>> zcols_local(comm().size());
286 
287  /* use already existing distribution of base G-vector set */
288  if (gvec_base_) {
289  for (int rank = 0; rank < comm().size(); rank++) {
290  for (int i = 0; i < gvec_base_->zcol_count(rank); i++) {
291  int icol = gvec_base_->zcol_offset(rank) + i;
292  /* assign column to the found rank */
293  zcols_local[rank].push_back(z_columns_[icol]);
294  /* count local number of z-columns */
295  zcol_distr_.counts[rank] += 1;
296  /* count local number of G-vectors */
297  gvec_distr_.counts[rank] += static_cast<int>(z_columns_[icol].z.size());
298  }
299  }
300  }
301 
302  int n = (gvec_base_) ? gvec_base_->num_zcol() : 0;
303 
304  std::vector<int> ranks;
305  for (int i = n; i < static_cast<int>(z_columns_.size()); i++) {
306  /* initialize the list of ranks to 0,1,2,... */
307  if (ranks.empty()) {
308  ranks.resize(comm().size());
309  std::iota(ranks.begin(), ranks.end(), 0);
310  }
311  /* find rank with minimum number of G-vectors */
312  auto rank_with_min_gvec = std::min_element(ranks.begin(), ranks.end(), [this](const int& a, const int& b) {
313  return gvec_distr_.counts[a] < gvec_distr_.counts[b];
314  });
315 
316  /* assign column to the found rank */
317  zcols_local[*rank_with_min_gvec].push_back(z_columns_[i]);
318  /* count local number of z-columns */
319  zcol_distr_.counts[*rank_with_min_gvec] += 1;
320  /* count local number of G-vectors */
321  gvec_distr_.counts[*rank_with_min_gvec] += static_cast<int>(z_columns_[i].z.size());
322  /* exclude this rank from the search */
323  ranks.erase(rank_with_min_gvec);
324  }
325  gvec_distr_.calc_offsets();
326  zcol_distr_.calc_offsets();
327 
328  /* save new ordering of z-columns */
329  z_columns_.clear();
330  for (int rank = 0; rank < comm().size(); rank++) {
331  z_columns_.insert(z_columns_.end(), zcols_local[rank].begin(), zcols_local[rank].end());
332  }
333 
334  /* sanity check */
335  int ng{0};
336  for (int rank = 0; rank < comm().size(); rank++) {
337  ng += gvec_distr_.counts[rank];
338  }
339  if (ng != num_gvec_) {
340  TERMINATE("wrong number of G-vectors");
341  }
342  }

344  /// Find a list of G-vector shells.
345  /** G or G+k vectors belonging to the same shell have the same length */
346  inline void find_gvec_shells()
347  {
348  PROFILE("sddk::Gvec::find_gvec_shells");
349 
350  /* list of pairs (length, index of G-vector) */
351  std::vector<std::pair<uint64_t, int>> tmp(num_gvec_);
352  #pragma omp parallel for schedule(static)
353  for (int ig = 0; ig < num_gvec(); ig++) {
354  /* take G+k */
355  auto gk = gkvec_cart<index_domain_t::global>(ig);
356  /* make some reasonable roundoff */
357  uint64_t len = static_cast<uint64_t>(gk.length() * 1e8);
358 
359  tmp[ig] = std::make_pair(len, ig);
360  }
361  /* sort by first element in pair (length) */
362  std::sort(tmp.begin(), tmp.end());
363 
364  gvec_shell_ = mdarray<int, 1>(num_gvec_);
365  /* index of the first shell */
366  gvec_shell_(tmp[0].second) = 0;
367  num_gvec_shells_ = 1;
368  /* temporary vector to store G-shell radius */
369  std::vector<double> tmp_len;
370  /* radius of the first shell */
371  tmp_len.push_back(static_cast<double>(tmp[0].first) * 1e-8);
372  for (int ig = 1; ig < num_gvec_; ig++) {
373  /* if this G+k-vector has a different length */
374  if (tmp[ig].first != tmp[ig - 1].first) {
375  /* increment number of shells */
376  num_gvec_shells_++;
377  /* save the radius of the new shell */
378  tmp_len.push_back(static_cast<double>(tmp[ig].first) * 1e-8);
379  }
380  /* assign the index of the current shell */
381  gvec_shell_(tmp[ig].second) = num_gvec_shells_ - 1;
382  }
383  gvec_shell_len_ = mdarray<double, 1>(num_gvec_shells_);
384  std::copy(tmp_len.begin(), tmp_len.end(), gvec_shell_len_.at(memory_t::host));
385  }

387  /// Compute the Cartesian coordinates.
388  void init_gvec_cart()
389  {
390  gvec_cart_ = mdarray<double, 2>(3, count());
391  gkvec_cart_ = mdarray<double, 2>(3, count());
392 
393  for (int igloc = 0; igloc < count(); igloc++) {
394  int ig = offset() + igloc;
395  auto G = gvec_by_full_index(gvec_full_index_(ig));
396  auto gc = lattice_vectors_ * vector3d<double>(G[0], G[1], G[2]);
397  auto gkc = lattice_vectors_ * (vector3d<double>(G[0], G[1], G[2]) + vk_);
398  for (int x: {0, 1, 2}) {
399  gvec_cart_(x, igloc) = gc[x];
400  gkvec_cart_(x, igloc) = gkc[x];
401  }
402  }
403  }

410  /// Initialize everything.
411  void init(FFT3D_grid const& fft_grid)
412  {
413  PROFILE("sddk::Gvec::init");
414 
415  find_z_columns(Gmax_, fft_grid);
416 
417  distribute_z_columns();
418 
419  gvec_index_by_xy_ = mdarray<int, 3>(2, fft_grid.limits(0), fft_grid.limits(1), memory_t::host, "Gvec.gvec_index_by_xy_");
420  std::fill(gvec_index_by_xy_.at(memory_t::host), gvec_index_by_xy_.at(memory_t::host) + gvec_index_by_xy_.size(), -1);
421 
422  /* build the full G-vector index and reverse mapping */
423  gvec_full_index_ = mdarray<uint32_t, 1>(num_gvec_);
424  int ig{0};
425  for (size_t i = 0; i < z_columns_.size(); i++) {
426  /* starting G-vector index for a z-stick */
427  gvec_index_by_xy_(0, z_columns_[i].x, z_columns_[i].y) = ig;
428  /* pack size of a z-stick and column index in one number */
429  gvec_index_by_xy_(1, z_columns_[i].x, z_columns_[i].y) =
430  static_cast<int>((z_columns_[i].z.size() << 20) + i);
431  for (size_t j = 0; j < z_columns_[i].z.size(); j++) {
432  gvec_full_index_[ig++] = static_cast<uint32_t>((i << 12) + j);
433  }
434  }
435  if (ig != num_gvec_) {
436  TERMINATE("wrong G-vector count");
437  }
438  for (int ig = 0; ig < num_gvec_; ig++) {
439  auto gv = gvec(ig);
440  if (index_by_gvec(gv) != ig) {
441  std::stringstream s;
442  s << "wrong G-vector index: ig=" << ig << " gv=" << gv << " index_by_gvec(gv)=" << index_by_gvec(gv);
443  TERMINATE(s);
444  }
445  }
446 
447  /* first G-vector must be (0, 0, 0); never reomove this check!!! */
448  auto g0 = gvec_by_full_index(gvec_full_index_(0));
449  if (g0[0] || g0[1] || g0[2]) {
450  TERMINATE("first G-vector is not zero");
451  }
452 
453  init_gvec_cart();
454 
455  find_gvec_shells();
456 
457  if (gvec_base_) {
458  /* the size of the mapping is equal to the local number of G-vectors in the base set */
459  gvec_base_mapping_ = mdarray<int, 1>(gvec_base_->count());
460  /* loop over local G-vectors of a base set */
461  for (int igloc = 0; igloc < gvec_base_->count(); igloc++) {
462  /* G-vector in lattice coordinates */
463  auto G = gvec_base_->gvec(gvec_base_->offset() + igloc);
464  /* global index of G-vector in the current set */
465  int ig = index_by_gvec(G);
466  /* the same MPI rank must store this G-vector */
467  ig -= offset();
468  if (ig >= 0 && ig < count()) {
469  gvec_base_mapping_(igloc) = ig;
470  } else {
471  std::stringstream s;
472  s << "local G-vector index is not found" << std::endl
473  << " G-vector: " << G << std::endl
474  << " G-vector index in base distribution : " << gvec_base_->offset() + igloc << std::endl
475  << " G-vector index in base distribution (by G-vector): " << gvec_base_->index_by_gvec(G) << std::endl
476  << " G-vector index in new distribution : " << index_by_gvec(G) << std::endl
477  << " offset in G-vector index for this rank: " << offset() << std::endl
478  << " local number of G-vectors for this rank: " << count();
479  TERMINATE(s);
480  }
481  }
482  }
483  // TODO: add a check for gvec_base (there is already a test for this).
484  }

770  /// Return a global G-vector index in the range [0, num_gvec) by the G-vector.
771  /** The information about a G-vector index is encoded by two numbers: a starting index for the
772  * column of G-vectors and column's size. Depending on the geometry of the reciprocal lattice,
773  * z-columns may have only negative, only positive or both negative and positive frequencies for
774  * a given x and y. This information is used to compute the offset which is added to the starting index
775  * in order to get a full G-vector index. */
776  inline int index_by_gvec(vector3d<int> const& G__) const
777  {
778  /* reduced G-vector set does not have negative z for x=y=0 */
779  if (reduced() && G__[0] == 0 && G__[1] == 0 && G__[2] < 0) {
780  return -1;
781  }
782  int ig0 = gvec_index_by_xy_(0, G__[0], G__[1]);
783  if (ig0 == -1) {
784  return -1;
785  }
786  /* index of the column */
787  int icol = gvec_index_by_xy_(1, G__[0], G__[1]) & 0xFFFFF;
788  /* size of the column */
789  int col_size = gvec_index_by_xy_(1, G__[0], G__[1]) >> 20;
790 
791  /* three possible options for the z-column location
792 
793  frequency ... -4, -3, -2, -1, 0, 1, 2, 3, 4 ...
794  -----------------------------------------------------------------------------
795  G-vec ordering
796  #1 (all negative) ___ 0 1 2 3 __________________
797  #2 (negative and positive) ____________ 3 4 0 1 2 _________
798  #3 (all positive) _____________________ 0 1 2 3 ___
799 
800  Remember how FFT frequencies are stored: firs positive frequences, then negative in the reverse order
801 
802  subtract first z-coordinate in column from the current z-coordinate of G-vector: in case #1 or #3 this
803  already gives a proper offset, in case #2 storage of FFT frequencies must be taken into account
804  */
805  int z0 = G__[2] - z_columns_[icol].z[0];
806  /* calculate proper offset */
807  int offs = (z0 >= 0) ? z0 : z0 + col_size;
808  /* full index */
809  int ig = ig0 + offs;
810  assert(ig >= 0 && ig < num_gvec());
811  return ig;
812  }

946 /// Stores information about G-vector partitioning between MPI ranks for the FFT transformation.
947 /** FFT driver works with a small communicator. G-vectors are distributed over the entire communicator which is
948  larger than the FFT communicator. In order to transform the functions, G-vectors must be redistributed to the
949  FFT-friendly "fat" slabs based on the FFT communicator size. */
950 class Gvec_partition
951 {
952  private:
953  /// Pointer to the G-vector instance.
954  Gvec const& gvec_;
955 
956  /// Communicator for the FFT.
957  Communicator const& fft_comm_;
958 
959  /// Communicator which is orthogonal to FFT communicator.
960  Communicator const& comm_ortho_fft_;
961 
962  /// Distribution of G-vectors for FFT.
963  block_data_descriptor gvec_distr_fft_;
964 
965  /// Distribution of z-columns for FFT.
966  block_data_descriptor zcol_distr_fft_;
967 
968  /// Distribution of G-vectors inside FFT-friendly "fat" slab.
969  block_data_descriptor gvec_fft_slab_;
970 
971  /// Offset of the z-column in the local data buffer.
972  /** Global index of z-column is expected */
973  mdarray<int, 1> zcol_offs_;
974 
975  /// Mapping of MPI ranks used to split G-vectors to a 2D grid.
976  mdarray<int, 2> rank_map_;
977 
978  /// Global index of z-column in new (fat-slab) distrubution.
979  /** This is a mapping between new and original ordering of z-columns. */
980  mdarray<int, 1> idx_zcol_;

985  inline void build_fft_distr()
986  {
987  /* calculate distribution of G-vectors and z-columns for the FFT communicator */
988  gvec_distr_fft_ = block_data_descriptor(fft_comm().size());
989  zcol_distr_fft_ = block_data_descriptor(fft_comm().size());
990 
991  for (int rank = 0; rank < fft_comm().size(); rank++) {
992  for (int i = 0; i < comm_ortho_fft().size(); i++) {
993  /* fine-grained rank */
994  int r = rank_map_(rank, i);
995  gvec_distr_fft_.counts[rank] += gvec().gvec_count(r);
996  zcol_distr_fft_.counts[rank] += gvec().zcol_count(r);
997  }
998  }
999  /* get offsets of z-columns */
1000  zcol_distr_fft_.calc_offsets();
1001  /* get offsets of G-vectors */
1002  gvec_distr_fft_.calc_offsets();
1003  }

1005  /// Calculate offsets of z-columns inside each local buffer of PW coefficients.
1006  inline void calc_offsets()
1007  {
1008  zcol_offs_ = mdarray<int, 1>(gvec().num_zcol(), memory_t::host, "Gvec_partition.zcol_offs_");
1009  for (int rank = 0; rank < fft_comm().size(); rank++) {
1010  int offs{0};
1011  /* loop over local number of z-columns */
1012  for (int i = 0; i < zcol_count_fft(rank); i++) {
1013  /* global index of z-column */
1014  int icol = idx_zcol_[zcol_distr_fft_.offsets[rank] + i];
1015  zcol_offs_[icol] = offs;
1016  offs += static_cast<int>(gvec().zcol(icol).z.size());
1017  }
1018  assert(offs == gvec_distr_fft_.counts[rank]);
1019  }
1020  }

1022  /// Stack together the G-vector slabs to make a larger ("fat") slab for a FFT driver.
1023  inline void pile_gvec()
1024  {
1025  /* build a table of {offset, count} values for G-vectors in the swapped distribution;
1026  * we are preparing to swap plane-wave coefficients from a default slab distribution to a FFT-friendly
1027  * distribution
1028  * +--------------+ +----+----+----+
1029  * | : : | | | | |
1030  * +--------------+ |....|....|....|
1031  * | : : | -> | | | |
1032  * +--------------+ |....|....|....|
1033  * | : : | | | | |
1034  * +--------------+ +----+----+----+
1035  *
1036  * i.e. we will make G-vector slabs more fat (pile-of-slabs) and at the same time reshulffle wave-functions
1037  * between columns of the 2D MPI grid */
1038  gvec_fft_slab_ = block_data_descriptor(comm_ortho_fft_.size());
1039  for (int i = 0; i < comm_ortho_fft_.size(); i++) {
1040  gvec_fft_slab_.counts[i] = gvec().gvec_count(rank_map_(fft_comm_.rank(), i));
1041  }
1042  gvec_fft_slab_.calc_offsets();
1043 
1044  assert(gvec_fft_slab_.offsets.back() + gvec_fft_slab_.counts.back() == gvec_distr_fft_.counts[fft_comm().rank()]);
1045  }

1058  << " gvec_.comm().size() = " << gvec_.comm().size();
1059  TERMINATE(s);
1060  }
1061  rank_map_ = mdarray<int, 2>(fft_comm_.size(), comm_ortho_fft_.size());
1062  rank_map_.zero();
1063  /* get a global rank */
1064  rank_map_(fft_comm_.rank(), comm_ortho_fft_.rank()) = gvec_.comm().rank();
1065  gvec_.comm().allreduce(&rank_map_(0, 0), gvec_.comm().size());
1066 
1067  build_fft_distr();
1068 
1069  idx_zcol_ = mdarray<int, 1>(gvec().num_zcol());
1070  int icol{0};
1071  for (int rank = 0; rank < fft_comm().size(); rank++) {
1072  for (int i = 0; i < comm_ortho_fft().size(); i++) {
1073  for (int k = 0; k < gvec_.zcol_count(rank_map_(rank, i)); k++) {
1074  idx_zcol_(icol) = gvec_.zcol_offset(rank_map_(rank, i)) + k;
1075  icol++;
1076  }
1077  }
1078  assert(icol == zcol_distr_fft_.counts[rank] + zcol_distr_fft_.offsets[rank]);
1079  }
1080  assert(icol == gvec().num_zcol());

1083  int ig{0};
1084  for (int i = 0; i < comm_ortho_fft_.size(); i++) {
1085  for (int k = 0; k < gvec_.gvec_count(rank_map_(fft_comm_.rank(), i)); k++) {
1086  idx_gvec_(ig) = gvec_.gvec_offset(rank_map_(fft_comm_.rank(), i)) + k;
1087  ig++;
1088  }
1089  }
1090 
1091  calc_offsets();
1092  pile_gvec();
1093  }
