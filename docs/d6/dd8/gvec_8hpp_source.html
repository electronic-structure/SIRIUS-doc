49/// Descriptor of the z-column (x,y fixed, z varying) of the G-vectors.
50/** Sphere of G-vectors within a given plane-wave cutoff is represented as a set of z-columns with different lengths. */
51struct z_column_descriptor
52{
53 /// X-coordinate (can be negative and positive).
54 int x;
55 /// Y-coordinate (can be negative and positive).
56 int y;
57 /// List of the Z-coordinates of the column.
58 std::vector<int> z;
59 /// Constructor.
60 z_column_descriptor(int x__, int y__, std::vector<int> z__)
61 : x(x__)
62 , y(y__)
63 , z(z__)
64 {
65 }
66 /// Default constructor.
67 z_column_descriptor()
68 {
69 }
70};
71
72/// Serialize a single z-column descriptor.
73inline void serialize(serializer& s__, z_column_descriptor const& zcol__)
74{
75 serialize(s__, zcol__.x);
76 serialize(s__, zcol__.y);
77 serialize(s__, zcol__.z);
78}
79
80/// Deserialize a single z-column descriptor.
81inline void deserialize(serializer& s__, z_column_descriptor& zcol__)
82{
83 deserialize(s__, zcol__.x);
84 deserialize(s__, zcol__.y);
85 deserialize(s__, zcol__.z);
86}
87
88/// Serialize a vector of z-column descriptors.
89inline void serialize(serializer& s__, std::vector<z_column_descriptor> const& zcol__)
90{
91 serialize(s__, zcol__.size());
92 for (auto& e: zcol__) {
93 serialize(s__, e);
94 }
95}

97/// Deserialize a vector of z-column descriptors.
98inline void deserialize(serializer& s__, std::vector<z_column_descriptor>& zcol__)
99{
100 size_t sz;
101 deserialize(s__, sz);
102 zcol__.resize(sz);
103 for (size_t i = 0; i < sz; i++) {
104 deserialize(s__, zcol__[i]);
105 }
106}

114/// A set of G-vectors for FFTs and G+k basis functions.
115/** Current implemntation supports up to 2^12 (4096) z-dimension of the FFT grid and 2^20 (1048576) number of
116 * z-columns. The order of z-sticks and G-vectors is not fixed and depends on the number of MPI ranks used
117 * for the parallelization. */
118class Gvec
119{
120 private:
121 /// k-vector of G+k.

123
124 /// Cutoff for |G+k| vectors.

141
142 /// Mapping between G-vector index [0:num_gvec_) and a full index.
143 /** Full index is used to store x,y,z coordinates in a packed form in a single integer number.
144 * The index is equal to ((i << 12) + j) where i is the global index of z_column and j is the
145 * index of G-vector z-coordinate in the column i. This is a global array: each MPI rank stores exactly the
146 * same copy of the gvec_full_index_.

148 * Limitations: size of z-dimension of FFT grid: 4096, number of z-columns: 1048576

179 block_data_descriptor gvec_distr_;
180
181 /// Fine-grained distribution of z-columns.

183
184 /// Set of G-vectors on which the current G-vector distribution can be based.

186 * without MPI communication. */

198 \endcode

204

699 int rank = gvec().comm().rank();
700 /* collect scattered PW coefficients */
701 comm_ortho_fft().allgather(f_pw_local__, gvec().gvec_count(rank), f_pw_fft__, gvec_fft_slab().counts.data(),
702 gvec_fft_slab().offsets.data());
703
704 }

798 template <typename T>
799 std::vector<T> remap_forward(T* data__) const
800 {
801 PROFILE("sddk::Gvec_shells::remap_forward");
802
803 std::vector<T> send_buf(gvec_.count());
804 std::vector<int> counts(comm_.size(), 0);
805 for (int igloc = 0; igloc < gvec_.count(); igloc++) {
806 int ig = gvec_.offset() + igloc;
807 int igsh = gvec_.shell(ig);
808 int r = spl_num_gsh_.local_rank(igsh);
809 send_buf[a2a_send_.offsets[r] + counts[r]] = data__[igloc];
810 counts[r]++;
811 }
812
813 std::vector<T> recv_buf(gvec_count_remapped());
814
815 comm_.alltoall(send_buf.data(), a2a_send_.counts.data(), a2a_send_.offsets.data(), recv_buf.data(),
816 a2a_recv_.counts.data(), a2a_recv_.offsets.data());
817
818 return recv_buf;
819 }

827
828 comm_.alltoall(buf__.data(), a2a_recv_.counts.data(), a2a_recv_.offsets.data(), recv_buf.data(),
829 a2a_send_.counts.data(), a2a_send_.offsets.data());
830
831 std::vector<int> counts(comm_.size(), 0);
832 for (int igloc = 0; igloc < gvec_.count(); igloc++) {
833 int ig = gvec_.offset() + igloc;
834 int igsh = gvec_.shell(ig);
835 int r = spl_num_gsh_.local_rank(igsh);
836 data__[igloc] = recv_buf[a2a_send_.offsets[r] + counts[r]];
837 counts[r]++;
838 }
839 }
