49 /// Descriptor of the z-column (x,y fixed, z varying) of the G-vectors.
50 /** Sphere of G-vectors within a given plane-wave cutoff is represented as a set of z-columns with different lengths. */
51 struct z_column_descriptor
52 {
53  /// X-coordinate (can be negative and positive).
54  int x;
55  /// Y-coordinate (can be negative and positive).
56  int y;
57  /// List of the Z-coordinates of the column.
58  std::vector<int> z;
59  /// Constructor.
60  z_column_descriptor(int x__, int y__, std::vector<int> z__)
61  : x(x__)
62  , y(y__)
63  , z(z__)
64  {
65  }
66  /// Default constructor.
67  z_column_descriptor()
68  {
69  }
70 };

88 /// Serialize a vector of z-column descriptors.
89 inline void serialize(serializer& s__, std::vector<z_column_descriptor> const& zcol__)
90 {
91  serialize(s__, zcol__.size());
92  for (auto& e: zcol__) {
93  serialize(s__, e);
94  }

97 /// Deserialize a vector of z-column descriptors.
98 inline void deserialize(serializer& s__, std::vector<z_column_descriptor>& zcol__)
99 {
100  size_t sz;
101  deserialize(s__, sz);
102  zcol__.resize(sz);
103  for (size_t i = 0; i < sz; i++) {
104  deserialize(s__, zcol__[i]);
105  }

108 /// A set of G-vectors for FFTs and G+k basis functions.
109 /** Current implemntation supports up to 2^12 (4096) z-dimension of the FFT grid and 2^20 (1048576) number of
110  * z-columns. The order of z-sticks and G-vectors is not fixed and depends on the number of MPI ranks used
111  * for the parallelization. */
112 class Gvec
113 {

115  /// k-vector of G+k.

135 
136  /// Mapping between G-vector index [0:num_gvec_) and a full index.
137  /** Full index is used to store x,y,z coordinates in a packed form in a single integer number.
138  * The index is equal to ((i << 12) + j) where i is the global index of z_column and j is the
139  * index of G-vector z-coordinate in the column i. This is a global array: each MPI rank stores exactly the
140  * same copy of the gvec_full_index_.

619 /// Stores information about G-vector partitioning between MPI ranks for the FFT transformation.
620 /** FFT driver works with a small communicator. G-vectors are distributed over the entire communicator which is
621  larger than the FFT communicator. In order to transform the functions, G-vectors must be redistributed to the
622  FFT-friendly "fat" slabs based on the FFT communicator size. */
623 class Gvec_partition
624 {
