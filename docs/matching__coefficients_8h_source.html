30 /** The following matching conditions must be fulfilled:
31  * \f[
32  * \frac{\partial^j}{\partial r^j} \sum_{L \nu} A_{L \nu}^{\bf k}({\bf G})u_{\ell \nu}(r)
33  * Y_{L}(\hat {\bf r}) \bigg|_{R^{MT}} = \frac{\partial^j}{\partial r^j} \frac{4 \pi}{\sqrt \Omega}
34  * e^{i{\bf (G+k)\tau}} \sum_{L}i^{\ell} j_{\ell}(|{\bf G+k}|r) Y_{L}^{*}(\widehat {\bf G+k}) Y_{L}(\hat {\bf r}) \bigg|_{R^{MT}}
35  * \f]
36  * where \f$ L = \{ \ell, m \} \f$. Dropping sum over L we arrive to the following system of linear equations:
37  * \f[
38  * \sum_{\nu} \frac{\partial^j u_{\ell \nu}(r)}{\partial r^j} \bigg|_{R^{MT}} A_{L \nu}^{\bf k}({\bf G}) =
39  * \frac{4 \pi}{\sqrt \Omega} e^{i{\bf (G+k)\tau}} i^{\ell} \frac{\partial^j j_{\ell}(|{\bf G+k}|r)}{\partial r^j}
40  * \bigg|_{R^{MT}} Y_{L}^{*}(\widehat {\bf G+k})
41  * \f]
42  * The matching coefficients are then equal to:
43  * \f[
44  * A_{L \nu}^{\bf k}({\bf G}) = \sum_{j} \bigg[ \frac{\partial^j u_{\ell \nu}(r)}{\partial r^j} \bigg|_{R^{MT}} \bigg]_{\nu j}^{-1}
45  * \frac{\partial^j j_{\ell}(|{\bf G+k}|r)}{\partial r^j} \bigg|_{R^{MT}} \frac{4 \pi}{\sqrt \Omega} i^{\ell}
46  * e^{i{\bf (G+k)\tau}} Y_{L}^{*}(\widehat {\bf G+k})
47  * \f]
48  */

69  /// Generate matching coefficients for a specific \f$ \ell \f$ and order.
70  /** \param [in] ngk Number of G+k vectors.
71  * \param [in] ia Index of atom.
72  * \param [in] iat Index of atom type.
73  * \param [in] l Orbital quantum nuber.
74  * \param [in] lm Composite l,m index.
75  * \param [in] nu Order of radial function \f$ u_{\ell \nu}(r) \f$ for which coefficients are generated.
76  * \param [inout] A Matrix of radial derivatives.
77  * \param [out] alm Pointer to alm coefficients.
78  */
79  template <int N>
80  inline void generate(int ngk,
81  std::vector<double_complex> const& phase_factors__,
82  int iat,
83  int l,
84  int lm,
85  int nu,
86  matrix3d<double>& A,
87  double_complex* alm) const
88  {
89  /* invert matrix of radial derivatives */
90  switch (N) {
91  case 1: {
92  #if (__VERIFICATION > 0)
93  if (std::abs(A(0, 0)) < 1.0 / std::sqrt(unit_cell_.omega())) {
94  std::stringstream s;
95  s << "Ill defined plane wave matching problem for atom type " << iat << ", l = " << l << std::endl
96  << " radial function value at the MT boundary : " << A(0, 0);
97  WARNING(s.str());
98  }
99  #endif
100 
101  A(0, 0) = 1.0 / A(0, 0);
102  break;
103  }
104  case 2: {
105  double det = A(0, 0) * A(1, 1) - A(0, 1) * A(1, 0);
106 
107  #if (__VERIFICATION > 0)
108  if (std::abs(det) < 1.0 / std::sqrt(unit_cell_.omega())) {
109  std::stringstream s;
110  s << "Ill defined plane wave matching problem for atom type " << iat << ", l = " << l << std::endl
111  << " radial function value at the MT boundary : " << A(0 ,0);
112  WARNING(s.str());
113  }
114  #endif
115 
116  std::swap(A(0, 0), A(1, 1));
117  A(0, 0) /= det;
118  A(1, 1) /= det;
119  A(0, 1) = -A(0, 1) / det;
120  A(1, 0) = -A(1, 0) / det;
121  break;
122  }
123  case 3: {
124  A = inverse(A);
125  break;
126  }
127  }
128 
129  double_complex zt;
130 
131  for (int igk = 0; igk < ngk; igk++) {
132  switch (N) {
133  case 1: {
134  zt = alm_b_(0, igk, l, iat) * A(0, 0);
135  break;
136  }
137  case 2: {
138  zt = alm_b_(0, igk, l, iat) * A(nu, 0) +
139  alm_b_(1, igk, l, iat) * A(nu, 1);
140  break;
141  }
142  case 3: {
143  zt = alm_b_(0, igk, l, iat) * A(nu, 0) +
144  alm_b_(1, igk, l, iat) * A(nu, 1) +
145  alm_b_(2, igk, l, iat) * A(nu, 2);
146  break;
147  }
148  }
149  alm[igk] = phase_factors__[igk] * std::conj(gkvec_ylm_(igk, lm)) * zt;
150  }
151  }

155  /// Constructor
156  Matching_coefficients(Unit_cell const& unit_cell__,
157  int lmax_apw__,
158  int num_gkvec__,
159  std::vector<int>& igk__,
160  Gvec const& gkvec__)
161  : unit_cell_(unit_cell__),
162  num_gkvec_(num_gkvec__),
163  igk_(igk__),
164  gkvec_(gkvec__)
165  {
166  int lmmax_apw = Utils::lmmax(lmax_apw__);
167 
168  gkvec_ylm_ = mdarray<double_complex, 2>(num_gkvec_, lmmax_apw);
169 
170  gkvec_len_.resize(num_gkvec_);
171 
172  /* get length and Ylm harmonics of G+k vectors */
173  #pragma omp parallel
174  {
175  std::vector<double_complex> ylm(lmmax_apw);
176 
177  #pragma omp for
178  for (int i = 0; i < num_gkvec_; i++) {
179  auto gkvec_cart = gkvec_.gkvec_cart(igk_[i]);
180  /* get r, theta, phi */
181  auto vs = SHT::spherical_coordinates(gkvec_cart);
182 
183  /* get spherical harmonics */
184  SHT::spherical_harmonics(lmax_apw__, vs[1], vs[2], &ylm[0]);
185  gkvec_len_[i] = vs[0];
186 
187  for (int lm = 0; lm < lmmax_apw; lm++) {
188  gkvec_ylm_(i, lm) = ylm[lm];
189  }
190  }
191  }
192 
193  alm_b_ = mdarray<double_complex, 4>(3, num_gkvec_, lmax_apw__ + 1, unit_cell_.num_atom_types());
194  alm_b_.zero();
195 
196  /* value and first two derivatives of spherical Bessel functions */
197  mdarray<double, 2> sbessel_mt(lmax_apw__ + 2, 3);
198 
199  for (int igk = 0; igk < num_gkvec_; igk++) {
200  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
201  double R = unit_cell_.atom_type(iat).mt_radius();
202 
203  double RGk = R * gkvec_len_[igk];
204 
205  /* compute values and first and second derivatives of the spherical Bessel functions at the MT boundary */
206  gsl_sf_bessel_jl_array(lmax_apw__ + 1, RGk, &sbessel_mt(0, 0));
207 
208  /* Bessel function derivative: f_{{n}}^{{\prime}}(z)=-f_{{n+1}}(z)+(n/z)f_{{n}}(z)
209  *
210  * In[]:= FullSimplify[D[SphericalBesselJ[n,a*x],{x,1}]]
211  * Out[]= (n SphericalBesselJ[n,a x])/x-a SphericalBesselJ[1+n,a x]
212  *
213  * In[]:= FullSimplify[D[SphericalBesselJ[n,a*x],{x,2}]]
214  * Out[]= (((-1+n) n-a^2 x^2) SphericalBesselJ[n,a x]+2 a x SphericalBesselJ[1+n,a x])/x^2
215  */
216  for (int l = 0; l <= lmax_apw__; l++) {
217  sbessel_mt(l, 1) = -sbessel_mt(l + 1, 0) * gkvec_len_[igk] + (l / R) * sbessel_mt(l, 0);
218  sbessel_mt(l, 2) = 2 * gkvec_len_[igk] * sbessel_mt(l + 1, 0) / R +
219  ((l - 1) * l - std::pow(RGk, 2)) * sbessel_mt(l, 0) / std::pow(R, 2);
220  }
221 
222  for (int l = 0; l <= lmax_apw__; l++) {
223  double_complex z = std::pow(double_complex(0, 1), l);
224  double f = fourpi / std::sqrt(unit_cell_.omega());
225  alm_b_(0, igk, l, iat) = z * f * sbessel_mt(l, 0);
226  alm_b_(1, igk, l, iat) = z * f * sbessel_mt(l, 1);
227  alm_b_(2, igk, l, iat) = z * f * sbessel_mt(l, 2);
228  }
229  }
230  }
231  }

233  /// Generate plane-wave matching coefficents for the radial solutions of a given atom.
234  /** \param [in] ia Index of atom.
235  * \param [out] alm Array of matching coefficients with dimension indices \f$ ({\bf G+k}, \xi) \f$.
236  */
237  void generate(int ia, mdarray<double_complex, 2>& alm) const
238  {
239  auto& atom = unit_cell_.atom(ia);
240  auto& type = atom.type();
241 
242  assert(type.max_aw_order() <= 3);
243 
244  int iat = type.id();
245 
246  std::vector<double_complex> phase_factors(num_gkvec_);
247  for (int i = 0; i < num_gkvec_; i++) {
248  double phase = twopi * (gkvec_.gkvec(igk_[i]) * unit_cell_.atom(ia).position());
249  phase_factors[i] = std::exp(double_complex(0, phase));
250  }
251 
252  matrix3d<double> A;
253  for (int xi = 0; xi < type.mt_aw_basis_size(); xi++) {
254  int l = type.indexb(xi).l;
255  int lm = type.indexb(xi).lm;
256  int nu = type.indexb(xi).order;
257 
258  /* order of augmentation for a given orbital quantum number */
259  int num_aw = static_cast<int>(type.aw_descriptor(l).size());
260 
261  /* create matrix of radial derivatives */
262  for (int order = 0; order < num_aw; order++) {
263  for (int dm = 0; dm < num_aw; dm++) {
264  A(dm, order) = atom.symmetry_class().aw_surface_dm(l, order, dm);
265  }
266  }
267 
268  switch (num_aw) {
269  /* APW */
270  case 1: {
271  generate<1>(num_gkvec_, phase_factors, iat, l, lm, nu, A, &alm(0, xi));
272  break;
273  }
274  /* LAPW */
275  case 2: {
276  generate<2>(num_gkvec_, phase_factors, iat, l, lm, nu, A, &alm(0, xi));
277  break;
278  }
279  /* Super LAPW */
280  case 3: {
281  generate<3>(num_gkvec_, phase_factors, iat, l, lm, nu, A, &alm(0, xi));
282  break;
283  }
284  default: {
285  TERMINATE("wrong order of augmented wave");
286  }
287  }
288  }
289  }
290 };
