52  inline void calc_stress_kin()
53  {
54  PROFILE("sirius::Stress|kin");
55 
56  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
57  int ik = kset_.spl_num_kpoints(ikloc);
58  auto kp = kset_[ik];
59  if (kp->gkvec().reduced()) {
60  TERMINATE("fix this");
61  }
62 
63  for (int igloc = 0; igloc < kp->num_gkvec_loc(); igloc++) {
64  int ig = kp->idxgk(igloc);
65  auto Gk = kp->gkvec().gkvec_cart(ig);
66 
67  double d{0};
68  for (int i = 0; i < ctx_.num_bands(); i++) {
69  double f = kp->band_occupancy(i);
70  if (f > 1e-12) {
71  auto z = kp->spinor_wave_functions(0).pw_coeffs().prime(igloc, i);
72  d += f * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
73  }
74  }
75  d *= kp->weight();
76  for (int mu: {0, 1, 2}) {
77  for (int nu: {0, 1, 2}) {
78  stress_kin_(mu, nu) += Gk[mu] * Gk[nu] * d;
79  }
80  }
81  } // igloc
82  } // ikloc
83 
84  ctx_.comm().allreduce(&stress_kin_(0, 0), 9 * sizeof(double));
85 
86  stress_kin_ *= (-1.0 / ctx_.unit_cell().omega());
87 
88  symmetrize(stress_kin_);
89  }

91  inline void calc_stress_har()
92  {
93  PROFILE("sirius::Stress|har");
94 
95  for (int igloc = 0; igloc < ctx_.gvec_count(); igloc++) {
96  int ig = ctx_.gvec_offset() + igloc;
97  if (!ig) {
98  continue;
99  }
100 
101  auto G = ctx_.gvec().gvec_cart(ig);
102  double g2 = std::pow(G.length(), 2);
103  auto z = density_.rho()->f_pw(ig);
104  double d = twopi * (std::pow(z.real(), 2) + std::pow(z.imag(), 2)) / g2;
105 
106  for (int mu: {0, 1, 2}) {
107  for (int nu: {0, 1, 2}) {
108  stress_har_(mu, nu) += d * 2 * G[mu] * G[nu] / g2;
109  }
110  }
111  for (int mu: {0, 1, 2}) {
112  stress_har_(mu, mu) -= d;
113  }
114  }
115 
116  if (ctx_.gvec().reduced()) {
117  stress_har_ *= 2;
118  }
119 
120  ctx_.comm().allreduce(&stress_har_(0, 0), 9 * sizeof(double));
121 
122  symmetrize(stress_har_);
123  }

125  inline void calc_stress_ewald()
126  {
127  PROFILE("sirius::Stress|ewald");
128 
129  double lambda = 1.5;
130 
131  auto& uc = ctx_.unit_cell();
132 
133  for (int igloc = 0; igloc < ctx_.gvec_count(); igloc++) {
134  int ig = ctx_.gvec_offset() + igloc;
135  if (!ig) {
136  continue;
137  }
138 
139  auto G = ctx_.gvec().gvec_cart(ig);
140  double g2 = std::pow(G.length(), 2);
141  double g2lambda = g2 / 4.0 / lambda;
142 
143  double_complex rho(0, 0);
144 
145  for (int ia = 0; ia < uc.num_atoms(); ia++) {
146  rho += ctx_.gvec_phase_factor(ig, ia) * static_cast<double>(uc.atom(ia).zn());
147  }
148 
149  double a1 = twopi * std::pow(std::abs(rho) / uc.omega(), 2) * std::exp(-g2lambda) / g2;
150 
151  for (int mu: {0, 1, 2}) {
152  for (int nu: {0, 1, 2}) {
153  stress_ewald_(mu, nu) += a1 * G[mu] * G[nu] * 2 * (g2lambda + 1) / g2;
154  }
155  }
156 
157  for (int mu: {0, 1, 2}) {
158  stress_ewald_(mu, mu) -= a1;
159  }
160  }
161 
162  if (ctx_.gvec().reduced()) {
163  stress_ewald_ *= 2;
164  }
165 
166  ctx_.comm().allreduce(&stress_ewald_(0, 0), 9 * sizeof(double));
167 
168  for (int mu: {0, 1, 2}) {
169  stress_ewald_(mu, mu) += twopi * std::pow(uc.num_electrons() / uc.omega(), 2) / 4 / lambda;
170  }
171 
172  for (int ia = 0; ia < uc.num_atoms(); ia++) {
173  for (int i = 1; i < uc.num_nearest_neighbours(ia); i++) {
174  int ja = uc.nearest_neighbour(i, ia).atom_id;
175  double d = uc.nearest_neighbour(i, ia).distance;
176 
177  vector3d<double> v1 = uc.atom(ja).position() - uc.atom(ia).position() + uc.nearest_neighbour(i, ia).translation;
178  auto r1 = uc.get_cartesian_coordinates(v1);
179  double len = r1.length();
180 
181  if (std::abs(d - len) > 1e-12) {
182  STOP();
183  }
184 
185  double a1 = (0.5 * uc.atom(ia).zn() * uc.atom(ja).zn() / uc.omega() / std::pow(len, 3)) *
186  (-2 * std::exp(-lambda * std::pow(len, 2)) * std::sqrt(lambda / pi) * len - gsl_sf_erfc(std::sqrt(lambda) * len));
187 
188  for (int mu: {0, 1, 2}) {
189  for (int nu: {0, 1, 2}) {
190  stress_ewald_(mu, nu) += a1 * r1[mu] * r1[nu];
191  }
192  }
193  }
194  }
195 
196  symmetrize(stress_ewald_);
197  }

199  inline void calc_stress_vloc()
200  {
201  PROFILE("sirius::Stress|vloc");
202 
203  Radial_integrals_vloc ri_vloc(ctx_.unit_cell(), ctx_.pw_cutoff(), 100);
204  Radial_integrals_vloc_dg ri_vloc_dg(ctx_.unit_cell(), ctx_.pw_cutoff(), 100);
205 
206  auto v = ctx_.make_periodic_function<index_domain_t::local>([&ri_vloc](int iat, double g)
207  {
208  return ri_vloc.value(iat, g);
209  });
210 
211  auto dv = ctx_.make_periodic_function<index_domain_t::local>([&ri_vloc_dg](int iat, double g)
212  {
213  return ri_vloc_dg.value(iat, g);
214  });
215 
216  double sdiag{0};
217 
218  for (int igloc = 0; igloc < ctx_.gvec_count(); igloc++) {
219  int ig = ctx_.gvec_offset() + igloc;
220 
221  if (!ig) {
222  continue;
223  }
224 
225  auto G = ctx_.gvec().gvec_cart(ig);
226 
227  for (int mu: {0, 1, 2}) {
228  for (int nu: {0, 1, 2}) {
229  stress_vloc_(mu, nu) += std::real(std::conj(density_.rho()->f_pw(ig)) * dv[igloc]) * G[mu] * G[nu];
230  }
231  }
232 
233  sdiag += std::real(std::conj(density_.rho()->f_pw(ig)) * v[igloc]);
234  }
235 
236  if (ctx_.gvec().reduced()) {
237  stress_vloc_ *= 2;
238  sdiag *= 2;
239  }
240  if (ctx_.comm().rank() == 0) {
241  sdiag += std::real(std::conj(density_.rho()->f_pw(0)) * v[0]);
242  }
243 
244  for (int mu: {0, 1, 2}) {
245  stress_vloc_(mu, mu) -= sdiag;
246  }
247 
248  ctx_.comm().allreduce(&stress_vloc_(0, 0), 9 * sizeof(double));
249 
250  symmetrize(stress_vloc_);
251  }

253  template <typename T>
254  inline void calc_stress_nonloc()
255  {
256  PROFILE("sirius::Stress|nonloc");
257 
258  auto& bchunk = ctx_.beta_projector_chunks();
259 
260  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
261  int ik = kset_.spl_num_kpoints(ikloc);
262  auto kp = kset_[ik];
263  if (kp->gkvec().reduced()) {
264  TERMINATE("fix this");
265  }
266 
267  auto d_mtrx = [&](Atom& atom, int ibnd, int i, int j)
268  {
269  if (atom.type().pp_desc().augment) {
270  return atom.d_mtrx(i, j, 0) - kp->band_energy(ibnd) * ctx_.augmentation_op(atom.type_id()).q_mtrx(i, j);
271  } else {
272  return atom.d_mtrx(i, j, 0);
273  }
274  };
275 
276  Beta_projectors_strain_deriv bp_strain_deriv(ctx_, kp->gkvec());
277 
278  kp->beta_projectors().prepare();
279  bp_strain_deriv.prepare();
280 
281  for (int ichunk = 0; ichunk < bchunk.num_chunks(); ichunk++) {
282  kp->beta_projectors().generate(ichunk);
283 
284  int nbnd = kp->num_occupied_bands(0);
285  splindex<block> spl_nbnd(nbnd, kp->comm().size(), kp->comm().rank());
286 
287  /* compute <beta|psi> */
288  auto beta_psi = kp->beta_projectors().inner<T>(ichunk, kp->spinor_wave_functions(0), 0, nbnd);
289 
290  for (int mu = 0; mu < 3; mu++) {
291  for (int nu = 0; nu < 3; nu++) {
292  bp_strain_deriv.generate(ichunk, mu + nu * 3);
293  auto dbeta_psi = bp_strain_deriv.inner<T>(ichunk, kp->spinor_wave_functions(0), 0, nbnd);
294 
295  for (int i = 0; i < bchunk(ichunk).num_atoms_; i++) {
296  int ia = bchunk(ichunk).desc_(beta_desc_idx::ia, i);
297  int offs = bchunk(ichunk).desc_(beta_desc_idx::offset, i);
298  int nbf = bchunk(ichunk).desc_(beta_desc_idx::nbf, i);
299 
300  auto& atom = ctx_.unit_cell().atom(ia);
301 
302  for (int ibnd_loc = 0; ibnd_loc < spl_nbnd.local_size(); ibnd_loc++) {
303  int ibnd = spl_nbnd[ibnd_loc];
304  double w = kp->weight() * kp->band_occupancy(ibnd);
305 
306  for (int xi1 = 0; xi1 < nbf; xi1++) {
307  for (int xi2 = 0; xi2 < nbf; xi2++) {
308  stress_nonloc_(mu, nu) += w * std::real(
309  std::conj(dbeta_psi(offs + xi1, ibnd)) * d_mtrx(atom, ibnd, xi1, xi2) * beta_psi(offs + xi2, ibnd) +
310  std::conj(beta_psi(offs + xi1, ibnd)) * d_mtrx(atom, ibnd, xi1, xi2) * dbeta_psi(offs + xi2, ibnd)
311  );
312  }
313  }
314  }
315  }
316  }
317  }
318  }
319  kp->beta_projectors().dismiss();
320  }
321  ctx_.comm().allreduce(&stress_nonloc_(0, 0), 9 * sizeof(double));
322 
323  stress_nonloc_ *= (1.0 / ctx_.unit_cell().omega());
324 
325  symmetrize(stress_nonloc_);
326  }

328  inline void symmetrize(matrix3d<double>& mtrx__)
329  {
330  if (!ctx_.use_symmetry()) {
331  return;
332  }
333 
334  matrix3d<double> result;
335 
336  for (int i = 0; i < ctx_.unit_cell().symmetry().num_mag_sym(); i++) {
337  auto R = ctx_.unit_cell().symmetry().magnetic_group_symmetry(i).spg_op.rotation;
338  result = result + transpose(R) * mtrx__ * R;
339  }
340 
341  mtrx__ = result * (1.0 / ctx_.unit_cell().symmetry().num_mag_sym());
342  }

398 /**
399 \page stress Stress tensor
400 \section stress_section1 Preliminary notes
401 Derivative of the G-vector in Cartesian coordinates over the lattice vector components:
402 \f[
403  \frac{\partial G_{\tau}}{\partial a_{\mu\nu}} + ({\bf a}^{-1})_{\nu \tau} G_{\mu} = 0
404 \f]
405 Mathematica proof script:
406 \verbatim
407 A = Table[Subscript[a, i, j], {i, 1, 3}, {j, 1, 3}];
408 invA = Inverse[A];
409 B = 2*Pi*Transpose[Inverse[A]];
410 G = Table[Subscript[g, i], {i, 1, 3}];
411 gvec = B.G;
412 Do[
413  Print[FullSimplify[
414  D[gvec[[tau]], Subscript[a, mu, nu]] + invA[[nu]][[tau]]*gvec[[mu]]]],
415 {beta, 1, 3}, {mu, 1, 3}, {nu, 1,3}]
416 \endverbatim
417 Another relation:
418 \f[
419  \frac{\partial}{\partial a_{\mu \nu}} \frac{1}{\sqrt{\Omega}} + \frac{1}{2} \frac{1}{\sqrt{\Omega}} ({\bf a}^{-1})_{\nu \mu} = 0
420 \f]
421 Mathematica proof script:
422 \verbatim
423 A = Table[Subscript[a, i, j], {i, 1, 3}, {j, 1, 3}];
424 invA = Inverse[A];
425 Do[
426  Print[FullSimplify[
427  D[1/Sqrt[Det[A]], Subscript[a, mu, nu]] + (1/2)*(1/Sqrt[Det[A]]) * invA[[nu]][[mu]]
428  ]
429  ],
430 {mu, 1, 3}, {nu, 1, 3}]
431 \endverbatim
432 
433 Strain tensor describes the deformation of a crystal:
434 \f[
435  r_{\mu} \rightarrow \sum_{\nu} (\delta_{\mu \nu} + \varepsilon_{\mu \nu}) r_{\nu}
436 \f]
437 Strain derivative of general position vector component:
438 \f[
439  \frac{\partial r_{\tau}}{\partial \varepsilon_{\mu \nu}} = \delta_{\tau \mu} r_{\nu}
440 \f]
441 Strain derivative of the lengths of general position vector:
442 \f[
443  \frac{\partial |{\bf r}|}{\partial \varepsilon_{\mu \nu}} = \sum_{\tau} \frac{\partial |{\bf r}|}{\partial r_{\tau}}
444  \frac{\partial r_{\tau}}{\partial \varepsilon_{\mu \nu}} = \sum_{\tau} \frac{ r_{\tau}}{|{\bf r}|}
445  \delta_{\tau \mu} r_{\nu} = \frac{r_{\mu} r_{\nu}}{|{\bf r}|}
446 \f]
447 Strain derivative of unit cell volume:
448 \f[
449  \frac{\partial \Omega}{\partial \varepsilon_{\mu \nu}} = \delta_{\mu \nu} \Omega
450 \f]
451 Strain derivative of reciprocal vector:
452 \f[
453  \frac{\partial G_{\tau}}{\partial \varepsilon_{\mu \nu}} = -\delta_{\tau \nu} G_{\mu}
454 \f]
455 Strain derivative of the length of reciprocal vector:
456 \f[
457  \frac{\partial |{\bf G}|}{\partial \varepsilon_{\mu \nu}} = \sum_{\tau} \frac{\partial |{\bf G}|}{\partial G_{\tau}}
458  \frac{\partial G_{\tau}}{\partial \varepsilon_{\mu \nu}} = -\frac{1}{|{\bf G}|}G_{\nu}G_{\mu}
459 \f]
460 Stress tensor is a reaction to strain:
461 \f[
462  \sigma_{\mu \nu} = \frac{1}{\Omega} \frac{\partial E_{tot}}{\partial \varepsilon_{\mu \nu}} =
463  \frac{1}{\Omega} \sum_{\mu' \nu'} \frac{\partial E_{tot}}{\partial a_{\mu' \nu'}} \frac{\partial a_{\mu' \nu'}}{\partial \varepsilon_{\mu \nu}} =
464  \frac{1}{\Omega} \sum_{\nu'} \frac{\partial E_{tot}}{\partial a_{\mu \nu'}} a_{\nu \nu'}
465 \f]
466 
467 \section stress_section2 Derivative of beta-projectors.
468 We need to compute derivative of beta-projectors
469 \f[
470  \frac{\partial}{\partial a_{\mu \nu}} \langle {\bf G+k} | \beta_{\ell m} \rangle
471 \f]
472 
473 Derivative of the G-vector real spherical harmonics over the lattice vector components:
474 \f[
475  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial a_{\mu \nu}} =
476  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \frac{\partial \theta} {\partial a_{\mu \nu}} +
477  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{\partial \phi} {\partial a_{\mu \nu}}
478 \f]
479 Derivatives of the \f$ R_{\ell m} \f$ with respect to the \f$ \theta,\, \phi\f$ angles can be tabulated up to a given \f$ \ell_{max} \f$.
480 The derivatives of angles are computed as following:
481 \f[
482  \frac{\partial \theta} {\partial a_{\mu \nu}} = \sum_{\beta=1}^{3} \frac{\partial \theta}{\partial G_{\beta}} \frac{\partial G_{\beta}} {\partial a_{\mu \nu}}
483 \f]
484 \f[
485  \frac{\partial \phi} {\partial a_{\mu \nu}} = \sum_{\beta=1}^{3} \frac{\partial \phi}{\partial G_{\beta}} \frac{\partial G_{\beta}} {\partial a_{\mu \nu}}
486 \f]
487 where
488 \f[
489  \frac{\partial \theta}{\partial G_{x}} = \frac{\cos(\phi) \cos(\theta)}{G} \\
490  \frac{\partial \theta}{\partial G_{y}} = \frac{\cos(\theta) \sin(\phi)}{G} \\
491  \frac{\partial \theta}{\partial G_{z}} = -\frac{\sin(\theta)}{G}
492 \f]
493 and
494 \f[
495  \frac{\partial \phi}{\partial G_{x}} = -\frac{\sin(\phi)}{\sin(\theta) G} \\
496  \frac{\partial \phi}{\partial G_{y}} = \frac{\cos(\phi)}{\sin(\theta) G} \\
497  \frac{\partial \phi}{\partial G_{z}} = 0
498 \f]
499 The derivative of \f$ phi \f$ has discontinuities at \f$ \theta = 0, \theta=\pi \f$. This, however, is not a problem, because
500 multiplication by the the derivative of \f$ R_{\ell m} \f$ removes it. The following functions have to be hardcoded:
501 \f[
502  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \\
503  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{1}{\sin(\theta)}
504 \f]
505 
506 Derivatives of the spherical Bessel functions are computed in the same fashion:
507 \f[
508  \frac{\partial j_{\ell}(Gx)}{\partial a_{\mu \nu}} =
509  \frac{\partial j_{\ell}(Gx)}{\partial G} \frac{\partial G} {\partial a_{\mu \nu}} =
510  \frac{\partial j_{\ell}(Gx)}{\partial G} \sum_{\beta=1}^{3}\frac{\partial G}{\partial G_{\beta}} \frac{\partial G_{\beta}} {\partial a_{\mu \nu}}
511 \f]
512 The derivatives of \f$ G \f$ are:
513 \f[
514  \frac{\partial G}{\partial G_{x}} = \sin(\theta)\cos(\phi) \\
515  \frac{\partial G}{\partial G_{y}} = \sin(\theta)\sin(\phi) \\
516  \frac{\partial G}{\partial G_{z}} = \cos(\theta)
517 \f]
518 
519 Let's write the full expression for the derivative of beta-projector matrix elements with respect to lattice vector
520 components:
521 \f[
522  \frac{\partial \langle {\bf G+k}|\beta_{\ell m} \rangle} {\partial a_{\mu \nu}} =
523  \frac{\partial} {\partial a_{\mu \nu}} \frac{4\pi}{\sqrt{\Omega}}(-i)^{\ell} R_{\ell m}(\theta_{G+k}, \phi_{G+k}) \int \beta_{\ell}(r) j_{\ell}(Gr) r^2 dr =\\
524  \frac{4\pi}{\sqrt{\Omega}} (-i)^{\ell} \Bigg[ \int \beta_{\ell}(r) j_{\ell}(Gr) r^2 dr
525  \Big( \frac{\partial R_{\ell m}(\theta, \phi)}{\partial a_{\mu \nu}} - \frac{1}{2} R_{\ell m}(\theta, \phi) ({\bf a}^{-1})_{\nu \mu} \Big) +
526  R_{\ell m}(\theta, \phi) \int \beta_{\ell}(r) \frac{\partial j_{\ell}(Gr)}{\partial a_{\mu \nu}} r^2 dr \Bigg]
527 \f]
528 
529 \section stress_section3 Kinetic energy contribution to stress.
530 Kinetic contribution to stress tensor:
531 \f[
532  E^{kin} = \sum_{{\bf k}} w_{\bf k} \sum_i f_i \frac{1}{2} |{\bf G+k}|^2 |\psi_i({\bf G + k})|^2
533 \f]
534 Derivative over lattice vectors
535 \f[
536  \frac{\partial E^{kin}}{\partial a_{\mu \nu}} =
537  \sum_{{\bf k}} w_{\bf k} \sum_i f_i \sum_{\tau} (G+k)_{\tau} (-{\bf a}^{-1})_{\nu \tau} (G+k)_{\mu} |\psi_i({\bf G + k})|^2
538 \f]
539 Contribution to the stress tensor
540 \f[
541  \sigma_{\mu \nu}^{kin} = \frac{1}{\Omega} \sum_{\nu'} \frac{\partial E^{kin}}{\partial a_{\mu \nu'}} a_{\nu \nu'} =

547 Hartree energy:
548 \f[

551  \frac{2 \pi}{\Omega} \sum_{\bf G} \frac{|\tilde \rho({\bf G})|^2}{G^2}
552 \f]
553 where
554 \f[
555  \langle {\bf G} | \rho \rangle = \int e^{-i{\bf Gr}}\rho({\bf r}) d {\bf r} = \tilde \rho({\bf G})
556 \f]
557 and
558 \f[
559  \langle {\bf G} | V^{H} \rangle = \int e^{-i{\bf Gr}}V^{H}({\bf r}) d {\bf r} = \frac{4\pi}{G^2} \tilde \rho({\bf G})

569  \frac{1}{\Omega} \sum_{\bf G} |\tilde \rho({\bf G})|^2 \sum_{\tau} \frac{-2 G_{\tau}}{G^4} \frac{\partial G_{\tau}}{\partial \varepsilon_{\mu \nu}} \Big) = \\

575 \f[

612 \f[
613  E^{loc} = \int \rho({\bf r}) V^{loc}({\bf r}) d{\bf r} = \frac{1}{\Omega} \sum_{\bf G} \langle \rho | {\bf G} \rangle \langle {\bf G}| V^{loc} \rangle =

615 \f]
616 where
617 \f[
618  \tilde \rho({\bf G}) = \langle {\bf G} | \rho \rangle = \int e^{-i{\bf Gr}}\rho({\bf r}) d {\bf r}
619 \f]
620 and
621 \f[
622  \tilde V^{loc}({\bf G}) = \langle {\bf G} | V^{loc} \rangle = \int e^{-i{\bf Gr}}V^{loc}({\bf r}) d {\bf r}
623 \f]

625 \f[
626  E^{loc} = \frac{1}{\Omega} \sum_{\bf G} \tilde \rho^{*}({\bf G}) \sum_{\alpha} e^{-{\bf G\tau}_{\alpha}} 4 \pi
627  \int V_{\alpha}^{loc}(r)\frac{\sin(Gr)}{Gr} r^2 dr =
628  \frac{4\pi}{\Omega}\sum_{\bf G}\tilde \rho^{*}({\bf G})\sum_{\alpha} e^{-{\bf G\tau}_{\alpha}}

634 \f[

649 \section stress_section7 Non-local contribution to stress.
650 Energy contribution from the non-local part of pseudopotential:
651 \f[
652  E^{nl} = \sum_{i{\bf k}} \sum_{\alpha}\sum_{\xi \xi'}P_{\xi}^{\alpha,i{\bf k}} D_{\xi\xi'}^{\alpha}P_{\xi'}^{\alpha,i{\bf k}*}
653 \f]
654 where
655 \f[
656  P_{\xi}^{\alpha,i{\bf k}} = \langle \psi_{i{\bf k}} | \beta_{\xi}^{\alpha} \rangle =
657  \frac{1}{\Omega} \sum_{\bf G} \langle \psi_{i{\bf k}} | {\bf G+k} \rangle \langle {\bf G+k} | \beta_{\xi}^{\alpha} \rangle =
658  \frac{1}{\Omega} \sum_{\bf G} \tilde \psi_i^{*}({\bf G+k}) \tilde \beta_{\xi}^{\alpha}({\bf G+k})
659 \f]
660 We need to compute strain derivatives of \f$ P_{\xi}^{\alpha,i{\bf k}} \f$:
661 \f[
662  \frac{\partial}{\partial \varepsilon_{\mu \nu}} P_{\xi}^{\alpha,i{\bf k}} = -\frac{1}{\Omega}\delta_{\mu \nu}
663  \sum_{\bf G} \tilde \psi_i^{*}({\bf G+k}) \tilde \beta_{\xi}^{\alpha}({\bf G+k}) + \frac{1}{\Omega}
664  \sum_{\bf G} \tilde \psi_i^{*}({\bf G+k}) \frac{\partial}{\partial \varepsilon_{\mu \nu}} \tilde \beta_{\xi}^{\alpha}({\bf G+k})
665 \f]
666 \f[
667  \tilde \beta_{\xi}^{\alpha}({\bf G+k}) = 4\pi e^{-i{\bf G r_{\alpha}}}(-i)^{\ell} R_{\ell m}(\theta_{G+k}, \phi_{G+k})
668  \int \beta_{\ell}(r) j_{\ell}(|{\bf G+k}|r) r^2 dr
669 \f]
670 First way to take strain derivative of beta-projectors:
671 \f[
672  \frac{\partial}{\partial \varepsilon_{\mu \nu}} \tilde \beta_{\xi}^{\alpha}({\bf G+k}) = 4\pi e^{-i{\bf G r_{\alpha}}}(-i)^{\ell}
673  \Big( \frac{\partial R_{\ell m}(\theta_{G+k}, \phi_{G+k})}{\partial \varepsilon_{\mu \nu}}
674  \int \beta_{\ell}(r) j_{\ell}(|{\bf G+k}|r) r^2 dr + R_{\ell m}(\theta_{G+k}, \phi_{G+k})
675  \int \beta_{\ell}(r) \frac{\partial j_{\ell}(|{\bf G+k}|r) }{\partial \varepsilon_{\mu \nu}} r^2 dr
676  \Big)
677 \f]
678 
679 Strain derivative of the real spherical harmonics (here and below \f$ {\bf G+k} = {\bf q} \f$):
680 \f[
681  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \varepsilon_{\mu \nu}} =
682  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \sum_{\tau} \frac{\partial \theta}{\partial q_{\tau}} \frac{\partial{q_{\tau}}}{\partial \varepsilon_{\mu \nu}} +
683  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \sum_{\tau} \frac{\partial \phi}{\partial q_{\tau}}\frac{\partial q_{\tau}}{\partial \varepsilon_{\mu \nu}} =
684  -q_{\mu} \Big( \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \frac{\partial \theta}{\partial q_{\nu}} +
685  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{\partial \phi}{\partial q_{\nu}}\Big)
686 \f]
687 
688 The derivatives of angles are:
689 \f[
690  \frac{\partial \theta}{\partial q_{x}} = \frac{\cos(\phi) \cos(\theta)}{q} \\
691  \frac{\partial \theta}{\partial q_{y}} = \frac{\cos(\theta) \sin(\phi)}{q} \\
692  \frac{\partial \theta}{\partial q_{z}} = -\frac{\sin(\theta)}{q}
693 \f]
694 and
695 \f[
696  \frac{\partial \phi}{\partial q_{x}} = -\frac{\sin(\phi)}{\sin(\theta) q} \\
697  \frac{\partial \phi}{\partial q_{y}} = \frac{\cos(\phi)}{\sin(\theta) q} \\
698  \frac{\partial \phi}{\partial q_{z}} = 0
699 \f]
700 The derivative of \f$ \phi \f$ has discontinuities at \f$ \theta = 0, \theta=\pi \f$. This, however, is not a problem, because
701 multiplication by the the derivative of \f$ R_{\ell m} \f$ removes it. The following functions have to be hardcoded:
702 \f[
703  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \theta} \\
704  \frac{\partial R_{\ell m}(\theta, \phi)}{\partial \phi} \frac{1}{\sin(\theta)}
705 \f]
