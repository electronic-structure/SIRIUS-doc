107 /// Implementation of FFT3D.
108 /** FFT convention:
109  * \f[
110  * f({\bf r}) = \sum_{{\bf G}} e^{i{\bf G}{\bf r}} f({\bf G})
111  * \f]
112  * is a \em backward transformation from a set of pw coefficients to a function.
113  *
114  * \f[
115  * f({\bf G}) = \frac{1}{\Omega} \int e^{-i{\bf G}{\bf r}} f({\bf r}) d {\bf r} =
116  * \frac{1}{N} \sum_{{\bf r}_j} e^{-i{\bf G}{\bf r}_j} f({\bf r}_j)
117  * \f]
118  * is a \em forward transformation from a function to a set of coefficients.
119  *
120  * The following cases are handeled by the FFT driver:
121  * - transformation of a single real / complex function (serial / parallel, cpu / gpu)
122  * - transformation of two real functions (serial / parallel, cpu / gpu)
123  * - input / ouput data buffer pointer (cpu / gpu)
124  *
125  * The transformation of two real functions is done as one transformation of complex function:
126  * \f[
127  * \Psi({\bf r}) = \psi_1({\bf r}) + i \psi_2({\bf r})
128  * \f]
129  * For each of the real wave functions the following condition is fulfilled:
130  * \f[
131  * \psi_{1,2}({\bf r}) = \psi_{1,2}^{*}({\bf r})
132  * \f]
133  * from which it follows that
134  * \f[
135  * \psi_{1,2}({\bf G}) = \psi_{1,2}^{*}(-{\bf G})
136  * \f]
137  * When z-direction is transformed
138  * \f[
139  * \psi_{1,2}(G_x, G_y, z) = \sum_{G_z} e^{izG_z} \psi_{1,2}(G_x, G_y, G_z)
140  * \f]
141  * it leads to the following symmetry
142  * \f[
143  * \psi_{1,2}^{*}(G_x, G_y, z) = \sum_{G_z} e^{-izG_z} \psi_{1,2}^{*}(G_x, G_y, G_z) =
144  * \sum_{G_z} e^{-izG_z} \psi_{1,2}(-G_x, -G_y, -G_z) = \psi_{1,2}(-G_x, -G_y, z)
145  * \f]
146  */

213  /// Serial part of 1D transformation of columns.
214  template <int direction, device_t data_ptr_type>
215  void transform_z_serial(Gvec_partition const& gvec__,
216  double_complex* data__,
217  mdarray<double_complex, 1>& fft_buffer_aux__) // TODO: prepare data for mpi_a2a in case of gpu_ptr
218  {
219  PROFILE("sddk::FFT3D::transform_z_serial");
220 
221  int num_zcol_local = gvec__.zcol_count_fft();
222  double norm = 1.0 / size();
223 
224  bool is_reduced = gvec__.reduced();
225 
226  assert(static_cast<int>(fft_buffer_aux__.size()) >= gvec__.zcol_count_fft() * grid_.size(2));
227 
228  /* input/output data buffer is on GPU */
229  if (data_ptr_type == GPU) {
230  #ifdef __GPU
231  switch (direction) {
232  case 1: {
233  /* load all columns into FFT buffer */
234  cufft_batch_load_gpu(gvec__.zcol_count_fft() * grid_.size(2),
235  gvec__.gvec_count_fft(),
236  1,
237  map_zcol_to_fft_buffer_.at<GPU>(),
238  data__,
239  fft_buffer_aux__.at<GPU>(),
240  0);
241  if (is_reduced && comm_.rank() == 0) {
242  cufft_load_x0y0_col_gpu(static_cast<int>(gvec__.zcol(0).z.size()),
243  map_zcol_to_fft_buffer_x0y0_.at<GPU>(),
244  data__,
245  fft_buffer_aux__.at<GPU>(),
246  0);
247  }
248  /* transform all columns */
249  cufft_backward_transform(cufft_plan_z_, fft_buffer_aux__.at<GPU>());
250  break;
251  }
252  case -1: {
253  /* transform all columns */
254  cufft_forward_transform(cufft_plan_z_, fft_buffer_aux__.at<GPU>());
255  acc::zero(data__, gvec__.gvec_count_fft()); // TODO: this should happen in cufft_batch_unload_gpu()
256  /* get all columns from FFT buffer */
257  cufft_batch_unload_gpu(gvec__.zcol_count_fft() * grid_.size(2), gvec__.gvec_count_fft(), 1,
258  map_zcol_to_fft_buffer_.at<GPU>(), fft_buffer_aux__.at<GPU>(), data__, 0.0, norm, 0);
259  break;
260  }
261  default: {
262  TERMINATE("wrong direction");
263  }
264  }
265  acc::sync_stream(0);
266  #endif
267  }
268 
269  if (data_ptr_type == CPU) {
270  #pragma omp parallel
271  {
272  int tid = omp_get_thread_num();
273  #pragma omp for schedule(dynamic, 1)
274  for (int i = 0; i < num_zcol_local; i++) {
275  /* global index of column */
276  int icol = gvec__.zcol_offset_fft() + i;
277  /* offset of the PW coeffs in the input/output data buffer */
278  int data_offset = gvec__.zcol_offs(icol);
279 
280  switch (direction) {
281  case 1: {
282  /* clear z buffer */
283  std::fill(fftw_buffer_z_[tid], fftw_buffer_z_[tid] + grid_.size(2), 0);
284  /* load z column of PW coefficients into buffer */
285  for (size_t j = 0; j < gvec__.zcol(icol).z.size(); j++) {
286  int z = grid().coord_by_gvec(gvec__.zcol(icol).z[j], 2);
287  fftw_buffer_z_[tid][z] = data__[data_offset + j];
288  }
289 
290  /* column with {x,y} = {0,0} has only non-negative z components */
291  if (is_reduced && !icol) {
292  /* load remaining part of {0,0,z} column */
293  for (size_t j = 0; j < gvec__.zcol(icol).z.size(); j++) {
294  int z = grid().coord_by_gvec(-gvec__.zcol(icol).z[j], 2);
295  fftw_buffer_z_[tid][z] = std::conj(data__[data_offset + j]);
296  }
297  }
298 
299  /* perform local FFT transform of a column */
300  fftw_execute(plan_backward_z_[tid]);
301 
302  /* redistribute z-column for a forthcoming all-to-all or just load the
303  * full column into auxiliary buffer in serial case */
304  for (int r = 0; r < comm_.size(); r++) {
305  int lsz = spl_z_.local_size(r);
306  int offs = spl_z_.global_offset(r);
307 
308  std::copy(&fftw_buffer_z_[tid][offs],
309  &fftw_buffer_z_[tid][offs] + lsz,
310  &fft_buffer_aux__[offs * num_zcol_local + i * lsz]);
311  }
312 
313  break;
314 
315  }
316  case -1: {
317  /* collect full z-column or just load it from the auxiliary buffer is serial case */
318  for (int r = 0; r < comm_.size(); r++) {
319  int lsz = spl_z_.local_size(r);
320  int offs = spl_z_.global_offset(r);
321 
322  std::copy(&fft_buffer_aux__[offs * num_zcol_local + i * lsz],
323  &fft_buffer_aux__[offs * num_zcol_local + i * lsz] + lsz,
324  &fftw_buffer_z_[tid][offs]);
325  }
326 
327  /* perform local FFT transform of a column */
328  fftw_execute(plan_forward_z_[tid]);
329 
330  /* save z column of PW coefficients */
331  for (size_t j = 0; j < gvec__.zcol(icol).z.size(); j++) {
332  int z = grid().coord_by_gvec(gvec__.zcol(icol).z[j], 2);
333  data__[data_offset + j] = fftw_buffer_z_[tid][z] * norm;
334  }
335 
336  break;
337  }
338  default: {
339  TERMINATE("wrong direction");
340  }
341  }
342  }
343  }
344  }

347  /// Transformation of z-columns.
348  template <int direction, device_t data_ptr_type>
349  void transform_z(Gvec_partition const& gvec__,
350  double_complex* data__,
351  mdarray<double_complex, 1>& fft_buffer_aux__)
352  {
353  PROFILE("sddk::FFT3D::transform_z");
354 
355  int rank = comm_.rank();
356  int num_zcol_local = gvec__.zcol_distr_fft().counts[rank];
357 
358  if (direction == -1) {
359  #ifdef __GPU
360  /* copy z-sticks to CPU */
361  if ((data_ptr_type == CPU && pu_ == GPU) || (data_ptr_type == GPU && comm_.size() > 1)) {
362  fft_buffer_aux__.copy_to_host(local_size_z_ * gvec__.num_zcol());
363  }
364  #endif
365 
366  /* collect full sticks */
367  if (comm_.size() > 1) {
368  sddk::timer t("sddk::FFT3D::transform_z|comm");
369 
370  block_data_descriptor send(comm_.size());
371  block_data_descriptor recv(comm_.size());
372  for (int r = 0; r < comm_.size(); r++) {
373  send.counts[r] = spl_z_.local_size(rank) * gvec__.zcol_distr_fft().counts[r];
374  recv.counts[r] = spl_z_.local_size(r) * gvec__.zcol_distr_fft().counts[rank];
375  }
376  send.calc_offsets();
377  recv.calc_offsets();
378 
379  /* copy auxiliary buffer because it will be use as the output buffer in the following mpi_a2a */
380  std::copy(&fft_buffer_aux__[0], &fft_buffer_aux__[0] + gvec__.num_zcol() * local_size_z_,
381  &fft_buffer_[0]);
382 
383  comm_.alltoall(&fft_buffer_[0], &send.counts[0], &send.offsets[0], &fft_buffer_aux__[0], &recv.counts[0], &recv.offsets[0]);
384  }
385 
386  /* buffer is on CPU after mpi_a2a and has to be copied to GPU */
387  #ifdef __GPU
388  if (data_ptr_type == GPU && comm_.size() > 1) {
389  fft_buffer_aux__.copy_to_device(gvec__.zcol_count_fft() * grid_.size(2));
390  }
391  #endif
392  }
393 
394  transform_z_serial<direction, data_ptr_type>(gvec__, data__, fft_buffer_aux__);
395 
396  if (direction == 1) {
397  /* data is on GPU but we need to perform mpi_a2a */
398  #ifdef __GPU
399  if (data_ptr_type == GPU && comm_.size() > 1) {
400  fft_buffer_aux__.copy_to_host(gvec__.zcol_count_fft() * grid_.size(2));
401  }
402  #endif
403 
404  /* scatter z-columns between slabs of FFT buffer */
405  if (comm_.size() > 1) {
406  sddk::timer t("sddk::FFT3D::transform_z|comm");
407 
408  block_data_descriptor send(comm_.size());
409  block_data_descriptor recv(comm_.size());
410  for (int r = 0; r < comm_.size(); r++) {
411  send.counts[r] = spl_z_.local_size(r) * gvec__.zcol_distr_fft().counts[rank];
412  recv.counts[r] = spl_z_.local_size(rank) * gvec__.zcol_distr_fft().counts[r];
413  }
414  send.calc_offsets();
415  recv.calc_offsets();
416 
417  /* scatter z-columns */
418  comm_.alltoall(&fft_buffer_aux__[0], &send.counts[0], &send.offsets[0], &fft_buffer_[0], &recv.counts[0], &recv.offsets[0]);
419 
420  /* copy local fractions of z-columns into auxiliary buffer */
421  std::copy(&fft_buffer_[0], &fft_buffer_[0] + gvec__.num_zcol() * local_size_z_,
422  &fft_buffer_aux__[0]);
423  }
424 
425  #ifdef __GPU
426  if ((data_ptr_type == CPU && pu_ == GPU) || (data_ptr_type == GPU && comm_.size() > 1)) {
427  fft_buffer_aux__.copy_to_device(local_size_z_ * gvec__.num_zcol());
428  }
429  #endif
430  }
431  }

433  /// Apply 2D FFT transformation to z-columns of one complex function.
434  template <int direction>
435  void transform_xy(Gvec_partition const& gvec__,
436  mdarray<double_complex, 1>& fft_buffer_aux__)
437  {
438  PROFILE("sddk::FFT3D::transform_xy");
439 
440  int size_xy = grid_.size(0) * grid_.size(1);
441 
442  int is_reduced = gvec__.reduced();
443 
444  #ifdef __GPU
445  if (pu_ == GPU) {
446  /* stream #0 will be doing cuFFT */
447  switch (direction) {
448  case 1: {
449  /* srteam #0 unpacks z-columns into proper position of FFT buffer */
450  unpack_z_cols_gpu(fft_buffer_aux__.at<GPU>(), fft_buffer_.at<GPU>(), grid_.size(0), grid_.size(1),
451  local_size_z_, gvec__.num_zcol(), z_col_pos_.at<GPU>(), is_reduced, 0);
452  /* stream #0 executes FFT */
453  cufft_backward_transform(cufft_plan_xy_, fft_buffer_.at<GPU>());
454  break;
455  }
456  case -1: {
457  /* stream #0 executes FFT */
458  cufft_forward_transform(cufft_plan_xy_, fft_buffer_.at<GPU>());
459  /* stream #0 packs z-columns */
460  pack_z_cols_gpu(fft_buffer_aux__.at<GPU>(), fft_buffer_.at<GPU>(), grid_.size(0), grid_.size(1),
461  local_size_z_, gvec__.num_zcol(), z_col_pos_.at<GPU>(), 0);
462  break;
463  }
464  }
465  acc::sync_stream(0);
466  }
467  #endif
468 
469  if (pu_ == CPU) {
470  #pragma omp parallel
471  {
472  int tid = omp_get_thread_num();
473  #pragma omp for schedule(static)
474  for (int iz = 0; iz < local_size_z_; iz++) {
475  switch (direction) {
476  case 1: {
477  /* clear xy-buffer */
478  std::fill(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, 0);
479  /* load z-columns into proper location */
480  for (int i = 0; i < gvec__.num_zcol(); i++) {
481  fftw_buffer_xy_[tid][z_col_pos_(i, 0)] = fft_buffer_aux__[iz + i * local_size_z_];
482 
483  if (is_reduced && i) {
484  fftw_buffer_xy_[tid][z_col_pos_(i, 1)] = std::conj(fftw_buffer_xy_[tid][z_col_pos_(i, 0)]);
485  }
486  }
487 
488  /* execute local FFT transform */
489  fftw_execute(plan_backward_xy_[tid]);
490 
491  /* copy xy plane to the main FFT buffer */
492  std::copy(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, &fft_buffer_[iz * size_xy]);
493 
494  break;
495  }
496  case -1: {
497  /* copy xy plane from the main FFT buffer */
498  std::copy(&fft_buffer_[iz * size_xy], &fft_buffer_[iz * size_xy] + size_xy, fftw_buffer_xy_[tid]);
499 
500  /* execute local FFT transform */
501  fftw_execute(plan_forward_xy_[tid]);
502 
503  /* get z-columns */
504  for (int i = 0; i < gvec__.num_zcol(); i++) {
505  fft_buffer_aux__[iz + i * local_size_z_] = fftw_buffer_xy_[tid][z_col_pos_(i, 0)];
506  }
507 
508  break;
509  }
510  default: {
511  TERMINATE("wrong direction");
512  }
513  }
514  }
515  }
516  }
517  }

522  mdarray<double_complex, 1>& fft_buffer_aux1__,

526 
527  if (!gvec__.reduced()) {
528  TERMINATE("reduced set of G-vectors is required");
529  }
530 
531  int size_xy = grid_.size(0) * grid_.size(1);
532 
533  #ifdef __GPU
534  if (pu_ == GPU) {
535  /* stream #0 will be doing cuFFT */
536  switch (direction) {
537  case 1: {
538  /* srteam #0 unpacks z-columns into proper position of FFT buffer */
539  unpack_z_cols_2_gpu(fft_buffer_aux1__.at<GPU>(), fft_buffer_aux2__.at<GPU>(), fft_buffer_.at<GPU>(),
540  grid_.size(0), grid_.size(1), local_size_z_,
541  gvec__.num_zcol(), z_col_pos_.at<GPU>(), 0);
542  /* stream #0 executes FFT */
543  cufft_backward_transform(cufft_plan_xy_, fft_buffer_.at<GPU>());
544  break;
545  }
546  case -1: {
547  /* stream #0 executes FFT */
548  cufft_forward_transform(cufft_plan_xy_, fft_buffer_.at<GPU>());
549  /* stream #0 packs z-columns */
550  pack_z_cols_2_gpu(fft_buffer_aux1__.at<GPU>(), fft_buffer_aux2__.at<GPU>(), fft_buffer_.at<GPU>(),
551  grid_.size(0), grid_.size(1), local_size_z_,
552  gvec__.num_zcol(), z_col_pos_.at<GPU>(), 0);
553  break;
554  }
555  }
556  acc::sync_stream(0);
557  }
558  #endif
559 
560  if (pu_ == CPU) {
561  #pragma omp parallel
562  {
563  int tid = omp_get_thread_num();
564  #pragma omp for schedule(static)
565  for (int iz = 0; iz < local_size_z_; iz++) {
566  switch (direction) {
567  case 1: {
568  /* clear xy-buffer */
569  std::fill(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, 0);
570 
571  /* load first z-column into proper location */
572  fftw_buffer_xy_[tid][z_col_pos_(0, 0)] = fft_buffer_aux1__[iz] +
573  double_complex(0, 1) * fft_buffer_aux2__[iz];
574 
575  /* load remaining z-columns into proper location */
576  for (int i = 1; i < gvec__.num_zcol(); i++) {
577  /* {x, y} part */
578  fftw_buffer_xy_[tid][z_col_pos_(i, 0)] = fft_buffer_aux1__[iz + i * local_size_z_] +
579  double_complex(0, 1) * fft_buffer_aux2__[iz + i * local_size_z_];
580 
581  /* {-x, -y} part */
582  fftw_buffer_xy_[tid][z_col_pos_(i, 1)] = std::conj(fft_buffer_aux1__[iz + i * local_size_z_]) +
583  double_complex(0, 1) * std::conj(fft_buffer_aux2__[iz + i * local_size_z_]);
584  }
585 
586  /* execute local FFT transform */
587  fftw_execute(plan_backward_xy_[tid]);
588 
589  /* copy xy plane to the main FFT buffer */
590  std::copy(fftw_buffer_xy_[tid], fftw_buffer_xy_[tid] + size_xy, &fft_buffer_[iz * size_xy]);
591 
592  break;
593  }
594  case -1: {
595  /* copy xy plane from the main FFT buffer */
596  std::copy(&fft_buffer_[iz * size_xy], &fft_buffer_[iz * size_xy] + size_xy, fftw_buffer_xy_[tid]);
597 
598  /* execute local FFT transform */
599  fftw_execute(plan_forward_xy_[tid]);
600 
601  /* get z-columns */
602  for (int i = 0; i < gvec__.num_zcol(); i++) {
603  fft_buffer_aux1__[iz + i * local_size_z_] = 0.5 *
604  (fftw_buffer_xy_[tid][z_col_pos_(i, 0)] + std::conj(fftw_buffer_xy_[tid][z_col_pos_(i, 1)]));
605 
606  fft_buffer_aux2__[iz + i * local_size_z_] = double_complex(0, -0.5) *
607  (fftw_buffer_xy_[tid][z_col_pos_(i, 0)] - std::conj(fftw_buffer_xy_[tid][z_col_pos_(i, 1)]));
608  }
609 
610  break;
611  }
612  default: {
613  TERMINATE("wrong direction");
614  }
615  }
616  }
617  }
618  }

812  int nc = gvec__.reduced() ? 2 : 1;
813 
814  /* get positions of z-columns in xy plane */
815  z_col_pos_ = mdarray<int, 2>(gvec__.num_zcol(), nc, memory_t::host, "FFT3D.z_col_pos_");

819  int y = grid().coord_by_gvec(gvec__.zcol(i).y, 1);
820  assert(x >= 0 && x < grid().size(0));

1016 /** \page ft_pw Fourier transform and plane wave normalization
1017  *
1018  * We use plane waves in two different cases: a) plane waves (or augmented plane waves in the case of APW+lo method)
1019  * as a basis for expanding Kohn-Sham wave functions and b) plane waves are used to expand charge density and
1020  * potential. When we are dealing with plane wave basis functions it is convenient to adopt the following
1021  * normalization:
1022  * \f[
1023  * \langle {\bf r} |{\bf G+k} \rangle = \frac{1}{\sqrt \Omega} e^{i{\bf (G+k)r}}
1024  * \f]
1025  * such that
1026  * \f[
1027  * \langle {\bf G+k} |{\bf G'+k} \rangle_{\Omega} = \delta_{{\bf GG'}}
1028  * \f]
1029  * in the unit cell. However, for the expansion of periodic functions such as density or potential, the following
1030  * convention is more appropriate:
1031  * \f[
1032  * \rho({\bf r}) = \sum_{\bf G} e^{i{\bf Gr}} \rho({\bf G})
1033  * \f]
1034  * where
1035  * \f[
1036  * \rho({\bf G}) = \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} \rho({\bf r}) d{\bf r} =
1037  * \frac{1}{\Omega} \sum_{{\bf r}_i} e^{-i{\bf Gr}_i} \rho({\bf r}_i) \frac{\Omega}{N} =
1038  * \frac{1}{N} \sum_{{\bf r}_i} e^{-i{\bf Gr}_i} \rho({\bf r}_i)
1039  * \f]
1040  * i.e. with such convention the plane-wave expansion coefficients are obtained with a normalized FFT.
1041  */
