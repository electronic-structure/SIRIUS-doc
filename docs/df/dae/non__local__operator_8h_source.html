56  Non_local_operator(Simulation_context const& ctx_)
57  : ctx__(ctx_)

59  PROFILE("sirius::Non_local_operator::Non_local_operator");
60  pu_ = this->ctx__.processing_unit();
61  auto& uc = this->ctx__.unit_cell();
62  packed_mtrx_offset_ = mdarray<int, 1>(uc.num_atoms());
63  packed_mtrx_size_ = 0;
64  for (int ia = 0; ia < uc.num_atoms(); ia++) {
65  int nbf = uc.atom(ia).mt_basis_size();
66  packed_mtrx_offset_(ia) = packed_mtrx_size_;
67  packed_mtrx_size_ += nbf * nbf;
68  }
69 
70  if (pu_ == GPU) {
71  packed_mtrx_offset_.allocate(memory_t::device);
72  packed_mtrx_offset_.template copy<memory_t::host, memory_t::device>();
73  }

100 template <>
101 inline void Non_local_operator<double_complex>::apply(int chunk__,
102  int ispn_block__,
103  Wave_functions& op_phi__,
104  int idx0__,
105  int n__,
106  Beta_projectors& beta_,
107  matrix<double_complex>& beta_phi__)
108 {
109  PROFILE("sirius::Non_local_operator::apply");
110 
111  if (is_null_) {
112  return;
113  }
114 
115  int jspn = ispn_block__ & 1;
116 
117  auto& beta_gk = beta_.pw_coeffs_a();
118  int num_gkvec_loc = beta_.num_gkvec_loc();
119  int nbeta = beta_.chunk(chunk__).num_beta_;
120 
121  if (static_cast<size_t>(nbeta * n__) > work_.size()) {
122  work_ = mdarray<double_complex, 1>(nbeta * n__);
123  if (pu_ == GPU) {
124  work_.allocate(memory_t::device);
125  }
126  }
127  /* compute O * <beta|phi> for atoms in a chunk */
128  #pragma omp parallel for
129  for (int i = 0; i < beta_.chunk(chunk__).num_atoms_; i++) {
130  /* number of beta functions for a given atom */
131  int nbf = beta_.chunk(chunk__).desc_(beta_desc_idx::nbf, i);
132  int offs = beta_.chunk(chunk__).desc_(beta_desc_idx::offset, i);
133  int ia = beta_.chunk(chunk__).desc_(beta_desc_idx::ia, i);
134  switch (pu_) {
135  case CPU: {
136  linalg<CPU>::gemm(0, 0, nbf, n__, nbf, op_.at<CPU>(packed_mtrx_offset_(ia), ispn_block__), nbf,
137  beta_phi__.at<CPU>(offs, 0), nbeta, work_.at<CPU>(offs), nbeta);
138 
139  break;
140  }
141  case GPU: {
142 #ifdef __GPU
143  linalg<GPU>::gemm(0, 0, nbf, n__, nbf, op_.at<GPU>(packed_mtrx_offset_(ia), ispn_block__), nbf,
144  beta_phi__.at<GPU>(offs, 0), nbeta, work_.at<GPU>(offs), nbeta, omp_get_thread_num());
145 #endif
146  break;
147  }
148  }
149  }
150 
151  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
152  switch (pu_) {
153  case CPU: {
154  linalg<CPU>::gemm(0, 0, num_gkvec_loc, n__, nbeta, linalg_const<double_complex>::one(), beta_gk.at<CPU>(),
155  num_gkvec_loc, work_.at<CPU>(), nbeta, linalg_const<double_complex>::one(),
156  op_phi__.pw_coeffs(jspn).prime().at<CPU>(0, idx0__),
157  op_phi__.pw_coeffs(jspn).prime().ld());
158  break;
159  }
160  case GPU: {
161 #ifdef __GPU
162  /* wait for previous zgemms */
163  #pragma omp parallel
164  acc::sync_stream(omp_get_thread_num());
165 
166  linalg<GPU>::gemm(0, 0, num_gkvec_loc, n__, nbeta, &linalg_const<double_complex>::one(), beta_gk.at<GPU>(),
167  beta_gk.ld(), work_.at<GPU>(), nbeta, &linalg_const<double_complex>::one(),
168  op_phi__.pw_coeffs(jspn).prime().at<GPU>(0, idx0__),
169  op_phi__.pw_coeffs(jspn).prime().ld());
170  acc::sync_stream(-1);
171 #endif
172  break;
173  }
174  }
175 }

177 template <>
178 inline void Non_local_operator<double>::apply(int chunk__,
179  int ispn_block__,
180  Wave_functions& op_phi__,
181  int idx0__,
182  int n__,
183  Beta_projectors& beta_,
184  matrix<double>& beta_phi__)
185 {
186  PROFILE("sirius::Non_local_operator::apply");
187 
188  if (is_null_) {
189  return;
190  }
191 
192  int jspn = ispn_block__ & 1;
193 
194  auto& beta_gk = beta_.pw_coeffs_a();
195  int num_gkvec_loc = beta_.num_gkvec_loc();
196  int nbeta = beta_.chunk(chunk__).num_beta_;
197 
198  if (static_cast<size_t>(nbeta * n__) > work_.size()) {
199  work_ = mdarray<double, 1>(nbeta * n__);
200  if (pu_ == GPU) {
201  work_.allocate(memory_t::device);
202  }
203  }
204 
205  /* compute O * <beta|phi> */
206  #pragma omp parallel for
207  for (int i = 0; i < beta_.chunk(chunk__).num_atoms_; i++) {
208  /* number of beta functions for a given atom */
209  int nbf = beta_.chunk(chunk__).desc_(beta_desc_idx::nbf, i);
210  int offs = beta_.chunk(chunk__).desc_(beta_desc_idx::offset, i);
211  int ia = beta_.chunk(chunk__).desc_(beta_desc_idx::ia, i);
212 
213  switch (pu_) {
214  case CPU: {
215  linalg<CPU>::gemm(0, 0, nbf, n__, nbf, op_.at<CPU>(packed_mtrx_offset_(ia), ispn_block__), nbf,
216  beta_phi__.at<CPU>(offs, 0), nbeta, work_.at<CPU>(offs), nbeta);
217  break;
218  }
219  case GPU: {
220 #ifdef __GPU
221  linalg<GPU>::gemm(0, 0, nbf, n__, nbf, op_.at<GPU>(packed_mtrx_offset_(ia), ispn_block__), nbf,
222  beta_phi__.at<GPU>(offs, 0), nbeta, work_.at<GPU>(offs), nbeta, omp_get_thread_num());
223  break;
224 #endif
225  }
226  }
227  }
228 
229  /* compute <G+k|beta> * O * <beta|phi> and add to op_phi */
230  switch (pu_) {
231  case CPU: {
232  linalg<CPU>::gemm(0, 0, 2 * num_gkvec_loc, n__, nbeta, 1.0, reinterpret_cast<double*>(beta_gk.at<CPU>()),
233  2 * num_gkvec_loc, work_.at<CPU>(), nbeta, 1.0,
234  reinterpret_cast<double*>(op_phi__.pw_coeffs(jspn).prime().at<CPU>(0, idx0__)),
235  2 * op_phi__.pw_coeffs(jspn).prime().ld());
236  break;
237  }
238  case GPU: {
239 #ifdef __GPU
240  /* wait for previous zgemms */
241  #pragma omp parallel
242  acc::sync_stream(omp_get_thread_num());
243 
244  linalg<GPU>::gemm(0, 0, 2 * num_gkvec_loc, n__, nbeta, &linalg_const<double>::one(),
245  reinterpret_cast<double*>(beta_gk.at<GPU>()), 2 * num_gkvec_loc, work_.at<GPU>(), nbeta,
246  &linalg_const<double>::one(),
247  reinterpret_cast<double*>(op_phi__.pw_coeffs(jspn).prime().at<GPU>(0, idx0__)),
248  2 * num_gkvec_loc);
249  acc::sync_stream(-1);
250 #endif
251  break;
252  }
253  }

256 template <typename T>
257 class D_operator : public Non_local_operator<T>
258 {
259  private:
260  void initialize()
261  {
262 
263  auto& uc = this->ctx__.unit_cell();
264 
265  for (int ia = 0; ia < uc.num_atoms(); ia++) {
266  int nbf = uc.atom(ia).mt_basis_size();
267  if (uc.atom(ia).type().spin_orbit_coupling()) {
268 
269  // the pseudo potential contains information about
270  // spin orbit coupling so we use a different formula
271  // Eq.19 PRB 71 115106 for calculating the D matrix
272 
273  // Note that the D matrices are stored and
274  // calculated in the up-down basis already not the
275  // (Veff,Bx,By,Bz) one.
276  for (int xi2 = 0; xi2 < nbf; xi2++) {
277  for (int xi1 = 0; xi1 < nbf; xi1++) {
278  int idx = xi2 * nbf + xi1;
279  for (int s = 0; s < 4; s++) {
280  this->op_(this->packed_mtrx_offset_(ia) + idx, s) =
281  type_wrapper<T>::bypass(uc.atom(ia).d_mtrx_so(xi1, xi2, s));
282  }
283  }
284  }
285  } else {
286  // No spin orbit coupling for this atom \f[D = D(V_{eff})
287  // I + D(B_x) \sigma_x + D(B_y) sigma_y + D(B_z)
288  // sigma_z\f] since the D matrices are calculated that
289  // way.
290  for (int xi2 = 0; xi2 < nbf; xi2++) {
291  for (int xi1 = 0; xi1 < nbf; xi1++) {
292  int idx = xi2 * nbf + xi1;
293  switch (this->ctx__.num_mag_dims()) {
294  case 3: {
295  double bx = uc.atom(ia).d_mtrx(xi1, xi2, 2);
296  double by = uc.atom(ia).d_mtrx(xi1, xi2, 3);
297  this->op_(this->packed_mtrx_offset_(ia) + idx, 2) =
298  type_wrapper<T>::bypass(double_complex(bx, -by));
299  this->op_(this->packed_mtrx_offset_(ia) + idx, 3) =
300  type_wrapper<T>::bypass(double_complex(bx, by));
301  }
302  case 1: {
303  double v = uc.atom(ia).d_mtrx(xi1, xi2, 0);
304  double bz = uc.atom(ia).d_mtrx(xi1, xi2, 1);
305  this->op_(this->packed_mtrx_offset_(ia) + idx, 0) = v + bz;
306  this->op_(this->packed_mtrx_offset_(ia) + idx, 1) = v - bz;
307  break;
308  }
309  case 0: {
310  this->op_(this->packed_mtrx_offset_(ia) + idx, 0) = uc.atom(ia).d_mtrx(xi1, xi2, 0);
311  break;
312  }
313  default: {
314  TERMINATE("wrong number of magnetic dimensions");
315  }
316  }
317  }
318  }
319  }
320  }
321 
322  if (this->ctx__.control().print_checksum_ && this->ctx__.comm().rank() == 0) {
323  auto cs = this->op_.checksum();
324  print_checksum("D_operator", cs);
325  }
326 
327  if (this->pu_ == GPU) {
328  this->op_.allocate(memory_t::device);
329  this->op_.template copy<memory_t::host, memory_t::device>();
330  }
331  }
332 
333  public:
334  D_operator(Simulation_context const& ctx_)
335  : Non_local_operator<T>(ctx_)
336  {
337  this->op_ = mdarray<T, 2>(this->packed_mtrx_size_, ctx_.num_mag_dims() + 1);
338  this->op_.zero();
339  /* D-matrix is complex in non-collinear case */
340  if (ctx_.num_mag_dims() == 3) {
341  assert((std::is_same<T, double_complex>::value));
342  }
343  initialize();
344  };

348 template <typename T>
349 class Q_operator : public Non_local_operator<T>
350 {
351  private:
352  void initialize()
353  {
354  auto& uc = this->ctx__.unit_cell();
355  for (int ia = 0; ia < uc.num_atoms(); ia++) {
356  int iat = uc.atom(ia).type().id();
357  if (!uc.atom_type(iat).augment()) {
358  continue;
359  }
360  int nbf = uc.atom(ia).mt_basis_size();
361  for (int xi2 = 0; xi2 < nbf; xi2++) {
362  for (int xi1 = 0; xi1 < nbf; xi1++) {
363  /* The ultra soft pseudo potential has spin orbit coupling incorporated to it, so we
364  need to rotate the Q matrix */
365  if (this->ctx__.unit_cell().atom_type(iat).spin_orbit_coupling()) {
366  /* this is nothing else than Eq.18 of Ref PRB 71, 115106 */
367  for (auto si = 0; si < 2; si++) {
368  for (auto sj = 0; sj < 2; sj++) {
369 
370  double_complex result(0, 0);
371 
372  for (int xi2p = 0; xi2p < nbf; xi2p++) {
373  if (uc.atom(ia).type().compare_index_beta_functions(xi2, xi2p)) {
374  for (int xi1p = 0; xi1p < nbf; xi1p++) {
375  /* The F coefficients are already "block diagonal" so we do a full
376  summation. We actually rotate the q_matrices only */
377  if (uc.atom(ia).type().compare_index_beta_functions(xi1, xi1p)) {
378  result += this->ctx__.augmentation_op(iat).q_mtrx(xi1p, xi2p) *
379  (uc.atom(ia).type().f_coefficients(xi1, xi1p, sj, 0) *
380  uc.atom(ia).type().f_coefficients(xi2p, xi2, 0, si) +
381  uc.atom(ia).type().f_coefficients(xi1, xi1p, sj, 1) *
382  uc.atom(ia).type().f_coefficients(xi2p, xi2, 1, si));
383  }
384  }
385  }
386  }
387 
388  /* the order of the index is important */
389  const int ind = (si == sj) ? si : sj + 2;
390  /* this gives
391  ind = 0 if si = up and sj = up
392  ind = 1 if si = sj = down
393  ind = 2 if si = down and sj = up
394  ind = 3 if si = up and sj = down */
395  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ind) =
396  type_wrapper<T>::bypass(result);
397  }
398  }
399  } else {
400  for (int ispn = 0; ispn < this->ctx__.num_spins(); ispn++) {
401  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, ispn) =
402  this->ctx__.augmentation_op(iat).q_mtrx(xi1, xi2);
403  }
404  }
405  }
406  }
407  }
408  if (this->ctx__.control().print_checksum_ && this->ctx__.comm().rank() == 0) {
409  auto cs = this->op_.checksum();
410  print_checksum("Q_operator", cs);
411  }
412 
413  if (this->pu_ == GPU) {
414  this->op_.allocate(memory_t::device);
415  this->op_.template copy<memory_t::host, memory_t::device>();
416  }
417  }
418 
419  public:
420  Q_operator(Simulation_context const& ctx_)
421  : Non_local_operator<T>(ctx_)
422  {
423  /* Q-operator is independent of spin if there is no spin-orbit; however, it simplifies the apply()
424  * method if the Q-operator has a spin index */
425  this->op_ = mdarray<T, 2>(this->packed_mtrx_size_, ctx_.num_mag_dims() + 1);
426  this->op_.zero();
427  initialize();
428  }
429 };

432 class P_operator : public Non_local_operator<T>
433 {
434  public:
435  P_operator(Simulation_context const& ctx_, mdarray<double_complex, 3>& p_mtrx__)
436  : Non_local_operator<T>(ctx_)
437  {

446  continue;

448  int nbf = uc.atom(ia).mt_basis_size();
449  for (int xi2 = 0; xi2 < nbf; xi2++) {
450  for (int xi1 = 0; xi1 < nbf; xi1++) {
451  this->op_(this->packed_mtrx_offset_(ia) + xi2 * nbf + xi1, 0) = -p_mtrx__(xi1, xi2, iat).real();
