44template <typename T>
45void
46Band::diag_pseudo_potential_exact(int ispn__, Hamiltonian_k<real_type<T>>& Hk__) const
47{
48 PROFILE("sirius::Band::diag_pseudo_potential_exact");
49
50 auto& kp = Hk__.kp();
51
52 if (ctx_.gamma_point()) {
53 TERMINATE("exact diagonalization for Gamma-point case is not implemented");
54 }
55
56 const int bs = ctx_.cyclic_block_size();
57 dmatrix<T> hmlt(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
58 dmatrix<T> ovlp(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
59 dmatrix<T> evec(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
60 std::vector<real_type<T>> eval(kp.num_gkvec());
61
62 hmlt.zero();
63 ovlp.zero();
64
65 auto& gen_solver = ctx_.gen_evp_solver();
66
67 for (int ig = 0; ig < kp.num_gkvec(); ig++) {
68 hmlt.set(ig, ig, 0.5 * std::pow(kp.gkvec().template gkvec_cart<index_domain_t::global>(ig).length(), 2));
69 ovlp.set(ig, ig, 1);
70 }
71
72 auto veff = Hk__.H0().potential().effective_potential().gather_f_pw();
73 std::vector<double_complex> beff;
74 if (ctx_.num_mag_dims() == 1) {
75 beff = Hk__.H0().potential().effective_magnetic_field(0).gather_f_pw();
76 for (int ig = 0; ig < ctx_.gvec().num_gvec(); ig++) {
77 auto z1 = veff[ig];
78 auto z2 = beff[ig];
79 veff[ig] = z1 + z2;
80 beff[ig] = z1 - z2;
81 }
82 }
83
84 #pragma omp parallel for schedule(static)
85 for (int igk_col = 0; igk_col < kp.num_gkvec_col(); igk_col++) {
86 int ig_col = kp.igk_col(igk_col);
87 auto gvec_col = kp.gkvec().gvec(ig_col);
88 for (int igk_row = 0; igk_row < kp.num_gkvec_row(); igk_row++) {
89 int ig_row = kp.igk_row(igk_row);
90 auto gvec_row = kp.gkvec().gvec(ig_row);
91 auto ig12 = ctx_.gvec().index_g12_safe(gvec_row, gvec_col);
92
93 if (ispn__ == 0) {
94 if (ig12.second) {
95 hmlt(igk_row, igk_col) += std::conj(veff[ig12.first]);
96 } else {
97 hmlt(igk_row, igk_col) += veff[ig12.first];
98 }
99 } else {
100 if (ig12.second) {
101 hmlt(igk_row, igk_col) += std::conj(beff[ig12.first]);
102 } else {
103 hmlt(igk_row, igk_col) += beff[ig12.first];
104 }
105 }
106 }
107 }
108
109 auto& Dop = Hk__.H0().D();
110 auto& Qop = Hk__.H0().Q();
111
112 sddk::mdarray<T, 2> dop(ctx_.unit_cell().max_mt_basis_size(), ctx_.unit_cell().max_mt_basis_size());
113 sddk::mdarray<T, 2> qop(ctx_.unit_cell().max_mt_basis_size(), ctx_.unit_cell().max_mt_basis_size());
114
115 sddk::mdarray<T, 2> btmp(kp.num_gkvec_row(), ctx_.unit_cell().max_mt_basis_size());
116
117 kp.beta_projectors_row().prepare();
118 kp.beta_projectors_col().prepare();
119 for (int ichunk = 0; ichunk < kp.beta_projectors_row().num_chunks(); ichunk++) {
120 /* generate beta-projectors for a block of atoms */
121 kp.beta_projectors_row().generate(ichunk);
122 kp.beta_projectors_col().generate(ichunk);
123
124 auto& beta_row = kp.beta_projectors_row().pw_coeffs_a();
125 auto& beta_col = kp.beta_projectors_col().pw_coeffs_a();
126
127 for (int i = 0; i < kp.beta_projectors_row().chunk(ichunk).num_atoms_; i++) {
128 /* number of beta functions for a given atom */
129 int nbf = kp.beta_projectors_row().chunk(ichunk).desc_(static_cast<int>(beta_desc_idx::nbf), i);
130 int offs = kp.beta_projectors_row().chunk(ichunk).desc_(static_cast<int>(beta_desc_idx::offset), i);
131 int ia = kp.beta_projectors_row().chunk(ichunk).desc_(static_cast<int>(beta_desc_idx::ia), i);
132
133 for (int xi1 = 0; xi1 < nbf; xi1++) {
134 for (int xi2 = 0; xi2 < nbf; xi2++) {
135 dop(xi1, xi2) = Dop.template value<T>(xi1, xi2, ispn__, ia);
136 qop(xi1, xi2) = Qop.template value<T>(xi1, xi2, ispn__, ia);
137 }
138 }
139 /* compute <G+k|beta> D */
140 linalg(linalg_t::blas).gemm('N', 'N', kp.num_gkvec_row(), nbf, nbf,
141 &linalg_const<T>::one(), &beta_row(0, offs), beta_row.ld(), &dop(0, 0), dop.ld(),
142 &linalg_const<T>::zero(), &btmp(0, 0), btmp.ld());
143 /* compute (<G+k|beta> D ) <beta|G+k> */
144 linalg(linalg_t::blas).gemm('N', 'C', kp.num_gkvec_row(), kp.num_gkvec_col(), nbf,
145 &linalg_const<T>::one(), &btmp(0, 0), btmp.ld(), &beta_col(0, offs), beta_col.ld(),
146 &linalg_const<T>::one(), &hmlt(0, 0), hmlt.ld());
147 /* update the overlap matrix */
148 if (ctx_.unit_cell().atom(ia).type().augment()) {
149 linalg(linalg_t::blas).gemm('N', 'N', kp.num_gkvec_row(), nbf, nbf,
150 &linalg_const<T>::one(), &beta_row(0, offs), beta_row.ld(), &qop(0, 0), qop.ld(),
151 &linalg_const<T>::zero(), &btmp(0, 0), btmp.ld());
152 linalg(linalg_t::blas).gemm('N', 'C', kp.num_gkvec_row(), kp.num_gkvec_col(), nbf,
153 &linalg_const<T>::one(), &btmp(0, 0), btmp.ld(), &beta_col(0, offs), beta_col.ld(),
154 &linalg_const<T>::one(), &ovlp(0, 0), ovlp.ld());
155 }
156 } // i (atoms in chunk)
157 }
158 kp.beta_projectors_row().dismiss();
159 kp.beta_projectors_col().dismiss();
160
161 if (ctx_.cfg().control().verification() >= 1) {
162 double max_diff = check_hermitian(ovlp, kp.num_gkvec());
163 if (max_diff > 1e-12) {
164 std::stringstream s;
165 s << "overlap matrix is not hermitian, max_err = " << max_diff;
166 TERMINATE(s);
167 }
168 max_diff = check_hermitian(hmlt, kp.num_gkvec());
169 if (max_diff > 1e-12) {
170 std::stringstream s;
171 s << "Hamiltonian matrix is not hermitian, max_err = " << max_diff;
172 TERMINATE(s);
173 }
174 }
175 if (ctx_.cfg().control().verification() >= 2) {
176 ctx_.message(1, __function_name__, "%s", "checking eigen-values of S-matrix\n");
177
178 dmatrix<T> ovlp1(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
179 dmatrix<T> evec(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
180
181 ovlp >> ovlp1;
182
183 std::vector<real_type<T>> eo(kp.num_gkvec());
184
185 auto solver = Eigensolver_factory("scalapack", nullptr);
186 solver->solve(kp.num_gkvec(), ovlp1, eo.data(), evec);
187
188 for (int i = 0; i < kp.num_gkvec(); i++) {
189 if (eo[i] < 1e-6) {
190 ctx_.message(1, __function_name__, "small eigen-value: %18.10f\n", eo[i]);
191 }
192 }
193 }
194
195 if (gen_solver.solve(kp.num_gkvec(), ctx_.num_bands(), hmlt, ovlp, eval.data(), evec)) {
196 std::stringstream s;
197 s << "error in full diagonalziation";
198 TERMINATE(s);
199 }
200
201 for (int j = 0; j < ctx_.num_bands(); j++) {
202 kp.band_energy(j, ispn__, eval[j]);
203 }
204
205 kp.spinor_wave_functions().pw_coeffs(ispn__).remap_from(evec, 0);
206}

208template <typename T>
209sddk::mdarray<real_type<T>, 1>
210Band::diag_S_davidson(Hamiltonian_k<real_type<T>>& Hk__) const
211{
212 PROFILE("sirius::Band::diag_S_davidson");
213
214 auto& kp = Hk__.kp();
215
216 auto& itso = ctx_.cfg().iterative_solver();
217
218 double iterative_solver_tolerance = 1e-12;
219
220 /* true if this is a non-collinear case */
221 const bool nc_mag = (ctx_.num_mag_dims() == 3);
222
223 /* number of spin components, treated simultaneously
224 * 1 - in case of non-magnetic or collinear calculation
225 * 2 - in case of non-collinear calculation
226 */
227 const int num_sc = nc_mag ? 2 : 1;
228
229 /* number of eigen-vectors to find */
230 const int nevec = 1;
231
232 /* maximum subspace size */
233 int num_phi = itso.subspace_size() * nevec;
234
235 if (num_phi > kp.num_gkvec()) {
236 std::stringstream s;
237 s << "subspace size is too large!";
238 TERMINATE(s);
239 }
240 /* alias for memory pool */
241 auto& mp = ctx_.mem_pool(ctx_.host_memory_t());
242
243 /* eigen-vectors */
244 Wave_functions<real_type<T>> psi(mp, kp.gkvec_partition(), nevec, ctx_.aux_preferred_memory_t(), num_sc);
245 for (int i = 0; i < nevec; i++) {
246 for (int ispn = 0; ispn < num_sc; ispn++) {
247 for (int igk_loc = 0; igk_loc < kp.num_gkvec_loc(); igk_loc++) {
248 /* global index of G+k vector */
249 int igk = kp.idxgk(igk_loc);
250 if (igk == i + 1) {
251 psi.pw_coeffs(ispn).prime(igk_loc, i) = 1.0;
252 }
253 if (igk == i + 2) {
254 psi.pw_coeffs(ispn).prime(igk_loc, i) = 0.5;
255 }
256 if (igk == i + 3) {
257 psi.pw_coeffs(ispn).prime(igk_loc, i) = 0.25;
258 }
259 if (igk == i + 4) {
260 psi.pw_coeffs(ispn).prime(igk_loc, i) = 0.125;
261 }
262 }
263 }
264 }
265 std::vector<double> tmp(4096);
266 for (int i = 0; i < 4096; i++) {
267 tmp[i] = utils::random<double>();
268 }
269 #pragma omp parallel for schedule(static)
270 for (int i = 0; i < nevec; i++) {
271 for (int ispn = 0; ispn < num_sc; ispn++) {
272 for (int igk_loc = kp.gkvec().skip_g0(); igk_loc < kp.num_gkvec_loc(); igk_loc++) {
273 /* global index of G+k vector */
274 int igk = kp.idxgk(igk_loc);
275 psi.pw_coeffs(ispn).prime(igk_loc, i) += tmp[igk & 0xFFF] * 1e-5;
276 }
277 }
278 }
279
280 /* auxiliary wave-functions */
281 Wave_functions<real_type<T>> phi(mp, kp.gkvec_partition(), num_phi, ctx_.aux_preferred_memory_t(), num_sc);
282
283 /* S operator, applied to auxiliary wave-functions */
284 Wave_functions<real_type<T>> sphi(mp, kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t(), num_sc);
285
286 /* S operator, applied to new Psi wave-functions */
287 Wave_functions<real_type<T>> spsi(mp, kp.gkvec_partition(), nevec, ctx_.preferred_memory_t(), num_sc);
288
289 /* residuals */
290 Wave_functions<real_type<T>> res(mp, kp.gkvec_partition(), nevec, ctx_.preferred_memory_t(), num_sc);
291
292 const int bs = ctx_.cyclic_block_size();
293
294 dmatrix<T> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
295 dmatrix<T> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
296 dmatrix<T> ovlp_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
297
298 if (is_device_memory(ctx_.aux_preferred_memory_t())) {
299 auto& mpd = ctx_.mem_pool(memory_t::device);
300 for (int ispn = 0; ispn < num_sc; ispn++) {
301 phi.pw_coeffs(ispn).allocate(mpd);
302 }
303 }
304
305 if (is_device_memory(ctx_.preferred_memory_t())) {
306 auto& mpd = ctx_.mem_pool(memory_t::device);
307 for (int ispn = 0; ispn < num_sc; ispn++) {
308 psi.pw_coeffs(ispn).allocate(mpd);
309 psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, nevec);
310 }
311
312 for (int i = 0; i < num_sc; i++) {
313 res.pw_coeffs(i).allocate(mpd);
314 sphi.pw_coeffs(i).allocate(mpd);
315 spsi.pw_coeffs(i).allocate(mpd);
316 }
317
318 if (ctx_.blacs_grid().comm().size() == 1) {
319 evec.allocate(mpd);
320 ovlp.allocate(mpd);
321 }
322 }
323
324 /* allocate memory for the hubbard orbitals on device */
325 Hk__.kp().copy_hubbard_orbitals_on_device();
326
327 auto o_diag = Hk__.template get_h_o_diag_pw<T, 2>().second;
328
329 mdarray<real_type<T>, 2> o_diag1(kp.num_gkvec_loc(), num_sc);
330 for (int ispn = 0; ispn < num_sc; ispn++) {
331 for (int ig = 0; ig < kp.num_gkvec_loc(); ig++) {
332 o_diag1(ig, ispn) = 1.0;
333 }
334 }
335 if (ctx_.processing_unit() == device_t::GPU) {
336 o_diag1.allocate(memory_t::device).copy_to(memory_t::device);
337 }
338
339 auto& std_solver = ctx_.std_evp_solver();
340
341 for (int ispn = 0; ispn < num_sc; ispn++) {
342 /* trial basis functions */
343 phi.copy_from(psi, nevec, ispn, 0, ispn, 0);
344 }
345
346 /* current subspace size */
347 int N{0};
348
349 /* number of newly added basis functions */
350 int n = nevec;
351
352 mdarray<real_type<T>, 1> eval(nevec);
353 mdarray<real_type<T>, 1> eval_old(nevec);
354 eval_old = [](){return 1e10;};
355
356 /* tolerance for the norm of L2-norms of the residuals, used for
357 * relative convergence criterion. We can only compute this after
358 * we have the first residual norms available */
359 double relative_frobenius_tolerance{0};
360 double current_frobenius_norm{0};
361
362 for (int k = 0; k < itso.num_steps(); k++) {
363
364 /* apply Hamiltonian and S operators to the basis functions */
365 Hk__.template apply_h_s<T>(spin_range(nc_mag ? 2 : 0), N, n, phi, nullptr, &sphi);
366
367 orthogonalize<T>(ctx_.spla_context(), ctx_.preferred_memory_t(), ctx_.blas_linalg_t(),
368 spin_range(nc_mag ? 2 : 0), phi, sphi, N, n, ovlp, res);
369
370 /* setup eigen-value problem
371 * N is the number of previous basis functions
372 * n is the number of new basis functions */
373 set_subspace_mtrx<T, T>(N, n, 0, phi, sphi, ovlp, &ovlp_old);
374 if (ctx_.cfg().control().verification() >= 2 && ctx_.verbosity() >= 2) {
375 ovlp.serialize("<i|S|j> subspace matrix", N + n);
376 }
377
378 /* increase size of the variation space */
379 N += n;
380
381 eval >> eval_old;
382
383 /* solve standard eigen-value problem with the size N */
384 if (std_solver.solve(N, nevec, ovlp, &eval[0], evec)) {
385 std::stringstream s;
386 s << "error in diagonalziation";
387 TERMINATE(s);
388 }
389
390 kp.message(3, __function_name__, "step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
391 for (int i = 0; i < nevec; i++) {
392 kp.message(4, __function_name__, "eval[%i]=%20.16f, diff=%20.16f\n", i, eval[i], std::abs(eval[i] - eval_old[i]));
393 }
394
395 /* don't compute residuals on last iteration */
396 if (k != itso.num_steps() - 1) {
397 if (ctx_.processing_unit() == device_t::GPU) {
398 o_diag.allocate(memory_t::device).copy_to(memory_t::device);
399 o_diag1.allocate(memory_t::device).copy_to(memory_t::device);
400 }
401
402 /* get new preconditionined residuals, and also opsi and psi as a by-product */
403 auto result = sirius::residuals<T>(ctx_, ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), spin_range(nc_mag ? 2 : 0),
404 N, nevec, 0, eval, evec, sphi, phi, spsi, psi, res, o_diag, o_diag1,
405 itso.converge_by_energy(), itso.residual_tolerance(),
406 [&](int i, int ispn){return std::abs(eval[i] - eval_old[i]) < iterative_solver_tolerance;});
407 n = result.unconverged_residuals;
408 current_frobenius_norm = result.frobenius_norm;
409
410 /* set the relative tolerance convergence criterion */
411 if (k == 0) {
412 relative_frobenius_tolerance = current_frobenius_norm * itso.relative_tolerance();
413 }
414 }
415
416 /* verify convergence criteria */
417 bool converged_by_relative_tol = k > 0 && current_frobenius_norm < relative_frobenius_tolerance ;
418 bool converged_by_absolute_tol = n <= itso.min_num_res();
419 bool converged = converged_by_absolute_tol || converged_by_relative_tol;
420
421 /* check if running out of space */
422 bool should_restart = N + n > num_phi;
423
424 bool last_iteration = k == (itso.num_steps() - 1);
425
426 /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
427 if (should_restart || converged || last_iteration) {
428 /* recompute wave-functions */
429 /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
430 transform<T, T>(ctx_.spla_context(), nc_mag ? 2 : 0, phi, 0, N, evec, 0, 0, psi, 0, nevec);
431
432 /* exit the loop if the eigen-vectors are converged or this is a last iteration */
433 if (converged || last_iteration) {
434 break;
435 } else { /* otherwise, set Psi as a new trial basis */
436 kp.message(3, __function_name__, "%s", "subspace size limit reached\n");
437
438 if (itso.converge_by_energy()) {
439 transform<T, T>(ctx_.spla_context(), nc_mag ? 2 : 0, sphi, 0, N, evec, 0, 0, spsi, 0, nevec);
440 }
441
442 ovlp_old.zero();
443 for (int i = 0; i < nevec; i++) {
444 ovlp_old.set(i, i, eval[i]);
445 }
446 /* update basis functions */
447 for (int ispn = 0; ispn < num_sc; ispn++) {
448 phi.copy_from(ctx_.processing_unit(), nevec, psi, ispn, 0, ispn, 0);
449 sphi.copy_from(ctx_.processing_unit(), nevec, spsi, ispn, 0, ispn, 0);
450 }
451 /* number of basis functions that we already have */
452 N = nevec;
453 }
454 }
455 for (int ispn = 0; ispn < num_sc; ispn++) {
456 /* expand variational subspace with new basis vectors obtatined from residuals */
457 phi.copy_from(ctx_.processing_unit(), n, res, ispn, 0, ispn, N);
458 }
459 }
460
461 Hk__.kp().release_hubbard_orbitals_on_device();
462
463 return eval;
464}
