90 template <typename T>
91 void
92 Band::diag_pseudo_potential_exact(int ispn__, Hamiltonian_k& Hk__) const
93 {
94  PROFILE("sirius::Band::diag_pseudo_potential_exact");
95 
96  auto& kp = Hk__.kp();
97 
98  if (ctx_.gamma_point()) {
99  TERMINATE("exact diagonalization for Gamma-point case is not implemented");
100  }
101 
102  const int bs = ctx_.cyclic_block_size();
103  dmatrix<T> hmlt(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
104  dmatrix<T> ovlp(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
105  dmatrix<T> evec(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
106  std::vector<double> eval(kp.num_gkvec());
107 
108  hmlt.zero();
109  ovlp.zero();
110 
111  auto& gen_solver = ctx_.gen_evp_solver();
112 
113  for (int ig = 0; ig < kp.num_gkvec(); ig++) {
114  hmlt.set(ig, ig, 0.5 * std::pow(kp.gkvec().gkvec_cart<index_domain_t::global>(ig).length(), 2));
115  ovlp.set(ig, ig, 1);
116  }
117 
118  auto veff = Hk__.H0().potential().effective_potential().gather_f_pw();
119  std::vector<double_complex> beff;
120  if (ctx_.num_mag_dims() == 1) {
121  beff = Hk__.H0().potential().effective_magnetic_field(0).gather_f_pw();
122  for (int ig = 0; ig < ctx_.gvec().num_gvec(); ig++) {
123  auto z1 = veff[ig];
124  auto z2 = beff[ig];
125  veff[ig] = z1 + z2;
126  beff[ig] = z1 - z2;
127  }
128  }
129 
130  #pragma omp parallel for schedule(static)
131  for (int igk_col = 0; igk_col < kp.num_gkvec_col(); igk_col++) {
132  int ig_col = kp.igk_col(igk_col);
133  auto gvec_col = kp.gkvec().gvec(ig_col);
134  for (int igk_row = 0; igk_row < kp.num_gkvec_row(); igk_row++) {
135  int ig_row = kp.igk_row(igk_row);
136  auto gvec_row = kp.gkvec().gvec(ig_row);
137  auto ig12 = ctx_.gvec().index_g12_safe(gvec_row, gvec_col);
138 
139  if (ispn__ == 0) {
140  if (ig12.second) {
141  hmlt(igk_row, igk_col) += std::conj(veff[ig12.first]);
142  } else {
143  hmlt(igk_row, igk_col) += veff[ig12.first];
144  }
145  } else {
146  if (ig12.second) {
147  hmlt(igk_row, igk_col) += std::conj(beff[ig12.first]);
148  } else {
149  hmlt(igk_row, igk_col) += beff[ig12.first];
150  }
151  }
152  }
153  }
154 
155  auto& Dop = Hk__.H0().D();
156  auto& Qop = Hk__.H0().Q();
157 
158  mdarray<double_complex, 2> dop(ctx_.unit_cell().max_mt_basis_size(), ctx_.unit_cell().max_mt_basis_size());
159  mdarray<double_complex, 2> qop(ctx_.unit_cell().max_mt_basis_size(), ctx_.unit_cell().max_mt_basis_size());
160 
161  mdarray<double_complex, 2> btmp(kp.num_gkvec_row(), ctx_.unit_cell().max_mt_basis_size());
162 
163  kp.beta_projectors_row().prepare();
164  kp.beta_projectors_col().prepare();
165  for (int ichunk = 0; ichunk < kp.beta_projectors_row().num_chunks(); ichunk++) {
166  /* generate beta-projectors for a block of atoms */
167  kp.beta_projectors_row().generate(ichunk);
168  kp.beta_projectors_col().generate(ichunk);
169 
170  auto& beta_row = kp.beta_projectors_row().pw_coeffs_a();
171  auto& beta_col = kp.beta_projectors_col().pw_coeffs_a();
172 
173  for (int i = 0; i < kp.beta_projectors_row().chunk(ichunk).num_atoms_; i++) {
174  /* number of beta functions for a given atom */
175  int nbf = kp.beta_projectors_row().chunk(ichunk).desc_(static_cast<int>(beta_desc_idx::nbf), i);
176  int offs = kp.beta_projectors_row().chunk(ichunk).desc_(static_cast<int>(beta_desc_idx::offset), i);
177  int ia = kp.beta_projectors_row().chunk(ichunk).desc_(static_cast<int>(beta_desc_idx::ia), i);
178 
179  for (int xi1 = 0; xi1 < nbf; xi1++) {
180  for (int xi2 = 0; xi2 < nbf; xi2++) {
181  dop(xi1, xi2) = Dop.value<T>(xi1, xi2, ispn__, ia);
182  qop(xi1, xi2) = Qop.value<T>(xi1, xi2, ispn__, ia);
183  }
184  }
185  /* compute <G+k|beta> D */
186  linalg(linalg_t::blas).gemm('N', 'N', kp.num_gkvec_row(), nbf, nbf,
187  &linalg_const<T>::one(), &beta_row(0, offs), beta_row.ld(), &dop(0, 0), dop.ld(),
188  &linalg_const<T>::zero(), &btmp(0, 0), btmp.ld());
189  /* compute (<G+k|beta> D ) <beta|G+k> */
190  linalg(linalg_t::blas).gemm('N', 'C', kp.num_gkvec_row(), kp.num_gkvec_col(), nbf,
191  &linalg_const<double_complex>::one(), &btmp(0, 0), btmp.ld(), &beta_col(0, offs), beta_col.ld(),
192  &linalg_const<double_complex>::one(), &hmlt(0, 0), hmlt.ld());
193  /* update the overlap matrix */
194  if (ctx_.unit_cell().atom(ia).type().augment()) {
195  linalg(linalg_t::blas).gemm('N', 'N', kp.num_gkvec_row(), nbf, nbf,
196  &linalg_const<T>::one(), &beta_row(0, offs), beta_row.ld(), &qop(0, 0), qop.ld(),
197  &linalg_const<T>::zero(), &btmp(0, 0), btmp.ld());
198  linalg(linalg_t::blas).gemm('N', 'C', kp.num_gkvec_row(), kp.num_gkvec_col(), nbf,
199  &linalg_const<double_complex>::one(), &btmp(0, 0), btmp.ld(), &beta_col(0, offs), beta_col.ld(),
200  &linalg_const<double_complex>::one(), &ovlp(0, 0), ovlp.ld());
201  }
202  } // i (atoms in chunk)
203  }
204  kp.beta_projectors_row().dismiss();
205  kp.beta_projectors_col().dismiss();
206 
207  if (ctx_.cfg().control().verification() >= 1) {
208  double max_diff = check_hermitian(ovlp, kp.num_gkvec());
209  if (max_diff > 1e-12) {
210  std::stringstream s;
211  s << "overlap matrix is not hermitian, max_err = " << max_diff;
212  TERMINATE(s);
213  }
214  max_diff = check_hermitian(hmlt, kp.num_gkvec());
215  if (max_diff > 1e-12) {
216  std::stringstream s;
217  s << "Hamiltonian matrix is not hermitian, max_err = " << max_diff;
218  TERMINATE(s);
219  }
220  }
221  if (ctx_.cfg().control().verification() >= 2) {
222  ctx_.message(1, __function_name__, "%s", "checking eigen-values of S-matrix\n");
223 
224  dmatrix<T> ovlp1(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
225  dmatrix<T> evec(kp.num_gkvec(), kp.num_gkvec(), ctx_.blacs_grid(), bs, bs);
226 
227  ovlp >> ovlp1;
228 
229  std::vector<double> eo(kp.num_gkvec());
230 
231  auto solver = Eigensolver_factory("scalapack", nullptr);
232  solver->solve(kp.num_gkvec(), ovlp1, eo.data(), evec);
233 
234  for (int i = 0; i < kp.num_gkvec(); i++) {
235  if (eo[i] < 1e-6) {
236  ctx_.message(1, __function_name__, "small eigen-value: %18.10f\n", eo[i]);
237  }
238  }
239  }
240 
241  if (gen_solver.solve(kp.num_gkvec(), ctx_.num_bands(), hmlt, ovlp, eval.data(), evec)) {
242  std::stringstream s;
243  s << "error in full diagonalziation";
244  TERMINATE(s);
245  }
246 
247  for (int j = 0; j < ctx_.num_bands(); j++) {
248  kp.band_energy(j, ispn__, eval[j]);
249  }
250 
251  kp.spinor_wave_functions().pw_coeffs(ispn__).remap_from(evec, 0);
252 }

254 template <typename T>
255 int
256 Band::diag_pseudo_potential_davidson(Hamiltonian_k& Hk__) const
257 {
258  PROFILE("sirius::Band::diag_pseudo_potential_davidson");
259 
260  auto& kp = Hk__.kp();
261 
262  ctx_.print_memory_usage(__FILE__, __LINE__);
263 
264  auto& itso = ctx_.cfg().iterative_solver();
265 
266  bool converge_by_energy = (itso.converge_by_energy() == 1);
267 
268  /* true if this is a non-collinear case */
269  const bool nc_mag = (ctx_.num_mag_dims() == 3);
270 
271  /* number of spin components, treated simultaneously
272  * 1 - in case of non-magnetic or collinear calculation
273  * 2 - in case of non-collinear calculation
274  */
275  const int num_sc = nc_mag ? 2 : 1;
276 
277  /* short notation for number of target wave-functions */
278  const int num_bands = ctx_.num_bands();
279 
280  /* short notation for target wave-functions */
281  auto& psi = kp.spinor_wave_functions();
282 
283  /* maximum subspace size */
284  int num_phi = itso.subspace_size() * num_bands;
285 
286  if (num_phi > kp.num_gkvec()) {
287  std::stringstream s;
288  s << "subspace size is too large!";
289  TERMINATE(s);
290  }
291  /* alias for memory pool */
292  auto& mp = ctx_.mem_pool(ctx_.host_memory_t());
293 
294  /* allocate wave-functions */
295 
296  /* auxiliary wave-functions */
297  Wave_functions phi(mp, kp.gkvec_partition(), num_phi, ctx_.aux_preferred_memory_t(), num_sc);
298 
299  /* Hamiltonian, applied to auxiliary wave-functions */
300  Wave_functions hphi(mp, kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t(), num_sc);
301 
302  /* S operator, applied to auxiliary wave-functions */
303  Wave_functions sphi(mp, kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t(), num_sc);
304 
305  /* Hamiltonian, applied to new Psi wave-functions, plus some extra space */
306  Wave_functions hpsi(mp, kp.gkvec_partition(), num_bands, ctx_.preferred_memory_t(), num_sc);
307 
308  /* S operator, applied to new Psi wave-functions */
309  Wave_functions spsi(mp, kp.gkvec_partition(), num_bands, ctx_.preferred_memory_t(), num_sc);
310 
311  /* residuals */
312  Wave_functions res(mp, kp.gkvec_partition(), num_bands, ctx_.preferred_memory_t(), num_sc);
313 
314  const int bs = ctx_.cyclic_block_size();
315 
316  dmatrix<T> hmlt(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
317  dmatrix<T> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
318  dmatrix<T> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
319  dmatrix<T> hmlt_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
320  dmatrix<T> ovlp_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
321 
322  if (is_device_memory(ctx_.aux_preferred_memory_t())) {
323  auto& mpd = ctx_.mem_pool(memory_t::device);
324  for (int i = 0; i < num_sc; i++) {
325  phi.pw_coeffs(i).allocate(mpd);
326  }
327  }
328 
329  if (is_device_memory(ctx_.preferred_memory_t())) {
330  auto& mpd = ctx_.mem_pool(memory_t::device);
331  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
332  psi.pw_coeffs(ispn).allocate(mpd);
333  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, num_bands);
334  }
335 
336  for (int i = 0; i < num_sc; i++) {
337  res.pw_coeffs(i).allocate(mpd);
338 
339  hphi.pw_coeffs(i).allocate(mpd);
340  sphi.pw_coeffs(i).allocate(mpd);
341 
342  hpsi.pw_coeffs(i).allocate(mpd);
343  spsi.pw_coeffs(i).allocate(mpd);
344  }
345 
346  if (ctx_.blacs_grid().comm().size() == 1) {
347  evec.allocate(mpd);
348  ovlp.allocate(mpd);
349  hmlt.allocate(mpd);
350  }
351  }
352 
353  kp.copy_hubbard_orbitals_on_device();
354 
355  ctx_.print_memory_usage(__FILE__, __LINE__);
356 
357  /* get diagonal elements for preconditioning */
358  auto h_o_diag = Hk__.get_h_o_diag_pw<T, 3>();
359 
360  if (ctx_.print_checksum()) {
361  auto cs1 = h_o_diag.first.checksum();
362  auto cs2 = h_o_diag.second.checksum();
363  kp.comm().allreduce(&cs1, 1);
364  kp.comm().allreduce(&cs2, 1);
365  if (kp.comm().rank() == 0) {
366  utils::print_checksum("h_diag", cs1);
367  utils::print_checksum("o_diag", cs2);
368  }
369  }
370 
371  auto& std_solver = ctx_.std_evp_solver();
372 
373  if (ctx_.print_checksum()) {
374  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
375  auto cs = psi.checksum_pw(get_device_t(psi.preferred_memory_t()), ispn, 0, num_bands);
376  std::stringstream s;
377  s << "input spinor_wave_functions_" << ispn;
378  if (kp.comm().rank() == 0) {
379  utils::print_checksum(s.str(), cs);
380  }
381  }
382  }
383 
384  int niter{0};
385 
386  PROFILE_START("sirius::Band::diag_pseudo_potential_davidson|iter");
387  for (int ispin_step = 0; ispin_step < ctx_.num_spinors(); ispin_step++) {
388 
389  /* converged vectors */
390  int num_locked = 0;
391 
392  sddk::mdarray<double, 1> eval(num_bands);
393  sddk::mdarray<double, 1> eval_old(num_bands);
394  eval_old = [](){return 1e10;};
395 
396  /* check if band energy is converged */
397  auto is_converged = [&](int j__, int ispn__) -> bool
398  {
399  double tol = ctx_.iterative_solver_tolerance();
400  double empy_tol = std::max(tol * ctx_.cfg().settings().itsol_tol_ratio(), itso.empty_states_tolerance());
401  /* if band is empty, decrease the tolerance */
402  // note: j__ indexes the unconverged eigenpairs -- excluding locked ones.
403  if (std::abs(kp.band_occupancy(j__ + num_locked, ispn__)) < ctx_.min_occupancy() * ctx_.max_occupancy()) {
404  tol += empy_tol;
405  }
406  return std::abs(eval[j__] - eval_old[j__]) <= tol;
407  };
408 
409  if (itso.init_eval_old()) {
410  eval_old = [&](int64_t j) {return kp.band_energy(j, ispin_step);};
411  }
412 
413  /* trial basis functions */
414  for (int ispn = 0; ispn < num_sc; ispn++) {
415  phi.copy_from(psi, num_bands, nc_mag ? ispn : ispin_step, 0, ispn, 0);
416  }
417  if (ctx_.print_checksum()) {
418  for (int ispn = 0; ispn < num_sc; ispn++) {
419  auto cs = phi.checksum_pw(get_device_t(phi.preferred_memory_t()), ispn, 0, num_bands);
420  std::stringstream s;
421  s << "input phi" << ispn;
422  if (kp.comm().rank() == 0) {
423  utils::print_checksum(s.str(), cs);
424  }
425  }
426  }
427 
428  /* first phase: setup and diagonalize reduced Hamiltonian and get eigen-values;
429  * this is done before the main iterative loop */
430 
431  /* apply Hamiltonian and S operators to the basis functions */
432  Hk__.apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), 0, num_bands, phi, &hphi, &sphi);
433 
434  orthogonalize<T>(ctx_.spla_context(), ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), nc_mag ? 2 : 0, phi,
435  hphi, sphi, 0, num_bands, ovlp, res);
436 
437  /* setup eigen-value problem */
438  set_subspace_mtrx(0, num_bands, 0, phi, hphi, hmlt, &hmlt_old);
439 
440  if (ctx_.cfg().control().verification() >= 1) {
441  double max_diff = check_hermitian(hmlt, num_bands);
442  if (max_diff > 1e-12) {
443  std::stringstream s;
444  s << "H matrix is not Hermitian, max_err = " << max_diff;
445  WARNING(s);
446  }
447  }
448  //if (ctx_.control().verification_ >= 2 && ctx_.control().verbosity_ >= 2) {
449  // hmlt.serialize("H matrix", num_bands);
450  // ovlp.serialize("S matrix", num_bands);
451  //}
452 
453  /* current subspace size */
454  int N = num_bands;
455 
456  /* number of residuals to add to the search subspace.
457  the idea here is to expand the search subspace with
458  a number of vectors significantly small than num_bands,
459  such that we converge quickly to the first so many
460  vectors, which can then be locked at restart. */
461  //int block_size = std::max(1, itso.locking_ ? num_bands / 2 : num_bands);
462 
463  // Seems like a smaller block size is not always improving time to solution much,
464  // so keep it num_bands.
465  int block_size = num_bands;
466 
467  /* solve generalized eigen-value problem with the size N and get lowest num_bands eigen-vectors */
468  if (std_solver.solve(N, num_bands, hmlt, &eval[0], evec)) {
469  std::stringstream s;
470  s << "error in diagonalization";
471  TERMINATE(s);
472  }
473 
474  ctx_.evp_work_count(1);
475 
476  for (int i = 0; i < num_bands; i++) {
477  kp.message(4, __function_name__, "eval[%i]=%20.16f\n", i, eval[i]);
478  }
479 
480  /* number of newly added basis functions */
481  int num_unconverged{0};
482 
483  /* tolerance for the norm of L2-norms of the residuals, used for
484  * relative convergence criterion. We can only compute this after
485  * we have the first residual norms available */
486  double relative_frobenius_tolerance{0};
487  double current_frobenius_norm{0};
488 
489  /* second phase: start iterative diagonalization */
490  for (int k = 0; k < itso.num_steps(); k++) {
491  int num_lockable = 0;
492 
493  bool last_iteration = k == (itso.num_steps() - 1);
494 
495  int num_ritz = num_bands - num_locked;
496 
497  /* don't compute residuals on last iteration */
498  if (!last_iteration) {
499  /* get new preconditioned residuals, and also hpsi and opsi as a by-product */
500  auto result = sirius::residuals<T>(ctx_, ctx_.preferred_memory_t(), ctx_.blas_linalg_t(),
501  spin_range(nc_mag ? 2 : ispin_step), N, num_ritz, num_locked, eval, evec, hphi,
502  sphi, hpsi, spsi, res, h_o_diag.first, h_o_diag.second,
503  itso.converge_by_energy(), itso.residual_tolerance(), is_converged);
504 
505  num_unconverged = result.unconverged_residuals;
506  num_lockable = result.num_consecutive_smallest_converged;
507  current_frobenius_norm = result.frobenius_norm;
508 
509  /* set the relative tolerance convergence criterion */
510  if (k == 0) {
511  relative_frobenius_tolerance = current_frobenius_norm * itso.relative_tolerance();
512  }
513  }
514 
515  /* verify convergence criteria */
516  int num_converged = num_ritz - num_unconverged;

518  bool converged_by_absolute_tol = num_locked + num_converged + itso.min_num_res() >= num_bands;

561  break;

571  std::vector<Wave_functions*>({&hphi, &sphi}), num_locked, N - num_locked, evec, 0,
572  0, 0.0, {&hpsi, &spsi}, 0, num_ritz);
573  }
574

576  for (int ispn = 0; ispn < num_sc; ispn++) {
577  phi.copy_from( psi, keep - num_locked, nc_mag ? ispn : ispin_step, num_locked, nc_mag ? ispn : 0, num_locked);
578  hphi.copy_from(hpsi, keep - num_locked, ispn, 0, ispn, num_locked);
579  sphi.copy_from(spsi, keep - num_locked, ispn, 0, ispn, num_locked);
580  }
581

583  if (itso.locking() && num_lockable > 0) {
584  for (int i = num_lockable; i < num_ritz; ++i) {
585  eval[i - num_lockable] = eval[i];
586  }
587  }
588

590  hmlt_old.zero();
591  for (int i = 0; i < keep - num_locked; i++) {
592  hmlt_old.set(i, i, eval[i]);
593  }
594 
595  /* number of basis functions that we already have */
596  N = keep;
597 
598  // Only when we do orthogonalization we can lock vecs
599  if (itso.locking()) {
600  num_locked += num_lockable;
601  }
602  }
603  }
604

606  for (int ispn = 0; ispn < num_sc; ispn++) {

608  }

611  Hk__.apply_h_s<T>(spin_range(nc_mag ? 2 : ispin_step), N, expand_with, phi, &hphi, &sphi);
612

614

616  hphi, sphi, N, expand_with, ovlp, res);
617

619  * N is the number of previous basis functions
620  * expand_with is the number of new basis functions */

629  } else {
630  kp.message(1, __function_name__, "OK! H matrix of size %i is Hermitian\n", N + expand_with - num_locked);
631  }
632  }
633

642  if (std_solver.solve(N - num_locked, num_bands - num_locked, hmlt, &eval[0], evec)) {
643  std::stringstream s;
644  s << "error in diagonalization";
645  TERMINATE(s);
646  }
647 
648  ctx_.evp_work_count(std::pow(static_cast<double>(N - num_locked) / num_bands, 3));
649 
650  kp.message(2, __function_name__, "step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
651  for (int i = 0; i < num_bands - num_locked; i++) {
652  kp.message(4, __function_name__, "eval[%i]=%20.16f, diff=%20.16f, occ=%20.16f\n", i, eval[i],
653  std::abs(eval[i] - eval_old[i]), kp.band_occupancy(i, ispin_step));
654  }

656  }

694 sddk::mdarray<double, 1>
695 Band::diag_S_davidson(Hamiltonian_k& Hk__) const

700 
701  auto& itso = ctx_.cfg().iterative_solver();
702 
703  double iterative_solver_tolerance = 1e-12;
704 
705  /* true if this is a non-collinear case */
706  const bool nc_mag = (ctx_.num_mag_dims() == 3);
707 
708  /* number of spin components, treated simultaneously
709  * 1 - in case of non-magnetic or collinear calculation
710  * 2 - in case of non-collinear calculation
711  */
712  const int num_sc = nc_mag ? 2 : 1;
713 
714  /* number of eigen-vectors to find */
715  const int nevec = 1;
716 
717  /* maximum subspace size */
718  int num_phi = itso.subspace_size() * nevec;
719 
720  if (num_phi > kp.num_gkvec()) {
721  std::stringstream s;
722  s << "subspace size is too large!";
723  TERMINATE(s);
724  }
725  /* alias for memory pool */
726  auto& mp = ctx_.mem_pool(ctx_.host_memory_t());
727 
728  /* eigen-vectors */
729  Wave_functions psi(mp, kp.gkvec_partition(), nevec, ctx_.aux_preferred_memory_t(), num_sc);
730  for (int i = 0; i < nevec; i++) {
731  for (int ispn = 0; ispn < num_sc; ispn++) {
732  for (int igk_loc = 0; igk_loc < kp.num_gkvec_loc(); igk_loc++) {
733  /* global index of G+k vector */
734  int igk = kp.idxgk(igk_loc);
735  if (igk == i + 1) {
736  psi.pw_coeffs(ispn).prime(igk_loc, i) = 1.0;
737  }
738  if (igk == i + 2) {
739  psi.pw_coeffs(ispn).prime(igk_loc, i) = 0.5;
740  }
741  if (igk == i + 3) {
742  psi.pw_coeffs(ispn).prime(igk_loc, i) = 0.25;
743  }
744  if (igk == i + 4) {
745  psi.pw_coeffs(ispn).prime(igk_loc, i) = 0.125;
746  }
747  }
748  }
749  }
750  std::vector<double> tmp(4096);
751  for (int i = 0; i < 4096; i++) {
752  tmp[i] = utils::random<double>();
753  }
754  #pragma omp parallel for schedule(static)
755  for (int i = 0; i < nevec; i++) {
756  for (int ispn = 0; ispn < num_sc; ispn++) {
757  for (int igk_loc = kp.gkvec().skip_g0(); igk_loc < kp.num_gkvec_loc(); igk_loc++) {
758  /* global index of G+k vector */
759  int igk = kp.idxgk(igk_loc);
760  psi.pw_coeffs(ispn).prime(igk_loc, i) += tmp[igk & 0xFFF] * 1e-5;
761  }
762  }
763  }
764 
765  /* auxiliary wave-functions */
766  Wave_functions phi(mp, kp.gkvec_partition(), num_phi, ctx_.aux_preferred_memory_t(), num_sc);
767 
768  /* S operator, applied to auxiliary wave-functions */
769  Wave_functions sphi(mp, kp.gkvec_partition(), num_phi, ctx_.preferred_memory_t(), num_sc);
770 
771  /* S operator, applied to new Psi wave-functions */
772  Wave_functions spsi(mp, kp.gkvec_partition(), nevec, ctx_.preferred_memory_t(), num_sc);
773 
774  /* residuals */
775  Wave_functions res(mp, kp.gkvec_partition(), nevec, ctx_.preferred_memory_t(), num_sc);
776 
777  const int bs = ctx_.cyclic_block_size();
778 
779  dmatrix<T> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
780  dmatrix<T> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
781  dmatrix<T> ovlp_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs, mp);
782 
783  if (is_device_memory(ctx_.aux_preferred_memory_t())) {
784  auto& mpd = ctx_.mem_pool(memory_t::device);
785  for (int ispn = 0; ispn < num_sc; ispn++) {
786  phi.pw_coeffs(ispn).allocate(mpd);

788  }
789 
790  if (is_device_memory(ctx_.preferred_memory_t())) {
791  auto& mpd = ctx_.mem_pool(memory_t::device);
792  for (int ispn = 0; ispn < num_sc; ispn++) {
793  psi.pw_coeffs(ispn).allocate(mpd);
794  psi.pw_coeffs(ispn).copy_to(memory_t::device, 0, nevec);
795  }
796 
797  for (int i = 0; i < num_sc; i++) {
798  res.pw_coeffs(i).allocate(mpd);
799  sphi.pw_coeffs(i).allocate(mpd);
800  spsi.pw_coeffs(i).allocate(mpd);
801  }
802 
803  if (ctx_.blacs_grid().comm().size() == 1) {
804  evec.allocate(mpd);
805  ovlp.allocate(mpd);
806  }
807  }
808 
809  /* allocate memory for the hubbard orbitals on device */
810  Hk__.kp().copy_hubbard_orbitals_on_device();
811 
812  auto o_diag = Hk__.get_h_o_diag_pw<T, 2>().second;
813 
814  mdarray<double, 2> o_diag1(kp.num_gkvec_loc(), num_sc);
815  for (int ispn = 0; ispn < num_sc; ispn++) {
816  for (int ig = 0; ig < kp.num_gkvec_loc(); ig++) {
817  o_diag1(ig, ispn) = 1.0;
818  }
819  }
820  if (ctx_.processing_unit() == device_t::GPU) {
821  o_diag1.allocate(memory_t::device).copy_to(memory_t::device);
822  }
823 
824  auto& std_solver = ctx_.std_evp_solver();
825 
826  for (int ispn = 0; ispn < num_sc; ispn++) {
827  /* trial basis functions */
828  phi.copy_from(psi, nevec, ispn, 0, ispn, 0);
829  }
830 
831  /* current subspace size */
832  int N{0};
833 
834  /* number of newly added basis functions */
835  int n = nevec;
836 
837  mdarray<double, 1> eval(nevec);
838  mdarray<double, 1> eval_old(nevec);
839  eval_old = [](){return 1e10;};
840 
841  /* tolerance for the norm of L2-norms of the residuals, used for
842  * relative convergence criterion. We can only compute this after
843  * we have the first residual norms available */
844  double relative_frobenius_tolerance{0};
845  double current_frobenius_norm{0};
846 
847  for (int k = 0; k < itso.num_steps(); k++) {
848 
849  /* apply Hamiltonian and S operators to the basis functions */
850  Hk__.apply_h_s<T>(spin_range(nc_mag ? 2 : 0), N, n, phi, nullptr, &sphi);
851 
852  orthogonalize<T>(ctx_.spla_context(), ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), nc_mag ? 2 : 0, phi,
853  sphi, N, n, ovlp, res);
854 
855  /* setup eigen-value problem
856  * N is the number of previous basis functions
857  * n is the number of new basis functions */
858  set_subspace_mtrx(N, n, 0, phi, sphi, ovlp, &ovlp_old);
859  if (ctx_.cfg().control().verification() >= 2 && ctx_.verbosity() >= 2) {
860  ovlp.serialize("<i|S|j> subspace matrix", N + n);
861  }
862 
863  /* increase size of the variation space */
864  N += n;
865 
866  eval >> eval_old;
867 
868  /* solve standard eigen-value problem with the size N */
869  if (std_solver.solve(N, nevec, ovlp, &eval[0], evec)) {
870  std::stringstream s;
871  s << "error in diagonalziation";
872  TERMINATE(s);
873  }
874 
875  kp.message(3, __function_name__, "step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
876  for (int i = 0; i < nevec; i++) {
877  kp.message(4, __function_name__, "eval[%i]=%20.16f, diff=%20.16f\n", i, eval[i], std::abs(eval[i] - eval_old[i]));
878  }
879 
880  /* don't compute residuals on last iteration */
881  if (k != itso.num_steps() - 1) {
882  if (ctx_.processing_unit() == device_t::GPU) {
883  o_diag.allocate(memory_t::device).copy_to(memory_t::device);
884  o_diag1.allocate(memory_t::device).copy_to(memory_t::device);
885  }
886 
887  /* get new preconditionined residuals, and also opsi and psi as a by-product */
888  auto result = sirius::residuals<T>(ctx_, ctx_.preferred_memory_t(), ctx_.blas_linalg_t(), spin_range(nc_mag ? 2 : 0),
889  N, nevec, 0, eval, evec, sphi, phi, spsi, psi, res, o_diag, o_diag1,
890  itso.converge_by_energy(), itso.residual_tolerance(),
891  [&](int i, int ispn){return std::abs(eval[i] - eval_old[i]) < iterative_solver_tolerance;});
892  n = result.unconverged_residuals;
893  current_frobenius_norm = result.frobenius_norm;
894 
895  /* set the relative tolerance convergence criterion */
896  if (k == 0) {
897  relative_frobenius_tolerance = current_frobenius_norm * itso.relative_tolerance();
898  }
899  }
900 
901  /* verify convergence criteria */
902  bool converged_by_relative_tol = k > 0 && current_frobenius_norm < relative_frobenius_tolerance ;
903  bool converged_by_absolute_tol = n <= itso.min_num_res();
904  bool converged = converged_by_absolute_tol || converged_by_relative_tol;
905 
906  /* check if running out of space */
907  bool should_restart = N + n > num_phi;
908 
909  bool last_iteration = k == (itso.num_steps() - 1);
910 
911  /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
912  if (should_restart || converged || last_iteration) {
913  /* recompute wave-functions */
914  /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
915  transform(ctx_.spla_context(), nc_mag ? 2 : 0, phi, 0, N, evec, 0, 0, psi, 0, nevec);
916 
917  /* exit the loop if the eigen-vectors are converged or this is a last iteration */
918  if (converged || last_iteration) {
919  break;
920  } else { /* otherwise, set Psi as a new trial basis */
921  kp.message(3, __function_name__, "%s", "subspace size limit reached\n");
922 
923  if (itso.converge_by_energy()) {
924  transform(ctx_.spla_context(), nc_mag ? 2 : 0, sphi, 0, N, evec, 0, 0, spsi, 0, nevec);
925  }
926 
927  ovlp_old.zero();
928  for (int i = 0; i < nevec; i++) {
929  ovlp_old.set(i, i, eval[i]);
930  }
931  /* update basis functions */
932  for (int ispn = 0; ispn < num_sc; ispn++) {
933  phi.copy_from(ctx_.processing_unit(), nevec, psi, ispn, 0, ispn, 0);
934  sphi.copy_from(ctx_.processing_unit(), nevec, spsi, ispn, 0, ispn, 0);
935  }
936  /* number of basis functions that we already have */
937  N = nevec;
938  }
939  }
940  for (int ispn = 0; ispn < num_sc; ispn++) {
941  /* expand variational subspace with new basis vectors obtatined from residuals */
942  phi.copy_from(ctx_.processing_unit(), n, res, ispn, 0, ispn, N);
943  }
944  }
945 
946  Hk__.kp().release_hubbard_orbitals_on_device();
947 
948  return eval;
949 }
