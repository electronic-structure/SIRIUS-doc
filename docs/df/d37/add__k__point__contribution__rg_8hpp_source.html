28 inline void Density::add_k_point_contribution_rg(K_point* kp__)
29 {
30  PROFILE("sirius::Density::add_k_point_contribution_rg");
31 
32  double omega = unit_cell_.omega();
33 
34  auto& fft = ctx_.fft_coarse();
35 
36  /* get preallocated memory */
37  mdarray<double, 2> density_rg(ctx_.mem_pool().allocate<double, memory_t::host>(fft.local_size() * (ctx_.num_mag_dims() + 1)),
38  fft.local_size(), ctx_.num_mag_dims() + 1, "density_rg");
39  density_rg.zero();
40 
41  if (fft.pu() == GPU) {
42  density_rg.allocate(memory_t::device);
43  density_rg.zero<memory_t::device>();
44  }
45 
46  fft.prepare(kp__->gkvec_partition());
47 
48  /* non-magnetic or collinear case */
49  if (ctx_.num_mag_dims() != 3) {
50  /* loop over pure spinor components */
51  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
52  /* trivial case */
53  if (!kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col().global_index_size()) {
54  continue;
55  }
56 
57  for (int i = 0; i < kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col().local_size(); i++) {
58  int j = kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col()[i];
59  double w = kp__->band_occupancy(j, ispn) * kp__->weight() / omega;
60 
61  ///* transform to real space; in case of GPU wave-function stays in GPU memory */
62  fft.transform<1>(kp__->spinor_wave_functions().pw_coeffs(ispn).extra().template at<CPU>(0, i));
63  //switch (fft.pu()) {
64  // case CPU: {
65  // fft.transform<1>(kp__->gkvec().partition(),
66  // kp__->spinor_wave_functions(ispn).pw_coeffs().extra().template at<CPU>(0, i));
67  // break;
68  // }
69  // case GPU: {
70  // fft.transform<1, GPU>(kp__->gkvec().partition(),
71  // kp__->spinor_wave_functions(ispn).pw_coeffs().extra().template at<GPU>(0, i));
72  // break;
73  // }
74  //}
75 
76  /* add to density */
77  switch (fft.pu()) {
78  case CPU: {
79  #pragma omp parallel for schedule(static)
80  for (int ir = 0; ir < fft.local_size(); ir++) {
81  auto z = fft.buffer(ir);
82  density_rg(ir, ispn) += w * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
83  }
84  break;
85  }
86  case GPU: {
87 #ifdef __GPU
88  update_density_rg_1_gpu(fft.local_size(), fft.buffer().at<GPU>(), w, density_rg.at<GPU>(0, ispn));
89 #else
90  TERMINATE_NO_GPU
91 #endif
92  break;
93  }
94  }
95  }
96  }
97  } else { /* non-collinear case */
98  assert(kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col().local_size() ==
99  kp__->spinor_wave_functions().pw_coeffs(1).spl_num_col().local_size());
100 
101  /* allocate on CPU or GPU */
102  mdarray<double_complex, 1> psi_r(fft.local_size(), ctx_.main_memory_t());
103 
104  for (int i = 0; i < kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col().local_size(); i++) {
105  int j = kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col()[i];
106  double w = kp__->band_occupancy(j, 0) * kp__->weight() / omega;
107 
108  /* transform up- component of spinor function to real space; in case of GPU wave-function stays in GPU memory */
109  fft.transform<1>(kp__->spinor_wave_functions().pw_coeffs(0).extra().template at<CPU>(0, i));
110  /* save in auxiliary buffer */
111  switch (fft.pu()) {
112  case CPU: {
113  fft.output(&psi_r[0]);
114  break;
115  }
116  case GPU: {
117 #ifdef __GPU
118  acc::copyout(psi_r.at<GPU>(), fft.buffer().at<GPU>(), fft.local_size());
119 #endif
120  break;
121  }
122  }
123 
124  /* transform dn- component of spinor wave function */
125  fft.transform<1>(kp__->spinor_wave_functions().pw_coeffs(1).extra().template at<CPU>(0, i));
126 
127  switch (fft.pu()) {
128  case CPU: {
129  #pragma omp parallel for schedule(static)
130  for (int ir = 0; ir < fft.local_size(); ir++) {
131  auto r0 = (std::pow(psi_r[ir].real(), 2) + std::pow(psi_r[ir].imag(), 2)) * w;
132  auto r1 = (std::pow(fft.buffer(ir).real(), 2) + std::pow(fft.buffer(ir).imag(), 2)) * w;
133 
134  auto z2 = psi_r[ir] * std::conj(fft.buffer(ir)) * w;
135 
136  density_rg(ir, 0) += r0;
137  density_rg(ir, 1) += r1;
138  density_rg(ir, 2) += 2.0 * std::real(z2);
139  density_rg(ir, 3) -= 2.0 * std::imag(z2);
140  }
141  break;
142  }
143  case GPU: {
144 #ifdef __GPU
145  /* add up-up contribution */
146  update_density_rg_1_gpu(fft.local_size(), psi_r.at<GPU>(), w, density_rg.at<GPU>(0, 0));
147  /* add dn-dn contribution */
148  update_density_rg_1_gpu(fft.local_size(), fft.buffer().at<GPU>(), w, density_rg.at<GPU>(0, 1));
149  /* add off-diagonal contribution */
150  update_density_rg_2_gpu(fft.local_size(), psi_r.at<GPU>(), fft.buffer().at<GPU>(), w,
151  density_rg.at<GPU>(0, 2), density_rg.at<GPU>(0, 3));
152 #endif
153  break;
154  }
155  }
156  }
157  }
158 
159  if (fft.pu() == GPU) {
160  density_rg.copy<memory_t::device, memory_t::host>();
161  }
162 
163  /* switch from real density matrix to density and magnetization */
164  switch (ctx_.num_mag_dims()) {
165  case 3: {
166  #pragma omp parallel for schedule(static)
167  for (int ir = 0; ir < fft.local_size(); ir++) {
168  rho_mag_coarse_[2]->f_rg(ir) += density_rg(ir, 2); // Mx
169  rho_mag_coarse_[3]->f_rg(ir) += density_rg(ir, 3); // My
170  }
171  }
172  case 1: {
173  #pragma omp parallel for schedule(static)
174  for (int ir = 0; ir < fft.local_size(); ir++) {
175  rho_mag_coarse_[0]->f_rg(ir) += (density_rg(ir, 0) + density_rg(ir, 1)); // rho
176  rho_mag_coarse_[1]->f_rg(ir) += (density_rg(ir, 0) - density_rg(ir, 1)); // Mz
177  }
178  break;
179  }
180  case 0: {
181  #pragma omp parallel for schedule(static)
182  for (int ir = 0; ir < fft.local_size(); ir++) {
183  rho_mag_coarse_[0]->f_rg(ir) += density_rg(ir, 0); // rho
184  }
185  }
186  }
187 
188  fft.dismiss();
189  ctx_.mem_pool().reset<memory_t::host>();
190 }
