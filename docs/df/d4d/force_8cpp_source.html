46 template <typename T>
47 void Force::add_k_point_contribution(K_point& kpoint, mdarray<double, 2>& forces__) const
48 {
49  /* if there are no beta projectors then get out there */
50  if (ctx_.unit_cell().mt_lo_basis_size() == 0) {
51  return;
52  }
53 
54  Beta_projectors_gradient bp_grad(ctx_, kpoint.gkvec(), kpoint.igk_loc(), kpoint.beta_projectors());
55  if (is_device_memory(ctx_.preferred_memory_t())) {
56  int nbnd = ctx_.num_bands();
57  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
58  /* allocate GPU memory */
59  kpoint.spinor_wave_functions().pw_coeffs(ispn).allocate(ctx_.mem_pool(memory_t::device));
60  kpoint.spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, nbnd);
61  }
62  }
63 
64  Non_local_functor<T> nlf(ctx_, bp_grad);
65 
66  nlf.add_k_point_contribution(kpoint, forces__);
67  if (is_device_memory(ctx_.preferred_memory_t())) {
68  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
69  /* deallocate GPU memory */
70  kpoint.spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
71  }
72  }
73 }

75 void Force::compute_dmat(K_point* kp__, dmatrix<double_complex>& dm__) const
76 {
77  dm__.zero();
78 
79  /* trivial case */
80  if (!ctx_.need_sv()) {
81  for (int i = 0; i < ctx_.num_fv_states(); i++) {
82  dm__.set(i, i, double_complex(kp__->band_occupancy(i, 0), 0));
83  }
84  } else {
85  if (ctx_.num_mag_dims() != 3) {
86  dmatrix<double_complex> ev1(ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.blacs_grid(),
87  ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
88  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
89  auto& ev = kp__->sv_eigen_vectors(ispn);
90  /* multiply second-variational eigen-vectors with band occupancies */
91  for (int j = 0; j < ev.num_cols_local(); j++) {
92  /* up- or dn- band index */
93  int jb = ev.icol(j);
94  for (int i = 0; i < ev.num_rows_local(); i++) {
95  ev1(i, j) = std::conj(ev(i, j)) * kp__->band_occupancy(jb, ispn);
96  }
97  }
98 
99  linalg(linalg_t::scalapack).gemm('N', 'T', ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.num_bands(),
100  &linalg_const<double_complex>::one(), ev1, 0, 0, ev, 0, 0,
101  &linalg_const<double_complex>::one(), dm__, 0, 0);
102  }
103  } else {
104  dmatrix<double_complex> ev1(ctx_.num_bands(), ctx_.num_bands(), ctx_.blacs_grid(), ctx_.cyclic_block_size(),
105  ctx_.cyclic_block_size());
106  auto& ev = kp__->sv_eigen_vectors(0);
107  /* multiply second-variational eigen-vectors with band occupancies */
108  for (int j = 0; j < ev.num_cols_local(); j++) {
109  /* band index */
110  int jb = ev.icol(j);
111  for (int i = 0; i < ev.num_rows_local(); i++) {
112  ev1(i, j) = std::conj(ev(i, j)) * kp__->band_occupancy(jb, 0);
113  }
114  }
115  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
116  int offs = ispn * ctx_.num_fv_states();
117 
118  linalg(linalg_t::scalapack).gemm('N', 'T', ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.num_bands(),
119  &linalg_const<double_complex>::one(), ev1, offs, 0, ev, offs, 0,
120  &linalg_const<double_complex>::one(), dm__, 0, 0);
121  }
122  }
123  }

126 mdarray<double, 2> const& Force::calc_forces_total()
127 {
128  forces_total_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
129  if (ctx_.full_potential()) {
130  calc_forces_rho();
131  calc_forces_hf();
132  calc_forces_ibs();
133  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
134  for (int x : {0, 1, 2}) {
135  forces_total_(x, ia) = forces_ibs_(x, ia) + forces_hf_(x, ia) + forces_rho_(x, ia);
136  }
137  }
138  } else {
139  calc_forces_vloc();
140  calc_forces_us();
141  calc_forces_nonloc();
142  calc_forces_core();
143  calc_forces_ewald();
144  calc_forces_scf_corr();
145  calc_forces_hubbard();
146 
147  forces_total_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
148  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
149  for (int x : {0, 1, 2}) {
150  forces_total_(x, ia) = forces_vloc_(x, ia) + forces_us_(x, ia) + forces_nonloc_(x, ia) +
151  forces_core_(x, ia) + forces_ewald_(x, ia) + forces_scf_corr_(x, ia) +
152  forces_hubbard_(x, ia);
153  }
154  }
155  }
156  return forces_total_;
157 }

159 mdarray<double, 2> const& Force::calc_forces_ibs()
160 {
161  forces_ibs_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
162  forces_ibs_.zero();
163 
164  mdarray<double, 2> ffac(ctx_.unit_cell().num_atom_types(), ctx_.gvec().num_shells());
165  #pragma omp parallel for
166  for (int igs = 0; igs < ctx_.gvec().num_shells(); igs++) {
167  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
168  ffac(iat, igs) = unit_step_function_form_factors(ctx_.unit_cell().atom_type(iat).mt_radius(),
169  ctx_.gvec().shell_len(igs));
170  }
171  }
172 
173  Hamiltonian0 H0(potential_);
174  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
175  int ik = kset_.spl_num_kpoints(ikloc);
176  auto hk = H0(*kset_[ik]);
177  add_ibs_force(kset_[ik], hk, ffac, forces_ibs_);
178  }
179  ctx_.comm().allreduce(&forces_ibs_(0, 0), (int)forces_ibs_.size());
180  symmetrize(forces_ibs_);
181 
182  if (ctx_.control().verbosity_ > 2 && ctx_.comm().rank() == 0) {
183  std::printf("ibs force\n");
184  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
185  std::printf("ia : %i, IBS : %12.6f %12.6f %12.6f\n", ia, forces_ibs_(0, ia), forces_ibs_(1, ia),
186  forces_ibs_(2, ia));
187  }
188  }
189 
190  return forces_ibs_;
191 }

271 mdarray<double, 2> const& Force::calc_forces_ewald()
272 {
273  PROFILE("sirius::Force::calc_forces_ewald");
274 
275  forces_ewald_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
276  forces_ewald_.zero();
277 
278  Unit_cell& unit_cell = ctx_.unit_cell();
279 
280  double alpha = ctx_.ewald_lambda();
281 
282  double prefac = (ctx_.gvec().reduced() ? 4.0 : 2.0) * (twopi / unit_cell.omega());
283 
284  int ig0{0};
285  if (ctx_.comm().rank() == 0) {
286  ig0 = 1;
287  }
288 
289  mdarray<double_complex, 1> rho_tmp(ctx_.gvec().count());
290  rho_tmp.zero();
291  #pragma omp parallel for schedule(static)
292  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
293  int ig = ctx_.gvec().offset() + igloc;
294 
295  double_complex rho(0, 0);
296 
297  for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
298  rho += ctx_.gvec_phase_factor(ig, ja) * static_cast<double>(unit_cell.atom(ja).zn());
299  }
300 
301  rho_tmp[igloc] = std::conj(rho);
302  }
303 
304  #pragma omp parallel for
305  for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
306  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
307  int ig = ctx_.gvec().offset() + igloc;
308 
309  double g2 = std::pow(ctx_.gvec().gvec_len(ig), 2);
310 
311  /* cartesian form for getting cartesian force components */
312  vector3d<double> gvec_cart = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
313 
314  double scalar_part = prefac * (rho_tmp[igloc] * ctx_.gvec_phase_factor(ig, ja)).imag() *
315  static_cast<double>(unit_cell.atom(ja).zn()) * std::exp(-g2 / (4 * alpha)) / g2;
316 
317  for (int x : {0, 1, 2}) {
318  forces_ewald_(x, ja) += scalar_part * gvec_cart[x];
319  }
320  }
321  }
322 
323  ctx_.comm().allreduce(&forces_ewald_(0, 0), 3 * ctx_.unit_cell().num_atoms());
324 
325  double invpi = 1. / pi;
326 
327  #pragma omp parallel for
328  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
329  for (int i = 1; i < unit_cell.num_nearest_neighbours(ia); i++) {
330  int ja = unit_cell.nearest_neighbour(i, ia).atom_id;
331 
332  double d = unit_cell.nearest_neighbour(i, ia).distance;
333  double d2 = d * d;
334 
335  vector3d<double> t =
336  unit_cell.lattice_vectors() * vector3d<int>(unit_cell.nearest_neighbour(i, ia).translation);
337 
338  double scalar_part =
339  static_cast<double>(unit_cell.atom(ia).zn() * unit_cell.atom(ja).zn()) / d2 *
340  (std::erfc(std::sqrt(alpha) * d) / d + 2.0 * std::sqrt(alpha * invpi) * std::exp(-d2 * alpha));
341 
342  for (int x : {0, 1, 2}) {
343  forces_ewald_(x, ia) += scalar_part * t[x];
344  }
345  }
346  }
347 
348  return forces_ewald_;
349 }

351 mdarray<double, 2> const& Force::calc_forces_us()
352 {
353  PROFILE("sirius::Force::calc_forces_us");
354 
355  forces_us_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
356  forces_us_.zero();
357 
358  potential_.fft_transform(-1);
359 
360  Unit_cell& unit_cell = ctx_.unit_cell();
361 
362  double reduce_g_fact = ctx_.gvec().reduced() ? 2.0 : 1.0;
363 
364  linalg_t la{linalg_t::none};
365 
366  memory_pool* mp{nullptr};
367  switch (ctx_.processing_unit()) {
368  case device_t::CPU: {
369  mp = &ctx_.mem_pool(memory_t::host);
370  la = linalg_t::blas;
371  break;
372  }
373  case device_t::GPU: {
374 #ifdef __ROCM
375  // ROCm does not support cubblasxt functionality
376  mp = &ctx_.mem_pool(memory_t::host);
377  la = linalg_t::blas;
378 #else
379  mp = &ctx_.mem_pool(memory_t::host_pinned);
380  la = linalg_t::cublasxt;
381 #endif
382  break;
383  }
384  }
385 
386  /* over atom types */
387  for (int iat = 0; iat < unit_cell.num_atom_types(); iat++) {
388  auto& atom_type = unit_cell.atom_type(iat);
389 
390  auto aug_op = ctx_.augmentation_op(iat);
391 
392  if (!aug_op) {
393  continue;
394  }
395 
396  int nbf = atom_type.mt_basis_size();
397 
398  /* get auxiliary density matrix */
399  auto dm = density_.density_matrix_aux(iat);
400 
401  mdarray<double, 2> v_tmp(atom_type.num_atoms(), ctx_.gvec().count() * 2, *mp);
402  mdarray<double, 2> tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms(), *mp);
403 
404  /* over spin components, can be from 1 to 4*/
405  for (int ispin = 0; ispin < ctx_.num_mag_dims() + 1; ispin++) {
406  /* over 3 components of the force/G - vectors */
407  for (int ivec = 0; ivec < 3; ivec++) {
408  /* over local rank G vectors */
409  #pragma omp parallel for schedule(static)
410  for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
411  int ig = ctx_.gvec().offset() + igloc;
412  auto gvc = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
413  for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
414  /* here we write in v_tmp -i * G * exp[ iGRn] Veff(G)
415  * but in formula we have i * G * exp[-iGRn] Veff*(G)
416  * the differences because we unfold complex array in the real one
417  * and need negative imagine part due to a multiplication law of complex numbers */
418  auto z = double_complex(0, -gvc[ivec]) * ctx_.gvec_phase_factor(ig, atom_type.atom_id(ia)) *
419  potential_.component(ispin).f_pw_local(igloc);
420  v_tmp(ia, 2 * igloc) = z.real();
421  v_tmp(ia, 2 * igloc + 1) = z.imag();
422  }
423  }
424 
425  /* multiply tmp matrices, or sum over G */
426  linalg(la).gemm('N', 'T', nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
427  &linalg_const<double>::one(),
428  aug_op->q_pw().at(memory_t::host), aug_op->q_pw().ld(),
429  v_tmp.at(memory_t::host), v_tmp.ld(),
430  &linalg_const<double>::zero(),
431  tmp.at(memory_t::host), tmp.ld());
432 
433  #pragma omp parallel for
434  for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
435  for (int i = 0; i < nbf * (nbf + 1) / 2; i++) {
436  forces_us_(ivec, atom_type.atom_id(ia)) += ctx_.unit_cell().omega() * reduce_g_fact *
437  dm(i, ia, ispin) * aug_op->sym_weight(i) * tmp(i, ia);
438  }
439  }
440  }
441  }
442  }
443 
444  ctx_.comm().allreduce(&forces_us_(0, 0), 3 * ctx_.unit_cell().num_atoms());
445 
446  return forces_us_;
447 }

449 mdarray<double, 2> const& Force::calc_forces_scf_corr()
450 {
451  PROFILE("sirius::Force::calc_forces_scf_corr");
452 
453  forces_scf_corr_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
454  forces_scf_corr_.zero();
455 
456  /* get main arrays */
457  auto& dveff = potential_.dveff();
458 
459  Unit_cell& unit_cell = ctx_.unit_cell();
460 
461  Gvec const& gvec = ctx_.gvec();
462 
463  int gvec_count = gvec.count();
464  int gvec_offset = gvec.offset();
465 
466  double fact = gvec.reduced() ? 2.0 : 1.0;
467 
468  int ig0 = (ctx_.comm().rank() == 0) ? 1 : 0;
469 
470  auto& ri = ctx_.ps_rho_ri();
471 
472  #pragma omp parallel for
473  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
474  Atom& atom = unit_cell.atom(ia);
475 
476  int iat = atom.type_id();
477 
478  for (int igloc = ig0; igloc < gvec_count; igloc++) {
479  int ig = gvec_offset + igloc;
480 
481  /* cartesian form for getting cartesian force components */
482  vector3d<double> gvec_cart = gvec.gvec_cart<index_domain_t::local>(igloc);
483 
484  /* scalar part of a force without multipying by G-vector */
485  double_complex z = fact * fourpi * ri.value<int>(iat, gvec.gvec_len(ig)) *
486  std::conj(dveff.f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia));
487 
488  /* get force components multiplying by cartesian G-vector */
489  for (int x : {0, 1, 2}) {
490  forces_scf_corr_(x, ia) -= (gvec_cart[x] * z).imag();
491  }
492  }
493  }
494  ctx_.comm().allreduce(&forces_scf_corr_(0, 0), 3 * ctx_.unit_cell().num_atoms());
495 
496  return forces_scf_corr_;
497 }

499 mdarray<double, 2> const& Force::calc_forces_core()
500 {
501  PROFILE("sirius::Force::calc_forces_core");
502 
503  forces_core_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
504  forces_core_.zero();
505 
506  /* get main arrays */
507  auto& xc_pot = potential_.xc_potential();
508 
509  /* transform from real space to reciprocal */
510  xc_pot.fft_transform(-1);
511 
512  Unit_cell& unit_cell = ctx_.unit_cell();
513 
514  Gvec const& gvecs = ctx_.gvec();
515 
516  int gvec_count = gvecs.count();
517  int gvec_offset = gvecs.offset();
518 
519  double fact = gvecs.reduced() ? 2.0 : 1.0;
520 
521  auto& ri = ctx_.ps_core_ri();
522 
523  /* here the calculations are in lattice vectors space */
524  #pragma omp parallel for
525  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
526  Atom& atom = unit_cell.atom(ia);
527  if (atom.type().ps_core_charge_density().empty()) {
528  continue;
529  }
530  int iat = atom.type_id();
531 
532  for (int igloc = 0; igloc < gvec_count; igloc++) {
533  int ig = gvec_offset + igloc;
534 
535  if (ig == 0) {
536  continue;
537  }
538 
539  /* cartesian form for getting cartesian force components */
540  vector3d<double> gvec_cart = gvecs.gvec_cart<index_domain_t::local>(igloc);
541 
542  /* scalar part of a force without multipying by G-vector */
543  double_complex z = fact * fourpi * ri.value<int>(iat, gvecs.gvec_len(ig)) *
544  std::conj(xc_pot.f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia));
545

549  }

580  }

588 {
589  PROFILE("sirius::Force::calc_forces_vloc");
590 
591  forces_vloc_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
592  forces_vloc_.zero();
593 
594  auto& valence_rho = density_.rho();
595 
596  auto& ri = ctx_.vloc_ri();
597 
598  Unit_cell& unit_cell = ctx_.unit_cell();
599

619 
620  /* scalar part of a force without multiplying by G-vector */

626  forces_vloc_(x, ia) -= (gvec_cart[x] * z).imag();
627  }
628  }
629  }
630 
631  ctx_.comm().allreduce(&forces_vloc_(0, 0), 3 * ctx_.unit_cell().num_atoms());
632

651 void Force::add_ibs_force(K_point* kp__, Hamiltonian_k& Hk__, mdarray<double, 2>& ffac__, mdarray<double, 2>& forcek__) const
652 {
653  PROFILE("sirius::Force::ibs_force");
654 
655  auto& uc = ctx_.unit_cell();
656 
657  auto& bg = ctx_.blacs_grid();
658 
659  auto bs = ctx_.cyclic_block_size();
660 
661  auto nfv = ctx_.num_fv_states();
662 
663  auto ngklo = kp__->gklo_basis_size();
664 
665  /* compute density matrix for a k-point */
666  dmatrix<double_complex> dm(nfv, nfv, bg, bs, bs);
667  compute_dmat(kp__, dm);
668 
669  /* first-variational eigen-vectors in scalapack distribution */
670  auto& fv_evec = kp__->fv_eigen_vectors();
671 
672  dmatrix<double_complex> h(ngklo, ngklo, bg, bs, bs);
673  dmatrix<double_complex> o(ngklo, ngklo, bg, bs, bs);
674 
675  dmatrix<double_complex> h1(ngklo, ngklo, bg, bs, bs);
676  dmatrix<double_complex> o1(ngklo, ngklo, bg, bs, bs);
677 
678  dmatrix<double_complex> zm1(ngklo, nfv, bg, bs, bs);
679  dmatrix<double_complex> zf(nfv, nfv, bg, bs, bs);
680 
681  mdarray<double_complex, 2> alm_row(kp__->num_gkvec_row(), uc.max_mt_aw_basis_size());
682  mdarray<double_complex, 2> alm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
683  mdarray<double_complex, 2> halm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
684 
685  for (int ia = 0; ia < uc.num_atoms(); ia++) {
686  h.zero();
687  o.zero();
688 
689  auto& atom = uc.atom(ia);
690  auto& type = atom.type();
691 
692  /* generate matching coefficients for current atom */
693  kp__->alm_coeffs_row().generate<true>(atom, alm_row);
694  kp__->alm_coeffs_col().generate<false>(atom, alm_col);
695 
696  /* setup apw-lo and lo-apw blocks */
697  Hk__.set_fv_h_o_apw_lo(atom, ia, alm_row, alm_col, h, o);
698 
699  /* apply MT Hamiltonian to column coefficients */
700  Hk__.H0().apply_hmt_to_apw<spin_block_t::nm>(atom, kp__->num_gkvec_col(), alm_col, halm_col);
701 
702  /* apw-apw block of the overlap matrix */
703  linalg(linalg_t::blas).gemm('N', 'T', kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
704  &linalg_const<double_complex>::one(), alm_row.at(memory_t::host), alm_row.ld(),
705  alm_col.at(memory_t::host), alm_col.ld(), &linalg_const<double_complex>::zero(),
706  o.at(memory_t::host), o.ld());
707 
708  /* apw-apw block of the Hamiltonian matrix */
709  linalg(linalg_t::blas).gemm('N', 'T', kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
710  &linalg_const<double_complex>::one(), alm_row.at(memory_t::host), alm_row.ld(), halm_col.at(memory_t::host),
711  halm_col.ld(), &linalg_const<double_complex>::zero(), h.at(memory_t::host), h.ld());
712 
713  int iat = type.id();
714 
715  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) { // loop over columns
716  auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
717  auto gkvec_col_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_col(igk_col));
718  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) { // for each column loop over rows
719  auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
720  auto gkvec_row_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_row(igk_row));
721 
722  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
723 
724  int igs = ctx_.gvec().shell(ig12);
725 
726  auto zt = std::conj(ctx_.gvec_phase_factor(ig12, ia)) * ffac__(iat, igs) * fourpi / uc.omega();
727 
728  double t1 = 0.5 * dot(gkvec_row_cart, gkvec_col_cart);
729 
730  h(igk_row, igk_col) -= t1 * zt;
731  o(igk_row, igk_col) -= zt;
732  }
733  }
734 
735  for (int x = 0; x < 3; x++) {
736  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) { // loop over columns
737  auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
738  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) { // loop over rows
739  auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
740  /* compute index of G-G' */
741  int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
742  /* get G-G' */
743  auto vg = ctx_.gvec().gvec_cart<index_domain_t::global>(ig12);
744  /* multiply by i(G-G') */
745  h1(igk_row, igk_col) = double_complex(0.0, vg[x]) * h(igk_row, igk_col);
746  /* multiply by i(G-G') */
747  o1(igk_row, igk_col) = double_complex(0.0, vg[x]) * o(igk_row, igk_col);
748  }
749  }
750 
751  for (int icol = 0; icol < kp__->num_lo_col(); icol++) {
752  for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) {
753  auto gkvec_row_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_row(igk_row));
754  /* multiply by i(G+k) */
755  h1(igk_row, icol + kp__->num_gkvec_col()) =
756  double_complex(0.0, gkvec_row_cart[x]) * h(igk_row, icol + kp__->num_gkvec_col());
757  /* multiply by i(G+k) */
758  o1(igk_row, icol + kp__->num_gkvec_col()) =
759  double_complex(0.0, gkvec_row_cart[x]) * o(igk_row, icol + kp__->num_gkvec_col());
760  }
761  }
762 
763  for (int irow = 0; irow < kp__->num_lo_row(); irow++) {
764  for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) {
765  auto gkvec_col_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_col(igk_col));
766  /* multiply by i(G+k) */
767  h1(irow + kp__->num_gkvec_row(), igk_col) =
768  double_complex(0.0, -gkvec_col_cart[x]) * h(irow + kp__->num_gkvec_row(), igk_col);
769  /* multiply by i(G+k) */
770  o1(irow + kp__->num_gkvec_row(), igk_col) =
771  double_complex(0.0, -gkvec_col_cart[x]) * o(irow + kp__->num_gkvec_row(), igk_col);
772  }
773  }
774 
775  /* zm1 = dO * V */
776  linalg(linalg_t::scalapack).gemm('N', 'N', ngklo, nfv, ngklo, &linalg_const<double_complex>::one(),
777  o1, 0, 0, fv_evec, 0, 0, &linalg_const<double_complex>::zero(), zm1, 0, 0);
778  /* multiply by energy: zm1 = E * (dO * V) */
779  for (int i = 0; i < zm1.num_cols_local(); i++) {
780  int ist = zm1.icol(i);
781  for (int j = 0; j < kp__->gklo_basis_size_row(); j++) {
782  zm1(j, i) *= kp__->fv_eigen_value(ist);
783  }
784  }
785  /* compute zm1 = dH * V - E * (dO * V) */
786  linalg(linalg_t::scalapack).gemm('N', 'N', ngklo, nfv, ngklo, &linalg_const<double_complex>::one(),
787  h1, 0, 0, fv_evec, 0, 0, &linalg_const<double_complex>::m_one(), zm1, 0, 0);
788 
789  /* compute zf = V^{+} * zm1 = V^{+} * (dH * V - E * (dO * V)) */
790  linalg(linalg_t::scalapack).gemm('C', 'N', nfv, nfv, ngklo, &linalg_const<double_complex>::one(),
791  fv_evec, 0, 0, zm1, 0, 0, &linalg_const<double_complex>::zero(), zf, 0, 0);
792 
793  for (int i = 0; i < dm.num_cols_local(); i++) {
794  for (int j = 0; j < dm.num_rows_local(); j++) {
795  forcek__(x, ia) += kp__->weight() * std::real(dm(j, i) * zf(j, i));
796  }
797  }
798  }
799  } // ia
800 }

802 void Force::symmetrize(mdarray<double, 2>& forces__) const
803 {
804  if (!ctx_.use_symmetry()) {
805  return;
806  }
807 
808  mdarray<double, 2> sym_forces(3, ctx_.unit_cell().num_atoms());
809  sym_forces.zero();
810 
811  auto& lattice_vectors = ctx_.unit_cell().symmetry().lattice_vectors();
812  auto& inverse_lattice_vectors = ctx_.unit_cell().symmetry().inverse_lattice_vectors();
813 
814  sym_forces.zero();
815 
816  #pragma omp parallel for
817  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
818  vector3d<double> cart_force(&forces__(0, ia));
819  vector3d<double> lat_force =
820  inverse_lattice_vectors * (cart_force / (double)ctx_.unit_cell().symmetry().num_mag_sym());
821 
822  for (int isym = 0; isym < ctx_.unit_cell().symmetry().num_mag_sym(); isym++) {
823  int ja = ctx_.unit_cell().symmetry().sym_table(ia, isym);
824  auto& R = ctx_.unit_cell().symmetry().magnetic_group_symmetry(isym).spg_op.R;
825  vector3d<double> rot_force = lattice_vectors * (R * lat_force);
826 
827  #pragma omp atomic update
828  sym_forces(0, ja) += rot_force[0];
829 
830  #pragma omp atomic update
831  sym_forces(1, ja) += rot_force[1];
832 
833  #pragma omp atomic update
834  sym_forces(2, ja) += rot_force[2];
835  }
836  }
837  sym_forces >> forces__;
838 }
