47void Force::symmetrize(mdarray<double, 2>& forces__) const
48{
49 if (!ctx_.use_symmetry()) {
50 return;
51 }
52
53 PROFILE("sirius::Force::symmetrize");
54
55 sddk::mdarray<double, 2> sym_forces(3, ctx_.unit_cell().spl_num_atoms().local_size());
56 sym_forces.zero();
57
58 for (int isym = 0; isym < ctx_.unit_cell().symmetry().size(); isym++) {
59 auto Rc = dot(dot(ctx_.unit_cell().symmetry().lattice_vectors(),
60 matrix3d<double>(ctx_.unit_cell().symmetry()[isym].spg_op.R)),
61 ctx_.unit_cell().symmetry().inverse_lattice_vectors());
62
63 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
64 vector3d<double> force_ia(&forces__(0, ia));
65 int ja = ctx_.unit_cell().symmetry()[isym].spg_op.sym_atom[ia];
66 auto location = ctx_.unit_cell().spl_num_atoms().location(ja);
67 if (location.rank == ctx_.comm().rank()) {
68 auto force_ja = dot(Rc, force_ia);
69 for (int x: {0, 1, 2}) {
70 sym_forces(x, location.local_index) += force_ja[x];
71 }
72 }
73 }
74 }
75
76 double alpha = 1.0 / double(ctx_.unit_cell().symmetry().size());
77 for (int ia = 0; ia < ctx_.unit_cell().spl_num_atoms().local_size(); ia++) {
78 for (int x: {0, 1, 2}) {
79 sym_forces(x, ia) *= alpha;
80 }
81 }
82 double* sbuf = ctx_.unit_cell().spl_num_atoms().local_size() ? sym_forces.at(memory_t::host) : nullptr;
83 ctx_.comm().allgather(sbuf, forces__.at(memory_t::host), 3 * ctx_.unit_cell().spl_num_atoms().local_size(),
84 3 * ctx_.unit_cell().spl_num_atoms().global_offset());
85}

87template <typename T>
88void Force::calc_forces_nonloc_aux()
89{
90 forces_nonloc_ = sddk::mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
91 forces_nonloc_.zero();
92
93 auto& spl_num_kp = kset_.spl_num_kpoints();
94
95 for (int ikploc = 0; ikploc < spl_num_kp.local_size(); ikploc++) {
96 auto* kp = kset_.get<T>(spl_num_kp[ikploc]);
97
98 if (ctx_.gamma_point()) {
99 add_k_point_contribution<T>(*kp, forces_nonloc_);
100 } else {
101 add_k_point_contribution<std::complex<T>>(*kp, forces_nonloc_);
102 }
103 }
104
105 ctx_.comm().allreduce(&forces_nonloc_(0, 0), 3 * ctx_.unit_cell().num_atoms());
106
107 symmetrize(forces_nonloc_);
108}

124template <typename T>
125void Force::add_k_point_contribution(K_point<real_type<T>>& kp__, mdarray<double, 2>& forces__) const
126{
127 /* if there are no beta projectors then get out there */
128 if (ctx_.unit_cell().mt_lo_basis_size() == 0) {
129 return;
130 }
131
132 Beta_projectors_gradient<real_type<T>> bp_grad(ctx_, kp__.gkvec(), kp__.igk_loc(), kp__.beta_projectors());
133 if (is_device_memory(ctx_.preferred_memory_t())) {
134 int nbnd = ctx_.num_bands();
135 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
136 /* allocate GPU memory */
137 kp__.spinor_wave_functions().pw_coeffs(ispn).allocate(ctx_.mem_pool(memory_t::device));
138 kp__.spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, nbnd);
139 }
140 }
141
142 Non_local_functor<T> nlf(ctx_, bp_grad);
143
144 sddk::mdarray<real_type<T>, 2> f(3, ctx_.unit_cell().num_atoms());
145 f.zero();
146
147 nlf.add_k_point_contribution(kp__, f);
148 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
149 for (int x: {0, 1, 2}) {
150 forces__(x, ia) += f(x, ia);
151 }
152 }
153 if (is_device_memory(ctx_.preferred_memory_t())) {
154 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
155 /* deallocate GPU memory */
156 kp__.spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
157 }
158 }
159}

161void Force::compute_dmat(K_point<double>* kp__, dmatrix<double_complex>& dm__) const
162{
163 dm__.zero();
164
165 /* trivial case */
166 if (!ctx_.need_sv()) {
167 for (int i = 0; i < ctx_.num_fv_states(); i++) {
168 dm__.set(i, i, double_complex(kp__->band_occupancy(i, 0), 0));
169 }
170 } else {
171 if (ctx_.num_mag_dims() != 3) {
172 dmatrix<double_complex> ev1(ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.blacs_grid(),
173 ctx_.cyclic_block_size(), ctx_.cyclic_block_size());
174 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
175 auto& ev = kp__->sv_eigen_vectors(ispn);
176 /* multiply second-variational eigen-vectors with band occupancies */
177 for (int j = 0; j < ev.num_cols_local(); j++) {
178 /* up- or dn- band index */
179 int jb = ev.icol(j);
180 for (int i = 0; i < ev.num_rows_local(); i++) {
181 ev1(i, j) = std::conj(ev(i, j)) * kp__->band_occupancy(jb, ispn);
182 }
183 }
184
185 linalg(linalg_t::scalapack).gemm('N', 'T', ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.num_bands(),
186 &linalg_const<double_complex>::one(), ev1, 0, 0, ev, 0, 0,
187 &linalg_const<double_complex>::one(), dm__, 0, 0);
188 }
189 } else {
190 dmatrix<double_complex> ev1(ctx_.num_bands(), ctx_.num_bands(), ctx_.blacs_grid(), ctx_.cyclic_block_size(),
191 ctx_.cyclic_block_size());
192 auto& ev = kp__->sv_eigen_vectors(0);
193 /* multiply second-variational eigen-vectors with band occupancies */
194 for (int j = 0; j < ev.num_cols_local(); j++) {
195 /* band index */
196 int jb = ev.icol(j);
197 for (int i = 0; i < ev.num_rows_local(); i++) {
198 ev1(i, j) = std::conj(ev(i, j)) * kp__->band_occupancy(jb, 0);
199 }
200 }
201 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
202 int offs = ispn * ctx_.num_fv_states();
203
204 linalg(linalg_t::scalapack).gemm('N', 'T', ctx_.num_fv_states(), ctx_.num_fv_states(), ctx_.num_bands(),
205 &linalg_const<double_complex>::one(), ev1, offs, 0, ev, offs, 0,
206 &linalg_const<double_complex>::one(), dm__, 0, 0);
207 }
208 }
209 }

212mdarray<double, 2> const& Force::calc_forces_total()
213{
214 forces_total_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
215 if (ctx_.full_potential()) {
216 calc_forces_rho();
217 calc_forces_hf();
218 calc_forces_ibs();
219 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
220 for (int x : {0, 1, 2}) {
221 forces_total_(x, ia) = forces_ibs_(x, ia) + forces_hf_(x, ia) + forces_rho_(x, ia);
222 }
223 }
224 } else {
225 calc_forces_vloc();
226 calc_forces_us();
227 calc_forces_nonloc();
228 calc_forces_core();
229 calc_forces_ewald();
230 calc_forces_scf_corr();
231 calc_forces_hubbard();
232
233 forces_total_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
234 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
235 for (int x : {0, 1, 2}) {
236 forces_total_(x, ia) = forces_vloc_(x, ia) + forces_us_(x, ia) + forces_nonloc_(x, ia) +
237 forces_core_(x, ia) + forces_ewald_(x, ia) + forces_scf_corr_(x, ia) +
238 forces_hubbard_(x, ia);
239 }
240 }
241 }
242 return forces_total_;
243}

245mdarray<double, 2> const& Force::calc_forces_ibs()
246{
247 forces_ibs_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
248 forces_ibs_.zero();
249
250 mdarray<double, 2> ffac(ctx_.unit_cell().num_atom_types(), ctx_.gvec().num_shells());
251 #pragma omp parallel for
252 for (int igs = 0; igs < ctx_.gvec().num_shells(); igs++) {
253 for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
254 ffac(iat, igs) = unit_step_function_form_factors(ctx_.unit_cell().atom_type(iat).mt_radius(),
255 ctx_.gvec().shell_len(igs));
256 }
257 }
258
259 Hamiltonian0<double> H0(potential_);
260 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
261 int ik = kset_.spl_num_kpoints(ikloc);
262 auto hk = H0(*kset_.get<double>(ik));
263 add_ibs_force(kset_.get<double>(ik), hk, ffac, forces_ibs_);
264 }
265 ctx_.comm().allreduce(&forces_ibs_(0, 0), (int)forces_ibs_.size());
266 symmetrize(forces_ibs_);
267
268 if (ctx_.verbosity() > 2 && ctx_.comm().rank() == 0) {
269 std::printf("ibs force\n");
270 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
271 std::printf("ia : %i, IBS : %12.6f %12.6f %12.6f\n", ia, forces_ibs_(0, ia), forces_ibs_(1, ia),
272 forces_ibs_(2, ia));
273 }
274 }
275
276 return forces_ibs_;
277}

279mdarray<double, 2> const& Force::calc_forces_rho()
280{
281
282 forces_rho_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
283 forces_rho_.zero();
284 for (int ialoc = 0; ialoc < ctx_.unit_cell().spl_num_atoms().local_size(); ialoc++) {
285 int ia = ctx_.unit_cell().spl_num_atoms(ialoc);
286 auto g = gradient(density_.density_mt(ialoc));
287 for (int x = 0; x < 3; x++) {
288 forces_rho_(x, ia) = inner(potential_.effective_potential_mt(ialoc), g[x]);
289 }
290 }
291 ctx_.comm().allreduce(&forces_rho_(0, 0), (int)forces_rho_.size());
292 symmetrize(forces_rho_);
293
294 if (ctx_.verbosity() > 2 && ctx_.comm().rank() == 0) {
295 std::printf("rho force\n");
296 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
297 std::printf("ia : %i, density contribution : %12.6f %12.6f %12.6f\n", ia, forces_rho_(0, ia), forces_rho_(1, ia),
298 forces_rho_(2, ia));
299 }
300 }
301 return forces_rho_;
302}

304mdarray<double, 2> const& Force::calc_forces_hf()
305{
306 forces_hf_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
307 forces_hf_.zero();
308
309 for (int ialoc = 0; ialoc < ctx_.unit_cell().spl_num_atoms().local_size(); ialoc++) {
310 int ia = ctx_.unit_cell().spl_num_atoms(ialoc);
311 auto g = gradient(potential_.hartree_potential_mt(ialoc));
312 for (int x = 0; x < 3; x++) {
313 forces_hf_(x, ia) = ctx_.unit_cell().atom(ia).zn() * g[x](0, 0) * y00;
314 }
315 }
316 ctx_.comm().allreduce(&forces_hf_(0, 0), (int)forces_hf_.size());
317 symmetrize(forces_hf_);
318
319 if (ctx_.cfg().control().verbosity() > 2 && ctx_.comm().rank() == 0) {
320 std::printf("H-F force\n");
321 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
322 std::printf("ia : %i, Hellmann–Feynman : %12.6f %12.6f %12.6f\n", ia, forces_hf_(0, ia), forces_hf_(1, ia),
323 forces_hf_(2, ia));
324 }
325 }
326
327 return forces_hf_;
328}

330mdarray<double, 2> const& Force::calc_forces_hubbard()
331{
332 PROFILE("sirius::Force::hubbard_force");
333 forces_hubbard_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
334 forces_hubbard_.zero();
335
336 if (ctx_.hubbard_correction()) {
337 /* recompute the hubbard potential */
338 ::sirius::generate_potential(density_.occupation_matrix(), potential_.hubbard_potential());
339
340 Q_operator<double> q_op(ctx_);
341
342 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
343
344 int ik = kset_.spl_num_kpoints(ikloc);
345 auto kp = kset_.get<double>(ik);
346 kp->beta_projectors().prepare();
347 if (ctx_.num_mag_dims() == 3) {
348 TERMINATE("Hubbard forces are only implemented for the simple hubbard correction.");
349 }
350 hubbard_force_add_k_contribution_collinear(*kp, q_op, forces_hubbard_);
351 kp->beta_projectors().dismiss();
352 }
353
354 /* global reduction */
355 kset_.comm().allreduce(forces_hubbard_.at(memory_t::host), 3 * ctx_.unit_cell().num_atoms());
356 }
357
358 return forces_hubbard_;
359}

361mdarray<double, 2> const& Force::calc_forces_ewald()
362{
363 PROFILE("sirius::Force::calc_forces_ewald");
364
365 forces_ewald_ = sddk::mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
366 forces_ewald_.zero();
367
368 Unit_cell& unit_cell = ctx_.unit_cell();
369
370 double alpha = ctx_.ewald_lambda();
371
372 double prefac = (ctx_.gvec().reduced() ? 4.0 : 2.0) * (twopi / unit_cell.omega());
373
374 int ig0 = ctx_.gvec().skip_g0();
375
376 sddk::mdarray<double_complex, 1> rho_tmp(ctx_.gvec().count());
377 rho_tmp.zero();
378 #pragma omp parallel for schedule(static)
379 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
380 int ig = ctx_.gvec().offset() + igloc;
381
382 double_complex rho(0, 0);
383
384 for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
385 rho += ctx_.gvec_phase_factor(ig, ja) * static_cast<double>(unit_cell.atom(ja).zn());
386 }
387
388 rho_tmp[igloc] = std::conj(rho);
389 }
390
391 #pragma omp parallel for
392 for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
393 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
394 int ig = ctx_.gvec().offset() + igloc;
395
396 double g2 = std::pow(ctx_.gvec().gvec_len(ig), 2);
397
398 /* cartesian form for getting cartesian force components */
399 auto gvec_cart = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
400
401 double scalar_part = prefac * (rho_tmp[igloc] * ctx_.gvec_phase_factor(ig, ja)).imag() *
402 static_cast<double>(unit_cell.atom(ja).zn()) * std::exp(-g2 / (4 * alpha)) / g2;
403
404 for (int x : {0, 1, 2}) {
405 forces_ewald_(x, ja) += scalar_part * gvec_cart[x];
406 }
407 }
408 }
409
410 ctx_.comm().allreduce(&forces_ewald_(0, 0), 3 * ctx_.unit_cell().num_atoms());
411
412 double invpi = 1. / pi;
413
414 #pragma omp parallel for
415 for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
416 for (int i = 1; i < unit_cell.num_nearest_neighbours(ia); i++) {
417 int ja = unit_cell.nearest_neighbour(i, ia).atom_id;
418
419 double d = unit_cell.nearest_neighbour(i, ia).distance;
420 double d2 = d * d;
421
422 auto t = dot(unit_cell.lattice_vectors(), vector3d<int>(unit_cell.nearest_neighbour(i, ia).translation));
423
424 double scalar_part =
425 static_cast<double>(unit_cell.atom(ia).zn() * unit_cell.atom(ja).zn()) / d2 *
426 (std::erfc(std::sqrt(alpha) * d) / d + 2.0 * std::sqrt(alpha * invpi) * std::exp(-d2 * alpha));
427
428 for (int x : {0, 1, 2}) {
429 forces_ewald_(x, ia) += scalar_part * t[x];
430 }
431 }
432 }
433
434 symmetrize(forces_ewald_);
435
436 return forces_ewald_;
437}

439mdarray<double, 2> const& Force::calc_forces_us()
440{
441 PROFILE("sirius::Force::calc_forces_us");
442
443 forces_us_ = sddk::mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
444 forces_us_.zero();
445
446 potential_.fft_transform(-1);
447
448 Unit_cell& unit_cell = ctx_.unit_cell();
449
450 double reduce_g_fact = ctx_.gvec().reduced() ? 2.0 : 1.0;
451
452 linalg_t la{linalg_t::none};
453
454 memory_pool* mp{nullptr};
455 switch (ctx_.processing_unit()) {
456 case device_t::CPU: {
457 mp = &ctx_.mem_pool(memory_t::host);
458 la = linalg_t::blas;
459 break;
460 }
461 case device_t::GPU: {
462 mp = &ctx_.mem_pool(memory_t::host_pinned);
463 la = linalg_t::spla;
464 break;
465 }
466 }
467
468 /* over atom types */
469 for (int iat = 0; iat < unit_cell.num_atom_types(); iat++) {
470 auto& atom_type = unit_cell.atom_type(iat);
471
472 auto aug_op = ctx_.augmentation_op(iat);
473
474 if (!aug_op) {
475 continue;
476 }
477
478 int nbf = atom_type.mt_basis_size();
479
480 /* get auxiliary density matrix */
481 auto dm = density_.density_matrix_aux(density_.density_matrix(), iat);
482
483 sddk::mdarray<double, 2> v_tmp(atom_type.num_atoms(), ctx_.gvec().count() * 2, *mp);
484 sddk::mdarray<double, 2> tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms(), *mp);
485
486 /* over spin components, can be from 1 to 4*/
487 for (int ispin = 0; ispin < ctx_.num_mag_dims() + 1; ispin++) {
488 /* over 3 components of the force/G - vectors */
489 for (int ivec = 0; ivec < 3; ivec++) {
490 /* over local rank G vectors */
491 #pragma omp parallel for schedule(static)
492 for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
493 int ig = ctx_.gvec().offset() + igloc;
494 auto gvc = ctx_.gvec().gvec_cart<index_domain_t::local>(igloc);
495 for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
496 /* here we write in v_tmp -i * G * exp[ iGRn] Veff(G)
497 * but in formula we have i * G * exp[-iGRn] Veff*(G)
498 * the differences because we unfold complex array in the real one
499 * and need negative imagine part due to a multiplication law of complex numbers */
500 auto z = double_complex(0, -gvc[ivec]) * ctx_.gvec_phase_factor(ig, atom_type.atom_id(ia)) *
501 potential_.component(ispin).f_pw_local(igloc);
502 v_tmp(ia, 2 * igloc) = z.real();
503 v_tmp(ia, 2 * igloc + 1) = z.imag();
504 }
505 }
506
507 /* multiply tmp matrices, or sum over G */
508 linalg(la).gemm('N', 'T', nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
509 &linalg_const<double>::one(),
510 aug_op->q_pw().at(memory_t::host), aug_op->q_pw().ld(),
511 v_tmp.at(memory_t::host), v_tmp.ld(),
512 &linalg_const<double>::zero(),
513 tmp.at(memory_t::host), tmp.ld());
514
515 #pragma omp parallel for
516 for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
517 for (int i = 0; i < nbf * (nbf + 1) / 2; i++) {
518 forces_us_(ivec, atom_type.atom_id(ia)) += ctx_.unit_cell().omega() * reduce_g_fact *
519 dm(i, ia, ispin) * aug_op->sym_weight(i) * tmp(i, ia);
520 }
521 }
522 }
523 }
524 }
525
526 ctx_.comm().allreduce(&forces_us_(0, 0), 3 * ctx_.unit_cell().num_atoms());
527
528 symmetrize(forces_us_);
529
530 return forces_us_;
531}

533mdarray<double, 2> const& Force::calc_forces_scf_corr()
534{
535 PROFILE("sirius::Force::calc_forces_scf_corr");
536
537 auto q = ctx_.gvec().shells_len();
538 /* get form-factors for all G shells */
539 auto ff = ctx_.ps_rho_ri().values(q, ctx_.comm());
540
541 forces_scf_corr_ = mdarray<double, 2>(3, ctx_.unit_cell().num_atoms());
542 forces_scf_corr_.zero();
543
544 /* get main arrays */
545 auto& dveff = potential_.dveff();
546
547 Unit_cell& unit_cell = ctx_.unit_cell();
548
549 Gvec const& gvec = ctx_.gvec();
550
551 int gvec_count = gvec.count();
552 int gvec_offset = gvec.offset();
553
554 double fact = gvec.reduced() ? 2.0 : 1.0;
555
556 int ig0 = ctx_.gvec().skip_g0();
557
558 #pragma omp parallel for
559 for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
560 Atom& atom = unit_cell.atom(ia);
561
562 int iat = atom.type_id();
563
564 for (int igloc = ig0; igloc < gvec_count; igloc++) {
565 int ig = gvec_offset + igloc;
566 int igsh = ctx_.gvec().shell(ig);
567
568 /* cartesian form for getting cartesian force components */
569 auto gvec_cart = gvec.gvec_cart<index_domain_t::local>(igloc);
570
571 /* scalar part of a force without multipying by G-vector */
572 double_complex z = fact * fourpi * ff(igsh, iat) *
573 std::conj(dveff.f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia));
574
575 /* get force components multiplying by cartesian G-vector */
576 for (int x : {0, 1, 2}) {
577 forces_scf_corr_(x, ia) -= (gvec_cart[x] * z).imag();
578 }
579 }
580 }
581 ctx_.comm().allreduce(&forces_scf_corr_(0, 0), 3 * ctx_.unit_cell().num_atoms());
582
583 symmetrize(forces_scf_corr_);
584
585 return forces_scf_corr_;
586}

647void Force::hubbard_force_add_k_contribution_collinear(K_point<double>& kp__, Q_operator<double>& q_op__, mdarray<double, 2>& forceh_)
648{
649 mdarray<double_complex, 6> dn(potential_.U().max_number_of_orbitals_per_atom(), potential_.U().max_number_of_orbitals_per_atom(), 2,
650 ctx_.unit_cell().num_atoms(), 3, ctx_.unit_cell().num_atoms());
651
652 potential_.U().compute_occupancies_derivatives(kp__, q_op__, dn);
653
654 #pragma omp parallel for
655 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
656 /* compute the derivative of the occupancies numbers */
657 for (int dir = 0; dir < 3; dir++) {
658 double_complex d{0.0};
659 for (int ia1 = 0; ia1 < ctx_.unit_cell().num_atoms(); ia1++) {
660 auto const& atom = ctx_.unit_cell().atom(ia1);
661 if (atom.type().hubbard_correction()) {
662 int const lmax_at = 2 * atom.type().lo_descriptor_hub(0).l + 1;
663 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
664 for (int m1 = 0; m1 < lmax_at; m1++) {
665 for (int m2 = 0; m2 < lmax_at; m2++) {
666 d += potential_.hubbard_potential().local(ia1)(m2, m1, ispn) *
667 dn(m1, m2, ispn, ia1, dir, ia);
668 }
669 }
670 }
671 }
672 }
673 forceh_(dir, ia) -= std::real(d);
674 }
675 }

747void Force::add_ibs_force(K_point<double>* kp__, Hamiltonian_k<double>& Hk__, mdarray<double, 2>& ffac__, mdarray<double, 2>& forcek__) const
748{
749 PROFILE("sirius::Force::ibs_force");
750
751 auto& uc = ctx_.unit_cell();
752
753 auto& bg = ctx_.blacs_grid();
754
755 auto bs = ctx_.cyclic_block_size();
756
757 auto nfv = ctx_.num_fv_states();
758
759 auto ngklo = kp__->gklo_basis_size();
760
761 /* compute density matrix for a k-point */
762 dmatrix<double_complex> dm(nfv, nfv, bg, bs, bs);
763 compute_dmat(kp__, dm);
764
765 /* first-variational eigen-vectors in scalapack distribution */
766 auto& fv_evec = kp__->fv_eigen_vectors();
767
768 dmatrix<double_complex> h(ngklo, ngklo, bg, bs, bs);
769 dmatrix<double_complex> o(ngklo, ngklo, bg, bs, bs);
770
771 dmatrix<double_complex> h1(ngklo, ngklo, bg, bs, bs);
772 dmatrix<double_complex> o1(ngklo, ngklo, bg, bs, bs);
773
774 dmatrix<double_complex> zm1(ngklo, nfv, bg, bs, bs);
775 dmatrix<double_complex> zf(nfv, nfv, bg, bs, bs);
776
777 mdarray<double_complex, 2> alm_row(kp__->num_gkvec_row(), uc.max_mt_aw_basis_size());
778 mdarray<double_complex, 2> alm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
779 mdarray<double_complex, 2> halm_col(kp__->num_gkvec_col(), uc.max_mt_aw_basis_size());
780
781 for (int ia = 0; ia < uc.num_atoms(); ia++) {
782 h.zero();
783 o.zero();
784
785 auto& atom = uc.atom(ia);
786 auto& type = atom.type();
787
788 /* generate matching coefficients for current atom */
789 kp__->alm_coeffs_row().generate<true>(atom, alm_row);
790 kp__->alm_coeffs_col().generate<false>(atom, alm_col);
791
792 /* setup apw-lo and lo-apw blocks */
793 Hk__.set_fv_h_o_apw_lo(atom, ia, alm_row, alm_col, h, o);
794
795 /* apply MT Hamiltonian to column coefficients */
796 Hk__.H0().apply_hmt_to_apw<spin_block_t::nm>(atom, kp__->num_gkvec_col(), alm_col, halm_col);
797
798 /* apw-apw block of the overlap matrix */
799 linalg(linalg_t::blas).gemm('N', 'T', kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
800 &linalg_const<double_complex>::one(), alm_row.at(memory_t::host), alm_row.ld(),
801 alm_col.at(memory_t::host), alm_col.ld(), &linalg_const<double_complex>::zero(),
802 o.at(memory_t::host), o.ld());
803
804 /* apw-apw block of the Hamiltonian matrix */
805 linalg(linalg_t::blas).gemm('N', 'T', kp__->num_gkvec_row(), kp__->num_gkvec_col(), type.mt_aw_basis_size(),
806 &linalg_const<double_complex>::one(), alm_row.at(memory_t::host), alm_row.ld(), halm_col.at(memory_t::host),
807 halm_col.ld(), &linalg_const<double_complex>::zero(), h.at(memory_t::host), h.ld());
808
809 int iat = type.id();
810
811 for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) { // loop over columns
812 auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
813 auto gkvec_col_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_col(igk_col));
814 for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) { // for each column loop over rows
815 auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
816 auto gkvec_row_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_row(igk_row));
817
818 int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
819
820 int igs = ctx_.gvec().shell(ig12);
821
822 auto zt = std::conj(ctx_.gvec_phase_factor(ig12, ia)) * ffac__(iat, igs) * fourpi / uc.omega();
823
824 double t1 = 0.5 * dot(gkvec_row_cart, gkvec_col_cart);
825
826 h(igk_row, igk_col) -= t1 * zt;
827 o(igk_row, igk_col) -= zt;
828 }
829 }
830
831 for (int x = 0; x < 3; x++) {
832 for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) { // loop over columns
833 auto gvec_col = kp__->gkvec().gvec(kp__->igk_col(igk_col));
834 for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) { // loop over rows
835 auto gvec_row = kp__->gkvec().gvec(kp__->igk_row(igk_row));
836 /* compute index of G-G' */
837 int ig12 = ctx_.gvec().index_g12(gvec_row, gvec_col);
838 /* get G-G' */
839 auto vg = ctx_.gvec().gvec_cart<index_domain_t::global>(ig12);
840 /* multiply by i(G-G') */
841 h1(igk_row, igk_col) = double_complex(0.0, vg[x]) * h(igk_row, igk_col);
842 /* multiply by i(G-G') */
843 o1(igk_row, igk_col) = double_complex(0.0, vg[x]) * o(igk_row, igk_col);
844 }
845 }
846
847 for (int icol = 0; icol < kp__->num_lo_col(); icol++) {
848 for (int igk_row = 0; igk_row < kp__->num_gkvec_row(); igk_row++) {
849 auto gkvec_row_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_row(igk_row));
850 /* multiply by i(G+k) */
851 h1(igk_row, icol + kp__->num_gkvec_col()) =
852 double_complex(0.0, gkvec_row_cart[x]) * h(igk_row, icol + kp__->num_gkvec_col());
853 /* multiply by i(G+k) */
854 o1(igk_row, icol + kp__->num_gkvec_col()) =
855 double_complex(0.0, gkvec_row_cart[x]) * o(igk_row, icol + kp__->num_gkvec_col());
856 }
857 }
858
859 for (int irow = 0; irow < kp__->num_lo_row(); irow++) {
860 for (int igk_col = 0; igk_col < kp__->num_gkvec_col(); igk_col++) {
861 auto gkvec_col_cart = kp__->gkvec().gkvec_cart<index_domain_t::global>(kp__->igk_col(igk_col));
862 /* multiply by i(G+k) */
863 h1(irow + kp__->num_gkvec_row(), igk_col) =
864 double_complex(0.0, -gkvec_col_cart[x]) * h(irow + kp__->num_gkvec_row(), igk_col);
865 /* multiply by i(G+k) */
866 o1(irow + kp__->num_gkvec_row(), igk_col) =
867 double_complex(0.0, -gkvec_col_cart[x]) * o(irow + kp__->num_gkvec_row(), igk_col);
868 }
869 }
870
871 /* zm1 = dO * V */
872 linalg(linalg_t::scalapack).gemm('N', 'N', ngklo, nfv, ngklo, &linalg_const<double_complex>::one(),
873 o1, 0, 0, fv_evec, 0, 0, &linalg_const<double_complex>::zero(), zm1, 0, 0);
874 /* multiply by energy: zm1 = E * (dO * V) */
875 for (int i = 0; i < zm1.num_cols_local(); i++) {
876 int ist = zm1.icol(i);
877 for (int j = 0; j < kp__->gklo_basis_size_row(); j++) {
878 zm1(j, i) *= kp__->fv_eigen_value(ist);
879 }
880 }
881 /* compute zm1 = dH * V - E * (dO * V) */
882 linalg(linalg_t::scalapack).gemm('N', 'N', ngklo, nfv, ngklo, &linalg_const<double_complex>::one(),
883 h1, 0, 0, fv_evec, 0, 0, &linalg_const<double_complex>::m_one(), zm1, 0, 0);
884
885 /* compute zf = V^{+} * zm1 = V^{+} * (dH * V - E * (dO * V)) */
886 linalg(linalg_t::scalapack).gemm('C', 'N', nfv, nfv, ngklo, &linalg_const<double_complex>::one(),
887 fv_evec, 0, 0, zm1, 0, 0, &linalg_const<double_complex>::zero(), zf, 0, 0);
888
889 for (int i = 0; i < dm.num_cols_local(); i++) {
890 for (int j = 0; j < dm.num_rows_local(); j++) {
891 forcek__(x, ia) += kp__->weight() * std::real(dm(j, i) * zf(j, i));
892 }
893 }
894 }
895 } // ia
896}
