176  /// Stop the timer and update the statistics.
177  double stop()
178  {
179  if (!active_) {
180  return 0;
181  }
182 
183  /* remove this timer name from the list; now last element contains
184  the name of the parent timer */
185  stack().pop_back();
186 
187  /* measure the time difference */
188  auto t2 = std::chrono::high_resolution_clock::now();
189  auto tdiff = std::chrono::duration_cast<std::chrono::duration<double>>(t2 - starting_time_);
190  double val = tdiff.count();
191 
192  auto& ts = timer_values(label_);
193 #ifdef __TIMER_SEQUENCE
194  ts.sequence.push_back(starting_time_);
195  ts.sequence.push_back(t2);
196 #endif
197  ts.min_val = std::min(ts.min_val, val);
198  ts.max_val = std::max(ts.max_val, val);
199  ts.tot_val += val;
200  ts.count++;
201 
202  if (stack().size() != 0) {
203  /* last element contains the name of the parent timer */
204  auto parent_label = stack().back();
205  /* add value to the parent timer */
206  if (timer_values_ex().count(parent_label) == 0) {
207  timer_values_ex()[parent_label] = std::map<std::string, double>();
208  }
209  if (timer_values_ex()[parent_label].count(label_) == 0) {
210  timer_values_ex()[parent_label][label_] = 0;
211  }
212  timer_values_ex()[parent_label][label_] += val;
213  }
214 #if defined(__APEX)
215  apex::stop(apex_p_);
216 #endif
217  active_ = false;
218  return val;
219  }

270  /* collect external times */
271  double te{0};
272  for (auto& it2: timer_values_ex()[it.first]) {
273  te += it2.second;
274  }
275  double f = it.second.tot_val / ttot;
276  if (f > 0.01) {

278  it.first.c_str(), it.second.tot_val, (it.second.tot_val - te) / it.second.tot_val * 100, f * 100);
279 
280  std::vector<std::pair<double, std::string>> tmp;
281 
282  for (auto& it2: timer_values_ex()[it.first]) {
283  tmp.push_back(std::pair<double, std::string>(it2.second / it.second.tot_val, it2.first));
284  }
285  std::sort(tmp.rbegin(), tmp.rend());
286  for (auto& e: tmp) {

294  static nlohmann::json serialize()
295  {
296  nlohmann::json dict;
297 
298  /* collect local timers */
299  for (auto& it: timer::timer_values()) {
300  timer_stats_t ts;
301  nlohmann::json node;
302  node["count"] = it.second.count;
303  node["total"] = it.second.tot_val;
304  node["min"] = it.second.min_val;
305  node["max"] = it.second.max_val;
306  node["avg"] = it.second.tot_val / it.second.count;
307 #ifdef __TIMER_SEQUENCE
308  std::vector<double> tseq;
309  for (auto& s: it.second.sequence) {
310  auto tdiff = std::chrono::duration_cast<std::chrono::duration<double>>(s - global_starting_time());
311  tseq.push_back(tdiff.count());
312  }
313  node["sequence"] = tseq;
314 #endif
315  dict[it.first] = node;
316  }
317  return std::move(dict);
318  }

320  static nlohmann::json serialize_tree()
321  {
322  nlohmann::json dict;
323 
324  if (!timer_values().count(main_timer_label)) {
325  return {};
326  }
327  /* total execution time */
328  double ttot = timer_values()[main_timer_label].tot_val;
329 
330  /* loop over the timer; iterator `it` is a <key, valu> pair */
331  for (auto& it: timer_values()) {
332  /* if this timer is a parent timer for somebody and timer has a non-negligible contribution */
333  if (timer_values_ex().count(it.first) && (it.second.tot_val / ttot) > 0.01) {
334  /* collect child (external) times */
335  double te{0};
336  for (auto& it2: timer_values_ex()[it.first]) {
337  te += it2.second;
338  }
339  nlohmann::json node;
340 
341  node["total_time"] = it.second.tot_val;
342  node["child_time"] = te;
343  node["self_time"] = it.second.tot_val - te;
344  node["fraction_of_global_time"] = it.second.tot_val / ttot;
345  node["call"] = {};
346 
347  /* add all children values */
348  for (auto& it2: timer_values_ex()[it.first]) {
349  nlohmann::json n;
350  n["time"] = timer_values_ex()[it.first][it2.first];
351  n["fraction_of_parent"] = it2.second / it.second.tot_val;
352  node["call"][it2.first] = n;
353  }
354  dict[it.first] = node;
355  }
356  }
357 
358  return std::move(dict);
359  }
