109  /// Stop the timer and update the statistics.
110  double stop()
111  {
112  if (!active_) {
113  return 0;
114  }
115 
116  /* remove this timer name from the list; now last element contains
117  the name of the parent timer */
118  stack().pop_back();
119 
120  /* measure the time difference */
121  auto t2 = std::chrono::high_resolution_clock::now();
122  auto tdiff = std::chrono::duration_cast<std::chrono::duration<double>>(t2 - starting_time_);
123  double val = tdiff.count();
124 
125  auto& ts = timer_values(label_);
126 #ifdef __TIMER_SEQUENCE
127  auto abs_time = std::chrono::duration_cast<std::chrono::duration<double>>(starting_time_.time_since_epoch());
128  double abs_time_val = abs_time.count();
129  ts.sequence.push_back(abs_time_val);
130  abs_time = std::chrono::duration_cast<std::chrono::duration<double>>(t2.time_since_epoch());
131  abs_time_val = abs_time.count();
132  ts.sequence.push_back(abs_time_val);
133 #endif
134  ts.min_val = std::min(ts.min_val, val);
135  ts.max_val = std::max(ts.max_val, val);
136  ts.tot_val += val;
137  ts.count++;
138 
139  if (stack().size() != 0) {
140  /* last element contains the name of the parent timer */
141  auto parent_label = stack().back();
142  /* add value to the parent timer */
143  if (timer_values_ex().count(parent_label) == 0) {
144  timer_values_ex()[parent_label] = std::map<std::string, double>();
145  }
146  if (timer_values_ex()[parent_label].count(label_) == 0) {
147  timer_values_ex()[parent_label][label_] = 0;
148  }
149  timer_values_ex()[parent_label][label_] += val;
150  }
151  active_ = false;
152  return val;
153  }
