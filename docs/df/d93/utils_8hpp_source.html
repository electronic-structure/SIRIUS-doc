153 /// Pack two indices into one for symmetric matrices.
154 inline int packed_index(int i__, int j__)
155 {
156  /* suppose we have a symmetric matrix: M_{ij} = M_{ji}
157  j
158  +-------+
159  | + + + |
160  i| + + | -> idx = j * (j + 1) / 2 + i for i <= j
161  | + |
162  +-------+
163 
164  i, j are row and column indices
165  */
166 
167  if (i__ > j__) {
168  std::swap(i__, j__);
169  }
170  return j__ * (j__ + 1) / 2 + i__;
171 }

173 inline std::string double_to_string(double val, int precision = -1)
174 {
175  char buf[100];
176 
177  double abs_val = std::abs(val);
178 
179  if (precision == -1) {
180  if (abs_val > 1.0) {
181  precision = 6;
182  } else if (abs_val > 1e-14) {
183  precision = int(-std::log(abs_val) / std::log(10.0)) + 7;
184  } else {
185  return std::string("0.0");
186  }
187  }
188 
189  std::stringstream fmt;
190  fmt << "%." << precision << "f";
191 
192  int len = snprintf(buf, 100, fmt.str().c_str(), val);
193  for (int i = len - 1; i >= 1; i--) {
194  if (buf[i] == '0' && buf[i - 1] == '0') {
195  buf[i] = 0;
196  } else {
197  break;
198  }
199  }
200  return std::string(buf);
201 }

245 {

266 inline double confined_polynomial(double r, double R, int p1, int p2, int dm)
267 {
268  double t = 1.0 - std::pow(r / R, 2);
269  switch (dm) {
270  case 0: {
271  return (std::pow(r, p1) * std::pow(t, p2));
272  }
273  case 2: {
274  return (-4 * p1 * p2 * std::pow(r, p1) * std::pow(t, p2 - 1) / std::pow(R, 2) +
275  p1 * (p1 - 1) * std::pow(r, p1 - 2) * std::pow(t, p2) +
276  std::pow(r, p1) * (4 * (p2 - 1) * p2 * std::pow(r, 2) * std::pow(t, p2 - 2) / std::pow(R, 4) -
277  2 * p2 * std::pow(t, p2 - 1) / std::pow(R, 2)));
278  }
279  default: {
280  TERMINATE("wrong derivative order");
281  return 0.0;
282  }
283  }

307  std::stringstream s;
