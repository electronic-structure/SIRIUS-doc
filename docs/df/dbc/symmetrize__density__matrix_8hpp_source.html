25 inline void Density::symmetrize_density_matrix()
26 {
27  PROFILE("sirius::Density::symmetrize_density_matrix");
28 
29  auto& sym = unit_cell_.symmetry();
30 
31  int ndm = ctx_.num_mag_comp();
32 
33  mdarray<double_complex, 4> dm(unit_cell_.max_mt_basis_size(), unit_cell_.max_mt_basis_size(),
34  ndm, unit_cell_.num_atoms());
35  dm.zero();
36 
37  int lmax = unit_cell_.lmax();
38  int lmmax = utils::lmmax(lmax);
39 
40  mdarray<double, 2> rotm(lmmax, lmmax);
41 
42  double alpha = 1.0 / double(sym.num_mag_sym());
43 
44  for (int i = 0; i < sym.num_mag_sym(); i++) {
45  int pr = sym.magnetic_group_symmetry(i).spg_op.proper;
46  auto eang = sym.magnetic_group_symmetry(i).spg_op.euler_angles;
47  int isym = sym.magnetic_group_symmetry(i).isym;
48  SHT::rotation_matrix(lmax, eang, pr, rotm);
49  auto spin_rot_su2 = SHT::rotation_matrix_su2(sym.magnetic_group_symmetry(i).spin_rotation);
50 
51  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
52  auto& atom_type = unit_cell_.atom(ia).type();
53  int ja = sym.sym_table(ia, isym);
54 
55  for (int xi1 = 0; xi1 < unit_cell_.atom(ia).mt_basis_size(); xi1++) {
56  int l1 = atom_type.indexb(xi1).l;
57  int lm1 = atom_type.indexb(xi1).lm;
58  int o1 = atom_type.indexb(xi1).order;
59 
60  for (int xi2 = 0; xi2 < unit_cell_.atom(ia).mt_basis_size(); xi2++) {
61  int l2 = atom_type.indexb(xi2).l;
62  int lm2 = atom_type.indexb(xi2).lm;
63  int o2 = atom_type.indexb(xi2).order;
64 
65  std::array<double_complex, 3> dm_rot_spatial = {0, 0, 0};
66 
67  for (int j = 0; j < ndm; j++) {
68  for (int m3 = -l1; m3 <= l1; m3++) {
69  int lm3 = utils::lm(l1, m3);
70  int xi3 = atom_type.indexb().index_by_lm_order(lm3, o1);
71  for (int m4 = -l2; m4 <= l2; m4++) {
72  int lm4 = utils::lm(l2, m4);
73  int xi4 = atom_type.indexb().index_by_lm_order(lm4, o2);
74  dm_rot_spatial[j] += density_matrix_(xi3, xi4, j, ja) * rotm(lm1, lm3) * rotm(lm2, lm4) * alpha;
75  }
76  }
77  }
78 
79  /* magnetic symmetrization */
80  if (ndm == 1) {
81  dm(xi1, xi2, 0, ia) += dm_rot_spatial[0];
82  } else {
83  double_complex spin_dm[2][2] = {
84  {dm_rot_spatial[0], dm_rot_spatial[2]},
85  {std::conj(dm_rot_spatial[2]), dm_rot_spatial[1]}};
86 
87  /* spin blocks of density matrix are: uu, dd, ud
88  the mapping from linear index (0, 1, 2) of density matrix components is:
89  for the first spin index: k & 1, i.e. (0, 1, 2) -> (0, 1, 0)
90  for the second spin index: min(k, 1), i.e. (0, 1, 2) -> (0, 1, 1)
91  */
92  for (int k = 0; k < ndm; k++) {
93  for (int is = 0; is < 2; is++) {
94  for (int js = 0; js < 2; js++) {
95  dm(xi1, xi2, k, ia) += spin_rot_su2(k & 1, is) * spin_dm[is][js] * std::conj(spin_rot_su2(std::min(k, 1), js));
96  }
97  }
98  }
99  }
100  }
101  }
102  }
103  }
104 
105  dm >> density_matrix_;
106 
107  if (ctx_.control().print_checksum_ && ctx_.comm().rank() == 0) {
108  auto cs = dm.checksum();
109  utils::print_checksum("density_matrix", cs);
110  //for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
111  // auto cs = mdarray<double_complex, 1>(&dm(0, 0, 0, ia), dm.size(0) * dm.size(1) * dm.size(2)).checksum();
112  // DUMP("checksum(density_matrix(%i)): %20.14f %20.14f", ia, cs.real(), cs.imag());
113  //}
114  }
115 
116  if (ctx_.control().print_hash_ && ctx_.comm().rank() == 0) {
117  auto h = dm.hash();
118  utils::print_hash("density_matrix", h);
119  }
120 }
