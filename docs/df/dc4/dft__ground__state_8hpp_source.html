40/// The whole DFT ground state implementation.
41/** The DFT cycle consists of four basic steps: solution of the Kohn-Sham equations, summation of the occupied states
42 * in order to get a system's charge density and magnetization, mixing and finally gneneration of the effective
43 * potential and effective magnetic field. \image html dft_cycle.png "DFT self-consistency cycle"
44 */
45class DFT_ground_state
46{
47 private:
48 /// Context of simulation.
49 Simulation_context& ctx_;
50
51 /// Set of k-points that are used to generate density.
52 K_point_set& kset_;
53
54 /// Alias of the unit cell.
55 Unit_cell& unit_cell_;
56
57 /// Instance of the Potential class.
58 Potential potential_;
59
60 /// Instance of the Density class.
61 Density density_;
62
63 /// Lattice stress.
64 Stress stress_;
65
66 /// Atomic forces.
67 Force forces_;
68
69 /// Store Ewald energy which is computed once and which doesn't change during the run.
70 double ewald_energy_{0};
71
72 /// Correction to total energy from the SCF density minimisation.
73 double scf_energy_{0};
74
75 public:
76 /// Constructor.
77 DFT_ground_state(K_point_set& kset__)
78 : ctx_(kset__.ctx())
79 , kset_(kset__)
80 , unit_cell_(ctx_.unit_cell())
81 , potential_(ctx_)
82 , density_(ctx_)
83 , stress_(ctx_, density_, potential_, kset__)
84 , forces_(ctx_, density_, potential_, kset__)
85
86 {
87 if (!ctx_.full_potential()) {
88 ewald_energy_ = sirius::ewald_energy(ctx_, ctx_.gvec(), ctx_.unit_cell());
89 }
90 }
91 ~DFT_ground_state()
92 {
93 int n = ctx_.num_loc_op_applied();
94 kset_.comm().allreduce(&n, 1);
95 ctx_.message(2, __function_name__, "local op. applied: %i\n", n);
96 double d = ctx_.evp_work_count();
97 kset_.comm().allreduce(&d, 1);
98 ctx_.message(2, __function_name__, "evp. work count: %f\n", d);
99 n = ctx_.num_itsol_steps();
100 kset_.comm().allreduce(&n, 1);
101 ctx_.message(2, __function_name__, "number of iterative solver steps: %i\n", n);
102 }

171/** \page dft Spin-polarized DFT
172 * \section s1dft Preliminary notes
173 *
174 * \note Here and below sybol \f$ {\boldsymbol \sigma} \f$ is reserved for the vector of Pauli matrices. Spin
175 * components are labeled with \f$ \alpha \f$ or \f$ \beta \f$.
176 *
177 * Wave-function of spin-1/2 particle is a two-component spinor:
178 * \f[
179 * {\bf \Psi}({\bf r}) = \left( \begin{array}{c} \psi^{\uparrow}({\bf r}) \\
180 * \psi^{\downarrow}({\bf r})
181 * \end{array}
182 * \right)
183 * \f]
184 * Operator of spin:
185 * \f[
186 * {\bf \hat S}=\frac{\hbar}{2}{\boldsymbol \sigma},
187 * \f]
188 * Pauli matrices:
189 * \f[
190 * \sigma_x=\left( \begin{array}{cc}
191 * 0 & 1 \\
192 * 1 & 0 \\ \end{array} \right) \,
193 * \sigma_y=\left( \begin{array}{cc}
194 * 0 & -i \\
195 * i & 0 \\ \end{array} \right) \,
196 * \sigma_z=\left( \begin{array}{cc}
197 * 1 & 0 \\
198 * 0 & -1 \\ \end{array} \right)
199 * \f]
200 *
201 * Spin moment of an electron in quantum state \f$ | \Psi \rangle \f$:
202 * \f[
203 * {\bf S}=\langle \Psi | {\bf \hat S} | \Psi \rangle = \frac{\hbar}{2} \langle \Psi | {\boldsymbol \sigma} | \Psi
204 * \rangle
205 * \f]
206 *
207 * Spin magnetic moment of electron:
208 * \f[
209 * {\bf \mu}_e=\gamma_e {\bf S},
210 * \f]
211 * where \f$ \gamma_e \f$ is the gyromagnetic ratio for the electron.
212 * \f[
213 * \gamma_e=-\frac{g_e \mu_B}{\hbar} \;\;\; \mu_B=\frac{e\hbar}{2m_ec}
214 * \f]
215 * Here \f$ g_e \f$ is a g-factor for electron which is ~2, and \f$ \mu_B \f$ - Bohr magneton (defined as positive
216 * constant). Finally, magnetic moment of electron: \f[
217 * {\bf \mu}_e=-{\bf \mu}_B \langle \Psi | {\boldsymbol \sigma} | \Psi \rangle
218 * \f]
219 * Potential energy of magnetic dipole in magnetic field:
220 * \f[
221 * U=-{\bf B}{\bf \mu}={\bf \mu}_B {\bf B} \langle \Psi | {\boldsymbol \sigma} | \Psi \rangle
222 * \f]
223 *
224 * \section s2dft Density and magnetization
225 * In magnetic calculations we have charge density \f$ \rho({\bf r}) \f$ (scalar function) and magnetization density
226 * \f$ {\bf m}({\bf r}) \f$ (vector function). Density is defined as:
227 * \f[
228 * \rho({\bf r}) = \sum_{j}^{occ} \Psi_{j}^{*}({\bf r}){\bf I} \Psi_{j}({\bf r}) =
229 * \sum_{j}^{occ} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) +
230 * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r})
231 * \f]
232 * Magnetization is defined as:
233 * \f[
234 * {\bf m}({\bf r}) = \sum_{j}^{occ} \Psi_{j}^{*}({\bf r}) {\boldsymbol \sigma} \Psi_{j}({\bf r})
235 * \f]
236 * \f[
237 * m_x({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) +
238 * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r})
239 * \f]
240 * \f[
241 * m_y({\bf r}) = \sum_{j}^{occ} -i \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) +
242 * i \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r})
243 * \f]
244 * \f[
245 * m_z({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) -
246 * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r})
247 * \f]
248 * Density and magnetization can be grouped into a \f$ 2 \times 2 \f$ density matrix, which is defined as:
249 * \f[
250 * {\boldsymbol \rho}({\bf r}) = \frac{1}{2} \Big( {\bf I}\rho({\bf r}) + {\boldsymbol \sigma} {\bf m}({\bf
251 * r})\Big) =
252 * \frac{1}{2} \left( \begin{array}{cc} \rho({\bf r}) + m_z({\bf r}) & m_x({\bf r}) - i m_y({\bf r}) \\
253 * m_x({\bf r}) + i m_y({\bf r}) & \rho({\bf r}) - m_z({\bf r}) \end{array}
254 * \right) = \sum_{j}^{occ} \left( \begin{array}{cc} \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) &
255 * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\uparrow}({\bf r}) \\
256 * \psi_{j}^{\uparrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r}) &
257 * \psi_{j}^{\downarrow *}({\bf r}) \psi_{j}^{\downarrow}({\bf r})
258 * \end{array} \right) \f] or simply \f[ \rho_{\alpha \beta}({\bf r}) = \sum_{j}^{occ} \psi_{j}^{\beta *}({\bf
259 * r})\psi_{j}^{\alpha}({\bf r}) \f] Pay attention to the order of spin indices in the \f$ 2 \times 2 \f$ density
260 * matrix. External potential \f$ v^{ext}({\bf r}) \f$ and external magnetic field \f$ {\bf B}^{ext}({\bf r}) \f$ can
261 * also be grouped into a \f$ 2 \times 2 \f$ matrix:
262 * \f[
263 * V_{\alpha\beta}^{ext}({\bf r})=\Big({\bf I}v^{ext}({\bf r})+\mu_{B}{\boldsymbol \sigma}{\bf B}^{ext}({\bf r})
264 * \Big) =
265 * \left( \begin{array}{cc} v^{ext}({\bf r})+\mu_{B}B_z^{ext}({\bf r}) & \mu_{B} \Big( B_x^{ext}({\bf
266 * r})-iB_y^{exp}({\bf r}) \Big) \\ \mu_{B} \Big( B_x^{ext}({\bf r})+iB_y^{ext}({\bf r}) \Big) & v^{ext}({\bf
267 * r})-\mu_{B}B_z^{ext}({\bf r}) \end{array} \right) \f] Let's check that potential energy in external fields can be
268 * written in the following way: \f[ E^{ext}=\int Tr \Big( {\boldsymbol \rho}({\bf r}) {\bf V}^{ext}({\bf r}) \Big)
269 * d{\bf r} = \sum_{\alpha\beta} \int \rho_{\alpha\beta}({\bf r})V_{\beta\alpha}^{ext}({\bf r}) d{\bf r} \f] (below \f$
270 * {\bf r} \f$, \f$ ext \f$ and \f$ \int d{\bf r} \f$ are dropped for simplicity) \f[ \begin{eqnarray}
271 * \rho_{11}V_{11} &= \frac{1}{2}(\rho+m_z)(v+\mu_{B}B_z) = \frac{1}{2}(\rho v +\mu_{B} \rho B_z + m_z v +
272 * \mu_{B}m_zB_z) \\
273 * \rho_{22}V_{22} &= \frac{1}{2}(\rho-m_z)(v-\mu_{B}B_z) = \frac{1}{2}(\rho v -\mu_{B} \rho B_z - m_z v +
274 * \mu_{B}m_zB_z) \\
275 * \rho_{12}V_{21} &= \frac{1}{2}(m_x-im_y)\Big( \mu_{B}( B_x+iB_y) \Big) =
276 * \frac{\mu_B}{2}(m_xB_x+im_xB_y-im_yB_x+m_yB_y) \\ \rho_{21}V_{12} &= \frac{1}{2}(m_x+im_y)\Big( \mu_{B}( B_x-iB_y)
277 * \Big) = \frac{\mu_B}{2}(m_xB_x-im_xB_y+im_yB_x+m_yB_y) \end{eqnarray} \f] The sum of this four terms will give
278 * exactly \f$ \int \rho({\bf r}) v^{ext}({\bf r}) + \mu_{B}{\bf m}({\bf r})
279 * {\bf B}^{ext}({\bf r}) d{\bf r}\f$
280 *
281 * \section s3dft Total energy variation
282 *
283 * To derive Kohn-Sham equations we need to write total energy functional of density matrix \f$ \rho_{\alpha\beta}({\bf
284 * r}) \f$: \f[ E^{tot}[\rho_{\alpha\beta}] = E^{kin}[\rho_{\alpha\beta}] + E^{H}[\rho_{\alpha\beta}] +
285 * E^{ext}[\rho_{\alpha\beta}] + E^{XC}[\rho_{\alpha\beta}] \f] Kinetic energy of non-interacting electrons is written
286 * in the following way: \f[ E^{kin}[\rho_{\alpha\beta}] \equiv E^{kin}[\Psi[\rho_{\alpha\beta}]] =
287 * -\frac{1}{2} \sum_{i}^{occ}\sum_{\alpha} \int \psi_{i}^{\alpha*}({\bf r}) \nabla^{2} \psi_{i}^{\alpha}({\bf
288 * r})d^3{\bf r} \f] Hartree energy: \f[ E^{H}[\rho_{\alpha\beta}]= \frac{1}{2} \iint \frac{\rho({\bf r})\rho({\bf
289 * r'})}{|{\bf r}-{\bf r'}|} d{\bf r} d{\bf r'} = \frac{1}{2} \iint \sum_{\alpha\beta}\delta_{\alpha\beta}
290 * \frac{\rho_{\alpha\beta}({\bf r}) \rho({\bf r'})}{|{\bf r}-{\bf r'}|} d{\bf r} d{\bf r'} \f] where \f$ \rho({\bf r})
291 * = Tr \rho_{\alpha\beta}({\bf r}) \f$.
292 *
293 * Exchange-correlation energy:
294 * \f[
295 * E^{XC}[\rho_{\alpha\beta}({\bf r})] \equiv E^{XC}[\rho({\bf r}),|{\bf m}({\bf r})|] =
296 * \int \rho({\bf r}) \eta_{XC}(\rho({\bf r}), m({\bf r})) d{\bf r} =
297 * \int \rho({\bf r}) \eta_{XC}(\rho^{\uparrow}({\bf r}), \rho_{\downarrow}({\bf r})) d{\bf r}
298 * \f]
299 * Now we can write the total energy variation over auxiliary orbitals with constrain of orbital normalization:
300 * \f[
301 * \frac{\delta \Big( E^{tot}+\varepsilon_i \big( 1-\sum_{\alpha} \int \psi^{\alpha*}_{i}({\bf r})
302 * \psi^{\alpha}_{i}({\bf r})d{\bf r} \big) \Big)} {\delta \psi_{i}^{\gamma*}({\bf r})} = 0
303 * \f]
304 * We will use the following chain rule:
305 * \f[
306 * \frac{\delta F[\rho_{\alpha\beta}]}{\delta \psi_{i}^{\gamma *}({\bf r})} =
307 * \sum_{\alpha' \beta'} \frac{\delta F[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\beta'}({\bf r})}
308 * \frac{\delta \rho_{\alpha'\beta'}({\bf r})}{\delta \psi_{i}^{\gamma *}({\bf r})} =
309 * \sum_{\alpha'\beta'}\frac{\delta F[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\beta'}({\bf r})}
310 * \psi_{i}^{\alpha'}({\bf r}) \delta_{\beta'\gamma} =
311 * \sum_{\alpha'}\frac{\delta F[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\gamma}({\bf r})}\psi_{i}^{\alpha'}({\bf
312 * r}) \f] Variation of the normalization integral: \f[ \frac{\delta \sum_{\alpha} \int \psi_{i}^{\alpha*}({\bf r})
313 * \psi_{i}^{\alpha}({\bf r}) d {\bf r} }{\delta \psi_{i}^{\gamma *}({\bf r})} = \psi_{i}^{\gamma}({\bf r}) \f]
314 * Variation of the kinetic energy functional:
315 * \f[
316 * \frac{\delta E^{kin}}{\delta \psi_{i}^{\gamma*}({\bf r})} = -\frac{1}{2} \sum_{\alpha} \nabla^{2}
317 * \psi_{i}^{\alpha}({\bf r}) \delta_{\alpha\gamma} = -\frac{1}{2}\nabla^{2}\psi_{i}^{\gamma}({\bf r}) \f] Variation of
318 * the Hartree energy functional: \f[ \frac{\delta E^{H}[\rho_{\alpha\beta}]}{\delta \psi_{i}^{\gamma *}({\bf r})} =
319 * \sum_{\alpha'} \sum_{\alpha\beta} \delta_{\alpha\beta} \frac{1}{2} \int \frac{ \rho({\bf r'})}{|{\bf r}-{\bf r'}|}
320 * d{\bf r'} \delta_{\alpha\alpha'}\delta_{\beta\gamma} \psi_{i}^{\alpha'}({\bf r}) = v^{H}({\bf r})
321 * \psi_{i}^{\gamma}({\bf r}) \f] Variation of the external energy functional: \f[ \frac{\delta
322 * E^{ext}[\rho_{\alpha\beta}]}{\delta \psi_{i}^{\gamma*}({\bf r}) } = \sum_{\alpha'} \sum_{\alpha\beta}
323 * V_{\beta\alpha}^{ext}({\bf r}) \delta_{\alpha\alpha'} \delta_{\beta\gamma} \psi_{i}^{\alpha'}({\bf r})= \sum_{\alpha}
324 * V_{\gamma\alpha}^{ext}({\bf r}) \psi_{i}^{\alpha}({\bf r}) \f]
325 *
326 * Variation of the exchange-correlation functional:
327 * \f[
328 * \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{ \delta \psi_{i}^{\gamma*}({\bf r}) } =
329 * \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{ \delta \rho({\bf r})} \frac{\delta \rho({\bf r})}{\delta
330 * \psi_{i}^{\gamma*}({\bf r})} + \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{ \delta m({\bf r})} \sum_{p=x,y,z}
331 * \frac{\delta m({\bf r})}{ \delta m_p({\bf r})}
332 * \frac{\delta m_p({\bf r})}{\delta \psi_{i}^{\gamma*}({\bf r})}
333 * \f]
334 * where \f$ m({\bf r}) = |{\bf m}({\bf r})|\f$ is the length of magnetization vector.
335 *
336 * First term:
337 * \f[
338 * \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{ \delta \rho({\bf r})} \frac{\delta \rho({\bf r})}{\delta
339 * \psi_{i}^{\gamma*}({\bf r})} = v^{XC}({\bf r}) \psi_{i}^{\gamma}({\bf r})
340 * \f]
341 * Second term:
342 * \f[
343 * \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{ \delta m({\bf r})} \sum_{p=x,y,z} \frac{\delta m({\bf r})}{ \delta
344 * m_p({\bf r})} \frac{\delta m_p({\bf r})}{\delta \psi_{i}^{\gamma*}({\bf r})} = B^{XC}({\bf r}) \hat {\bf m}
345 * \sum_{\beta} {\boldsymbol \sigma}_{\gamma \beta} \psi_{i}^{\beta}({\bf r}) \f] where \f$ B^{XC}({\bf r}) =
346 * \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{ \delta m({\bf r})} \f$, \f$ \hat {\bf m}({\bf r}) = \frac{\delta m({\bf
347 * r})}{ \delta {\bf m}({\bf r})} \f$ is the unit vector, parallel to \f$ {\bf m}({\bf r}) \f$ and \f$ {\bf m}({\bf r})
348 * = \sum_{i} \sum_{\alpha \beta} \psi_{i}^{\alpha*}({\bf r}) {\boldsymbol \sigma}_{\alpha \beta} \psi_i^{\beta}({\bf
349 * r}) \f$
350 *
351 * Similarly to external potential, exchange-correlation potential can be grouped into \f$ 2 \times 2 \f$ matrix::
352 * \f[
353 * \frac{\delta E^{XC}[\rho_{\alpha\beta}]}{\delta \rho_{\alpha'\beta'}({\bf r})} \equiv V^{XC}_{\beta'\alpha'}({\bf
354 * r}) = \Big( {\bf I}v^{XC}({\bf r}) + {\bf B}^{XC}({\bf r}) {\boldsymbol \sigma} \Big)_{\beta'\alpha'} \f] where
355 * \f${\bf B}^{XC}({\bf r}) = \hat {\bf m}({\bf r})B^{XC}({\bf r}) \f$ -- exchange-correlation magnetic field, parallel
356 * to \f$ {\bf m}({\bf r}) \f$ at each point in space. We can now collect \f$ v^{H}({\bf r}) \f$, \f$
357 * V_{\alpha\beta}^{ext}({\bf r}) \f$ and \f$V_{\alpha\beta}^{XC}({\bf r}) \f$ to one effective potential: \f[
358 * V^{eff}_{\alpha\beta}({\bf r}) = v^{H}({\bf r})\delta_{\alpha\beta} + V_{\alpha\beta}^{ext}({\bf r}) +
359 * V_{\alpha\beta}^{XC}({\bf r}) =
360 * \Big({\bf I}\big(v^{H}({\bf r})+v^{ext}({\bf r})+v^{XC}({\bf r})\big) +
361 * {\boldsymbol \sigma}\big( \mu_{B}{\bf B}^{ext}({\bf r}) + {\bf B}^{XC}({\bf r})\big)\Big)_{\alpha\beta}
362 * \f]
363 * and finally, we arrive to the following Kohn-Sham equation for each component \f$ \gamma \f$ of spinor
364 * wave-functions: \f[
365 * -\frac{1}{2}\nabla^{2}\psi_{i}^{\gamma}({\bf r}) + \sum_{\alpha} V_{\gamma\alpha}^{eff}({\bf r})
366 * \psi_{i}^{\alpha}({\bf r}) = \varepsilon_i \psi_{i}^{\gamma}({\bf r}) \f] or in matrix form \f[
367 * \left( \begin{array}{cc} -\frac{1}{2}\nabla^2+V^{eff}_{\uparrow \uparrow} & V^{eff}_{\uparrow \downarrow} \\
368 * V^{eff}_{\downarrow \uparrow} & -\frac{1}{2}\nabla^2+V^{eff}_{\downarrow \downarrow} \end{array}\right)
369 * \left(\begin{array}{c} \psi_{i}^{\uparrow}({\bf r}) \\ \psi_{i}^{\downarrow} ({\bf r}) \end{array} \right) =
370 * \varepsilon_i \left(\begin{array}{c} \psi_{i}^{\uparrow}({\bf r}) \\ \psi_{i}^{\downarrow}({\bf r}) \end{array}
371 * \right) \f]
372 *
373 * \section s4dft Second-variational approach
374 *
375 * Suppose that we know first \f$ N_{fv} \f$ solutions of the following equation (so-called first variational
376 * equation): \f[ \Big(-\frac{1}{2}\nabla^2+v^{H}({\bf r})+v^{ext}({\bf r})+v^{XC}({\bf r}) \Big)\phi_{i}({\bf r}) =
377 * \epsilon_i \phi_{i}({\bf r}) \f] We can write expansion of the components of spinor wave-functions \f$
378 * \psi^{\alpha}_j({\bf r}) \f$ in terms of first-variational states \f$ \phi_i({\bf r}) \f$: \f[ \psi_{j}^{\alpha}({\bf
379 * r}) = \sum_{i}^{N_{fv}}C_{ij}^{\alpha}\phi_{i}({\bf r}) \f] Next, we switch to matrix equation: \f[
380 * \langle \Psi_{j'}| \hat H | \Psi_{j} \rangle = \varepsilon_j \delta_{j'j} \\
381 * \sum_{\alpha \alpha'} \sum_{ii'} C_{i'j'}^{\alpha'*} C_{ij}^{\alpha} \langle \phi_{i'} | \hat H_{\alpha' \alpha} |
382 * \phi_{i} \rangle = \sum_{\alpha \alpha'} \sum_{ii'} C_{i'j'}^{\alpha'*} C_{ij}^{\alpha} H_{\alpha'i', \alpha i} =
383 * \varepsilon_j \delta_{j'j} \f]
384 *
385 * We can combine indices \f$ \{i,\alpha\} \f$ into one 'flat' index \f$ \nu \f$. If we also assume that the number of
386 * spinor wave-functions is equal to \f$ 2 N_{fv} \f$ then we arrive to the well-known eigen decomposition:
387 * \f[
388 * \sum_{\nu'\nu} C_{\nu' j'}^{*} H_{\nu'\nu} C_{\nu j} = \varepsilon_j \delta_{j'j}
389 * \f]
390 *
391 * The expression for second-variational Hamiltonian is simple:
392 * \f[
393 * \langle \phi_{i'}|\hat H_{\alpha'\alpha} |\phi_{i} \rangle =
394 * \langle \phi_{i'} | \Big(-\frac{1}{2}\nabla^2 + v^{H}({\bf r}) + v^{ext}({\bf r}) + v^{XC}({\bf r}) \Big)
395 * \delta_{\alpha\alpha'}|\phi_{i}\rangle +
396 * \langle \phi_{i'} | {\boldsymbol \sigma}_{\alpha\alpha'} \Big( \mu_{B}{\bf B}^{ext}({\bf r})+
397 * {\bf B}^{XC}({\bf r})\Big) | \phi_{i}\rangle =\\
398 * \epsilon_{i}\delta_{i'i}\delta_{\alpha\alpha'} + {\boldsymbol \sigma}_{\alpha\alpha'} \langle \phi_{i'} |
399 * \Big( \mu_{B}{\bf B}^{ext}({\bf r}) +
400 * {\bf B}^{XC}({\bf r})\Big) | \phi_{i}\rangle
401 * \f]
402 */
