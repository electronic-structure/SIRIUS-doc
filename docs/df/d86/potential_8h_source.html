201  /// Generate local part of pseudo potential.
202  /** Total local potential is a lattice sum:
203  * \f[
204  * V({\bf r}) = \sum_{{\bf T},\alpha} V_{\alpha}({\bf r} - {\bf T} - {\bf \tau}_{\alpha})
205  * \f]
206  * We want to compute it's plane-wave expansion coefficients:
207  * \f[
208  * V({\bf G}) = \frac{1}{V} \int e^{-i{\bf Gr}} V({\bf r}) d{\bf r} =
209  * \frac{1}{V} \sum_{{\bf T},\alpha} \int e^{-i{\bf Gr}}V_{\alpha}({\bf r} - {\bf T} - {\bf \tau}_{\alpha})d{\bf r}
210  * \f]
211  * Standard change of variables: \f$ {\bf r}' = {\bf r} - {\bf T} - {\bf \tau}_{\alpha},\; {\bf r} = {\bf r}' + {\bf T} + {\bf \tau}_{\alpha} \f$
212  * leads to:
213  * \f[
214  * V({\bf G}) = \frac{1}{V} \sum_{{\bf T},\alpha} \int e^{-i{\bf G}({\bf r}' + {\bf T} + {\bf \tau}_{\alpha})}V_{\alpha}({\bf r}')d{\bf r'} =
215  * \frac{N}{V} \sum_{\alpha} \int e^{-i{\bf G}({\bf r}' + {\bf \tau}_{\alpha})}V_{\alpha}({\bf r}')d{\bf r'} =
216  * \frac{1}{\Omega} \sum_{\alpha} e^{-i {\bf G} {\bf \tau}_{\alpha} } \int e^{-i{\bf G}{\bf r}}V_{\alpha}({\bf r})d{\bf r}
217  * \f]
218  * Using the well-known expansion of a plane wave in terms of spherical Bessel functions:
219  * \f[
220  * e^{i{\bf G}{\bf r}}=4\pi \sum_{\ell m} i^\ell j_{\ell}(Gr)Y_{\ell m}^{*}({\bf \hat G})Y_{\ell m}({\bf \hat r})
221  * \f]
222  * and remembering that for \f$ \ell = 0 \f$ (potential is sphericla) \f$ j_{0}(x) = \sin(x) / x \f$ we have:
223  * \f[
224  * V_{\alpha}({\bf G}) = \int V_{\alpha}(r) 4\pi \frac{\sin(Gr)}{Gr} Y^{*}_{00} Y_{00} r^2 \sin(\theta) dr d \phi d\theta =
225  * 4\pi \int V_{\alpha}(r) \frac{\sin(Gr)}{Gr} r^2 dr
226  * \f]
227  * The tricky part comes next: \f$ V_{\alpha}({\bf r}) \f$ is a long-range potential -- it decays slowly as
228  * \f$ -Z_{\alpha}^{p}/r \f$ and the straightforward integration with sperical Bessel function is numerically
229  * unstable. For \f$ {\bf G} = 0 \f$ an extra term \f$ Z_{\alpha}^p/r \f$, corresponding to the potential of
230  * pseudo-ion, is added to and removed from the local part of the atomic pseudopotential \f$ V_{\alpha}({\bf r}) \f$:
231  * \f[
232  * V_{\alpha}({\bf G} = 0) = \int V_{\alpha}({\bf r})d{\bf r} \Rightarrow
233  * 4\pi \int \Big( V_{\alpha}(r) + \frac{Z_{\alpha}^p}{r} \Big) r^2 dr -
234  * 4\pi \int \Big( \frac{Z_{\alpha}^p}{r} \Big) r^2 dr
235  * \f]
236  * Second term corresponds to the average electrostatic potential of ions and it is ignored
237  * (like the \f$ {\bf G} = 0 \f$ term in the Hartree potential of electrons).
238  * For \f$ G \ne 0 \f$ the following trick is done: \f$ Z_{\alpha}^p {\rm erf}(r) / r \f$ is added to and
239  * removed from \f$ V_{\alpha}(r) \f$. The idea is to make potential decay quickly and then take the extra
240  * contribution analytically. We have:
241  * \f[
242  * V_{\alpha}({\bf G}) = 4\pi \int \Big(V_{\alpha}(r) + Z_{\alpha}^p \frac{{\rm erf}(r)} {r} -
243  * Z_{\alpha}^p \frac{{\rm erf}(r)}{r}\Big) \frac{\sin(Gr)}{Gr} r^2 dr
244  * \f]
245  * Analytical contribution from the error function is computed using the 1D Fourier transform in complex plane:
246  * \f[
247  * \frac{1}{\sqrt{2 \pi}} \int_{-\infty}^{\infty} {\rm erf}(t) e^{i\omega t} dt =
248  * \frac{i e^{-\frac{\omega ^2}{4}} \sqrt{\frac{2}{\pi }}}{\omega }
249  * \f]
250  * from which we immediately get
251  * \f[
252  * \int_{0}^{\infty} \frac{{\rm erf}(r)}{r} \frac{\sin(Gr)}{Gr} r^2 dr = \frac{e^{-\frac{G^2}{4}}}{G^2}
253  * \f]
254  * The final expression for the local potential radial integrals for \f$ G \ne 0 \f$ take the following form:
255  * \f[
256  * 4\pi \int \Big(V_{\alpha}(r) r + Z_{\alpha}^p {\rm erf}(r) \Big) \frac{\sin(Gr)}{G} dr - Z_{\alpha}^p \frac{e^{-\frac{G^2}{4}}}{G^2}
257  * \f]
258  */

475  /// Solve Poisson equation for a single atom.
476  template <bool free_atom, typename T>
477  inline std::vector<T>
478  poisson_vmt(Atom const& atom__,
479  Spheric_function<function_domain_t::spectral, T> const& rho_mt__,
480  Spheric_function<function_domain_t::spectral, T>& vha_mt__) const
481  {
482  const bool use_r_prefact{false};
483 
484  int lmmax_rho = rho_mt__.angular_domain_size();
485  int lmmax_pot = vha_mt__.angular_domain_size();
486  assert((int)l_by_lm_.size() >= lmmax_rho);
487  if (lmmax_rho > ctx_.lmmax_rho()) {
488  std::stringstream s;
489  s << "wrong angular size of rho_mt for atom of " << atom__.type().symbol() << std::endl
490  << " lmmax_rho: " << lmmax_rho << std::endl
491  << " ctx.lmmax_rho(): " << ctx_.lmmax_rho();
492  TERMINATE(s);
493  }
494  std::vector<T> qmt(ctx_.lmmax_rho(), 0);
495 
496  double R = atom__.mt_radius();
497  int nmtp = atom__.num_mt_points();
498 
499  #pragma omp parallel
500  {
501  std::vector<T> g1;
502  std::vector<T> g2;
503 
504  #pragma omp for
505  for (int lm = 0; lm < lmmax_rho; lm++) {
506  int l = l_by_lm_[lm];
507 
508  auto rholm = rho_mt__.component(lm);
509 
510  if (use_r_prefact) {
511  /* save multipole moment */
512  qmt[lm] = rholm.integrate(g1, l + 2);
513  } else {
514  for (int ir = 0; ir < nmtp; ir++) {
515  double r = atom__.radial_grid(ir);
516  rholm(ir) *= std::pow(r, l + 2);
517  }
518  qmt[lm] = rholm.interpolate().integrate(g1, 0);
519  }
520 
521  if (lm < lmmax_pot) {
522 
523  if (use_r_prefact) {
524  rholm.integrate(g2, 1 - l);
525  } else {
526  rholm = rho_mt__.component(lm);
527  for (int ir = 0; ir < nmtp; ir++) {
528  double r = atom__.radial_grid(ir);
529  rholm(ir) *= std::pow(r, 1 - l);
530  }
531  rholm.interpolate().integrate(g2, 0);
532  }
533 
534  double fact = fourpi / double(2 * l + 1);
535  T vlm;
536 
537  for (int ir = 0; ir < nmtp; ir++) {
538  double r = atom__.radial_grid(ir);
539 
540  if (free_atom) {
541  vlm = g1[ir] / std::pow(r, l + 1) + (g2.back() - g2[ir]) * std::pow(r, l);
542  } else {
543  double d1 = 1.0 / std::pow(R, 2 * l + 1);
544  vlm = (1.0 - std::pow(r / R, 2 * l + 1)) * g1[ir] / std::pow(r, l + 1) +
545  (g2.back() - g2[ir]) * std::pow(r, l) - (g1.back() - g1[ir]) * std::pow(r, l) * d1;
546  }
547  vha_mt__(lm, ir) = vlm * fact;
548  }
549  }
550  }
551  }
552  if (!free_atom) {
553  /* constant part of nuclear potential -z*(1/r - 1/R) */
554  for (int ir = 0; ir < nmtp; ir++) {
555 #ifdef __VHA_AUX
556  double r = atom__.radial_grid(ir);
557  vha_mt__(0, ir) += atom__.zn() * (1 / R - 1 / r) / y00;
558 #else
559  vha_mt__(0, ir) += atom__.zn() / R / y00;
560 #endif
561  }
562  }
563  /* nuclear multipole moment */
564  qmt[0] -= atom__.zn() * y00;
565 
566  return std::move(qmt);

569  /// Poisson solver.
570  /** Detailed explanation is available in:
571  * - Weinert, M. (1981). Solution of Poisson's equation: beyond Ewald-type methods.
572  * Journal of Mathematical Physics, 22(11), 2433–2439. doi:10.1063/1.524800
573  * - Classical Electrodynamics Third Edition by J. D. Jackson.
574  *
575  * Solution of Poisson's equation for the muffin-tin geometry is carried out in several steps:
576  * - True multipole moments \f$ q_{\ell m}^{\alpha} \f$ of the muffin-tin charge density are computed.
577  * - Pseudocharge density is introduced. Pseudocharge density coincides with the true charge density
578  * in the interstitial region and it's multipole moments inside muffin-tin spheres coincide with the
579  * true multipole moments.
580  * - Poisson's equation for the pseudocharge density is solved in the plane-wave domain. It gives the
581  * correct interstitial potential and correct muffin-tin boundary values.
582  * - Finally, muffin-tin part of potential is found by solving Poisson's equation in spherical coordinates
583  * with Dirichlet boundary conditions.
584  *
585  * We start by computing true multipole moments of the charge density inside the muffin-tin spheres:
586  * \f[
587  * q_{\ell m}^{\alpha} = \int Y_{\ell m}^{*}(\hat {\bf r}) r^{\ell} \rho({\bf r}) d {\bf r} =
588  * \int \rho_{\ell m}^{\alpha}(r) r^{\ell + 2} dr
589  * \f]
590  * and for the nucleus with charge density \f$ \rho(r, \theta, \phi) = -\frac{Z \delta(r)}{4 \pi r^2} \f$:
591  * \f[
592  * q_{00}^{\alpha} = \int Y_{0 0} \frac{-Z_{\alpha} \delta(r)}{4 \pi r^2} r^2 \sin \theta dr d\phi d\theta =
593  * -Z_{\alpha} Y_{00}
594  * \f]
595  *
596  * Now we need to get the multipole moments of the interstitial charge density \f$ \rho^{I}({\bf r}) \f$ inside
597  * muffin-tin spheres. We need this in order to estimate the amount of pseudocharge to be added to
598  * \f$ \rho^{I}({\bf r}) \f$ to get the pseudocharge multipole moments equal to the true multipole moments.
599  * We want to compute
600  * \f[
601  * q_{\ell m}^{I,\alpha} = \int Y_{\ell m}^{*}(\hat {\bf r}) r^{\ell} \rho^{I}({\bf r}) d {\bf r}
602  * \f]
603  * where
604  * \f[
605  * \rho^{I}({\bf r}) = \sum_{\bf G}e^{i{\bf Gr}} \rho({\bf G})
606  * \f]
607  *
608  * Recall the spherical plane wave expansion:
609  * \f[
610  * e^{i{\bf G r}}=4\pi e^{i{\bf G r}_{\alpha}} \sum_{\ell m} i^\ell
611  * j_{\ell}(G|{\bf r}-{\bf r}_{\alpha}|)
612  * Y_{\ell m}^{*}({\bf \hat G}) Y_{\ell m}(\widehat{{\bf r}-{\bf r}_{\alpha}})
613  * \f]
614  * Multipole moments of each plane-wave are computed as:
615  * \f[
616  * q_{\ell m}^{\alpha}({\bf G}) = 4 \pi e^{i{\bf G r}_{\alpha}} Y_{\ell m}^{*}({\bf \hat G}) i^{\ell}
617  * \int_{0}^{R} j_{\ell}(Gr) r^{\ell + 2} dr = 4 \pi e^{i{\bf G r}_{\alpha}} Y_{\ell m}^{*}({\bf \hat G}) i^{\ell}
618  * \left\{\begin{array}{ll} \frac{R^{\ell + 2} j_{\ell + 1}(GR)}{G} & G \ne 0 \\
619  * \frac{R^3}{3} \delta_{\ell 0} & G = 0 \end{array} \right.
620  * \f]
621  *
622  * Final expression for the muffin-tin multipole moments of the interstitial charge denisty:
623  * \f[
624  * q_{\ell m}^{I,\alpha} = \sum_{\bf G}\rho({\bf G}) q_{\ell m}^{\alpha}({\bf G})
625  * \f]
626  *
627  * Now we are going to modify interstitial charge density inside the muffin-tin region in order to
628  * get the true multipole moments. We will add a pseudodensity of the form:
629  * \f[
630  * P({\bf r}) = \sum_{\ell m} p_{\ell m}^{\alpha} Y_{\ell m}(\hat {\bf r}) r^{\ell} \left(1-\frac{r^2}{R^2}\right)^n
631  * \f]
632  * Radial functions of the pseudodensity are chosen in a special way. First, they produce a confined and
633  * smooth functions inside muffin-tins and second (most important) plane-wave coefficients of the
634  * pseudodensity can be computed analytically. Let's find the relation between \f$ p_{\ell m}^{\alpha} \f$
635  * coefficients and true and interstitial multipole moments first. We are searching for the pseudodensity which restores
636  * the true multipole moments:
637  * \f[
638  * \int Y_{\ell m}^{*}(\hat {\bf r}) r^{\ell} \Big(\rho^{I}({\bf r}) + P({\bf r})\Big) d {\bf r} = q_{\ell m}^{\alpha}
639  * \f]
640  * Then
641  * \f[
642  * p_{\ell m}^{\alpha} = \frac{q_{\ell m}^{\alpha} - q_{\ell m}^{I,\alpha}}
643  * {\int r^{2 \ell + 2} \left(1-\frac{r^2}{R^2}\right)^n dr} =
644  * (q_{\ell m}^{\alpha} - q_{\ell m}^{I,\alpha}) \frac{2 \Gamma(5/2 + \ell + n)}{R^{2\ell + 3}\Gamma(3/2 + \ell) \Gamma(n + 1)}
645  * \f]
646  *
647  * Now let's find the plane-wave coefficients of \f$ P({\bf r}) \f$ inside each muffin-tin:
648  * \f[
649  * P^{\alpha}({\bf G}) = \frac{4\pi e^{-i{\bf G r}_{\alpha}}}{\Omega} \sum_{\ell m} (-i)^{\ell} Y_{\ell m}({\bf \hat G})
650  * p_{\ell m}^{\alpha} \int_{0}^{R} j_{\ell}(G r) r^{\ell} \left(1-\frac{r^2}{R^2}\right)^n r^2 dr
651  * \f]
652  *
653  * Integral of the spherical Bessel function with the radial pseudodensity component is taken analytically:
654  * \f[
655  * \int_{0}^{R} j_{\ell}(G r) r^{\ell} \left(1-\frac{r^2}{R^2}\right)^n r^2 dr =
656  * 2^n R^{\ell + 3} (GR)^{-n - 1} \Gamma(n + 1) j_{n + \ell + 1}(GR)
657  * \f]
658  *
659  * The final expression for the pseudodensity plane-wave component is:
660  * \f[
661  * P^{\alpha}({\bf G}) = \frac{4\pi e^{-i{\bf G r}_{\alpha}}}{\Omega} \sum_{\ell m} (-i)^{\ell} Y_{\ell m}({\bf \hat G})
662  * (q_{\ell m}^{\alpha} - q_{\ell m}^{I,\alpha}) \Big( \frac{2}{GR} \Big)^{n+1}
663  * \frac{ \Gamma(5/2 + n + \ell) } {R^{\ell} \Gamma(3/2+\ell)} j_{n + \ell + 1}(GR)
664  * \f]
665  *
666  * For \f$ G=0 \f$ only \f$ \ell = 0 \f$ contribution survives:
667  * \f[
668  * P^{\alpha}({\bf G}=0) = \frac{4\pi}{\Omega} Y_{00} (q_{00}^{\alpha} - q_{00}^{I,\alpha})
669  * \f]
670  *
671  * We can now sum the contributions from all muffin-tin spheres and obtain a modified charge density,
672  * which is equal to the exact charge density in the interstitial region and which has correct multipole
673  * moments inside muffin-tin spheres:
674  * \f[
675  * \tilde \rho({\bf G}) = \rho({\bf G}) + \sum_{\alpha} P^{\alpha}({\bf G})
676  * \f]
677  * This density is used to solve the Poisson's equation in the plane-wave domain:
678  * \f[
679  * V_{H}({\bf G}) = \frac{4 \pi \tilde \rho({\bf G})}{G^2}
680  * \f]
681  * The potential is correct in the interstitial region and also on the muffin-tin surface. We will use
682  * it to find the boundary conditions for the potential inside the muffin-tins. Using spherical
683  * plane-wave expansion we get:
684  * \f[
685  * V^{\alpha}_{\ell m}(R) = \sum_{\bf G} V_{H}({\bf G})
686  * 4\pi e^{i{\bf G r}_{\alpha}} i^\ell
687  * j_{\ell}^{\alpha}(GR) Y_{\ell m}^{*}({\bf \hat G})
688  * \f]
689  *
690  * As soon as the muffin-tin boundary conditions for the potential are known, we can find the potential
691  * inside spheres using Dirichlet Green's function:
692  * \f[
693  * V({\bf x}) = \int \rho({\bf x'})G_D({\bf x},{\bf x'}) d{\bf x'} - \frac{1}{4 \pi} \int_{S} V({\bf x'})
694  * \frac{\partial G_D}{\partial n'} d{\bf S'}
695  * \f]
696  * where Dirichlet Green's function for the sphere is defined as:
697  * \f[
698  * G_D({\bf x},{\bf x'}) = 4\pi \sum_{\ell m} \frac{Y_{\ell m}^{*}({\bf \hat x'})
699  * Y_{\ell m}(\hat {\bf x})}{2\ell + 1}
700  * \frac{r_{<}^{\ell}}{r_{>}^{\ell+1}}\Biggl(1 - \Big( \frac{r_{>}}{R} \Big)^{2\ell + 1} \Biggr)
701  * \f]
702  * and it's normal derivative at the surface is equal to:
703  * \f[
704  * \frac{\partial G_D}{\partial n'} = -\frac{4 \pi}{R^2} \sum_{\ell m} \Big( \frac{r}{R} \Big)^{\ell}
705  * Y_{\ell m}^{*}({\bf \hat x'}) Y_{\ell m}(\hat {\bf x})
706  * \f]
707  */
708  inline void poisson(Periodic_function<double> const& rho);

710  /// Generate XC potential and energy density
711  /** In case of spin-unpolarized GGA the XC potential has the following expression:
712  * \f[
713  * V_{XC}({\bf r}) = \frac{\partial}{\partial \rho} \varepsilon_{xc}(\rho, \nabla \rho) -
714  * \nabla \frac{\partial}{\partial (\nabla \rho)} \varepsilon_{xc}(\rho, \nabla \rho)
715  * \f]
716  * LibXC packs the gradient information into the so-called \a sigma array:
717  * \f[
718  * \sigma = \nabla \rho \nabla \rho
719  * \f]
720  * Changing variables in \f$ V_{XC} \f$ expression gives:
721  * \f{eqnarray*}{
722  * V_{XC}({\bf r}) &=& \frac{\partial}{\partial \rho} \varepsilon_{xc}(\rho, \sigma) -
723  * \nabla \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma}
724  * \frac{\partial \sigma}{ \partial (\nabla \rho)} \\
725  * &=& \frac{\partial}{\partial \rho} \varepsilon_{xc}(\rho, \sigma) -
726  * 2 \nabla \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \nabla \rho -
727  * 2 \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \nabla^2 \rho
728  * \f}
729  * The following sequence of functions must be computed:
730  * - density on the real space grid
731  * - gradient of density (in spectral representation)
732  * - gradient of density on the real space grid
733  * - laplacian of density (in spectral representation)
734  * - laplacian of density on the real space grid
735  * - \a sigma array
736  * - a call to Libxc must be performed \a sigma derivatives must be obtained
737  * - \f$ \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \f$ in spectral representation
738  * - gradient of \f$ \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \f$ in spectral representation
739  * - gradient of \f$ \frac{\partial \varepsilon_{xc}(\rho, \sigma)}{\partial \sigma} \f$ on the real space grid
740  *
741  * Expression for spin-polarized potential has a bit more complicated form:
742  * \f{eqnarray*}
743  * V_{XC}^{\gamma} &=& \frac{\partial \varepsilon_{xc}}{\partial \rho_{\gamma}} - \nabla
744  * \Big( 2 \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \gamma}} \nabla \rho_{\gamma} +
745  * \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \delta}} \nabla \rho_{\delta} \Big) \\
746  * &=& \frac{\partial \varepsilon_{xc}}{\partial \rho_{\gamma}}
747  * -2 \nabla \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \gamma}} \nabla \rho_{\gamma}
748  * -2 \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \gamma}} \nabla^2 \rho_{\gamma}
749  * - \nabla \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \delta}} \nabla \rho_{\delta}
750  * - \frac{\partial \varepsilon_{xc}}{\partial \sigma_{\gamma \delta}} \nabla^2 \rho_{\delta}
751  * \f}
752  * In magnetic case the "up" and "dn" density and potential decomposition is used. Using the fact that the
753  * effective magnetic field is parallel to magnetization at each point in space, we can write the coupling
754  * of density and magnetization with XC potential and XC magentic field as:
755  * \f[
756  * V_{xc}({\bf r}) \rho({\bf r}) + {\bf B}_{xc}({\bf r}){\bf m}({\bf r}) =
757  * V_{xc}({\bf r}) \rho({\bf r}) + {\rm B}_{xc}({\bf r}) {\rm m}({\bf r}) =
758  * V^{\uparrow}({\bf r})\rho^{\uparrow}({\bf r}) + V^{\downarrow}({\bf r})\rho^{\downarrow}({\bf r})
759  * \f]
760  * where
761  * \f{eqnarray*}{
762  * \rho^{\uparrow}({\bf r}) &=& \frac{1}{2}\Big( \rho({\bf r}) + {\rm m}({\bf r}) \Big) \\
763  * \rho^{\downarrow}({\bf r}) &=& \frac{1}{2}\Big( \rho({\bf r}) - {\rm m}({\bf r}) \Big)
764  * \f}
765  * and
766  * \f{eqnarray*}{
767  * V^{\uparrow}({\bf r}) &=& V_{xc}({\bf r}) + {\rm B}_{xc}({\bf r}) \\
768  * V^{\downarrow}({\bf r}) &=& V_{xc}({\bf r}) - {\rm B}_{xc}({\bf r})
769  * \f}
770  */

928  /// Calculate D operator from potential and augmentation charge.
929  /** The following real symmetric matrix is computed:
930  * \f[
931  * D_{\xi \xi'}^{\alpha} = \int V({\bf r}) Q_{\xi \xi'}^{\alpha}({\bf r}) d{\bf r}
932  * \f]
933  * In the plane-wave domain this integrals transform into sum over Fourier components:
934  * \f[
935  * D_{\xi \xi'}^{\alpha} = \sum_{\bf G} \langle V |{\bf G}\rangle \langle{\bf G}|Q_{\xi \xi'}^{\alpha} \rangle =
936  * \sum_{\bf G} V^{*}({\bf G}) e^{-i{\bf r}_{\alpha}{\bf G}} Q_{\xi \xi'}^{A}({\bf G}) =
937  * \sum_{\bf G} Q_{\xi \xi'}^{A}({\bf G}) \tilde V_{\alpha}^{*}({\bf G})
938  * \f]
939  * where \f$ \alpha \f$ is the atom, \f$ A \f$ is the atom type and
940  * \f[
941  * \tilde V_{\alpha}({\bf G}) = e^{i{\bf r}_{\alpha}{\bf G}} V({\bf G})
942  * \f]
943  * Both \f$ V({\bf r}) \f$ and \f$ Q({\bf r}) \f$ functions are real and the following condition is fulfilled:
944  * \f[
945  * \tilde V_{\alpha}({\bf G}) = \tilde V_{\alpha}^{*}(-{\bf G})
946  * \f]
947  * \f[
948  * Q_{\xi \xi'}({\bf G}) = Q_{\xi \xi'}^{*}(-{\bf G})
949  * \f]
950  * In the sum over plane-wave coefficients the \f$ {\bf G} \f$ and \f$ -{\bf G} \f$ contributions will give:
951  * \f[
952  * Q_{\xi \xi'}^{A}({\bf G}) \tilde V_{\alpha}^{*}({\bf G}) + Q_{\xi \xi'}^{A}(-{\bf G}) \tilde V_{\alpha}^{*}(-{\bf G}) =
953  * 2 \Re \Big( Q_{\xi \xi'}^{A}({\bf G}) \Big) \Re \Big( \tilde V_{\alpha}({\bf G}) \Big) +
954  * 2 \Im \Big( Q_{\xi \xi'}^{A}({\bf G}) \Big) \Im \Big( \tilde V_{\alpha}({\bf G}) \Big)
955  * \f]
956  * This allows the use of a <b>dgemm</b> instead of a <b>zgemm</b> when \f$ D_{\xi \xi'}^{\alpha} \f$ matrix
957  * is calculated for all atoms of the same type.
958  */
