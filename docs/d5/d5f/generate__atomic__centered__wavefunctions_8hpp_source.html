5 inline void K_point::generate_atomic_centered_wavefunctions(const int num_ao__, Wave_functions& phi)
6 {
7  if (!num_ao__) {
8  return;
9  }
10 
11  int lmax{0};
12  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
13  auto& atom_type = unit_cell_.atom_type(iat);
14  lmax = std::max(lmax, atom_type.lmax_ps_atomic_wf());
15  }
16  lmax = std::max(lmax, unit_cell_.lmax());
17 
18  #pragma omp parallel for schedule(static)
19  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
20  /* global index of G+k vector */
21  int igk = this->idxgk(igk_loc);
22  /* vs = {r, theta, phi} */
23  auto vs = SHT::spherical_coordinates(this->gkvec().gkvec_cart(igk));
24  /* compute real spherical harmonics for G+k vector */
25  std::vector<double> rlm(Utils::lmmax(lmax));
26  SHT::spherical_harmonics(lmax, vs[1], vs[2], &rlm[0]);
27  /* get values of radial integrals for a given G+k vector length */
28  std::vector<mdarray<double, 1>> ri_values(unit_cell_.num_atom_types());
29  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
30  ri_values[iat] = ctx_.atomic_wf_ri().values(iat, vs[0]);
31  }
32 
33  int n{0};
34  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
35  auto phase = twopi * dot(gkvec().gkvec(igk), unit_cell_.atom(ia).position());
36  auto phase_factor = std::exp(double_complex(0.0, phase));
37  auto& atom_type = unit_cell_.atom(ia).type();
38 
39  for (int i = 0; i < atom_type.num_ps_atomic_wf(); i++) {
40  auto l = std::abs(atom_type.ps_atomic_wf(i).first);
41  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
42  for (int m = -l; m <= l; m++) {
43  int lm = Utils::lm_by_l_m(l, m);
44  phi.pw_coeffs(0).prime(igk_loc, n) = z * std::conj(phase_factor) * rlm[lm] * ri_values[atom_type.id()][i];
45  n++;
46  }
47  } // i
48  } // ia
49  } // igk_loc
50 }
