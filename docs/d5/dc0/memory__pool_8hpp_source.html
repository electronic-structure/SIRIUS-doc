42  /// Allocate n elements of type T in a specified memory.
43  template <typename T, memory_t mem_type>
44  T* allocate(size_t n__)
45  {
46  size_t sz = n__ * sizeof(T);
47 
48  if (!pool_pos_.count(mem_type)) {
49  pool_pos_[mem_type] = 0;
50  }
51  //if (!pool_.count(mem_type)) {

53  //}
54  //if (!tmp_pool_.count(mem_type)) {
55  // tmp_pool_[mem_type];
56  //}
57 
58  size_t pos = pool_pos_[mem_type];
59  if (pos + sz <= pool_[mem_type].size()) {
60  T* ptr = reinterpret_cast<T*>(pool_[mem_type].template at<device<mem_type>::type>(pos));
61  pool_pos_[mem_type] += sz;
62  return ptr;
63  } else {
64  tmp_pool_[mem_type].emplace_back(mdarray<int8_t, 1>(sz, mem_type));
65  return reinterpret_cast<T*>(tmp_pool_[mem_type].back().template at<device<mem_type>::type>());
66  }
67  }
68 
69  template <memory_t mem_type>
70  void reset()
71  {
72  size_t sz{0};
73  if (tmp_pool_.count(mem_type)) {
74  for (auto& e: tmp_pool_[mem_type]) {
75  sz += e.size();
76  }
77  }
78  if (sz != 0) {
79  sz += pool_[mem_type].size();
80  tmp_pool_[mem_type].clear();
81  pool_[mem_type] = mdarray<int8_t, 1>(sz, mem_type);
82  }
83  pool_pos_[mem_type] = 0;
84  }
85 };
