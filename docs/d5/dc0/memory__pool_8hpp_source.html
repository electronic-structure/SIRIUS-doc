33 /// Descriptor of the allocated memory block.
34 struct memory_block_descriptor
35 {
36  /// Raw pointer to the block of memory.
37  uint8_t* ptr_{nullptr};
38  /// Size of the memory block.
39  size_t size_{0};
40  /// True if this block is used.
41  bool used_{false};
42  /// Shared pointer to an allocated memory block.
43  std::shared_ptr<mdarray<uint8_t, 1>> buf_;
44  /// Keep track of the used sub-blocks of the shared memory buffer.
45  std::shared_ptr<int> used_count_;
46 };

49 {
50  private:
51  /// List of blocks of allocated memory.
52  std::map<memory_t, std::list<memory_block_descriptor>> memory_blocks_;
53  size_t size_allocated_{0};
54  size_t max_allocated_{0};

56  template <memory_t mem_type>
57  std::list<memory_block_descriptor>::iterator remove_block(std::list<memory_block_descriptor>::iterator it__)
58  {
59  (*it__->used_count_)--;
60  size_allocated_ -= it__->size_;
61  if (it__->buf_.use_count() > 1) {
62  /* this is not the last sub-block: just remove it */
63  return memory_blocks_[mem_type].erase(it__);
64  } else {
65  /* this is the last sub-block in the series */
66  it__->used_ = false;
67  it__->ptr_ = it__->buf_->at<device<mem_type>::type>();
68  it__->size_ = it__->buf_->size();
69 
70  auto merge_blocks = [&](std::list<memory_block_descriptor>::iterator& it0)
71  {
72  if (!it0->used_ && it0->buf_.use_count() == 1) {
73  it__->size_ = it__->buf_->size() + it0->buf_->size();
74  it__->buf_ = std::shared_ptr<mdarray<uint8_t, 1>>(new mdarray<uint8_t, 1>(it__->size_, mem_type));
75  it__->ptr_ = it__->buf_->at<device<mem_type>::type>();
76  /* erase the merged block */
77  memory_blocks_[mem_type].erase(it0);
78  }
79  };
80 
81  if (it__ != memory_blocks_[mem_type].begin()) {
82  /* check previous block */
83  auto it0 = it__;
84  it0--;
85  /* if previous block is not used and its memory is not shared with other blocks */
86  merge_blocks(it0);
87  }
88  /* check next block */
89  auto it1 = it__;
90  it1++;
91  if (it1 != memory_blocks_[mem_type].end()) {
92  /* if next block is not used and its memory is not shared with other blocks */
93  merge_blocks(it1);
94  }
95  it__++;
96  return it__;
97  }
98  }

106  template <typename T, memory_t mem_type>
107  T* allocate(size_t num_elements__)
108  {
109  if (memory_blocks_.count(mem_type) == 0) {
110  memory_blocks_[mem_type] = std::list<memory_block_descriptor>();
111  }
112  /* size of the memory block in bytes */
113  size_t size = num_elements__ * sizeof(T);
114  size_allocated_ += size;
115  max_allocated_ = std::max(max_allocated_, size_allocated_);
116  /* iterate over existing blocks */
117  for (auto it = memory_blocks_[mem_type].begin(); it != memory_blocks_[mem_type].end(); ++it) {
118  /* if unused block with suitable size is found */
119  if (!it->used_ && it->size_ >= size) {
120  /* divide block into two parts (head and tail) */
121  memory_block_descriptor tail_block;
122  tail_block.ptr_ = it->ptr_ + size;
123  tail_block.size_ = it->size_ - size;
124  tail_block.used_ = false;
125  tail_block.buf_ = it->buf_;
126  tail_block.used_count_ = it->used_count_;
127  /* shrink the size of the current block */
128  it->size_ = size;
129  it->used_ = true;
130  (*it->used_count_)++;
131  auto ptr = it->ptr_;
132  /* insert the tail block */
133  if (tail_block.size_ > 0) {
134  it++;
135  memory_blocks_[mem_type].insert(it, std::move(tail_block));
136  }
137  return reinterpret_cast<T*>(ptr);
138  }
139  }
140  /* if the free block is not found, create a new one */
141  memory_block_descriptor new_block;
142  new_block.buf_ = std::shared_ptr<mdarray<uint8_t, 1>>(new mdarray<uint8_t, 1>(size, mem_type));
143  new_block.ptr_ = new_block.buf_->at<device<mem_type>::type>();
144  new_block.size_ = size;
145  new_block.used_ = true;
146  new_block.used_count_ = std::shared_ptr<int>(new int);
147  (*new_block.used_count_) = 1;
148  auto ptr = new_block.ptr_;
149  memory_blocks_[mem_type].push_back(std::move(new_block));
150  return reinterpret_cast<T*>(ptr);
151  }

153  template <memory_t mem_type>
154  void free(void* ptr__)
155  {
156  /* iterate over memory blocks */
157  for (auto it = memory_blocks_[mem_type].begin(); it != memory_blocks_[mem_type].end(); ++it) {
158  if (it->ptr_ == ptr__) {
159  if (!it->used_) {
160  throw std::runtime_error("wrong pointer");
161  }
162  remove_block<mem_type>(it);
163  return;
164  }
165  }
166  throw std::runtime_error("wrong pointer");
167  }

169  template <memory_t mem_type>
170  void reset()
171  {
172  auto mem_type_entry = memory_blocks_.find(mem_type);
173  if (mem_type_entry != memory_blocks_.end()) {
174  /* iterate over memory blocks */
175  auto it = mem_type_entry->second.begin();
176  while (it != mem_type_entry->second.end()) {
177  it = remove_block<mem_type>(it);
178  }
179  it = mem_type_entry->second.begin();
180  if (mem_type_entry->second.size() != 1 || it->used_ || (it->size_ != it->buf_->size())) {
181  std::stringstream s;
182  s << "error in memory_pool::reset()\n"
183  << " list size : " << mem_type_entry->second.size() << " (expecting 1)\n"
184  << " used : " << it->used_ << " (expecting false)\n"
185  << " buffer size : " << it->size_ << " " << it->buf_->size() << " (expecting equal)";
186  TERMINATE(s);
187  }
188  }
189  }
