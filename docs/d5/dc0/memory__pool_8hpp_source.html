46  /// Allocate n elements of type T.
47  template <typename T>
48  T* allocate(size_t n__)
49  {
50  //static_assert(std::is_pod<T>::value, "not a simple data type");
51 
52  size_t sz = n__ * sizeof(T);
53  if (pos_ + sz <= pool_.size()) {
54  T* ptr = reinterpret_cast<T*>(&pool_[pos_]);
55  pos_ += sz;
56  return ptr;
57  } else {
58  tmp_pool_.emplace_back(new sddk::mdarray<int8_t, 1>(sz));
59  return reinterpret_cast<T*>(tmp_pool_.back()->template at<CPU>());
60  }
61  }
62 
63  void reset()
64  {
65  size_t sz{0};
66  for (auto& e: tmp_pool_) {
67  sz += e->size();
68  }
69  if (sz != 0) {
70  sz += pool_.size();
71  tmp_pool_.clear();
72  pool_ = sddk::mdarray<int8_t, 1>(sz);
73  }
74  pos_ = 0;
75  }
76 };
