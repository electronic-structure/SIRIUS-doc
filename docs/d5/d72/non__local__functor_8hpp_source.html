48  /// collect summation result in an array
49  void add_k_point_contribution(K_point& kpoint__, mdarray<double, 2>& collect_res__)
50  {
51  auto& unit_cell = ctx_.unit_cell();
52 
53  auto& bp = kpoint__.beta_projectors();
54 
55  double main_two_factor{-2};
56 
57  for (int icnk = 0; icnk < bp_base_.num_chunks(); icnk++) {
58 
59  bp.prepare();
60  /* generate chunk for inner product of beta */
61  bp.generate(icnk);
62 
63  /* store <beta|psi> for spin up and down */
64  matrix<T> beta_phi_chunks[2];
65 
66  for(int ispn = 0; ispn < ctx_.num_spins(); ispn++){
67  int nbnd = kpoint__.num_occupied_bands(ispn);
68  beta_phi_chunks[ispn] = bp.inner<T>(icnk, kpoint__.spinor_wave_functions(), ispn, 0, nbnd);
69  }
70  bp.dismiss();
71 
72  bp_base_.prepare();
73  for (int x = 0; x < bp_base_.num_comp(); x++) {
74  /* generate chunk for inner product of beta gradient */
75  bp_base_.generate(icnk, x);
76 
77  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
78  int spin_factor = (ispn == 0 ? 1 : -1);
79 
80  int nbnd = kpoint__.num_occupied_bands(ispn);
81 
82  /* inner product of beta gradient and WF */
83  auto bp_base_phi_chunk = bp_base_.template inner<T>(icnk, kpoint__.spinor_wave_functions(), ispn, 0, nbnd);
84 
85  splindex<block> spl_nbnd(nbnd, kpoint__.comm().size(), kpoint__.comm().rank());
86 
87  int nbnd_loc = spl_nbnd.local_size();
88 
89  #pragma omp parallel for
90  for (int ia_chunk = 0; ia_chunk < bp_base_.chunk(icnk).num_atoms_; ia_chunk++) {
91  int ia = bp_base_.chunk(icnk).desc_(beta_desc_idx::ia, ia_chunk);
92  int offs = bp_base_.chunk(icnk).desc_(beta_desc_idx::offset, ia_chunk);
93  int nbf = bp_base_.chunk(icnk).desc_(beta_desc_idx::nbf, ia_chunk);
94  int iat = unit_cell.atom(ia).type_id();
95 
96  /* helper lambda to calculate for sum loop over bands for different beta_phi and dij combinations*/
97  auto for_bnd = [&](int ibf, int jbf, double_complex dij, double_complex qij, matrix<T>& beta_phi_chunk)
98  {
99  /* gather everything = - 2 Re[ occ(k,n) weight(k) beta_phi*(i,n) [ Dij - E(n)Qij] beta_base_phi(j,n) ]*/
100  for (int ibnd_loc = 0; ibnd_loc < nbnd_loc; ibnd_loc++) {
101  int ibnd = spl_nbnd[ibnd_loc];
102 
103  double_complex scalar_part = main_two_factor * kpoint__.band_occupancy(ibnd, ispn) * kpoint__.weight() *
104  std::conj(beta_phi_chunk(offs + jbf, ibnd)) * bp_base_phi_chunk(offs + ibf, ibnd) *
105  (dij - kpoint__.band_energy(ibnd, ispn) * qij);
106 
107  /* get real part and add to the result array*/
108  collect_res__(x, ia) += scalar_part.real();
109  }
110  };
111 
112  for (int ibf = 0; ibf < nbf; ibf++) {
113  for (int jbf = 0; jbf < nbf; jbf++) {
114 
115  /* Qij exists only in the case of ultrasoft/PAW */
116  double qij = unit_cell.atom(ia).type().augment() ? ctx_.augmentation_op(iat).q_mtrx(ibf, jbf) : 0.0;
117  double_complex dij = 0.0;
118 
119  /* get non-magnetic or collinear spin parts of dij*/
120  switch (ctx_.num_spins()) {
121  case 1: {
122  dij = unit_cell.atom(ia).d_mtrx(ibf, jbf, 0);
123  break;
124  }
125 
126  case 2: {
127  /* Dij(00) = dij + dij_Z ; Dij(11) = dij - dij_Z*/
128  dij = (unit_cell.atom(ia).d_mtrx(ibf, jbf, 0) + spin_factor * unit_cell.atom(ia).d_mtrx(ibf, jbf, 1));
129  break;
130  }
131 
132  default: {
133  TERMINATE("Error in non_local_functor, D_aug_mtrx. ");
134  break;
135  }
136  }
137 
138  /* add non-magnetic or diagonal spin components (or collinear part) */
139  for_bnd(ibf, jbf, dij, double_complex(qij, 0.0), beta_phi_chunks[ispn]);
140 
141  /* for non-collinear case*/
142  if (ctx_.num_mag_dims() == 3) {
143  /* Dij(10) = dij_X + i dij_Y ; Dij(01) = dij_X - i dij_Y */
144  dij = double_complex( unit_cell.atom(ia).d_mtrx(ibf, jbf, 2), spin_factor * unit_cell.atom(ia).d_mtrx(ibf, jbf, 3));
145  /* add non-diagonal spin components*/
146  for_bnd(ibf, jbf, dij, double_complex(0.0, 0.0), beta_phi_chunks[ispn + spin_factor] );
147  }
148  } // jbf
149  } // ibf
150  } // ia_chunk
151  } // ispn
152  } // x
153  }
154 
155  //bp.dismiss();
156  bp_base_.dismiss();
157  }
158 };
