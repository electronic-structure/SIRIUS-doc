50  /// collect summation result in an array
51  void add_k_point_contribution(K_point& kpoint__, mdarray<double, 2>& collect_res__)
52  {
53  Unit_cell& unit_cell = ctx_.unit_cell();
54 
55  Beta_projectors& bp = kpoint__.beta_projectors();
56 
57  double main_two_factor = -2.0;
58 
59  bp_base_.prepare();
60  bp.prepare();
61 
62  for (int icnk = 0; icnk < bp_base_.num_chunks(); icnk++) {
63  /* generate chunk for inner product of beta */
64  bp.generate(icnk);
65 
66  /* store <beta|psi> for spin up and down */
67  matrix<T> beta_phi_chunks[2];
68 
69  for(int ispn = 0; ispn < ctx_.num_spins(); ispn++){
70  int nbnd = kpoint__.num_occupied_bands(ispn);
71  auto beta_phi_tmp = bp.inner<T>(icnk, kpoint__.spinor_wave_functions(), ispn, 0, nbnd);
72  beta_phi_chunks[ispn] = matrix<T>(beta_phi_tmp.size(0), beta_phi_tmp.size(1)) ;
73  beta_phi_tmp >> beta_phi_chunks[ispn];
74  }
75 
76  for (int x = 0; x < N; x++) {
77  /* generate chunk for inner product of beta gradient */
78  bp_base_.generate(icnk, x);
79 
80  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
81  int spin_factor = (ispn == 0 ? 1 : -1);
82 
83  int nbnd = kpoint__.num_occupied_bands(ispn);
84 
85  /* inner product of beta gradient and WF */
86  auto bp_base_phi_chunk = bp_base_.template inner<T>(icnk, kpoint__.spinor_wave_functions(), ispn, 0, nbnd);
87 
88  splindex<block> spl_nbnd(nbnd, kpoint__.comm().size(), kpoint__.comm().rank());
89 
90  int nbnd_loc = spl_nbnd.local_size();
91 
92  #pragma omp parallel for
93  for (int ia_chunk = 0; ia_chunk < bp_base_.chunk(icnk).num_atoms_; ia_chunk++) {
94  int ia = bp_base_.chunk(icnk).desc_(beta_desc_idx::ia, ia_chunk);
95  int offs = bp_base_.chunk(icnk).desc_(beta_desc_idx::offset, ia_chunk);
96  int nbf = bp_base_.chunk(icnk).desc_(beta_desc_idx::nbf, ia_chunk);
97  int iat = unit_cell.atom(ia).type_id();
98 
99  /* helper lambda to calculate for sum loop over bands for different beta_phi and dij combinations*/
100  auto for_bnd = [&](int ibf, int jbf, double_complex dij, double_complex qij, matrix<T>& beta_phi_chunk)
101  {
102  /* gather everything = - 2 Re[ occ(k,n) weight(k) beta_phi*(i,n) [ Dij - E(n)Qij] beta_base_phi(j,n) ]*/
103  for (int ibnd_loc = 0; ibnd_loc < nbnd_loc; ibnd_loc++) {
104  int ibnd = spl_nbnd[ibnd_loc];
105 
106  double_complex scalar_part = main_two_factor * kpoint__.band_occupancy(ibnd, ispn) * kpoint__.weight() *
107  std::conj(beta_phi_chunk(offs + jbf, ibnd)) * bp_base_phi_chunk(offs + ibf, ibnd) *
108  (dij - kpoint__.band_energy(ibnd, ispn) * qij);
109 
110  /* get real part and add to the result array*/
111  collect_res__(x, ia) += scalar_part.real();
112  }
113  };
114 
115  for (int ibf = 0; ibf < nbf; ibf++) {
116  for (int jbf = 0; jbf < nbf; jbf++) {
117 
118  /* Qij exists only in the case of ultrasoft/PAW */
119  double qij = unit_cell.atom(ia).type().augment() ? ctx_.augmentation_op(iat).q_mtrx(ibf, jbf) : 0.0;
120  double_complex dij = 0.0;
121 
122  /* get non-magnetic or collinear spin parts of dij*/
123  switch (ctx_.num_spins()) {
124  case 1: {
125  dij = unit_cell.atom(ia).d_mtrx(ibf, jbf, 0);
126  break;
127  }
128 
129  case 2: {
130  /* Dij(00) = dij + dij_Z ; Dij(11) = dij - dij_Z*/
131  dij = (unit_cell.atom(ia).d_mtrx(ibf, jbf, 0) + spin_factor * unit_cell.atom(ia).d_mtrx(ibf, jbf, 1));
132  break;
133  }
134 
135  default: {
136  TERMINATE("Error in non_local_functor, D_aug_mtrx. ");
137  break;
138  }
139  }
140 
141  /* add non-magnetic or diagonal spin components ( or collinear part) */
142  for_bnd(ibf, jbf, dij, double_complex(qij, 0.0), beta_phi_chunks[ispn]);
143 
144  /* for non-collinear case*/
145  if (ctx_.num_mag_dims() == 3) {
146  /* Dij(10) = dij_X + i dij_Y ; Dij(01) = dij_X - i dij_Y */
147  dij = double_complex( unit_cell.atom(ia).d_mtrx(ibf, jbf, 2), spin_factor * unit_cell.atom(ia).d_mtrx(ibf, jbf, 3));
148  /* add non-diagonal spin components*/
149  for_bnd(ibf, jbf, dij, double_complex(0.0, 0.0), beta_phi_chunks[ispn + spin_factor] );
150  }
151  } // jbf
152  } // ibf
153  } // ia_chunk
154  } // ispn
155  } // x
156  }
157 
158  bp.dismiss();
159  bp_base_.dismiss();
160  }
161 };
