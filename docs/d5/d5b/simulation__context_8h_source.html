36 /// Simulation context is a set of parameters and objects describing a single simulation.
37 /** The order of initialization of the simulation context is the following: first, the default parameter
38  * values are set in the constructor, then (optionally) import() method is called and the parameters are
39  * overwritten with the those from the input file, and finally, the user sets the values with set_...() metods.
40  * Then the unit cell can be populated and the context can be initialized. */

73  /// Initialize the similation (can only be called once).
74  void initialize()
75  {
76  PROFILE("sirius::Simulation_context::initialize");
77 
78  Simulation_context_base::initialize();
79 
80  if (full_potential()) {
81  step_function_ = std::unique_ptr<Step_function>(new Step_function(*this));
82  }
83 
84  if (!full_potential()) {
85 
86  if (comm().rank() == 0 && control().print_memory_usage_) {
87  MEMORY_USAGE_INFO();
88  }
89 
90  /* create augmentation operator Q_{xi,xi'}(G) here */
91  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
92  augmentation_op_.push_back(std::move(Augmentation_operator(*this, iat)));
93 
94  if (comm().rank() == 0 && control().print_memory_usage_) {
95  MEMORY_USAGE_INFO();
96  }
97  }
98  }
99 
100  if (comm().rank() == 0 && control().print_memory_usage_) {
101  MEMORY_USAGE_INFO();
102  }
103  }
