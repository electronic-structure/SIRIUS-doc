224  /// Unit step function is defined to be 1 in the interstitial and 0 inside muffin-tins.
225  /** Unit step function is constructed from it's plane-wave expansion coefficients which are computed
226  * analytically:
227  * \f[
228  * \Theta({\bf r}) = \sum_{\bf G} \Theta({\bf G}) e^{i{\bf Gr}},
229  * \f]
230  * where
231  * \f[
232  * \Theta({\bf G}) = \frac{1}{\Omega} \int \Theta({\bf r}) e^{-i{\bf Gr}} d{\bf r} =
233  * \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} d{\bf r} - \frac{1}{\Omega} \int_{MT} e^{-i{\bf Gr}} d{\bf r} =
234  * \delta_{\bf G, 0} - \sum_{\alpha} \frac{1}{\Omega} \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r}
235  * \f]
236  * Integral of a plane-wave over the muffin-tin volume is taken using the spherical expansion of the plane-wave
237  * around central point \f$ \tau_{\alpha} \f$:
238  * \f[
239  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} =
240  * e^{-i{\bf G\tau_{\alpha}}} \int_{MT_{\alpha}} 4\pi \sum_{\ell m} (-i)^{\ell} j_{\ell}(Gr)
241  * Y_{\ell m}(\hat {\bf G}) Y_{\ell m}^{*}(\hat {\bf r}) r^2 \sin \theta dr d\phi d\theta
242  * \f]
243  * In the above integral only \f$ \ell=m=0 \f$ term survives. So we have:
244  * \f[
245  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = 4\pi e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
246  * \f]
247  * where
248  * \f[
249  * \Theta(\alpha, G) = \int_{0}^{R_{\alpha}} \frac{\sin(Gr)}{Gr} r^2 dr =
250  * \left\{ \begin{array}{ll} \displaystyle R_{\alpha}^3 / 3 & G=0 \\
251  * \Big( \sin(GR_{\alpha}) - GR_{\alpha}\cos(GR_{\alpha}) \Big) / G^3 & G \ne 0 \end{array} \right.
252  * \f]
253  * are the so-called step function form factors. With this we have a final expression for the plane-wave coefficients
254  * of the unit step function:
255  * \f[
256  * \Theta({\bf G}) = \delta_{\bf G, 0} - \sum_{\alpha} \frac{4\pi}{\Omega} e^{-i{\bf G\tau_{\alpha}}}
257  * \Theta(\alpha, G)
258  * \f]
259  */
260  void init_step_function()
261  {
262  auto v = make_periodic_function<index_domain_t::global>([&](int iat, double g) {
263  auto R = unit_cell().atom_type(iat).mt_radius();
264  if (g < 1e-12) {
265  return std::pow(R, 3) / 3.0;
266  } else {
267  return (std::sin(g * R) - g * R * std::cos(g * R)) / std::pow(g, 3);
268  }});
269 
270  theta_ = mdarray<double, 1>(fft().local_size());
271  theta_pw_ = mdarray<double_complex, 1>(gvec().num_gvec());
272 
273  for (int ig = 0; ig < gvec().num_gvec(); ig++) {
274  theta_pw_[ig] = -v[ig];
275  }
276  theta_pw_[0] += 1.0;
277 
278  std::vector<double_complex> ftmp(gvec_partition().gvec_count_fft());
279  for (int i = 0; i < gvec_partition().gvec_count_fft(); i++) {
280  ftmp[i] = theta_pw_[gvec_partition().idx_gvec(i)];
281  }
282  fft().transform<1>(ftmp.data());
283  fft().output(&theta_[0]);
284 
285  double vit{0};
286  for (int i = 0; i < fft().local_size(); i++) {
287  vit += theta_[i];
288  }
289  vit *= (unit_cell().omega() / fft().size());
290  fft().comm().allreduce(&vit, 1);
291 
292  if (std::abs(vit - unit_cell().volume_it()) > 1e-10) {
293  std::stringstream s;
294  s << "step function gives a wrong volume for IT region" << std::endl
295  << " difference with exact value : " << std::abs(vit - unit_cell().volume_it());
296  if (comm().rank() == 0) {
297  WARNING(s);
298  }
299  }
300  if (control().print_checksum_) {
301  double_complex z1 = theta_pw_.checksum();
302  double d1 = theta_.checksum();
303  fft().comm().allreduce(&d1, 1);
304  if (comm().rank() == 0) {
305  utils::print_checksum("theta", d1);
306  utils::print_checksum("theta_pw", z1);
307  }
308  }
309  }

318  void init_atoms_to_grid_idx()
319  {
320  PROFILE("sirius::Simulation_context::init_atoms_to_grid_idx");
321 
322  atoms_to_grid_idx_.resize(unit_cell_.num_atoms());
323 
324  vector3d<double> delta(1.0 / fft_->size(0), 1.0 / fft_->size(1), 1.0 / fft_->size(2));
325 
326  int z_off = fft_->offset_z();
327  vector3d<int> grid_beg(0, 0, z_off);
328  vector3d<int> grid_end(fft_->size(0), fft_->size(1), z_off + fft_->local_size_z());
329  /* approximate atom radius in bohr */
330  double R = av_atom_radius_;
331  std::vector<vector3d<double>> verts_cart{{-R,-R,-R},{R,-R,-R},{-R,R,-R},{R,R,-R},{-R,-R,R},{R,-R,R},{-R,R,R},{R,R,R}};
332 
333  auto bounds_box = [&](vector3d<double> pos)
334  {
335  std::vector<vector3d<double>> verts;
336 
337  for (auto v : verts_cart) {
338  verts.push_back(pos + unit_cell_.get_fractional_coordinates(v));
339  }
340 
341  std::pair<vector3d<int>, vector3d<int>> bounds_ind;
342 
343  size_t size = verts.size();
344  for (int x: {0, 1, 2}) {
345  std::sort(verts.begin(), verts.end(), [x](vector3d<double>& a, vector3d<double>& b){return a[x] < b[x];});
346  bounds_ind.first[x] = std::max((int)(verts[0][x] / delta[x]) - 1, grid_beg[x]);
347  bounds_ind.second[x] = std::min((int)(verts[size - 1][x] / delta[x]) + 1, grid_end[x]);
348  }
349 
350  return bounds_ind;
351  };
352 
353  #pragma omp parallel for
354  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
355 
356  std::vector<std::pair<int, double>> atom_to_inds_map;
357 
358  for (int t0 = -1; t0 <= 1; t0++) {
359  for (int t1 = -1; t1 <= 1; t1++) {
360  for (int t2 = -1; t2 <= 1; t2++) {
361  auto position = unit_cell_.atom(ia).position() + vector3d<double>(t0, t1, t2);
362 
363  auto box = bounds_box(position);
364 
365  for (int j0 = box.first[0]; j0 < box.second[0]; j0++) {
366  for (int j1 = box.first[1]; j1 < box.second[1]; j1++) {
367  for (int j2 = box.first[2]; j2 < box.second[2]; j2++) {
368  auto dist = position - vector3d<double>(delta[0] * j0, delta[1] * j1, delta[2] * j2);
369  auto r = unit_cell_.get_cartesian_coordinates(dist).length();
370  auto ir = fft_->index_by_coord(j0, j1, j2 - z_off);
371 
372  if (r <= R) {
373  atom_to_inds_map.push_back({ir, r});
374  }
375  }
376  }
377  }
378  }
379  }
380  }
381 
382  atoms_to_grid_idx_[ia] = std::move(atom_to_inds_map);
383  }

431  /// Update context after setting new lattice vectors or atomic coordinates.
432  void update()
433  {
434  PROFILE("sirius::Simulation_context::update");
435 
436  gvec_->lattice_vectors(unit_cell().reciprocal_lattice_vectors());
437  gvec_coarse_->lattice_vectors(unit_cell().reciprocal_lattice_vectors());
438 
439  unit_cell().update();
440 
441  if (unit_cell_.num_atoms() != 0 && use_symmetry() && control().verification_ >= 1) {
442  unit_cell_.symmetry().check_gvec_symmetry(gvec(), comm());
443  if (!full_potential()) {
444  unit_cell_.symmetry().check_gvec_symmetry(gvec_coarse(), comm());
445  }
446  }
447 
448  init_atoms_to_grid_idx();
449 
450  std::pair<int, int> limits(0, 0);
451  for (int x: {0, 1, 2}) {
452  limits.first = std::min(limits.first, fft().limits(x).first);
453  limits.second = std::max(limits.second, fft().limits(x).second);
454  }
455 
456  phase_factors_ = mdarray<double_complex, 3>(3, limits, unit_cell().num_atoms(), memory_t::host, "phase_factors_");
457  #pragma omp parallel for
458  for (int i = limits.first; i <= limits.second; i++) {
459  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
460  auto pos = unit_cell_.atom(ia).position();
461  for (int x: {0, 1, 2}) {
462  phase_factors_(x, i, ia) = std::exp(double_complex(0.0, twopi * (i * pos[x])));
463  }
464  }
465  }
466 
467  phase_factors_t_ = mdarray<double_complex, 2>(gvec().count(), unit_cell().num_atom_types());
468  #pragma omp parallel for schedule(static)
469  for (int igloc = 0; igloc < gvec().count(); igloc++) {
470  /* global index of G-vector */
471  int ig = gvec().offset() + igloc;
472  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
473  double_complex z(0, 0);
474  for (int ia = 0; ia < unit_cell().atom_type(iat).num_atoms(); ia++) {
475  z += gvec_phase_factor(ig, unit_cell().atom_type(iat).atom_id(ia));
476  }
477  phase_factors_t_(igloc, iat) = z;
478  }
479  }
480 
481  if (use_symmetry()) {
482  sym_phase_factors_ = mdarray<double_complex, 3>(3, limits, unit_cell().symmetry().num_mag_sym());
483 
484  #pragma omp parallel for
485  for (int i = limits.first; i <= limits.second; i++) {
486  for (int isym = 0; isym < unit_cell().symmetry().num_mag_sym(); isym++) {
487  auto t = unit_cell().symmetry().magnetic_group_symmetry(isym).spg_op.t;
488  for (int x: {0, 1, 2}) {
489  sym_phase_factors_(x, i, isym) = std::exp(double_complex(0.0, twopi * (i * t[x])));
490  }
491  }
492  }
493  }
494 #if defined(__GPU)
495  if (processing_unit() == GPU) {
496  acc::set_device();
497  gvec_coord_ = mdarray<int, 2>(gvec().count(), 3, memory_t::host | memory_t::device, "gvec_coord_");
498  for (int igloc = 0; igloc < gvec().count(); igloc++) {
499  int ig = gvec().offset() + igloc;
500  auto G = gvec().gvec(ig);
501  for (int x: {0, 1, 2}) {
502  gvec_coord_(igloc, x) = G[x];
503  }
504  }
505  gvec_coord_.copy<memory_t::host, memory_t::device>();
506  }
507 #endif
508  if (full_potential()) {
509  init_step_function();
510  }
511 
512  if (!full_potential()) {
513  augmentation_op_.clear();
514  /* create augmentation operator Q_{xi,xi'}(G) here */
515  for (int iat = 0; iat < unit_cell().num_atom_types(); iat++) {
516  augmentation_op_.push_back(std::move(Augmentation_operator(unit_cell().atom_type(iat), gvec(), comm())));
517  augmentation_op_.back().generate_pw_coeffs(aug_ri());
518  }
519  }
520  }

816  /// Sum over the plane-wave coefficients and spherical harmonics that apperas in Poisson solver and finding of the MT boundary values.
817  /** The following operation is performed:
818  * \f[
819  * q_{\ell m}^{\alpha} = \sum_{\bf G} 4\pi \rho({\bf G}) e^{i{\bf G}{\bf r}_{\alpha}}i^{\ell}f_{\ell}^{\alpha}(G) Y_{\ell m}^{*}(\hat{\bf G})
820  * \f]
821  */
822  inline mdarray<double_complex, 2> sum_fg_fl_yg(int lmax__,
823  double_complex const* fpw__,
824  mdarray<double, 3>& fl__,
825  matrix<double_complex>& gvec_ylm__)
826  {
827  PROFILE("sirius::Simulation_context::sum_fg_fl_yg");
828 
829  int ngv_loc = gvec().count();
830 
831  int na_max{0};
832  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
833  na_max = std::max(na_max, unit_cell_.atom_type(iat).num_atoms());
834  }
835 
836  int lmmax = utils::lmmax(lmax__);
837  /* resuling matrix */
838  mdarray<double_complex, 2> flm(lmmax, unit_cell().num_atoms());
839 
840  matrix<double_complex> phase_factors;
841  matrix<double_complex> zm;
842  matrix<double_complex> tmp;
843  switch (processing_unit()) {
844  case CPU: {
845  phase_factors = matrix<double_complex>(mem_pool().allocate<double_complex, memory_t::host>(ngv_loc * na_max),
846  ngv_loc, na_max);
847  zm = matrix<double_complex>(mem_pool().allocate<double_complex, memory_t::host>(lmmax * ngv_loc),
848  lmmax, ngv_loc);
849  tmp = matrix<double_complex>(mem_pool().allocate<double_complex, memory_t::host>(lmmax * na_max),
850  lmmax, na_max);
851  break;
852  }
853  case GPU: {
854  phase_factors = matrix<double_complex>(nullptr, mem_pool().allocate<double_complex, memory_t::device>(ngv_loc * na_max),
855  ngv_loc, na_max);
856  zm = matrix<double_complex>(mem_pool().allocate<double_complex, memory_t::host>(lmmax * ngv_loc),
857  mem_pool().allocate<double_complex, memory_t::device>(lmmax * ngv_loc),
858  lmmax, ngv_loc);
859  tmp = matrix<double_complex>(mem_pool().allocate<double_complex, memory_t::host>(lmmax * na_max),
860  mem_pool().allocate<double_complex, memory_t::device>(lmmax * na_max),
861  lmmax, na_max);
862  break;
863  }
864  }
865 
866  std::vector<double_complex> zil(lmax__ + 1);
867  for (int l = 0; l <= lmax__; l++) {
868  zil[l] = std::pow(double_complex(0, 1), l);
869  }
870 
871  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
872  int na = unit_cell_.atom_type(iat).num_atoms();
873  generate_phase_factors(iat, phase_factors);
874  utils::timer t1("sirius::Simulation_context::sum_fg_fl_yg|zm");
875  #pragma omp parallel for schedule(static)
876  for (int igloc = 0; igloc < ngv_loc; igloc++) {
877  for (int l = 0, lm = 0; l <= lmax__; l++) {
878  double_complex z = fourpi * fl__(l, igloc, iat) * zil[l] * fpw__[igloc];
879  for (int m = -l; m <= l; m++, lm++) {
880  zm(lm, igloc) = z * std::conj(gvec_ylm__(lm, igloc));
881  }
882  }
883  }
884  t1.stop();
885  utils::timer t2("sirius::Simulation_context::sum_fg_fl_yg|mul");
886  switch (processing_unit()) {
887  case CPU: {
888  linalg<CPU>::gemm(0, 0, lmmax, na, ngv_loc,
889  zm.at<CPU>(), zm.ld(),
890  phase_factors.at<CPU>(), phase_factors.ld(),
891  tmp.at<CPU>(), tmp.ld());
892  break;
893  }
894  case GPU: {
895 #if defined(__GPU)
896  zm.copy<memory_t::host, memory_t::device>();
897  linalg<GPU>::gemm(0, 0, lmmax, na, ngv_loc,
898  zm.at<GPU>(), zm.ld(),
899  phase_factors.at<GPU>(), phase_factors.ld(),
900  tmp.at<GPU>(), tmp.ld());
901  tmp.copy<memory_t::device, memory_t::host>();
902 #endif
903  break;
904  }
905  }
906  t2.stop();
907 
908  for (int i = 0; i < na; i++) {
909  int ia = unit_cell_.atom_type(iat).atom_id(i);
910  std::copy(&tmp(0, i), &tmp(0, i) + lmmax, &flm(0, ia));
911  }
912  }
913 
914  comm().allreduce(&flm(0, 0), (int)flm.size());
915 
916  mem_pool().reset<memory_t::host>();
917  if (processing_unit() == GPU) {
918  mem_pool().reset<memory_t::device>();
919  }
920  return std::move(flm);
921  }

978  /// Find the lambda parameter used in the Ewald summation.
979  /** lambda parameter scales the erfc function argument:
980  * \f[
981  * {\rm erf}(\sqrt{\lambda}x)
982  * \f]
983  */
984  double ewald_lambda()
985  {
986  /* alpha = 1 / (2*sigma^2), selecting alpha here for better convergence */
987  double lambda{1};
988  double gmax = pw_cutoff();
989  double upper_bound{0};
990  double charge = unit_cell_.num_electrons();
991 
992  /* iterate to find lambda */
993  do {
994  lambda += 0.1;
995  upper_bound = charge * charge * std::sqrt(2.0 * lambda / twopi) * std::erfc(gmax * std::sqrt(1.0 / (4.0 * lambda)));
996  } while (upper_bound < 1.0e-8);
997 
998  if (lambda < 1.5) {
999  std::stringstream s;
1000  s << "Ewald forces error: probably, pw_cutoff is too small.";
1001  WARNING(s);
1002  }
1003  return lambda;
1004  }
