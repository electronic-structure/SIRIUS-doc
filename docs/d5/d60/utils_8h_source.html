71  {

192  static void check_hermitian(const std::string& name, matrix<T> const& mtrx, int n = -1)
193  {
194  assert(mtrx.size(0) == mtrx.size(1));
195 
196  double maxdiff = 0.0;
197  int i0 = -1;
198  int j0 = -1;
199 
200  if (n == -1) {
201  n = static_cast<int>(mtrx.size(0));
202  }
203 
204  for (int i = 0; i < n; i++) {
205  for (int j = 0; j < n; j++) {
206  double diff = std::abs(mtrx(i, j) - type_wrapper<T>::conjugate(mtrx(j, i)));
207  if (diff > maxdiff) {
208  maxdiff = diff;
209  i0 = i;
210  j0 = j;
211  }
212  }
213  }
214 
215  if (maxdiff > 1e-10) {
216  std::stringstream s;
217  s << name << " is not a symmetric or hermitian matrix" << std::endl
218  << " maximum error: i, j : " << i0 << " " << j0 << " diff : " << maxdiff;
219 
220  WARNING(s);
221  }
222  }

224  static double confined_polynomial(double r, double R, int p1, int p2, int dm)
225  {
226  double t = 1.0 - std::pow(r / R, 2);
227  switch (dm) {
228  case 0: {
229  return (std::pow(r, p1) * std::pow(t, p2));
230  }
231  case 2: {
232  return (-4 * p1 * p2 * std::pow(r, p1) * std::pow(t, p2 - 1) / std::pow(R, 2) +
233  p1 * (p1 - 1) * std::pow(r, p1 - 2) * std::pow(t, p2) +
234  std::pow(r, p1) * (4 * (p2 - 1) * p2 * std::pow(r, 2) * std::pow(t, p2 - 2) / std::pow(R, 4) -
235  2 * p2 * std::pow(t, p2 - 1) / std::pow(R, 2)));
236  }
237  default: {
238  TERMINATE("wrong derivative order");
239  return 0.0;
240  }
241  }
