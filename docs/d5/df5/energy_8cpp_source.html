29 double ewald_energy(const Simulation_context& ctx, const Gvec& gvec, const Unit_cell& unit_cell)
30 {
31  double alpha{ctx.ewald_lambda()};
32  double ewald_g{0};
33 
34  #pragma omp parallel for reduction(+ : ewald_g)
35  for (int igloc = 0; igloc < gvec.count(); igloc++) {
36  int ig = gvec.offset() + igloc;
37  if (!ig) {
38  continue;
39  }
40 
41  double g2 = std::pow(gvec.gvec_len(ig), 2);
42 
43  double_complex rho(0, 0);
44 
45  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
46  rho += ctx.gvec_phase_factor(gvec.gvec(ig), ia) * static_cast<double>(unit_cell.atom(ia).zn());
47  }
48 
49  ewald_g += std::pow(std::abs(rho), 2) * std::exp(-g2 / 4 / alpha) / g2;
50  }
51 
52  ctx.comm().allreduce(&ewald_g, 1);
53  if (gvec.reduced()) {
54  ewald_g *= 2;
55  }
56  /* remaining G=0 contribution */
57  ewald_g -= std::pow(unit_cell.num_electrons(), 2) / alpha / 4;
58  ewald_g *= (twopi / unit_cell.omega());
59 
60  /* remove self-interaction */
61  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
62  ewald_g -= std::sqrt(alpha / pi) * std::pow(unit_cell.atom(ia).zn(), 2);
63  }
64 
65  double ewald_r{0};
66  #pragma omp parallel for reduction(+ : ewald_r)
67  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
68  for (int i = 1; i < unit_cell.num_nearest_neighbours(ia); i++) {
69  int ja = unit_cell.nearest_neighbour(i, ia).atom_id;
70  double d = unit_cell.nearest_neighbour(i, ia).distance;
71  ewald_r += 0.5 * unit_cell.atom(ia).zn() * unit_cell.atom(ja).zn() * std::erfc(std::sqrt(alpha) * d) / d;
72  }
73  }
74 
75  return (ewald_g + ewald_r);
76 }

102 double energy_enuc(Simulation_context const& ctx, Potential const& potential)
103 {
104  auto& unit_cell = ctx.unit_cell();
105  double enuc{0};
106  if (ctx.full_potential()) {
107  for (int ialoc = 0; ialoc < unit_cell.spl_num_atoms().local_size(); ialoc++) {
108  int ia = unit_cell.spl_num_atoms(ialoc);
109  int zn = unit_cell.atom(ia).zn();
110  enuc -= 0.5 * zn * potential.vh_el(ia);
111  }
112  ctx.comm().allreduce(&enuc, 1);
113  }
114  return enuc;
115 }

148 double total_energy(Simulation_context const& ctx, K_point_set const& kset, Density const& density,
149  Potential const& potential, double ewald_energy)
150 {
151  double tot_en{0};
152 
153  switch (ctx.electronic_structure_method()) {
154  case electronic_structure_method_t::full_potential_lapwlo: {
155  tot_en = (energy_kin(ctx, kset, density, potential) + energy_exc(density, potential) +
156  0.5 * energy_vha(potential) + energy_enuc(ctx, potential));
157  break;
158  }
159 
160  case electronic_structure_method_t::pseudopotential: {
161  tot_en = (kset.valence_eval_sum() - energy_vxc(density, potential) -
162  energy_bxc(density, potential) - potential.PAW_one_elec_energy(density)) -
163  0.5 * energy_vha(potential) + energy_exc(density, potential) + potential.PAW_total_energy() +
164  ewald_energy + kset.entropy_sum();
165  break;
166  }
167  }
168 
169  if (ctx.hubbard_correction()) {
170  tot_en += potential.U().hubbard_energy(density.occupation_matrix());
171  }
172 
173  return tot_en;
174 }
