4 double ewald_energy(const Simulation_context& ctx, const Gvec& gvec, const Unit_cell& unit_cell)
5 {
6  double alpha{ctx.ewald_lambda()};
7  double ewald_g{0};
8 
9  #pragma omp parallel for reduction(+ : ewald_g)
10  for (int igloc = 0; igloc < gvec.count(); igloc++) {
11  int ig = gvec.offset() + igloc;
12  if (!ig) {
13  continue;
14  }
15 
16  double g2 = std::pow(gvec.gvec_len(ig), 2);
17 
18  double_complex rho(0, 0);
19 
20  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
21  rho += ctx.gvec_phase_factor(gvec.gvec(ig), ia) * static_cast<double>(unit_cell.atom(ia).zn());
22  }
23 
24  ewald_g += std::pow(std::abs(rho), 2) * std::exp(-g2 / 4 / alpha) / g2;
25  }
26 
27  ctx.comm().allreduce(&ewald_g, 1);
28  if (gvec.reduced()) {
29  ewald_g *= 2;
30  }
31  /* remaining G=0 contribution */
32  ewald_g -= std::pow(unit_cell.num_electrons(), 2) / alpha / 4;
33  ewald_g *= (twopi / unit_cell.omega());
34 
35  /* remove self-interaction */
36  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
37  ewald_g -= std::sqrt(alpha / pi) * std::pow(unit_cell.atom(ia).zn(), 2);
38  }
39 
40  double ewald_r{0};
41  #pragma omp parallel for reduction(+ : ewald_r)
42  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
43  for (int i = 1; i < unit_cell.num_nearest_neighbours(ia); i++) {
44  int ja = unit_cell.nearest_neighbour(i, ia).atom_id;
45  double d = unit_cell.nearest_neighbour(i, ia).distance;
46  ewald_r += 0.5 * unit_cell.atom(ia).zn() * unit_cell.atom(ja).zn() * std::erfc(std::sqrt(alpha) * d) / d;
47  }
48  }
49 
50  return (ewald_g + ewald_r);
51 }
