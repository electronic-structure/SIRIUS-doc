27 // Load a scalar. If the argument is a pointer, dereference it; otherwise copy
28 // it. Allows the same kernels to be used for host and device scalars.
29 
30 // For host scalars
31 template <typename T>
32 __forceinline__ __device__ __host__ T load_scalar(T x)
33 {
34  return x;
35 }
36 
37 // For device scalars
38 template <typename T>
39 __forceinline__ __device__ __host__ T load_scalar(const T* xp)
40 {
41  return *xp;
42 }
43 
44 // For rocblas_half2, we broadcast a fp16 across two halves
45 template <>
46 __forceinline__ __device__ __host__ rocblas_half2 load_scalar(const rocblas_half2* xp)
47 {
48  auto x = *reinterpret_cast<const _Float16*>(xp);
49  return {x, x};
50 }

52 #endif // GOOGLE_TEST
