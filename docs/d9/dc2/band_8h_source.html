260  /** Compute \f$ O_{ii'} = \langle \phi_i | \hat O | \phi_{i'} \rangle \f$ operator matrix
261  * for the subspace spanned by the wave-functions \f$ \phi_i \f$. The matrix is always returned
262  * in the CPU pointer because most of the standard math libraries start from the CPU. */
263  template <typename T, typename W>
264  inline void set_subspace_mtrx(int N__,
265  int n__,
266  W& phi__,
267  W& op_phi__,
268  dmatrix<T>& mtrx__,
269  dmatrix<T>& mtrx_old__) const
270  {
271  PROFILE("sirius::Band::set_subspace_mtrx");
272 
273  assert(n__ != 0);
274  if (mtrx_old__.size()) {
275  assert(&mtrx__.blacs_grid() == &mtrx_old__.blacs_grid());
276  }
277 
278  /* copy old N x N distributed matrix */
279  if (N__ > 0) {
280  splindex<block_cyclic> spl_row(N__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
281  splindex<block_cyclic> spl_col(N__, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
282 
283  #pragma omp parallel for
284  for (int i = 0; i < spl_col.local_size(); i++) {
285  std::memcpy(&mtrx__(0, i), &mtrx_old__(0, i), spl_row.local_size() * sizeof(T));
286  }
287  }
288 
289  /* <{phi,phi_new}|Op|phi_new> */
290  inner(phi__, 0, N__ + n__, op_phi__, N__, n__, mtrx__, 0, N__);
291 
292  /* restore lower part */
293  if (N__ > 0) {
294  if (mtrx__.blacs_grid().comm().size() == 1) {
295  #pragma omp parallel for
296  for (int i = 0; i < N__; i++) {
297  for (int j = N__; j < N__ + n__; j++) {
298  mtrx__(j, i) = type_wrapper<T>::bypass(std::conj(mtrx__(i, j)));
299  }
300  }
301  } else {
302  linalg<CPU>::tranc(n__, N__, mtrx__, 0, N__, mtrx__, N__, 0);
303  }
304  }
305 
306  if (ctx_.control().print_checksum_) {
307  splindex<block_cyclic> spl_row(N__ + n__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
308  splindex<block_cyclic> spl_col(N__ + n__, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
309  double_complex cs(0, 0);
310  for (int i = 0; i < spl_col.local_size(); i++) {
311  for (int j = 0; j < spl_row.local_size(); j++) {
312  cs += mtrx__(j, i);
313  }
314  }
315  mtrx__.blacs_grid().comm().allreduce(&cs, 1);
316  if (mtrx__.blacs_grid().comm().rank() == 0) {
317  print_checksum("subspace_mtrx", cs);
318  }
319  }
320 
321  mtrx__.make_real_diag(N__ + n__);
322 
323  /* save new matrix */
324  if (mtrx_old__.size()) {
325  splindex<block_cyclic> spl_row(N__ + n__, mtrx__.blacs_grid().num_ranks_row(), mtrx__.blacs_grid().rank_row(), mtrx__.bs_row());
326  splindex<block_cyclic> spl_col(N__ + n__, mtrx__.blacs_grid().num_ranks_col(), mtrx__.blacs_grid().rank_col(), mtrx__.bs_col());
327 
328  #pragma omp parallel for
329  for (int i = 0; i < spl_col.local_size(); i++) {
330  std::memcpy(&mtrx_old__(0, i), &mtrx__(0, i), spl_row.local_size() * sizeof(T));
331  }
332  }
333  }

335  /// Diagonalize a pseudo-potential Hamiltonian.
336  template <typename T>
337  int diag_pseudo_potential(K_point* kp__) const
338  {
339  PROFILE("sirius::Band::diag_pseudo_potential");
340 
341  local_op_->prepare(kp__->gkvec());
342  ctx_.fft_coarse().prepare(kp__->gkvec().partition());
343 
344  D_operator<T> d_op(ctx_, kp__->beta_projectors());
345  Q_operator<T> q_op(ctx_, kp__->beta_projectors());
346 
347  int niter{0};
348 
349  auto& itso = ctx_.iterative_solver_input();
350  if (itso.type_ == "exact") {
351  if (ctx_.num_mag_dims() != 3) {
352  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
353  diag_pseudo_potential_exact(kp__, ispn, d_op, q_op);
354  }
355  } else {
356  STOP();
357  }
358  } else if (itso.type_ == "davidson") {
359  niter = diag_pseudo_potential_davidson(kp__, d_op, q_op);
360  } else if (itso.type_ == "rmm-diis") {
361  if (ctx_.num_mag_dims() != 3) {
362  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
363  diag_pseudo_potential_rmm_diis(kp__, ispn, d_op, q_op);
364  }
365  } else {
366  STOP();
367  }
368  } else if (itso.type_ == "chebyshev") {
369  P_operator<T> p_op(ctx_, kp__->beta_projectors(), kp__->p_mtrx());
370  if (ctx_.num_mag_dims() != 3) {
371  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
372  diag_pseudo_potential_chebyshev(kp__, ispn, d_op, q_op, p_op);
373 
374  }
375  } else {
376  STOP();
377  }
378  } else {
379  TERMINATE("unknown iterative solver type");
380  }
381 
382  ctx_.fft_coarse().dismiss();
383  return niter;
384  }

498  /// Apply the muffin-tin part of the Hamiltonian to the apw basis functions of an atom.
499  /** The following matrix is computed:
500  * \f[
501  * b_{L_2 \nu_2}^{\alpha}({\bf G'}) = \sum_{L_1 \nu_1} \sum_{L_3}
502  * a_{L_1\nu_1}^{\alpha}({\bf G'})
503  * \langle u_{\ell_1\nu_1}^{\alpha} | h_{L3}^{\alpha} | u_{\ell_2\nu_2}^{\alpha}
504  * \rangle \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
505  * \f]
506  */
507  template <spin_block_t sblock>
508  void apply_hmt_to_apw(Atom const& atom__,
509  int num_gkvec__,
510  mdarray<double_complex, 2>& alm__,
511  mdarray<double_complex, 2>& halm__) const
512  {
513  auto& type = atom__.type();
514 
515  // TODO: this is k-independent and can in principle be precomputed together with radial integrals if memory is available
516  // TODO: for spin-collinear case hmt is Hermitian; compute upper triangular part and use zhemm
517  mdarray<double_complex, 2> hmt(type.mt_aw_basis_size(), type.mt_aw_basis_size());
518  /* compute the muffin-tin Hamiltonian */
519  for (int j2 = 0; j2 < type.mt_aw_basis_size(); j2++) {
520  int lm2 = type.indexb(j2).lm;
521  int idxrf2 = type.indexb(j2).idxrf;
522  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
523  int lm1 = type.indexb(j1).lm;
524  int idxrf1 = type.indexb(j1).idxrf;
525  hmt(j1, j2) = atom__.radial_integrals_sum_L3<sblock>(idxrf1, idxrf2, gaunt_coefs_->gaunt_vector(lm1, lm2));
526  }
527  }
528  linalg<CPU>::gemm(0, 1, num_gkvec__, type.mt_aw_basis_size(), type.mt_aw_basis_size(), alm__, hmt, halm__);
529  }

531  void apply_o1mt_to_apw(Atom const& atom__,
532  int num_gkvec__,
533  mdarray<double_complex, 2>& alm__,
534  mdarray<double_complex, 2>& oalm__) const
535  {
536  auto& type = atom__.type();
537 
538  for (int j = 0; j < type.mt_aw_basis_size(); j++) {
539  int l = type.indexb(j).l;
540  int lm = type.indexb(j).lm;
541  int idxrf = type.indexb(j).idxrf;
542  for (int order = 0; order < type.aw_order(l); order++) {
543  int j1 = type.indexb().index_by_lm_order(lm, order);
544  int idxrf1 = type.indexr().index_by_l_order(l, order);
545  for (int ig = 0; ig < num_gkvec__; ig++) {
546  oalm__(ig, j) += atom__.symmetry_class().o1_radial_integral(idxrf, idxrf1) * alm__(ig, j1);
547  }
548  }
549  }

570  /// Setup the Hamiltonian and overlap matrices in APW+lo basis
571  /** The Hamiltonian matrix has the following expression:
572  * \f[
573  * H_{\mu' \mu}=\langle \varphi_{\mu' } | \hat H | \varphi_{\mu } \rangle =
574  * \left( \begin{array}{cc}
575  * H_{\bf G'G} & H_{{\bf G'}j} \\
576  * H_{j'{\bf G}} & H_{j'j}
577  * \end{array} \right)
578  * \f]
579  * APW-APW block:
580  * \f{eqnarray*}{
581  * H_{{\bf G'} {\bf G}}^{\bf k} &=& \sum_{\alpha} \sum_{L'\nu', L\nu} a_{L'\nu'}^{\alpha *}({\bf G'+k})
582  * \langle u_{\ell' \nu'}^{\alpha}Y_{\ell' m'}|\hat h^{\alpha} | u_{\ell \nu}^{\alpha}Y_{\ell m} \rangle
583  * a_{L\nu}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde V_{eff}({\bf G - G'}) \\
584  * &=& \sum_{\alpha} \sum_{\xi' } a_{\xi'}^{\alpha *}({\bf G'+k})
585  * b_{\xi'}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde V_{eff}({\bf G - G'})
586  * \f}
587  * APW-lo block:
588  * \f[
589  * H_{{\bf G'} j}^{\bf k} = \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'+k})
590  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle
591  * \f]
592  * lo-APW block:
593  * \f[
594  * H_{j' {\bf G}}^{\bf k} = \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
595  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G+k})
596  * \f]
597  * lo-lo block:
598  * \f[
599  * H_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
600  * |\hat h^{\alpha_{j}} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle \delta_{\alpha_j \alpha_{j'}}
601  * \f]
602  *
603  * The overlap matrix has the following expression:
604  * \f[
605  * O_{\mu' \mu} = \langle \varphi_{\mu'} | \varphi_{\mu} \rangle
606  * \f]
607  * APW-APW block:
608  * \f[
609  * O_{{\bf G'} {\bf G}}^{\bf k} = \sum_{\alpha} \sum_{L\nu} a_{L\nu}^{\alpha *}({\bf G'+k})
610  * a_{L\nu}^{\alpha}({\bf G+k}) + \Theta({\bf G-G'})
611  * \f]
612  *
613  * APW-lo block:
614  * \f[
615  * O_{{\bf G'} j}^{\bf k} = \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'+k})
616  * \langle u_{\ell_j \nu'}^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} \rangle
617  * \f]
618  *
619  * lo-APW block:
620  * \f[
621  * O_{j' {\bf G}}^{\bf k} =
622  * \sum_{\nu'} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} | u_{\ell_{j'} \nu'}^{\alpha_{j'}} \rangle
623  * a_{\ell_{j'} m_{j'} \nu'}^{\alpha_{j'}}({\bf G+k})
624  * \f]
625  *
626  * lo-lo block:
627  * \f[
628  * O_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} |
629  * \phi_{\ell_{j}}^{\zeta_{j} \alpha_{j}} \rangle \delta_{\alpha_{j'} \alpha_j}
630  * \delta_{\ell_{j'} \ell_j} \delta_{m_{j'} m_j}
631  * \f]
632  */

639  /// Apply LAPW Hamiltonain and overlap to the trial wave-functions.
640  /** Check the documentation of Band::set_fv_h_o() for the expressions of Hamiltonian and overlap
641  * matrices and \ref basis for the definition of the LAPW+lo basis.
642  *
643  * For the set of wave-functions expanded in LAPW+lo basis (k-point index is dropped for simplicity)
644  * \f[
645  * \psi_{i} = \sum_{\mu} \phi_{\mu} C_{\mu i}
646  * \f]
647  * where \f$ \mu = \{ {\bf G}, j \} \f$ is a combined index of LAPW and local orbitals we want to contrusct
648  * a subspace Hamiltonian and overlap matrices:
649  * \f[
650  * H_{i' i} = \langle \psi_{i'} | \hat H | \psi_i \rangle =
651  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i} =
652  * \sum_{\mu'} C_{\mu' i'}^{*} h_{\mu' i}(\psi)
653  * \f]
654  * \f[
655  * O_{i' i} = \langle \psi_{i'} | \psi_i \rangle =
656  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i} =
657  * \sum_{\mu'} C_{\mu' i'}^{*} o_{\mu' i}(\psi)
658  * \f]
659  * where
660  * \f[
661  * h_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i}
662  * \f]
663  * and
664  * \f[
665  * o_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i}
666  * \f]
667  * For the APW block of \f$ h_{\mu' i}(\psi) \f$ and \f$ o_{\mu' i}(\psi) \f$ we have:
668  * \f[
669  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
670  * \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i}
671  * \f]
672  * \f[
673  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
674  * \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i}
675  * \f]
676  * and for the lo block:
677  * \f[
678  * h_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
679  * \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i}
680  * \f]
681  * \f[
682  * o_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
683  * \sum_{j} \langle \phi_{j'} | \phi_{j} \rangle C_{j i}
684  * \f]
685  *
686  * APW-APW contribution, muffin-tin part:
687  * \f[
688  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
689  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) b_{\xi'}^{\alpha}({\bf G})
690  * C_{{\bf G} i}
691  * \f]
692  * \f[
693  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
694  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) a_{\xi'}^{\alpha}({\bf G})
695  * C_{{\bf G} i}
696  * \f]
697  * APW-APW contribution, interstitial effective potential part:
698  * \f[
699  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} V({\bf r}) \psi_{i}({\bf r}) d{\bf r}
700  * \f]
701  * This is done by transforming \f$ \psi_i({\bf G}) \f$ to the real space, multiplying by effectvive potential
702  * and step function and transforming the result back to the \f$ {\bf G} \f$ domain.
703  *
704  * APW-APW contribution, interstitial kinetic energy part:
705  * \f[
706  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \Big( -\frac{1}{2} \nabla \Big)
707  * \Big( \nabla \psi_{i}({\bf r}) \Big) d{\bf r}
708  * \f]
709  * and the gradient of the wave-function is computed with FFT as:
710  * \f[
711  * \Big( \nabla \psi_{i}({\bf r}) \Big) = \sum_{\bf G} i{\bf G}e^{i{\bf G}{\bf r}}\psi_i({\bf G})
712  * \f]
713  *
714  * APW-APW contribution, interstitial overlap:
715  * \f[
716  * o_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \psi_{i}({\bf r}) d{\bf r}
717  * \f]
718  *
719  * APW-lo contribution:
720  * \f[
721  * h_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i} =
722  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
723  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle =
724  * \sum_{j} C_{j i} \sum_{\xi'} a_{\xi'}^{\alpha_j *}({\bf G'}) h_{\xi' \xi_j}^{\alpha_j}
725  * \f]
726  * \f[
727  * o_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i} =
728  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
729  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}| \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle =
730  * \sum_{j} C_{j i} \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'}) o_{\nu' \zeta_j \ell_j}^{\alpha_j}
731  * \f]
732  * lo-APW contribution:
733  * \f[
734  * h_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
735  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
736  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
737  * \sum_{\bf G} C_{{\bf G} i} \sum_{\xi} h_{\xi_{j'} \xi}^{\alpha_{j'}} a_{\xi}^{\alpha_{j'}}({\bf G})
738  * \f]
739  * \f[
740  * o_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
741  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
742  * | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
743  * \sum_{\bf G} C_{{\bf G} i} \sum_{\nu} o_{\zeta_{j'} \nu \ell_{j'}}^{\alpha_{j'}} a_{\ell_{j'} m_{j'} \nu}^{\alpha_{j'}}({\bf G})
744  * \f]
745  * lo-lo contribution:
746  * \f[
747  * h_{j' i}(\psi) = \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i} = \sum_{j} C_{j i} h_{\xi_{j'} \xi_j}^{\alpha_j}
748  * \delta_{\alpha_j \alpha_{j'}}
749  * \f]
750  * \f[
751  * o_{j' i}(\psi) = \sum_{j} \langle \phi_{j'} | \phi_{j} \rangle C_{j i} = \sum_{j} C_{j i}
752  * o_{\zeta_{j'} \zeta_{j} \ell_j}^{\alpha_j}
753  * \delta_{\alpha_j \alpha_{j'}} \delta_{\ell_j \ell_{j'}} \delta_{m_j m_{j'}}
754  * \f]
755  */
