32 __global__ void sum_q_pw_dm_pw_gpu_kernel
33 (
34  int nbf__,
35  double const* q_pw__,
36  double const* dm_pw__,
37  double const* sym_weight__,
38  acc_complex_double_t* rho_pw__
39 )
40 {
41  ACC_DYNAMIC_SHARED( char, sdata_ptr)
42  double* rho_re = (double*)&sdata_ptr[0];
43  double* rho_im = (double*)&sdata_ptr[sizeof(double) * blockDim.x];
44 
45  int igloc = blockIdx.x;
46 
47  rho_re[threadIdx.x] = 0;
48  rho_im[threadIdx.x] = 0;
49 
50  int ld = nbf__ * (nbf__ + 1) / 2;
51 
52  int N = num_blocks(ld, blockDim.x);
53 
54  for (int n = 0; n < N; n++) {
55  int i = n * blockDim.x + threadIdx.x;
56  if (i < ld) {
57  double qx = q_pw__[array2D_offset(i, 2 * igloc, ld)];
58  double qy = q_pw__[array2D_offset(i, 2 * igloc + 1, ld)];
59  double dx = dm_pw__[array2D_offset(i, 2 * igloc, ld)];
60  double dy = dm_pw__[array2D_offset(i, 2 * igloc + 1, ld)];
61 
62  rho_re[threadIdx.x] += sym_weight__[i] * (dx * qx - dy * qy);
63  rho_im[threadIdx.x] += sym_weight__[i] * (dy * qx + dx * qy);
64  }
65  }
66  __syncthreads();
67 
68  for (int s = 1; s < blockDim.x; s *= 2) {
69  if (threadIdx.x % (2 * s) == 0) {
70  rho_re[threadIdx.x] = rho_re[threadIdx.x] + rho_re[threadIdx.x + s];
71  rho_im[threadIdx.x] = rho_im[threadIdx.x] + rho_im[threadIdx.x + s];
72  }
73  __syncthreads();
74  }
75  if (threadIdx.x == 0) {
76  rho_pw__[igloc] = accCadd(rho_pw__[igloc], make_accDoubleComplex(rho_re[0], rho_im[0]));
77  }
78 }
