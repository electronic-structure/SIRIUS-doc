35 /// Compute a metric tensor.
36 inline matrix3d<double>
37 metric_tensor(matrix3d<double> const& lat_vec__)
38 {
39  return dot(transpose(lat_vec__), lat_vec__);
40 }

42 /// Compute error of the symmetry-transformed metric tensor.
43 inline double
44 metric_tensor_error(matrix3d<double> const& lat_vec__, matrix3d<int> const& R__)
45 {
46  auto mt = metric_tensor(lat_vec__);
47 
48  double diff{0};
49  auto mt1 = dot(dot(transpose(R__), mt), R__);
50  for (int i: {0, 1, 2}) {
51  for (int j: {0, 1, 2}) {
52  diff = std::max(diff, std::abs(mt1(i, j) - mt(i, j)));
53  }
54  }
55  return diff;
56 }

58 inline std::vector<matrix3d<int>>
59 find_lat_sym(matrix3d<double> const& lat_vec__, double tol__)
60 {
61  std::vector<matrix3d<int>> lat_sym;
62 
63  auto r = {-1, 0, 1};
64 
65  for (int i00: r) {
66  for (int i01: r) {
67  for (int i02: r) {
68  for (int i10: r) {
69  for (int i11: r) {
70  for (int i12: r) {
71  for (int i20: r) {
72  for (int i21: r) {
73  for (int i22: r) {
74  /* build a trial symmetry operation */
75  matrix3d<int> R({{i00, i01, i02}, {i10, i11, i12}, {i20, i21, i22}});
76  /* valid symmetry operation has a determinant of +/- 1 */
77  if (std::abs(R.det()) == 1) {
78  /* metric tensor should be invariant under symmetry operation */
79  if (metric_tensor_error(lat_vec__, R) < tol__) {
80  lat_sym.push_back(R);
81  }
82  }
83  }
84  }
85  }
86  }
87  }
88  }
89  }
90  }
91  }
92 
93  if (lat_sym.size() == 0 || lat_sym.size() > 48) {
94  std::stringstream s;
95  s << "wrong number of lattice symmetries: " << lat_sym.size();
96  RTE_THROW(s);
97  }

99  return lat_sym;
