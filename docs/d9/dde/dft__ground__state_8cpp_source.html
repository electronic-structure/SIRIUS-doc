54 double DFT_ground_state::energy_kin_sum_pw() const
55 {
56  double ekin{0};
57 
58  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
59  int ik = kset_.spl_num_kpoints(ikloc);
60  auto kp = kset_[ik];
61 
62  #pragma omp parallel for schedule(static) reduction(+:ekin)
63  for (int igloc = 0; igloc < kp->num_gkvec_loc(); igloc++) {
64  auto Gk = kp->gkvec().gkvec_cart<index_domain_t::local>(igloc);
65 
66  double d{0};
67  for (int ispin = 0; ispin < ctx_.num_spins(); ispin++) {
68  for (int i = 0; i < kp->num_occupied_bands(ispin); i++) {
69  double f = kp->band_occupancy(i, ispin);
70  auto z = kp->spinor_wave_functions().pw_coeffs(ispin).prime(igloc, i);
71  d += f * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
72  }
73  }
74  if (kp->gkvec().reduced()) {
75  d *= 2;
76  }
77  ekin += 0.5 * d * kp->weight() * Gk.length2();
78  } // igloc
79  } // ikloc
80  ctx_.comm().allreduce(&ekin, 1);
81  return ekin;
82 }

128 /// A quick check of self-constent density in case of pseudopotential.
129 json DFT_ground_state::check_scf_density()
130 {
131  if (ctx_.full_potential()) {
132  return json();
133  }
134  std::vector<double_complex> rho_pw(ctx_.gvec().count());
135  for (int ig = 0; ig < ctx_.gvec().count(); ig++) {
136  rho_pw[ig] = density_.rho().f_pw_local(ig);
137  }
138 
139  double etot = total_energy();
140 
141  /* create new potential */
142  Potential pot(ctx_);
143  /* generate potential from existing density */
144  pot.generate(density_);
145  /* create new Hamiltonian */
146  Hamiltonian0 H0(pot);
147  /* set the high tolerance */
148  ctx_.iterative_solver_tolerance(ctx_.settings().itsol_tol_min_);
149  /* initialize the subspace */
150  Band(ctx_).initialize_subspace(kset_, H0);
151  /* find new wave-functions */
152  Band(ctx_).solve(kset_, H0, true);
153  /* find band occupancies */
154  kset_.find_band_occupancies();
155  /* generate new density from the occupied wave-functions */
156  density_.generate(kset_, true, false);
157  /* symmetrize density and magnetization */
158  if (ctx_.use_symmetry()) {
159  density_.symmetrize();
160  if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {
161  density_.symmetrize_density_matrix();
162  }
163  }
164  density_.fft_transform(1);
165  double rms{0};
166  for (int ig = 0; ig < ctx_.gvec().count(); ig++) {
167  rms += std::pow(std::abs(density_.rho().f_pw_local(ig) - rho_pw[ig]), 2);
168  }
169  ctx_.comm().allreduce(&rms, 1);
170  json dict;
171  dict["rss"] = rms;
172  dict["rms"] = std::sqrt(rms / ctx_.gvec().num_gvec());
173  dict["detot"] = total_energy() - etot;
174 
175  ctx_.message(1, __function_name__, "RSS: %18.12E\n", dict["rss"].get<double>());
176  ctx_.message(1, __function_name__, "RMS: %18.12E\n", dict["rms"].get<double>());
177  ctx_.message(1, __function_name__, "dEtot: %18.12E\n", dict["detot"].get<double>());
178  ctx_.message(1, __function_name__, "Eold: %18.12E Enew: %18.12E\n", etot, total_energy());
179 
180  return dict;
181 }

183 json DFT_ground_state::find(double rms_tol, double energy_tol, double initial_tolerance, int num_dft_iter, bool write_state)
184 {
185  PROFILE("sirius::DFT_ground_state::scf_loop");
186 
187  auto tstart = std::chrono::high_resolution_clock::now();
188 
189  double eold{0}, rms{0};
190 
191  density_.mixer_init(ctx_.mixer_input());
192 
193  int num_iter{-1};
194  std::vector<double> rms_hist;
195  std::vector<double> etot_hist;
196 
197  if (ctx_.hubbard_correction()) { // TODO: move to inititialization functions
198  potential_.U().hubbard_compute_occupation_numbers(kset_);
199  potential_.U().calculate_hubbard_potential_and_energy();
200  }
201 
202  ctx_.iterative_solver_tolerance(initial_tolerance);
203 
204  for (int iter = 0; iter < num_dft_iter; iter++) {
205  PROFILE("sirius::DFT_ground_state::scf_loop|iteration");
206 
207  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
208  std::printf("\n");
209  std::printf("+------------------------------+\n");
210  std::printf("| SCF iteration %3i out of %3i |\n", iter, num_dft_iter);
211  std::printf("+------------------------------+\n");
212  }
213  Hamiltonian0 H0(potential_);
214  /* find new wave-functions */
215  Band(ctx_).solve(kset_, H0, true);
216  /* find band occupancies */
217  kset_.find_band_occupancies();
218  /* generate new density from the occupied wave-functions */
219  density_.generate(kset_, ctx_.use_symmetry(), true, true);
220 
221  /* mix density */
222  rms = density_.mix();
223 
224  double old_tol = ctx_.iterative_solver_tolerance();
225  /* estimate new tolerance of iterative solver */
226  double tol = std::min(ctx_.settings().itsol_tol_scale_[0] * rms, ctx_.settings().itsol_tol_scale_[1] * old_tol);
227  tol = std::max(ctx_.settings().itsol_tol_min_, tol);
228  /* set new tolerance of iterative solver */
229  ctx_.iterative_solver_tolerance(tol);
230 
231  /* check number of elctrons */
232  density_.check_num_electrons();
233 
234  /* compute new potential */
235  potential_.generate(density_);
236 
237  if (!ctx_.full_potential() && ctx_.control().verification_ >= 2) {
238  ctx_.message(1, __function_name__, "%s", "checking functional derivative of Exc\n");
239  double eps{0.1};
240  for (int i = 0; i < 10; i++) {
241  Potential p1(ctx_);
242  p1.scale_rho_xc(1 + eps);
243  p1.generate(density_);
244 
245  double evxc = potential_.energy_vxc(density_) + potential_.energy_vxc_core(density_) + energy_bxc(density_, potential_);
246  double deriv = (p1.energy_exc(density_) - potential_.energy_exc(density_)) / eps;
247 
248  std::printf("eps : %18.12f\n", eps);
249  std::printf("Energy Vxc : %18.12f\n", evxc);
250  std::printf("numerical deriv : %18.12f\n", deriv);
251  std::printf("difference : %18.12f\n", std::abs(evxc - deriv));
252  eps /= 10;
253  }
254  }
255 
256  /* symmetrize potential and effective magnetic field */
257  if (ctx_.use_symmetry()) {
258  potential_.symmetrize();
259  }
260 
261  /* transform potential to real space after symmetrization */
262  potential_.fft_transform(1);
263 
264  /* compute new total energy for a new density */
265  double etot = total_energy();
266 
267  etot_hist.push_back(etot);
268 
269  rms_hist.push_back(rms);
270 
271  /* write some information */
272  print_info();
273  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
274  std::printf("iteration : %3i, RMS %18.12E, energy difference : %18.12E\n", iter, rms, etot - eold);
275  }
276  /* check if the calculation has converged */
277  if (std::abs(eold - etot) < energy_tol && rms < rms_tol) {
278  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
279  std::printf("\n");
280  std::printf("converged after %i SCF iterations!\n", iter + 1);
281  }
282  num_iter = iter;
283  break;
284  }
285 
286  /* Compute the hubbard correction */
287  if (ctx_.hubbard_correction()) {
288  potential_.U().hubbard_compute_occupation_numbers(kset_);
289  potential_.U().calculate_hubbard_potential_and_energy();
290  }
291 
292  eold = etot;
293  }
294 
295  if (write_state) {
296  ctx_.create_storage_file();
297  if (ctx_.full_potential()) { // TODO: why this is necessary?
298  density_.rho().fft_transform(-1);
299  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
300  density_.magnetization(j).fft_transform(-1);
301  }
302  }
303  potential_.save();
304  density_.save();
305  //kset_.save(storage_file_name);
306  }
307 
308  auto tstop = std::chrono::high_resolution_clock::now();
309 
310  json dict = serialize();
311  dict["scf_time"] = std::chrono::duration_cast<std::chrono::duration<double>>(tstop - tstart).count();
312  dict["etot_history"] = etot_hist;
313  if (num_iter >= 0) {
314  dict["converged"] = true;
315  dict["num_scf_iterations"] = num_iter;
316  dict["rms_history"] = rms_hist;
317  } else {
318  dict["converged"] = false;
319  }
320 
321  //if (ctx_.control().verification_ >= 1) {
322  // check_scf_density();
323  //}
324 
325  // dict["volume"] = ctx.unit_cell().omega() * std::pow(bohr_radius, 3);
326  // dict["volume_units"] = "angstrom^3";
327  // dict["energy"] = dft.total_energy() * ha2ev;
328  // dict["energy_units"] = "eV";
329 
330  return dict;
