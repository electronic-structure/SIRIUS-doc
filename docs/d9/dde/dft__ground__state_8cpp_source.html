29 double DFT_ground_state::ewald_energy() const
30 {
31  PROFILE("sirius::DFT_ground_state::ewald_energy");
32 
33  double alpha = ctx_.ewald_lambda();
34 
35  double ewald_g{0};
36 
37  int ig0 = ctx_.gvec().skip_g0();
38 
39  #pragma omp parallel for schedule(static) reduction(+:ewald_g)
40  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
41  int ig = ctx_.gvec().offset() + igloc;
42 
43  double g2 = std::pow(ctx_.gvec().gvec_len(ig), 2);
44 
45  double_complex rho(0, 0);
46 
47  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
48  rho += ctx_.gvec_phase_factor(ig, ia) * static_cast<double>(unit_cell_.atom(ia).zn());
49  }
50 
51  ewald_g += std::pow(std::abs(rho), 2) * std::exp(-g2 / 4 / alpha) / g2;
52  }
53 
54  ctx_.comm().allreduce(&ewald_g, 1);
55  if (ctx_.gvec().reduced()) {
56  ewald_g *= 2;
57  }
58  /* remaining G=0 contribution */
59  ewald_g -= std::pow(unit_cell_.num_electrons(), 2) / alpha / 4;
60  ewald_g *= (twopi / unit_cell_.omega());
61 
62  /* remove self-interaction */
63  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
64  ewald_g -= std::sqrt(alpha / pi) * std::pow(unit_cell_.atom(ia).zn(), 2);
65  }
66 
67  double ewald_r{0};
68  #pragma omp parallel for reduction(+:ewald_r)
69  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
70  for (int i = 1; i < unit_cell_.num_nearest_neighbours(ia); i++) {
71  int ja = unit_cell_.nearest_neighbour(i, ia).atom_id;
72  double d = unit_cell_.nearest_neighbour(i, ia).distance;
73  ewald_r += 0.5 * unit_cell_.atom(ia).zn() * unit_cell_.atom(ja).zn() *
74  std::erfc(std::sqrt(alpha) * d) / d;
75  }
76  }
77 
78  return (ewald_g + ewald_r);
79 }

132 /// Return nucleus energy in the electrostatic field.
133 /** Compute energy of nucleus in the electrostatic potential generated by the total (electrons + nuclei)
134  * charge density. Diverging self-interaction term z*z/|r=0| is excluded. */
135 double DFT_ground_state::energy_enuc() const
136 {
137  double enuc{0};
138  if (ctx_.full_potential()) {
139  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
140  int ia = unit_cell_.spl_num_atoms(ialoc);
141  int zn = unit_cell_.atom(ia).zn();
142  enuc -= 0.5 * zn * potential_.vh_el(ia);
143  }
144  ctx_.comm().allreduce(&enuc, 1);
145  }
146  return enuc;
147 }

159 double DFT_ground_state::energy_kin_sum_pw() const
160 {
161  double ekin{0};
162 
163  for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
164  int ik = kset_.spl_num_kpoints(ikloc);
165  auto kp = kset_[ik];
166 
167  #pragma omp parallel for schedule(static) reduction(+:ekin)
168  for (int igloc = 0; igloc < kp->num_gkvec_loc(); igloc++) {
169  auto Gk = kp->gkvec().gkvec_cart<index_domain_t::local>(igloc);
170 
171  double d{0};
172  for (int ispin = 0; ispin < ctx_.num_spins(); ispin++) {
173  for (int i = 0; i < kp->num_occupied_bands(ispin); i++) {
174  double f = kp->band_occupancy(i, ispin);
175  auto z = kp->spinor_wave_functions().pw_coeffs(ispin).prime(igloc, i);
176  d += f * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
177  }
178  }
179  if (kp->gkvec().reduced()) {
180  d *= 2;
181  }
182  ekin += 0.5 * d * kp->weight() * Gk.length2();
183  } // igloc
184  } // ikloc
185  ctx_.comm().allreduce(&ekin, 1);
186  return ekin;
187 }

189 double DFT_ground_state::total_energy() const
190 {
191  double tot_en{0};
192 
193  switch (ctx_.electronic_structure_method()) {
194  case electronic_structure_method_t::full_potential_lapwlo: {
195  tot_en = (energy_kin() + energy_exc() + 0.5 * energy_vha() + energy_enuc());
196  break;
197  }
198 
199  case electronic_structure_method_t::pseudopotential: {
200  tot_en = (kset_.valence_eval_sum() - energy_vxc() - energy_bxc() - potential_.PAW_one_elec_energy()) -
201  0.5 * energy_vha() + energy_exc() + potential_.PAW_total_energy() + ewald_energy_;
202  break;
203  }
204  }
205 
206  if (ctx_.hubbard_correction()) {
207  tot_en += potential_.U().hubbard_energy();
208  }
209 
210  return tot_en;
211 }

252 /// A quick check of self-constent density in case of pseudopotential.
253 json DFT_ground_state::check_scf_density()
254 {
255  if (ctx_.full_potential()) {
256  return json();
257  }
258  std::vector<double_complex> rho_pw(ctx_.gvec().count());
259  for (int ig = 0; ig < ctx_.gvec().count(); ig++) {
260  rho_pw[ig] = density_.rho().f_pw_local(ig);
261  }
262 
263  double etot = total_energy();
264 
265  /* create new potential */
266  Potential pot(ctx_);
267  /* generate potential from existing density */
268  pot.generate(density_);
269  /* create new Hamiltonian */
270  Hamiltonian0 H0(pot);
271  /* set the high tolerance */
272  ctx_.iterative_solver_tolerance(ctx_.settings().itsol_tol_min_);
273  /* initialize the subspace */
274  Band(ctx_).initialize_subspace(kset_, H0);
275  /* find new wave-functions */
276  Band(ctx_).solve(kset_, H0, true);
277  /* find band occupancies */
278  kset_.find_band_occupancies();
279  /* generate new density from the occupied wave-functions */
280  density_.generate(kset_, true, false);
281  /* symmetrize density and magnetization */
282  if (ctx_.use_symmetry()) {
283  density_.symmetrize();
284  if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {
285  density_.symmetrize_density_matrix();
286  }
287  }
288  density_.fft_transform(1);
289  double rms{0};
290  for (int ig = 0; ig < ctx_.gvec().count(); ig++) {
291  rms += std::pow(std::abs(density_.rho().f_pw_local(ig) - rho_pw[ig]), 2);
292  }
293  ctx_.comm().allreduce(&rms, 1);
294  json dict;
295  dict["rss"] = rms;
296  dict["rms"] = std::sqrt(rms / ctx_.gvec().num_gvec());
297  dict["detot"] = total_energy() - etot;
298 
299  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
300  printf("[check_scf_density] RSS: %18.12E\n", dict["rss"].get<double>());
301  printf("[check_scf_density] RMS: %18.12E\n", dict["rms"].get<double>());
302  printf("[check_scf_density] dEtot: %18.12E\n", dict["detot"].get<double>());
303  printf("[check_scf_density] Eold: %18.12E Enew: %18.12E\n", etot, total_energy());
304  }
305 
306  return dict;
307 }

309 json DFT_ground_state::find(double rms_tol, double energy_tol, double initial_tolerance, int num_dft_iter, bool write_state)
310 {
311  PROFILE("sirius::DFT_ground_state::scf_loop");
312 
313  double eold{0}, rms{0};
314 
315  bool mix_density_and_potential{false};
316 
317  if (ctx_.full_potential()) {
318  potential_.mixer_init(ctx_.mixer_input());
319  if (mix_density_and_potential) {
320  Mixer_input i1 = ctx_.mixer_input();
321  i1.type_ = "linear";
322  i1.beta_ = 0.1;
323  density_.mixer_init(i1);
324  }
325  } else {
326  density_.mixer_init(ctx_.mixer_input());
327  }
328 
329  int num_iter{-1};
330  std::vector<double> rms_hist;
331 
332  if (ctx_.hubbard_correction()) {
333  potential_.U().hubbard_compute_occupation_numbers(kset_);
334  potential_.U().calculate_hubbard_potential_and_energy();
335  }
336 
337  ctx_.iterative_solver_tolerance(initial_tolerance);
338 
339  for (int iter = 0; iter < num_dft_iter; iter++) {
340  utils::timer t1("sirius::DFT_ground_state::scf_loop|iteration");
341 
342  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
343  printf("\n");
344  printf("+------------------------------+\n");
345  printf("| SCF iteration %3i out of %3i |\n", iter, num_dft_iter);
346  printf("+------------------------------+\n");
347  }
348  Hamiltonian0 H0(potential_);
349  /* find new wave-functions */
350  Band(ctx_).solve(kset_, H0, true);
351  /* find band occupancies */
352  kset_.find_band_occupancies();
353  /* generate new density from the occupied wave-functions */
354  density_.generate(kset_, true, false);
355  /* symmetrize density and magnetization */
356  if (ctx_.use_symmetry()) {
357  density_.symmetrize();
358  if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {
359  density_.symmetrize_density_matrix();
360  }
361  }
362 
363  if (!ctx_.full_potential()) {
364  /* mix density */
365  rms = density_.mix();
366  /* estimate new tolerance of iterative solver */
367  //double tol = std::max(1e-12, 0.1 * density_.dr2() / ctx_.unit_cell().num_valence_electrons());
368  double tol = std::max(ctx_.settings().itsol_tol_min_, 0.0001 * rms);
369  /* print dr2 of mixer and current iterative solver tolerance */
370  //if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
371  // printf("dr2: %18.12E, tol: %18.12E\n", density_.dr2(), tol);
372  //}
373  /* set new tolerance of iterative solver */
374  ctx_.iterative_solver_tolerance(std::min(ctx_.iterative_solver_tolerance(), tol));
375  // TODO: this is horrible when PAW density is generated from the mixed
376  // density matrix here; better solution: generate in Density and
377  // then mix
378  /* generate PAW density from density matrix */
379  density_.generate_paw_loc_density();
380  }
381 
382  /* transform density to real space after mixing and symmetrization */
383  density_.fft_transform(1);
384 
385  /* check number of elctrons */
386  density_.check_num_electrons();
387 
388  if (ctx_.full_potential() && mix_density_and_potential) {
389  density_.mix();
390  }
391 
392  /* compute new potential */
393  potential_.generate(density_);
394 
395  if (!ctx_.full_potential() && ctx_.control().verification_ >= 1) {
396  double eps{0.1};
397  for (int i = 0; i < 10; i++) {
398  Potential p1(ctx_);
399  p1.scale_rho_xc(1 + eps);
400  p1.generate(density_);
401 
402  double evxc = potential_.energy_vxc(density_) + potential_.energy_vxc_core(density_) + energy_bxc();
403  double deriv = (p1.energy_exc(density_) - potential_.energy_exc(density_)) / eps;
404 
405  printf("eps : %18.12f\n", eps);
406  printf("Energy Vxc : %18.12f\n", evxc);
407  printf("numerical deriv : %18.12f\n", deriv);
408  printf("difference : %18.12f\n", std::abs(evxc - deriv));
409  eps /= 10;
410  }
411  }
412 
413  /* symmetrize potential and effective magnetic field */
414  if (ctx_.use_symmetry()) {
415  potential_.symmetrize();
416  }
417 
418  /* transform potential to real space after symmetrization */
419  potential_.fft_transform(1);
420 
421  /* compute new total energy for a new density */
422  double etot = total_energy();
423 
424  if (ctx_.full_potential()) {
425  rms = potential_.mix(ctx_.settings().mixer_rss_min_);
426  double tol = std::max(ctx_.settings().itsol_tol_min_, 0.001 * rms);
427  ctx_.iterative_solver_tolerance(std::min(ctx_.iterative_solver_tolerance(), tol));
428  }
429  rms_hist.push_back(rms);
430 
431  /* write some information */
432  print_info();
433  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
434  printf("iteration : %3i, RMS %18.12E, energy difference : %18.12E\n", iter, rms, etot - eold);
435  }
436  /* check if the calculation has converged */
437  if (std::abs(eold - etot) < energy_tol && rms < rms_tol) {
438  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 1) {
439  printf("\n");
440  printf("converged after %i SCF iterations!\n", iter + 1);
441  }
442  num_iter = iter;
443  break;
444  }
445 
446  /* Compute the hubbard correction */
447  if (ctx_.hubbard_correction()) {
448  potential_.U().hubbard_compute_occupation_numbers(kset_);
449  potential_.U().calculate_hubbard_potential_and_energy();
450  }
451 
452  eold = etot;
453  }
454 
455  if (write_state) {
456  ctx_.create_storage_file();
457  if (ctx_.full_potential()) { // TODO: why this is necessary?
458  density_.rho().fft_transform(-1);
459  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
460  density_.magnetization(j).fft_transform(-1);
461  }
462  }
463  potential_.save();
464  density_.save();
465  //kset_.save(storage_file_name);
466  }
467 
468  json dict = serialize();
469  if (num_iter >= 0) {
470  dict["converged"] = true;
471  dict["num_scf_iterations"] = num_iter;
472  dict["rms_history"] = rms_hist;
473  } else {
474  dict["converged"] = false;
475  }
476 
477  //if (ctx_.control().verification_ >= 1) {
478  // check_scf_density();
479  //}
480 
481  // dict["volume"] = ctx.unit_cell().omega() * std::pow(bohr_radius, 3);
482  // dict["volume_units"] = "angstrom^3";
483  // dict["energy"] = dft.total_energy() * ha2ev;
484  // dict["energy_units"] = "eV";
485 
486  return dict;
487 }
