65double DFT_ground_state::energy_kin_sum_pw() const
66{
67 double ekin{0};
68
69 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
70 int ik = kset_.spl_num_kpoints(ikloc);
71 auto kp = kset_.get<double>(ik);
72
73 #pragma omp parallel for schedule(static) reduction(+:ekin)
74 for (int igloc = 0; igloc < kp->num_gkvec_loc(); igloc++) {
75 auto Gk = kp->gkvec().gkvec_cart<index_domain_t::local>(igloc);
76
77 double d{0};
78 for (int ispin = 0; ispin < ctx_.num_spins(); ispin++) {
79 for (int i = 0; i < kp->num_occupied_bands(ispin); i++) {
80 double f = kp->band_occupancy(i, ispin);
81 auto z = kp->spinor_wave_functions().pw_coeffs(ispin).prime(igloc, i);
82 d += f * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
83 }
84 }
85 if (kp->gkvec().reduced()) {
86 d *= 2;
87 }
88 ekin += 0.5 * d * kp->weight() * Gk.length2();
89 } // igloc
90 } // ikloc
91 ctx_.comm().allreduce(&ekin, 1);
92 return ekin;
93}

128/// A quick check of self-constent density in case of pseudopotential.
129json DFT_ground_state::check_scf_density()
130{
131 if (ctx_.full_potential()) {
132 return json();
133 }
134 std::vector<double_complex> rho_pw(ctx_.gvec().count());
135 for (int ig = 0; ig < ctx_.gvec().count(); ig++) {
136 rho_pw[ig] = density_.rho().f_pw_local(ig);
137 }
138
139 double etot = total_energy();
140
141 /* create new potential */
142 Potential pot(ctx_);
143 /* generate potential from existing density */
144 pot.generate(density_, ctx_.use_symmetry(), true);
145 /* create new Hamiltonian */
146 Hamiltonian0<double> H0(pot);
147 /* initialize the subspace */
148 Band(ctx_).initialize_subspace(kset_, H0);
149 /* find new wave-functions */
150 Band(ctx_).solve<double, double>(kset_, H0, true, ctx_.cfg().settings().itsol_tol_min());
151 /* find band occupancies */
152 kset_.find_band_occupancies<double>();
153 /* generate new density from the occupied wave-functions */
154 density_.generate<double>(kset_, true, true, false);
155 double rms{0};
156 for (int ig = 0; ig < ctx_.gvec().count(); ig++) {
157 rms += std::pow(std::abs(density_.rho().f_pw_local(ig) - rho_pw[ig]), 2);
158 }
159 ctx_.comm().allreduce(&rms, 1);
160 json dict;
161 dict["rss"] = rms;
162 dict["rms"] = std::sqrt(rms / ctx_.gvec().num_gvec());
163 dict["detot"] = total_energy() - etot;
164
165 ctx_.message(1, __function_name__, "RSS: %18.12E\n", dict["rss"].get<double>());
166 ctx_.message(1, __function_name__, "RMS: %18.12E\n", dict["rms"].get<double>());
167 ctx_.message(1, __function_name__, "dEtot: %18.12E\n", dict["detot"].get<double>());
168 ctx_.message(1, __function_name__, "Eold: %18.12E Enew: %18.12E\n", etot, total_energy());
169
170 return dict;
171}

173json DFT_ground_state::find(double density_tol, double energy_tol, double itsol_tol, int num_dft_iter, bool write_state)
174{
175 PROFILE("sirius::DFT_ground_state::scf_loop");
176
177 auto tstart = std::chrono::high_resolution_clock::now();
178
179 double eold{0}, rms{0};
180
181 density_.mixer_init(ctx_.cfg().mixer());
182
183 int num_iter{-1};
184 std::vector<double> rms_hist;
185 std::vector<double> etot_hist;
186
187 Density rho1(ctx_);
188
189 std::stringstream s;
190 s << "density_tol : " << density_tol << std::endl
191 << "energy_tol : " << energy_tol << std::endl
192 << "itsol_tol (initial) : " << itsol_tol << std::endl
193 << "itsol_tol_min : " << ctx_.cfg().settings().itsol_tol_min() << std::endl
194 << "num_dft_iter : " << num_dft_iter;
195 ctx_.message(1, __func__, s);
196
197 for (int iter = 0; iter < num_dft_iter; iter++) {
198 PROFILE("sirius::DFT_ground_state::scf_loop|iteration");
199 std::stringstream s;
200 s << std::endl;
201 s << "+------------------------------+" << std::endl
202 << "| SCF iteration " << std::setw(3) << iter << " out of " << std::setw(3) << num_dft_iter << '|' << std::endl
203 << "+------------------------------+" << std::endl;
204 ctx_.message(2, __func__, s);
205
206 if (ctx_.cfg().parameters().precision_wf() == "fp32") {
207#if defined(USE_FP32)
208 Hamiltonian0<float> H0(potential_);
209 /* find new wave-functions */
210 if (ctx_.cfg().parameters().precision_hs() == "fp32") {
211 Band(ctx_).solve<float, float>(kset_, H0, true, itsol_tol);
212 } else {
213 Band(ctx_).solve<float, double>(kset_, H0, true, itsol_tol);
214 }
215 /* find band occupancies */
216 kset_.find_band_occupancies<float>();
217 /* generate new density from the occupied wave-functions */
218 density_.generate<float>(kset_, ctx_.use_symmetry(), true, true);
219#else
220 RTE_THROW("not compiled with FP32 support");
221#endif
222 } else {
223 Hamiltonian0<double> H0(potential_);
224 /* find new wave-functions */
225 Band(ctx_).solve<double, double>(kset_, H0, true, itsol_tol);
226 /* find band occupancies */
227 kset_.find_band_occupancies<double>();
228 /* generate new density from the occupied wave-functions */
229 density_.generate<double>(kset_, ctx_.use_symmetry(), true, true);
230 }
231
232 double e1 = energy_potential(density_, potential_);
233 copy(density_, rho1);
234
235 /* mix density */
236 rms = density_.mix();
237
238 double eha_res = density_residual_hartree_energy(density_, rho1);
239
240 /* estimate new tolerance of the iterative solver */
241 double tol = rms;
242 if (ctx_.cfg().mixer().use_hartree()) {
243 //tol = rms * rms / std::max(1.0, unit_cell_.num_electrons());
244 tol = eha_res / std::max(1.0, unit_cell_.num_electrons());
245 }
246 tol = std::min(ctx_.cfg().settings().itsol_tol_scale()[0] * tol,
247 ctx_.cfg().settings().itsol_tol_scale()[1] * itsol_tol);
248 /* tolerance can't be too small */
249 itsol_tol = std::max(ctx_.cfg().settings().itsol_tol_min(), tol);
250
251#if defined(USE_FP32)
252 /* if the final precision is not equal to the current precision */
253 if (ctx_.cfg().parameters().precision_gs() == "fp64" && ctx_.cfg().parameters().precision_wf() == "fp32") {
254 /* if we reached the mimimum tolerance for fp32 */
255 if ((ctx_.cfg().settings().fp32_to_fp64_rms() == 0 && itsol_tol <= ctx_.cfg().settings().itsol_tol_min()) ||
256 (rms < ctx_.cfg().settings().fp32_to_fp64_rms())) {
257 std::cout << "switching to FP64" << std::endl;
258 ctx_.cfg().unlock();
259 ctx_.cfg().settings().itsol_tol_min(std::numeric_limits<double>::epsilon() * 10);
260 ctx_.cfg().parameters().precision_wf("fp64");
261 ctx_.cfg().parameters().precision_hs("fp64");
262 ctx_.cfg().lock();
263
264 for (int ikloc = 0; ikloc < kset_.spl_num_kpoints().local_size(); ikloc++) {
265 int ik = kset_.spl_num_kpoints(ikloc);
266 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
267 kset_.get<double>(ik)->spinor_wave_functions().copy_from(device_t::CPU, ctx_.num_bands(),
268 kset_.get<float>(ik)->spinor_wave_functions(), ispn, 0, ispn, 0);
269 }
270 }
271 for (int ik = 0; ik < kset_.num_kpoints(); ik++) {
272 for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
273 for (int j = 0; j < ctx_.num_bands(); j++) {
274 kset_.get<double>(ik)->band_energy(j, ispn, kset_.get<float>(ik)->band_energy(j, ispn));
275 kset_.get<double>(ik)->band_occupancy(j, ispn, kset_.get<float>(ik)->band_occupancy(j, ispn));
276 }
277 }
278 }
279 }
280 }
281#endif
282 if (ctx_.cfg().control().verification() >= 1) {
283 /* check number of electrons */
284 density_.check_num_electrons();
285 }
286
287 /* compute new potential */
288 potential_.generate(density_, ctx_.use_symmetry(), true);
289
290 if (!ctx_.full_potential() && ctx_.cfg().control().verification() >= 2) {
291 ctx_.message(1, __function_name__, "%s", "checking functional derivative of Exc\n");
292 sirius::check_xc_potential(density_);
293 }
294
295 if (ctx_.cfg().parameters().use_scf_correction()) {
296 double e2 = energy_potential(rho1, potential_);
297 this->scf_energy_ = e2 - e1;
298 }
299
300 /* compute new total energy for a new density */
301 double etot = total_energy();
302
303 etot_hist.push_back(etot);
304
305 rms_hist.push_back(rms);
306
307 /* write some information */
308 std::stringstream out;
309 out << std::endl;
310 print_info(out);
311 out << std::endl;
312 out << "iteration : " << iter << ", RMS : " << std::setprecision(12) << std::scientific << rms
313 << ", energy difference : " << std::setprecision(12) << std::scientific << etot - eold << std::endl
314 << "residual density Hartree energy : " << eha_res;
315 ctx_.message(2, __func__, out);
316 /* check if the calculation has converged */
317 bool converged{true};
318 converged = converged && (std::abs(eold - etot) < energy_tol);
319 if (ctx_.cfg().mixer().use_hartree()) {
320 converged = converged && (eha_res < density_tol);
321 } else {
322 converged = converged && (rms < density_tol);
323 }
324 if (converged) {
325 std::stringstream out;
326 out << std::endl;
327 out << "converged after " << iter + 1 << " SCF iterations!";
328 ctx_.message(1, __func__, out);
329 density_.check_num_electrons();
330 num_iter = iter;
331 break;
332 }
333
334 eold = etot;
335 }
336 std::stringstream out;
337 out << std::endl;
338 print_info(out);
339 ctx_.message(1, __func__, out);
340
341 if (write_state) {
342 ctx_.create_storage_file();
343 if (ctx_.full_potential()) { // TODO: why this is necessary?
344 density_.rho().fft_transform(-1);
345 for (int j = 0; j < ctx_.num_mag_dims(); j++) {
346 density_.magnetization(j).fft_transform(-1);
347 }
348 }
349 potential_.save();
350 density_.save();
351 //kset_.save(storage_file_name);
352 }
353
354 auto tstop = std::chrono::high_resolution_clock::now();
355
356 json dict = serialize();
357 dict["scf_time"] = std::chrono::duration_cast<std::chrono::duration<double>>(tstop - tstart).count();
358 dict["etot_history"] = etot_hist;
359 if (num_iter >= 0) {
360 dict["converged"] = true;
361 dict["num_scf_iterations"] = num_iter;
362 dict["rms_history"] = rms_hist;
363 } else {
364 dict["converged"] = false;
365 }
366
367 //if (ctx_.control().verification_ >= 1) {
368 // check_scf_density();
369 //}
370
371 // dict["volume"] = ctx.unit_cell().omega() * std::pow(bohr_radius, 3);
372 // dict["volume_units"] = "angstrom^3";
373 // dict["energy"] = dft.total_energy() * ha2ev;
374 // dict["energy_units"] = "eV";
375
376 return dict;
377}

379void DFT_ground_state::print_info(std::ostream& out__) const
380{
381 double evalsum1 = kset_.valence_eval_sum();
382 double evalsum2 = core_eval_sum(ctx_.unit_cell());
383 double s_sum = kset_.entropy_sum();
384 double ekin = energy_kin(ctx_, kset_, density_, potential_);
385 double evxc = energy_vxc(density_, potential_);
386 double eexc = energy_exc(density_, potential_);
387 double ebxc = energy_bxc(density_, potential_);
388 double evha = energy_vha(potential_);
389 double etot = total_energy();
390 double gap = kset_.band_gap() * ha2ev;
391 double ef = kset_.energy_fermi();
392 double enuc = energy_enuc(ctx_, potential_);
393
394 double one_elec_en = evalsum1 - (evxc + evha);
395
396 if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {
397 one_elec_en -= potential_.PAW_one_elec_energy(density_);
398 }
399
400 auto result = density_.rho().integrate();
401
402 auto total_charge = std::get<0>(result);
403 auto it_charge = std::get<1>(result);
404 auto mt_charge = std::get<2>(result);
405
406 auto result_mag = density_.get_magnetisation();
407 auto total_mag = std::get<0>(result_mag);
408 auto it_mag = std::get<1>(result_mag);
409 auto mt_mag = std::get<2>(result_mag);
410
411 auto draw_bar = [&](int w)
412 {
413 out__ << std::setfill('-') << std::setw(w) << '-' << std::setfill(' ') << std::endl;
414 };
415
416 auto write_vector = [&](vector3d<double> v__)
417 {
418 out__ << "["
419 << std::setw(9) << std::setprecision(5) << std::fixed << v__[0] << ", "
420 << std::setw(9) << std::setprecision(5) << std::fixed << v__[1] << ", "
421 << std::setw(9) << std::setprecision(5) << std::fixed << v__[2] << "]";
422 };
423
424 out__ << "Charges and magnetic moments" << std::endl;
425 draw_bar(80);
426 if (ctx_.full_potential()) {
427 double total_core_leakage{0.0};
428 out__ << "atom charge core leakage";
429 if (ctx_.num_mag_dims()) {
430 out__ << " moment |moment|";
431 }
432 out__ << std::endl;
433 draw_bar(80);
434
435 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
436 double core_leakage = unit_cell_.atom(ia).symmetry_class().core_leakage();
437 total_core_leakage += core_leakage;
438 out__ << std::setw(4) << ia
439 << std::setw(12) << std::setprecision(6) << std::fixed << mt_charge[ia]
440 << std::setw(16) << std::setprecision(6) << std::scientific << core_leakage;
441 if (ctx_.num_mag_dims()) {
442 vector3d<double> v(mt_mag[ia]);
443 out__ << " ";
444 write_vector(v);
445 out__ << std::setw(12) << std::setprecision(6) << std::fixed << v.length();
446 }
447 out__ << std::endl;
448 }
449 out__ << std::endl;
450 out__ << "total core leakage : " << std::setprecision(8) << std::scientific << total_core_leakage << std::endl
451 << "interstitial charge : " << std::setprecision(6) << std::fixed << it_charge << std::endl;
452 if (ctx_.num_mag_dims()) {
453 vector3d<double> v(it_mag);
454 out__ << "interstitial moment : ";
455 write_vector(v);
456 out__ << ", magnitude : " << std::setprecision(6) << std::fixed << v.length() << std::endl;
457 }
458 } else {
459 if (ctx_.num_mag_dims()) {
460 out__ << "atom moment |moment|" << std::endl;
461 draw_bar(80);
462
463 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
464 vector3d<double> v(mt_mag[ia]);
465 out__ << std::setw(4) << ia << " ";
466 write_vector(v);
467 out__ << std::setw(12) << std::setprecision(6) << std::fixed << v.length() << std::endl;
468 }
469 out__ << std::endl;
470 }
471 }
472 out__ << "total charge : " << std::setprecision(6) << std::fixed << total_charge << std::endl;
473
474 if (ctx_.num_mag_dims()) {
475 vector3d<double> v(total_mag);
476 out__ << "total moment : ";
477 write_vector(v);
478 out__ << ", magnitude : " << std::setprecision(6) << std::fixed << v.length() << std::endl;
479 }
480
481 out__ << std::endl;
482 out__ << "Energy" << std::endl;
483 draw_bar(80);
484
485 auto write_energy = [&](std::string label__, double value__)
486 {
487 out__ << std::left << std::setw(30) << label__ << " : "
488 << std::right << std::setw(16) << std::setprecision(8) << std::fixed << value__ << std::endl;
489 };
490
491 auto write_energy2 = [&](std::string label__, double value__)
492 {
493 out__ << std::left << std::setw(30) << label__ << " : " << std::right
494 << std::setw(16) << std::setprecision(8) << std::fixed << value__ << " (Ha), "
495 << std::setw(16) << std::setprecision(8) << std::fixed << value__ * 2 << " (Ry)" << std::endl;
496 };
497
498 write_energy("valence_eval_sum", evalsum1);
499 if (ctx_.full_potential()) {
500 write_energy("core_eval_sum", evalsum2);
501 write_energy("kinetic energy", ekin);
502 write_energy("enuc", enuc);
503 }
504 write_energy("<rho|V^{XC}>", evxc);
505 write_energy("<rho|E^{XC}>", eexc);
506 write_energy("<mag|B^{XC}>", ebxc);
507 write_energy("<rho|V^{H}>", evha);
508 if (!ctx_.full_potential()) {
509 write_energy2("one-electron contribution", one_elec_en); // eband + deband in QE
510 write_energy("hartree contribution", 0.5 * evha);
511 write_energy("xc contribution", eexc);
512 write_energy("ewald contribution", ewald_energy_);
513 write_energy("PAW contribution", potential_.PAW_total_energy());
514 }
515 write_energy("smearing (-TS)", s_sum);
516 write_energy("SCF correction", this->scf_energy_);
517 if (ctx_.hubbard_correction()) {
518 auto e = ::sirius::energy(density_.occupation_matrix());
519 write_energy2("Hubbard energy", e);
520 }
521 write_energy2("Total energy", etot);
522 out__ << std::endl;
523 write_energy("band gap (eV)", gap);
524 write_energy("Efermi", ef);
525}
