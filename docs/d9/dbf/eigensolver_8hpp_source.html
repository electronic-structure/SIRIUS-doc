58 /// Get type of an eigen solver by name (provided as a string).
59 inline ev_solver_t get_ev_solver_t(std::string name__)
60 {
61  std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
62 
63  static const std::map<std::string, ev_solver_t> map_to_type = {
64  {"lapack", ev_solver_t::lapack}, {"scalapack", ev_solver_t::scalapack}, {"elpa1", ev_solver_t::elpa},
65  {"elpa2", ev_solver_t::elpa}, {"magma", ev_solver_t::magma}, {"magma_gpu", ev_solver_t::magma_gpu},
66  {"plasma", ev_solver_t::plasma}, {"cusolver", ev_solver_t::cusolver}};
67 
68  if (map_to_type.count(name__) == 0) {
69  std::stringstream s;
70  s << "wrong label of eigen-solver : " << name__;
71  TERMINATE(s);
72  }
73 
74  return map_to_type.at(name__);
75 }
