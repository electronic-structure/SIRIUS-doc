33/// Compute gradient of beta-projectors over atomic positions \f$ d \langle {\bf G+k} | \beta \rangle / d \tau_{\alpha} \f$.
34template <typename T>
35class Beta_projectors_gradient: public Beta_projectors_base<T>
36{
37 private:
38 void generate_pw_coefs_t(Beta_projectors<T>& beta__, std::vector<int> const& igk__)
39 {
40 if (!this->num_beta_t()) {
41 return;
42 }
43
44 for (int x = 0; x < 3; x++) {
45 #pragma omp parallel for
46 for (int i = 0; i < this->num_beta_t(); i++) {
47 for (int igkloc = 0; igkloc < this->num_gkvec_loc(); igkloc++) {
48 int igk = igk__[igkloc];
49 auto vgc = this->gkvec_.template gkvec_cart<index_domain_t::global>(igk);
50 this->pw_coeffs_t_(igkloc, i, x) = std::complex<T>(0, -vgc[x]) * beta__.pw_coeffs_t(igkloc, i, 0);
51 }
52 }
53 }
54 }
55
56 public:
57 Beta_projectors_gradient(Simulation_context& ctx__,
58 Gvec const& gkvec__,
59 std::vector<int> const& igk__,
60 Beta_projectors<T>& beta__)
61 : Beta_projectors_base<T>(ctx__, gkvec__, igk__, 3)
62 {
63 generate_pw_coefs_t(beta__, igk__);
64 }
65};
