246  /// Find G+k vectors within the cutoff.
247  inline void generate_gkvec(double gk_cutoff__)
248  {
249  PROFILE("sirius::K_point::generate_gkvec");
250 
251  if (ctx_.full_potential() && (gk_cutoff__ * unit_cell_.max_mt_radius() > ctx_.lmax_apw()) &&
252  comm_.rank() == 0 && ctx_.control().verbosity_ >= 0) {
253  std::stringstream s;
254  s << "G+k cutoff (" << gk_cutoff__ << ") is too large for a given lmax ("
255  << ctx_.lmax_apw() << ") and a maximum MT radius (" << unit_cell_.max_mt_radius() << ")" << std::endl
256  << "suggested minimum value for lmax : " << int(gk_cutoff__ * unit_cell_.max_mt_radius()) + 1;
257  WARNING(s);
258  }
259 
260  if (gk_cutoff__ * 2 > ctx_.pw_cutoff()) {
261  std::stringstream s;
262  s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
263  << " pw cutoff : " << ctx_.pw_cutoff() << std::endl
264  << " doubled G+k cutoff : " << gk_cutoff__ * 2;
265  TERMINATE(s);
266  }
267 
268  /* create G+k vectors; communicator of the coarse FFT grid is used because wave-functions will be transformed
269  * only on the coarse grid; G+k-vectors will be distributed between MPI ranks assigned to the k-point */
270  gkvec_ = std::unique_ptr<Gvec>(new Gvec(vk_, ctx_.unit_cell().reciprocal_lattice_vectors(), gk_cutoff__, comm(),
271  ctx_.gamma_point()));
272 
273  gkvec_partition_ = std::unique_ptr<Gvec_partition>(new Gvec_partition(*gkvec_, ctx_.comm_fft_coarse(),
274  ctx_.comm_band_ortho_fft_coarse()));
275 
276  gkvec_offset_ = gkvec().gvec_offset(comm().rank());
277  }

317  /// Get the number of occupied bands for each spin channel.
318  int num_occupied_bands(int ispn__ = -1)
319  {
320  for (int j = ctx_.num_bands() - 1; j >= 0; j--) {
321  if (std::abs(band_occupancy(j, ispn__) * weight()) > 1e-14) {
322  return j + 1;
323  }
324  }
325  return 0;
326  }

334  /// Total number of muffin-tin and plane-wave expansion coefficients for the wave-functions.
335  /** APW+lo basis \f$ \varphi_{\mu {\bf k}}({\bf r}) = \{ \varphi_{\bf G+k}({\bf r}),
336  * \varphi_{j{\bf k}}({\bf r}) \} \f$ is used to expand first-variational wave-functions:
337  *
338  * \f[
339  * \psi_{i{\bf k}}({\bf r}) = \sum_{\mu} c_{\mu i}^{\bf k} \varphi_{\mu \bf k}({\bf r}) =
340  * \sum_{{\bf G}}c_{{\bf G} i}^{\bf k} \varphi_{\bf G+k}({\bf r}) +
341  * \sum_{j}c_{j i}^{\bf k}\varphi_{j{\bf k}}({\bf r})
342  * \f]
343  *
344  * Inside muffin-tins the expansion is converted into the following form:
345  * \f[
346  * \psi_{i {\bf k}}({\bf r})= \begin{array}{ll}
347  * \displaystyle \sum_{L} \sum_{\lambda=1}^{N_{\ell}^{\alpha}}
348  * F_{L \lambda}^{i {\bf k},\alpha}f_{\ell \lambda}^{\alpha}(r)
349  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in MT_{\alpha} \end{array}
350  * \f]
351  *
352  * Thus, the total number of coefficients representing a wave-funstion is equal
353  * to the number of muffin-tin basis functions of the form \f$ f_{\ell \lambda}^{\alpha}(r)
354  * Y_{\ell m}(\hat {\bf r}) \f$ plust the number of G+k plane waves. */

357  // if (ctx_.full_potential()) {
358  // return unit_cell_.mt_basis_size() + num_gkvec();
359  // } else {

701 /** \page basis Basis functions for Kohn-Sham wave-functions expansion
702  *
703  * \section basis1 LAPW+lo basis
704  *
705  * LAPW+lo basis consists of two different sets of functions: LAPW functions \f$ \varphi_{{\bf G+k}} \f$ defined over
706  * entire unit cell:
707  * \f[
708  * \varphi_{{\bf G+k}}({\bf r}) = \left\{ \begin{array}{ll}
709  * \displaystyle \sum_{L} \sum_{\nu=1}^{O_{\ell}^{\alpha}} a_{L\nu}^{\alpha}({\bf G+k})u_{\ell \nu}^{\alpha}(r)
710  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in {\rm MT} \alpha \\
711  * \displaystyle \frac{1}{\sqrt \Omega} e^{i({\bf G+k}){\bf r}} & {\bf r} \in {\rm I} \end{array} \right.
712  * \f]
713  * and Bloch sums of local orbitals defined inside muffin-tin spheres only:
714  * \f[
715  * \begin{array}{ll} \displaystyle \varphi_{j{\bf k}}({\bf r})=\sum_{{\bf T}} e^{i{\bf kT}}
716  * \varphi_{j}({\bf r - T}) & {\rm {\bf r} \in MT} \end{array}
717  * \f]
718  * Each local orbital is composed of radial and angular parts:
719  * \f[
720  * \varphi_{j}({\bf r}) = \phi_{\ell_j}^{\zeta_j,\alpha_j}(r) Y_{\ell_j m_j}(\hat {\bf r})
721  * \f]
722  * Radial part of local orbital is defined as a linear combination of radial functions (minimum two radial functions
723  * are required) such that local orbital vanishes at the sphere boundary:
724  * \f[
725  * \phi_{\ell}^{\zeta, \alpha}(r) = \sum_{p}\gamma_{p}^{\zeta,\alpha} u_{\ell \nu_p}^{\alpha}(r)
726  * \f]
727  *
728  * Arbitrary number of local orbitals can be introduced for each angular quantum number (this is highlighted by
729  * the index \f$ \zeta \f$).
730  *
731  * Radial functions are m-th order (with zero-order being a function itself) energy derivatives of the radial
732  * Schrdinger equation:
733  * \f[
734  * u_{\ell \nu}^{\alpha}(r) = \frac{\partial^{m_{\nu}}}{\partial^{m_{\nu}}E}u_{\ell}^{\alpha}(r,E)\Big|_{E=E_{\nu}}
735  * \f]
736  */
737
