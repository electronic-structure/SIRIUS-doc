78 void* create_batch_plan(int rank, int* dims, int* embed, int stride, int dist, int nfft,
79  bool auto_alloc)
80 {
81  // TODO: check how allocation could be implemented
82  if (auto_alloc) throw std::runtime_error("Auto allocation for rocfft not implemented!");
83 
84  // check input
85  for (size_t i = 0; i < (size_t)rank; i++) {
86  if (dims[i] > embed[i])
87  throw std::runtime_error("Illegal dims or embed parameters for ROCFFT plan creation!");
88  }
89 
90  rocfft_plan_description desc = nullptr;
91 
92  // ROCFFT appears to expect dimension to be ordered in reverse (see hipFFT implementation)
93  size_t lengths[3] = {1, 1, 1};
94  for (size_t i = 0; i < (size_t)rank; i++) lengths[i] = dims[rank - 1 - i];
95 
96  if (embed != nullptr) {
97  rocfft_plan_description_create(&desc);
98 
99  size_t strides[3] = {(size_t)stride, 1, 1};
100 
101  size_t nembed_lengths[3] = {1, 1, 1};
102  for (size_t i = 0; i < (size_t)rank; i++) nembed_lengths[i] = embed[rank - 1 - i];
103 
104  for (size_t i = 1; i < (size_t)rank; i++) strides[i] = nembed_lengths[i - 1] * strides[i - 1];
105 
106  CALL_ROCFFT(
107  rocfft_plan_description_set_data_layout,
108  (desc, rocfft_array_type_complex_interleaved, rocfft_array_type_complex_interleaved, 0,
109  0, rank, strides, dist, rank, strides, dist));
110  }
111 
112  rocfft_handler* handler = new rocfft_handler();
113 
114  // create plans
115  CALL_ROCFFT(rocfft_execution_info_create, (&handler->info));
116 
117  CALL_ROCFFT(rocfft_plan_create, (&handler->plan_forward, rocfft_placement_inplace,
118  rocfft_transform_type_complex_forward, rocfft_precision_double,
119  rank, lengths, nfft, desc));
120  CALL_ROCFFT(rocfft_plan_create, (&handler->plan_backward, rocfft_placement_inplace,
121  rocfft_transform_type_complex_inverse, rocfft_precision_double,
122  rank, lengths, nfft, desc));
123 
124  // description no longer needed
125  CALL_ROCFFT(rocfft_plan_description_destroy, (desc));
126 
127  // calculate workbuffer size
128  size_t work_size_forward, work_size_backward;
129  CALL_ROCFFT(rocfft_plan_get_work_buffer_size, (handler->plan_forward, &work_size_forward));
130  CALL_ROCFFT(rocfft_plan_get_work_buffer_size, (handler->plan_backward, &work_size_backward));
131  handler->work_size = std::max(work_size_forward, work_size_backward);
132 
133  return static_cast<void*>(handler);
134 }
