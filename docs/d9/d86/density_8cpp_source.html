118 void Density::initial_density_pseudo()
119 {
120  auto v = ctx_.make_periodic_function<index_domain_t::local>(
121  [&](int iat, double g) { return ctx_.ps_rho_ri().value<int>(iat, g); });
122 
123  if (ctx_.control().print_checksum_) {
124  auto z1 = mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).checksum();
125  ctx_.comm().allreduce(&z1, 1);
126  if (ctx_.comm().rank() == 0) {
127  utils::print_checksum("rho_pw_init", z1);
128  }
129  }
130  std::copy(v.begin(), v.end(), &rho().f_pw_local(0));
131 
132  if (ctx_.control().print_hash_ && ctx_.comm().rank() == 0) {
133  auto h = sddk::mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).hash();
134  utils::print_hash("rho_pw_init", h);
135  }
136 
137  double charge = rho().f_0().real() * unit_cell_.omega();
138 
139  if (std::abs(charge - unit_cell_.num_valence_electrons()) > 1e-6) {
140  std::stringstream s;
141  s << "wrong initial charge density" << std::endl
142  << " integral of the density : " << std::setprecision(12) << charge << std::endl
143  << " target number of electrons : " << std::setprecision(12) << unit_cell_.num_valence_electrons();
144  if (ctx_.comm().rank() == 0) {
145  WARNING(s);
146  }
147  }
148  rho().fft_transform(1);
149  if (ctx_.control().print_hash_ && ctx_.comm().rank() == 0) {
150  auto h = rho().f_rg().hash();
151  utils::print_hash("rho_rg_init", h);
152  }
153 
154  /* remove possible negative noise */
155  for (int ir = 0; ir < ctx_.spfft().local_slice_size(); ir++) {
156  rho().f_rg(ir) = std::max(rho().f_rg(ir), 0.0);
157  }
158  /* renormalize charge */
159  normalize();
160 
161  if (ctx_.control().print_checksum_) {
162  auto cs = rho().checksum_rg();
163  if (ctx_.comm().rank() == 0) {
164  utils::print_checksum("rho_rg_init", cs);
165  }
166  }
167 
168  /* initialize the magnetization */
169  if (ctx_.num_mag_dims()) {
170  double R = ctx_.control().rmt_max_;
171 
172  auto w = [R](double x) {
173  /* the constants are picked in such a way that the volume integral of the
174  weight function is equal to the volume of the atomic sphere;
175  in this case the starting magnetiation in the atomic spehre
176  integrates to the starting magnetization vector */
177 
178  /* volume of the sphere */
179  const double norm = fourpi * std::pow(R, 3) / 3.0;
180  return (35.0 / 8) * std::pow(1 - std::pow(x / R, 2), 2) / norm;
181  // return 10 * std::pow(1 - x / R, 2) / norm;
182  // const double b = 1.1016992073677703;
183  // return b * 1.0 / (std::exp(10 * (a - R)) + 1) / norm;
184  // const double norm = pi * std::pow(R, 3) / 3.0;
185  // return 1.0 / (std::exp(10 * (x - R)) + 1) / norm;
186  };
187 
188  #pragma omp parallel for
189  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
190  auto& atom_to_grid_map = ctx_.atoms_to_grid_idx_map(ia);
191  vector3d<double> v = unit_cell_.atom(ia).vector_field();
192 
193  for (auto coord : atom_to_grid_map) {
194  int ir = coord.first;
195  double a = coord.second;
196  magnetization(0).f_rg(ir) += v[2] * w(a);
197  if (ctx_.num_mag_dims() == 3) {
198  magnetization(1).f_rg(ir) += v[0] * w(a);
199  magnetization(2).f_rg(ir) += v[1] * w(a);
200  }
201  }
202  }
203  }
204  this->fft_transform(-1);
205 
206  if (ctx_.control().print_checksum_) {
207  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
208  auto cs = component(i).checksum_rg();
209  auto cs1 = component(i).checksum_pw();
210  if (ctx_.comm().rank() == 0) {
211  std::stringstream s;
212  s << "component[" << i << "]_rg";
213  utils::print_checksum(s.str(), cs);
214  std::stringstream s1;
215  s1 << "component[" << i << "]_pw";
216  utils::print_checksum(s1.str(), cs1);
217  }
218  }
219  }

222 void Density::initial_density_full_pot()
223 {
224  /* initialize smooth density of free atoms */
225  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
226  unit_cell_.atom_type(iat).init_free_atom_density(true);
227  }
228 
229  /* compute radial integrals */
230  Radial_integrals_rho_free_atom ri(ctx_.unit_cell(), ctx_.pw_cutoff(), 40);
231 
232  /* compute contribution from free atoms to the interstitial density */
233  auto v = ctx_.make_periodic_function<index_domain_t::local>([&ri](int iat, double g) { return ri.value(iat, g); });
234 
235  /* initialize density of free atoms (not smoothed) */
236  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
237  unit_cell_.atom_type(iat).init_free_atom_density(false);
238  }
239 
240  if (ctx_.control().print_checksum_) {
241  auto z = mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).checksum();
242  ctx_.comm().allreduce(&z, 1);
243  if (ctx_.comm().rank() == 0) {
244  utils::print_checksum("rho_pw", z);
245  }
246  }
247 
248  /* set plane-wave coefficients of the charge density */
249  std::copy(v.begin(), v.end(), &rho().f_pw_local(0));
250  /* convert charge density to real space mesh */
251  rho().fft_transform(1);
252 
253  if (ctx_.control().print_checksum_) {
254  auto cs = rho().checksum_rg();
255  if (ctx_.comm().rank() == 0) {
256  utils::print_checksum("rho_rg", cs);
257  }
258  }
259 
260  /* remove possible negative noise */
261  for (int ir = 0; ir < ctx_.spfft().local_slice_size(); ir++) {
262  rho().f_rg(ir) = std::max(0.0, rho().f_rg(ir));
263  }
264 
265  /* set Y00 component of charge density */
266  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
267  int nmtp = ctx_.unit_cell().atom(ia).num_mt_points();
268 
269  for (int ir = 0; ir < nmtp; ir++) {
270  double x = ctx_.unit_cell().atom(ia).radial_grid(ir);
271  rho().f_mt<index_domain_t::global>(0, ir, ia) = unit_cell_.atom(ia).type().free_atom_density(x) / y00;
272  }
273  }
274 
275  int lmax = ctx_.lmax_rho();
276  int lmmax = utils::lmmax(lmax);
277 
278  auto l_by_lm = utils::l_by_lm(lmax);
279 
280  std::vector<double_complex> zil(lmax + 1);
281  for (int l = 0; l <= lmax; l++) {
282  zil[l] = std::pow(double_complex(0, 1), l);
283  }
284 
285  /* compute boundary value at MT sphere from the plane-wave expansion */
286  auto gvec_ylm = ctx_.generate_gvec_ylm(lmax);
287 
288  auto sbessel_mt = ctx_.generate_sbessel_mt(lmax);
289 
290  auto flm = ctx_.sum_fg_fl_yg(lmax, v.data(), sbessel_mt, gvec_ylm);
291 
292  /* this is the difference between the value of periodic charge density at MT boundary and
293  a value of the atom's free density at the boundary */
294  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
295  double R = ctx_.unit_cell().atom(ia).mt_radius();
296  double c = unit_cell_.atom(ia).type().free_atom_density(R) / y00;
297  flm(0, ia) -= c;
298  }
299 
300  /* match density at MT */
301  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
302  mdarray<double, 2> rRl(ctx_.unit_cell().max_num_mt_points(), lmax + 1);
303  double R = ctx_.unit_cell().atom_type(iat).mt_radius();
304  int nmtp = ctx_.unit_cell().atom_type(iat).num_mt_points();
305 
306  #pragma omp parallel for default(shared)
307  for (int l = 0; l <= lmax; l++) {
308  for (int ir = 0; ir < nmtp; ir++) {
309  rRl(ir, l) = std::pow(ctx_.unit_cell().atom_type(iat).radial_grid(ir) / R, 2);
310  }
311  }
312  #pragma omp parallel for default(shared)
313  for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
314  int ia = unit_cell_.atom_type(iat).atom_id(i);
315  std::vector<double> glm(lmmax);
316  SHT::convert(lmax, &flm(0, ia), &glm[0]);
317  for (int lm = 0; lm < lmmax; lm++) {
318  int l = l_by_lm[lm];
319  for (int ir = 0; ir < nmtp; ir++) {
320  rho().f_mt<index_domain_t::global>(lm, ir, ia) += glm[lm] * rRl(ir, l);
321  }
322  }
323  }
324  }
325 
326  /* normalize charge density */
327  normalize();
328 
329  check_num_electrons();
330

332  // for (int i = 0; i <= 10000; i++) {
333  // vector3d<double> v = (i / 10000.0) * vector3d<double>({10.26, 10.26, 10.26});
334  // double val = rho().value(v);
335  // fprintf(fout, "%18.12f %18.12f\n", v.length(), val);

342  // double val = rho().value_rg(v);
343  // fprintf(fout2, "%18.12f %18.12f\n", v.length(), val);
344  //}

347  /* initialize the magnetization */
348  if (ctx_.num_mag_dims()) {
349  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
350  int ia = unit_cell_.spl_num_atoms(ialoc);
351  vector3d<double> v = unit_cell_.atom(ia).vector_field();
352  double len = v.length();
353 
354  int nmtp = unit_cell_.atom(ia).num_mt_points();
355  Spline<double> rho_s(unit_cell_.atom(ia).type().radial_grid());
356  double R = unit_cell_.atom(ia).mt_radius();
357  for (int ir = 0; ir < nmtp; ir++) {
358  double x = unit_cell_.atom(ia).type().radial_grid(ir);
359  rho_s(ir) = this->rho().f_mt<index_domain_t::local>(0, ir, ialoc) * y00 *
360  (1 - 3 * std::pow(x / R, 2) + 2 * std::pow(x / R, 3));
361  }
362 
363  /* maximum magnetization which can be achieved if we smooth density towards MT boundary */
364  double q = fourpi * rho_s.interpolate().integrate(2);
365 
366  /* if very strong initial magnetization is given */
367  if (q < len) {
368  /* renormalize starting magnetization */
369  for (int x : {0, 1, 2}) {
370  v[x] *= (q / len);
371  }
372  len = q;
373  }
374 
375  if (len > 1e-8) {
376  for (int ir = 0; ir < nmtp; ir++) {
377  magnetization(0).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[2] / q / y00;
378  }
379  if (ctx_.num_mag_dims() == 3) {
380  for (int ir = 0; ir < nmtp; ir++) {
381  magnetization(1).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[0] / q / y00;
382  magnetization(2).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[1] / q / y00;
383  }
384  }
385  }
386  }
387  }

390 void Density::init_density_matrix_for_paw()
391 {
392  density_matrix_.zero();
393 
394  for (int ipaw = 0; ipaw < unit_cell_.num_paw_atoms(); ipaw++) {
395  int ia = unit_cell_.paw_atom_index(ipaw);
396 
397  auto& atom = unit_cell_.atom(ia);
398  auto& atom_type = atom.type();
399 
400  int nbf = atom_type.mt_basis_size();
401 
402  auto& occupations = atom_type.paw_wf_occ();
403 
404  /* magnetization vector */
405  auto magn = atom.vector_field();
406 
407  for (int xi = 0; xi < nbf; xi++) {
408  auto& basis_func_index_dsc = atom_type.indexb()[xi];
409 
410  int rad_func_index = basis_func_index_dsc.idxrf;
411 
412  double occ = occupations[rad_func_index];
413 
414  int l = basis_func_index_dsc.l;
415 
416  switch (ctx_.num_mag_dims()) {
417  case 0: {
418  density_matrix_(xi, xi, 0, ia) = occ / double(2 * l + 1);
419  break;
420  }
421 
422  case 3:
423  case 1: {
424  double nm = (std::abs(magn[2]) < 1.0) ? magn[2] : std::copysign(1, magn[2]);
425  density_matrix_(xi, xi, 0, ia) = 0.5 * (1.0 + nm) * occ / double(2 * l + 1);
426  density_matrix_(xi, xi, 1, ia) = 0.5 * (1.0 - nm) * occ / double(2 * l + 1);
427  break;
428  }
429  }
430  }

434 void Density::generate_paw_atom_density(int idx__)
435 {
436  int ia_paw = ctx_.unit_cell().spl_num_paw_atoms(idx__);
437  int ia = ctx_.unit_cell().paw_atom_index(ia_paw);
438 
439  auto& atom_type = ctx_.unit_cell().atom(ia).type();
440 
441  auto l_by_lm = utils::l_by_lm(2 * atom_type.indexr().lmax_lo());
442 
443  /* get gaunt coefficients */
444  Gaunt_coefficients<double> GC(atom_type.indexr().lmax_lo(), 2 * atom_type.indexr().lmax_lo(),
445  atom_type.indexr().lmax_lo(), SHT::gaunt_rrr);
446 
447  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
448  paw_density_.ae_density(i, idx__).zero();
449  paw_density_.ps_density(i, idx__).zero();
450  }
451 
452  /* get radial grid to divide density over r^2 */
453  auto& grid = atom_type.radial_grid();
454 
455  auto& paw_ae_wfs = atom_type.ae_paw_wfs_array();
456  auto& paw_ps_wfs = atom_type.ps_paw_wfs_array();
457 
458  /* iterate over local basis functions (or over lm1 and lm2) */
459  for (int xi2 = 0; xi2 < atom_type.indexb().size(); xi2++) {
460  int lm2 = atom_type.indexb(xi2).lm;
461  int irb2 = atom_type.indexb(xi2).idxrf;
462 
463  for (int xi1 = 0; xi1 <= xi2; xi1++) {
464  int lm1 = atom_type.indexb(xi1).lm;
465  int irb1 = atom_type.indexb(xi1).idxrf;
466 
467  /* get num of non-zero GC */
468  int num_non_zero_gc = GC.num_gaunt(lm1, lm2);
469 
470  double diag_coef = (xi1 == xi2) ? 1.0 : 2.0;
471 
472  /* store density matrix in aux form */
473  double dm[4] = {0, 0, 0, 0};
474  switch (ctx_.num_mag_dims()) {
475  case 3: {
476  dm[2] = 2 * std::real(density_matrix_(xi1, xi2, 2, ia));
477  dm[3] = -2 * std::imag(density_matrix_(xi1, xi2, 2, ia));
478  }
479  case 1: {
480  dm[0] = std::real(density_matrix_(xi1, xi2, 0, ia) + density_matrix_(xi1, xi2, 1, ia));
481  dm[1] = std::real(density_matrix_(xi1, xi2, 0, ia) - density_matrix_(xi1, xi2, 1, ia));
482  break;
483  }
484  case 0: {
485  dm[0] = std::real(density_matrix_(xi1, xi2, 0, ia));
486  }
487  }
488 
489  for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
490  auto& ae_dens = paw_density_.ae_density(imagn, idx__);
491  auto& ps_dens = paw_density_.ps_density(imagn, idx__);
492 
493  /* add nonzero coefficients */
494  for (int inz = 0; inz < num_non_zero_gc; inz++) {
495  auto& lm3coef = GC.gaunt(lm1, lm2, inz);
496 
497  /* iterate over radial points */
498  for (int irad = 0; irad < grid.num_points(); irad++) {
499 
500  /* we need to divide density over r^2 since wave functions are stored multiplied by r */
501  double inv_r2 = diag_coef / (grid[irad] * grid[irad]);
502 
503  /* calculate unified density/magnetization
504  * dm_ij * GauntCoef * ( phi_i phi_j + Q_ij) */
505  ae_dens(lm3coef.lm3, irad) +=
506  dm[imagn] * inv_r2 * lm3coef.coef * paw_ae_wfs(irad, irb1) * paw_ae_wfs(irad, irb2);
507  ps_dens(lm3coef.lm3, irad) +=
508  dm[imagn] * inv_r2 * lm3coef.coef *
509  (paw_ps_wfs(irad, irb1) * paw_ps_wfs(irad, irb2) +
510  atom_type.q_radial_function(irb1, irb2, l_by_lm[lm3coef.lm3])(irad));
511  }
512  }
513  }
514  }
515  }

535

537

540 
541  /* get preallocated memory */

552  /* non-magnetic or collinear case */
553  if (ctx_.num_mag_dims() != 3) {
554  /* loop over pure spinor components */
555  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
556  /* trivial case */
557  if (!kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col().global_index_size()) {
558  continue;
559  }
560  int ncols =kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col().local_size();
561  for (int ii = 0; ii < ncols; ii++) {
562  /* global index of the band */
563  int i = ncols-1-ii;
564  int j = kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col()[i];
565  double w = kp__->band_occupancy(j, ispn) * kp__->weight() / omega;
566 
567  auto inp_wf = kp__->spinor_wave_functions().pw_coeffs(ispn).extra().at(memory_t::host, 0, i);
568 
569  /* transform to real space */
570  kp__->spfft_transform().backward(reinterpret_cast<const double*>(inp_wf),
571  kp__->spfft_transform().processing_unit());
572 
573  switch (kp__->spfft_transform().processing_unit()) {
574  case SPFFT_PU_HOST: {
575  if (ctx_.gamma_point()) {
576  #pragma omp parallel for schedule(static)
577  for (int ir = 0; ir < nr; ir++) {
578  density_rg(ir, ispn) += w * std::pow(data_ptr[ir], 2);
579  }
580  } else {
581  auto data = reinterpret_cast<double_complex*>(data_ptr);
582  #pragma omp parallel for schedule(static)
583  for (int ir = 0; ir < nr; ir++) {
584  auto z = data[ir];
585  density_rg(ir, ispn) += w * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
586  }
587  }
588  break;
589  }
590  case SPFFT_PU_GPU: {
591 #if defined(__GPU)
592  if (ctx_.gamma_point()) {
593  update_density_rg_1_real_gpu(nr, data_ptr, w, density_rg.at(memory_t::device, 0, ispn));
594  } else {
595  auto data = reinterpret_cast<double_complex*>(data_ptr);
596  update_density_rg_1_complex_gpu(nr, data, w, density_rg.at(memory_t::device, 0, ispn));
597  }
598 #endif
599  break;
600  }
601  }
602  }
603  }
604  } else { /* non-collinear case */
605  assert(kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col().local_size() ==
606  kp__->spinor_wave_functions().pw_coeffs(1).spl_num_col().local_size());
607 
608  /* allocate on CPU or GPU */
609  mdarray<double_complex, 1> psi_r_up(nr, ctx_.mem_pool(memory_t::host));
610  if (fft.processing_unit() == SPFFT_PU_GPU) {
611  psi_r_up.allocate(ctx_.mem_pool(memory_t::device));
612  }
613  for (int i = 0; i < kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col().local_size(); i++) {
614  int j = kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col()[i];
615  double w = kp__->band_occupancy(j, 0) * kp__->weight() / omega;
616 
617  /* transform up- component of spinor function to real space; in case of GPU wave-function stays in GPU
618  * memory */
619  auto inp_wf_up = kp__->spinor_wave_functions().pw_coeffs(0).extra().at(memory_t::host, 0, i);
620  /* transform to real space */
621  kp__->spfft_transform().backward(reinterpret_cast<const double*>(inp_wf_up),
622  kp__->spfft_transform().processing_unit());
623 
624  /* this is a non-collinear case, so the wave-functions and FFT buffer are complex and
625  we can copy memory */
626  switch (kp__->spfft_transform().processing_unit()) {
627  case SPFFT_PU_HOST: {
628  auto inp = reinterpret_cast<double_complex*>(data_ptr);
629  std::copy(inp, inp + nr, psi_r_up.at(memory_t::host));
630  break;
631  }
632  case SPFFT_PU_GPU: {
633  acc::copy(psi_r_up.at(memory_t::device), reinterpret_cast<double_complex*>(data_ptr), nr);
634  break;
635  }
636  }
637 
638  /* transform dn- component of spinor wave function */
639  auto inp_wf_dn = kp__->spinor_wave_functions().pw_coeffs(1).extra().at(memory_t::host, 0, i);
640  kp__->spfft_transform().backward(reinterpret_cast<const double*>(inp_wf_dn),
641  kp__->spfft_transform().processing_unit());
642 
643  auto psi_r_dn = reinterpret_cast<double_complex*>(data_ptr);
644 
645  switch (fft.processing_unit()) {
646  case SPFFT_PU_HOST: {
647  #pragma omp parallel for schedule(static)
648  for (int ir = 0; ir < nr; ir++) {
649  auto r0 = (std::pow(psi_r_up[ir].real(), 2) + std::pow(psi_r_up[ir].imag(), 2)) * w;
650  auto r1 = (std::pow(psi_r_dn[ir].real(), 2) + std::pow(psi_r_dn[ir].imag(), 2)) * w;
651 
652  auto z2 = psi_r_up[ir] * std::conj(psi_r_dn[ir]) * w;
653 
654  density_rg(ir, 0) += r0;
655  density_rg(ir, 1) += r1;
656  density_rg(ir, 2) += 2.0 * std::real(z2);
657  density_rg(ir, 3) -= 2.0 * std::imag(z2);
658  }
659  break;
660  }
661  case SPFFT_PU_GPU: {
662 #ifdef __GPU
663  /* add up-up contribution */
664  update_density_rg_1_complex_gpu(nr, psi_r_up.at(memory_t::device), w,
665  density_rg.at(memory_t::device, 0, 0));
666  /* add dn-dn contribution */
667  update_density_rg_1_complex_gpu(nr, psi_r_dn, w, density_rg.at(memory_t::device, 0, 1));
668  /* add off-diagonal contribution */
669  update_density_rg_2_gpu(nr, psi_r_up.at(memory_t::device), psi_r_dn, w,
670  density_rg.at(memory_t::device, 0, 2),
671  density_rg.at(memory_t::device, 0, 3));
672 #endif
673  break;
674  }
675  }
676  }
677  }
678 
679  if (fft.processing_unit() == SPFFT_PU_GPU) {
680  density_rg.copy_to(memory_t::host);
681  }
682 
683  /* switch from real density matrix to density and magnetization */
684  switch (ctx_.num_mag_dims()) {
685  case 3: {
686  #pragma omp parallel for schedule(static)
687  for (int ir = 0; ir < fft.local_slice_size(); ir++) {
688  rho_mag_coarse_[2]->f_rg(ir) += density_rg(ir, 2); // Mx
689  rho_mag_coarse_[3]->f_rg(ir) += density_rg(ir, 3); // My
690  }
691  }
692  case 1: {
693  #pragma omp parallel for schedule(static)
694  for (int ir = 0; ir < fft.local_slice_size(); ir++) {
695  rho_mag_coarse_[0]->f_rg(ir) += (density_rg(ir, 0) + density_rg(ir, 1)); // rho
696  rho_mag_coarse_[1]->f_rg(ir) += (density_rg(ir, 0) - density_rg(ir, 1)); // Mz
697  }
698  break;
699  }
700  case 0: {
701  #pragma omp parallel for schedule(static)
702  for (int ir = 0; ir < fft.local_slice_size(); ir++) {
703  rho_mag_coarse_[0]->f_rg(ir) += density_rg(ir, 0); // rho
704  }
705  }
706  }

959  &bp1(offs, 0, 0), bp1.ld(), &bp2(offs, 0, 1), bp2.ld(),
960  &linalg_const<double_complex>::one(), &density_matrix__(0, 0, 2, ja),
961  density_matrix__.ld());
962  }
963  }
964  }
965  }
966  kp__->beta_projectors().dismiss();
967  }
