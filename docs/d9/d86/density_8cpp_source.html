152{
153 double sum = 0.0;
154 for (int ic = 0; ic < unit_cell_.num_atom_symmetry_classes(); ic++) {

185void Density::initial_density_pseudo()
186{
187 /* get lenghts of all G shells */
188 auto q = ctx_.gvec().shells_len();
189 /* get form-factors for all G shells */
190 // TODO: MPI parallelise over G-shells
191 auto ff = ctx_.ps_rho_ri().values(q, ctx_.comm());
192 /* make Vloc(G) */
193 auto v = ctx_.make_periodic_function<index_domain_t::local>(ff);
194
195 if (ctx_.cfg().control().print_checksum()) {
196 auto z1 = mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).checksum();
197 ctx_.comm().allreduce(&z1, 1);
198 if (ctx_.comm().rank() == 0) {
199 utils::print_checksum("rho_pw_init", z1);
200 }
201 }
202 std::copy(v.begin(), v.end(), &rho().f_pw_local(0));
203
204 if (ctx_.cfg().control().print_hash() && ctx_.comm().rank() == 0) {
205 auto h = sddk::mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).hash();
206 utils::print_hash("rho_pw_init", h);
207 }
208
209 double charge = rho().f_0().real() * unit_cell_.omega();
210
211 if (std::abs(charge - unit_cell_.num_valence_electrons()) > 1e-6) {
212 std::stringstream s;
213 s << "wrong initial charge density" << std::endl
214 << " integral of the density : " << std::setprecision(12) << charge << std::endl
215 << " target number of electrons : " << std::setprecision(12) << unit_cell_.num_valence_electrons();
216 if (ctx_.comm().rank() == 0) {
217 WARNING(s);
218 }
219 }
220 rho().fft_transform(1);
221 if (ctx_.cfg().control().print_hash() && ctx_.comm().rank() == 0) {
222 auto h = rho().f_rg().hash();
223 utils::print_hash("rho_rg_init", h);
224 }
225
226 /* remove possible negative noise */
227 for (int ir = 0; ir < ctx_.spfft<double>().local_slice_size(); ir++) {
228 rho().f_rg(ir) = std::max(rho().f_rg(ir), 0.0);
229 }
230 /* renormalize charge */
231 normalize();
232
233 if (ctx_.cfg().control().print_checksum()) {
234 auto cs = rho().checksum_rg();
235 if (ctx_.comm().rank() == 0) {
236 utils::print_checksum("rho_rg_init", cs);
237 }
238 }
239
240 /* initialize the magnetization */
241 if (ctx_.num_mag_dims()) {
242 double R = ctx_.cfg().control().rmt_max();
243
244 auto w = [R](double x) {
245 /* the constants are picked in such a way that the volume integral of the
246 weight function is equal to the volume of the atomic sphere;
247 in this case the starting magnetiation in the atomic spehre
248 integrates to the starting magnetization vector */
249
250 /* volume of the sphere */
251 const double norm = fourpi * std::pow(R, 3) / 3.0;
252 return (35.0 / 8) * std::pow(1 - std::pow(x / R, 2), 2) / norm;
253 // return 10 * std::pow(1 - x / R, 2) / norm;
254 // const double b = 1.1016992073677703;
255 // return b * 1.0 / (std::exp(10 * (a - R)) + 1) / norm;
256 // const double norm = pi * std::pow(R, 3) / 3.0;
257 // return 1.0 / (std::exp(10 * (x - R)) + 1) / norm;
258 };
259
260 #pragma omp parallel for
261 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
262 auto& atom_to_grid_map = ctx_.atoms_to_grid_idx_map(ia);
263 vector3d<double> v = unit_cell_.atom(ia).vector_field();
264
265 for (auto coord : atom_to_grid_map) {
266 int ir = coord.first;
267 double a = coord.second;
268 magnetization(0).f_rg(ir) += v[2] * w(a);
269 if (ctx_.num_mag_dims() == 3) {
270 magnetization(1).f_rg(ir) += v[0] * w(a);
271 magnetization(2).f_rg(ir) += v[1] * w(a);
272 }
273 }
274 }
275 }
276 this->fft_transform(-1);
277
278 if (ctx_.cfg().control().print_checksum()) {
279 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
280 auto cs = component(i).checksum_rg();
281 auto cs1 = component(i).checksum_pw();
282 if (ctx_.comm().rank() == 0) {
283 std::stringstream s;
284 s << "component[" << i << "]_rg";
285 utils::print_checksum(s.str(), cs);
286 std::stringstream s1;
287 s1 << "component[" << i << "]_pw";
288 utils::print_checksum(s1.str(), cs1);
289 }
290 }
291 }

294void Density::initial_density_full_pot()
295{
296 /* initialize smooth density of free atoms */
297 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
298 unit_cell_.atom_type(iat).init_free_atom_density(true);
299 }
300
301 /* compute radial integrals */
302 Radial_integrals_rho_free_atom ri(ctx_.unit_cell(), ctx_.pw_cutoff(), 40);
303
304 /* compute contribution from free atoms to the interstitial density */
305 auto v = ctx_.make_periodic_function<index_domain_t::local>([&ri](int iat, double g) { return ri.value(iat, g); });
306
307 /* initialize density of free atoms (not smoothed) */
308 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
309 unit_cell_.atom_type(iat).init_free_atom_density(false);
310 }
311
312 if (ctx_.cfg().control().print_checksum()) {
313 auto z = mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).checksum();
314 ctx_.comm().allreduce(&z, 1);
315 if (ctx_.comm().rank() == 0) {
316 utils::print_checksum("rho_pw", z);
317 }
318 }
319
320 /* set plane-wave coefficients of the charge density */
321 std::copy(v.begin(), v.end(), &rho().f_pw_local(0));
322 /* convert charge density to real space mesh */
323 rho().fft_transform(1);
324
325 if (ctx_.cfg().control().print_checksum()) {
326 auto cs = rho().checksum_rg();
327 if (ctx_.comm().rank() == 0) {
328 utils::print_checksum("rho_rg", cs);
329 }
330 }
331
332 /* remove possible negative noise */
333 for (int ir = 0; ir < ctx_.spfft<double>().local_slice_size(); ir++) {
334 rho().f_rg(ir) = std::max(0.0, rho().f_rg(ir));
335 }
336
337 /* set Y00 component of charge density */
338 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
339 int nmtp = ctx_.unit_cell().atom(ia).num_mt_points();
340
341 for (int ir = 0; ir < nmtp; ir++) {
342 double x = ctx_.unit_cell().atom(ia).radial_grid(ir);
343 rho().f_mt<index_domain_t::global>(0, ir, ia) = unit_cell_.atom(ia).type().free_atom_density(x) / y00;
344 }
345 }
346
347 int lmax = ctx_.lmax_rho();
348 int lmmax = utils::lmmax(lmax);
349
350 auto l_by_lm = utils::l_by_lm(lmax);
351
352 std::vector<double_complex> zil(lmax + 1);
353 for (int l = 0; l <= lmax; l++) {
354 zil[l] = std::pow(double_complex(0, 1), l);
355 }
356
357 /* compute boundary value at MT sphere from the plane-wave expansion */
358 auto gvec_ylm = ctx_.generate_gvec_ylm(lmax);
359
360 auto sbessel_mt = ctx_.generate_sbessel_mt(lmax);
361
362 auto flm = ctx_.sum_fg_fl_yg(lmax, v.data(), sbessel_mt, gvec_ylm);
363
364 /* this is the difference between the value of periodic charge density at MT boundary and
365 a value of the atom's free density at the boundary */
366 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
367 double R = ctx_.unit_cell().atom(ia).mt_radius();
368 double c = unit_cell_.atom(ia).type().free_atom_density(R) / y00;
369 flm(0, ia) -= c;
370 }
371
372 /* match density at MT */
373 for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
374 mdarray<double, 2> rRl(ctx_.unit_cell().max_num_mt_points(), lmax + 1);
375 double R = ctx_.unit_cell().atom_type(iat).mt_radius();
376 int nmtp = ctx_.unit_cell().atom_type(iat).num_mt_points();
377
378 #pragma omp parallel for default(shared)
379 for (int l = 0; l <= lmax; l++) {
380 for (int ir = 0; ir < nmtp; ir++) {
381 rRl(ir, l) = std::pow(ctx_.unit_cell().atom_type(iat).radial_grid(ir) / R, 2);
382 }
383 }
384 #pragma omp parallel for default(shared)
385 for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
386 int ia = unit_cell_.atom_type(iat).atom_id(i);
387 std::vector<double> glm(lmmax);
388 SHT::convert(lmax, &flm(0, ia), &glm[0]);
389 for (int lm = 0; lm < lmmax; lm++) {
390 int l = l_by_lm[lm];
391 for (int ir = 0; ir < nmtp; ir++) {
392 rho().f_mt<index_domain_t::global>(lm, ir, ia) += glm[lm] * rRl(ir, l);
393 }
394 }
395 }
396 }
397
398 /* normalize charge density */
399 normalize();
400
401 check_num_electrons();
402
403 // FILE* fout = fopen("rho.dat", "w");
404 // for (int i = 0; i <= 10000; i++) {
405 // vector3d<double> v = (i / 10000.0) * vector3d<double>({10.26, 10.26, 10.26});
406 // double val = rho().value(v);
407 // fprintf(fout, "%18.12f %18.12f\n", v.length(), val);
408 //}
409 // fclose(fout);
410
411 // FILE* fout2 = fopen("rho_rg.dat", "w");
412 // for (int i = 0; i <= 10000; i++) {
413 // vector3d<double> v = (i / 10000.0) * vector3d<double>({10.26, 10.26, 10.26});
414 // double val = rho().value_rg(v);
415 // fprintf(fout2, "%18.12f %18.12f\n", v.length(), val);
416 //}
417 // fclose(fout2);
418
419 /* initialize the magnetization */
420 if (ctx_.num_mag_dims()) {
421 for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
422 int ia = unit_cell_.spl_num_atoms(ialoc);
423 vector3d<double> v = unit_cell_.atom(ia).vector_field();
424 double len = v.length();
425
426 int nmtp = unit_cell_.atom(ia).num_mt_points();
427 Spline<double> rho_s(unit_cell_.atom(ia).type().radial_grid());
428 double R = unit_cell_.atom(ia).mt_radius();
429 for (int ir = 0; ir < nmtp; ir++) {
430 double x = unit_cell_.atom(ia).type().radial_grid(ir);
431 rho_s(ir) = this->rho().f_mt<index_domain_t::local>(0, ir, ialoc) * y00 *
432 (1 - 3 * std::pow(x / R, 2) + 2 * std::pow(x / R, 3));
433 }
434
435 /* maximum magnetization which can be achieved if we smooth density towards MT boundary */
436 double q = fourpi * rho_s.interpolate().integrate(2);
437
438 /* if very strong initial magnetization is given */
439 if (q < len) {
440 /* renormalize starting magnetization */
441 for (int x : {0, 1, 2}) {
442 v[x] *= (q / len);
443 }
444 len = q;
445 }
446
447 if (len > 1e-8) {
448 for (int ir = 0; ir < nmtp; ir++) {
449 magnetization(0).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[2] / q / y00;
450 }
451 if (ctx_.num_mag_dims() == 3) {
452 for (int ir = 0; ir < nmtp; ir++) {
453 magnetization(1).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[0] / q / y00;
454 magnetization(2).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[1] / q / y00;
455 }
456 }
457 }
458 }
459 }

462void Density::init_density_matrix_for_paw()
463{
464 density_matrix_.zero();
465
466 for (int ipaw = 0; ipaw < unit_cell_.num_paw_atoms(); ipaw++) {
467 int ia = unit_cell_.paw_atom_index(ipaw);
468
469 auto& atom = unit_cell_.atom(ia);
470 auto& atom_type = atom.type();
471
472 int nbf = atom_type.mt_basis_size();
473
474 auto& occupations = atom_type.paw_wf_occ();
475
476 /* magnetization vector */
477 auto magn = atom.vector_field();
478
479 for (int xi = 0; xi < nbf; xi++) {
480 auto& basis_func_index_dsc = atom_type.indexb()[xi];
481
482 int rad_func_index = basis_func_index_dsc.idxrf;
483
484 double occ = occupations[rad_func_index];
485
486 int l = basis_func_index_dsc.l;
487
488 switch (ctx_.num_mag_dims()) {
489 case 0: {
490 density_matrix_(xi, xi, 0, ia) = occ / double(2 * l + 1);
491 break;
492 }
493
494 case 3:
495 case 1: {
496 double nm = (std::abs(magn[2]) < 1.0) ? magn[2] : std::copysign(1, magn[2]);
497 density_matrix_(xi, xi, 0, ia) = 0.5 * (1.0 + nm) * occ / double(2 * l + 1);
498 density_matrix_(xi, xi, 1, ia) = 0.5 * (1.0 - nm) * occ / double(2 * l + 1);
499 break;
500 }
501 }
502 }
503 }

506void Density::generate_paw_atom_density(int idx__)
507{
508 int ia_paw = ctx_.unit_cell().spl_num_paw_atoms(idx__);
509 int ia = ctx_.unit_cell().paw_atom_index(ia_paw);
510
511 auto& atom_type = ctx_.unit_cell().atom(ia).type();
512
513 auto l_by_lm = utils::l_by_lm(2 * atom_type.indexr().lmax_lo());
514
515 /* get gaunt coefficients */
516 Gaunt_coefficients<double> GC(atom_type.indexr().lmax_lo(), 2 * atom_type.indexr().lmax_lo(),
517 atom_type.indexr().lmax_lo(), SHT::gaunt_rrr);
518
519 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
520 paw_density_.ae_density(i, idx__).zero();
521 paw_density_.ps_density(i, idx__).zero();
522 }
523
524 /* get radial grid to divide density over r^2 */
525 auto& grid = atom_type.radial_grid();
526
527 auto& paw_ae_wfs = atom_type.ae_paw_wfs_array();
528 auto& paw_ps_wfs = atom_type.ps_paw_wfs_array();
529
530 /* iterate over local basis functions (or over lm1 and lm2) */
531 for (int xi2 = 0; xi2 < atom_type.indexb().size(); xi2++) {
532 int lm2 = atom_type.indexb(xi2).lm;
533 int irb2 = atom_type.indexb(xi2).idxrf;
534
535 for (int xi1 = 0; xi1 <= xi2; xi1++) {
536 int lm1 = atom_type.indexb(xi1).lm;
537 int irb1 = atom_type.indexb(xi1).idxrf;
538
539 /* get num of non-zero GC */
540 int num_non_zero_gc = GC.num_gaunt(lm1, lm2);
541
542 double diag_coef = (xi1 == xi2) ? 1.0 : 2.0;
543
544 /* store density matrix in aux form */
545 double dm[4] = {0, 0, 0, 0};
546 switch (ctx_.num_mag_dims()) {
547 case 3: {
548 dm[2] = 2 * std::real(density_matrix_(xi1, xi2, 2, ia));
549 dm[3] = -2 * std::imag(density_matrix_(xi1, xi2, 2, ia));
550 }
551 case 1: {
552 dm[0] = std::real(density_matrix_(xi1, xi2, 0, ia) + density_matrix_(xi1, xi2, 1, ia));
553 dm[1] = std::real(density_matrix_(xi1, xi2, 0, ia) - density_matrix_(xi1, xi2, 1, ia));
554 break;
555 }
556 case 0: {
557 dm[0] = std::real(density_matrix_(xi1, xi2, 0, ia));
558 }
559 }
560
561 for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
562 auto& ae_dens = paw_density_.ae_density(imagn, idx__);
563 auto& ps_dens = paw_density_.ps_density(imagn, idx__);
564
565 /* add nonzero coefficients */
566 for (int inz = 0; inz < num_non_zero_gc; inz++) {
567 auto& lm3coef = GC.gaunt(lm1, lm2, inz);
568
569 /* iterate over radial points */
570 for (int irad = 0; irad < grid.num_points(); irad++) {
571
572 /* we need to divide density over r^2 since wave functions are stored multiplied by r */
573 double inv_r2 = diag_coef / (grid[irad] * grid[irad]);
574
575 /* calculate unified density/magnetization
576 * dm_ij * GauntCoef * ( phi_i phi_j + Q_ij) */
577 ae_dens(lm3coef.lm3, irad) +=
578 dm[imagn] * inv_r2 * lm3coef.coef * paw_ae_wfs(irad, irb1) * paw_ae_wfs(irad, irb2);
579 ps_dens(lm3coef.lm3, irad) +=
580 dm[imagn] * inv_r2 * lm3coef.coef *
581 (paw_ps_wfs(irad, irb1) * paw_ps_wfs(irad, irb2) +
582 atom_type.q_radial_function(irb1, irb2, l_by_lm[lm3coef.lm3])(irad));
583 }
584 }
585 }
586 }

602template <typename T>
603void Density::add_k_point_contribution_rg(K_point<T>* kp__)
604{
605 PROFILE("sirius::Density::add_k_point_contribution_rg");
606
607 double omega = unit_cell_.omega();
608
609 auto& fft = ctx_.spfft_coarse<T>();
610
611 /* local number of real-space points */
612 int nr = fft.local_slice_size();
613
614 /* get preallocated memory */
615 mdarray<T, 2> density_rg(nr, ctx_.num_mag_dims() + 1, ctx_.mem_pool(memory_t::host), "density_rg");
616 density_rg.zero();
617
618 if (fft.processing_unit() == SPFFT_PU_GPU) {
619 density_rg.allocate(ctx_.mem_pool(memory_t::device)).zero(memory_t::device);
620 }
621
622 /* location of the real-space wave-functions psi(r) */
623 auto data_ptr = kp__->spfft_transform().space_domain_data(kp__->spfft_transform().processing_unit());
624
625 /* non-magnetic or collinear case */
626 if (ctx_.num_mag_dims() != 3) {
627 /* loop over pure spinor components */
628 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
629 /* trivial case */
630 if (!kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col().global_index_size()) {
631 continue;
632 }
633 int ncols = kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col().local_size();
634 for (int i = 0; i < ncols; i++) {
635 /* global index of the band */
636 int j = kp__->spinor_wave_functions().pw_coeffs(ispn).spl_num_col()[i];
637 double w = kp__->band_occupancy(j, ispn) * kp__->weight() / omega;
638
639 auto inp_wf = kp__->spinor_wave_functions().pw_coeffs(ispn).extra().at(memory_t::host, 0, i);
640
641 /* transform to real space */
642 kp__->spfft_transform().backward(reinterpret_cast<const T*>(inp_wf),
643 kp__->spfft_transform().processing_unit());
644
645 switch (kp__->spfft_transform().processing_unit()) {
646 case SPFFT_PU_HOST: {
647 if (ctx_.gamma_point()) {
648 #pragma omp parallel for schedule(static)
649 for (int ir = 0; ir < nr; ir++) {
650 density_rg(ir, ispn) += w * std::pow(data_ptr[ir], 2);
651 }
652 } else {
653 auto data = reinterpret_cast<std::complex<T>*>(data_ptr);
654 #pragma omp parallel for schedule(static)
655 for (int ir = 0; ir < nr; ir++) {
656 auto z = data[ir];
657 density_rg(ir, ispn) += w * (std::pow(z.real(), 2) + std::pow(z.imag(), 2));
658 }
659 }
660 break;
661 }
662 case SPFFT_PU_GPU: {
663#if defined(SIRIUS_GPU)
664 if (ctx_.gamma_point()) {
665 update_density_rg_1_real_gpu(nr, data_ptr, w, density_rg.at(memory_t::device, 0, ispn));
666 } else {
667 auto data = reinterpret_cast<std::complex<T>*>(data_ptr);
668 update_density_rg_1_complex_gpu(nr, data, w, density_rg.at(memory_t::device, 0, ispn));
669 }
670#endif
671 break;
672 }
673 }
674 }
675 }
676 } else { /* non-collinear case */
677 assert(kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col().local_size() ==
678 kp__->spinor_wave_functions().pw_coeffs(1).spl_num_col().local_size());
679
680 /* allocate on CPU or GPU */
681 mdarray<std::complex<T>, 1> psi_r_up(nr, ctx_.mem_pool(memory_t::host));
682 if (fft.processing_unit() == SPFFT_PU_GPU) {
683 psi_r_up.allocate(ctx_.mem_pool(memory_t::device));
684 }
685 for (int i = 0; i < kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col().local_size(); i++) {
686 int j = kp__->spinor_wave_functions().pw_coeffs(0).spl_num_col()[i];
687 T w = kp__->band_occupancy(j, 0) * kp__->weight() / omega;
688
689 /* transform up- component of spinor function to real space; in case of GPU wave-function stays in GPU
690 * memory */
691 auto inp_wf_up = kp__->spinor_wave_functions().pw_coeffs(0).extra().at(memory_t::host, 0, i);
692 /* transform to real space */
693 kp__->spfft_transform().backward(reinterpret_cast<const T*>(inp_wf_up),
694 kp__->spfft_transform().processing_unit());
695
696 /* this is a non-collinear case, so the wave-functions and FFT buffer are complex and
697 we can copy memory */
698 switch (kp__->spfft_transform().processing_unit()) {
699 case SPFFT_PU_HOST: {
700 auto inp = reinterpret_cast<std::complex<T>*>(data_ptr);
701 std::copy(inp, inp + nr, psi_r_up.at(memory_t::host));
702 break;
703 }
704 case SPFFT_PU_GPU: {
705 acc::copy(psi_r_up.at(memory_t::device), reinterpret_cast<std::complex<T>*>(data_ptr), nr);
706 break;
707 }
708 }
709
710 /* transform dn- component of spinor wave function */
711 auto inp_wf_dn = kp__->spinor_wave_functions().pw_coeffs(1).extra().at(memory_t::host, 0, i);
712 kp__->spfft_transform().backward(reinterpret_cast<const T*>(inp_wf_dn),
713 kp__->spfft_transform().processing_unit());
714
715 auto psi_r_dn = reinterpret_cast<std::complex<T>*>(data_ptr);
716
717 switch (fft.processing_unit()) {
718 case SPFFT_PU_HOST: {
719 #pragma omp parallel for schedule(static)
720 for (int ir = 0; ir < nr; ir++) {
721 auto r0 = (std::pow(psi_r_up[ir].real(), 2) + std::pow(psi_r_up[ir].imag(), 2)) * w;
722 auto r1 = (std::pow(psi_r_dn[ir].real(), 2) + std::pow(psi_r_dn[ir].imag(), 2)) * w;
723
724 auto z2 = psi_r_up[ir] * std::conj(psi_r_dn[ir]) * w;
725
726 density_rg(ir, 0) += r0;
727 density_rg(ir, 1) += r1;
728 density_rg(ir, 2) += 2.0 * std::real(z2);
729 density_rg(ir, 3) -= 2.0 * std::imag(z2);
730 }
731 break;
732 }
733 case SPFFT_PU_GPU: {
734#ifdef SIRIUS_GPU
735 /* add up-up contribution */
736 update_density_rg_1_complex_gpu(nr, psi_r_up.at(memory_t::device), w,
737 density_rg.at(memory_t::device, 0, 0));
738 /* add dn-dn contribution */
739 update_density_rg_1_complex_gpu(nr, psi_r_dn, w, density_rg.at(memory_t::device, 0, 1));
740 /* add off-diagonal contribution */
741 update_density_rg_2_gpu(nr, psi_r_up.at(memory_t::device), psi_r_dn, w,
742 density_rg.at(memory_t::device, 0, 2),
743 density_rg.at(memory_t::device, 0, 3));
744#endif
745 break;
746 }
747 }
748 }
749 }
750
751 if (fft.processing_unit() == SPFFT_PU_GPU) {
752 density_rg.copy_to(memory_t::host);
753 }
754
755 /* switch from real density matrix to density and magnetization */
756 switch (ctx_.num_mag_dims()) {
757 case 3: {
758 #pragma omp parallel for schedule(static)
759 for (int ir = 0; ir < fft.local_slice_size(); ir++) {
760 rho_mag_coarse_[2]->f_rg(ir) += density_rg(ir, 2); // Mx
761 rho_mag_coarse_[3]->f_rg(ir) += density_rg(ir, 3); // My
762 }
763 }
764 case 1: {
765 #pragma omp parallel for schedule(static)
766 for (int ir = 0; ir < fft.local_slice_size(); ir++) {
767 rho_mag_coarse_[0]->f_rg(ir) += (density_rg(ir, 0) + density_rg(ir, 1)); // rho
768 rho_mag_coarse_[1]->f_rg(ir) += (density_rg(ir, 0) - density_rg(ir, 1)); // Mz
769 }
770 break;
771 }
772 case 0: {
773 #pragma omp parallel for schedule(static)
774 for (int ir = 0; ir < fft.local_slice_size(); ir++) {
775 rho_mag_coarse_[0]->f_rg(ir) += density_rg(ir, 0); // rho
776 }
777 }
778 }
779}

781template <typename T>
782void Density::add_k_point_contribution_dm(K_point<real_type<T>>* kp__, sddk::mdarray<double_complex, 4>& density_matrix__)
783{
784 PROFILE("sirius::Density::add_k_point_contribution_dm");
785
786 if (ctx_.full_potential()) {
787 /* non-magnetic or spin-collinear case */
788 if (ctx_.num_mag_dims() != 3) {
789 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
790 int nbnd = kp__->num_occupied_bands(ispn);
791 if (nbnd) {
792 sddk::mdarray<double_complex, 2> wf1(unit_cell_.max_mt_basis_size(), nbnd);
793 sddk::mdarray<double_complex, 2> wf2(unit_cell_.max_mt_basis_size(), nbnd);
794
795 for (int ialoc = 0; ialoc < kp__->spinor_wave_functions().spl_num_atoms().local_size(); ialoc++) {
796 int ia = kp__->spinor_wave_functions().spl_num_atoms()[ialoc];
797 int mt_basis_size = unit_cell_.atom(ia).type().mt_basis_size();
798 int offset_wf = kp__->spinor_wave_functions().offset_mt_coeffs(ialoc);
799 if (mt_basis_size != 0) {
800 for (int i = 0; i < nbnd; i++) {
801 for (int xi = 0; xi < mt_basis_size; xi++) {
802 auto c = kp__->spinor_wave_functions().mt_coeffs(ispn).prime(offset_wf + xi, i);
803 wf1(xi, i) = std::conj(c);
804 wf2(xi, i) = static_cast<double_complex>(c) * kp__->band_occupancy(i, ispn) * kp__->weight();
805 }
806 }
807 /* add |psi_j> n_j <psi_j| to density matrix */
808 linalg(linalg_t::blas).gemm(
809 'N', 'T', mt_basis_size, mt_basis_size, nbnd, &linalg_const<double_complex>::one(), &wf1(0, 0),
810 wf1.ld(), &wf2(0, 0), wf2.ld(), &linalg_const<double_complex>::one(),
811 density_matrix__.at(memory_t::host, 0, 0, ispn, ia), density_matrix__.ld());
812 }
813 }
814 }
815 }
816 } else {
817 int nbnd = kp__->num_occupied_bands();
818 if (nbnd) {
819 sddk::mdarray<double_complex, 3> wf1(unit_cell_.max_mt_basis_size(), nbnd, ctx_.num_spins());
820 sddk::mdarray<double_complex, 3> wf2(unit_cell_.max_mt_basis_size(), nbnd, ctx_.num_spins());
821
822 for (int ialoc = 0; ialoc < kp__->spinor_wave_functions().spl_num_atoms().local_size(); ialoc++) {
823 int ia = kp__->spinor_wave_functions().spl_num_atoms()[ialoc];
824 int mt_basis_size = unit_cell_.atom(ia).type().mt_basis_size();
825 int offset_wf = kp__->spinor_wave_functions().offset_mt_coeffs(ialoc);
826
827 if (mt_basis_size != 0) {
828 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
829 for (int i = 0; i < nbnd; i++) {
830
831 for (int xi = 0; xi < mt_basis_size; xi++) {
832 auto c = kp__->spinor_wave_functions().mt_coeffs(ispn).prime(offset_wf + xi, i);
833 wf1(xi, i, ispn) = std::conj(c);
834 wf2(xi, i, ispn) = static_cast<double_complex>(c) * kp__->band_occupancy(i, 0) * kp__->weight();
835 }
836 }
837 }
838 /* compute diagonal terms */
839 for (int ispn = 0; ispn < 2; ispn++) {
840 linalg(linalg_t::blas).gemm(
841 'N', 'T', mt_basis_size, mt_basis_size, nbnd, &linalg_const<double_complex>::one(),
842 &wf1(0, 0, ispn), wf1.ld(), &wf2(0, 0, ispn), wf2.ld(), &linalg_const<double_complex>::one(),
843 density_matrix__.at(memory_t::host, 0, 0, ispn, ia), density_matrix__.ld());
844 }
845 /* offdiagonal term */
846 linalg(linalg_t::blas).gemm(
847 'N', 'T', mt_basis_size, mt_basis_size, nbnd, &linalg_const<double_complex>::one(), &wf1(0, 0, 1),
848 wf1.ld(), &wf2(0, 0, 0), wf2.ld(), &linalg_const<double_complex>::one(),
849 density_matrix__.at(memory_t::host, 0, 0, 2, ia), density_matrix__.ld());
850 }
851 }
852 }
853 }
854 } else { /* pseudopotential */
855 if (!ctx_.unit_cell().mt_lo_basis_size()) {
856 return;
857 }
858
859 kp__->beta_projectors().prepare();
860
861 if (ctx_.num_mag_dims() != 3) {
862 for (int chunk = 0; chunk < kp__->beta_projectors().num_chunks(); chunk++) {
863 kp__->beta_projectors().generate(chunk);
864
865 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
866 /* total number of occupied bands for this spin */
867 int nbnd = kp__->num_occupied_bands(ispn);
868 /* compute <beta|psi> */
869 auto beta_psi =
870 kp__->beta_projectors().template inner<T>(chunk, kp__->spinor_wave_functions(), ispn, 0, nbnd);
871
872 /* number of beta projectors */
873 int nbeta = kp__->beta_projectors().chunk(chunk).num_beta_;
874
875 /* use communicator of the k-point to split band index */
876 splindex<splindex_t::block> spl_nbnd(nbnd, kp__->comm().size(), kp__->comm().rank());
877
878 int nbnd_loc = spl_nbnd.local_size();
879 if (nbnd_loc) { // TODO: this part can also be moved to GPU
880 #pragma omp parallel
881 {
882 /* auxiliary arrays */
883 mdarray<double_complex, 2> bp1(nbeta, nbnd_loc);
884 mdarray<double_complex, 2> bp2(nbeta, nbnd_loc);
885 #pragma omp for
886 for (int ia = 0; ia < kp__->beta_projectors().chunk(chunk).num_atoms_; ia++) {
887 int nbf = kp__->beta_projectors().chunk(chunk).desc_(
888 static_cast<int>(beta_desc_idx::nbf), ia);
889 if (!nbf) {
890 continue;
891 }
892 int offs = kp__->beta_projectors().chunk(chunk).desc_(
893 static_cast<int>(beta_desc_idx::offset), ia);
894 int ja =
895 kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::ia), ia);
896
897 for (int i = 0; i < nbnd_loc; i++) {
898 int j = spl_nbnd[i];
899
900 for (int xi = 0; xi < nbf; xi++) {
901 bp1(xi, i) = beta_psi(offs + xi, j);
902 bp2(xi, i) =
903 std::conj(bp1(xi, i)) * kp__->weight() * kp__->band_occupancy(j, ispn);
904 }
905 }
906
907 linalg(linalg_t::blas).gemm('N', 'T', nbf, nbf, nbnd_loc,
908 &linalg_const<double_complex>::one(), &bp1(0, 0), bp1.ld(),
909 &bp2(0, 0), bp2.ld(), &linalg_const<double_complex>::one(),
910 &density_matrix__(0, 0, ispn, ja), density_matrix__.ld());
911 }
912 }
913 }
914 }
915 }
916 } else {
917 for (int chunk = 0; chunk < kp__->beta_projectors().num_chunks(); chunk++) {
918 kp__->beta_projectors().generate(chunk);
919
920 /* number of beta projectors */
921 int nbeta = kp__->beta_projectors().chunk(chunk).num_beta_;
922
923 /* total number of occupied bands */
924 int nbnd = kp__->num_occupied_bands();
925
926 splindex<splindex_t::block> spl_nbnd(nbnd, kp__->comm().size(), kp__->comm().rank());
927 int nbnd_loc = spl_nbnd.local_size();
928
929 /* auxiliary arrays */
930 mdarray<double_complex, 3> bp1(nbeta, nbnd_loc, ctx_.num_spins());
931 mdarray<double_complex, 3> bp2(nbeta, nbnd_loc, ctx_.num_spins());
932
933 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
934 /* compute <beta|psi> */
935 auto beta_psi =
936 kp__->beta_projectors().template inner<T>(chunk, kp__->spinor_wave_functions(), ispn, 0, nbnd);
937 #pragma omp parallel for schedule(static)
938 for (int i = 0; i < nbnd_loc; i++) {
939 int j = spl_nbnd[i];
940
941 for (int m = 0; m < nbeta; m++) {
942 bp1(m, i, ispn) = beta_psi(m, j);
943 bp2(m, i, ispn) = static_cast<double_complex>(std::conj(beta_psi(m, j))) * kp__->weight() * kp__->band_occupancy(j, 0);
944 }
945 }
946 }
947 for (int ia = 0; ia < kp__->beta_projectors().chunk(chunk).num_atoms_; ia++) {
948 int nbf = kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::nbf), ia);
949 if (!nbf) {
950 continue;
951 }
952 int offs = kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::offset), ia);
953 int ja = kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::ia), ia);
954 if (ctx_.unit_cell().atom(ja).type().spin_orbit_coupling()) {
955 mdarray<double_complex, 3> bp3(nbf, nbnd_loc, 2);
956 bp3.zero();
957 /* We already have the <beta|psi> but we need to rotate
958 * them when the spin orbit interaction is included in the
959 * pseudo potential.
960 *
961 * We rotate \f[\langle\beta|\psi\rangle\f] accordingly by multiplying it with
962 * the \f[f^{\sigma\sigma^{'}}_{\xi,\xi^'}\f]
963 */
964
965 for (int xi1 = 0; xi1 < nbf; xi1++) {
966 for (int i = 0; i < nbnd_loc; i++) {
967 for (int xi1p = 0; xi1p < nbf; xi1p++) {
968 if (ctx_.unit_cell().atom(ja).type().compare_index_beta_functions(xi1, xi1p)) {
969 bp3(xi1, i, 0) +=
970 bp1(offs + xi1p, i, 0) *
971 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 0, 0) +
972 bp1(offs + xi1p, i, 1) *
973 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 0, 1);
974 bp3(xi1, i, 1) +=
975 bp1(offs + xi1p, i, 0) *
976 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 1, 0) +
977 bp1(offs + xi1p, i, 1) *
978 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1, xi1p, 1, 1);
979 }
980 }
981 }
982 }
983
984 for (int xi1 = 0; xi1 < nbf; xi1++) {
985 for (int i = 0; i < nbnd_loc; i++) {
986 bp1(offs + xi1, i, 0) = bp3(xi1, i, 0);
987 bp1(offs + xi1, i, 1) = bp3(xi1, i, 1);
988 }
989 }
990
991 bp3.zero();
992
993 for (int xi1 = 0; xi1 < nbf; xi1++) {
994 for (int i = 0; i < nbnd_loc; i++) {
995 for (int xi1p = 0; xi1p < nbf; xi1p++) {
996 if (ctx_.unit_cell().atom(ja).type().compare_index_beta_functions(xi1, xi1p)) {
997 bp3(xi1, i, 0) +=
998 bp2(offs + xi1p, i, 0) *
999 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 0, 0) +
1000 bp2(offs + xi1p, i, 1) *
1001 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 1, 0);
1002 bp3(xi1, i, 1) +=
1003 bp2(offs + xi1p, i, 0) *
1004 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 0, 1) +
1005 bp2(offs + xi1p, i, 1) *
1006 ctx_.unit_cell().atom(ja).type().f_coefficients(xi1p, xi1, 1, 1);
1007 }
1008 }
1009 }
1010 }
1011
1012 for (int xi1 = 0; xi1 < nbf; xi1++) {
1013 for (int i = 0; i < nbnd_loc; i++) {
1014 bp2(offs + xi1, i, 0) = bp3(xi1, i, 0);
1015 bp2(offs + xi1, i, 1) = bp3(xi1, i, 1);
1016 }
1017 }
1018 }
1019 }
1020
1021 if (nbnd_loc) {
1022 #pragma omp parallel for
1023 for (int ia = 0; ia < kp__->beta_projectors().chunk(chunk).num_atoms_; ia++) {
1024 int nbf = kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::nbf), ia);
1025 int offs =
1026 kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::offset), ia);
1027 int ja = kp__->beta_projectors().chunk(chunk).desc_(static_cast<int>(beta_desc_idx::ia), ia);
1028 /* compute diagonal spin blocks */
1029 for (int ispn = 0; ispn < 2; ispn++) {
1030 linalg(linalg_t::blas).gemm('N', 'T', nbf, nbf, nbnd_loc, &linalg_const<double_complex>::one(),
1031 &bp1(offs, 0, ispn), bp1.ld(), &bp2(offs, 0, ispn), bp2.ld(),
1032 &linalg_const<double_complex>::one(),
1033 &density_matrix__(0, 0, ispn, ja), density_matrix__.ld());
1034 }
1035 /* off-diagonal spin block */
1036 linalg(linalg_t::blas).gemm('N', 'T', nbf, nbf, nbnd_loc, &linalg_const<double_complex>::one(),
1037 &bp1(offs, 0, 0), bp1.ld(), &bp2(offs, 0, 1), bp2.ld(),
1038 &linalg_const<double_complex>::one(), &density_matrix__(0, 0, 2, ja),
1039 density_matrix__.ld());
1040 }
1041 }
1042 }
1043 }
1044 kp__->beta_projectors().dismiss();
1045 }

1130
1131 if (ctx_.full_potential()) {
1132 if (add_core__) {
1133 /* find the core states */

1145 for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
1146 this->component(iv).sync_mt();
1147 }
1148 }
1149 if (symmetrize__) {

1151 if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {

1153 std::unique_ptr<Occupation_matrix> om_ref;

1175 return nullptr;
1176 }
1177 };

1180 {

1185 }

1196 }

1198 occupation_matrix_) {

1201 if (ctx_.unit_cell().atom(ia).type().hubbard_correction()) {
1202 for (size_t i = 0; i < occupation_matrix_->local(ia).size(); i++) {
1203 diff = std::max(diff, std::abs(om_ref->local(ia)[i] - occupation_matrix_->local(ia)[i]));
1204 }
1205 }
1206 }
1207 std::string status = (diff > 1e-8) ? "Fail" : "OK";
1208 ctx_.message(1, __function_name__, "error of the LDA+U occupation matrix symmetrization: %12.6e %s\n",
1209 diff, status.c_str());
1210 }
1211 }
1212 }
1213
1214 if (occupation_matrix_) {
1215 /* non-local part; TODO: move to symmetrize.hpp */
1216 occupation_matrix_->symmetrize();
1217 occupation_matrix_->print_occupancies(2);
1218 }
1219
1220 generate_paw_loc_density();
1221
1222 if (transform_to_rg__) {
1223 this->fft_transform(1);
1224 }

1251template <typename T>
1252void Density::generate_valence(K_point_set const& ks__)
1253{
1254 PROFILE("sirius::Density::generate_valence");
1255
1256 /* check weights */
1257 double wt{0};
1258 double occ_val{0};
1259 for (int ik = 0; ik < ks__.num_kpoints(); ik++) {
1260 wt += ks__.get<T>(ik)->weight();
1261 for (int ispn = 0; ispn < ctx_.num_spinors(); ispn++) {
1262 for (int j = 0; j < ctx_.num_bands(); j++) {
1263 occ_val += ks__.get<T>(ik)->weight() * ks__.get<T>(ik)->band_occupancy(j, ispn);
1264 }
1265 }
1266 }
1267
1268 if (std::abs(wt - 1.0) > 1e-12) {
1269 std::stringstream s;
1270 s << "K_point weights don't sum to one" << std::endl << " obtained sum: " << wt;
1271 TERMINATE(s);
1272 }
1273
1274 if (std::abs(occ_val - unit_cell_.num_valence_electrons() + ctx_.cfg().parameters().extra_charge()) > 1e-8 &&
1275 ctx_.comm().rank() == 0) {
1276 std::stringstream s;
1277 s << "wrong band occupancies" << std::endl
1278 << " computed : " << occ_val << std::endl
1279 << " required : " << unit_cell_.num_valence_electrons() - ctx_.cfg().parameters().extra_charge() << std::endl
1280 << " difference : "
1281 << std::abs(occ_val - unit_cell_.num_valence_electrons() + ctx_.cfg().parameters().extra_charge());
1282 WARNING(s);
1283 }
1284
1285 density_matrix_.zero();
1286
1287 if (occupation_matrix_) {
1288 occupation_matrix_->zero();
1289 }
1290
1291 /* zero density and magnetization */
1292 zero();
1293 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
1294 rho_mag_coarse_[i]->zero();
1295 }
1296
1297 /* start the main loop over k-points */
1298 for (int ikloc = 0; ikloc < ks__.spl_num_kpoints().local_size(); ikloc++) {
1299 int ik = ks__.spl_num_kpoints(ikloc);
1300 auto kp = ks__.get<T>(ik);
1301
1302 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
1303 int nbnd = kp->num_occupied_bands(ispn);
1304 /* swap wave functions for the FFT transformation */
1305 kp->spinor_wave_functions().pw_coeffs(ispn).remap_forward(nbnd, 0, &ctx_.mem_pool(memory_t::host));
1306 }
1307
1308 /*
1309 * GPU memory allocation
1310 */
1311 if (is_device_memory(ctx_.preferred_memory_t())) {
1312 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
1313 int nbnd = kp->num_occupied_bands(ispn);
1314 /* allocate GPU memory */
1315 kp->spinor_wave_functions().pw_coeffs(ispn).prime().allocate(ctx_.mem_pool(memory_t::device));
1316 /* copy to GPU */
1317 kp->spinor_wave_functions().pw_coeffs(ispn).copy_to(memory_t::device, 0, nbnd);
1318 }
1319 }
1320 if (!ctx_.full_potential() && ctx_.hubbard_correction() &&
1321 is_device_memory(kp->wave_functions_S_hub().preferred_memory_t())) {
1322 int nwfu = kp->wave_functions_S_hub().num_wf();
1323 for (int ispn = 0; ispn < kp->wave_functions_S_hub().num_sc(); ispn++) {
1324 /* allocate GPU memory */
1325 kp->wave_functions_S_hub().pw_coeffs(ispn).prime().allocate(ctx_.mem_pool(memory_t::device));
1326 /* copy to GPU */
1327 kp->wave_functions_S_hub().pw_coeffs(ispn).copy_to(memory_t::device, 0, nwfu);
1328 }
1329 }
1330
1331 if (ctx_.electronic_structure_method() == electronic_structure_method_t::full_potential_lapwlo) {
1332 add_k_point_contribution_dm_complex<T>(kp, density_matrix_);
1333 }
1334
1335 if (ctx_.electronic_structure_method() == electronic_structure_method_t::pseudopotential) {
1336 if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
1337 add_k_point_contribution_dm_real<T>(kp, density_matrix_);
1338 } else {
1339 add_k_point_contribution_dm_complex<T>(kp, density_matrix_);
1340 }
1341 if (occupation_matrix_) {
1342 occupation_matrix_->add_k_point_contribution(*kp);
1343 }
1344 }
1345
1346 /* add contribution from regular space grid */
1347 add_k_point_contribution_rg(kp);
1348
1349 /*
1350 * GPU memory deallocation
1351 */
1352 if (is_device_memory(ctx_.preferred_memory_t())) {
1353 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
1354 /* deallocate GPU memory */
1355 kp->spinor_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
1356 }
1357 }
1358 if (!ctx_.full_potential() && ctx_.hubbard_correction() &&
1359 is_device_memory(kp->wave_functions_S_hub().preferred_memory_t())) {
1360 for (int ispn = 0; ispn < kp->wave_functions_S_hub().num_sc(); ispn++) {
1361 kp->wave_functions_S_hub().pw_coeffs(ispn).deallocate(memory_t::device);
1362 }
1363 }
1364 }
1365
1366 if (density_matrix_.size()) {
1367 ctx_.comm().allreduce(density_matrix_.at(memory_t::host), static_cast<int>(density_matrix_.size()));
1368 }
1369
1370 if (occupation_matrix_) {
1371 occupation_matrix_->reduce();
1372 }
1373
1374 auto& comm = ctx_.gvec_coarse_partition().comm_ortho_fft();
1375 for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
1376 auto ptr = (ctx_.spfft_coarse<double>().local_slice_size() == 0) ? nullptr : &rho_mag_coarse_[j]->f_rg(0);
1377 /* reduce arrays; assume that each rank did its own fraction of the density */
1378 /* comm_ortho_fft is identical to a product of column communicator inside k-point with k-point communicator */
1379 comm.allreduce(ptr, ctx_.spfft_coarse<double>().local_slice_size());
1380 /* print checksum if needed */
1381 if (ctx_.cfg().control().print_checksum()) {
1382 auto cs = mdarray<double, 1>(ptr, ctx_.spfft_coarse<double>().local_slice_size()).checksum();
1383 Communicator(ctx_.spfft_coarse<double>().communicator()).allreduce(&cs, 1);
1384 if (ctx_.comm().rank() == 0) {
1385 utils::print_checksum("rho_mag_coarse_rg", cs);
1386 }
1387 }
1388 /* transform to PW domain */
1389 rho_mag_coarse_[j]->fft_transform(-1);
1390 /* map to fine G-vector grid */
1391 for (int igloc = 0; igloc < ctx_.gvec_coarse().count(); igloc++) {
1392 component(j).f_pw_local(ctx_.gvec().gvec_base_mapping(igloc)) = rho_mag_coarse_[j]->f_pw_local(igloc);
1393 }
1394 }
1395
1396 if (!ctx_.full_potential()) {
1397 augment();
1398
1399 /* remove extra chanrge */
1400 if (ctx_.gvec().comm().rank() == 0) {
1401 rho().f_pw_local(0) += ctx_.cfg().parameters().extra_charge() / ctx_.unit_cell().omega();
1402 }
1403
1404 if (ctx_.cfg().control().print_hash() && ctx_.comm().rank() == 0) {
1405 auto h = mdarray<double_complex, 1>(&rho().f_pw_local(0), ctx_.gvec().count()).hash();
1406 utils::print_hash("rho", h);
1407 }
1408
1409 double nel = rho().f_0().real() * unit_cell_.omega();
1410 /* check the number of electrons */
1411 if (std::abs(nel - unit_cell_.num_electrons()) > 1e-8 && ctx_.comm().rank() == 0) {
1412 std::stringstream s;
1413 s << "wrong unsymmetrized density" << std::endl
1414 << " obtained value : " << std::scientific << nel << std::endl
1415 << " target value : " << std::scientific << unit_cell_.num_electrons() << std::endl
1416 << " difference : " << std::scientific << std::abs(nel - unit_cell_.num_electrons()) << std::endl;
1417 WARNING(s);
1418 }
1419 }
1420
1421 /* for muffin-tin part */
1422 if (ctx_.full_potential()) {
1423 generate_valence_mt();
1424 }
1425}

1430

1432

1434 switch (ctx_.processing_unit()) {
1435 case device_t::CPU: {
1436 rho_aug.zero(memory_t::host);

1438 }
1439 case device_t::GPU: {
1440 rho_aug.allocate(ctx_.mem_pool(memory_t::device)).zero(memory_t::device);
1441 break;
1442 }
1443 }
1444
1445 // TODO: the GPU memory consumption here is huge, rewrite this; split gloc in blocks and

1450 }

1453 auto& atom_type = unit_cell_.atom_type(iat);
1454
1455 if (ctx_.processing_unit() == device_t::GPU) {

1470
1471 if (ctx_.cfg().control().print_checksum()) {
1472 auto cs = dm.checksum();
1473 if (ctx_.comm().rank() == 0) {
1474 utils::print_checksum("density_matrix_aux", cs);
1475 }
1476 }
1477 /* treat auxiliary array as double with x2 size */

1479 sddk::mdarray<double, 2> phase_factors(atom_type.num_atoms(), spl_ngv_loc.local_size() * 2,
1480 ctx_.mem_pool(memory_t::host));
1481
1482 ctx_.print_memory_usage(__FILE__, __LINE__);
1483
1484 switch (ctx_.processing_unit()) {
1485 case device_t::CPU: {
1486 break;
1487 }
1488 case device_t::GPU: {
1489 phase_factors.allocate(ctx_.mem_pool(memory_t::device));
1490 dm_pw.allocate(ctx_.mem_pool(memory_t::device));
1491 dm.allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
1492 break;
1493 }
1494 }
1495
1496 ctx_.print_memory_usage(__FILE__, __LINE__);
1497
1498 for (int ib = 0; ib < spl_ngv_loc.num_ranks(); ib++) {
1499 int g_begin = spl_ngv_loc.global_index(0, ib);
1500 int g_end = g_begin + spl_ngv_loc.local_size(ib);
1501
1502 switch (ctx_.processing_unit()) {
1503 case device_t::CPU: {
1504 #pragma omp parallel for schedule(static)
1505 for (int igloc = g_begin; igloc < g_end; igloc++) {
1506 int ig = ctx_.gvec().offset() + igloc;
1507 for (int i = 0; i < atom_type.num_atoms(); i++) {
1508 int ia = atom_type.atom_id(i);
1509 double_complex z = std::conj(ctx_.gvec_phase_factor(ig, ia));
1510 phase_factors(i, 2 * (igloc - g_begin)) = z.real();
1511 phase_factors(i, 2 * (igloc - g_begin) + 1) = z.imag();
1512 }
1513 }
1514 for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
1515 PROFILE_START("sirius::Density::generate_rho_aug|gemm");
1516 linalg(linalg_t::blas).gemm('N', 'N', nbf * (nbf + 1) / 2, 2 * spl_ngv_loc.local_size(ib),
1517 atom_type.num_atoms(), &linalg_const<double>::one(), dm.at(memory_t::host, 0, 0, iv),
1518 dm.ld(), phase_factors.at(memory_t::host), phase_factors.ld(),
1519 &linalg_const<double>::zero(), dm_pw.at(memory_t::host, 0, 0), dm_pw.ld());
1520 PROFILE_STOP("sirius::Density::generate_rho_aug|gemm");
1521 PROFILE_START("sirius::Density::generate_rho_aug|sum");
1522 #pragma omp parallel for
1523 for (int igloc = g_begin; igloc < g_end; igloc++) {

1526 for (int i = 0; i < nbf * (nbf + 1) / 2; i++) {
1527 double_complex z1 = double_complex(ctx_.augmentation_op(iat)->q_pw(i, 2 * igloc),
1528 ctx_.augmentation_op(iat)->q_pw(i, 2 * igloc + 1));
1529 double_complex z2(dm_pw(i, 2 * (igloc - g_begin)), dm_pw(i, 2 * (igloc - g_begin) + 1));
1530
1531 zsum += z1 * z2 * ctx_.augmentation_op(iat)->sym_weight(i);
1532 }
1533 rho_aug(igloc, iv) += zsum;
1534 }
1535 PROFILE_STOP("sirius::Density::generate_rho_aug|sum");
1536 }
1537 break;
1538 }
1539 case device_t::GPU: {
1540#if defined(SIRIUS_GPU)
1541 for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
1542 generate_dm_pw_gpu(atom_type.num_atoms(), spl_ngv_loc.local_size(ib), nbf,
1543 ctx_.unit_cell().atom_coord(iat).at(memory_t::device),
1544 ctx_.gvec_coord().at(memory_t::device, g_begin, 0),
1545 ctx_.gvec_coord().at(memory_t::device, g_begin, 1),
1546 ctx_.gvec_coord().at(memory_t::device, g_begin, 2),
1547 phase_factors.at(memory_t::device), dm.at(memory_t::device, 0, 0, iv),
1548 dm_pw.at(memory_t::device), 1);
1549 sum_q_pw_dm_pw_gpu(spl_ngv_loc.local_size(ib), nbf,
1550 ctx_.augmentation_op(iat)->q_pw().at(memory_t::device, 0, 2 * g_begin),
1551 dm_pw.at(memory_t::device),
1552 ctx_.augmentation_op(iat)->sym_weight().at(memory_t::device),
1553 rho_aug.at(memory_t::device, g_begin, iv), 1);
1554 }
1555#endif
1556 break;
1557 }
1558 }
1559 }
1560
1561 if (ctx_.processing_unit() == device_t::GPU) {
1562 acc::sync_stream(stream_id(1));
1563 ctx_.augmentation_op(iat)->dismiss();
1564 }
1565 }
1566
1567 if (ctx_.processing_unit() == device_t::GPU) {
1568 rho_aug.copy_to(memory_t::host);
1569 }
1570
1571 if (ctx_.cfg().control().print_checksum()) {
1572 auto cs = rho_aug.checksum();
1573 ctx_.comm().allreduce(&cs, 1);
1574 if (ctx_.comm().rank() == 0) {
1575 utils::print_checksum("rho_aug", cs);
1576 }
1577 }
1578
1579 if (ctx_.cfg().control().print_hash()) {
1580 auto h = rho_aug.hash();
1581 if (ctx_.comm().rank() == 0) {
1582 utils::print_hash("rho_aug", h);
1583 }
1584 }
1585
1586 return rho_aug;
1587}
