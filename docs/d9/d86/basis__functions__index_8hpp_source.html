77 /// A helper class to establish various index mappings for the atomic basis functions.
78 /** Atomic basis function is a radial function multiplied by a spherical harmonic:
79  \f[
80  \phi_{\ell m \nu}({\bf r}) = f_{\ell \nu}(r) Y_{\ell m}(\hat {\bf r})
81  \f]
82  Multiple radial functions for each \f$ \ell \f$ channel are allowed. This is reflected by
83  the \f$ \nu \f$ index and called "order".
84  */
85 class basis_functions_index
86 {
87  private:
88  std::vector<basis_function_index_descriptor> basis_function_index_descriptors_;
89 
90  mdarray<int, 2> index_by_lm_order_;
91 
92  mdarray<int, 1> index_by_idxrf_; // TODO: rename to first_lm_index_by_idxrf_ or similar
93 
94  /// Number of augmented wave basis functions.
95  int size_aw_{0};
96 
97  /// Number of local orbital basis functions.
98  int size_lo_{0};
99 
100  /// Maximum l of the radial basis functions.
101  int lmax_{-1};
102 
103  public:
104 
105  void init(radial_functions_index& indexr__)
106  {
107  basis_function_index_descriptors_.clear();
108 
109  index_by_idxrf_ = mdarray<int, 1>(indexr__.size());
110 
111  for (int idxrf = 0; idxrf < indexr__.size(); idxrf++) {
112  int l = indexr__[idxrf].l;
113  int order = indexr__[idxrf].order;
114  int idxlo = indexr__[idxrf].idxlo;
115 
116  index_by_idxrf_(idxrf) = (int)basis_function_index_descriptors_.size();
117 
118  for (int m = -l; m <= l; m++) {
119  basis_function_index_descriptors_.push_back(
120  basis_function_index_descriptor(l, m, indexr__[idxrf].j, order, idxlo, idxrf));
121  }
122  }
123  index_by_lm_order_ = mdarray<int, 2>(utils::lmmax(indexr__.lmax()), indexr__.max_num_rf());
124 
125  for (int i = 0; i < (int)basis_function_index_descriptors_.size(); i++) {
126  int lm = basis_function_index_descriptors_[i].lm;
127  int order = basis_function_index_descriptors_[i].order;
128  index_by_lm_order_(lm, order) = i;
129 
130  /* get number of aw basis functions */
131  if (basis_function_index_descriptors_[i].idxlo < 0) {
132  size_aw_ = i + 1;
133  }
134  }
135 
136  size_lo_ = (int)basis_function_index_descriptors_.size() - size_aw_;
137 
138  lmax_ = indexr__.lmax();
139 
140  assert(size_aw_ >= 0);
141  assert(size_lo_ >= 0);
142  }
143
