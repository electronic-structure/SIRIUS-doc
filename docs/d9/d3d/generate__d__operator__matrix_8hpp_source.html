35 inline void Potential::generate_D_operator_matrix()
36 {
37  PROFILE("sirius::Potential::generate_D_operator_matrix");
38 
39  /* store effective potential and magnetic field in a vector */
40  std::vector<Periodic_function<double>*> veff_vec(ctx_.num_mag_dims() + 1);
41  veff_vec[0] = effective_potential_.get();
42  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
43  veff_vec[1 + j] = effective_magnetic_field_[j];
44  }
45 
46  mdarray<double_complex, 1> veff_tmp(nullptr, ctx_.gvec().count());
47  if (ctx_.processing_unit() == GPU) {
48  veff_tmp.allocate(memory_t::device);
49  }
50 
51  if (ctx_.unit_cell().atom_type(0).augment() && ctx_.unit_cell().atom_type(0).num_atoms() > 0) {
52  ctx_.augmentation_op(0).prepare(0);
53  }
54 
55  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
56  auto& atom_type = unit_cell_.atom_type(iat);
57  int nbf = atom_type.mt_basis_size();
58 
59 #ifdef __GPU
60  /* start copy of Q(G) for the next atom type */
61  if (ctx_.processing_unit() == GPU) {
62  acc::sync_stream(0);
63  if (iat + 1 != unit_cell_.num_atom_types() && ctx_.unit_cell().atom_type(iat + 1).augment() &&
64  ctx_.unit_cell().atom_type(iat + 1).num_atoms() > 0) {
65  ctx_.augmentation_op(iat + 1).prepare(0);
66  }
67  }
68 #endif
69 
70  /* trivial case */
71  if (!atom_type.augment() || atom_type.num_atoms() == 0) {
72  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
73  for (int i = 0; i < atom_type.num_atoms(); i++) {
74  int ia = atom_type.atom_id(i);
75  auto& atom = unit_cell_.atom(ia);
76 
77  for (int xi2 = 0; xi2 < nbf; xi2++) {
78  for (int xi1 = 0; xi1 < nbf; xi1++) {
79  atom.d_mtrx(xi1, xi2, iv) = 0;
80  if (atom_type.spin_orbit_coupling()) {
81  atom.d_mtrx_so(xi1, xi2, iv) = 0;
82  }
83  }
84  }
85  }
86  }
87  continue;
88  }
89  matrix<double> d_tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms());
90  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
91  switch (ctx_.processing_unit()) {
92  case CPU: {
93  matrix<double> veff_a(2 * ctx_.gvec().count(), atom_type.num_atoms());
94 
95  #pragma omp parallel for schedule(static)
96  for (int i = 0; i < atom_type.num_atoms(); i++) {
97  int ia = atom_type.atom_id(i);
98 
99  for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
100  int ig = ctx_.gvec().offset() + igloc;
101  /* V(G) * exp(i * G * r_{alpha}) */
102  auto z = veff_vec[iv]->f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia);
103  veff_a(2 * igloc, i) = z.real();
104  veff_a(2 * igloc + 1, i) = z.imag();
105  }
106  }
107 
108  linalg<CPU>::gemm(0, 0, nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
109  ctx_.augmentation_op(iat).q_pw(), veff_a, d_tmp);
110  break;
111  }
112  case GPU: {
113 #ifdef __GPU
114  /* copy plane wave coefficients of effective potential to GPU */
115  mdarray<double_complex, 1> veff(&veff_vec[iv]->f_pw_local(0), veff_tmp.at<GPU>(),
116  ctx_.gvec().count());
117  veff.copy<memory_t::host, memory_t::device>();
118 
119  matrix<double> veff_a(2 * ctx_.gvec().count(), atom_type.num_atoms(), memory_t::device);
120 
121  d_tmp.allocate(memory_t::device);
122 
123  mul_veff_with_phase_factors_gpu(atom_type.num_atoms(), ctx_.gvec().count(), veff.at<GPU>(),
124  ctx_.gvec_coord().at<GPU>(), ctx_.atom_coord(iat).at<GPU>(),
125  veff_a.at<GPU>(), 1);
126 
127  linalg<GPU>::gemm(0, 0, nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
128  ctx_.augmentation_op(iat).q_pw(), veff_a, d_tmp, 1);
129 
130  d_tmp.copy<memory_t::device, memory_t::host>();
131 #endif
132  break;
133  }
134  }
135 
136  if (ctx_.gvec().reduced()) {
137  if (comm_.rank() == 0) {
138  for (int i = 0; i < atom_type.num_atoms(); i++) {
139  for (int j = 0; j < nbf * (nbf + 1) / 2; j++) {
140  d_tmp(j, i) = 2 * d_tmp(j, i) -
141  veff_vec[iv]->f_pw_local(0).real() * ctx_.augmentation_op(iat).q_pw(j, 0);
142  }
143  }
144  } else {
145  for (int i = 0; i < atom_type.num_atoms(); i++) {
146  for (int j = 0; j < nbf * (nbf + 1) / 2; j++) {
147  d_tmp(j, i) *= 2;
148  }
149  }
150  }
151  }
152 
153  comm_.allreduce(d_tmp.at<CPU>(), static_cast<int>(d_tmp.size()));
154 
155  if (ctx_.control().print_checksum_ && ctx_.comm().rank() == 0) {
156  for (int i = 0; i < atom_type.num_atoms(); i++) {
157  std::stringstream s;
158  s << "D_mtrx_val(atom_t" << iat << "_i" << i << "_c" << iv << ")";
159  auto cs = mdarray<double, 1>(&d_tmp(0, i), nbf * (nbf + 1) / 2).checksum();
160  print_checksum(s.str(), cs);
161  }
162  //auto cs = d_tmp.checksum();
163  //print_checksum("D_mtrx_valence", cs);
164  }
165 
166  #pragma omp parallel for schedule(static)
167  for (int i = 0; i < atom_type.num_atoms(); i++) {
168  int ia = atom_type.atom_id(i);
169  auto& atom = unit_cell_.atom(ia);
170 
171  for (int xi2 = 0; xi2 < nbf; xi2++) {
172  for (int xi1 = 0; xi1 <= xi2; xi1++) {
173  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
174  /* D-matix is symmetric */
175  atom.d_mtrx(xi1, xi2, iv) = atom.d_mtrx(xi2, xi1, iv) = d_tmp(idx12, i) * unit_cell_.omega();
176  }
177  }
178  }
179  }
180 
181  // Now compute the d operator for atoms with so interactions
182  if (atom_type.spin_orbit_coupling()) {
183  #pragma omp parallel for schedule(static)
184  for (int i = 0; i < atom_type.num_atoms(); i++) {
185  int ia = atom_type.atom_id(i);
186  auto& atom = unit_cell_.atom(ia);
187  for (int xi2 = 0; xi2 < nbf; xi2++) {
188  for (int xi1 = 0; xi1 < nbf; xi1++) {
189 
190  // first compute \f[A^_\alpha I^{I,\alpha}_{xi,xi}\f]
191  // cf Eq.19 PRB 71 115106
192 
193  // note that the I integrals are already calculated and
194  // stored in atom.d_mtrx
195  for (int sigma = 0; sigma < 2; sigma++) {
196  for (int sigmap = 0; sigmap < 2; sigmap++) {
197  double_complex Pauli_vector[4][2][2];
198  // Id
199  Pauli_vector[0][0][0] = double_complex(1.0, 0.0);
200  Pauli_vector[0][0][1] = double_complex(0.0, 0.0);
201  Pauli_vector[0][1][0] = double_complex(0.0, 0.0);
202  Pauli_vector[0][1][1] = double_complex(1.0, 0.0);
203  // sigma_z
204  Pauli_vector[1][0][0] = double_complex(1.0, 0.0);
205  Pauli_vector[1][0][1] = double_complex(0.0, 0.0);
206  Pauli_vector[1][1][0] = double_complex(0.0, 0.0);
207  Pauli_vector[1][1][1] = double_complex(-1.0, 0.0);
208  // sigma_x
209  Pauli_vector[2][0][0] = double_complex(0.0, 0.0);
210  Pauli_vector[2][0][1] = double_complex(1.0, 0.0);
211  Pauli_vector[2][1][0] = double_complex(1.0, 0.0);
212  Pauli_vector[2][1][1] = double_complex(0.0, 0.0);
213  // sigma_y
214  Pauli_vector[3][0][0] = double_complex(0.0, 0.0);
215  Pauli_vector[3][0][1] = double_complex(0.0, -1.0);
216  Pauli_vector[3][1][0] = double_complex(0.0, 1.0);
217  Pauli_vector[3][1][1] = double_complex(0.0, 0.0);
218  double_complex result = {0.0, 0.0};
219  for (auto xi2p = 0; xi2p < nbf; xi2p++) {
220  if (atom_type.compare_index_beta_functions(xi2, xi2p)) {
221  // just sum over m2, all other indices are the same
222  for (auto xi1p = 0; xi1p < nbf; xi1p++) {
223  if (atom_type.compare_index_beta_functions(xi1, xi1p)) {
224  // just sum over m1, all other indices are the same
225 
226  for (int alpha = 0; alpha < 4; alpha++) { // loop over the 0, z,x,y coordinates
227  for (int sigma1 = 0; sigma1 < 2; sigma1++) {
228  for (int sigma2 = 0; sigma2 < 2; sigma2++) {
229  result +=
230  atom.d_mtrx(xi1p, xi2p, alpha) *
231  Pauli_vector[alpha][sigma1][sigma2] *
232  atom.type().f_coefficients(xi1, xi1p, sigma, sigma1) *
233  atom.type().f_coefficients(xi2p, xi2, sigma2, sigmap);
234  }
235  }
236  }
237  }
238  }
239  }
240  }
241  const int ind =
242  (sigma == sigmap) * sigma + (1 + 2 * sigma + sigmap) * (sigma != sigmap);
243  atom.d_mtrx_so(xi1, xi2, ind) = result;
244  }
245  }
246  }
247  }
248  }
249  }
250  ctx_.augmentation_op(iat).dismiss();
251  }
252 
253  /* add d_ion to the effective potential component of D-operator */
254  #pragma omp parallel for schedule(static)
255  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
256  auto& atom_type = unit_cell_.atom(ia).type();
257  int nbf = unit_cell_.atom(ia).mt_basis_size();
258 
259  auto& dion = atom_type.d_mtrx_ion();
260 
261  if (atom_type.spin_orbit_coupling()) {
262  // spin orbit coupling mixes this term
263 
264  // keep the order of the indices because it is crucial
265  // here. Permuting the indices makes things wrong
266 
267  for (int xi2 = 0; xi2 < nbf; xi2++) {
268  int l2 = atom_type.indexb(xi2).l;
269  double j2 = atom_type.indexb(xi2).j;
270  int idxrf2 = atom_type.indexb(xi2).idxrf;
271  for (int xi1 = 0; xi1 < nbf; xi1++) {
272  int l1 = atom_type.indexb(xi1).l;
273  double j1 = atom_type.indexb(xi1).j;
274  int idxrf1 = atom_type.indexb(xi1).idxrf;
275  if ((l1 == l2) && (std::abs(j1 - j2) < 1e-8)) {
276  // up-up down-down
277  unit_cell_.atom(ia).d_mtrx_so(xi1, xi2, 0) +=
278  dion(idxrf1, idxrf2) * atom_type.f_coefficients(xi1, xi2, 0, 0);
279  unit_cell_.atom(ia).d_mtrx_so(xi1, xi2, 1) +=
280  dion(idxrf1, idxrf2) * atom_type.f_coefficients(xi1, xi2, 1, 1);
281 
282  // up-down down-up
283  unit_cell_.atom(ia).d_mtrx_so(xi1, xi2, 2) +=
284  dion(idxrf1, idxrf2) * atom_type.f_coefficients(xi1, xi2, 0, 1);
285  unit_cell_.atom(ia).d_mtrx_so(xi1, xi2, 3) +=
286  dion(idxrf1, idxrf2) * atom_type.f_coefficients(xi1, xi2, 1, 0);
287  }
288  }
289  }
290  } else {
291  for (int xi2 = 0; xi2 < nbf; xi2++) {
292  int lm2 = atom_type.indexb(xi2).lm;
293  int idxrf2 = atom_type.indexb(xi2).idxrf;
294  for (int xi1 = 0; xi1 < nbf; xi1++) {
295  int lm1 = atom_type.indexb(xi1).lm;
296  int idxrf1 = atom_type.indexb(xi1).idxrf;
297 
298  if (lm1 == lm2) {
299  unit_cell_.atom(ia).d_mtrx(xi1, xi2, 0) += dion(idxrf1, idxrf2);
300  }
301  }
302  }
303  }
304  }
305 
306  if (ctx_.control().print_checksum_ && ctx_.comm().rank() == 0) {
307  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
308  auto cs = unit_cell_.atom(ia).d_mtrx().checksum();
309  std::stringstream s;
310  s << "D_mtrx_tot(atom_" << ia << ")";
311  print_checksum(s.str(), cs);
312  }
313  }
