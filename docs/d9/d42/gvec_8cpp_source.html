30 vector3d<int> sddk::Gvec::gvec_by_full_index(uint32_t idx__) const
31 {
32  /* index of the z coordinate of G-vector: first 12 bits */
33  uint32_t j = idx__ & 0xFFF;
34  /* index of z-column: last 20 bits */
35  uint32_t i = idx__ >> 12;
36  assert(i < (uint32_t)z_columns_.size());
37  assert(j < (uint32_t)z_columns_[i].z.size());
38  int x = z_columns_[i].x;
39  int y = z_columns_[i].y;
40  int z = z_columns_[i].z[j];
41  return vector3d<int>(x, y, z);
42 }

44 void sddk::Gvec::find_z_columns(double Gmax__, const FFT3D_grid& fft_box__)
45 {
46  mdarray<int, 2> non_zero_columns(fft_box__.limits(0), fft_box__.limits(1));
47  non_zero_columns.zero();
48 
49  num_gvec_ = 0;
50 
51  auto add_new_column = [&](int i, int j) {
52  std::vector<int> zcol;
53 
54  /* in general case take z in [0, Nz) */
55  int zmax = fft_box__[2] - 1;
56  /* in case of G-vector reduction take z in [0, Nz/2] for {x=0,y=0} stick */
57  if (reduce_gvec_ && !i && !j) {
58  zmax = fft_box__.limits(2).second;
59  }
60  /* loop over z-coordinates of FFT grid */
61  for (int iz = 0; iz <= zmax; iz++) {
62  /* get z-coordinate of G-vector */
63  int k = fft_box__.freq_by_coord<2>(iz);
64  /* take G+k */
65  auto vgk = dot(lattice_vectors_, (vector3d<double>(i, j, k) + vk_));
66  /* add z-coordinate of G-vector to the list */
67  if (vgk.length() <= Gmax__) {
68  zcol.push_back(k);
69  }
70  }
71 
72  /* add column to the list */
73  if (zcol.size() && !non_zero_columns(i, j)) {
74  z_columns_.push_back(z_column_descriptor(i, j, zcol));
75  num_gvec_ += static_cast<int>(zcol.size());
76 
77  non_zero_columns(i, j) = 1;
78  if (reduce_gvec_) {
79  int mi = -i;
80  int mj = -j;
81  if (mi >= fft_box__.limits(0).first && mi <= fft_box__.limits(0).second &&
82  mj >= fft_box__.limits(1).first && mj <= fft_box__.limits(1).second) {
83  non_zero_columns(mi, mj) = 1;
84  }
85  }
86  }
87  };
88 
89  /* copy column order from previous G-vector set */
90  if (gvec_base_) {
91  for (int icol = 0; icol < gvec_base_->num_zcol(); icol++) {
92  int i = gvec_base_->zcol(icol).x;
93  int j = gvec_base_->zcol(icol).y;
94  add_new_column(i, j);
95  }
96  }
97 
98  // Check all z-columns and add if within sphere. Only allow non-negative x-indices for reduced case
99  for (int i = reduce_gvec_? 0 : fft_box__.limits(0).first; i <= fft_box__.limits(0).second; i++) {
100  for (int j = fft_box__.limits(1).first; j <= fft_box__.limits(1).second; j++) {
101  add_new_column(i, j);
102  }
103  }
104 
105  if (!gvec_base_) {
106  /* put column with {x, y} = {0, 0} to the beginning */
107  for (size_t i = 0; i < z_columns_.size(); i++) {
108  if (z_columns_[i].x == 0 && z_columns_[i].y == 0) {
109  std::swap(z_columns_[i], z_columns_[0]);
110  break;
111  }
112  }
113  }
114 
115  /* sort z-columns starting from the second or skip num_zcol of base distribution */
116  int n = (gvec_base_) ? gvec_base_->num_zcol() : 1;
117  std::sort(z_columns_.begin() + n, z_columns_.end(),
118  [](z_column_descriptor const& a, z_column_descriptor const& b) { return a.z.size() > b.z.size(); });
119 }

121 void Gvec::distribute_z_columns()
122 {
123  gvec_distr_ = block_data_descriptor(comm().size());
124  zcol_distr_ = block_data_descriptor(comm().size());
125  /* local number of z-columns for each rank */
126  std::vector<std::vector<z_column_descriptor>> zcols_local(comm().size());
127 
128  /* use already existing distribution of base G-vector set */
129  if (gvec_base_) {
130  for (int rank = 0; rank < comm().size(); rank++) {
131  for (int i = 0; i < gvec_base_->zcol_count(rank); i++) {
132  int icol = gvec_base_->zcol_offset(rank) + i;
133  /* assign column to the found rank */
134  zcols_local[rank].push_back(z_columns_[icol]);
135  /* count local number of z-columns */
136  zcol_distr_.counts[rank] += 1;
137  /* count local number of G-vectors */
138  gvec_distr_.counts[rank] += static_cast<int>(z_columns_[icol].z.size());
139  }
140  }
141  }
142 
143  int n = (gvec_base_) ? gvec_base_->num_zcol() : 0;
144 
145  std::vector<int> ranks;
146  for (int i = n; i < static_cast<int>(z_columns_.size()); i++) {
147  /* initialize the list of ranks to 0,1,2,... */
148  if (ranks.empty()) {
149  ranks.resize(comm().size());
150  std::iota(ranks.begin(), ranks.end(), 0);
151  }
152  /* find rank with minimum number of G-vectors */
153  auto rank_with_min_gvec = std::min_element(ranks.begin(), ranks.end(), [this](const int& a, const int& b) {
154  return gvec_distr_.counts[a] < gvec_distr_.counts[b];
155  });
156 
157  /* assign column to the found rank */
158  zcols_local[*rank_with_min_gvec].push_back(z_columns_[i]);
159  /* count local number of z-columns */
160  zcol_distr_.counts[*rank_with_min_gvec] += 1;
161  /* count local number of G-vectors */
162  gvec_distr_.counts[*rank_with_min_gvec] += static_cast<int>(z_columns_[i].z.size());
163  /* exclude this rank from the search */
164  ranks.erase(rank_with_min_gvec);
165  }
166  gvec_distr_.calc_offsets();
167  zcol_distr_.calc_offsets();
168 
169  /* save new ordering of z-columns */
170  z_columns_.clear();
171  for (int rank = 0; rank < comm().size(); rank++) {
172  z_columns_.insert(z_columns_.end(), zcols_local[rank].begin(), zcols_local[rank].end());
173  }
174 
175  /* sanity check */
176  int ng{0};
177  for (int rank = 0; rank < comm().size(); rank++) {
178  ng += gvec_distr_.counts[rank];
179  }
180  if (ng != num_gvec_) {
181  throw std::runtime_error("wrong number of G-vectors");
182  }
183 }

185 void Gvec::find_gvec_shells()
186 {
187  if (!bare_gvec_) {
188  return;
189  }
190 
191  PROFILE("sddk::Gvec::find_gvec_shells");
192 
193  auto lat_sym = sirius::find_lat_sym(lattice_vectors_, 1e-6);
194 
195  num_gvec_shells_ = 0;
196  gvec_shell_ = sddk::mdarray<int, 1>(num_gvec_);
197 
198  std::fill(&gvec_shell_[0], &gvec_shell_[0] + num_gvec_, -1);
199 
200  /* find G-vector shells using symmetry consideration */
201  for (int ig = 0; ig < num_gvec_; ig++) {
202  if (gvec_shell_[ig] == -1) {
203  auto G = gvec(ig);
204  for (auto& R: lat_sym) {
205  auto G1 = dot(R, G);
206  auto ig1 = index_by_gvec(G1);
207  if (ig1 == -1) {
208  ig1 = index_by_gvec(G1 * (-1));
209  if (ig1 == -1) {
210  throw std::runtime_error("[sddk::Gvec] symmetry-related G-vector is not found");
211  }
212  }
213  gvec_shell_[ig1] = num_gvec_shells_;
214  }
215  num_gvec_shells_++;
216  }
217  }
218  for (int ig = 0; ig < num_gvec_; ig++) {
219  if (gvec_shell_[ig] == -1) {
220  throw std::runtime_error("[sddk::Gvec] wrong G-vector shell");
221  }
222  }
223 
224  gvec_shell_len_ = mdarray<double, 1>(num_gvec_shells_);
225  std::fill(&gvec_shell_len_[0], &gvec_shell_len_[0] + num_gvec_shells_, -1);
226 
227  for (int ig = 0; ig < num_gvec_; ig++) {
228  auto g = gvec_cart<index_domain_t::global>(ig).length();
229  int igsh = gvec_shell_[ig];
230  if (gvec_shell_len_[igsh] < 0) {
231  gvec_shell_len_[igsh] = g;
232  } else {
233  /* lattice symmetries were found with 1e-6 tolererance for the metric tensor,
234  so tolerance on length should be square root of that */
235  if (std::abs(gvec_shell_len_[igsh] - g) > 1e-3) {
236  std::stringstream s;
237  s << "[sddk::Gvec] wrong G-vector length\n"
238  << " length of G-shell : " << gvec_shell_len_[igsh] << "\n"
239  << " length of current G-vector: " << g << "\n"
240  << " index of G-vector: " << ig << "\n"
241  << " index of G-shell: " << igsh << "\n"
242  << " length difference: " << std::abs(gvec_shell_len_[igsh] - g);
243  throw std::runtime_error(s.str());
244  }
245  }
246  }
247 
248  // TODO: maybe, make an average G-shell length.
249 
250  /* list of pairs (length, index of G-vector) */
251  std::vector<std::pair<uint64_t, int>> tmp(num_gvec_);
252  #pragma omp parallel for schedule(static)
253  for (int ig = 0; ig < num_gvec(); ig++) {
254  /* make some reasonable roundoff */
255  uint64_t len = static_cast<uint64_t>(gvec_shell_len_[gvec_shell_[ig]] * 1e10);
256  tmp[ig] = std::make_pair(len, ig);
257  }
258  /* sort by first element in pair (length) */
259  std::sort(tmp.begin(), tmp.end());
260 
261  /* index of the first shell */
262  gvec_shell_(tmp[0].second) = 0;
263  num_gvec_shells_ = 1;
264  /* temporary vector to store G-shell radius */
265  std::vector<double> tmp_len;
266  /* radius of the first shell */
267  tmp_len.push_back(static_cast<double>(tmp[0].first) * 1e-10);
268  for (int ig = 1; ig < num_gvec_; ig++) {
269  /* if this G-vector has a different length */
270  if (tmp[ig].first != tmp[ig - 1].first) {
271  /* increment number of shells */
272  num_gvec_shells_++;
273  /* save the radius of the new shell */
274  tmp_len.push_back(static_cast<double>(tmp[ig].first) * 1e-10);
275  }
276  /* assign the index of the current shell */
277  gvec_shell_(tmp[ig].second) = num_gvec_shells_ - 1;
278  }
279  gvec_shell_len_ = mdarray<double, 1>(num_gvec_shells_);
280  std::copy(tmp_len.begin(), tmp_len.end(), gvec_shell_len_.at(memory_t::host));
281 
282  /* map from global index of G-shell to a list of local G-vectors */
283  std::map<int, std::vector<int>> gshmap;
284  for (int igloc = 0; igloc < this->count(); igloc++) {
285  int igsh = this->shell(this->offset() + igloc);
286  if (gshmap.count(igsh) == 0) {
287  gshmap[igsh] = std::vector<int>();
288  }
289  gshmap[igsh].push_back(igloc);
290  }
291 
292  num_gvec_shells_local_ = 0;
293  gvec_shell_idx_local_.resize(this->count());
294  gvec_shell_len_local_.clear();
295  for (auto it = gshmap.begin(); it != gshmap.end(); ++it) {
296  int igsh = it->first;
297  gvec_shell_len_local_.push_back(this->shell_len(igsh));
298  for (auto igloc: it->second) {
299  gvec_shell_idx_local_[igloc] = num_gvec_shells_local_;
300  }
301  num_gvec_shells_local_++;
302  }
303 }

305 void Gvec::init_gvec_cart()
306 {
307  gvec_cart_ = mdarray<double, 2>(3, count());
308  gkvec_cart_ = mdarray<double, 2>(3, count());
309 
310  for (int igloc = 0; igloc < count(); igloc++) {
311  int ig = offset() + igloc;
312  auto G = gvec_by_full_index(gvec_full_index_(ig));
313  auto gc = dot(lattice_vectors_, vector3d<double>(G[0], G[1], G[2]));
314  auto gkc = dot(lattice_vectors_, (vector3d<double>(G[0], G[1], G[2]) + vk_));
315  for (int x : {0, 1, 2}) {
316  gvec_cart_(x, igloc) = gc[x];
317  gkvec_cart_(x, igloc) = gkc[x];
318  }
319  }

322 void Gvec::init(FFT3D_grid const& fft_grid)
323 {
324  PROFILE("sddk::Gvec::init");
325 
326  find_z_columns(Gmax_, fft_grid);
327 
328  distribute_z_columns();
329 
330  gvec_index_by_xy_ =
331  mdarray<int, 3>(2, fft_grid.limits(0), fft_grid.limits(1), memory_t::host, "Gvec.gvec_index_by_xy_");
332  std::fill(gvec_index_by_xy_.at(memory_t::host), gvec_index_by_xy_.at(memory_t::host) + gvec_index_by_xy_.size(),
333  -1);
334 
335  /* build the full G-vector index and reverse mapping */
336  gvec_full_index_ = mdarray<uint32_t, 1>(num_gvec_);
337  int ig{0};
338  for (size_t i = 0; i < z_columns_.size(); i++) {
339  /* starting G-vector index for a z-stick */
340  gvec_index_by_xy_(0, z_columns_[i].x, z_columns_[i].y) = ig;
341  /* pack size of a z-stick and column index in one number */
342  gvec_index_by_xy_(1, z_columns_[i].x, z_columns_[i].y) = static_cast<int>((z_columns_[i].z.size() << 20) + i);
343  for (size_t j = 0; j < z_columns_[i].z.size(); j++) {
344  gvec_full_index_[ig++] = static_cast<uint32_t>((i << 12) + j);
345  }
346  }
347  if (ig != num_gvec_) {
348  throw std::runtime_error("wrong G-vector count");
349  }
350  for (int ig = 0; ig < num_gvec_; ig++) {
351  auto gv = gvec(ig);
352  if (index_by_gvec(gv) != ig) {
353  std::stringstream s;
354  s << "wrong G-vector index: ig=" << ig << " gv=" << gv << " index_by_gvec(gv)=" << index_by_gvec(gv);
355  throw std::runtime_error(s.str());
356  }
357  }
358 
359  /* first G-vector must be (0, 0, 0); never remove this check!!! */
360  auto g0 = gvec_by_full_index(gvec_full_index_(0));
361  if (g0[0] || g0[1] || g0[2]) {
362  throw std::runtime_error("first G-vector is not zero");
363  }
364 
365  init_gvec_cart();
366 
367  find_gvec_shells();
368 
369  if (gvec_base_) {
370  /* the size of the mapping is equal to the local number of G-vectors in the base set */
371  gvec_base_mapping_ = mdarray<int, 1>(gvec_base_->count());
372  /* loop over local G-vectors of a base set */
373  for (int igloc = 0; igloc < gvec_base_->count(); igloc++) {
374  /* G-vector in lattice coordinates */
375  auto G = gvec_base_->gvec(gvec_base_->offset() + igloc);
376  /* global index of G-vector in the current set */
377  int ig = index_by_gvec(G);
378  /* the same MPI rank must store this G-vector */
379  ig -= offset();
380  if (ig >= 0 && ig < count()) {
381  gvec_base_mapping_(igloc) = ig;
382  } else {
383  std::stringstream s;
384  s << "local G-vector index is not found" << std::endl
385  << " G-vector: " << G << std::endl
386  << " G-vector index in base distribution : " << gvec_base_->offset() + igloc << std::endl
387  << " G-vector index in base distribution (by G-vector): " << gvec_base_->index_by_gvec(G) << std::endl
388  << " G-vector index in new distribution : " << index_by_gvec(G) << std::endl
389  << " offset in G-vector index for this rank: " << offset() << std::endl
390  << " local number of G-vectors for this rank: " << count();
391  throw std::runtime_error(s.str());
392  }
393  }
394  }
395  // TODO: add a check for gvec_base (there is already a test for this).
396 }

420 std::pair<int, bool> Gvec::index_g12_safe(vector3d<int> const& g1__, vector3d<int> const& g2__) const
421 {
422  auto v = g1__ - g2__;
423  int idx = index_by_gvec(v);
424  bool conj{false};
425  if (idx < 0) {
426  idx = index_by_gvec(v * (-1));
427  conj = true;
428  }
429  if (idx < 0 || idx >= num_gvec()) {
430  std::stringstream s;
431  s << "wrong index of G-G' vector" << std::endl
432  << " G: " << g1__ << std::endl
433  << " G': " << g2__ << std::endl
434  << " G - G': " << v << std::endl
435  << " idx: " << idx;
436  throw std::runtime_error(s.str());
437  }
438  return std::make_pair(idx, conj);
439 }

441 int Gvec::index_by_gvec(vector3d<int> const& G__) const
442 {
443  /* reduced G-vector set does not have negative z for x=y=0 */
444  if (reduced() && G__[0] == 0 && G__[1] == 0 && G__[2] < 0) {
445  return -1;
446  }
447  int ig0 = gvec_index_by_xy_(0, G__[0], G__[1]);
448  if (ig0 == -1) {
449  return -1;
450  }
451  /* index of the column */
452  int icol = gvec_index_by_xy_(1, G__[0], G__[1]) & 0xFFFFF;
453  /* size of the column */
454  int col_size = gvec_index_by_xy_(1, G__[0], G__[1]) >> 20;
455 
456  /* three possible options for the z-column location
457 
458  frequency ... -4, -3, -2, -1, 0, 1, 2, 3, 4 ...
459  -----------------------------------------------------------------------------
460  G-vec ordering
461  #1 (all negative) ___ 0 1 2 3 __________________
462  #2 (negative and positive) ____________ 3 4 0 1 2 _________
463  #3 (all positive) _____________________ 0 1 2 3 ___
464 
465  Remember how FFT frequencies are stored: first positive frequences, then negative in the reverse order
466 
467  subtract first z-coordinate in column from the current z-coordinate of G-vector: in case #1 or #3 this
468  already gives a proper offset, in case #2 storage of FFT frequencies must be taken into account
469  */
470  int z0 = G__[2] - z_columns_[icol].z[0];
471  /* calculate proper offset */
472  int offs = (z0 >= 0) ? z0 : z0 + col_size;
473  /* full index */
474  int ig = ig0 + offs;
475  assert(ig >= 0 && ig < num_gvec());
476  return ig;
477 }

532 void Gvec_partition::build_fft_distr()
533 {
534  /* calculate distribution of G-vectors and z-columns for the FFT communicator */
535  gvec_distr_fft_ = block_data_descriptor(fft_comm().size());
536  zcol_distr_fft_ = block_data_descriptor(fft_comm().size());
537 
538  for (int rank = 0; rank < fft_comm().size(); rank++) {
539  for (int i = 0; i < comm_ortho_fft().size(); i++) {
540  /* fine-grained rank */
541  int r = rank_map_(rank, i);
542  gvec_distr_fft_.counts[rank] += gvec().gvec_count(r);
543  zcol_distr_fft_.counts[rank] += gvec().zcol_count(r);
544  }
545  }
546  /* get offsets of z-columns */
547  zcol_distr_fft_.calc_offsets();
548  /* get offsets of G-vectors */
549  gvec_distr_fft_.calc_offsets();

552 void Gvec_partition::calc_offsets()
553 {
554  zcol_offs_ = mdarray<int, 1>(gvec().num_zcol(), memory_t::host, "Gvec_partition.zcol_offs_");
555  for (int rank = 0; rank < fft_comm().size(); rank++) {
556  int offs{0};
557  /* loop over local number of z-columns */
558  for (int i = 0; i < zcol_count_fft(rank); i++) {
559  /* global index of z-column */
560  int icol = idx_zcol_[zcol_distr_fft_.offsets[rank] + i];
561  zcol_offs_[icol] = offs;

568 void Gvec_partition::pile_gvec()
569 {
570  /* build a table of {offset, count} values for G-vectors in the swapped distribution;
571  * we are preparing to swap plane-wave coefficients from a default slab distribution to a FFT-friendly
572  * distribution
573  * +--------------+ +----+----+----+
574  * | : : | | | | |
575  * +--------------+ |....|....|....|
576  * | : : | -> | | | |
577  * +--------------+ |....|....|....|
578  * | : : | | | | |
579  * +--------------+ +----+----+----+
580  *
581  * i.e. we will make G-vector slabs more fat (pile-of-slabs) and at the same time reshulffle wave-functions
582  * between columns of the 2D MPI grid */
583  gvec_fft_slab_ = block_data_descriptor(comm_ortho_fft_.size());
584  for (int i = 0; i < comm_ortho_fft_.size(); i++) {
585  gvec_fft_slab_.counts[i] = gvec().gvec_count(rank_map_(fft_comm_.rank(), i));
586  }
587  gvec_fft_slab_.calc_offsets();
588 
589  assert(gvec_fft_slab_.offsets.back() + gvec_fft_slab_.counts.back() == gvec_distr_fft_.counts[fft_comm().rank()]);
590 }

592 Gvec_partition::Gvec_partition(Gvec const& gvec__, Communicator const& fft_comm__, Communicator const& comm_ortho_fft__)
593  : gvec_(gvec__)
594  , fft_comm_(fft_comm__)
595  , comm_ortho_fft_(comm_ortho_fft__)
596 {
597  if (fft_comm_.size() * comm_ortho_fft_.size() != gvec_.comm().size()) {
598  std::stringstream s;
599  s << "[sddk::Gvec] wrong size of communicators" << std::endl
600  << " fft_comm_.size() = " << fft_comm_.size() << std::endl
601  << " comm_ortho_fft_.size() = " << comm_ortho_fft_.size() << std::endl
602  << " gvec_.comm().size() = " << gvec_.comm().size();
603  throw std::runtime_error(s.str());
604  }
605  rank_map_ = mdarray<int, 2>(fft_comm_.size(), comm_ortho_fft_.size());
606  rank_map_.zero();
607  /* get a global rank */
608  rank_map_(fft_comm_.rank(), comm_ortho_fft_.rank()) = gvec_.comm().rank();
609  gvec_.comm().allreduce(&rank_map_(0, 0), gvec_.comm().size());
610 
611  build_fft_distr();
612 
613  idx_zcol_ = mdarray<int, 1>(gvec().num_zcol());
614  int icol{0};
615  for (int rank = 0; rank < fft_comm().size(); rank++) {
616  for (int i = 0; i < comm_ortho_fft().size(); i++) {
617  for (int k = 0; k < gvec_.zcol_count(rank_map_(rank, i)); k++) {
618  idx_zcol_(icol) = gvec_.zcol_offset(rank_map_(rank, i)) + k;
619  icol++;
620  }
621  }
622  assert(icol == zcol_distr_fft_.counts[rank] + zcol_distr_fft_.offsets[rank]);
623  }
624  assert(icol == gvec().num_zcol());
625 
626  idx_gvec_ = mdarray<int, 1>(gvec_count_fft());
627  int ig{0};
628  for (int i = 0; i < comm_ortho_fft_.size(); i++) {
629  for (int k = 0; k < gvec_.gvec_count(rank_map_(fft_comm_.rank(), i)); k++) {
630  idx_gvec_[ig] = gvec_.gvec_offset(rank_map_(fft_comm_.rank(), i)) + k;
631  ig++;
632  }
633  }
634 
635  calc_offsets();
636  pile_gvec();
637 }

670 Gvec_shells::Gvec_shells(Gvec const& gvec__)
671  : comm_(gvec__.comm())
672  , gvec_(gvec__)
673 {
674  PROFILE("sddk::Gvec_shells");
675 
676  a2a_send_ = block_data_descriptor(comm_.size());
677  a2a_recv_ = block_data_descriptor(comm_.size());
678 
679  /* split G-vector shells between ranks in cyclic order */
680  spl_num_gsh_ = splindex<splindex_t::block_cyclic>(gvec_.num_shells(), comm_.size(), comm_.rank(), 1);
681 
682  /* each rank sends a fraction of its local G-vectors to other ranks */
683  /* count this fraction */
684  for (int igloc = 0; igloc < gvec_.count(); igloc++) {
685  int ig = gvec_.offset() + igloc;
686  int igsh = gvec_.shell(ig);
687  a2a_send_.counts[spl_num_gsh_.local_rank(igsh)]++;
688  }
689  a2a_send_.calc_offsets();
690  /* sanity check: total number of elements to send is equal to the local number of G-vector */
691  if (a2a_send_.size() != gvec_.count()) {
692  throw std::runtime_error("wrong number of G-vectors");
693  }
694  /* count the number of elements to receive */
695  for (int r = 0; r < comm_.size(); r++) {
696  for (int igloc = 0; igloc < gvec_.gvec_count(r); igloc++) {
697  /* index of G-vector in the original distribution */
698  int ig = gvec_.gvec_offset(r) + igloc;
699  /* index of the G-vector shell */
700  int igsh = gvec_.shell(ig);
701  if (spl_num_gsh_.local_rank(igsh) == comm_.rank()) {
702  a2a_recv_.counts[r]++;
703  }
704  }
705  }
706  a2a_recv_.calc_offsets();
707  /* sanity check: sum of local sizes in the remapped order is equal to the total number of G-vectors */
708  int ng = gvec_count_remapped();
709  comm_.allreduce(&ng, 1);
710  if (ng != gvec_.num_gvec()) {
711  throw std::runtime_error("wrong number of G-vectors");
712  }
713 
714  /* local set of G-vectors in the remapped order */
715  gvec_remapped_ = sddk::mdarray<int, 2>(3, gvec_count_remapped());
716  gvec_shell_remapped_ = sddk::mdarray<int, 1>(gvec_count_remapped());
717  std::vector<int> counts(comm_.size(), 0);
718  for (int r = 0; r < comm_.size(); r++) {
719  for (int igloc = 0; igloc < gvec_.gvec_count(r); igloc++) {
720  int ig = gvec_.gvec_offset(r) + igloc;
721  int igsh = gvec_.shell(ig);
722  auto G = gvec_.gvec(ig);
723  if (spl_num_gsh_.local_rank(igsh) == comm_.rank()) {
724  for (int x : {0, 1, 2}) {
725  gvec_remapped_(x, a2a_recv_.offsets[r] + counts[r]) = G[x];
726  }
727  gvec_shell_remapped_(a2a_recv_.offsets[r] + counts[r]) = igsh;
728  counts[r]++;
729  }
730  }
731  }
732  for (int ig = 0; ig < gvec_count_remapped(); ig++) {
733  idx_gvec_[gvec_remapped(ig)] = ig;
734  }
735  /* sanity check */
736  for (int igloc = 0; igloc < this->gvec_count_remapped(); igloc++) {
737  auto G = this->gvec_remapped(igloc);
738  if (this->index_by_gvec(G) != igloc) {
739  throw std::runtime_error("Wrong remapped index of G-vector");
740  }
741  int igsh = this->gvec_shell_remapped(igloc);
742  if (igsh != this->gvec().shell(G)) {
743  throw std::runtime_error("Wrong remapped shell of G-vector");
744  }
745  }
746 }
