52 void Occupation_matrix::add_k_point_contribution(K_point& kp__)
53 {
54  if (!ctx_.hubbard_correction()) {
55  return;
56  }
57 
58  memory_t mem_host{memory_t::host};
59  memory_t mem{memory_t::host};
60  linalg_t la{linalg_t::blas};
61  /* find the appropriate linear algebra provider */
62  if (ctx_.processing_unit() == device_t::GPU) {
63  mem = memory_t::device;
64  mem_host = memory_t::host_pinned;
65  if (is_device_memory(ctx_.preferred_memory_t())) {
66  la = linalg_t::gpublas;
67  } else {
68  la = linalg_t::spla;
69  }
70  }
71 
72  int nwfu = kp__.hubbard_wave_functions().num_wf();
73 
74  sddk::matrix<double_complex> occ_mtrx(nwfu, nwfu, ctx_.mem_pool(memory_t::host), "occ_mtrx");
75  if (is_device_memory(mem)) {
76  occ_mtrx.allocate(ctx_.mem_pool(mem));
77  }
78 
79  auto r = ctx_.unit_cell().num_hubbard_wf();
80 
81  // TODO collnear and non-collinear cases have a lot of similar code; there should be a way to combine it
82 
83  /* full non colinear magnetism */
84  if (ctx_.num_mag_dims() == 3) {
85  dmatrix<double_complex> dm(kp__.num_occupied_bands(), nwfu, ctx_.mem_pool(mem_host), "dm");
86  if (is_device_memory(mem)) {
87  dm.allocate(ctx_.mem_pool(mem));
88  }
89  sddk::inner(ctx_.spla_context(), spin_range(2), kp__.spinor_wave_functions(), 0,
90  kp__.num_occupied_bands(), kp__.hubbard_wave_functions(), 0, nwfu, dm, 0, 0);
91 
92  // TODO: check if inner() already moved data to CPU
93 
94  //if (is_device_memory(mem)) {
95  // dm.copy_to(memory_t::host);
96  //}
97  dmatrix<double_complex> dm1(kp__.num_occupied_bands(), nwfu, ctx_.mem_pool(mem_host), "dm1");
98  #pragma omp parallel for
99  for (int m = 0; m < nwfu; m++) {
100  for (int j = 0; j < kp__.num_occupied_bands(); j++) {
101  dm1(j, m) = dm(j, m) * kp__.band_occupancy(j, 0);
102  }
103  }
104  if (is_device_memory(mem)) {
105  dm1.allocate(ctx_.mem_pool(mem)).copy_to(mem);
106  }
107 
108  /* now compute O_{ij}^{sigma,sigma'} = \sum_{nk} <psi_nk|phi_{i,sigma}><phi_{j,sigma^'}|psi_nk> f_{nk} */
109  auto alpha = double_complex(kp__.weight(), 0.0);
110  linalg(la).gemm('C', 'N', nwfu, nwfu, kp__.num_occupied_bands(), &alpha, dm.at(mem), dm.ld(),
111  dm1.at(mem), dm1.ld(), &linalg_const<double_complex>::zero(), occ_mtrx.at(mem), occ_mtrx.ld());
112  if (is_device_memory(mem)) {
113  occ_mtrx.copy_to(memory_t::host);
114  }
115 
116  #pragma omp parallel for schedule(static)
117  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
118  auto const& atom = ctx_.unit_cell().atom(ia);
119  if (atom.type().hubbard_correction()) {
120 
121  /* loop over the different channels */
122  /* note that for atom with SO interactions, we need to jump
123  by 2 instead of 1. This is due to the fact that the
124  relativistic wave functions have different total angular
125  momentum for the same n */
126 
127  //TODO: multi-orbital case
128  //for (int orb = 0; orb < atom.type().num_hubbard_orbitals(); orb += (atom.type().spin_orbit_coupling() ? 2 : 1)) {
129  /*
130  I know that the index of the hubbard wave functions (indexb_....) is
131  consistent with the index of the hubbard orbitals
132  */
133  const int lmmax_at = 2 * atom.type().indexr_hub()[0].aqn.l() + 1;
134  for (int s1 = 0; s1 < ctx_.num_spins(); s1++) {
135  for (int s2 = 0; s2 < ctx_.num_spins(); s2++) {
136  int s = (s1 == s2) * s1 + (s1 != s2) * (1 + 2 * s2 + s1);
137  for (int mp = 0; mp < lmmax_at; mp++) {
138  for (int m = 0; m < lmmax_at; m++) {
139  data_(m, mp, s, ia) +=
140  occ_mtrx(r.first * s1 + r.second[ia] + m, r.first * s2 + r.second[ia] + mp);
141  }
142  }
143  }
144  }
145  //}
146  }
147  }
148  } else {
149  /* SLDA + U, we need to do the explicit calculation. The hubbard
150  orbitals only have one component while the bloch wave functions
151  have two. The inner product takes care of this case internally. */
152 
153  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
154  PROFILE_START("sirius::Hubbard::compute_occupation_matrix|1");
155  dmatrix<double_complex> dm(kp__.num_occupied_bands(ispn), nwfu, ctx_.mem_pool(mem_host), "dm");
156  if (is_device_memory(mem)) {
157  dm.allocate(ctx_.mem_pool(mem));
158  }
159  sddk::inner(ctx_.spla_context(), spin_range(ispn), kp__.spinor_wave_functions(), 0, kp__.num_occupied_bands(ispn),
160  kp__.hubbard_wave_functions(), 0, nwfu, dm, 0, 0);
161  // TODO: check if inner() already moved data to CPU
162 
163  //if (is_device_memory(mem)) {
164  // dm.copy_to(memory_t::host);
165  //}
166  PROFILE_STOP("sirius::Hubbard::compute_occupation_matrix|1");
167 
168  PROFILE_START("sirius::Hubbard::compute_occupation_matrix|2");
169  dmatrix<double_complex> dm1(kp__.num_occupied_bands(ispn), nwfu, ctx_.mem_pool(mem_host), "dm1");
170  #pragma omp parallel for
171  for (int m = 0; m < nwfu; m++) {
172  for (int j = 0; j < kp__.num_occupied_bands(ispn); j++) {
173  dm1(j, m) = dm(j, m) * kp__.band_occupancy(j, ispn);
174  }
175  }
176  if (is_device_memory(mem)) {
177  dm1.allocate(ctx_.mem_pool(mem)).copy_to(mem);
178  }
179  PROFILE_STOP("sirius::Hubbard::compute_occupation_matrix|2");
180  /* now compute O_{ij}^{sigma,sigma'} = \sum_{nk} <psi_nk|phi_{i,sigma}><phi_{j,sigma^'}|psi_nk> f_{nk} */
181  /* We need to apply a factor 1/2 when we compute the occupancies for the LDA+U. It is because the
182  * calculations of E and U consider occupancies <= 1. Sirius for the LDA+U has a factor 2 in the
183  * band occupancies. We need to compensate for it because it is taken into account in the
184  * calculation of the hubbard potential */
185  PROFILE_START("sirius::Hubbard::compute_occupation_matrix|3");
186  auto alpha = double_complex(kp__.weight() / ctx_.max_occupancy(), 0.0);
187  linalg(la).gemm('C', 'N', nwfu, nwfu, kp__.num_occupied_bands(ispn), &alpha, dm.at(mem), dm.ld(),
188  dm1.at(mem), dm1.ld(), &linalg_const<double_complex>::zero(), occ_mtrx.at(mem), occ_mtrx.ld());
189  if (is_device_memory(mem)) {
190  occ_mtrx.copy_to(memory_t::host);
191  }
192  PROFILE_STOP("sirius::Hubbard::compute_occupation_matrix|3");
193  PROFILE_START("sirius::Hubbard::compute_occupation_matrix|4");
194  #pragma omp parallel for schedule(static)
195  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
196  const auto& atom = ctx_.unit_cell().atom(ia);
197  if (atom.type().hubbard_correction()) {
198  //for (int orb = 0; orb < atom.type().num_hubbard_orbitals(); orb++) {
199  const int lmmax_at = 2 * atom.type().indexr_hub()[0].aqn.l() + 1;
200  for (int mp = 0; mp < lmmax_at; mp++) {
201  const int mmp = r.second[ia] + mp;
202  for (int m = 0; m < lmmax_at; m++) {
203  const int mm = r.second[ia] + m;
204  data_(m, mp, ispn, ia) += occ_mtrx(mm, mmp);
205  }
206  }
207  //}
208  }
209  }
210  PROFILE_STOP("sirius::Hubbard::compute_occupation_matrix|4");
211  } // ispn
212  }

215 void Occupation_matrix::access(std::string const& what__, double_complex* occ__, int ld__) {
216  if (!(what__ == "get" || what__ == "set")) {
217  std::stringstream s;
218  s << "wrong access label: " << what__;
219  TERMINATE(s);
220  }
221 
222  sddk::mdarray<double_complex, 4> occ_mtrx;
223  /* in non-collinear case the occupancy matrix is complex */
224  if (ctx_.num_mag_dims() == 3) {
225  occ_mtrx = sddk::mdarray<double_complex, 4>(occ__, ld__, ld__, 4, ctx_.unit_cell().num_atoms());
226  } else {
227  occ_mtrx = sddk::mdarray<double_complex, 4>(occ__, ld__, ld__, ctx_.num_spins(), ctx_.unit_cell().num_atoms());
228  }
229  if (what__ == "get") {
230  occ_mtrx.zero();
231  }
232 
233  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
234  auto& atom = ctx_.unit_cell().atom(ia);
235  if (atom.type().hubbard_correction()) {
236  const int l = ctx_.unit_cell().atom(ia).type().indexr_hub()[0].aqn.l();
237  for (int m1 = -l; m1 <= l; m1++) {
238  for (int m2 = -l; m2 <= l; m2++) {
239  if (what__ == "get") {
240  for (int j = 0; j < ((ctx_.num_mag_dims() == 3) ? 4 : ctx_.num_spins()); j++) {
241  occ_mtrx(l + m1, l + m2, j, ia) = data_(l + m1, l + m2, j, ia);
242  }
243  } else {
244  for (int j = 0; j < ((ctx_.num_mag_dims() == 3) ? 4 : ctx_.num_spins()); j++) {
245  data_(l + m1, l + m2, j, ia) = occ_mtrx(l + m1, l + m2, j, ia);
246  }
247  }
248  }
249  }
250  }

254 void Occupation_matrix::init()
255 {
256  if (!data_.size()) {
257  return;
258  }
259 
260  data_.zero();
261  #pragma omp parallel for schedule(static)
262  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
263  const auto& atom = ctx_.unit_cell().atom(ia);
264  if (atom.type().hubbard_correction()) {
265  const int lmax_at = 2 * atom.type().indexr_hub()[0].aqn.l() + 1;
266  if (atom.type().lo_descriptor_hub(0).initial_occupancy.size()) {
267  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
268  for (int m = 0; m < lmax_at; m++) {
269  this->data_(m, m, ispn, ia) = atom.type().lo_descriptor_hub(0).initial_occupancy[m + ispn * lmax_at];
270  }
271  }
272  } else {
273  // compute the total charge for the hubbard orbitals
274  double charge = atom.type().lo_descriptor_hub(0).occupancy();
275  bool nm = true; // true if the atom is non magnetic
276  int majs, mins;
277  if (ctx_.num_spins() != 1) {
278  if (atom.vector_field()[2] > 0.0) {
279  nm = false;
280  majs = 0;
281  mins = 1;
282  } else if (atom.vector_field()[2] < 0.0) {
283  nm = false;
284  majs = 1;
285  mins = 0;
286  }
287  }
288 
289  if (!nm) {
290  if (ctx_.num_mag_dims() != 3) {
291  // colinear case
292  if (charge > (lmax_at)) {
293  for (int m = 0; m < lmax_at; m++) {
294  this->data_(m, m, majs, ia) = 1.0;
295  this->data_(m, m, mins, ia) =
296  (charge - static_cast<double>(lmax_at)) / static_cast<double>(lmax_at);
297  }
298  } else {
299  for (int m = 0; m < lmax_at; m++) {
300  data_(m, m, majs, ia) = charge / static_cast<double>(lmax_at);
301  }
302  }
303  } else {
304  // double c1, s1;
305  // sincos(atom.type().starting_magnetization_theta(), &s1, &c1);
306  double c1 = atom.vector_field()[2];
307  double_complex cs = double_complex(atom.vector_field()[0], atom.vector_field()[1]) / sqrt(1.0 - c1 * c1);
308  double_complex ns[4];
309 
310  if (charge > (lmax_at)) {
311  ns[majs] = 1.0;
312  ns[mins] = (charge - static_cast<double>(lmax_at)) / static_cast<double>(lmax_at);
313  } else {
314  ns[majs] = charge / static_cast<double>(lmax_at);
315  ns[mins] = 0.0;
316  }
317 
318  // charge and moment
319  double nc = ns[majs].real() + ns[mins].real();
320  double mag = ns[majs].real() - ns[mins].real();
321 
322  // rotate the occ matrix
323  ns[0] = (nc + mag * c1) * 0.5;
324  ns[1] = (nc - mag * c1) * 0.5;
325  ns[2] = mag * std::conj(cs) * 0.5;
326  ns[3] = mag * cs * 0.5;
327 
328  for (int m = 0; m < lmax_at; m++) {
329  this->data_(m, m, 0, ia) = ns[0];
330  this->data_(m, m, 1, ia) = ns[1];
331  this->data_(m, m, 2, ia) = ns[2];
332  this->data_(m, m, 3, ia) = ns[3];
333  }
334  }
335  } else {
336  for (int s = 0; s < ctx_.num_spins(); s++) {
337  for (int m = 0; m < lmax_at; m++) {
338  this->data_(m, m, s, ia) = charge * 0.5 / static_cast<double>(lmax_at);
339  }
340  }
341  }
342  }
343  }
344  }

349 void Occupation_matrix::print_occupancies() const
350 {
351  if (ctx_.verbosity() >= 2 && ctx_.comm().rank() == 0 && data_.size()) {
352  std::printf("\n");
353  for (int ci = 0; ci < 10; ci++) {
354  std::printf("--------");
355  }
356  std::printf("\n");
357  std::printf("hubbard occupancies\n");
358  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
359  std::printf("Atom : %d\n", ia);
360  std::printf("Mag Dim : %d\n", ctx_.num_mag_dims());
361  const auto& atom = ctx_.unit_cell().atom(ia);
362 
363  if (atom.type().hubbard_correction()) {
364  const int lmax_at = 2 * atom.type().lo_descriptor_hub(0).l + 1;
365  for (int m1 = 0; m1 < lmax_at; m1++) {
366  for (int m2 = 0; m2 < lmax_at; m2++) {
367  std::printf("%.3lf ", std::abs(this->data_(m1, m2, 0, ia)));
368  }
369 
370  if (ctx_.num_mag_dims() == 3) {
371  std::printf(" ");
372  for (int m2 = 0; m2 < lmax_at; m2++) {
373  std::printf("%.3lf ", std::abs(this->data_(m1, m2, 2, ia)));
374  }
375  }
376  std::printf("\n");
377  }
378 
379  if (ctx_.num_spins() == 2) {
380  for (int m1 = 0; m1 < lmax_at; m1++) {
381  if (ctx_.num_mag_dims() == 3) {
382  for (int m2 = 0; m2 < lmax_at; m2++) {
383  std::printf("%.3lf ", std::abs(this->data_(m1, m2, 3, ia)));
384  }
385  std::printf(" ");
386  }
387  for (int m2 = 0; m2 < lmax_at; m2++) {
388  std::printf("%.3lf ", std::abs(this->data_(m1, m2, 1, ia)));
389  }
390  std::printf("\n");
391  }
392  }
393 
394  double n_up, n_down, n_total;

399  }
400 
401  if (ctx_.num_spins() == 2) {
402  for (int m1 = 0; m1 < lmax_at; m1++) {
403  n_down += this->data_(m1, m1, 1, ia).real();
404  }
405  }
406  std::printf("\n");
407  n_total = n_up + n_down;
408  if (ctx_.num_spins() == 2) {
409  std::printf("Atom charge (total) %.5lf (n_up) %.5lf (n_down) %.5lf (mz) %.5lf\n", n_total, n_up, n_down, n_up - n_down);
410  } else {
411  std::printf("Atom charge (total) %.5lf\n", 2.0 * n_total);
412  }
413 
414  std::printf("\n");
415  for (int ci = 0; ci < 10; ci++) {
416  std::printf("--------");
417  }
418  std::printf("\n");
419  }
420  }
421  }
