31template <typename T>
32void Beta_projectors_base<T>::split_in_chunks()
33{
34 auto& uc = ctx_.unit_cell();
35
36 if (uc.mt_lo_basis_size() == 0) {
37 /* no beta projectors at all */
38 beta_chunks_ = std::vector<beta_chunk_t>(0);
39 num_beta_t_ = 0;
40 max_num_beta_ = 0;
41 return;
42 }
43
44 /* initial chunk size */
45 int chunk_size = std::min(uc.num_atoms(), ctx_.cfg().control().beta_chunk_size());
46 /* maximum number of chunks */
47 int num_chunks = uc.num_atoms() / chunk_size + std::min(1, uc.num_atoms() % chunk_size);
48 /* final maximum chunk size */
49 chunk_size = uc.num_atoms() / num_chunks + std::min(1, uc.num_atoms() % num_chunks);
50
51 int offset_in_beta_gk{0};
52 beta_chunks_ = std::vector<beta_chunk_t>(num_chunks);
53
54 for (int ib = 0; ib < num_chunks; ib++) {
55 /* number of atoms in this chunk */
56 int na = std::min(uc.num_atoms(), (ib + 1) * chunk_size) - ib * chunk_size;
57 beta_chunks_[ib].num_atoms_ = na;
58 beta_chunks_[ib].desc_ = mdarray<int, 2>(4, na);
59 beta_chunks_[ib].atom_pos_ = mdarray<double, 2>(3, na);
60
61 int num_beta{0};
62 for (int i = 0; i < na; i++) {
63 /* global index of atom by local index and chunk */
64 int ia = ib * chunk_size + i;
65 auto pos = uc.atom(ia).position();
66 auto& type = uc.atom(ia).type();
67 /* atom fractional coordinates */
68 for (int x: {0, 1, 2}) {
69 beta_chunks_[ib].atom_pos_(x, i) = pos[x];
70 }
71 /* number of beta functions for atom */
72 beta_chunks_[ib].desc_(static_cast<int>(beta_desc_idx::nbf), i) = type.mt_basis_size();
73 /* offset in beta_gk*/
74 beta_chunks_[ib].desc_(static_cast<int>(beta_desc_idx::offset), i) = num_beta;
75 /* offset in beta_gk_t */
76 beta_chunks_[ib].desc_(static_cast<int>(beta_desc_idx::offset_t), i) = type.offset_lo();
77 /* global index of atom */
78 beta_chunks_[ib].desc_(static_cast<int>(beta_desc_idx::ia), i) = ia;
79
80 num_beta += type.mt_basis_size();
81 }
82 /* number of beta-projectors in this chunk */
83 beta_chunks_[ib].num_beta_ = num_beta;
84 beta_chunks_[ib].offset_ = offset_in_beta_gk;
85 offset_in_beta_gk += num_beta;
86
87 if (ctx_.processing_unit() == device_t::GPU) {
88 beta_chunks_[ib].desc_.allocate(memory_t::device).copy_to(memory_t::device);
89 beta_chunks_[ib].atom_pos_.allocate(memory_t::device).copy_to(memory_t::device);
90 }
91 }
92
93 max_num_beta_ = 0;
94 for (auto& e: beta_chunks_) {
95 max_num_beta_ = std::max(max_num_beta_, e.num_beta_);
96 }
97
98 num_beta_t_ = 0;
99 for (int iat = 0; iat < uc.num_atom_types(); iat++) {
100 num_beta_t_ += uc.atom_type(iat).mt_lo_basis_size();
101 }
102}

104template <typename T>
105Beta_projectors_base<T>::Beta_projectors_base(Simulation_context& ctx__, Gvec const& gkvec__,
106 std::vector<int> const& igk__, int N__)
107 : ctx_(ctx__)
108 , gkvec_(gkvec__)
109 , igk_(igk__)
110 , N_(N__)
111{
112 split_in_chunks();
113
114 if (!num_beta_t()) {
115 return;
116 }
117
118 /* allocate memory */
119 pw_coeffs_t_ = mdarray<std::complex<T>, 3>(num_gkvec_loc(), num_beta_t(), N__, memory_t::host, "pw_coeffs_t_");
120
121 if (ctx_.processing_unit() == device_t::GPU) {
122 gkvec_coord_ = mdarray<double, 2>(3, num_gkvec_loc());
123 gkvec_coord_.allocate(memory_t::device);
124 /* copy G+k vectors */
125 for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
126 auto vgk = gkvec_.gkvec(igk_[igk_loc]);
127 for (auto x: {0, 1, 2}) {
128 gkvec_coord_(x, igk_loc) = vgk[x];
129 }
130 }
131 gkvec_coord_.copy_to(memory_t::device);
132 }
133}

135template <typename T> template <typename F, typename>
136matrix<F>
137Beta_projectors_base<T>::inner(int chunk__, Wave_functions<T>& phi__, int ispn__, int idx0__, int n__)
138{
139 PROFILE("sirius::Beta_projectors_base::inner");
140
141 assert(num_gkvec_loc() == phi__.pw_coeffs(ispn__).num_rows_loc());
142
143 int nbeta = chunk(chunk__).num_beta_;
144
145 matrix<F> beta_phi(nbeta, n__, ctx_.mem_pool(ctx_.host_memory_t()));
146
147 /* location of the beta-projectors is always on the memory of the processing unit being used */
148 F* pw_coeffs_a_ptr{nullptr};
149 switch (ctx_.processing_unit()) {
150 case device_t::CPU: {
151 pw_coeffs_a_ptr = reinterpret_cast<F*>(pw_coeffs_a().at(memory_t::host));
152 break;
153 }
154 case device_t::GPU: {
155 beta_phi.allocate(ctx_.mem_pool(memory_t::device));
156 pw_coeffs_a_ptr = reinterpret_cast<F*>(pw_coeffs_a().at(memory_t::device));
157 break;
158 }
159 }
160
161 local_inner_aux<F>(pw_coeffs_a_ptr, nbeta, phi__, ispn__, idx0__, n__, beta_phi);
162
163 /* copy to host in MPI sequential or parallel case */
164 if (is_device_memory(ctx_.preferred_memory_t())) {
165 beta_phi.copy_to(memory_t::host);
166 }
167
168 /* in parallel case do a reduction */
169 if (gkvec_.comm().size() > 1) {
170 PROFILE("sirius::Beta_projectors_base::inner|comm");
171 /* MPI reduction on the host */
172 gkvec_.comm().allreduce(beta_phi.at(memory_t::host), static_cast<int>(beta_phi.size()));
173 }
174
175 switch (ctx_.processing_unit()) {
176 case device_t::GPU: {
177 /* copy back to device */
178 if (gkvec_.comm().size() > 1 || is_host_memory(ctx_.preferred_memory_t())) {
179 beta_phi.copy_to(memory_t::device);
180 }
181 break;
182 }
183 case device_t::CPU: break;
184 }
185
186 return beta_phi;
187}

213template <typename T>
214void Beta_projectors_base<T>::generate(int ichunk__, int j__)
215{
216 PROFILE("sirius::Beta_projectors_base::generate");
217
218 switch (ctx_.processing_unit()) {
219 case device_t::CPU: {
220 #pragma omp parallel for
221 for (int i = 0; i < chunk(ichunk__).num_atoms_; i++) {
222 int ia = chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::ia), i);
223
224 double phase = twopi * dot(gkvec_.vk(), ctx_.unit_cell().atom(ia).position());
225 std::complex<double> phase_k = std::exp(std::complex<T>(0.0, phase));
226
227 std::vector<std::complex<double>> phase_gk(num_gkvec_loc());
228 for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
229 auto G = gkvec_.gvec(igk_[igk_loc]);
230 /* total phase e^{-i(G+k)r_{\alpha}} */
231 phase_gk[igk_loc] = std::conj(ctx_.gvec_phase_factor(G, ia) * phase_k);
232 }
233 for (int xi = 0; xi < chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::nbf), i); xi++) {
234 for (int igk_loc = 0; igk_loc < num_gkvec_loc(); igk_loc++) {
235 pw_coeffs_a_(igk_loc, chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::offset), i) + xi) =
236 pw_coeffs_t_(igk_loc, chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::offset_t), i) + xi, j__) * static_cast<std::complex<T>>(phase_gk[igk_loc]);
237 }
238 }
239 }
240 break;
241 }
242 case device_t::GPU: {
243#if defined(SIRIUS_GPU)
244 auto& desc = chunk(ichunk__).desc_;
245 create_beta_gk_gpu(chunk(ichunk__).num_atoms_,
246 num_gkvec_loc(),
247 desc.at(memory_t::device),
248 pw_coeffs_t_.at(memory_t::device, 0, 0, j__),
249 gkvec_coord_.at(memory_t::device),
250 chunk(ichunk__).atom_pos_.at(memory_t::device),
251 pw_coeffs_a().at(memory_t::device));
252#endif
253 /* wave-functions are on CPU but the beta-projectors are on GPU */
254 if (gkvec_.comm().rank() == 0 && is_host_memory(ctx_.preferred_memory_t())) {
255 /* make beta-projectors for G=0 on the CPU */
256 #pragma omp parallel for schedule(static)
257 for (int i = 0; i < chunk(ichunk__).num_atoms_; i++) {
258 for (int xi = 0; xi < chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::nbf), i); xi++) {
259 pw_coeffs_a_g0_(chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::offset), i) + xi) =
260 pw_coeffs_t_(0, chunk(ichunk__).desc_(static_cast<int>(beta_desc_idx::offset_t), i) + xi, j__);
261 }
262 }
263 }
264 break;
265 }
266 }
267}

269template <typename T>
270void Beta_projectors_base<T>::prepare()
271{
272 PROFILE("sirius::Beta_projectors_base::prepare");
273
274 if (max_num_beta() == 0) {
275 return;
276 }
277
278 switch (ctx_.processing_unit()) {
279 case device_t::CPU: {
280 pw_coeffs_a_ = matrix<std::complex<T>>(num_gkvec_loc(), max_num_beta(), ctx_.mem_pool(ctx_.host_memory_t()),
281 "pw_coeffs_a_");
282 pw_coeffs_a_g0_ = mdarray<std::complex<T>, 1>(max_num_beta(), ctx_.mem_pool(memory_t::host),
283 "pw_coeffs_a_g0_");
284 break;
285 }
286 case device_t::GPU: {
287 pw_coeffs_a_ = matrix<std::complex<T>>(num_gkvec_loc(), max_num_beta(), ctx_.mem_pool(memory_t::device),
288 "pw_coeffs_a_");
289 pw_coeffs_a_g0_ = mdarray<std::complex<T>, 1>(max_num_beta(), ctx_.mem_pool(memory_t::host),
290 "pw_coeffs_a_g0_");
291 pw_coeffs_a_g0_.allocate(ctx_.mem_pool(memory_t::device));
292 break;
293 }
294 }
295
296 if (ctx_.processing_unit() == device_t::GPU && reallocate_pw_coeffs_t_on_gpu_) {
297 pw_coeffs_t_.allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
298 }
299}

301template <typename T>
302void Beta_projectors_base<T>::dismiss()
303{
304 PROFILE("sirius::Beta_projectors_base::dismiss");
305
306 if (ctx_.processing_unit() == device_t::GPU && reallocate_pw_coeffs_t_on_gpu_) {
307 pw_coeffs_t_.deallocate(memory_t::device);
308 }
309 pw_coeffs_a_.deallocate(memory_t::device);
310 pw_coeffs_a_g0_.deallocate(memory_t::device);
311}
