45/// Compose JSON dictionary with default parameters based on input schema.
46/** Traverse the JSON schema and add nodes with default parameters to the output dictionary. The nodes without
47 * default parameters are ignored. Still, user has a possibility to add the missing nodes later by providing a
48 * corresponding input JSON dictionary. See compose_json() function. */
49void
50compose_default_json(nlohmann::json const& schema__, nlohmann::json& output__)
51{
52 for (auto it : schema__.items()) {
53 auto key = it.key();
54 /* this is a final node with the description of the data type */
55 if (it.value().contains("type") && it.value()["type"] != "object") {
56 /* check if default parameter is present */
57 if (it.value().contains("default")) {
58 output__[key] = it.value()["default"];
59 }
60 } else { /* otherwise continue to traverse the shcema */
61 if (!output__.contains(key)) {
62 output__[key] = nlohmann::json{};
63 }
64 if (it.value().contains("properties")) {
65 compose_default_json(it.value()["properties"], output__[key]);
66 }
67 }
68 }
69}

71/// Append the input dictionary to the existing dictionary.
72/** Use JSON schema to traverse the existing dictionary and add on top the values from the input dictionary. In this
73 * way we can add missing nodes which were not defined in the existing dictionary. */
74void
75compose_json(nlohmann::json const& schema__, nlohmann::json const& in__, nlohmann::json& inout__)
76{
77 std::unordered_set<std::string> visited;
78
79 for (auto it : in__.items()) {
80 visited.insert(it.key());
81 }
82
83 for (auto it : schema__.items()) {
84 auto key = it.key();
85
86 // Remove visited items.
87 auto found = visited.find(key);
88 if (found != visited.end()) {
89 visited.erase(found);
90 }
91
92 /* this is a final node with the description of the data type */
93 if (it.value().contains("type") && it.value()["type"] != "object") {
94 if (in__.contains(key)) {
95 /* copy the new input */
96 inout__[key] = in__[key];
97 }
98 } else { /* otherwise continue to traverse the shcema */
99 if (it.value().contains("properties")) {
100 compose_json(it.value()["properties"], in__.contains(key) ? in__[key] : nlohmann::json{}, inout__[key]);
101 } else if (in__.contains(key)) {
102 inout__[key] = in__[key];
103 } else {
104 inout__[key] = nlohmann::json();
105 }
106 }
107 }
108
109 // Emit warnings about keys that were set but unused.
110 if (!visited.empty()) {
111 std::stringstream ss;
112 ss << "The following configuration parameters were not recognized and ignored: ";
113 std::copy(visited.begin(), visited.end(), std::ostream_iterator<std::string>(ss, " "));
114 WARNING(ss)
115 }
116}

224void
225Simulation_parameters::electronic_structure_method(std::string name__)
226{
227 cfg_.parameters().electronic_structure_method(name__);
228
229 std::map<std::string, electronic_structure_method_t> m = {
230 {"full_potential_lapwlo", electronic_structure_method_t::full_potential_lapwlo},
231 {"pseudopotential", electronic_structure_method_t::pseudopotential}};
232
233 if (m.count(name__) == 0) {
234 std::stringstream s;
235 s << "wrong type of electronic structure method: " << name__;
236 TERMINATE(s);
237 }
238 electronic_structure_method_ = m[name__];
