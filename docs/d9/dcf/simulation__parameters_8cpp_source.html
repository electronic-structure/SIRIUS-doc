37 /// Compose JSON dictionary with default parameters based on input schema.
38 /** Traverse the JSON schema and add nodes with default parameters to the output dictionary. The nodes without
39  * default parameters are ignored. Still, user has a possibility to add the missing nodes later by providing a
40  * corresponding input JSON dictionary. See compose_json() function. */
41 void compose_default_json(nlohmann::json const& schema__, nlohmann::json& output__)
42 {
43  for (auto it: schema__.items()) {
44  auto key = it.key();
45  /* this is a final node with the description of the data type */
46  if (it.value().contains("type") && it.value()["type"] != "object") {
47  /* check if default parameter is present */
48  if (it.value().contains("default")) {
49  output__[key] = it.value()["default"];
50  }
51  } else { /* otherwise continue to traverse the shcema */
52  if (!output__.contains(key)) {
53  output__[key] = nlohmann::json{};
54  }
55  if (it.value().contains("properties")) {
56  compose_default_json(it.value()["properties"], output__[key]);
57  }
58  }
59  }
60 }

62 /// Append the input dictionary to the existing dictionary.
63 /** Use JSON schema to traverse the existing dictionary and add on top the values from the input dictionary. In this
64  * way we can add missing nodes which were not defined in the existing dictionary. */
65 void compose_json(nlohmann::json const& schema__, nlohmann::json const& in__, nlohmann::json& inout__)
66 {
67  std::unordered_set<std::string> visited;
68 
69  for (auto it: in__.items()) {
70  visited.insert(it.key());
71  }
72 
73  for (auto it: schema__.items()) {
74  auto key = it.key();
75 
76  // Remove visited items.
77  auto found = visited.find(key);
78  if (found != visited.end()) {
79  visited.erase(found);
80  }
81 
82  /* this is a final node with the description of the data type */
83  if (it.value().contains("type") && it.value()["type"] != "object") {
84  if (in__.contains(key)) {
85  /* copy the new input */
86  inout__[key] = in__[key];
87  }
88  } else { /* otherwise continue to traverse the shcema */
89  if (it.value().contains("properties")) {
90  compose_json(it.value()["properties"], in__.contains(key) ? in__[key] : nlohmann::json{}, inout__[key]);
91  } else if (in__.contains(key)) {
92  inout__[key] = in__[key];
93  } else {
94  inout__[key] = nlohmann::json();
95  }
96  }
97  }
98 
99  // Emit warnings about keys that were set but unused.
100  if (!visited.empty()) {
101  std::stringstream ss;
102  ss << "The following configuration parameters were not recognized and ignored: ";
103  std::copy(visited.begin(), visited.end(), std::ostream_iterator<std::string>(ss, " "));
104  WARNING(ss)
105  }
106 }

195 void Simulation_parameters::print_options() const
196 {
197  auto const& dict = get_options_dictionary();
198 
199  if (Communicator::world().rank() == 0) {
200  std::printf("The SIRIUS library or the mini apps can be initialized through the interface\n");
201  std::printf("using the API directly or through a json dictionary. The following contains\n");
202  std::printf("a description of all the runtime options, that can be used directly to\n");
203  std::printf("initialize SIRIUS.\n");
204 
205  for (auto& el : dict.items()) {
206  std::cout << "============================================================================\n";
207  std::cout << " ";
208  std::cout << " section : " << el.key() << " \n";
209  std::cout << " \n";
210  std::cout << "============================================================================\n";
211 
212  for (size_t s = 0; s < dict[el.key()].size(); s++) {
213  std::cout << "name of the option : " << dict[el.key()][s]["name"].get<std::string>() << std::endl;
214  std::cout << "description : " << dict[el.key()][s]["description"].get<std::string>() << std::endl;
215  if (dict[el.key()][s].count("possible_values")) {
216  const auto& v = dict[el.key()][s]["description"].get<std::vector<std::string>>();
217  std::cout << "possible values : " << v[0];
218  for (size_t st = 1; st < v.size(); st++)
219  std::cout << " " << v[st];
220  }
221  std::cout << "default value : " << dict[el.key()]["default_values"].get<std::string>() << std::endl;
222  }
223  }
224  }
225  Communicator::world().barrier();
226 }
