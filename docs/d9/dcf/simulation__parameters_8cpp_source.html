108 void Simulation_parameters::set_processing_unit(std::string name__)
109 {
110  std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
111 
112  /* set the default value */
113  if (name__ == "") {
114  if (acc::num_devices() > 0) {
115  name__ = "gpu";
116  } else {
117  name__ = "cpu";
118  }
119  }
120  control_input_.processing_unit_ = name__;
121  if (name__ == "cpu") {
122  this->set_processing_unit(device_t::CPU);
123  } else if (name__ == "gpu") {
124  this->set_processing_unit(device_t::GPU);
125  } else {
126  std::stringstream s;
127  s << "wrong processing unit name: " << name__;
128  TERMINATE(s);
129  }
130 }

143  } else {
144  std::stringstream s;

147  }
148  }

151 void Simulation_parameters::print_options() const
152 {
153  json const& dict = get_options_dictionary();
154 
155  if (Communicator::world().rank() == 0) {
156  std::printf("The SIRIUS library or the mini apps can be initialized through the interface\n");
157  std::printf("using the API directly or through a json dictionary. The following contains\n");
158  std::printf("a description of all the runtime options, that can be used directly to\n");
159  std::printf("initialize SIRIUS.\n");
160 
161  for (auto& el : dict.items()) {
162  std::cout << "============================================================================\n";
163  std::cout << " ";
164  std::cout << " section : " << el.key() << " \n";
165  std::cout << " \n";
166  std::cout << "============================================================================\n";
167 
168  for (size_t s = 0; s < dict[el.key()].size(); s++) {
169  std::cout << "name of the option : " << dict[el.key()][s]["name"].get<std::string>() << std::endl;
170  std::cout << "description : " << dict[el.key()][s]["description"].get<std::string>() << std::endl;
171  if (dict[el.key()][s].count("possible_values")) {
172  const auto& v = dict[el.key()][s]["description"].get<std::vector<std::string>>();
173  std::cout << "possible values : " << v[0];
174  for (size_t st = 1; st < v.size(); st++)
175  std::cout << " " << v[st];
176  }
177  std::cout << "default value : " << dict[el.key()]["default_values"].get<std::string>() << std::endl;
178  }
179  }
180  }
181  Communicator::world().barrier();
182 }
