62  : XC_functional_base(libxc_name__, num_spins__)
63  {
64 
65 #if defined(__USE_VDWXC)
66  /* return immediately if the functional_base class is initialized */
67  if (this->libxc_initialized_) {
68  return;
69  }
70 
71  /* test if have van der walls functionals types */
72 
73  bool test = (libxc_name_ == "XC_FUNC_VDWDF");
74  test = test || (libxc_name_ == "XC_FUNC_VDWDF2");
75  test = test || (libxc_name_ == "XC_FUNC_VDWDFCX");
76 
77  int func_ = -1;
78 
79  if (libxc_name__ == "XC_FUNC_VDWDF") {
80  func_ = FUNC_VDWDF;
81  }
82 
83  if (libxc_name__ == "XC_FUNC_VDWDF2") {
84  func_ = FUNC_VDWDF2;
85  }
86 
87  if (libxc_name__ == "XC_FUNC_VDWDFCX") {
88  func_ = FUNC_VDWDFCX;
89  }
90 
91  if (test) {
92  if (num_spins__ == 1) {
93  // non magnetic case
94  handler_vdw_ = vdwxc_new(func_);
95  } else {
96  // magnetic case
97  handler_vdw_ = vdwxc_new_spin(func_);
98  }
99 
100  if (!handler_vdw_) {
101  std::stringstream s;
102  s << "VDW functional lib could not be initialized";
103  TERMINATE(s);
104  }
105 
106  double v1[3] = { lattice_vectors_(0, 0), lattice_vectors_(1, 0), lattice_vectors_(2, 0) };
107  double v2[3] = { lattice_vectors_(0, 1), lattice_vectors_(1, 1), lattice_vectors_(2, 1) };
108  double v3[3] = { lattice_vectors_(0, 2), lattice_vectors_(1, 2), lattice_vectors_(2, 2) };
109  vdwxc_set_unit_cell(handler_vdw_,
110  fft.size(0),
111  fft.size(1),
112  fft.size(2),
113  v1[0], v1[1], v1[2],
114  v2[0], v2[1], v2[2],
115  v3[0], v3[1], v3[2]);
116 
117  if (fft.comm().size() == 1) {
118  vdwxc_init_serial(handler_vdw_);
119  } else {
120 #if __HAVE_VDWXC_MPI
121  vdwxc_init_mpi(handler_vdw_, fft.comm().mpi_comm());
122 #else
123  vdwxc_init_serial(handler_vdw_);
124 #endif
125  }
126  vdw_functional_ = true;
127  return;
128  } else {
129  /* it means that the functional does not exist either in vdw or xc libraries */
130  std::stringstream s;
131  s << "XC functional " << libxc_name__ << " is unknown";
132  TERMINATE(s);
133  }
134 #else
135  if (this->libxc_initialized_) {
136  return;
137  } else {
138  /* it means that the functional does not exist either in vdw or xc libraries */
139  std::stringstream s;
140  s << "XC functional " << libxc_name__ << " is unknown";
141  TERMINATE(s);
142  }
143 #endif /* __USE_VDWXC */
144  }
