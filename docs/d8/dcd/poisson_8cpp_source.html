30 void Potential::poisson_add_pseudo_pw(mdarray<double_complex, 2>& qmt__,
31  mdarray<double_complex, 2>& qit__,
32  double_complex* rho_pw__)
33 {
34  PROFILE("sirius::Potential::poisson_add_pseudo_pw");
35 
36  int lmmax = ctx_.lmmax_rho();
37  int ngv = ctx_.gvec().count();
38 
39  /* The following term is added to the plane-wave coefficients of the charge density:
40  * Integrate[SphericalBesselJ[l,a*x]*p[x,R]*x^2,{x,0,R},Assumptions->{l>=0,n>=0,R>0,a>0}] /
41  * Integrate[p[x,R]*x^(2+l),{x,0,R},Assumptions->{h>=0,n>=0,R>0}]
42  * i.e. contributon from pseudodensity to l-th channel of plane wave expansion multiplied by
43  * the difference bethween true and interstitial-in-the-mt multipole moments and divided by the
44  * moment of the pseudodensity.
45  */
46  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
47  double R = unit_cell_.atom_type(iat).mt_radius();
48  int na = unit_cell_.atom_type(iat).num_atoms();
49 
50  mdarray<double_complex, 2> pf;
51  mdarray<double_complex, 2> qa;
52  mdarray<double_complex, 2> qapf;
53 
54  switch (ctx_.processing_unit()) {
55  case device_t::CPU: {
56  auto& mp = ctx_.mem_pool(memory_t::host);
57  pf = mdarray<double_complex, 2>(ngv, na, mp);
58  qa = mdarray<double_complex, 2>(lmmax, na, mp);
59  qapf = mdarray<double_complex, 2>(lmmax, ngv, mp);
60  break;
61  }
62  case device_t::GPU: {
63  auto& mp = ctx_.mem_pool(memory_t::host);
64  auto& mpd = ctx_.mem_pool(memory_t::device);
65  /* allocate on GPU */
66  pf = mdarray<double_complex, 2>(nullptr, ngv, na);
67  pf.allocate(mpd);
68  /* allocate on CPU & GPU */
69  qa = mdarray<double_complex, 2>(lmmax, na, mp);
70  qa.allocate(mpd);
71  /* allocate on CPU & GPU */
72  qapf = mdarray<double_complex, 2>(lmmax, ngv, mp);
73  qapf.allocate(mpd);
74  break;
75  }
76  }
77 
78  ctx_.generate_phase_factors(iat, pf);
79 
80  for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
81  int ia = unit_cell_.atom_type(iat).atom_id(i);
82  for (int lm = 0; lm < ctx_.lmmax_rho(); lm++) {
83  qa(lm, i) = qmt__(lm, ia) - qit__(lm, ia);
84  }
85  }
86 
87  switch (ctx_.processing_unit()) {
88  case device_t::CPU: {
89  linalg(linalg_t::blas).gemm('N', 'C', ctx_.lmmax_rho(), ctx_.gvec().count(),
90  unit_cell_.atom_type(iat).num_atoms(), &linalg_const<double_complex>::one(),
91  qa.at(memory_t::host), qa.ld(), pf.at(memory_t::host), pf.ld(),
92  &linalg_const<double_complex>::zero(), qapf.at(memory_t::host), qapf.ld());
93  break;
94  }
95  case device_t::GPU: {
96  qa.copy_to(memory_t::device);
97  linalg(linalg_t::gpublas).gemm('N', 'C', ctx_.lmmax_rho(), ctx_.gvec().count(),
98  unit_cell_.atom_type(iat).num_atoms(), &linalg_const<double_complex>::one(),
99  qa.at(memory_t::device), qa.ld(), pf.at(memory_t::device), pf.ld(),
100  &linalg_const<double_complex>::zero(), qapf.at(memory_t::device), qapf.ld());
101  qapf.copy_to(memory_t::host);
102  break;
103  }
104  }
105 
106  double fourpi_omega = fourpi / unit_cell_.omega();
107 
108  /* add pseudo_density to interstitial charge density so that rho(G) has the correct
109  * multipole moments in the muffin-tins */
110  #pragma omp parallel for schedule(static)
111  for (int igloc = ctx_.gvec().skip_g0(); igloc < ctx_.gvec().count(); igloc++) {
112  int ig = ctx_.gvec().offset() + igloc;
113 
114  double gR = ctx_.gvec().gvec_len(ig) * R;
115  double gRn = std::pow(2.0 / gR, pseudo_density_order_ + 1);
116 
117  double_complex rho_G(0, 0);
118  /* loop over atoms of the same type */
119  for (int l = 0, lm = 0; l <= ctx_.lmax_rho(); l++) {
120  double_complex zt1(0, 0);
121  for (int m = -l; m <= l; m++, lm++) {
122  zt1 += gvec_ylm_(lm, igloc) * qapf(lm, igloc);
123  }
124  rho_G += fourpi_omega * std::conj(zil_[l]) * zt1 * gamma_factors_R_(l, iat) *
125  sbessel_mt_(l + pseudo_density_order_ + 1, igloc, iat) * gRn;
126  }
127  rho_pw__[igloc] += rho_G;
128  }
129  /* for G=0 case */
130  if (ctx_.comm().rank() == 0) {
131  double_complex rho_G(0, 0);
132  for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
133  int ia = unit_cell_.atom_type(iat).atom_id(i);
134  rho_G += fourpi_omega * y00 * (qmt__(0, ia) - qit__(0, ia));
135  }
136  rho_pw__[0] += rho_G;
137  }
138  }

141 void Potential::poisson(Periodic_function<double> const& rho)
142 {
143  PROFILE("sirius::Potential::poisson");
144 
145  /* in case of full potential we need to do pseudo-charge multipoles */
146  if (ctx_.full_potential()) {
147 
148  /* true multipole moments */
149  auto qmt = poisson_vmt(rho);
150 
151  if (ctx_.cfg().control().print_checksum()) {
152  if (ctx_.comm().rank() == 0) {
153  utils::print_checksum("qmt", qmt.checksum());
154  }
155  }
156 
157  /* compute multipoles of interstitial density in MT region */
158  auto qit = ctx_.sum_fg_fl_yg(ctx_.lmax_rho(), &rho.f_pw_local(0), sbessel_mom_, gvec_ylm_);
159 
160  if (ctx_.cfg().control().print_checksum()) {
161  if (ctx_.comm().rank() == 0) {
162  utils::print_checksum("qit", qit.checksum());
163  }
164  }
165 
166  /* add contribution from the pseudo-charge */
167  poisson_add_pseudo_pw(qmt, qit, const_cast<double_complex*>(&rho.f_pw_local(0)));
168 
169  if (ctx_.cfg().control().verification() >= 2) {
170  auto qit = ctx_.sum_fg_fl_yg(ctx_.lmax_rho(), &rho.f_pw_local(0), sbessel_mom_, gvec_ylm_);
171 
172  double d = 0.0;
173  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
174  for (int lm = 0; lm < ctx_.lmmax_rho(); lm++) {
175  d += std::abs(qmt(lm, ia) - qit(lm, ia));
176  }
177  }
178  ctx_.message(1, __function_name__, "pseudocharge error: %18.10f\n", d);
179  }
180  }
181 
182  /* compute pw coefficients of Hartree potential */
183  int ig0{0};
184  if (ctx_.gvec().comm().rank() == 0) {
185  hartree_potential_->f_pw_local(0) = 0.0;
186  ig0 = 1;
187  }
188  if (!ctx_.molecule()) {
189  #pragma omp parallel for schedule(static)
190  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
191  int ig = ctx_.gvec().offset() + igloc;
192  hartree_potential_->f_pw_local(igloc) = (fourpi * rho.f_pw_local(igloc) / std::pow(ctx_.gvec().gvec_len(ig), 2));
193  }
194  } else {
195  /* reference paper:
196 
197  Supercell technique for total-energy calculations of finite charged and polar systems
198  M. R. Jarvis, I. D. White, R. W. Godby, and M. C. Payne
199  Phys. Rev. B 56, 14972 â€“ Published 15 December 1997
200  DOI:https://doi.org/10.1103/PhysRevB.56.14972
201  */
202  double R_cut = 0.5 * std::pow(unit_cell_.omega(), 1.0 / 3);
203  #pragma omp parallel for schedule(static)
204  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
205  int ig = ctx_.gvec().offset() + igloc;
206  hartree_potential_->f_pw_local(igloc) = (fourpi * rho.f_pw_local(igloc) / std::pow(ctx_.gvec().gvec_len(ig), 2)) *
207  (1.0 - std::cos(ctx_.gvec().gvec_len(ig) * R_cut));
208  }
209  }
210 
211  //if (ctx_.control().print_checksum_) {
212  // auto z4 = mdarray<double_complex, 1>(&vh->f_pw(0), ctx_.gvec().num_gvec()).checksum();
213  // if (ctx_.comm().rank() == 0) {
214  // DUMP("checksum(vh_pw): %20.14f %20.14f", z4.real(), z4.imag());
215  // }
216  //}
217 
218  /* boundary condition for muffin-tins */
219  if (ctx_.full_potential()) {
220  /* compute V_lm at the MT boundary */
221  auto vmtlm = ctx_.sum_fg_fl_yg(ctx_.lmax_pot(), &hartree_potential_->f_pw_local(0), sbessel_mt_, gvec_ylm_);
222 
223  /* add boundary condition and convert to Rlm */
224  PROFILE("sirius::Potential::poisson|bc");
225  mdarray<double, 3> rRl(unit_cell_.max_num_mt_points(), ctx_.lmax_pot() + 1, unit_cell_.num_atom_types());
226  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
227  int nmtp = unit_cell_.atom_type(iat).num_mt_points();
228  double R = unit_cell_.atom_type(iat).mt_radius();
229 
230  #pragma omp parallel for default(shared)
231  for (int l = 0; l <= ctx_.lmax_pot(); l++) {
232  for (int ir = 0; ir < nmtp; ir++) {
233  rRl(ir, l, iat) = std::pow(unit_cell_.atom_type(iat).radial_grid(ir) / R, l);
234  }
235  }
236  }
237 
238  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
239  int ia = unit_cell_.spl_num_atoms(ialoc);
240  int nmtp = unit_cell_.atom(ia).num_mt_points();
241 
242  std::vector<double> vlm(ctx_.lmmax_pot());
243  SHT::convert(ctx_.lmax_pot(), &vmtlm(0, ia), &vlm[0]);
244 
245  #pragma omp parallel for default(shared)
246  for (int lm = 0; lm < ctx_.lmmax_pot(); lm++) {
247  int l = l_by_lm_[lm];
248 
249  for (int ir = 0; ir < nmtp; ir++) {
250  hartree_potential_->f_mt<index_domain_t::local>(lm, ir, ialoc) +=
251  vlm[lm] * rRl(ir, l, unit_cell_.atom(ia).type_id());
252  }
253  }
254  /* save electronic part of the potential at the point of origin */
255 #ifdef __VHA_AUX
256  vh_el_(ia) = y00 * hartree_potential_->f_mt<index_domain_t::local>(0, 0, ialoc) + unit_cell_.atom(ia).zn() / unit_cell_.atom(ia).radial_grid(0);
257 #else
258  vh_el_(ia) = y00 * hartree_potential_->f_mt<index_domain_t::local>(0, 0, ialoc);
259 #endif
260  }
261  ctx_.comm().allgather(vh_el_.at(memory_t::host), unit_cell_.spl_num_atoms().local_size(),
262  unit_cell_.spl_num_atoms().global_offset());
263  }
264 
265  /* transform Hartree potential to real space */
266  hartree_potential_->fft_transform(1);
267 
268  if (ctx_.cfg().control().print_checksum()) {
269  auto cs = hartree_potential_->checksum_rg();
270  auto cs1 = hartree_potential_->checksum_pw();
271  if (ctx_.comm().rank() == 0) {
272  utils::print_checksum("vha_rg", cs);
273  utils::print_checksum("vha_pw", cs1);
274  }
275  }
276 
277  /* compute contribution from the smooth part of Hartree potential */
278  energy_vha_ = sirius::inner(rho, hartree_potential());
279 
280 #ifndef __VHA_AUX
281  /* add nucleus potential and contribution to Hartree energy */
282  if (ctx_.full_potential()) {
283  double evha_nuc{0};
284  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
285  int ia = unit_cell_.spl_num_atoms(ialoc);
286  auto& atom = unit_cell_.atom(ia);
287  Spline<double> srho(atom.radial_grid());
288  for (int ir = 0; ir < atom.num_mt_points(); ir++) {
289  double r = atom.radial_grid(ir);
290  hartree_potential_->f_mt<index_domain_t::local>(0, ir, ialoc) -= atom.zn() / r / y00;
291  srho(ir) = rho.f_mt<index_domain_t::local>(0, ir, ialoc) * r;
292  }
293  evha_nuc -= atom.zn() * srho.interpolate().integrate(0) / y00;
294  }
295  ctx_.comm().allreduce(&evha_nuc, 1);
296  energy_vha_ += evha_nuc;
297  }
298 #endif
299

301  //if (true) {
302  // /* compute V_lm at the MT boundary */
303  // auto vmtlm = ctx_.sum_fg_fl_yg(ctx_.lmax_pot(), &hartree_potential_->f_pw_local(0), sbessel_mt_, gvec_ylm_);
304 
305  // for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
306  // int ia = unit_cell_.spl_num_atoms(ialoc);
307  // int nmtp = unit_cell_.atom(ia).num_mt_points();
308

311

313  // printf("ia=%i lm=%i vlmdiff=%20.16f\n", ia, lm,
314  // std::abs(hartree_potential_->f_mt<index_domain_t::local>(lm, nmtp - 1, ialoc) - vlm[lm]));
315  // }
316  // }
