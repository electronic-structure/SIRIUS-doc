30double density_residual_hartree_energy(Density const& rho1__, Density const& rho2__)
31{
32 double eh{0};
33 auto const& gv = rho1__.ctx().gvec();
34 #pragma omp parallel for reduction(+:eh)
35 for (int igloc = gv.skip_g0(); igloc < gv.count(); igloc++) {
36 int ig = gv.offset() + igloc;
37 double_complex z = rho1__.component(0).f_pw_local(igloc) - rho2__.component(0).f_pw_local(igloc);
38 double g = gv.gvec_len(ig);
39 eh += (std::pow(z.real(), 2) + std::pow(z.imag(), 2)) / std::pow(g, 2);
40 }
41 gv.comm().allreduce(&eh, 1);
42 eh *= twopi * rho1__.ctx().unit_cell().omega();
43 if (gv.reduced()) {
44 eh *= 2;
45 }
46 return eh;
47}

49void Potential::poisson_add_pseudo_pw(mdarray<double_complex, 2>& qmt__,
50 mdarray<double_complex, 2>& qit__,
51 double_complex* rho_pw__)
52{
53 PROFILE("sirius::Potential::poisson_add_pseudo_pw");
54
55 int lmmax = ctx_.lmmax_rho();
56 int ngv = ctx_.gvec().count();
57
58 /* The following term is added to the plane-wave coefficients of the charge density:
59 * Integrate[SphericalBesselJ[l,a*x]*p[x,R]*x^2,{x,0,R},Assumptions->{l>=0,n>=0,R>0,a>0}] /
60 * Integrate[p[x,R]*x^(2+l),{x,0,R},Assumptions->{h>=0,n>=0,R>0}]
61 * i.e. contributon from pseudodensity to l-th channel of plane wave expansion multiplied by
62 * the difference bethween true and interstitial-in-the-mt multipole moments and divided by the
63 * moment of the pseudodensity.
64 */
65 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
66 double R = unit_cell_.atom_type(iat).mt_radius();
67 int na = unit_cell_.atom_type(iat).num_atoms();
68
69 mdarray<double_complex, 2> pf;
70 mdarray<double_complex, 2> qa;
71 mdarray<double_complex, 2> qapf;
72
73 switch (ctx_.processing_unit()) {
74 case device_t::CPU: {
75 auto& mp = ctx_.mem_pool(memory_t::host);
76 pf = mdarray<double_complex, 2>(ngv, na, mp);
77 qa = mdarray<double_complex, 2>(lmmax, na, mp);
78 qapf = mdarray<double_complex, 2>(lmmax, ngv, mp);
79 break;
80 }
81 case device_t::GPU: {
82 auto& mp = ctx_.mem_pool(memory_t::host);
83 auto& mpd = ctx_.mem_pool(memory_t::device);
84 /* allocate on GPU */
85 pf = mdarray<double_complex, 2>(nullptr, ngv, na);
86 pf.allocate(mpd);
87 /* allocate on CPU & GPU */
88 qa = mdarray<double_complex, 2>(lmmax, na, mp);
89 qa.allocate(mpd);
90 /* allocate on CPU & GPU */
91 qapf = mdarray<double_complex, 2>(lmmax, ngv, mp);
92 qapf.allocate(mpd);
93 break;
94 }
95 }
96
97 ctx_.generate_phase_factors(iat, pf);
98
99 for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
100 int ia = unit_cell_.atom_type(iat).atom_id(i);
101 for (int lm = 0; lm < ctx_.lmmax_rho(); lm++) {
102 qa(lm, i) = qmt__(lm, ia) - qit__(lm, ia);
103 }
104 }
105
106 switch (ctx_.processing_unit()) {
107 case device_t::CPU: {
108 linalg(linalg_t::blas).gemm('N', 'C', ctx_.lmmax_rho(), ctx_.gvec().count(),
109 unit_cell_.atom_type(iat).num_atoms(), &linalg_const<double_complex>::one(),
110 qa.at(memory_t::host), qa.ld(), pf.at(memory_t::host), pf.ld(),
111 &linalg_const<double_complex>::zero(), qapf.at(memory_t::host), qapf.ld());
112 break;
113 }
114 case device_t::GPU: {
115 qa.copy_to(memory_t::device);
116 linalg(linalg_t::gpublas).gemm('N', 'C', ctx_.lmmax_rho(), ctx_.gvec().count(),
117 unit_cell_.atom_type(iat).num_atoms(), &linalg_const<double_complex>::one(),
118 qa.at(memory_t::device), qa.ld(), pf.at(memory_t::device), pf.ld(),
119 &linalg_const<double_complex>::zero(), qapf.at(memory_t::device), qapf.ld());
120 qapf.copy_to(memory_t::host);
121 break;
122 }
123 }
124
125 double fourpi_omega = fourpi / unit_cell_.omega();
126
127 /* add pseudo_density to interstitial charge density so that rho(G) has the correct
128 * multipole moments in the muffin-tins */
129 #pragma omp parallel for schedule(static)
130 for (int igloc = ctx_.gvec().skip_g0(); igloc < ctx_.gvec().count(); igloc++) {
131 int ig = ctx_.gvec().offset() + igloc;
132
133 double gR = ctx_.gvec().gvec_len(ig) * R;
134 double gRn = std::pow(2.0 / gR, pseudo_density_order_ + 1);
135
136 double_complex rho_G(0, 0);
137 /* loop over atoms of the same type */
138 for (int l = 0, lm = 0; l <= ctx_.lmax_rho(); l++) {
139 double_complex zt1(0, 0);
140 for (int m = -l; m <= l; m++, lm++) {
141 zt1 += gvec_ylm_(lm, igloc) * qapf(lm, igloc);
142 }
143 rho_G += fourpi_omega * std::conj(zil_[l]) * zt1 * gamma_factors_R_(l, iat) *
144 sbessel_mt_(l + pseudo_density_order_ + 1, igloc, iat) * gRn;
145 }
146 rho_pw__[igloc] += rho_G;
147 }
148 /* for G=0 case */
149 if (ctx_.comm().rank() == 0) {
150 double_complex rho_G(0, 0);
151 for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
152 int ia = unit_cell_.atom_type(iat).atom_id(i);
153 rho_G += fourpi_omega * y00 * (qmt__(0, ia) - qit__(0, ia));
154 }
155 rho_pw__[0] += rho_G;
156 }
157 }

160void Potential::poisson(Periodic_function<double> const& rho)
161{
162 PROFILE("sirius::Potential::poisson");
163
164 /* in case of full potential we need to do pseudo-charge multipoles */
165 if (ctx_.full_potential()) {
166
167 /* true multipole moments */
168 auto qmt = poisson_vmt(rho);
169
170 if (ctx_.cfg().control().print_checksum()) {
171 if (ctx_.comm().rank() == 0) {
172 utils::print_checksum("qmt", qmt.checksum());
173 }
174 }
175
176 /* compute multipoles of interstitial density in MT region */
177 auto qit = ctx_.sum_fg_fl_yg(ctx_.lmax_rho(), &rho.f_pw_local(0), sbessel_mom_, gvec_ylm_);
178
179 if (ctx_.cfg().control().print_checksum()) {
180 if (ctx_.comm().rank() == 0) {
181 utils::print_checksum("qit", qit.checksum());
182 }
183 }
184
185 /* add contribution from the pseudo-charge */
186 poisson_add_pseudo_pw(qmt, qit, const_cast<double_complex*>(&rho.f_pw_local(0)));
187
188 if (ctx_.cfg().control().verification() >= 2) {
189 auto qit = ctx_.sum_fg_fl_yg(ctx_.lmax_rho(), &rho.f_pw_local(0), sbessel_mom_, gvec_ylm_);
190
191 double d = 0.0;
192 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
193 for (int lm = 0; lm < ctx_.lmmax_rho(); lm++) {
194 d += std::abs(qmt(lm, ia) - qit(lm, ia));
195 }
196 }
197 ctx_.message(1, __function_name__, "pseudocharge error: %18.10f\n", d);
198 }
199 }
200
201 /* compute pw coefficients of Hartree potential */
202 int ig0{0};
203 if (ctx_.gvec().comm().rank() == 0) {
204 hartree_potential_->f_pw_local(0) = 0.0;
205 ig0 = 1;
206 }
207 if (!ctx_.molecule()) {
208 #pragma omp parallel for schedule(static)
209 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
210 int ig = ctx_.gvec().offset() + igloc;
211 hartree_potential_->f_pw_local(igloc) = (fourpi * rho.f_pw_local(igloc) / std::pow(ctx_.gvec().gvec_len(ig), 2));
212 }
213 } else {
214 /* reference paper:
215
216 Supercell technique for total-energy calculations of finite charged and polar systems
217 M. R. Jarvis, I. D. White, R. W. Godby, and M. C. Payne
218 Phys. Rev. B 56, 14972 Published 15 December 1997
219 DOI:https://doi.org/10.1103/PhysRevB.56.14972
220 */
221 double R_cut = 0.5 * std::pow(unit_cell_.omega(), 1.0 / 3);
222 #pragma omp parallel for schedule(static)
223 for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
224 int ig = ctx_.gvec().offset() + igloc;
225 hartree_potential_->f_pw_local(igloc) = (fourpi * rho.f_pw_local(igloc) / std::pow(ctx_.gvec().gvec_len(ig), 2)) *
226 (1.0 - std::cos(ctx_.gvec().gvec_len(ig) * R_cut));
227 }
228 }
229
230 //if (ctx_.control().print_checksum_) {
231 // auto z4 = mdarray<double_complex, 1>(&vh->f_pw(0), ctx_.gvec().num_gvec()).checksum();
232 // if (ctx_.comm().rank() == 0) {
233 // DUMP("checksum(vh_pw): %20.14f %20.14f", z4.real(), z4.imag());
234 // }
235 //}
236
237 /* boundary condition for muffin-tins */
238 if (ctx_.full_potential()) {
239 /* compute V_lm at the MT boundary */
240 auto vmtlm = ctx_.sum_fg_fl_yg(ctx_.lmax_pot(), &hartree_potential_->f_pw_local(0), sbessel_mt_, gvec_ylm_);
241
242 /* add boundary condition and convert to Rlm */
243 PROFILE("sirius::Potential::poisson|bc");
244 mdarray<double, 3> rRl(unit_cell_.max_num_mt_points(), ctx_.lmax_pot() + 1, unit_cell_.num_atom_types());
245 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
246 int nmtp = unit_cell_.atom_type(iat).num_mt_points();
247 double R = unit_cell_.atom_type(iat).mt_radius();
248
249 #pragma omp parallel for default(shared)
250 for (int l = 0; l <= ctx_.lmax_pot(); l++) {
251 for (int ir = 0; ir < nmtp; ir++) {
252 rRl(ir, l, iat) = std::pow(unit_cell_.atom_type(iat).radial_grid(ir) / R, l);
253 }
254 }
255 }
256
257 for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
258 int ia = unit_cell_.spl_num_atoms(ialoc);
259 int nmtp = unit_cell_.atom(ia).num_mt_points();
260
261 std::vector<double> vlm(ctx_.lmmax_pot());
262 SHT::convert(ctx_.lmax_pot(), &vmtlm(0, ia), &vlm[0]);
263
264 #pragma omp parallel for default(shared)
265 for (int lm = 0; lm < ctx_.lmmax_pot(); lm++) {
266 int l = l_by_lm_[lm];
267
268 for (int ir = 0; ir < nmtp; ir++) {
269 hartree_potential_->f_mt<index_domain_t::local>(lm, ir, ialoc) +=
270 vlm[lm] * rRl(ir, l, unit_cell_.atom(ia).type_id());
271 }
272 }
273 /* save electronic part of the potential at the point of origin */
274#ifdef __VHA_AUX
275 vh_el_(ia) = y00 * hartree_potential_->f_mt<index_domain_t::local>(0, 0, ialoc) + unit_cell_.atom(ia).zn() / unit_cell_.atom(ia).radial_grid(0);
276#else
277 vh_el_(ia) = y00 * hartree_potential_->f_mt<index_domain_t::local>(0, 0, ialoc);
278#endif
279 }
280 ctx_.comm().allgather(vh_el_.at(memory_t::host), unit_cell_.spl_num_atoms().local_size(),
281 unit_cell_.spl_num_atoms().global_offset());
282 }
283
284 /* transform Hartree potential to real space */
285 hartree_potential_->fft_transform(1);
286
287 if (ctx_.cfg().control().print_checksum()) {
288 auto cs = hartree_potential_->checksum_rg();
289 auto cs1 = hartree_potential_->checksum_pw();
290 if (ctx_.comm().rank() == 0) {
291 utils::print_checksum("vha_rg", cs);
292 utils::print_checksum("vha_pw", cs1);
293 }
294 }
295
296 /* compute contribution from the smooth part of Hartree potential */
297 energy_vha_ = sirius::inner(rho, hartree_potential());
298
299#ifndef __VHA_AUX
300 /* add nucleus potential and contribution to Hartree energy */
301 if (ctx_.full_potential()) {
302 double evha_nuc{0};
303 for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
304 int ia = unit_cell_.spl_num_atoms(ialoc);
305 auto& atom = unit_cell_.atom(ia);
306 Spline<double> srho(atom.radial_grid());
307 for (int ir = 0; ir < atom.num_mt_points(); ir++) {
308 double r = atom.radial_grid(ir);
309 hartree_potential_->f_mt<index_domain_t::local>(0, ir, ialoc) -= atom.zn() / r / y00;
310 srho(ir) = rho.f_mt<index_domain_t::local>(0, ir, ialoc) * r;
311 }
312 evha_nuc -= atom.zn() * srho.interpolate().integrate(0) / y00;
313 }
314 ctx_.comm().allreduce(&evha_nuc, 1);
315 energy_vha_ += evha_nuc;
316 }
317#endif
318
319 /* check values at MT boundary */
320 //if (true) {
321 // /* compute V_lm at the MT boundary */
322 // auto vmtlm = ctx_.sum_fg_fl_yg(ctx_.lmax_pot(), &hartree_potential_->f_pw_local(0), sbessel_mt_, gvec_ylm_);
323
324 // for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
325 // int ia = unit_cell_.spl_num_atoms(ialoc);
326 // int nmtp = unit_cell_.atom(ia).num_mt_points();
327
328 // std::vector<double> vlm(ctx_.lmmax_pot());
329 // SHT::convert(ctx_.lmax_pot(), &vmtlm(0, ia), &vlm[0]);
330
331 // for (int lm = 0; lm < ctx_.lmmax_pot(); lm++) {
332 // printf("ia=%i lm=%i vlmdiff=%20.16f\n", ia, lm,
333 // std::abs(hartree_potential_->f_mt<index_domain_t::local>(lm, nmtp - 1, ialoc) - vlm[lm]));
334 // }
335 // }
336
337 //}
338}
