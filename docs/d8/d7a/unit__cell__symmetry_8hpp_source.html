38 /// Descriptor of the space group symmetry operation.
39 struct space_group_symmetry_descriptor
40 {
41  /// Rotational part of symmetry operation (fractional coordinates).
42  matrix3d<int> R;
43  /// inverse of R
44  matrix3d<int> invR;
45  /// inverse transposed of R
46  matrix3d<int> invRT;
47 
48  /// Fractional translation.
49  vector3d<double> t;
50 
51  /// Proper (+1) or improper (-1) rotation.
52  int proper;
53 
54  /// Proper rotation matrix in Cartesian coordinates.
55  matrix3d<double> rotation;
56 
57  /// Three Euler angles that generate the proper rotation matrix.
58  vector3d<double> euler_angles;
59 };

113  /// Generate rotation matrix from three Euler angles
114  /** Euler angles \f$ \alpha, \beta, \gamma \f$ define the general rotation as three consecutive rotations:
115  * - about \f$ \hat e_z \f$ through the angle \f$ \gamma \f$ (\f$ 0 \le \gamma < 2\pi \f$)
116  * - about \f$ \hat e_y \f$ through the angle \f$ \beta \f$ (\f$ 0 \le \beta \le \pi \f$)
117  * - about \f$ \hat e_z \f$ through the angle \f$ \alpha \f$ (\f$ 0 \le \gamma < 2\pi \f$)
118  *
119  * The total rotation matrix is defined as a product of three rotation matrices:
120  * \f[
121  * R(\alpha, \beta, \gamma) =
122  * \left( \begin{array}{ccc} \cos(\alpha) & -\sin(\alpha) & 0 \\
123  * \sin(\alpha) & \cos(\alpha) & 0 \\
124  * 0 & 0 & 1 \end{array} \right)
125  * \left( \begin{array}{ccc} \cos(\beta) & 0 & \sin(\beta) \\
126  * 0 & 1 & 0 \\
127  * -\sin(\beta) & 0 & \cos(\beta) \end{array} \right)
128  * \left( \begin{array}{ccc} \cos(\gamma) & -\sin(\gamma) & 0 \\
129  * \sin(\gamma) & \cos(\gamma) & 0 \\
130  * 0 & 0 & 1 \end{array} \right) =
131  * \left( \begin{array}{ccc} \cos(\alpha) \cos(\beta) \cos(\gamma) - \sin(\alpha) \sin(\gamma) &
132  * -\sin(\alpha) \cos(\gamma) - \cos(\alpha) \cos(\beta) \sin(\gamma) &
133  * \cos(\alpha) \sin(\beta) \\
134  * \sin(\alpha) \cos(\beta) \cos(\gamma) + \cos(\alpha) \sin(\gamma) &
135  * \cos(\alpha) \cos(\gamma) - \sin(\alpha) \cos(\beta) \sin(\gamma) &
136  * \sin(\alpha) \sin(\beta) \\
137  * -\sin(\beta) \cos(\gamma) &
138  * \sin(\beta) \sin(\gamma) &
139  * \cos(\beta) \end{array} \right)
140  * \f]
141  */
142  matrix3d<double> rot_mtrx_cart(vector3d<double> euler_angles__) const;
143 
144  /// Get axis and angle from rotation matrix.
145  static std::pair<vector3d<double>, double> axis_angle(matrix3d<double> R__)
146  {
147  vector3d<double> u;
148  /* make proper rotation */
149  R__ = R__ * R__.det();
150  u[0] = R__(2, 1) - R__(1, 2);
151  u[1] = R__(0, 2) - R__(2, 0);
152  u[2] = R__(1, 0) - R__(0, 1);
153 
154  double sint = u.length() / 2.0;
155  double cost = (R__(0, 0) + R__(1, 1) + R__(2, 2) - 1) / 2.0;
156 
157  double theta = utils::phi_by_sin_cos(sint, cost);
158 
159  /* rotation angle is zero */
160  if (std::abs(theta) < 1e-12) {
161  u = vector3d<double>({0, 0, 1});
162  } else if (std::abs(theta - pi) < 1e-12) { /* rotation angle is Pi */
163  /* rotation matrix for Pi angle has this form
164 
165  [-1+2ux^2 | 2 ux uy | 2 ux uz]
166  [2 ux uy | -1+2uy^2 | 2 uy uz]
167  [2 ux uz | 2 uy uz | -1+2uz^2] */
168 
169  if (R__(0, 0) >= R__(1, 1) && R__(0, 0) >= R__(2, 2)) { /* x-component is largest */
170  u[0] = std::sqrt(std::abs(R__(0, 0) + 1) / 2);
171  u[1] = (R__(0, 1) + R__(1, 0)) / 4 / u[0];
172  u[2] = (R__(0, 2) + R__(2, 0)) / 4 / u[0];
173  } else if (R__(1, 1) >= R__(0, 0) && R__(1, 1) >= R__(2, 2)) { /* y-component is largest */
174  u[1] = std::sqrt(std::abs(R__(1, 1) + 1) / 2);
175  u[0] = (R__(1, 0) + R__(0, 1)) / 4 / u[1];
176  u[2] = (R__(1, 2) + R__(2, 1)) / 4 / u[1];
177  } else {
178  u[2] = std::sqrt(std::abs(R__(2, 2) + 1) / 2);
179  u[0] = (R__(2, 0) + R__(0, 2)) / 4 / u[2];
180  u[1] = (R__(2, 1) + R__(1, 2)) / 4 / u[2];
181  }
182  } else {
183  u = u * (1.0 / u.length());
184  }
185 
186  return std::pair<vector3d<double>, double>(u, theta);
187  }

189  static mdarray<double_complex, 2> spinor_rotation_matrix(vector3d<double> u__, double theta__)
190  {
191  mdarray<double_complex, 2> rotm(2, 2);
192 
193  auto cost = std::cos(theta__ / 2);
194  auto sint = std::sin(theta__ / 2);
195 
196  rotm(0, 0) = double_complex(cost, -u__[2] * sint);
197  rotm(1, 1) = double_complex(cost, u__[2] * sint);
198  rotm(0, 1) = double_complex(-u__[1] * sint, -u__[0] * sint);
199  rotm(1, 0) = double_complex( u__[1] * sint, -u__[0] * sint);
200 
201  return std::move(rotm);
202  }

279  /// Symmetrize scalar function.
280  /** Symmetrize scalar function. The following operation is performed:
281  * \f[
282  * f_{\mathrm{sym}}({\bf x}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} f({\bf \hat P x})
283  * \f]
284  * For the function expanded in plane-waves we have:
285  * \f[
286  * f_{\mathrm{sym}}({\bf x}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} \sum_{\bf G}
287  * e^{i{\bf G \hat P x}} \hat f({\bf G})
288  * = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G (Rx +
289  * t)}} \hat f({\bf G})
290  * = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} \sum_{\bf G} e^{i{\bf G t}}
291  * e^{i{\bf R^T G x}} \hat f({\bf G})
292  * \f]
293  * Substitute \f$\bf \tilde G = \bf R^T \bf G\f$
294  * \f[
295  * f_{\mathrm{sym}}({\bf x}) =
296  * \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \tilde G}} e^{i {\bf \tilde G} }\sum_{{\bf \hat P}}
297  * e^{i {\bf R}^{-T} {\bf t}} \hat f ({\bf R}^{-T} {\bf \tilde G}) \,,
298  * \f]
299  * to find the Fourier coefficients \f$ \hat f_{\mathrm{sym}} \f$ of \f$f_{\mathrm{sym}}\f$ in
300  * terms of \f$ \hat f \f$:
301  * \f[
302  * \hat f_{\mathrm{sym}}({\bf G}) = \frac{1}{N_{\mathrm{sym}}} \sum_{{\bf \hat P}} e^{i {\bf R}^{-T} {\bf t}} \hat f ({\bf R}^{-T} {\bf G})\,.
303  * \f]
304  * Once \f$\hat f_{\mathrm{sym}} \f$ has been calculated by the above formula for a single \f$\bf G \f$, its values at
305  * points \f${\bf R}^{-T} {\bf G}\f$, \f$\forall \, {\bf R}\f$ are given by the update formula
306  * \f[
307  * \hat f_{\mathrm{sym}} ({\bf R}^{-T} {\bf G}) = e^{-i {\bf R}^{-T} {\bf G} {\bf t}}
308  * \hat{f}_{\mathrm{sym}} ({\bf G})\,,
309  * \f]
310  *
311  * which follows by using that \f$f_{\mathrm{sym}}({\bf \hat P} {\bf x}) = f_{\mathrm{sym}}({\bf x})\f$:
312  * \f{eqnarray*}{
313  * f_{\mathrm{sym}}(\hat{P}{\bf x}) &=& \sum_G e^{i G ({\bf R}{\bf x} + {\bf t} )} \hat{f}_{\mathrm{sym}}({\bf G}) \ \
314  &=& \sum_G e^{i {\bf G} {\bf x}} e^{i {\bf R}^{-T} {\bf G} {\bf t}} \hat{f}_{\mathrm{sym}}({\bf R}^{-T} {\bf G}) \,.
315  * \f}
316  */
317  void symmetrize_function(double_complex* f_pw__,
318  remap_gvec_to_shells const& remap_gvec__,
319  mdarray<double_complex, 3> const& sym_phase_factors__) const;
320

378 inline Unit_cell_symmetry::Unit_cell_symmetry(matrix3d<double>& lattice_vectors__,
379  int num_atoms__,
380  mdarray<double, 2>& positions__,
381  mdarray<double, 2>& spins__,
382  std::vector<int>& types__,
383  double tolerance__)
384  : lattice_vectors_(lattice_vectors__)
385  , num_atoms_(num_atoms__)
386  , types_(types__)
387  , tolerance_(tolerance__)
388 {
389  PROFILE("sirius::Unit_cell_symmetry::Unit_cell_symmetry");
390 
391  if (lattice_vectors__.det() < 0) {
392  std::stringstream s;
393  s << "spglib requires positive determinant for a matrix of lattice vectors";
394  TERMINATE(s);
395  }
396 
397  double lattice[3][3];
398  for (int i: {0, 1, 2}) {
399  for (int j: {0, 1, 2}) {
400  lattice[i][j] = lattice_vectors_(i, j);
401  }
402  }
403  positions_ = mdarray<double, 2>(3, num_atoms_);
404  for (int ia = 0; ia < num_atoms_; ia++) {
405  for (int x: {0, 1, 2}) {
406  positions_(x, ia) = positions__(x, ia);
407  }
408  }
409 
410  utils::timer t1("sirius::Unit_cell_symmetry::Unit_cell_symmetry|spg");
411  spg_dataset_ = spg_get_dataset(lattice, (double(*)[3])&positions_(0, 0), &types_[0], num_atoms_, tolerance_);
412  if (spg_dataset_ == NULL) {
413  TERMINATE("spg_get_dataset() returned NULL");
414  }
415 
416  if (spg_dataset_->spacegroup_number == 0) {
417  TERMINATE("spg_get_dataset() returned 0 for the space group");
418  }
419 
420  if (spg_dataset_->n_atoms != num_atoms__) {
421  std::stringstream s;
422  s << "spg_get_dataset() returned wrong number of atoms (" << spg_dataset_->n_atoms << ")" << std::endl
423  << "expected number of atoms is " << num_atoms__;
424  TERMINATE(s);
425  }
426  t1.stop();
427 
428  inverse_lattice_vectors_ = inverse(lattice_vectors_);
429 
430  utils::timer t2("sirius::Unit_cell_symmetry::Unit_cell_symmetry|sym1");
431  for (int isym = 0; isym < spg_dataset_->n_operations; isym++) {
432  space_group_symmetry_descriptor sym_op;
433 
434  sym_op.R = matrix3d<int>(spg_dataset_->rotations[isym]);
435  sym_op.invRT = transpose(inverse(sym_op.R));
436  sym_op.t = vector3d<double>(spg_dataset_->translations[isym][0],
437  spg_dataset_->translations[isym][1],
438  spg_dataset_->translations[isym][2]);
439  int p = sym_op.R.det();
440  if (!(p == 1 || p == -1)) {
441  TERMINATE("wrong rotation matrix");
442  }
443  sym_op.proper = p;
444  sym_op.rotation = lattice_vectors_ * matrix3d<double>(sym_op.R * p) * inverse_lattice_vectors_;
445  sym_op.euler_angles = euler_angles(sym_op.rotation);
446 
447  for (int i = 0; i < spg_dataset_->n_operations; i++) {
448  auto m = matrix3d<int>(spg_dataset_->rotations[isym]) * matrix3d<int>(spg_dataset_->rotations[i]);
449 
450  if (m(0, 0) == 1 && m(1, 1) == 1 && m(2, 2) == 1 &&
451  m(0, 1) == 0 && m(1, 0) == 0 &&
452  m(0, 2) == 0 && m(2, 0) == 0 &&
453  m(1, 2) == 0 && m(2, 1) == 0)
454  {
455  sym_op.invR = matrix3d<int>(spg_dataset_->rotations[i]);
456  break;
457  }
458  }
459 
460  space_group_symmetry_.push_back(sym_op);
461  }
462  t2.stop();
463 
464  utils::timer t3("sirius::Unit_cell_symmetry::Unit_cell_symmetry|sym2");
465  sym_table_ = mdarray<int, 2>(num_atoms_, num_spg_sym());
466  /* loop over spatial symmetries */
467  #pragma omp parallel for schedule(static)
468  for (int isym = 0; isym < num_spg_sym(); isym++) {
469  for (int ia = 0; ia < num_atoms_; ia++) {
470  auto R = space_group_symmetry(isym).R;
471  auto t = space_group_symmetry(isym).t;
472  /* spatial transform */
473  vector3d<double> pos(positions__(0, ia), positions__(1, ia), positions__(2, ia));
474  auto v = reduce_coordinates(R * pos + t);
475  auto distance = [](const vector3d<double>& a, const vector3d<double>& b) {
476  auto diff = a-b;
477  vector3d<double> d;
478  for (int i=0; i < 3; ++i) {
479  double dl = std::abs(diff[i]);
480  d[i] = std::min(dl, 1-dl);
481  }
482  return std::sqrt(d[0]*d[0] + d[1]*d[1] + d[2]*d[2]);
483  };
484 
485  int ja = -1;
486  /* check for equivalent atom */
487  for (int k = 0; k < num_atoms_; k++) {
488  vector3d<double> pos1(positions__(0, k), positions__(1, k), positions__(2, k));
489  if (distance(v.first, pos1) < tolerance_) {
490  ja = k;
491  break;
492  }
493  }
494 
495  if (ja == -1) {
496  TERMINATE("equivalent atom was not found");
497  }
498  sym_table_(ia, isym) = ja;
499  }
500  }
501  t3.stop();
502 
503  utils::timer t4("sirius::Unit_cell_symmetry::Unit_cell_symmetry|sym3");
504  /* loop over spatial symmetries */
505  for (int isym = 0; isym < num_spg_sym(); isym++) {
506  /* loop over spin symmetries */
507  for (int jsym = 0; jsym < num_spg_sym(); jsym++) {
508  /* take proper part of rotation matrix */
509  auto Rspin = space_group_symmetry(jsym).rotation;
510 
511  int n{0};
512  /* check if all atoms transfrom under spatial and spin symmetries */
513  for (int ia = 0; ia < num_atoms_; ia++) {
514  int ja = sym_table_(ia, isym);
515 
516  /* now check that vector field transforms from atom ia to atom ja */
517  /* vector field of atom is expected to be in Cartesian coordinates */
518  auto vd = Rspin * vector3d<double>(spins__(0, ia), spins__(1, ia), spins__(2, ia)) -
519  vector3d<double>(spins__(0, ja), spins__(1, ja), spins__(2, ja));
520 
521  if (vd.length() < 1e-10) {
522  n++;
523  }
524  }
525  /* if all atoms transform under spin rotaion, add it to a list */
526  if (n == num_atoms_) {
527  magnetic_group_symmetry_descriptor mag_op;
528  mag_op.spg_op = space_group_symmetry(isym);
529  mag_op.isym = isym;
530  mag_op.spin_rotation = Rspin;
531  mag_op.spin_rotation_inv = inverse(Rspin);
532  magnetic_group_symmetry_.push_back(mag_op);
533  break;
534  }
535  }
536  }
537  t4.stop();
538 }

540 inline matrix3d<double> Unit_cell_symmetry::rot_mtrx_cart(vector3d<double> euler_angles) const
541 {
542  double alpha = euler_angles[0];
543  double beta = euler_angles[1];
544  double gamma = euler_angles[2];
545 
546  matrix3d<double> rm;
547  rm(0, 0) = std::cos(alpha) * std::cos(beta) * std::cos(gamma) - std::sin(alpha) * std::sin(gamma);
548  rm(0, 1) = -std::cos(gamma) * std::sin(alpha) - std::cos(alpha) * std::cos(beta) * std::sin(gamma);
549  rm(0, 2) = std::cos(alpha) * std::sin(beta);
550  rm(1, 0) = std::cos(beta) * std::cos(gamma) * std::sin(alpha) + std::cos(alpha) * std::sin(gamma);
551  rm(1, 1) = std::cos(alpha) * std::cos(gamma) - std::cos(beta) * std::sin(alpha) * std::sin(gamma);
552  rm(1, 2) = std::sin(alpha) * std::sin(beta);
553  rm(2, 0) = -std::cos(gamma) * std::sin(beta);
554  rm(2, 1) = std::sin(beta) * std::sin(gamma);
555  rm(2, 2) = std::cos(beta);
556 
557  return rm;

568  }

718 inline void Unit_cell_symmetry::symmetrize_function(double_complex* f_pw__,

720  mdarray<double_complex, 3> const& sym_phase_factors__) const
721 {
722  PROFILE("sirius::Unit_cell_symmetry::symmetrize_function_pw");
723

732  {
733  return sym_phase_factors__(0, G[0], isym) *
734  sym_phase_factors__(1, G[1], isym) *
735  sym_phase_factors__(2, G[2], isym);
736  };

770  }
771  } /* loop over symmetries */

789  } /* loop over symmetries */
790  }
791  } /* loop over igloc */
792  }

1135 /** \page sym Symmetry
1136  * \section section1 Definition of symmetry operation
1137  *
1138  * SIRIUS uses Spglib to find the spacial symmetry operations. Spglib defines symmetry operation in fractional
1139  * coordinates:
1140  * \f[
1141  * {\bf x'} = \{ {\bf R} | {\bf t} \} {\bf x} \equiv {\bf R}{\bf x} + {\bf t}
1142  * \f]
1143  * where \b R is the proper or improper rotation matrix with elements equal to -1,0,1 and determinant of 1
1144  * (pure rotation) or -1 (rotoreflection) and \b t is the fractional translation, associated with the symmetry
1145  * operation. The inverse of the symmetry operation is:
1146  * \f[
1147  * {\bf x} = \{ {\bf R} | {\bf t} \}^{-1} {\bf x'} = {\bf R}^{-1} ({\bf x'} - {\bf t}) =
1148  * {\bf R}^{-1} {\bf x'} - {\bf R}^{-1} {\bf t}
1149  * \f]
1150  *
1151  * We will always use an \a active transformation (transformation of vectors or functions) and never a passive
1152  * transformation (transformation of coordinate system). However one should remember definition of the function
1153  * transformation:
1154  * \f[
1155  * \hat {\bf P} f({\bf r}) \equiv f(\hat {\bf P}^{-1} {\bf r})
1156  * \f]
1157  *
1158  * It is straightforward to get the rotation matrix in Cartesian coordinates. We know how the vector in Cartesian
1159  * coordinates is obtained from the vector in fractional coordinates:
1160  * \f[
1161  * {\bf v} = {\bf L} {\bf x}
1162  * \f]
1163  * where \b L is the 3x3 matrix which clomuns are three lattice vectors. The backward transformation is simply
1164  * \f[
1165  * {\bf x} = {\bf L}^{-1} {\bf v}
1166  * \f]
1167  * Now we write rotation operation in fractional coordinates and apply the backward transformation to Cartesian
1168  * coordinates:
1169  * \f[
1170  * {\bf x'} = {\bf R}{\bf x} \rightarrow {\bf L}^{-1} {\bf v'} = {\bf R} {\bf L}^{-1} {\bf v}
1171  * \f]
1172  * from which we derive the rotation operation in Cartesian coordinates:
1173  * \f[
1174  * {\bf v'} = {\bf L} {\bf R} {\bf L}^{-1} {\bf v}
1175  * \f]
1176  */
