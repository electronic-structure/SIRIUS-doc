29 void K_point::initialize()
30 {
31  PROFILE("sirius::K_point::initialize");
32 
33  zil_.resize(ctx_.lmax_apw() + 1);
34  for (int l = 0; l <= ctx_.lmax_apw(); l++) {
35  zil_[l] = std::pow(double_complex(0, 1), l);
36  }
37 
38  l_by_lm_ = utils::l_by_lm(ctx_.lmax_apw());
39 
40  int bs = ctx_.cyclic_block_size();
41 
42  if (ctx_.control().use_second_variation_ && ctx_.full_potential()) {
43  assert(ctx_.num_fv_states() > 0);
44  fv_eigen_values_.resize(ctx_.num_fv_states());
45  }
46 
47  /* In case of collinear magnetism we store only non-zero spinor components.
48  *
49  * non magnetic case:
50  * .---.
51  * | |
52  * .---.
53  *
54  * collinear case:
55  * .---. .---.
56  * |uu | 0 |uu |
57  * .---.---. -> .---.
58  * 0 |dd | |dd |
59  * .---. .---.
60  *
61  * non collinear case:
62  * .-------.
63  * | |
64  * .-------.
65  * | |
66  * .-------.
67  */
68  int nst = ctx_.num_bands();
69 
70  auto mem_type_evp = (ctx_.std_evp_solver_type() == ev_solver_t::magma) ? memory_t::host_pinned : memory_t::host;
71  auto mem_type_gevp = (ctx_.gen_evp_solver_type() == ev_solver_t::magma) ? memory_t::host_pinned : memory_t::host;
72 
73  /* build a full list of G+k vectors for all MPI ranks */
74  generate_gkvec(ctx_.gk_cutoff());
75  /* build a list of basis functions */
76  generate_gklo_basis();
77 
78  if (ctx_.full_potential()) {
79  if (ctx_.control().use_second_variation_) {
80  if (ctx_.need_sv()) {
81  /* in case of collinear magnetism store pure up and pure dn components, otherwise store the full matrix
82  */
83  for (int is = 0; is < ctx_.num_spin_dims(); is++) {
84  sv_eigen_vectors_[is] = dmatrix<double_complex>(nst, nst, ctx_.blacs_grid(), bs, bs, mem_type_evp);
85  }
86  }
87  /* allocate fv eien vectors */
88  fv_eigen_vectors_slab_ = std::unique_ptr<Wave_functions>(
89  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(),
90  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); },
91  ctx_.num_fv_states(), ctx_.preferred_memory_t()));
92 
93  fv_eigen_vectors_slab_->pw_coeffs(0).prime().zero();
94  fv_eigen_vectors_slab_->mt_coeffs(0).prime().zero();
95  /* starting guess for wave-functions */
96  for (int i = 0; i < ctx_.num_fv_states(); i++) {
97  for (int igloc = 0; igloc < gkvec().gvec_count(comm().rank()); igloc++) {
98  int ig = igloc + gkvec().gvec_offset(comm().rank());
99  if (ig == i) {
100  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 1.0;
101  }
102  if (ig == i + 1) {
103  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.5;
104  }
105  if (ig == i + 2) {
106  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.125;
107  }
108  }
109  }
110  if (ctx_.iterative_solver_input().type_ == "exact") {
111  /* ELPA needs a full matrix of eigen-vectors as it uses it as a work space */
112  fv_eigen_vectors_ = dmatrix<double_complex>(gklo_basis_size(), gklo_basis_size(), ctx_.blacs_grid(), bs,
113  bs, mem_type_gevp);
114  } else {
115  int ncomp = ctx_.iterative_solver_input().num_singular_;
116  if (ncomp < 0) {
117  ncomp = ctx_.num_fv_states() / 2;
118  }
119 
120  singular_components_ = std::unique_ptr<Wave_functions>(
121  new Wave_functions(gkvec_partition(), ncomp, ctx_.preferred_memory_t()));
122  singular_components_->pw_coeffs(0).prime().zero();
123  /* starting guess for wave-functions */
124  for (int i = 0; i < ncomp; i++) {
125  for (int igloc = 0; igloc < gkvec().count(); igloc++) {
126  int ig = igloc + gkvec().offset();
127  if (ig == i) {
128  singular_components_->pw_coeffs(0).prime(igloc, i) = 1.0;
129  }
130  if (ig == i + 1) {
131  singular_components_->pw_coeffs(0).prime(igloc, i) = 0.5;
132  }
133  if (ig == i + 2) {
134  singular_components_->pw_coeffs(0).prime(igloc, i) = 0.125;
135  }
136  }
137  }
138  if (ctx_.control().print_checksum_) {
139  auto cs = singular_components_->checksum_pw(device_t::CPU, 0, 0, ncomp);
140  if (comm().rank() == 0) {
141  utils::print_checksum("singular_components", cs);
142  }
143  }
144  }
145 
146  fv_states_ = std::unique_ptr<Wave_functions>(
147  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(),
148  [this](int ia) { return unit_cell_.atom(ia).mt_basis_size(); }, ctx_.num_fv_states(),
149  ctx_.preferred_memory_t()));
150 
151  spinor_wave_functions_ = std::unique_ptr<Wave_functions>(
152  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(),
153  [this](int ia) { return unit_cell_.atom(ia).mt_basis_size(); }, nst,
154  ctx_.preferred_memory_t(), ctx_.num_spins()));
155  } else {
156  throw std::runtime_error("not implemented");
157  }
158  } else {
159  spinor_wave_functions_ = std::unique_ptr<Wave_functions>(
160  new Wave_functions(gkvec_partition(), nst, ctx_.preferred_memory_t(), ctx_.num_spins()));
161  if (ctx_.hubbard_correction()) {
162  auto r = unit_cell_.num_wf_with_U();
163  const int num_sc = ctx_.num_mag_dims() == 3 ? 2 : 1;
164  hubbard_wave_functions_ = std::unique_ptr<Wave_functions>(
165  new Wave_functions(gkvec_partition(), r.first, ctx_.preferred_memory_t(), num_sc));
166  }
167  }
168 
169  update();
170 }
