30 void Hubbard_input::read(json const& parser)
31 {
32  if (!parser.count("hubbard")) {
33  return;
34  }
35 
36  auto section = parser["hubbard"];
37 
38  orthogonalize_hubbard_orbitals_ =
39  section.value("orthogonalize_hubbard_wave_functions", orthogonalize_hubbard_orbitals_);
40 
41  normalize_hubbard_orbitals_ =
42  section.value("normalize_hubbard_wave_functions", normalize_hubbard_orbitals_);
43 
44  simplified_hubbard_correction_ =
45  section.value("simplified_hubbard_correction", simplified_hubbard_correction_);
46 
47  if (section.count("projection_method")) {
48  std::string projection_method = parser["hubbard"]["projection_method"].get<std::string>();
49  if (projection_method == "file") {
50  // they are provided by a external file
51  if (parser["hubbard"].count("wave_function_file")) {
52  this->wave_function_file_ = parser["hubbard"]["wave_function_file"].get<std::string>();
53  this->projection_method_ = 1;
54  } else {
55  throw std::runtime_error(
56  "The hubbard projection method 'file' requires the option 'wave_function_file' to be defined");
57  }
58  }
59 
60  if (projection_method == "pseudo") {
61  this->projection_method_ = 2;
62  }
63  }
64 
65  hubbard_U_plus_V_ = parser["hubbard"].value("hubbard_u_plus_v", hubbard_U_plus_V_);
66 
67  auto v = parser["unit_cell"]["atom_types"].get<std::vector<std::string>>();
68 
69  for (auto& label : v) {
70  if (section.count(label)) {
71  if (species_with_U.count(label)) {
72  throw std::runtime_error("U-correction for atom " + label + " has already been defined");
73  }
74 
75  hubbard_correction_ = true;
76 
77  hubbard_orbital_t ho;
78 
79  ho.coeff[0] = section[label].value("U", ho.coeff[0]);
80  ho.coeff[1] = section[label].value("J", ho.coeff[1]);
81  ho.coeff[2] = section[label].value("B", ho.coeff[2]);
82  ho.coeff[2] = section[label].value("E2", ho.coeff[2]);
83  ho.coeff[3] = section[label].value("E3", ho.coeff[3]);
84  ho.coeff[4] = section[label].value("alpha", ho.coeff[4]);
85  ho.coeff[5] = section[label].value("beta", ho.coeff[5]);
86 
87  /* now convert eV in Ha */
88  for (int s = 0; s < 6; s++) {
89  ho.coeff[s] /= ha2ev;
90  }
91  ho.l = section[label].value("l", ho.l);
92  ho.n = section[label].value("n", ho.n);
93 
94  if (ho.l == -1 || ho.n == -1) {
95  std::string level;
96  section[label].value("hubbard_orbital", level);
97  std::map<char, int> const map_l = {{'s', 0},{'p', 1}, {'d', 2}, {'f', 3}};
98 
99  std::istringstream iss(std::string(1, level[0]));
100  iss >> ho.n;
101  if (ho.n <= 0 || iss.fail()) {
102  std::stringstream s;
103  s << "wrong principal quantum number : " << std::string(1, level[0]);
104  throw std::runtime_error(s.str());
105  }
106  ho.l = map_l.at(level[1]);
107  }
108 
109  if (!section[label].count("occupancy")) {
110  throw std::runtime_error("initial occupancy of the Hubbard orbital is not set");
111  }
112  ho.occupancy = section[label].value("occupancy", ho.occupancy);
113  ho.initial_occupancy = section[label].value("initial_occupancy", ho.initial_occupancy);
114 
115  int sz = static_cast<int>(ho.initial_occupancy.size());
116  int lmmax = 2 * ho.l + 1;
117 
118  if (!(sz == 0 || sz == lmmax || sz == 2 * lmmax)) {
119  std::stringstream s;
120  s << "wrong size of initial occupacies vector (" << sz << ") for l = " << ho.l;
121  throw std::runtime_error(s.str());
122  }
123 
124  species_with_U[label] = ho;
125  }
126  }
127 }
