197 inline void Band::get_singular_components(K_point& kp__, Hamiltonian& H__) const
198 {
199  PROFILE("sirius::Band::get_singular_components");
200 
201  auto o_diag_tmp = H__.get_o_diag(&kp__, ctx_.theta_pw(0).real());
202 
203  mdarray<double, 2> o_diag(kp__.num_gkvec_loc(), 1, memory_t::host, "o_diag");
204  mdarray<double, 1> diag1(kp__.num_gkvec_loc(), memory_t::host, "diag1");
205  for (int ig = 0; ig < kp__.num_gkvec_loc(); ig++) {
206  o_diag[ig] = o_diag_tmp[ig];
207  diag1[ig] = 1;
208  }
209 
210  if (ctx_.processing_unit() == GPU) {
211  o_diag.allocate(memory_t::device).copy_to(memory_t::device);
212  diag1.allocate(memory_t::device).copy_to(memory_t::device);
213  }
214 
215  auto& psi = kp__.singular_components();
216 
217  int ncomp = psi.num_wf();
218 
219  if (ctx_.comm().rank() == 0 && ctx_.control().verbosity_ >= 3) {
220  printf("number of singular components: %i\n", ncomp);
221  }
222 
223  auto& itso = ctx_.iterative_solver_input();
224 
225  int num_phi = itso.subspace_size_ * ncomp;
226 
227  Wave_functions phi(kp__.gkvec_partition(), num_phi);
228  Wave_functions ophi(kp__.gkvec_partition(), num_phi);
229  Wave_functions opsi(kp__.gkvec_partition(), ncomp);
230  Wave_functions res(kp__.gkvec_partition(), ncomp);
231 
232  int bs = ctx_.cyclic_block_size();
233 
234  dmatrix<double_complex> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
235  dmatrix<double_complex> ovlp_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
236  dmatrix<double_complex> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
237 
238  if (ctx_.processing_unit() == device_t::GPU) {
239  psi.pw_coeffs(0).allocate(memory_t::device);
240  psi.pw_coeffs(0).copy_to(memory_t::device, 0, ncomp);
241  phi.pw_coeffs(0).allocate(memory_t::device);
242  res.pw_coeffs(0).allocate(memory_t::device);
243  ophi.pw_coeffs(0).allocate(memory_t::device);
244  opsi.pw_coeffs(0).allocate(memory_t::device);
245  if (ctx_.blacs_grid().comm().size() == 1) {
246  evec.allocate(memory_t::device);
247  ovlp.allocate(memory_t::device);
248  }
249  }
250 
251  std::vector<double> eval(ncomp, 1e10);
252  std::vector<double> eval_old(ncomp);
253 
254  if (ctx_.control().print_checksum_) {
255  auto cs2 = phi.checksum(ctx_.processing_unit(), 0, 0, ncomp);
256  if (kp__.comm().rank() == 0) {
257  utils::print_checksum("phi", cs2);
258  }
259  }
260 
261  phi.copy_from(ctx_.processing_unit(), ncomp, psi, 0, 0, 0, 0);
262 
263  /* current subspace size */
264  int N{0};
265 
266  /* number of newly added basis functions */
267  int n = ncomp;
268 
269  if (ctx_.control().verbosity_ >= 3 && kp__.comm().rank() == 0) {
270  printf("iterative solver tolerance: %18.12f\n", ctx_.iterative_solver_tolerance());
271  }
272 
273  if (kp__.comm().rank() == 0 && ctx_.control().print_memory_usage_) {
274  MEMORY_USAGE_INFO();
275  }
276 
277  auto std_solver = ctx_.std_evp_solver<double_complex>();
278 
279  /* start iterative diagonalization */
280  for (int k = 0; k < itso.num_steps_; k++) {
281  /* apply Hamiltonian and overlap operators to the new basis functions */
282  H__.apply_fv_h_o(&kp__, true, false, N, n, phi, nullptr, &ophi);
283 
284  if (ctx_.control().verification_ >= 1) {
285  dmatrix<double_complex> tmp;
286  set_subspace_mtrx(0, N + n, phi, ophi, ovlp, tmp);
287 
288  if (ctx_.control().verification_ >= 2) {
289  ovlp.serialize("overlap", N + n);
290  }
291 
292  double max_diff = check_hermitian(ovlp, N + n);
293  if (max_diff > 1e-12) {
294  std::stringstream s;
295  s << "overlap matrix is not hermitian, max_err = " << max_diff;
296  TERMINATE(s);
297  }
298  }
299 
300  orthogonalize(ctx_.processing_unit(), 0, phi, ophi, N, n, ovlp, res);
301 
302  /* setup eigen-value problem
303  * N is the number of previous basis functions
304  * n is the number of new basis functions */
305  set_subspace_mtrx(N, n, phi, ophi, ovlp, ovlp_old);
306 
307  if (ctx_.control().verification_ >= 1) {
308 
309  if (ctx_.control().verification_ >= 2) {
310  ovlp.serialize("overlap_ortho", N + n);
311  }
312 
313  double max_diff = check_hermitian(ovlp, N + n);
314  if (max_diff > 1e-12) {
315  std::stringstream s;
316  s << "overlap matrix is not hermitian, max_err = " << max_diff;
317  TERMINATE(s);
318  }
319  }
320 
321  /* increase size of the variation space */
322  N += n;
323 
324  eval_old = eval;
325 
326  /* solve standard eigen-value problem with the size N */
327  if (std_solver->solve(N, ncomp, ovlp, eval.data(), evec)) {
328  std::stringstream s;
329  s << "error in diagonalziation";
330  TERMINATE(s);
331  }
332 
333  for (auto e : eval) {
334  if (e < 0) {
335  std::stringstream s;
336  s << "overlap matrix is not positively defined";
337  TERMINATE(s);
338  }
339  }
340 
341  if (ctx_.control().verbosity_ >= 3 && kp__.comm().rank() == 0) {
342  printf("step: %i, current subspace size: %i, maximum subspace size: %i\n", k, N, num_phi);
343  if (ctx_.control().verbosity_ >= 4) {
344  for (int i = 0; i < ncomp; i++) {
345  printf("eval[%i]=%20.16f, diff=%20.16f\n", i, eval[i], std::abs(eval[i] - eval_old[i]));
346  }
347  }
348  }
349 
350  /* don't compute residuals on last iteration */
351  if (k != itso.num_steps_ - 1) {
352  /* get new preconditionined residuals, and also opsi and psi as a by-product */
353  n = residuals(&kp__, 0, N, ncomp, eval, eval_old, evec, ophi, phi, opsi, psi, res, o_diag, diag1);
354  }
355 
356  /* check if we run out of variational space or eigen-vectors are converged or it's a last iteration */
357  if (N + n > num_phi || n <= itso.min_num_res_ || k == (itso.num_steps_ - 1)) {
358  utils::timer t1("sirius::Band::get_singular_components|update_phi");
359  /* recompute wave-functions */
360  /* \Psi_{i} = \sum_{mu} \phi_{mu} * Z_{mu, i} */
361  transform(ctx_.processing_unit(), 0, phi, 0, N, evec, 0, 0, psi, 0, ncomp);
362 
363  /* exit the loop if the eigen-vectors are converged or this is a last iteration */
364  if (n <= itso.min_num_res_ || k == (itso.num_steps_ - 1)) {
365  break;
366  } else { /* otherwise, set Psi as a new trial basis */
367  if (ctx_.control().verbosity_ >= 3 && kp__.comm().rank() == 0) {
368  printf("subspace size limit reached\n");
369  }
370 
371  if (itso.converge_by_energy_) {
372  transform(ctx_.processing_unit(), 0, ophi, 0, N, evec, 0, 0, opsi, 0, ncomp);
373  }
374 
375  ovlp_old.zero();
376  for (int i = 0; i < ncomp; i++) {
377  ovlp_old.set(i, i, eval[i]);
378  }
379  /* update basis functions */
380  phi.copy_from(ctx_.processing_unit(), ncomp, psi, 0, 0, 0, 0);
381  ophi.copy_from(ctx_.processing_unit(), ncomp, opsi, 0, 0, 0, 0);
382  /* number of basis functions that we already have */
383  N = ncomp;
384  }
385  }
386  /* expand variational subspace with new basis vectors obtatined from residuals */
387  phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, N);
388  }
389 
390  if (ctx_.processing_unit() == device_t::GPU) {
391  psi.pw_coeffs(0).copy_to(memory_t::host, 0, ncomp);
392  psi.pw_coeffs(0).deallocate(memory_t::device);
393  }
394 
395  if (ctx_.control().verbosity_ >= 2 && kp__.comm().rank() == 0) {
396  printf("lowest and highest eigen-values of the singular components: %20.16f %20.16f\n", eval.front(), eval.back());
397  }
398 }

400 inline void Band::diag_full_potential_first_variation_davidson(K_point& kp__, Hamiltonian& H__) const
401 {
402  PROFILE("sirius::Band::diag_fv_davidson");
403 
404  H__.local_op().prepare(kp__.gkvec_partition());
405 
406  get_singular_components(kp__, H__);
407 
408  auto h_diag = H__.get_h_diag(&kp__, H__.local_op().v0(0), ctx_.theta_pw(0).real());
409  auto o_diag = H__.get_o_diag(&kp__, ctx_.theta_pw(0).real());
410 
411  /* short notation for number of target wave-functions */
412  int num_bands = ctx_.num_fv_states();
413 
414  auto& itso = ctx_.iterative_solver_input();
415 
416  /* short notation for target wave-functions */
417  auto& psi = kp__.fv_eigen_vectors_slab();
418 
419  /* total number of local orbitals */
420  int nlo = ctx_.unit_cell().mt_lo_basis_size();
421 
422  /* number of singular components */
423  int ncomp = kp__.singular_components().num_wf();
424 
425  /* number of auxiliary basis functions */
426  int num_phi = nlo + ncomp + itso.subspace_size_ * num_bands;
427  /* sanity check */
428  if (num_phi >= kp__.num_gkvec()) {
429  TERMINATE("subspace is too big");
430  }
431 
432  if (ctx_.control().verbosity_ >= 2 && kp__.comm().rank() == 0) {
433  printf("iterative solver tolerance: %18.12f\n", ctx_.iterative_solver_tolerance());
434  }
435 
436  /* allocate wave-functions */
437  Wave_functions phi(kp__.gkvec_partition(), unit_cell_.num_atoms(),
438  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi);
439  Wave_functions hphi(kp__.gkvec_partition(), unit_cell_.num_atoms(),
440  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi);
441  Wave_functions ophi(kp__.gkvec_partition(), unit_cell_.num_atoms(),
442  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_phi);
443  Wave_functions hpsi(kp__.gkvec_partition(), unit_cell_.num_atoms(),
444  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_bands);
445  Wave_functions opsi(kp__.gkvec_partition(), unit_cell_.num_atoms(),
446  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, num_bands);
447 
448  /* residuals */
449  /* res is also used as a temporary array in orthogonalize() and the first time nlo + ncomp + num_bands
450  * states will be orthogonalized */
451  Wave_functions res(kp__.gkvec_partition(), unit_cell_.num_atoms(),
452  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); }, nlo + ncomp + num_bands);
453 
454  //auto mem_type = (gen_evp_solver_->type() == ev_magma) ? memory_t::host_pinned : memory_t::host;
455 
456  int bs = ctx_.cyclic_block_size();
457 
458  dmatrix<double_complex> hmlt(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
459  dmatrix<double_complex> ovlp(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
460  dmatrix<double_complex> evec(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
461  dmatrix<double_complex> hmlt_old(num_phi, num_phi, ctx_.blacs_grid(), bs, bs);
462 
463  /* add pure local orbitals to the basis */
464  if (nlo) {
465  phi.pw_coeffs(0).zero(memory_t::host, 0, nlo);
466  phi.mt_coeffs(0).zero(memory_t::host, 0, nlo);
467  for (int ialoc = 0; ialoc < phi.spl_num_atoms().local_size(); ialoc++) {
468  int ia = phi.spl_num_atoms()[ialoc];
469  for (int xi = 0; xi < unit_cell_.atom(ia).mt_lo_basis_size(); xi++) {
470  phi.mt_coeffs(0).prime(phi.offset_mt_coeffs(ialoc) + xi, unit_cell_.atom(ia).offset_lo() + xi) = 1.0;
471  }
472  }
473  }
474 
475  /* add singular components to the basis */
476  if (ncomp != 0) {
477  phi.mt_coeffs(0).zero(memory_t::host, nlo, ncomp);
478  for (int j = 0; j < ncomp; j++) {
479  std::memcpy(phi.pw_coeffs(0).prime().at(memory_t::host, 0, nlo + j),
480  kp__.singular_components().pw_coeffs(0).prime().at(memory_t::host, 0, j),
481  phi.pw_coeffs(0).num_rows_loc() * sizeof(double_complex));
482  }
483  }
484 
485  if (ctx_.processing_unit() == device_t::GPU) {
486  psi.allocate(spin_idx(0), memory_t::device);

599  phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, nlo + ncomp + num_bands);
600  /* number of basis functions that we already have */
601  N = nlo + ncomp;
602  n += num_bands;
603  }
604  } else {
605  /* expand variational subspace with new basis vectors obtatined from residuals */
606  phi.copy_from(ctx_.processing_unit(), n, res, 0, 0, 0, N);
607  }
608  }
609 
610  if (ctx_.processing_unit() == device_t::GPU) {
611  psi.pw_coeffs(0).copy_to(memory_t::host, 0, num_bands);
612  psi.mt_coeffs(0).copy_to(memory_t::host, 0, num_bands);
613  psi.deallocate(0, memory_t::device);
614  }
615  kp__.set_fv_eigen_values(&eval[0]);
