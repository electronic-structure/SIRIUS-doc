33static std::pair<std::vector<int>, std::vector<vector3d<int>>>
34find_sym_atom(int num_atoms__, sddk::mdarray<double, 2> const& positions__, matrix3d<int> const& R__,
35 vector3d<double> const& t__, double tolerance__, bool inverse__ = false)
36{
37 PROFILE("sirius::find_sym_atom");
38
39 std::vector<int> sym_atom(num_atoms__);
40 std::vector<vector3d<int>> sym_atom_T(num_atoms__);
41
42 auto distance = [](const vector3d<double>& a, const vector3d<double>& b)
43 {
44 auto diff = a - b;
45 for (int x: {0, 1, 2}) {
46 double dl = std::abs(diff[x]);
47 diff[x] = std::min(dl, 1 - dl);
48 }
49 return diff.length();
50 };
51
52 for (int ia = 0; ia < num_atoms__; ia++) {
53 /* position of atom ia */
54 vector3d<double> pos(positions__(0, ia), positions__(1, ia), positions__(2, ia));
55 /* apply crystal symmetry */
56 auto v = (inverse__) ? reduce_coordinates(dot(inverse(R__), pos - t__)) : reduce_coordinates(dot(R__, pos) + t__);
57 double d0{1e10};
58 double j0{-1};
59 vector3d<double> p0;
60
61 int ja{-1};
62 /* check for equivalent atom; remember that the atomic positions are not necessarily in [0,1) interval
63 and the reduction of coordinates is required */
64 for (int k = 0; k < num_atoms__; k++) {
65 vector3d<double> pos1(positions__(0, k), positions__(1, k), positions__(2, k));
66 /* find the distance between atom k and trasformed atoms ia */
67 double dist = distance(v.first, reduce_coordinates(pos1).first);
68 if (dist < tolerance__) {
69 ja = k;
70 break;
71 }
72 if (dist < d0) {
73 d0 = dist;
74 j0 = k;
75 p0 = pos1;
76 }
77 }
78
79 if (ja == -1) {
80 std::stringstream s;
81 s << "equivalent atom was not found\n"
82 << " initial atom: " << ia << ", position : " << pos << ", reduced: " << v.first << "\n"
83 << " nearest atom: " << j0 << ", position : " << p0 << ", reduced: " << reduce_coordinates(p0).first << "\n"
84 << " distance between atoms: " << d0 << "\n"
85 << " tolerance: " << tolerance__;
86 RTE_THROW(s);
87 }
88 sym_atom[ia] = ja;
89 sym_atom_T[ia] = v.second;
90 }
91 return std::make_pair(sym_atom, sym_atom_T);
92}

94static space_group_symmetry_descriptor
95get_spg_sym_op(int isym_spg__, SpglibDataset* spg_dataset__, matrix3d<double> const& lattice_vectors__, int num_atoms__,
96 sddk::mdarray<double, 2> const& positions__, double tolerance__)
97{
98 space_group_symmetry_descriptor sym_op;
99
100 auto inverse_lattice_vectors = inverse(lattice_vectors__);
101
102 /* rotation matrix in lattice coordinates */
103 sym_op.R = matrix3d<int>(spg_dataset__->rotations[isym_spg__]);
104 /* sanity check */
105 int p = sym_op.R.det();
106 if (!(p == 1 || p == -1)) {
107 RTE_THROW("wrong rotation matrix");
108 }
109 /* inverse of the rotation matrix */
110 sym_op.invR = inverse(sym_op.R);
111 /* inverse transpose */
112 sym_op.invRT = transpose(sym_op.invR);
113 /* fractional translation */
114 sym_op.t = vector3d<double>(spg_dataset__->translations[isym_spg__][0],
115 spg_dataset__->translations[isym_spg__][1],
116 spg_dataset__->translations[isym_spg__][2]);
117 /* is this proper or improper rotation */
118 sym_op.proper = p;
119 /* rotation in Cartesian coordinates */
120 sym_op.Rc = dot(dot(lattice_vectors__, matrix3d<double>(sym_op.R)), inverse_lattice_vectors);
121 /* proper rotation in Cartesian coordinates */
122 sym_op.Rcp = sym_op.Rc * p;
123 try {
124 /* get Euler angles of the rotation */
125 sym_op.euler_angles = euler_angles(sym_op.Rcp, tolerance__);
126 } catch(std::exception const& e) {
127 std::stringstream s;
128 s << "number of symmetry operations found by spglib: " << spg_dataset__->n_operations << std::endl
129 << "symmetry operation: " << isym_spg__ << std::endl
130 << "rotation matrix in lattice coordinates: " << sym_op.R << std::endl
131 << "rotation matrix in Cartesian coordinates: " << sym_op.Rc << std::endl
132 << "lattice vectors: " << lattice_vectors__ << std::endl
133 << "metric tensor error: " << metric_tensor_error(lattice_vectors__, sym_op.R) << std::endl
134 << std::endl
135 << "possible solution: decrease the spglib_tolerance";
136 RTE_THROW(s, e.what());
137 }
138 try {
139 /* get symmetry related atoms */
140 auto result = find_sym_atom(num_atoms__, positions__, sym_op.R, sym_op.t, tolerance__);
141 sym_op.sym_atom = result.first;
142 result = find_sym_atom(num_atoms__, positions__, sym_op.R, sym_op.t, tolerance__, true);
143 sym_op.inv_sym_atom = result.first;
144 sym_op.inv_sym_atom_T = result.second;
145 for (int ia = 0; ia < num_atoms__; ia++) {
146 int ja = sym_op.sym_atom[ia];
147 if (sym_op.inv_sym_atom[ja] != ia) {
148 std::stringstream s;
149 s << "atom symmetry tables are not consistent" << std::endl
150 << "ia: " << ia << " sym_atom[ia]: " << ja << " inv_sym_atom[sym_atom[ia]]: "
151 << sym_op.inv_sym_atom[ja];
152 RTE_THROW(s);
153 }
154 }
155 } catch(std::exception const& e) {
156 std::stringstream s;
157 s << "R: " << sym_op.R << std::endl
158 << "t: " << sym_op.t << std::endl
159 << "tolerance: " << tolerance__;
160 RTE_THROW(s, e.what());
161 }
162
163 return sym_op;
164}

166static space_group_symmetry_descriptor
167get_identity_spg_sym_op(int num_atoms__)
168{
169 space_group_symmetry_descriptor sym_op;
170
171 sym_op.R = matrix3d<int>({{1, 0, 0}, {0, 1, 0}, {0, 0, 1}});
172 /* inverse of the rotation matrix */
173 sym_op.invR = inverse(sym_op.R);
174 /* inverse transpose */
175 sym_op.invRT = transpose(sym_op.invR);
176 /* fractional translation */
177 sym_op.t = vector3d<double>(0, 0, 0);
178 /* is this proper or improper rotation */
179 sym_op.proper = 1;
180 /* rotation in Cartesian coordinates */
181 sym_op.Rcp = sym_op.Rc = matrix3d<double>({{1, 0, 0}, {0, 1, 0}, {0, 0, 1}});
182 /* get Euler angles of the rotation */
183 sym_op.euler_angles = euler_angles(sym_op.Rc, 1e-10);
184 /* trivial atom symmetry table */
185 sym_op.sym_atom = std::vector<int>(num_atoms__);
186 std::iota(sym_op.sym_atom.begin(), sym_op.sym_atom.end(), 0);
187 /* trivial atom symmetry table for inverse operation */
188 sym_op.inv_sym_atom = std::vector<int>(num_atoms__);
189 std::iota(sym_op.inv_sym_atom.begin(), sym_op.inv_sym_atom.end(), 0);
190 sym_op.inv_sym_atom_T = std::vector<vector3d<int>>(num_atoms__, vector3d<int>(0, 0, 0));
191
192 return sym_op;
193}

195Crystal_symmetry::Crystal_symmetry(matrix3d<double> const& lattice_vectors__, int num_atoms__,
196 int num_atom_types__, std::vector<int> const& types__, sddk::mdarray<double, 2> const& positions__,
197 sddk::mdarray<double, 2> const& spins__, bool spin_orbit__, double tolerance__, bool use_sym__)
198 : lattice_vectors_(lattice_vectors__)
199 , num_atoms_(num_atoms__)
200 , num_atom_types_(num_atom_types__)
201 , types_(types__)
202 , tolerance_(tolerance__)
203{
204 PROFILE("sirius::Crystal_symmetry");
205
206 /* check lattice vectors */
207 if (lattice_vectors__.det() < 0 && use_sym__) {
208 std::stringstream s;
209 s << "spglib requires positive determinant for a matrix of lattice vectors";
210 RTE_THROW(s);
211 }
212
213 /* make inverse */
214 inverse_lattice_vectors_ = inverse(lattice_vectors_);
215
216 double lattice[3][3];
217 for (int i: {0, 1, 2}) {
218 for (int j: {0, 1, 2}) {
219 lattice[i][j] = lattice_vectors_(i, j);
220 }
221 }
222
223 positions_ = sddk::mdarray<double, 2>(3, num_atoms_);
224 positions__ >> positions_;
225
226 magnetization_ = sddk::mdarray<double, 2>(3, num_atoms_);
227 spins__ >> magnetization_;
228
229 PROFILE_START("sirius::Crystal_symmetry|spg");
230 if (use_sym__) {
231 /* make a call to spglib */
232 spg_dataset_ = spg_get_dataset(lattice, (double(*)[3])&positions_(0, 0), &types_[0], num_atoms_, tolerance_);
233 if (spg_dataset_ == NULL) {
234 RTE_THROW("spg_get_dataset() returned NULL");
235 }
236
237 if (spg_dataset_->spacegroup_number == 0) {
238 RTE_THROW("spg_get_dataset() returned 0 for the space group");
239 }
240
241 if (spg_dataset_->n_atoms != num_atoms__) {
242 std::stringstream s;
243 s << "spg_get_dataset() returned wrong number of atoms (" << spg_dataset_->n_atoms << ")" << std::endl
244 << "expected number of atoms is " << num_atoms__;
245 RTE_THROW(s);
246 }
247 }
248 PROFILE_STOP("sirius::Crystal_symmetry|spg");
249
250 if (spg_dataset_) {
251 auto lat_sym = find_lat_sym(lattice_vectors_, tolerance_);
252 /* make a list of crystal symmetries */
253 for (int isym = 0; isym < spg_dataset_->n_operations; isym++) {
254 matrix3d<int> R(spg_dataset_->rotations[isym]);
255 for (auto& e: lat_sym) {
256 if (e == R) {
257 auto sym_op = get_spg_sym_op(isym, spg_dataset_, lattice_vectors__, num_atoms__, positions__, tolerance__);
258 /* add symmetry operation to a list */
259 space_group_symmetry_.push_back(sym_op);
260 }
261 }
262 }
263 } else { /* add only identity element */
264 auto sym_op = get_identity_spg_sym_op(num_atoms__);
265 /* add symmetry operation to a list */
266 space_group_symmetry_.push_back(sym_op);
267 }
268
269 PROFILE_START("sirius::Crystal_symmetry|mag");
270 /* loop over spatial symmetries */
271 for (int isym = 0; isym < num_spg_sym(); isym++) {
272 int jsym0 = 0;
273 int jsym1 = num_spg_sym() - 1;
274 if (spin_orbit__) {
275 jsym0 = jsym1 = isym;
276 }
277 /* loop over spin symmetries */
278 for (int jsym = jsym0; jsym <= jsym1; jsym++) {
279 /* take proper part of rotation matrix */
280 auto Rspin = space_group_symmetry(jsym).Rcp;
281
282 int n{0};
283 /* check if all atoms transform under spatial and spin symmetries */
284 for (int ia = 0; ia < num_atoms_; ia++) {
285 int ja = space_group_symmetry(isym).sym_atom[ia];
286
287 /* now check that vector field transforms from atom ia to atom ja */
288 /* vector field of atom is expected to be in Cartesian coordinates */
289 auto vd = dot(Rspin, vector3d<double>(spins__(0, ia), spins__(1, ia), spins__(2, ia))) -
290 vector3d<double>(spins__(0, ja), spins__(1, ja), spins__(2, ja));
291
292 if (vd.length() < 1e-10) {
293 n++;
294 }
295 }
296 /* if all atoms transform under spin rotaion, add it to a list */
297 if (n == num_atoms_) {
298 magnetic_group_symmetry_descriptor mag_op;
299 mag_op.spg_op = space_group_symmetry(isym);
300 mag_op.spin_rotation = Rspin;
301 mag_op.spin_rotation_inv = inverse(Rspin);
302 mag_op.spin_rotation_su2 = rotation_matrix_su2(Rspin);
303 /* add symmetry to the list */
304 magnetic_group_symmetry_.push_back(std::move(mag_op));
305 break;
306 }
307 }
308 }
309 PROFILE_STOP("sirius::Crystal_symmetry|mag");
310}

312double
313Crystal_symmetry::metric_tensor_error() const
314{
315 double diff{0};
316 for (auto const& e: magnetic_group_symmetry_) {
317 diff = std::max(diff, sirius::metric_tensor_error(lattice_vectors_, e.spg_op.R));
318 }
319 return diff;
320}

322double
323Crystal_symmetry::sym_op_R_error() const
324{
325 double diff{0};
326 for (auto const& e: magnetic_group_symmetry_) {
327 auto R = e.spg_op.Rcp;
328 auto R1 = inverse(transpose(R));
329 for (int i: {0, 1, 2}) {
330 for (int j: {0, 1, 2}) {
331 diff = std::max(diff, std::abs(R1(i, j) - R(i, j)));
332 }
333 }
334 }
335 return diff;
336}
