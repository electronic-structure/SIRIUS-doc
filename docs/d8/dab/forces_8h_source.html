88  inline void symmetrize_forces(mdarray<double, 2>& unsym_forces, mdarray<double, 2>& sym_forces)
89  {
90  if (!ctx_.use_symmetry()) {
91  unsym_forces >> sym_forces;
92  return;
93  }
94 
95  matrix3d<double> const& lattice_vectors = ctx_.unit_cell().symmetry().lattice_vectors();
96  matrix3d<double> const& inverse_lattice_vectors = ctx_.unit_cell().symmetry().inverse_lattice_vectors();
97 
98  sym_forces.zero();
99 
100  #pragma omp parallel for
101  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
102  vector3d<double> cart_force(&unsym_forces(0, ia));
103  vector3d<double> lat_force =
104  inverse_lattice_vectors * (cart_force / (double)ctx_.unit_cell().symmetry().num_mag_sym());
105 
106  for (int isym = 0; isym < ctx_.unit_cell().symmetry().num_mag_sym(); isym++) {
107  int ja = ctx_.unit_cell().symmetry().sym_table(ia, isym);
108  auto& R = ctx_.unit_cell().symmetry().magnetic_group_symmetry(isym).spg_op.R;
109  vector3d<double> rot_force = lattice_vectors * (R * lat_force);
110 
111  #pragma omp atomic update
112  sym_forces(0, ja) += rot_force[0];
113 
114  #pragma omp atomic update
115  sym_forces(1, ja) += rot_force[1];
116 
117  #pragma omp atomic update
118  sym_forces(2, ja) += rot_force[2];
119  }
120  }
121  }

135  inline void calc_local_forces(mdarray<double, 2>& forces)
136  {
137  PROFILE("sirius::Forces_PS::calc_local_forces");
138 
139  auto& valence_rho = density_.rho();
140 
141  Radial_integrals_vloc<false> ri(ctx_.unit_cell(), ctx_.pw_cutoff(), ctx_.settings().nprii_vloc_);
142 
143  Unit_cell& unit_cell = ctx_.unit_cell();
144 
145  Gvec const& gvecs = ctx_.gvec();
146 
147  int gvec_count = gvecs.gvec_count(ctx_.comm().rank());
148  int gvec_offset = gvecs.gvec_offset(ctx_.comm().rank());
149 
150  if (forces.size(0) != 3 || (int)forces.size(1) != unit_cell.num_atoms()) {
151  TERMINATE("forces array has wrong number of elements");
152  }
153 
154  forces.zero();
155 
156  double fact = valence_rho.gvec().reduced() ? 2.0 : 1.0;
157 
158  /* here the calculations are in lattice vectors space */
159  #pragma omp parallel for
160  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
161  Atom& atom = unit_cell.atom(ia);
162 
163  int iat = atom.type_id();
164 
165  for (int igloc = 0; igloc < gvec_count; igloc++) {
166  int ig = gvec_offset + igloc;
167 
168  /* cartesian form for getting cartesian force components */
169  vector3d<double> gvec_cart = gvecs.gvec_cart(ig);
170 
171  /* scalar part of a force without multipying by G-vector */
172  double_complex z = fact * fourpi * ri.value(iat, gvecs.gvec_len(ig)) *
173  std::conj(valence_rho.f_pw_local(igloc)) *
174  std::conj(ctx_.gvec_phase_factor(ig, ia));
175 
176  /* get force components multiplying by cartesian G-vector */
177  forces(0, ia) -= (gvec_cart[0] * z).imag();
178  forces(1, ia) -= (gvec_cart[1] * z).imag();
179  forces(2, ia) -= (gvec_cart[2] * z).imag();
180  }
181  }
182 
183  ctx_.comm().allreduce(&forces(0, 0), static_cast<int>(forces.size()));
184  }

186  inline void calc_ultrasoft_forces(mdarray<double, 2>& forces)
187  {
188  PROFILE("sirius::Forces_PS::calc_ultrasoft_forces");
189 
190  /* pack v effective in one array of pointers*/
191  Periodic_function<double>* vfield_eff[4];
192  vfield_eff[0] = potential_.effective_potential();
193  vfield_eff[0]->fft_transform(-1);
194  for (int imagn = 0; imagn < ctx_.num_mag_dims(); imagn++){
195  vfield_eff[imagn + 1] = potential_.effective_magnetic_field(imagn);
196  vfield_eff[imagn + 1]->fft_transform(-1);
197  }
198 
199  Unit_cell& unit_cell = ctx_.unit_cell();
200 
201  forces.zero();
202 
203  double reduce_g_fact = ctx_.gvec().reduced() ? 2.0 : 1.0;
204 
205  /* over atom types */
206  for (int iat = 0; iat < unit_cell.num_atom_types(); iat++){
207  auto& atom_type = unit_cell.atom_type(iat);
208 
209  if (!atom_type.pp_desc().augment) {
210  continue;
211  }
212 
213  const Augmentation_operator& aug_op = ctx_.augmentation_op(iat);
214 
215  int nbf = atom_type.mt_basis_size();
216 
217  /* get auxiliary density matrix */
218  auto dm = density_.density_matrix_aux(iat);
219 
220  //mdarray<double, 2> q_tmp(nbf * (nbf + 1) / 2, ctx_.gvec().count() * 2);
221  mdarray<double, 2> v_tmp(atom_type.num_atoms(), ctx_.gvec().count() * 2);
222  mdarray<double, 2> tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms());
223 
224  /* over spin components, can be from 1 to 4*/
225  for (int ispin = 0; ispin < ctx_.num_mag_dims() + 1; ispin++ ){
226  /* over 3 components of the force/G - vectors */
227  for (int ivec = 0; ivec < 3; ivec++ ){
228  /* over local rank G vectors */
229  #pragma omp parallel for schedule(static)
230  for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
231  int ig = ctx_.gvec().offset() + igloc;
232  auto gvc = ctx_.gvec().gvec_cart(ig);
233  for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
234  /* here we write in v_tmp -i * G * exp[ iGRn] Veff(G)
235  * but in formula we have i * G * exp[-iGRn] Veff*(G)
236  * the differences because we unfold complex array in the real one
237  * and need negative imagine part due to a multiplication law of complex numbers */
238  auto z = double_complex(0,-gvc[ivec]) * ctx_.gvec_phase_factor(ig, atom_type.atom_id(ia)) * vfield_eff[ispin]->f_pw_local(igloc);
239  v_tmp(ia, 2 * igloc) = z.real();
240  v_tmp(ia, 2 * igloc + 1) = z.imag();
241  }
242  }
243 
244  /* multiply tmp matrices, or sum over G*/
245  linalg<CPU>::gemm(0, 1, nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
246  aug_op.q_pw(), v_tmp, tmp);
247 
248 
249  #pragma omp parallel for
250  for (int ia = 0; ia < atom_type.num_atoms(); ia++) {
251  for (int i = 0; i < nbf * (nbf + 1) / 2; i++) {
252  forces(ivec, atom_type.atom_id(ia)) += ctx_.unit_cell().omega() * reduce_g_fact * dm(i, ia, ispin) * aug_op.sym_weight(i) * tmp(i, ia);
253  }
254  }
255  }
256  }
257  }
258 
259  ctx_.comm().allreduce(&forces(0, 0), static_cast<int>(forces.size()));
260  }

288  inline void calc_nlcc_forces(mdarray<double, 2>& forces)
289  {
290  PROFILE("sirius::Forces_PS::calc_nlcc_force");
291 
292  /* get main arrays */
293  auto xc_pot = potential_.xc_potential();
294 
295  /* transform from real space to reciprocal */
296  xc_pot->fft_transform(-1);
297 
298  Unit_cell& unit_cell = ctx_.unit_cell();
299 
300  Gvec const& gvecs = ctx_.gvec();
301 
302  int gvec_count = gvecs.count();
303  int gvec_offset = gvecs.offset();
304 
305  forces.zero();
306 
307  double fact = gvecs.reduced() ? 2.0 : 1.0;
308 
309  Radial_integrals_rho_core_pseudo<false> ri(ctx_.unit_cell(), ctx_.pw_cutoff(), ctx_.settings().nprii_rho_core_);
310 
311  /* here the calculations are in lattice vectors space */
312  #pragma omp parallel for
313  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
314  Atom& atom = unit_cell.atom(ia);
315 
316  int iat = atom.type_id();
317 
318  for (int igloc = 0; igloc < gvec_count; igloc++) {
319  int ig = gvec_offset + igloc;
320 
321  if (ig == 0) {
322  continue;
323  }
324 
325  /* cartesian form for getting cartesian force components */
326  vector3d<double> gvec_cart = gvecs.gvec_cart(ig);
327 
328  /* scalar part of a force without multipying by G-vector */
329  double_complex z = fact * fourpi * ri.value<int>(iat, gvecs.gvec_len(ig)) *
330  std::conj(xc_pot->f_pw_local(igloc) * ctx_.gvec_phase_factor( ig, ia));
331 
332  /* get force components multiplying by cartesian G-vector */
333  forces(0, ia) -= (gvec_cart[0] * z).imag();
334  forces(1, ia) -= (gvec_cart[1] * z).imag();
335  forces(2, ia) -= (gvec_cart[2] * z).imag();
336  }
337  }
338  ctx_.comm().allreduce(&forces(0, 0), static_cast<int>(forces.size()));
339  }

341  inline void calc_ewald_forces(mdarray<double, 2>& forces)
342  {
343  PROFILE("sirius::Forces_PS::calc_ewald_forces");
344 
345  Unit_cell& unit_cell = ctx_.unit_cell();
346 
347  forces.zero();
348 
349  double alpha = ctx_.ewald_lambda();
350 
351  double prefac = (ctx_.gvec().reduced() ? 4.0 : 2.0) * (twopi / unit_cell.omega());
352 
353  int ig0{0};
354  if (ctx_.comm().rank() == 0) {
355  ig0 = 1;
356  }
357 
358  mdarray<double_complex, 1> rho_tmp(ctx_.gvec().count());
359  rho_tmp.zero();
360  #pragma omp parallel for schedule(static)
361  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
362  int ig = ctx_.gvec().offset() + igloc;
363 
364  double_complex rho(0, 0);
365 
366  for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
367  rho += ctx_.gvec_phase_factor(ig, ja) * static_cast<double>(unit_cell.atom(ja).zn());
368  }
369 
370  rho_tmp[igloc] = std::conj(rho);
371  }
372 
373  #pragma omp parallel for
374  for (int ja = 0; ja < unit_cell.num_atoms(); ja++) {
375  for (int igloc = ig0; igloc < ctx_.gvec().count(); igloc++) {
376  int ig = ctx_.gvec().offset() + igloc;
377 
378  double g2 = std::pow(ctx_.gvec().gvec_len(ig), 2);
379 
380  /* cartesian form for getting cartesian force components */
381  vector3d<double> gvec_cart = ctx_.gvec().gvec_cart(ig);
382  double_complex rho(0, 0);
383 
384  double scalar_part = prefac * (rho_tmp[igloc] * ctx_.gvec_phase_factor(ig, ja)).imag() *
385  static_cast<double>(unit_cell.atom(ja).zn()) * std::exp(-g2 / (4 * alpha)) / g2;
386 
387  for (int x : {0, 1, 2}) {
388  forces(x, ja) += scalar_part * gvec_cart[x];
389  }
390  }
391  }
392 
393  ctx_.comm().allreduce(&forces(0, 0), static_cast<int>(forces.size()));
394 
395  double invpi = 1. / pi;
396 
397  #pragma omp parallel for
398  for (int ia = 0; ia < unit_cell.num_atoms(); ia++) {
399  for (int i = 1; i < unit_cell.num_nearest_neighbours(ia); i++) {
400  int ja = unit_cell.nearest_neighbour(i, ia).atom_id;
401 
402  double d = unit_cell.nearest_neighbour(i, ia).distance;
403  double d2 = d * d;
404 
405  vector3d<double> t = unit_cell.lattice_vectors() * unit_cell.nearest_neighbour(i, ia).translation;
406 
407  double scalar_part =
408  static_cast<double>(unit_cell.atom(ia).zn() * unit_cell.atom(ja).zn()) / d2 *
409  (gsl_sf_erfc(std::sqrt(alpha) * d) / d + 2.0 * std::sqrt(alpha * invpi) * std::exp(-d2 * alpha));
410 
411  for (int x : {0, 1, 2}) {
412  forces(x, ia) += scalar_part * t[x];
413  }
414  }
415  }
416  }
