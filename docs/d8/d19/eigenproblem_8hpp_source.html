465  dmatrix<double>& Z__)
466  {
467  if (A__.num_cols_local() != Z__.num_cols_local()) {
468  std::stringstream s;
469  s << "number of columns in A and Z don't match" << std::endl
470  << " number of cols in A (local and global): " << A__.num_cols_local() << " " << A__.num_cols()
471  << std::endl
472  << " number of cols in B (local and global): " << B__.num_cols_local() << " " << B__.num_cols()
473  << std::endl

475  << std::endl
476  << " number of rows in A (local and global): " << A__.num_rows_local() << " " << A__.num_rows()
477  << std::endl
478  << " number of rows in B (local and global): " << B__.num_rows_local() << " " << B__.num_rows()
479  << std::endl
480  << " number of rows in Z (local and global): " << Z__.num_rows_local() << " " << Z__.num_rows()
481  << std::endl;

483  }

485  TERMINATE("wrong block size");
486  }

488  utils::timer t1("Eigensolver_elpa|to_std");
489  /* Cholesky factorization B = U^{H}*U */
490  linalg2(linalg_t::scalapack).potrf(matrix_size__, B__.at(memory_t::host), B__.ld(), B__.descriptor());
491  /* inversion of the triangular matrix */
492  linalg2(linalg_t::scalapack).trtri(matrix_size__, B__.at(memory_t::host), B__.ld(), B__.descriptor());
493  /* U^{-1} is upper triangular matrix */
494  for (int i = 0; i < matrix_size__; i++) {
495  for (int j = i + 1; j < matrix_size__; j++) {
496  B__.set(j, i, 0);
497  }
498  }
499  /* transform to standard eigen-problem */
500  /* A * U{-1} -> Z */
501  linalg<device_t::CPU>::gemm(0, 0, matrix_size__, matrix_size__, matrix_size__, linalg_const<double>::one(), A__, B__,
502  linalg_const<double>::zero(), Z__);
503  /* U^{-H} * Z = U{-H} * A * U^{-1} -> A */
504  linalg<device_t::CPU>::gemm(2, 0, matrix_size__, matrix_size__, matrix_size__, linalg_const<double>::one(), B__, Z__,
505  linalg_const<double>::zero(), A__);
506  t1.stop();
507 
508  /* solve a standard problem */
509  int result = this->solve(matrix_size__, nev__, A__, eval__, Z__);
510  if (result) {
511  return result;
512  }
513 
514  utils::timer t3("Eigensolver_elpa|bt");
515  /* back-transform of eigen-vectors */
516  linalg<device_t::CPU>::gemm(0, 0, matrix_size__, nev__, matrix_size__, linalg_const<double>::one(), B__, Z__,
517  linalg_const<double>::zero(), A__);
518  A__ >> Z__;
519  t3.stop();
520 
521  return 0;
522  }
