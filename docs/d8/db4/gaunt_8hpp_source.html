54 /// Compact storage of non-zero Gaunt coefficients \f$ \langle \ell_1 m_1 | \ell_3 m_3 | \ell_2 m_2 \rangle \f$.
55 /** Very important! The following notation is adopted and used everywhere: lm1 and lm2 represent 'bra' and 'ket'
56  * spherical harmonics of the Gaunt integral and lm3 represent the inner spherical harmonic.
57  */
58 template <typename T>
59 class Gaunt_coefficients
60 {
61  private:
62  /// lmax of <lm1|
63  int lmax1_;
64  /// lmmax of <lm1|
65  int lmmax1_;
66 
67  /// lmax of inner real or complex spherical harmonic
68  int lmax3_;
69  /// lmmax of inner real or complex spherical harmonic
70  int lmmax3_;
71 
72  /// lmax of |lm2>
73  int lmax2_;
74  /// lmmax of |lm2>
75  int lmmax2_;
76 
77  /// List of non-zero Gaunt coefficients for each lm3.
78  mdarray<std::vector<gaunt_L1_L2<T>>, 1> gaunt_packed_L1_L2_;
79 
80  /// List of non-zero Gaunt coefficients for each combination of lm1, lm2.
81  mdarray<std::vector<gaunt_L3<T>>, 2> gaunt_packed_L3_;
82 
83  public:
84  /// Class constructor.
85  Gaunt_coefficients(int lmax1__,
86  int lmax3__,
87  int lmax2__,
88  std::function<T(int, int, int, int, int, int)> get__)
89  : lmax1_(lmax1__)
90  , lmax3_(lmax3__)
91  , lmax2_(lmax2__)
92  {
93  lmmax1_ = utils::lmmax(lmax1_);
94  lmmax3_ = utils::lmmax(lmax3_);
95  lmmax2_ = utils::lmmax(lmax2_);
96 
97  gaunt_packed_L1_L2_ = mdarray<std::vector<gaunt_L1_L2<T>>, 1>(lmmax3_);
98  gaunt_L1_L2<T> g12;
99 
100  gaunt_packed_L3_ = mdarray<std::vector<gaunt_L3<T>>, 2>(lmmax1_, lmmax2_);
101  gaunt_L3<T> g3;
102 
103  for (int l1 = 0, lm1 = 0; l1 <= lmax1_; l1++) {
104  for (int m1 = -l1; m1 <= l1; m1++, lm1++) {
105  for (int l2 = 0, lm2 = 0; l2 <= lmax2_; l2++) {
106  for (int m2 = -l2; m2 <= l2; m2++, lm2++) {
107  for (int l3 = 0, lm3 = 0; l3 <= lmax3_; l3++) {
108  for (int m3 = -l3; m3 <= l3; m3++, lm3++) {
109 
110  T gc = get__(l1, l3, l2, m1, m3, m2);
111  if (std::abs(gc) > 1e-12) {
112  g12.lm1 = lm1;
113  g12.lm2 = lm2;
114  g12.coef = gc;
115  gaunt_packed_L1_L2_[lm3].push_back(g12);
116 
117  g3.lm3 = lm3;
118  g3.l3 = l3;
119  g3.coef = gc;
120  gaunt_packed_L3_(lm1, lm2).push_back(g3);
121  }
122  }
123  }
124  }
125  }
126  }
127  }
128  }
129 
130  /// Return number of non-zero Gaunt coefficients for a given lm3.
131  inline int num_gaunt(int lm3) const
132  {
133  assert(lm3 >= 0 && lm3 < lmmax3_);
134  return static_cast<int>(gaunt_packed_L1_L2_[lm3].size());
135  }

137  /// Return a structure containing {lm1, lm2, coef} for a given lm3 and index.
138  /** Example:
139  * \code{.cpp}
140  * for (int lm3 = 0; lm3 < lmmax3; lm3++)
141  * {
142  * for (int i = 0; i < gaunt_coefs.num_gaunt(lm3); i++) {
143  * int lm1 = gaunt_coefs.gaunt(lm3, i).lm1;
144  * int lm2 = gaunt_coefs.gaunt(lm3, i).lm2;
145  * double coef = gaunt_coefs.gaunt(lm3, i).coef;
146  *
147  * // do something with lm1,lm2,lm3 and coef
148  * }
149  * }
150  * \endcode
151  */
152  inline gaunt_L1_L2<T> const& gaunt(int lm3, int idx) const
153  {
154  assert(lm3 >= 0 && lm3 < lmmax3_);
155  assert(idx >= 0 && idx < (int)gaunt_packed_L1_L2_[lm3].size());
156  return gaunt_packed_L1_L2_[lm3][idx];
157  }

163  }

166  inline gaunt_L3<T> const& gaunt(int lm1, int lm2, int idx) const

168  return gaunt_packed_L3_(lm1, lm2)[idx];

171  /// Return a sum over L3 (lm3) index of Gaunt coefficients and a complex vector.
172  /** The following operation is performed:
173  * \f[
174  * \sum_{\ell_3 m_3} \langle \ell_1 m_1 | \ell_3 m_3 | \ell_2 m_2 \rangle v_{\ell_3 m_3}
175  * \f]
176  * Result is assumed to be complex.
177  */
178  inline double_complex sum_L3_gaunt(int lm1, int lm2, double_complex const* v) const
179  {
180  double_complex zsum(0, 0);
181  for (int k = 0; k < (int)gaunt_packed_L3_(lm1, lm2).size(); k++) {
182  zsum += gaunt_packed_L3_(lm1, lm2)[k].coef * v[gaunt_packed_L3_(lm1, lm2)[k].lm3];
183  }
184  return zsum;
185  }
186 
187  /// Return a sum over L3 (lm3) index of Gaunt coefficients and a real vector.
188  /** The following operation is performed:
189  * \f[
190  * \sum_{\ell_3 m_3} \langle \ell_1 m_1 | \ell_3 m_3 | \ell_2 m_2 \rangle v_{\ell_3 m_3}
191  * \f]
192  * Result is assumed to be of the same type as Gaunt coefficients.
193  */
194  inline T sum_L3_gaunt(int lm1, int lm2, double const* v) const
195  {
196  T sum = 0;
197  for (int k = 0; k < (int)gaunt_packed_L3_(lm1, lm2).size(); k++) {
198  sum += gaunt_packed_L3_(lm1, lm2)[k].coef * v[gaunt_packed_L3_(lm1, lm2)[k].lm3];
199  }
200  return sum;
201  }

203  /// Return vector of non-zero Gaunt coefficients for a given combination of lm1 and lm2
204  inline std::vector<gaunt_L3<T>> const& gaunt_vector(int lm1, int lm2) const
205  {
206  return gaunt_packed_L3_(lm1, lm2);
207  }
208 };
