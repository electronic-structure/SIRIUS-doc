33 /// MPI grid interface
34 /** The following terminology is used. Suppose we have a 4x5 grid of MPI ranks. We say it's a two-\em dimensional
35  * grid with the first dimension of the size 4 and the second dimensoion of the size 5. The \em actual number of
36  * grid dimensions is two, however we may also consider the grid as being a D-dimensional (D >= 2) with implicit
37  * dimension sizes equal to one, e.g. 4x5 := 4x5x1x1x1... The communication happens along single or multiple
38  * \em directions along the grid dimensions. We specify directions wth bits, eg. directions=00000101 reads as
39  * "communication along 1-st and 3-rd dimensions".
40  * \image html mpi_grid_comm.png "Communication along dimension d0 (between ranks of d0)."
41  * In the provided example the corresponding communicator is MPI_grid::communicator(1 << d0), where d0 is the integer
42  * index of dimension.
43  */
44 class MPI_grid
45 {
46  private:
47  /// Dimensions of the grid.
48  std::vector<int> dimensions_;
49 
50  /// Coordinates of the MPI rank in the grid.
51  std::vector<int> coordinates_;
52 
53  /// Parent communicator
54  Communicator const& parent_communicator_;
55 
56  /// Grid communicator of the enrire grid returned by MPI_Cart_create
57  Communicator base_grid_communicator_;
58 
59  /// grid communicators
60  /** Grid comminicators are built for all possible combinations of directions, i.e. 001, 010, 011, etc.
61  * First communicator is the trivial "self" communicator; the last communicator handles the entire grid. */
62  std::vector<Communicator> communicators_;
63 
64  /// Number of MPI ranks in each communicator.
65  std::vector<int> communicator_size_;
66 
67  /// True if this is the root of the communicator group.
68  std::vector<bool> communicator_root_;
69 
70  /// Return valid directions for the current grid dimensionality.
71  inline int valid_directions(int directions__) const
72  {
73  return (directions__ & ((1 << dimensions_.size()) - 1));
74  }
75 
76  /// Initialize the grid.
77  void initialize()
78  {
79  if (dimensions_.size() == 0) {
80  TERMINATE("no dimensions provided for the MPI grid");
81  }
82 
83  int sz{1};
84  for (size_t i = 0; i < dimensions_.size(); i++) {
85  sz *= dimensions_[i];
86  }
87 
88  if (parent_communicator_.size() != sz) {
89  std::stringstream s;
90  s << "Number of MPI ranks doesn't match the size of the grid." << std::endl << " grid dimensions :";
91  for (auto& e : dimensions_) {
92  s << " " << e;
93  }
94  s << std::endl << " available number of MPI ranks : " << parent_communicator_.size();
95 
96  TERMINATE(s);
97  }
98 
99  /* communicator of the entire grid */
100  std::vector<int> periods(dimensions_.size(), 0);
101  CALL_MPI(MPI_Cart_create, (parent_communicator_.mpi_comm(), (int)dimensions_.size(), &dimensions_[0],
102  &periods[0], 0, &base_grid_communicator_.mpi_comm()));
103 
104  /* total number of communicators inside the grid */
105  int num_comm = 1 << dimensions_.size();
106 
107  communicators_ = std::vector<Communicator>(num_comm); //.resize(num_comm);
108 
109  coordinates_ = std::vector<int>(dimensions_.size(), -1);
110 
111  communicator_size_ = std::vector<int>(num_comm, 0);
112 
113  communicator_root_ = std::vector<bool>(num_comm, false);
114 
115  /* get coordinates */
116  CALL_MPI(MPI_Cart_get, (base_grid_communicator_.mpi_comm(), (int)dimensions_.size(), &dimensions_[0],
117  &periods[0], &coordinates_[0]));
118 
119  /* get all possible communicators */
120  for (int i = 1; i < num_comm; i++) {
121  bool is_root = true;
122  int comm_size = 1;
123  std::vector<int> flg(dimensions_.size(), 0);
124 
125  /* each bit represents a directions */
126  for (int j = 0; j < (int)dimensions_.size(); j++) {
127  if (i & (1 << j)) {
128  flg[j] = 1;
129  is_root = is_root && (coordinates_[j] == 0);
130  comm_size *= dimensions_[j];
131  }
132  }
133 
134  communicator_root_[i] = is_root;
135 
136  communicator_size_[i] = comm_size;
137 
138  /* subcommunicators */
139  CALL_MPI(MPI_Cart_sub, (base_grid_communicator_.mpi_comm(), &flg[0], &communicators_[i].mpi_comm()));
140  }
141 
142  /* explicitly set the size of "self" communicator */
143  communicator_size_[0] = 1;
144 
145  /* explicitly set the root of "self" communicator */
146  communicator_root_[0] = true;
147 
148  /* expicitly set the "self" communicator */
149  communicators_[0].mpi_comm() = MPI_COMM_SELF;
150 
151  /* double check the size of communicators */
152  for (int i = 1; i < num_comm; i++) {
153  if (communicators_[i].size() != communicator_size_[i]) {
154  TERMINATE("communicator sizes don't match");
155  }
156  }
157 
158  for (int i = 0; i < (int)dimensions_.size(); i++) {
159  if (communicator(1 << i).rank() != coordinate(i)) {
160  TERMINATE("ranks don't match");
161  }
162  }
163 
164  if (communicator().cart_rank(coordinates_) != parent_communicator_.rank()) {
165  TERMINATE("cartesian and communicator ranks don't match");
166  }
167  }
168 
169  void finalize()
170  {
171  communicators_.clear();
172  communicator_root_.clear();
173  communicator_size_.clear();
174  coordinates_.clear();
175  dimensions_.clear();
176  }
177 
178  /* forbid copy constructor */
179  MPI_grid(MPI_grid const& src) = delete;
180  /* forbid assigment operator */
181  MPI_grid& operator=(MPI_grid const& src) = delete;
182 
183  public:
184  MPI_grid(std::vector<int> dimensions__, Communicator const& parent_communicator__)
185  : dimensions_(dimensions__)
186  , parent_communicator_(parent_communicator__)
187  {
188  initialize();
189  }
190 
191  MPI_grid(Communicator const& parent_communicator__)
192  : dimensions_({parent_communicator__.size()})
193  , parent_communicator_(parent_communicator__)
194  {
195  initialize();
196  }
197 
198  ~MPI_grid()
199  {
200  finalize();
201  }

204  inline int size(int directions = 0xFF) const
205  {
206  return communicator_size_[valid_directions(directions)];
207  }

231  }

233  inline Communicator const& communicator(int directions__ = 0xFF) const
234  {
235  assert(communicators_.size() != 0);
236 
237  return communicators_[valid_directions(directions__)];
238  }

241 /// A bundle of MPI communicators.
242 /** Each MPI rank is assigned to one of the MPI subgroups. */
243 class Communicator_bundle
244 {
245  private:
246  /// MPI subgroup communicator
247  Communicator comm_;
248 
249  /// Total number of communicators.
250  int size_;
251 
252  /// ID of a communication subgroup.
253  int id_;
254 
255  public:
256  Communicator_bundle()
257  : size_(-1)
258  , id_(-1)
259  {
260  }
261 
262  Communicator_bundle(Communicator const& base_comm__, int num_elements__)
263  {
264  /* number of communicators can't be larger than the number of MPI ranks */
265  size_ = std::min(base_comm__.size(), num_elements__);
266  int rank = base_comm__.rank();
267  /* assign ID in cyclic fasion */
268  id_ = rank % size_;
269  /* split parent communicator */
270  CALL_MPI(MPI_Comm_split, (base_comm__.mpi_comm(), id_, rank, &comm_.mpi_comm()));
271  }
272 
273  /// Return sub-communicator.
274  inline Communicator const& comm() const
275  {
276  return comm_;
277  }
278 
279  /// Return size of the communicator bundle.
280  inline int size() const
281  {
282  return size_;
283  }
284 
285  /// Return id of a subgroup for a given MPI rank.
286  inline int id() const
287  {
288  return id_;
289  }
290 };
