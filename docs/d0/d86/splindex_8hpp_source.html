155  /// Return "local index, rank" pair for a global index.
156  inline typename splindex_base<T>::location_t location(T idxglob__) const
157  {
158  assert(idxglob__ < this->global_index_size_);
159 
160  int rank = int(idxglob__ / block_size_);
161  T idxloc = idxglob__ - rank * block_size_;
162 
163  return typename splindex_base<T>::location_t(idxloc, rank);
164  }
165 
166  /// Return local size of the split index for an arbitrary rank.
167  inline T local_size(int rank__) const
168  {
169  assert(rank__ >= 0 && rank__ < this->num_ranks_);
170 
171  if (this->global_index_size_ == 0) {
172  return 0;
173  }
174 
175  int n = static_cast<int>(this->global_index_size_ / block_size_);
176  if (rank__ < n) {
177  return block_size_;
178  } else if (rank__ == n) {
179  return this->global_index_size_ - rank__ * block_size_;
180  } else {
181  return 0;
182  }
183  }

298  /// Return "local index, rank" pair for a global index.
299  inline typename splindex_base<T>::location_t location(T idxglob__) const
300  {
301  assert(idxglob__ < this->global_index_size_);
302 
303  /* number of full blocks */
304  T num_blocks = idxglob__ / block_size_;
305 
306  /* local index */
307  T idxloc = (num_blocks / this->num_ranks_) * block_size_ + idxglob__ % block_size_;
308 
309  /* corresponding rank */
310  int rank = static_cast<int>(num_blocks % this->num_ranks_);
311 
312  return typename splindex_base<T>::location_t(idxloc, rank);
313  }

315  /// Return local size of the split index for an arbitrary rank.
316  inline T local_size(int rank__) const
317  {
318  assert(rank__ >= 0 && rank__ < this->num_ranks_);
319 
320  /* number of full blocks */
321  T num_blocks = this->global_index_size_ / block_size_;
322 
323  T n = (num_blocks / this->num_ranks_) * block_size_;
324 
325  int rank_offs = static_cast<int>(num_blocks % this->num_ranks_);
326 
327  if (rank__ < rank_offs) {
328  n += block_size_;
329  } else if (rank__ == rank_offs) {
330  n += this->global_index_size_ % block_size_;
331  }
332  return n;
333  }

369 /// Specialization for the block distribution.
370 template <typename T>
371 class splindex<chunk, T> : public splindex_base<T>
372 {
373  private:
374  std::vector<std::vector<T>> global_index_;
375  std::vector<typename splindex_base<T>::location_t> locations_;
376 
377  public:
378  /// Default constructor.
379  splindex()
380  {
381  }
382 
383  /// Constructor with specific partitioning.
384  splindex(T global_index_size__, int num_ranks__, int rank__, std::vector<T> const& counts__)
385  {
386  this->global_index_size_ = global_index_size__;
387 
388  if (num_ranks__ < 0) {
389  std::stringstream s;
390  s << "wrong number of ranks: " << num_ranks__;
391  throw std::runtime_error(s.str());
392  }
393  this->num_ranks_ = num_ranks__;
394 
395  if (rank__ < 0 || rank__ >= num_ranks__) {
396  std::stringstream s;
397  s << "wrong rank: " << rank__;
398  throw std::runtime_error(s.str());
399  }
400  this->rank_ = rank__;
401 
402  for (int r = 0; r < num_ranks__; r++) {
403  global_index_.push_back(std::vector<T>());
404  for (int i = 0; i < counts__[r]; i++) {
405  global_index_.back().push_back(static_cast<T>(locations_.size()));
406  locations_.push_back(typename splindex_base<T>::location_t(i, r));
407  }
408  }
409 
410  assert(static_cast<T>(locations_.size()) == global_index_size__);
411  }
412 
413  inline T local_size(int rank__) const
414  {
415  assert(rank__ >= 0);
416  assert(rank__ < this->num_ranks_);
417  return static_cast<T>(global_index_[rank__].size());
418  }
419 
420  inline T local_size() const
421  {
422  return local_size(this->rank_);
423  }
424 
425  inline int local_rank(T idxglob__) const
426  {
427  return locations_[idxglob__].rank;
428  }
429 
430  inline T local_index(T idxglob__) const
431  {
432  return locations_[idxglob__].local_index;
433  }

436  {
