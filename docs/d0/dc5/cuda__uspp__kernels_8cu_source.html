30 __global__ void compute_chebyshev_order1_gpu_kernel
31 (
32  int num_gkvec__,
33  double c__,
34  double r__,
35  acc_complex_double_t* phi0__,
36  acc_complex_double_t* phi1__
37 )
38 {
39  int igk = blockDim.x * blockIdx.x + threadIdx.x;
40  int j = blockIdx.y;
41 
42  if (igk < num_gkvec__)
43  {
44  int i = array2D_offset(igk, j, num_gkvec__);
45  // phi0 * c
46  acc_complex_double_t z1 = accCmul(phi0__[i], make_accDoubleComplex(c__, 0));
47  // phi1 - phi0 * c
48  acc_complex_double_t z2 = accCsub(phi1__[i], z1);
49  // (phi1 - phi0 * c) / r
50  phi1__[i] = accCdiv(z2, make_accDoubleComplex(r__, 0));
51  }
52 }

254 __global__ void compute_inner_product_gpu_kernel
255 (
256  int num_gkvec_row,
257  acc_complex_double_t const* f1,
258  acc_complex_double_t const* f2,
259  double* prod
260 )
261 {
262  int N = num_blocks(num_gkvec_row, blockDim.x);
263 
264  ACC_DYNAMIC_SHARED( char, sdata_ptr)
265  double* sdata = (double*)&sdata_ptr[0];
266 
267  sdata[threadIdx.x] = 0.0;
268 
269  for (int n = 0; n < N; n++)
270  {
271  int igk = n * blockDim.x + threadIdx.x;
272  if (igk < num_gkvec_row)
273  {
274  int k = array2D_offset(igk, blockIdx.x, num_gkvec_row);
275  sdata[threadIdx.x] += f1[k].x * f2[k].x + f1[k].y *f2[k].y;
276  }
277  }
278 
279  __syncthreads();
280 
281  for (int s = 1; s < blockDim.x; s *= 2)
282  {
283  if (threadIdx.x % (2 * s) == 0) sdata[threadIdx.x] = sdata[threadIdx.x] + sdata[threadIdx.x + s];
284  __syncthreads();
285  }
286 
287  prod[blockIdx.x] = sdata[0];
288 }

308 __global__ void add_checksum_gpu_kernel
309 (
310  acc_complex_double_t const* wf__,
311  int num_rows_loc__,
312  acc_complex_double_t* result__
313 )
314 {
315  int N = num_blocks(num_rows_loc__, blockDim.x);
316 
317  ACC_DYNAMIC_SHARED( char, sdata_ptr)
318  double* sdata_x = (double*)&sdata_ptr[0];
319  double* sdata_y = (double*)&sdata_ptr[blockDim.x * sizeof(double)];
320 
321  sdata_x[threadIdx.x] = 0.0;
322  sdata_y[threadIdx.x] = 0.0;
323 
324  for (int n = 0; n < N; n++) {
325  int j = n * blockDim.x + threadIdx.x;
326  if (j < num_rows_loc__) {
327  int k = array2D_offset(j, blockIdx.x, num_rows_loc__);
328  sdata_x[threadIdx.x] += wf__[k].x;
329  sdata_y[threadIdx.x] += wf__[k].y;
330  }
331  }
332  __syncthreads();
333 
334  for (int s = 1; s < blockDim.x; s *= 2) {
335  if (threadIdx.x % (2 * s) == 0) {
336  sdata_x[threadIdx.x] = sdata_x[threadIdx.x] + sdata_x[threadIdx.x + s];
337  sdata_y[threadIdx.x] = sdata_y[threadIdx.x] + sdata_y[threadIdx.x + s];
338  }
339  __syncthreads();
340  }
341 
342  result__[blockIdx.x] = accCadd(result__[blockIdx.x], make_accDoubleComplex(sdata_x[0], sdata_y[0]));
343 }
