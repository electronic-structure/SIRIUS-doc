29void Potential::init_PAW()
30{
31 paw_potential_data_.clear();
32 if (!unit_cell_.num_paw_atoms()) {
33 return;
34 }
35
36 for (int i = 0; i < unit_cell_.spl_num_paw_atoms().local_size(); i++) {
37 int ia_paw = unit_cell_.spl_num_paw_atoms(i);
38 int ia = unit_cell_.paw_atom_index(ia_paw);
39
40 auto& atom = unit_cell_.atom(ia);
41
42 auto& atom_type = atom.type();
43
44 int l_max = 2 * atom_type.indexr().lmax_lo();
45 int lm_max_rho = utils::lmmax(l_max);
46
47 paw_potential_data_t ppd;
48
49 ppd.atom_ = &atom;
50
51 ppd.ia = ia;
52
53 ppd.ia_paw = ia_paw;
54
55 /* allocate potential arrays */
56 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
57 ppd.ae_potential_.push_back(sf(lm_max_rho, ppd.atom_->radial_grid()));
58 ppd.ps_potential_.push_back(sf(lm_max_rho, ppd.atom_->radial_grid()));
59 }
60
61 ppd.core_energy_ = atom_type.paw_core_energy();
62
63 paw_potential_data_.push_back(std::move(ppd));
64 }
65
66 for (int i = 0; i < unit_cell_.num_paw_atoms(); i++) {
67 int ia = unit_cell_.paw_atom_index(i);
68 int bs = unit_cell_.atom(ia).mt_basis_size();
69 max_paw_basis_size_ = std::max(max_paw_basis_size_, bs);
70 }
71
72 /* initialize dij matrix */
73 paw_dij_ = mdarray<double, 4>(max_paw_basis_size_, max_paw_basis_size_, ctx_.num_mag_dims() + 1,
74 unit_cell_.num_paw_atoms(), memory_t::host, "paw_dij_");
75
76 /* allocate PAW energy array */
77 paw_hartree_energies_.resize(unit_cell_.num_paw_atoms());
78 paw_xc_energies_.resize(unit_cell_.num_paw_atoms());
79 paw_core_energies_.resize(unit_cell_.num_paw_atoms());
80 paw_one_elec_energies_.resize(unit_cell_.num_paw_atoms());
81}

83void Potential::generate_PAW_effective_potential(Density const& density)
84{
85 PROFILE("sirius::Potential::generate_PAW_effective_potential");
86
87 if (!unit_cell_.num_paw_atoms()) {
88 return;
89 }
90
91 /* zero PAW arrays */
92 std::fill(paw_one_elec_energies_.begin(), paw_one_elec_energies_.end(), 0.0);
93 std::fill(paw_hartree_energies_.begin(), paw_hartree_energies_.end(), 0.0);
94 std::fill(paw_xc_energies_.begin(), paw_xc_energies_.end(), 0.0);
95
96 /* zero Dij */
97 paw_dij_.zero();
98
99 /* calculate xc and hartree for atoms */
100 for (int i = 0; i < unit_cell_.spl_num_paw_atoms().local_size(); i++) {
101 calc_PAW_local_potential(paw_potential_data_[i], density.paw_ae_density(i), density.paw_ps_density(i));
102 }
103
104 /* calculate PAW Dij matrix */
105 #pragma omp parallel for
106 for (int i = 0; i < unit_cell_.spl_num_paw_atoms().local_size(); i++) {
107 calc_PAW_local_Dij(paw_potential_data_[i], paw_dij_);
108
109 //calc_PAW_one_elec_energy(paw_potential_data_[i], density.density_matrix(), paw_dij_);
110 }
111
112 // collect Dij and add to atom d_mtrx
113 comm_.allreduce(&paw_dij_(0, 0, 0, 0), static_cast<int>(paw_dij_.size()));
114
115 if (ctx_.cfg().control().print_checksum() && comm_.rank() == 0) {
116 auto cs = paw_dij_.checksum();
117 utils::print_checksum("paw_dij", cs);
118 }
119
120 // add paw Dij to uspp Dij
121 add_paw_Dij_to_atom_Dmtrx();
122
123 // calc total energy
124 double energies[] = {0.0, 0.0, 0.0, 0.0};
125
126 for (int ia = 0; ia < unit_cell_.spl_num_paw_atoms().local_size(); ia++) {
127 energies[0] += paw_potential_data_[ia].hartree_energy_;
128 energies[1] += paw_potential_data_[ia].xc_energy_;
129 //energies[2] += paw_potential_data_[ia].one_elec_energy_;
130 energies[3] += paw_potential_data_[ia].core_energy_; // it is light operation
131 }
132
133 comm_.allreduce(&energies[0], 4);
134
135 paw_hartree_total_energy_ = energies[0];
136 paw_xc_total_energy_ = energies[1];
137 //paw_one_elec_energy_ = energies[2];
138 paw_total_core_energy_ = energies[3];
139}

141double xc_mt_paw(std::vector<XC_functional> const& xc_func__, int lmax__, int num_mag_dims__, SHT const& sht__,
142 Radial_grid<double> const& rgrid__, std::vector<Flm const*> rho__, std::vector<double> const& rho_core__,
143 std::vector<Flm>& vxc__)
144{
145 int lmmax = utils::lmmax(lmax__);
146
147 /* new array to store core and valence densities */
148 Flm rho0(lmmax, rgrid__);
149
150 assert(rho0.size(0) == rho__[0]->size(0));
151
152 rho0.zero();
153 rho0 += (*rho__[0]);
154
155 double invY00 = 1.0 / y00;
156
157 /* add core density */
158 for (int ir = 0; ir < rgrid__.num_points(); ir++) {
159 rho0(0, ir) += invY00 * rho_core__[ir];
160 }
161
162 std::vector<Flm const*> rho;
163 rho.push_back(&rho0);
164 for (int j = 0; j < num_mag_dims__; j++) {
165 rho.push_back(rho__[j + 1]);
166 }
167
168 std::vector<Flm*> vxc;
169 for (int j = 0; j < num_mag_dims__ + 1; j++) {
170 vxc.push_back(&vxc__[j]);
171 }
172
173 Flm exclm(lmmax, rgrid__);
174
175 sirius::xc_mt(rgrid__, sht__, xc_func__, num_mag_dims__, rho, vxc, &exclm);
176 return inner(exclm, rho0);
177}

179double Potential::calc_PAW_hartree_potential(Atom& atom, sf const& full_density, sf& full_potential)
180{
181 int lmsize_rho = static_cast<int>(full_density.size(0));
182
183 auto& grid = full_density.radial_grid();
184
185 /* array passed to poisson solver */
186 Spheric_function<function_domain_t::spectral, double> atom_pot_sf(lmsize_rho, grid);
187 atom_pot_sf.zero();
188
189 poisson_vmt<true>(atom, full_density, atom_pot_sf);
190
191 /* add hartree contribution */
192 full_potential += atom_pot_sf;
193
194 /* calculate energy */
195
196 auto l_by_lm = utils::l_by_lm(utils::lmax(lmsize_rho));
197
198 /* create array for integration */
199 std::vector<double> intdata(grid.num_points(), 0);
200
201 double hartree_energy{0};
202
203 for (int lm = 0; lm < lmsize_rho; lm++) {
204 /* fill data to integrate */
205 for (int irad = 0; irad < grid.num_points(); irad++) {
206 intdata[irad] = full_density(lm, irad) * full_potential(lm, irad) * grid[irad] * grid[irad];
207 }
208
209 /* create spline from the data */
210 Spline<double> h_spl(grid, intdata);
211 hartree_energy += 0.5 * h_spl.integrate(0);
212 }
213
214 return hartree_energy;
215}

217void Potential::calc_PAW_local_potential(paw_potential_data_t& ppd,
218 std::vector<Spheric_function<function_domain_t::spectral, double> const*> ae_density,
219 std::vector<Spheric_function<function_domain_t::spectral, double> const*> ps_density)
220{
221 /* calculation of Hartree potential */
222 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
223 ppd.ae_potential_[i].zero();
224 ppd.ps_potential_[i].zero();
225 }
226
227 double ae_hartree_energy = calc_PAW_hartree_potential(*ppd.atom_, *ae_density[0], ppd.ae_potential_[0]);
228
229 double ps_hartree_energy = calc_PAW_hartree_potential(*ppd.atom_, *ps_density[0], ppd.ps_potential_[0]);
230
231 ppd.hartree_energy_ = ae_hartree_energy - ps_hartree_energy;
232
233 /* calculation of XC potential */
234 auto& ps_core = ppd.atom_->type().ps_core_charge_density();
235 auto& ae_core = ppd.atom_->type().paw_ae_core_charge_density();
236
237 auto& rgrid = ppd.atom_->type().radial_grid();
238 int l_max = 2 * ppd.atom_->type().indexr().lmax_lo();
239
240 std::vector<Flm> vxc;
241 for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
242 vxc.emplace_back(utils::lmmax(l_max), rgrid);
243 }
244
245 auto ae_xc_energy = sirius::xc_mt_paw(xc_func_, l_max, ctx_.num_mag_dims(), *sht_, rgrid, ae_density,
246 ae_core, vxc);
247 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
248 ppd.ae_potential_[i] += vxc[i];
249 }
250
251 auto ps_xc_energy = sirius::xc_mt_paw(xc_func_, l_max, ctx_.num_mag_dims(), *sht_, rgrid, ps_density,
252 ps_core, vxc);
253
254 for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
255 ppd.ps_potential_[i] += vxc[i];
256 }
257
258 /* save xc energy in pdd structure */
259 ppd.xc_energy_ = ae_xc_energy - ps_xc_energy;
260}

262void Potential::calc_PAW_local_Dij(paw_potential_data_t& pdd, mdarray<double, 4>& paw_dij)
263{
264 int paw_ind = pdd.ia_paw;
265
266 auto& atom_type = pdd.atom_->type();
267
268 auto& paw_ae_wfs = atom_type.ae_paw_wfs_array();
269 auto& paw_ps_wfs = atom_type.ps_paw_wfs_array();
270
271 /* get lm size for density */
272 int lmax = atom_type.indexr().lmax_lo();
273 int lmsize_rho = utils::lmmax(2 * lmax);
274
275 auto l_by_lm = utils::l_by_lm(2 * lmax);
276
277 Gaunt_coefficients<double> GC(lmax, 2 * lmax, lmax, SHT::gaunt_rrr);
278
279 /* store integrals here */
280 mdarray<double, 3> integrals(
281 lmsize_rho, atom_type.num_beta_radial_functions() * (atom_type.num_beta_radial_functions() + 1) / 2,
282 ctx_.num_mag_dims() + 1);
283
284 auto& rgrid = atom_type.radial_grid();
285
286 for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
287 auto& ae_atom_pot = pdd.ae_potential_[imagn];
288 auto& ps_atom_pot = pdd.ps_potential_[imagn];
289
290 for (int irb2 = 0; irb2 < atom_type.num_beta_radial_functions(); irb2++) {
291 for (int irb1 = 0; irb1 <= irb2; irb1++) {
292 int iqij = (irb2 * (irb2 + 1)) / 2 + irb1;
293
294 /* create array for integration */
295 std::vector<double> intdata(rgrid.num_points());
296
297 for (int lm3 = 0; lm3 < lmsize_rho; lm3++) {
298 /* fill array */
299 for (int irad = 0; irad < rgrid.num_points(); irad++) {
300 double ae_part = paw_ae_wfs(irad, irb1) * paw_ae_wfs(irad, irb2);
301 double ps_part = paw_ps_wfs(irad, irb1) * paw_ps_wfs(irad, irb2) +
302 atom_type.q_radial_function(irb1, irb2, l_by_lm[lm3])(irad);
303
304 intdata[irad] = ae_atom_pot(lm3, irad) * ae_part - ps_atom_pot(lm3, irad) * ps_part;
305 }
306
307 /* create spline from data arrays */
308 Spline<double> dij_spl(rgrid, intdata);
309
310 /* integrate */
311 integrals(lm3, iqij, imagn) = dij_spl.integrate(0);
312 }
313 }
314 }
315 }
316
317 /* calculate Dij */
318 for (int ib2 = 0; ib2 < atom_type.mt_lo_basis_size(); ib2++) {
319 for (int ib1 = 0; ib1 <= ib2; ib1++) {
320
321 /* get lm quantum numbers (lm index) of the basis functions */
322 int lm1 = atom_type.indexb(ib1).lm;
323 int lm2 = atom_type.indexb(ib2).lm;
324
325 /* get radial basis functions indices */
326 int irb1 = atom_type.indexb(ib1).idxrf;
327 int irb2 = atom_type.indexb(ib2).idxrf;
328
329 /* common index */
330 int iqij = (irb2 * (irb2 + 1)) / 2 + irb1;
331
332 /* get num of non-zero GC */
333 int num_non_zero_gk = GC.num_gaunt(lm1, lm2);
334
335 for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
336 /* add nonzero coefficients */
337 for (int inz = 0; inz < num_non_zero_gk; inz++) {
338 auto& lm3coef = GC.gaunt(lm1, lm2, inz);
339
340 /* add to atom Dij an integral of dij array */
341 paw_dij(ib1, ib2, imagn, paw_ind) += lm3coef.coef * integrals(lm3coef.lm3, iqij, imagn);
342 }
343
344 if (ib1 != ib2) {
345 paw_dij(ib2, ib1, imagn, paw_ind) = paw_dij(ib1, ib2, imagn, paw_ind);
346 }
347 }
348 }
349 }

352double Potential::calc_PAW_one_elec_energy(paw_potential_data_t const& pdd, const mdarray<double_complex, 4>& density_matrix,
353 const mdarray<double, 4>& paw_dij) const
354{
355 int ia = pdd.ia;
356 int paw_ind = pdd.ia_paw;
357
358 double_complex energy = 0.0;
359
360 for (int ib2 = 0; ib2 < pdd.atom_->mt_lo_basis_size(); ib2++) {
361 for (int ib1 = 0; ib1 < pdd.atom_->mt_lo_basis_size(); ib1++) {
362 double dm[4] = {0, 0, 0, 0};
363 switch (ctx_.num_mag_dims()) {
364 case 3: {
365 dm[2] = 2 * std::real(density_matrix(ib1, ib2, 2, ia));
366 dm[3] = -2 * std::imag(density_matrix(ib1, ib2, 2, ia));
367 }
368 case 1: {
369 dm[0] = std::real(density_matrix(ib1, ib2, 0, ia) + density_matrix(ib1, ib2, 1, ia));
370 dm[1] = std::real(density_matrix(ib1, ib2, 0, ia) - density_matrix(ib1, ib2, 1, ia));
371 break;
372 }
373 case 0: {
374 dm[0] = density_matrix(ib1, ib2, 0, ia).real();
375 break;
376 }
377 default: {
378 TERMINATE("calc_PAW_one_elec_energy FATAL ERROR!");
379 break;
380 }
381 }
382 for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
383 energy += dm[imagn] * paw_dij(ib1, ib2, imagn, paw_ind);
384 }
385 }
386 }
387
388 if (std::abs(energy.imag()) > 1e-10) {
389 std::stringstream s;
390 s << "PAW energy is not real: " << energy;
391 TERMINATE(s.str());
392 }
393
394 return energy.real();
395}

397void Potential::add_paw_Dij_to_atom_Dmtrx()
398{
399 #pragma omp parallel for
400 for (int i = 0; i < unit_cell_.num_paw_atoms(); i++) {
401 int ia = unit_cell_.paw_atom_index(i);
402 auto& atom = unit_cell_.atom(ia);
403
404 for (int imagn = 0; imagn < ctx_.num_mag_dims() + 1; imagn++) {
405 for (int ib2 = 0; ib2 < atom.mt_lo_basis_size(); ib2++) {
406 for (int ib1 = 0; ib1 < atom.mt_lo_basis_size(); ib1++) {
407 atom.d_mtrx(ib1, ib2, imagn) += paw_dij_(ib1, ib2, imagn, i);
408 }
409 }
410 }
411 }
