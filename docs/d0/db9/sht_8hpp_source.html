263  /// Convert form Rlm to Ylm representation.
264  static void convert(int lmax__, double const* f_rlm__, double_complex* f_ylm__)
265  {
266  int lm = 0;
267  for (int l = 0; l <= lmax__; l++) {
268  for (int m = -l; m <= l; m++) {
269  if (m == 0) {
270  f_ylm__[lm] = f_rlm__[lm];
271  } else {
272  int lm1 = utils::lm(l, -m);
273  f_ylm__[lm] = ylm_dot_rlm(l, m, m) * f_rlm__[lm] + ylm_dot_rlm(l, m, -m) * f_rlm__[lm1];
274  }
275  lm++;
276  }
277  }
278  }

281  static void convert(int lmax__, double_complex const* f_ylm__, double* f_rlm__)
282  {

285  for (int m = -l; m <= l; m++) {

288  } else {
289  int lm1 = utils::lm(l, -m);
290  f_rlm__[lm] = std::real(rlm_dot_ylm(l, m, m) * f_ylm__[lm] + rlm_dot_ylm(l, m, -m) * f_ylm__[lm1]);
291  }
292  lm++;
293  }
294  }

332  /// Transform Cartesian coordinates [x,y,z] to spherical coordinates [r,theta,phi]
333  static geometry3d::vector3d<double> spherical_coordinates(geometry3d::vector3d<double> vc)
334  {
335  geometry3d::vector3d<double> vs;
336 
337  const double eps{1e-12};
338 
339  vs[0] = vc.length();
340 
341  if (vs[0] <= eps) {
342  vs[1] = 0.0;
343  vs[2] = 0.0;
344  } else {
345  vs[1] = std::acos(vc[2] / vs[0]); // theta = cos^{-1}(z/r)
346 
347  if (std::abs(vc[0]) > eps || std::abs(vc[1]) > eps) {
348  vs[2] = std::atan2(vc[1], vc[0]); // phi = tan^{-1}(y/x)
349  if (vs[2] < 0.0) {
350  vs[2] += twopi;
351  }
352  } else {
353  vs[2] = 0.0;
354  }
355  }
356 
357  return vs;
358  }

360  /// Compute element of the transformation matrix from complex to real spherical harmonics.
361  /** Real spherical harmonic can be written as a linear combination of complex harmonics:
362  \f[
363  R_{\ell m}(\theta, \phi) = \sum_{m'} a^{\ell}_{m' m}Y_{\ell m'}(\theta, \phi)
364  \f]
365  where
366  \f[
367  a^{\ell}_{m' m} = \langle Y_{\ell m'} | R_{\ell m} \rangle
368  \f]
369  which gives the name to this function.
370 
371  Transformation from real to complex spherical harmonics is conjugate transpose:
372 
373  \f[
374  Y_{\ell m}(\theta, \phi) = \sum_{m'} a^{\ell*}_{m m'}R_{\ell m'}(\theta, \phi)
375  \f]
376 
377  Mathematica code:
378  \verbatim
379  b[m1_, m2_] :=
380  If[m1 == 0, 1,
381  If[m1 < 0 && m2 < 0, -I/Sqrt[2],
382  If[m1 > 0 && m2 < 0, (-1)^m1*I/Sqrt[2],
383  If[m1 < 0 && m2 > 0, (-1)^m2/Sqrt[2],
384  If[m1 > 0 && m2 > 0, 1/Sqrt[2]]]]]]
385 
386  a[m1_, m2_] := If[Abs[m1] == Abs[m2], b[m1, m2], 0]
387 
388  Rlm[l_, m_, t_, p_] := Sum[a[m1, m]*SphericalHarmonicY[l, m1, t, p], {m1, -l, l}]
389  \endverbatim
390  */
391  static inline double_complex ylm_dot_rlm(int l, int m1, int m2)
392  {
393  double const isqrt2 = 1.0 / std::sqrt(2);
394 
395  assert(l >= 0 && std::abs(m1) <= l && std::abs(m2) <= l);
396 
397  if (!((m1 == m2) || (m1 == -m2))) {
398  return double_complex(0, 0);
399  }
400 
401  if (m1 == 0) {
402  return double_complex(1, 0);
403  }
404 
405  if (m1 < 0) {
406  if (m2 < 0) {
407  return -double_complex(0, isqrt2);
408  } else {
409  return std::pow(-1.0, m2) * double_complex(isqrt2, 0);
410  }
411  } else {
412  if (m2 < 0) {
413  return std::pow(-1.0, m1) * double_complex(0, isqrt2);
414  } else {
415  return double_complex(isqrt2, 0);
416  }
417  }
418  }

425  /// Gaunt coefficent of three complex spherical harmonics.
426  /**
427  * \f[
428  * \langle Y_{\ell_1 m_1} | Y_{\ell_2 m_2} | Y_{\ell_3 m_3} \rangle
429  * \f]
430  */
431  static double gaunt_yyy(int l1, int l2, int l3, int m1, int m2, int m3)
432  {
433  assert(l1 >= 0);
434  assert(l2 >= 0);
435  assert(l3 >= 0);
436  assert(m1 >= -l1 && m1 <= l1);
437  assert(m2 >= -l2 && m2 <= l2);
438  assert(m3 >= -l3 && m3 <= l3);
439 
440  return std::pow(-1.0, std::abs(m1)) * std::sqrt(double(2 * l1 + 1) * double(2 * l2 + 1) * double(2 * l3 + 1) / fourpi) *
441  gsl_sf_coupling_3j(2 * l1, 2 * l2, 2 * l3, 0, 0, 0) *
442  gsl_sf_coupling_3j(2 * l1, 2 * l2, 2 * l3, -2 * m1, 2 * m2, 2 * m3);
443  }
444 
445  /// Gaunt coefficent of three real spherical harmonics.
446  /**
447  * \f[
448  * \langle R_{\ell_1 m_1} | R_{\ell_2 m_2} | R_{\ell_3 m_3} \rangle
449  * \f]
450  */
451  static double gaunt_rrr(int l1, int l2, int l3, int m1, int m2, int m3)
452  {
453  assert(l1 >= 0);
454  assert(l2 >= 0);
455  assert(l3 >= 0);
456  assert(m1 >= -l1 && m1 <= l1);
457  assert(m2 >= -l2 && m2 <= l2);
458  assert(m3 >= -l3 && m3 <= l3);
459 
460  double d = 0;
461  for (int k1 = -l1; k1 <= l1; k1++) {
462  for (int k2 = -l2; k2 <= l2; k2++) {
463  for (int k3 = -l3; k3 <= l3; k3++) {
464  d += std::real(std::conj(SHT::ylm_dot_rlm(l1, k1, m1)) *
465  SHT::ylm_dot_rlm(l2, k2, m2) *
466  SHT::ylm_dot_rlm(l3, k3, m3)) *
467  SHT::gaunt_yyy(l1, l2, l3, k1, k2, k3);
468  }
469  }
470  }
471  return d;
472  }

476  * \f[
477  * \langle R_{\ell_1 m_1} | Y_{\ell_2 m_2} | R_{\ell_3 m_3} \rangle

480  static double gaunt_rlm_ylm_rlm(int l1, int l2, int l3, int m1, int m2, int m3)

500  /// Gaunt coefficent of two complex and one real spherical harmonics.
501  /**
502  * \f[
503  * \langle Y_{\ell_1 m_1} | R_{\ell_2 m_2} | Y_{\ell_3 m_3} \rangle
504  * \f]
505  */
506  static double_complex gaunt_hybrid(int l1, int l2, int l3, int m1, int m2, int m3)
507  {
508  assert(l1 >= 0);
509  assert(l2 >= 0);
510  assert(l3 >= 0);
511  assert(m1 >= -l1 && m1 <= l1);
512  assert(m2 >= -l2 && m2 <= l2);
513  assert(m3 >= -l3 && m3 <= l3);
514 
515  if (m2 == 0) {
516  return double_complex(gaunt_yyy(l1, l2, l3, m1, m2, m3), 0.0);
517  } else {
518  return (ylm_dot_rlm(l2, m2, m2) * gaunt_yyy(l1, l2, l3, m1, m2, m3) +
519  ylm_dot_rlm(l2, -m2, m2) * gaunt_yyy(l1, l2, l3, m1, -m2, m3));
520  }
521  }

523  void uniform_coverage()
524  {
525  tp_(0, 0) = pi;
526  tp_(1, 0) = 0;
527 
528  for (int k = 1; k < num_points_ - 1; k++) {
529  double hk = -1.0 + double(2 * k) / double(num_points_ - 1);
530  tp_(0, k) = std::acos(hk);
531  double t = tp_(1, k - 1) + 3.80925122745582 / std::sqrt(double(num_points_)) / std::sqrt(1 - hk * hk);
532  tp_(1, k) = std::fmod(t, twopi);
533  }
534 
535  tp_(0, num_points_ - 1) = 0;
536  tp_(1, num_points_ - 1) = 0;
537  }

539  /// Return Clebsch-Gordan coefficient.
540  /** Clebsch-Gordan coefficients arise when two angular momenta are combined into a
541  * total angular momentum.
542  */
543  static inline double clebsch_gordan(int l1, int l2, int l3, int m1, int m2, int m3)
544  {
545  assert(l1 >= 0);
546  assert(l2 >= 0);
547  assert(l3 >= 0);
548  assert(m1 >= -l1 && m1 <= l1);
549  assert(m2 >= -l2 && m2 <= l2);
550  assert(m3 >= -l3 && m3 <= l3);
551 
552  return std::pow(-1, l1 - l2 + m3) * std::sqrt(double(2 * l3 + 1)) *
