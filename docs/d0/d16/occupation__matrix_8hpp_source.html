42 * fill the d (f) states according to the hund's rules and with majority
43 * spin first and the remaining electrons distributed among the minority states. */
44 void init();

46 void reduce()
47 {
48 /* global reduction over k points */
49 for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
50 if (ctx_.unit_cell().atom(ia).type().hubbard_correction()) {
51 ctx_.comm_k().allreduce(this->local(ia).at(memory_t::host), static_cast<int>(this->local(ia).size()));
52 }
53 }
54 }
55
56 void zero()
57 {
58 Hubbard_matrix::zero();
59 for (auto& e: occ_mtrx_T_) {
60 e.second.zero();
61 }
62 }
