32 /// Stores <G+k | beta> expansion
33 class Beta_projectors : public Beta_projectors_base
34 {
35  protected:
36  bool prepared_{false};
37  matrix<double_complex> beta_pw_all_atoms_;
38  /// Generate plane-wave coefficients for beta-projectors of atom types.
39  void generate_pw_coefs_t(std::vector<int>& igk__)
40  {
41  PROFILE("sirius::Beta_projectors::generate_pw_coefs_t");
42  if (!num_beta_t()) {
43  return;
44  }
45 
46  auto& comm = gkvec_.comm();
47 
48  auto& beta_radial_integrals = ctx_.beta_ri();
49 
50  std::vector<double_complex> z(ctx_.unit_cell().lmax() + 1);
51  for (int l = 0; l <= ctx_.unit_cell().lmax(); l++) {
52  z[l] = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(ctx_.unit_cell().omega());
53  }
54 
55  /* compute <G+k|beta> */
56  #pragma omp parallel for
57  for (int igkloc = 0; igkloc < num_gkvec_loc(); igkloc++) {
58  int igk = igk__[igkloc];
59  /* vs = {r, theta, phi} */
60  auto vs = SHT::spherical_coordinates(gkvec_.gkvec_cart<index_domain_t::global>(igk));
61  /* compute real spherical harmonics for G+k vector */
62  std::vector<double> gkvec_rlm(utils::lmmax(ctx_.unit_cell().lmax()));
63  sf::spherical_harmonics(ctx_.unit_cell().lmax(), vs[1], vs[2], &gkvec_rlm[0]);
64  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
65  auto& atom_type = ctx_.unit_cell().atom_type(iat);
66  /* get all values of radial integrals */
67  auto ri_val = beta_radial_integrals.values(iat, vs[0]);
68  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
69  int l = atom_type.indexb(xi).l;
70  int lm = atom_type.indexb(xi).lm;
71  int idxrf = atom_type.indexb(xi).idxrf;
72 
73  pw_coeffs_t_(igkloc, atom_type.offset_lo() + xi, 0) = z[l] * gkvec_rlm[lm] * ri_val(idxrf);
74  }
75  }
76  }
77 
78  if (ctx_.cfg().control().print_checksum()) {
79  auto c1 = pw_coeffs_t_.checksum();
80  comm.allreduce(&c1, 1);
81  if (comm.rank() == 0) {
82  utils::print_checksum("beta_pw_coeffs_t", c1);
83  }
84  }
85 
86  }
87 
88  public:
89  Beta_projectors(Simulation_context& ctx__, Gvec const& gkvec__, std::vector<int>& igk__)
90  : Beta_projectors_base(ctx__, gkvec__, igk__, 1)
91  {
92  PROFILE("sirius::Beta_projectors");
93  /* generate phase-factor independent projectors for atom types */
94  generate_pw_coefs_t(igk__);
95  if (!num_beta_t()) {
96  return;
97  }
98  /* special treatment for beta-projectors as they are mostly often used */
99  switch (ctx_.processing_unit()) {
100  /* beta projectors for atom types will be stored on GPU for the entire run */
101  case device_t::GPU: {
102  reallocate_pw_coeffs_t_on_gpu_ = false;
103  pw_coeffs_t_.allocate(memory_t::device).copy_to(memory_t::device);
104  break;
105  }
106  /* generate beta projectors for all atoms */
107  case device_t::CPU: {
108  beta_pw_all_atoms_ = matrix<double_complex>(num_gkvec_loc(), ctx_.unit_cell().mt_lo_basis_size());
109  for (int ichunk = 0; ichunk < num_chunks(); ichunk++) {
110  /* wrap the the pointer in the big array beta_pw_all_atoms */
111  pw_coeffs_a_ = matrix<double_complex>(&beta_pw_all_atoms_(0, chunk(ichunk).offset_),
112  num_gkvec_loc(), chunk(ichunk).num_beta_);
113  Beta_projectors_base::generate(ichunk, 0);
114  }
115  break;
116  }
117  }
118  }

125  switch (ctx_.processing_unit()) {
126  case device_t::GPU: {
127  Beta_projectors_base::prepare();
128  break;

138  TERMINATE("beta projectors are already dismissed");
139  }
140  switch (ctx_.processing_unit()) {
141  case device_t::GPU: {
142  Beta_projectors_base::dismiss();
143  break;
144  }
145  case device_t::CPU: break;
146  }
147  prepared_ = false;
148  }
149 
150  void generate(int chunk__)
151  {
152  switch (ctx_.processing_unit()) {
153  case device_t::CPU: {
154  pw_coeffs_a_ = matrix<double_complex>(&beta_pw_all_atoms_(0, chunk(chunk__).offset_),
155  num_gkvec_loc(), chunk(chunk__).num_beta_);
156  break;
157  }
158  case device_t::GPU: {
159  Beta_projectors_base::generate(chunk__, 0);
160  break;
161  }
162  }
163  }

168  }
