32 /// Stores <G+k | beta> expansion
33 class Beta_projectors : public Beta_projectors_base
34 {
35  protected:
36  matrix<double_complex> beta_pw_all_atoms_;
37  /// Generate plane-wave coefficients for beta-projectors of atom types.
38  void generate_pw_coefs_t(std::vector<int>& igk__)
39  {
40  PROFILE("sirius::Beta_projectors::generate_pw_coefs_t");
41  if (!num_beta_t()) {
42  return;
43  }
44 
45  auto& comm = gkvec_.comm();
46 
47  auto& beta_radial_integrals = ctx_.beta_ri();
48 
49  std::vector<double_complex> z(ctx_.unit_cell().lmax() + 1);
50  for (int l = 0; l <= ctx_.unit_cell().lmax(); l++) {
51  z[l] = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(ctx_.unit_cell().omega());
52  }
53 
54  /* compute <G+k|beta> */
55  #pragma omp parallel for
56  for (int igkloc = 0; igkloc < num_gkvec_loc(); igkloc++) {
57  int igk = igk__[igkloc];
58  /* vs = {r, theta, phi} */
59  auto vs = SHT::spherical_coordinates(gkvec_.gkvec_cart<index_domain_t::global>(igk));
60  /* compute real spherical harmonics for G+k vector */
61  std::vector<double> gkvec_rlm(utils::lmmax(ctx_.unit_cell().lmax()));
62  SHT::spherical_harmonics(ctx_.unit_cell().lmax(), vs[1], vs[2], &gkvec_rlm[0]);
63  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
64  auto& atom_type = ctx_.unit_cell().atom_type(iat);
65  /* get all values of radial integrals */
66  auto ri_val = beta_radial_integrals.values(iat, vs[0]);
67  for (int xi = 0; xi < atom_type.mt_basis_size(); xi++) {
68  int l = atom_type.indexb(xi).l;
69  int lm = atom_type.indexb(xi).lm;
70  int idxrf = atom_type.indexb(xi).idxrf;
71 
72  pw_coeffs_t_(igkloc, atom_type.offset_lo() + xi, 0) = z[l] * gkvec_rlm[lm] * ri_val(idxrf);
73  }
74  }
75  }
76 
77  if (ctx_.control().print_checksum_) {
78  auto c1 = pw_coeffs_t_.checksum();
79  comm.allreduce(&c1, 1);
80  if (comm.rank() == 0) {
81  utils::print_checksum("beta_pw_coeffs_t", c1);
82  }
83  }
84 
85  }
86 
87  public:
88  Beta_projectors(Simulation_context& ctx__, Gvec const& gkvec__, std::vector<int>& igk__)
89  : Beta_projectors_base(ctx__, gkvec__, igk__, 1)
90  {
91  PROFILE("sirius::Beta_projectors::Beta_projectors");
92  /* generate phase-factor independent projectors for atom types */
93  generate_pw_coefs_t(igk__);
94  /* special treatment for beta-projectors as they are mostly often used */
95  switch (ctx_.processing_unit()) {
96  /* beta projectors for atom types will be stored on GPU for the entire run */
97  case device_t::GPU: {
98  reallocate_pw_coeffs_t_on_gpu_ = false;
99  pw_coeffs_t_.allocate(memory_t::device).copy_to(memory_t::device);
100  break;
101  }
102  /* generate beta projectors for all atoms */
103  case device_t::CPU: {
104  beta_pw_all_atoms_ = matrix<double_complex>(num_gkvec_loc(), ctx_.unit_cell().mt_lo_basis_size());
105  for (int ichunk = 0; ichunk < num_chunks(); ichunk++) {
106  /* wrap the the pointer in the big array beta_pw_all_atoms */
107  pw_coeffs_a_ = matrix<double_complex>(&beta_pw_all_atoms_(0, chunk(ichunk).offset_),
108  num_gkvec_loc(), chunk(ichunk).num_beta_);
109  Beta_projectors_base::generate(ichunk, 0);
110  }
111  break;
112  }
113  }
114  }

134  }

137  break;
138  }
