77 /// Use Kuebler's trick to get rho_up and rho_dn from density and magnetisation.
78 inline std::pair<double, double> get_rho_up_dn(int num_mag_dims__, double rho__, vector3d<double> mag__)
79 {
80  if (rho__ < 0.0) {
81  return std::make_pair<double, double>(0, 0);
82  }
83 
84  double mag{0};
85  if (num_mag_dims__ == 1) { /* collinear case */
86  mag = mag__[0];
87  /* fix numerical noise at high values of magnetization */
88  if (std::abs(mag) > rho__) {
89  mag = utils::sign(mag) * rho__;
90  }
91  } else { /* non-collinear case */
92  /* fix numerical noise at high values of magnetization */
93  mag = std::min(mag__.length(), rho__);
94  }
95 
96  return std::make_pair<double, double>(0.5 * (rho__ + mag), 0.5 * (rho__ - mag));
97 }

99 /// Generate charge density and magnetization from occupied spinor wave-functions.
100 /** Let's start from the definition of the complex density matrix:
101  \f[
102  \rho_{\sigma' \sigma}({\bf r}) =
103  \sum_{j{\bf k}} n_{j{\bf k}} \Psi_{j{\bf k}}^{\sigma*}({\bf r}) \Psi_{j{\bf k}}^{\sigma'}({\bf r}) =
104  \frac{1}{2} \left( \begin{array}{cc} \rho({\bf r})+m_z({\bf r}) &
105  m_x({\bf r})-im_y({\bf r}) \\ m_x({\bf r})+im_y({\bf r}) & \rho({\bf r})-m_z({\bf r}) \end{array} \right)
106  \f]
107  We notice that the diagonal components of the density matrix are actually real and the off-diagonal components are
108  expressed trough two independent functions \f$ m_x({\bf r}) \f$ and \f$ m_y({\bf r}) \f$. Having this in mind we
109  will work with a slightly different object, namely a real density matrix, defined as a 1-, 2- or 4-dimensional
110  (depending on the number of magnetic components) vector with the following elements:
111  - \f$ [ \rho({\bf r}) ] \f$ in case of non-magnetic configuration
112  - \f$ [ \rho_{\uparrow \uparrow}({\bf r}), \rho_{\downarrow \downarrow}({\bf r}) ] =
113  [ \frac{\rho({\bf r})+m_z({\bf r})}{2}, \frac{\rho({\bf r})-m_z({\bf r})}{2} ] \f$ in case of collinear
114  magnetic configuration
115  - \f$ [ \rho_{\uparrow \uparrow}({\bf r}), \rho_{\downarrow \downarrow}({\bf r}),
116  2 \Re \rho_{\uparrow \downarrow}({\bf r}), -2 \Im \rho_{\uparrow \downarrow}({\bf r}) ] =
117  [ \frac{\rho({\bf r})+m_z({\bf r})}{2}, \frac{\rho({\bf r})-m_z({\bf r})}{2},
118  m_x({\bf r}), m_y({\bf r}) ] \f$ in the general case of non-collinear magnetic configuration
119 
120  At this point it is straightforward to compute the density and magnetization in the interstitial
121  (see Density::add_k_point_contribution_rg()). The muffin-tin part of the density and magnetization is obtained
122  in a slighlty more complicated way. Recall the expansion of spinor wave-functions inside the muffin-tin
123  \f$ \alpha \f$:
124  \f[
125  \Psi_{j{\bf k}}^{\sigma}({\bf r}) = \sum_{\xi}^{N_{\xi}^{\alpha}} {S_{\xi}^{\sigma j {\bf k},\alpha}}
126  f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}(\hat {\bf r})
127  \f]
128  which we insert into expression for the complex density matrix:
129  \f[
130  \rho_{\sigma' \sigma}({\bf r}) = \sum_{j{\bf k}} n_{j{\bf k}} \sum_{\xi}^{N_{\xi}^{\alpha}}
131  S_{\xi}^{\sigma j {\bf k},\alpha*} f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)
132  Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r}) \sum_{\xi'}^{N_{\xi'}^{\alpha}} S_{\xi'}^{\sigma' j{\bf k},\alpha}
133  f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r})
134  \f]
135  First, we eliminate a sum over bands and k-points by forming an auxiliary density tensor:
136  \f[
137  D_{\xi \sigma, \xi' \sigma'}^{\alpha} = \sum_{j{\bf k}} n_{j{\bf k}} S_{\xi}^{\sigma j {\bf k},\alpha*}
138  S_{\xi'}^{\sigma' j {\bf k},\alpha}
139  \f]
140  The expression for complex density matrix simplifies to:
141  \f[
142  \rho_{\sigma' \sigma}({\bf r}) = \sum_{\xi \xi'} D_{\xi \sigma, \xi' \sigma'}^{\alpha}
143  f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r})
144  f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r})
145  \f]
146  Now we can switch to the real density matrix and write its' expansion in real spherical harmonics. Let's take
147  non-magnetic case as an example:
148  \f[
149  \rho({\bf r}) = \sum_{\xi \xi'} D_{\xi \xi'}^{\alpha}
150  f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r})
151  f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r}) =
152  \sum_{\ell_3 m_3} \rho_{\ell_3 m_3}^{\alpha}(r) R_{\ell_3 m_3}(\hat {\bf r})
153  \f]
154  where
155  \f[
156  \rho_{\ell_3 m_3}^{\alpha}(r) = \sum_{\xi \xi'} D_{\xi \xi'}^{\alpha} f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)
157  f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r) \langle Y_{\ell_{\xi}m_{\xi}} | R_{\ell_3 m_3} |
158  Y_{\ell_{\xi'}m_{\xi'}} \rangle
159  \f]
160  We are almost done. Now it is time to switch to the full index notation
161  \f$ \xi \rightarrow \{ \ell \lambda m \} \f$ and sum over \a m and \a m' indices:
162  \f[
163  \rho_{\ell_3 m_3}^{\alpha}(r) = \sum_{\ell \lambda, \ell' \lambda'} f_{\ell \lambda}^{\alpha}(r)
164  f_{\ell' \lambda'}^{\alpha}(r) d_{\ell \lambda, \ell' \lambda', \ell_3 m_3}^{\alpha}
165  \f]
166  where
167  \f[
168  d_{\ell \lambda, \ell' \lambda', \ell_3 m_3}^{\alpha} =
169  \sum_{mm'} D_{\ell \lambda m, \ell' \lambda' m'}^{\alpha}
170  \langle Y_{\ell m} | R_{\ell_3 m_3} | Y_{\ell' m'} \rangle
171  \f]
172  This is our final answer: radial components of density and magnetization are expressed as a linear combination of
173  quadratic forms in radial functions.
174 
175  \note density and potential are allocated as global function because it's easier to load and save them.
176  \note In case of full-potential calculation valence + core electron charge density is computed.
177  \note In tcase of pseudopotential valence charge density is computed.
178  */

346  \f[
347  \tilde \rho({\bf G}) = \sum_{\alpha} \sum_{\xi \xi'} d_{\xi \xi'}^{\alpha} Q_{\xi' \xi}^{\alpha}({\bf G})

684  /// Symmetrize density matrix.
685  /** We start from the spectral represntation of the occupancy operator defined for the irreducible Brillouin
686  * zone:
687  * \f[
688  * \hat N_{IBZ} = \sum_{j} \sum_{{\bf k}}^{IBZ} | \Psi_{j{\bf k}}^{\sigma} \rangle w_{\bf k} n_{j{\bf k}}
689  * \langle \Psi_{j{\bf k}}^{\sigma'} |
690  * \f]
691  * and a set of localized orbitals with the pure angular character (this can be LAPW functions,
692  * beta-projectors or localized Hubbard orbitals) :
693  * \f[
694  * |\phi_{\ell m}^{\alpha {\bf T}}\rangle
695  * \f]
696  * The orbitals are labeled by the angular and azimuthal quantum numbers (\f$ \ell m \f$),
697  * atom index (\f$ \alpha \f$) and a lattice translation vector (\f$ {\bf T} \f$) such that:
698  * \f[
699  * \langle {\bf r} | \phi_{\ell m}^{\alpha {\bf T}} \rangle = \phi_{\ell m}({\bf r - r_{\alpha} - T})
700  * \f]
701  *
702  * There might be several localized orbitals per atom. We wish to compute the symmetrized occupation matrix:
703  * \f[
704  * n_{\ell m \alpha {\bf T} \sigma, \ell' m' \alpha' {\bf T}' \sigma'} =
705  * \langle \phi_{\ell m}^{\alpha {\bf T}} | \hat N | \phi_{\ell' m'}^{\alpha' {\bf T'}} \rangle =
706  * \sum_{\bf P} \sum_{j} \sum_{{\bf k}}^{IBZ}
707  * \langle \phi_{\ell m}^{\alpha {\bf T}} | \hat{\bf P} \Psi_{j{\bf k}}^{\sigma} \rangle
708  * w_{\bf k} n_{j{\bf k}}
709  * \langle \hat{\bf P} \Psi_{j{\bf k}}^{\sigma'} | \phi_{\ell' m'}^{\alpha' {\bf T'}} \rangle
710  * \f]
711  *
712  * Let's now label the overlap integrals between localized orbitals and KS wave-functions:
713  * \f[
714  * A_{\ell m j{\bf k}}^{\alpha {\bf T} \sigma} = \langle \Psi_{j{\bf k}}^{\sigma} |
715  * \phi_{\ell m}^{\alpha {\bf T}} \rangle =
716  * \int \Psi_{j{\bf k}}^{\sigma *}({\bf r})
717  * \phi_{\ell m}({\bf r} - {\bf r}_{\alpha} - {\bf T}) d{\bf r}
718  * \f]
719  * and check how it transforms under the symmetry operation \f$ \hat {\bf P} = \{ {\bf R} | {\bf t} \} \f$
720  * applied to the KS states.
721  * \f[
722  * \int \big( \hat {\bf P}\Psi_{j {\bf k}}^{\sigma *}({\bf r}) \big)
723  * \phi_{\ell m}({\bf r} - {\bf r}_{\alpha} - {\bf T}) d{\bf r} =
724  * \int \Psi_{j {\bf k}}^{\sigma *}({\bf r})
725  * \big( \hat {\bf P}^{-1} \phi_{\ell m}({\bf r} - {\bf r}_{\alpha} - {\bf T}) \big) d{\bf r}
726  * \f]
727  *
728  * Let's first derive how the inverse symmetry operation acts on the localized orbital centered on the
729  * atom inside a unit cell (no <b>T</b>):
730  * \f[
731  * \hat {\bf P}^{-1} \phi\big( {\bf r} - {\bf r}_{\alpha} \big) =
732  * \phi\big( {\bf R} {\bf r} + {\bf t} - {\bf r}_{\alpha} \big) = \\
733  * \phi\big( {\bf R}({\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} - {\bf t})) \big) =
734  * \tilde \phi \big( {\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} - {\bf t}) \big)
735  * \f]
736  * This operation rotates the orbital and centers it at the position
737  * \f[
738  * {\bf r}_{\beta} = {\bf R}^{-1}{\bf r}_{\alpha} - {\bf R}^{-1}{\bf t} = \hat {\bf P}^{-1}{\bf r}_{\alpha}
739  * \f]
740  *
741  *
742  * For example, suppose thar we have y-orbital centered at \f$ {\bf r}_{\alpha} = [1, 0] \f$ (black dot)
743  * and a symmetry operation \f$ \hat {\bf P} = \{ {\bf R} | {\bf t} \} \f$ that rotates by
744  * \f$ \pi/2 \f$ counterclockwise and translates by [1/2, 1/2]:
745  *
746  * \image html sym_orbital1.png width=400px
747  *
748  * Under this symmetry operation the atom coordinate will transform into [1/2, 3/2] (red dot), but
749  * this is not(!) how the orbital is transformed. The origin of the atom will transform according to
750  * the inverse of \f$ \hat {\bf P} \f$ into \f$ {\bf r}_{\beta} = [-1/2, -1/2] \f$ (blue dot) such that
751  * \f$ \hat {\bf P} {\bf r}_{\beta} = {\bf r}_{\alpha} \f$:
752  *
753  * \image html sym_orbital2.png width=400px
754  *
755  * To be more precise, we should highlight that the transformed atom coordinate can go out of the original
756  * unit cell and can be brought back with a translation vector:
757  * \f[
758  * \hat {\bf P}^{-1}{\bf r}_{\alpha} = {\bf r}_{\beta} + {\bf T}_{P\alpha\beta}
759  * \f]
760  *
761  * Now let's derive how the inverse symmetry operation acts on the localized orbital \f$ \phi({\bf r}) \f$
762  * centered on atom in the arbitrary unit cell:
763  * \f[
764  * \hat {\bf P}^{-1} \phi\big( {\bf r} - {\bf r}_{\alpha} - {\bf T} \big) =
765  * \phi\big( {\bf R} {\bf r} + {\bf t} - {\bf r}_{\alpha} - {\bf T} \big) = \\
766  * \phi\big( {\bf R}({\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} + {\bf T} - {\bf t})) \big) =
767  * \tilde \phi\big( {\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} + {\bf T} - {\bf t}) \big) =
768  * \tilde \phi\big( {\bf r} - {\bf r}_{\beta} - {\bf T}_{P\alpha\beta} - {\bf R}^{-1}{\bf T} \big)
769  * \f]
770  *
771  * Now let's check how the atomic orbitals transfrom under the rotational part of the symmetry operation.
772  * The atomic functions of (\f$ \ell m \f$) character is expressed as a product of radial function and a
773  * spherical harmonic:
774  * \f[
775  * \phi_{\ell m}({\bf r}) = \phi_{\ell}(r) Y_{\ell m}(\theta, \phi)
776  * \f]
777  *
778  * Under rotation the spherical harmonic is transformed as:
779  * \f[
780  * Y_{\ell m}({\bf P} \hat{\bf r}) = {\bf P}^{-1}Y_{\ell m}(\hat {\bf r}) =
781  * \sum_{m'} D_{m'm}^{\ell}({\bf P}^{-1}) Y_{\ell m'}(\hat {\bf r}) =
782  * \sum_{m'} D_{mm'}^{\ell}({\bf P}) Y_{\ell m'}(\hat {\bf r})
783  * \f]
784  * so
785  * \f[
786  * \tilde \phi_{\ell m}({\bf r}) =\hat {\bf P}^{-1} \phi_{\ell}(r) Y_{\ell m}(\theta, \phi) =
787  * \sum_{m'} D_{mm'}^{\ell}({\bf P}) \phi_{\ell}(r) Y_{\ell m'}(\theta, \phi)
788  * \f]
789  *
790  * We will use Bloch theorem to get rid of the translations in the argument of \f$ \tilde \phi \f$:
791  * \f[
792  * \int \Psi_{j {\bf k}}^{\sigma *}({\bf r})
793  * \tilde \phi_{\ell m} \big( {\bf r} - {\bf r}_{\beta} - {\bf T}_{P\alpha\beta} - {\bf R}^{-1}{\bf T} \big)
794  * d{\bf r} =
795  * e^{i{\bf k}({\bf T}_{P\alpha\beta} + {\bf R}^{-1}{\bf T})} \int \Psi_{j {\bf k}}^{\sigma *}({\bf r})
796  * \tilde \phi_{\ell m} \big( {\bf r} - {\bf r}_{\beta} \big) d{\bf r}
797  * \f]
798  * and now we can write
799  * \f[
800  * A_{\ell m j\hat {\bf P}{\bf k}}^{\alpha {\bf T} \sigma} =
801  * e^{i{\bf k}({\bf T}_{P\alpha\beta} + {\bf R}^{-1}{\bf T})}
802  * \sum_{m'} D_{mm'}^{\ell}({\bf P}) A_{\ell m' j{\bf k}}^{\beta \sigma}
803  * \f]
804  *
805  * The final expression for the symmetrized matrix is then
806  * \f[
807  * n_{\ell m \alpha {\bf T} \sigma, \ell' m' \alpha' {\bf T}' \sigma'} =
808  * \sum_{\bf P} \sum_{j} \sum_{{\bf k}}^{IBZ}
809  * A_{\ell m j\hat {\bf P}{\bf k}}^{\alpha {\bf T} \sigma *}
810  * w_{\bf k} n_{j{\bf k}}
811  * A_{\ell' m' j\hat {\bf P}{\bf k}}^{\alpha' {\bf T'} \sigma'} = \\ = \sum_{\bf P} \sum_{j}
812  * \sum_{{\bf k}}^{IBZ}
813  * e^{-i{\bf k}({\bf T}_{P\alpha\beta} + {\bf R}^{-1}{\bf T})}
814  * e^{i{\bf k}({\bf T}_{P\alpha'\beta'} + {\bf R}^{-1}{\bf T'})}
815  * \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
816  * A_{\ell m_1 j{\bf k}}^{\beta \sigma *} A_{\ell' m_2 j{\bf k}}^{\beta' \sigma'} w_{\bf k} n_{j{\bf k}}
817  * \f]
818  *
819  * In the case of \f$ \alpha = \alpha' \f$ and \f$ {\bf T}={\bf T}' \f$ all the phase-factor exponents disappear
820  * and we get an expression for the "on-site" occupation matrix:
821  *
822  * \f[
823  * n_{\ell m \sigma, \ell' m' \sigma'}^{\alpha} =
824  * \sum_{\bf P} \sum_{j} \sum_{{\bf k}}^{IBZ}
825  * \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
826  * A_{\ell m_1 j{\bf k}}^{\beta \sigma *} A_{\ell' m_2 j{\bf k}}^{\beta \sigma'}
827  * w_{\bf k} n_{j{\bf k}} = \\ =
828  * \sum_{\bf P} \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
829  * \tilde n_{\ell m_1 \sigma, \ell' m_2 \sigma'}^{\beta}
830  * \f]
831  *
832  *
833  * To compute the overlap integrals between KS wave-functions and localized Hubbard orbitals we insert
834  * resolution of identity (in \f$ {\bf G+k} \f$ planve-waves) between bra and ket:
835  * \f[
836  * \langle \phi_{\ell m}^{\alpha} | \Psi_{j{\bf k}}^{\sigma} \rangle = \sum_{\bf G}
837  * \phi_{\ell m}^{\alpha *}({\bf G+k}) \Psi_{j}^{\sigma}({\bf G+k})
838  * \f]
839  *
840  */
841  void symmetrize_density_matrix();

953  PROFILE("sirius::get_rho_up_dn");
954 
955  auto& ctx = const_cast<Simulation_context&>(density__.ctx());
956  int num_points = ctx.spfft().local_slice_size();
957 
958  auto rho_up = std::unique_ptr<Smooth_periodic_function<double>>(new
959  Smooth_periodic_function<double>(ctx.spfft(), ctx.gvec_partition()));
960  auto rho_dn = std::unique_ptr<Smooth_periodic_function<double>>(new
961  Smooth_periodic_function<double>(ctx.spfft(), ctx.gvec_partition()));
962 
963  /* compute "up" and "dn" components and also check for negative values of density */
964  double rhomin{0};
965  #pragma omp parallel for reduction(min:rhomin)
966  for (int ir = 0; ir < num_points; ir++) {
967  vector3d<double> m;
968  for (int j = 0; j < ctx.num_mag_dims(); j++) {
969  m[j] = density__.magnetization(j).f_rg(ir) * (1 + add_delta_mag_xc__);
970  }
971 
972  double rho = density__.rho().f_rg(ir);
973  if (add_pseudo_core__) {
974  rho += density__.rho_pseudo_core().f_rg(ir);
975  }
976  rho *= (1 + add_delta_rho_xc__);
977  rhomin = std::min(rhomin, rho);
978  auto rud = get_rho_up_dn(ctx.num_mag_dims(), rho, m);
979 
980  rho_up->f_rg(ir) = rud.first;
981  rho_dn->f_rg(ir) = rud.second;
982  }
983 
984  Communicator(ctx.spfft().communicator()).allreduce<double, mpi_op_t::min>(&rhomin, 1);
985  if (rhomin< 0.0 && ctx.comm().rank() == 0) {
986  std::stringstream s;
987  s << "Interstitial charge density has negative values" << std::endl
988  << "most negatve value : " << rhomin;
989  WARNING(s);
990  }
991  std::array<std::unique_ptr<Smooth_periodic_function<double>>, 2> result;
992  result[0] = std::move(rho_up);
993  result[1] = std::move(rho_dn);
994  return result;
995 }
