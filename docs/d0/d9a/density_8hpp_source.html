94/// Use Kuebler's trick to get rho_up and rho_dn from density and magnetisation.
95inline std::pair<double, double> get_rho_up_dn(int num_mag_dims__, double rho__, vector3d<double> mag__)
96{
97 if (rho__ < 0.0) {
98 return std::make_pair<double, double>(0, 0);
99 }
100
101 double mag{0};
102 if (num_mag_dims__ == 1) { /* collinear case */
103 mag = mag__[0];
104 /* fix numerical noise at high values of magnetization */
105 if (std::abs(mag) > rho__) {
106 mag = utils::sign(mag) * rho__;
107 }
108 } else { /* non-collinear case */
109 /* fix numerical noise at high values of magnetization */
110 mag = std::min(mag__.length(), rho__);
111 }
112
113 return std::make_pair<double, double>(0.5 * (rho__ + mag), 0.5 * (rho__ - mag));
114}

116/// Generate charge density and magnetization from occupied spinor wave-functions.
117/** Let's start from the definition of the complex density matrix:
118 \f[
119 \rho_{\sigma' \sigma}({\bf r}) =
120 \sum_{j{\bf k}} n_{j{\bf k}} \Psi_{j{\bf k}}^{\sigma*}({\bf r}) \Psi_{j{\bf k}}^{\sigma'}({\bf r}) =
121 \frac{1}{2} \left( \begin{array}{cc} \rho({\bf r})+m_z({\bf r}) &
122 m_x({\bf r})-im_y({\bf r}) \\ m_x({\bf r})+im_y({\bf r}) & \rho({\bf r})-m_z({\bf r}) \end{array} \right)
123 \f]
124 We notice that the diagonal components of the density matrix are actually real and the off-diagonal components are
125 expressed trough two independent functions \f$ m_x({\bf r}) \f$ and \f$ m_y({\bf r}) \f$. Having this in mind we
126 will work with a slightly different object, namely a real density matrix, defined as a 1-, 2- or 4-dimensional
127 (depending on the number of magnetic components) vector with the following elements:
128 - \f$ [ \rho({\bf r}) ] \f$ in case of non-magnetic configuration
129 - \f$ [ \rho_{\uparrow \uparrow}({\bf r}), \rho_{\downarrow \downarrow}({\bf r}) ] =
130 [ \frac{\rho({\bf r})+m_z({\bf r})}{2}, \frac{\rho({\bf r})-m_z({\bf r})}{2} ] \f$ in case of collinear
131 magnetic configuration
132 - \f$ [ \rho_{\uparrow \uparrow}({\bf r}), \rho_{\downarrow \downarrow}({\bf r}),
133 2 \Re \rho_{\uparrow \downarrow}({\bf r}), -2 \Im \rho_{\uparrow \downarrow}({\bf r}) ] =
134 [ \frac{\rho({\bf r})+m_z({\bf r})}{2}, \frac{\rho({\bf r})-m_z({\bf r})}{2},
135 m_x({\bf r}), m_y({\bf r}) ] \f$ in the general case of non-collinear magnetic configuration
136
137 At this point it is straightforward to compute the density and magnetization in the interstitial
138 (see Density::add_k_point_contribution_rg()). The muffin-tin part of the density and magnetization is obtained
139 in a slighlty more complicated way. Recall the expansion of spinor wave-functions inside the muffin-tin
140 \f$ \alpha \f$:
141 \f[
142 \Psi_{j{\bf k}}^{\sigma}({\bf r}) = \sum_{\xi}^{N_{\xi}^{\alpha}} {S_{\xi}^{\sigma j {\bf k},\alpha}}
143 f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}(\hat {\bf r})
144 \f]
145 which we insert into expression for the complex density matrix:
146 \f[
147 \rho_{\sigma' \sigma}({\bf r}) = \sum_{j{\bf k}} n_{j{\bf k}} \sum_{\xi}^{N_{\xi}^{\alpha}}
148 S_{\xi}^{\sigma j {\bf k},\alpha*} f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)
149 Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r}) \sum_{\xi'}^{N_{\xi'}^{\alpha}} S_{\xi'}^{\sigma' j{\bf k},\alpha}
150 f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r})
151 \f]
152 First, we eliminate a sum over bands and k-points by forming an auxiliary density tensor:
153 \f[
154 D_{\xi \sigma, \xi' \sigma'}^{\alpha} = \sum_{j{\bf k}} n_{j{\bf k}} S_{\xi}^{\sigma j {\bf k},\alpha*}
155 S_{\xi'}^{\sigma' j {\bf k},\alpha}
156 \f]
157 The expression for complex density matrix simplifies to:
158 \f[
159 \rho_{\sigma' \sigma}({\bf r}) = \sum_{\xi \xi'} D_{\xi \sigma, \xi' \sigma'}^{\alpha}
160 f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r})
161 f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r})
162 \f]
163 Now we can switch to the real density matrix and write its' expansion in real spherical harmonics. Let's take
164 non-magnetic case as an example:
165 \f[
166 \rho({\bf r}) = \sum_{\xi \xi'} D_{\xi \xi'}^{\alpha}
167 f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)Y_{\ell_{\xi}m_{\xi}}^{*}(\hat {\bf r})
168 f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r)Y_{\ell_{\xi'}m_{\xi'}}(\hat {\bf r}) =
169 \sum_{\ell_3 m_3} \rho_{\ell_3 m_3}^{\alpha}(r) R_{\ell_3 m_3}(\hat {\bf r})
170 \f]
171 where
172 \f[
173 \rho_{\ell_3 m_3}^{\alpha}(r) = \sum_{\xi \xi'} D_{\xi \xi'}^{\alpha} f_{\ell_{\xi} \lambda_{\xi}}^{\alpha}(r)
174 f_{\ell_{\xi'} \lambda_{\xi'}}^{\alpha}(r) \langle Y_{\ell_{\xi}m_{\xi}} | R_{\ell_3 m_3} |
175 Y_{\ell_{\xi'}m_{\xi'}} \rangle
176 \f]
177 We are almost done. Now it is time to switch to the full index notation
178 \f$ \xi \rightarrow \{ \ell \lambda m \} \f$ and sum over \a m and \a m' indices:
179 \f[
180 \rho_{\ell_3 m_3}^{\alpha}(r) = \sum_{\ell \lambda, \ell' \lambda'} f_{\ell \lambda}^{\alpha}(r)
181 f_{\ell' \lambda'}^{\alpha}(r) d_{\ell \lambda, \ell' \lambda', \ell_3 m_3}^{\alpha}
182 \f]
183 where
184 \f[
185 d_{\ell \lambda, \ell' \lambda', \ell_3 m_3}^{\alpha} =
186 \sum_{mm'} D_{\ell \lambda m, \ell' \lambda' m'}^{\alpha}
187 \langle Y_{\ell m} | R_{\ell_3 m_3} | Y_{\ell' m'} \rangle
188 \f]
189 This is our final answer: radial components of density and magnetization are expressed as a linear combination of
190 quadratic forms in radial functions.
191
192 \note density and potential are allocated as global function because it's easier to load and save them.
193 \note In case of full-potential calculation valence + core electron charge density is computed.
194 \note In tcase of pseudopotential valence charge density is computed.
195 */

262 d_{\xi \sigma, \xi' \sigma'}^{\alpha} = \sum_{j{\bf k}} n_{j{\bf k}}

271 n_{\ell,mm'}^{\sigma \sigma'} = \sum_{i {\bf k}}^{occ} \int_{0}^{R_{MT}} r^2 dr

369 /// Add augmentation charge Q(r).
370 /** Restore valence density by adding the Q-operator constribution.
371 The following term is added to the valence density, generated by the pseudo wave-functions:
372 \f[
373 \tilde \rho({\bf G}) = \sum_{\alpha} \sum_{\xi \xi'} d_{\xi \xi'}^{\alpha} Q_{\xi' \xi}^{\alpha}({\bf G})
374 \f]
375 Plane-wave coefficients of the Q-operator for a given atom \f$ \alpha \f$ can be obtained from the
376 corresponding coefficients of the Q-operator for a given atom \a type A:
377 \f[
378 Q_{\xi' \xi}^{\alpha(A)}({\bf G}) = e^{-i{\bf G}\tau_{\alpha(A)}} Q_{\xi' \xi}^{A}({\bf G})
379 \f]

381 same type:
382 \f[
383 \tilde \rho({\bf G}) = \sum_{A} \sum_{\xi \xi'} Q_{\xi' \xi}^{A}({\bf G}) \sum_{\alpha(A)}
384 d_{\xi \xi'}^{\alpha(A)} e^{-i{\bf G}\tau_{\alpha(A)}} =
385 \sum_{A} \sum_{\xi \xi'} Q_{\xi' \xi}^{A}({\bf G}) d_{\xi \xi'}^{A}({\bf G})
386 \f]
387 where
388 \f[
389 d_{\xi \xi'}^{A}({\bf G}) = \sum_{\alpha(A)} d_{\xi \xi'}^{\alpha(A)} e^{-i{\bf G}\tau_{\alpha(A)}}

710 /// Symmetrize density matrix.
711 /** We start from the spectral represntation of the occupancy operator defined for the irreducible Brillouin
712 * zone:
713 * \f[
714 * \hat N_{IBZ} = \sum_{j} \sum_{{\bf k}}^{IBZ} | \Psi_{j{\bf k}}^{\sigma} \rangle w_{\bf k} n_{j{\bf k}}
715 * \langle \Psi_{j{\bf k}}^{\sigma'} |
716 * \f]
717 * and a set of localized orbitals with the pure angular character (this can be LAPW functions,
718 * beta-projectors or localized Hubbard orbitals) :
719 * \f[
720 * |\phi_{\ell m}^{\alpha {\bf T}}\rangle
721 * \f]
722 * The orbitals are labeled by the angular and azimuthal quantum numbers (\f$ \ell m \f$),
723 * atom index (\f$ \alpha \f$) and a lattice translation vector (\f$ {\bf T} \f$) such that:
724 * \f[
725 * \langle {\bf r} | \phi_{\ell m}^{\alpha {\bf T}} \rangle = \phi_{\ell m}({\bf r - r_{\alpha} - T})
726 * \f]
727 *
728 * There might be several localized orbitals per atom. We wish to compute the symmetrized occupation matrix:
729 * \f[
730 * n_{\ell m \alpha {\bf T} \sigma, \ell' m' \alpha' {\bf T}' \sigma'} =
731 * \langle \phi_{\ell m}^{\alpha {\bf T}} | \hat N | \phi_{\ell' m'}^{\alpha' {\bf T'}} \rangle =
732 * \sum_{\bf P} \sum_{j} \sum_{{\bf k}}^{IBZ}
733 * \langle \phi_{\ell m}^{\alpha {\bf T}} | \hat{\bf P} \Psi_{j{\bf k}}^{\sigma} \rangle
734 * w_{\bf k} n_{j{\bf k}}
735 * \langle \hat{\bf P} \Psi_{j{\bf k}}^{\sigma'} | \phi_{\ell' m'}^{\alpha' {\bf T'}} \rangle
736 * \f]
737 *
738 * Let's now label the overlap integrals between localized orbitals and KS wave-functions:
739 * \f[
740 * A_{\ell m j{\bf k}}^{\alpha {\bf T} \sigma} = \langle \Psi_{j{\bf k}}^{\sigma} |
741 * \phi_{\ell m}^{\alpha {\bf T}} \rangle =
742 * \int \Psi_{j{\bf k}}^{\sigma *}({\bf r})
743 * \phi_{\ell m}({\bf r} - {\bf r}_{\alpha} - {\bf T}) d{\bf r}
744 * \f]
745 * and check how it transforms under the symmetry operation \f$ \hat {\bf P} = \{ {\bf R} | {\bf t} \} \f$
746 * applied to the KS states.
747 * \f[
748 * \int \big( \hat {\bf P}\Psi_{j {\bf k}}^{\sigma *}({\bf r}) \big)
749 * \phi_{\ell m}({\bf r} - {\bf r}_{\alpha} - {\bf T}) d{\bf r} =
750 * \int \Psi_{j {\bf k}}^{\sigma *}({\bf r})
751 * \big( \hat {\bf P}^{-1} \phi_{\ell m}({\bf r} - {\bf r}_{\alpha} - {\bf T}) \big) d{\bf r}
752 * \f]
753 *
754 * Let's first derive how the inverse symmetry operation acts on the localized orbital centered on the
755 * atom inside a unit cell (no <b>T</b>):
756 * \f[
757 * \hat {\bf P}^{-1} \phi\big( {\bf r} - {\bf r}_{\alpha} \big) =
758 * \phi\big( {\bf R} {\bf r} + {\bf t} - {\bf r}_{\alpha} \big) = \\
759 * \phi\big( {\bf R}({\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} - {\bf t})) \big) =
760 * \tilde \phi \big( {\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} - {\bf t}) \big)
761 * \f]
762 * This operation rotates the orbital and centers it at the position
763 * \f[
764 * {\bf r}_{\beta} = {\bf R}^{-1}{\bf r}_{\alpha} - {\bf R}^{-1}{\bf t} = \hat {\bf P}^{-1}{\bf r}_{\alpha}
765 * \f]
766 *
767 *
768 * For example, suppose thar we have y-orbital centered at \f$ {\bf r}_{\alpha} = [1, 0] \f$ (black dot)
769 * and a symmetry operation \f$ \hat {\bf P} = \{ {\bf R} | {\bf t} \} \f$ that rotates by
770 * \f$ \pi/2 \f$ counterclockwise and translates by [1/2, 1/2]:
771 *
772 * \image html sym_orbital1.png width=400px
773 *
774 * Under this symmetry operation the atom coordinate will transform into [1/2, 3/2] (red dot), but
775 * this is not(!) how the orbital is transformed. The origin of the atom will transform according to
776 * the inverse of \f$ \hat {\bf P} \f$ into \f$ {\bf r}_{\beta} = [-1/2, -1/2] \f$ (blue dot) such that
777 * \f$ \hat {\bf P} {\bf r}_{\beta} = {\bf r}_{\alpha} \f$:
778 *
779 * \image html sym_orbital2.png width=400px
780 *
781 * To be more precise, we should highlight that the transformed atom coordinate can go out of the original
782 * unit cell and can be brought back with a translation vector:
783 * \f[
784 * \hat {\bf P}^{-1}{\bf r}_{\alpha} = {\bf r}_{\beta} + {\bf T}_{P\alpha\beta}
785 * \f]
786 *
787 * Now let's derive how the inverse symmetry operation acts on the localized orbital \f$ \phi({\bf r}) \f$
788 * centered on atom in the arbitrary unit cell:
789 * \f[
790 * \hat {\bf P}^{-1} \phi\big( {\bf r} - {\bf r}_{\alpha} - {\bf T} \big) =
791 * \phi\big( {\bf R} {\bf r} + {\bf t} - {\bf r}_{\alpha} - {\bf T} \big) = \\
792 * \phi\big( {\bf R}({\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} + {\bf T} - {\bf t})) \big) =
793 * \tilde \phi\big( {\bf r} - {\bf R}^{-1}({\bf r}_{\alpha} + {\bf T} - {\bf t}) \big) =
794 * \tilde \phi\big( {\bf r} - {\bf r}_{\beta} - {\bf T}_{P\alpha\beta} - {\bf R}^{-1}{\bf T} \big)
795 * \f]
796 *
797 * Now let's check how the atomic orbitals transfrom under the rotational part of the symmetry operation.
798 * The atomic functions of (\f$ \ell m \f$) character is expressed as a product of radial function and a
799 * spherical harmonic:
800 * \f[
801 * \phi_{\ell m}({\bf r}) = \phi_{\ell}(r) Y_{\ell m}(\theta, \phi)
802 * \f]
803 *
804 * Under rotation the spherical harmonic is transformed as:
805 * \f[
806 * Y_{\ell m}({\bf P} \hat{\bf r}) = {\bf P}^{-1}Y_{\ell m}(\hat {\bf r}) =
807 * \sum_{m'} D_{m'm}^{\ell}({\bf P}^{-1}) Y_{\ell m'}(\hat {\bf r}) =
808 * \sum_{m'} D_{mm'}^{\ell}({\bf P}) Y_{\ell m'}(\hat {\bf r})
809 * \f]
810 * so
811 * \f[
812 * \tilde \phi_{\ell m}({\bf r}) =\hat {\bf P}^{-1} \phi_{\ell}(r) Y_{\ell m}(\theta, \phi) =
813 * \sum_{m'} D_{mm'}^{\ell}({\bf P}) \phi_{\ell}(r) Y_{\ell m'}(\theta, \phi)
814 * \f]
815 *
816 * We will use Bloch theorem to get rid of the translations in the argument of \f$ \tilde \phi \f$:
817 * \f[
818 * \int \Psi_{j {\bf k}}^{\sigma *}({\bf r})
819 * \tilde \phi_{\ell m} \big( {\bf r} - {\bf r}_{\beta} - {\bf T}_{P\alpha\beta} - {\bf R}^{-1}{\bf T} \big)
820 * d{\bf r} =
821 * e^{-i{\bf k}({\bf T}_{P\alpha\beta} + {\bf R}^{-1}{\bf T})} \int \Psi_{j {\bf k}}^{\sigma *}({\bf r})
822 * \tilde \phi_{\ell m} \big( {\bf r} - {\bf r}_{\beta} \big) d{\bf r}
823 * \f]
824 * (the "-" in the phase factor appears because KS wave-functions are complex conjugate) and now we can write
825 * \f[
826 * A_{\ell m j\hat {\bf P}{\bf k}}^{\alpha {\bf T} \sigma} =
827 * e^{-i{\bf k}({\bf T}_{P\alpha\beta} + {\bf R}^{-1}{\bf T})}
828 * \sum_{m'} D_{mm'}^{\ell}({\bf P}) A_{\ell m' j{\bf k}}^{\beta \sigma}
829 * \f]
830 *
831 * The final expression for the symmetrized matrix is then
832 * \f[
833 * n_{\ell m \alpha {\bf T} \sigma, \ell' m' \alpha' {\bf T}' \sigma'} =
834 * \sum_{\bf P} \sum_{j} \sum_{{\bf k}}^{IBZ}
835 * A_{\ell m j\hat {\bf P}{\bf k}}^{\alpha {\bf T} \sigma *}
836 * w_{\bf k} n_{j{\bf k}}
837 * A_{\ell' m' j\hat {\bf P}{\bf k}}^{\alpha' {\bf T'} \sigma'} = \\ = \sum_{\bf P} \sum_{j}
838 * \sum_{{\bf k}}^{IBZ}
839 * e^{i{\bf k}({\bf T}_{P\alpha\beta} + {\bf R}^{-1}{\bf T})}
840 * e^{-i{\bf k}({\bf T}_{P\alpha'\beta'} + {\bf R}^{-1}{\bf T'})}
841 * \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
842 * A_{\ell m_1 j{\bf k}}^{\beta \sigma *} A_{\ell' m_2 j{\bf k}}^{\beta' \sigma'} w_{\bf k} n_{j{\bf k}}
843 * \f]
844 *
845 * In the case of \f$ \alpha = \alpha' \f$ and \f$ {\bf T}={\bf T}' \f$ all the phase-factor exponents disappear
846 * and we get an expression for the "on-site" occupation matrix:
847 *
848 * \f[
849 * n_{\ell m \sigma, \ell' m' \sigma'}^{\alpha} =
850 * \sum_{\bf P} \sum_{j} \sum_{{\bf k}}^{IBZ}
851 * \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
852 * A_{\ell m_1 j{\bf k}}^{\beta \sigma *} A_{\ell' m_2 j{\bf k}}^{\beta \sigma'}
853 * w_{\bf k} n_{j{\bf k}} = \\ =
854 * \sum_{\bf P} \sum_{m_1 m_2} D_{mm_1}^{\ell *}({\bf P}) D_{m'm_2}^{\ell'}({\bf P})
855 * \tilde n_{\ell m_1 \sigma, \ell' m_2 \sigma'}^{\beta}
856 * \f]
857 *
858 *
859 * To compute the overlap integrals between KS wave-functions and localized Hubbard orbitals we insert
860 * resolution of identity (in \f$ {\bf G+k} \f$ planve-waves) between bra and ket:
861 * \f[
862 * \langle \phi_{\ell m}^{\alpha} | \Psi_{j{\bf k}}^{\sigma} \rangle = \sum_{\bf G}
863 * \phi_{\ell m}^{\alpha *}({\bf G+k}) \Psi_{j}^{\sigma}({\bf G+k})
864 * \f]
865 *
866 */
867 void symmetrize_density_matrix();

975template <bool add_pseudo_core__>
976inline std::array<std::unique_ptr<Smooth_periodic_function<double>>, 2>
977get_rho_up_dn(Density const& density__, double add_delta_rho_xc__ = 0.0, double add_delta_mag_xc__ = 0.0)
978{
979 PROFILE("sirius::get_rho_up_dn");
980
981 auto& ctx = const_cast<Simulation_context&>(density__.ctx());
982 int num_points = ctx.spfft<double>().local_slice_size();
983
984 auto rho_up = std::unique_ptr<Smooth_periodic_function<double>>(new
985 Smooth_periodic_function<double>(ctx.spfft<double>(), ctx.gvec_partition()));
986 auto rho_dn = std::unique_ptr<Smooth_periodic_function<double>>(new
987 Smooth_periodic_function<double>(ctx.spfft<double>(), ctx.gvec_partition()));
988
989 /* compute "up" and "dn" components and also check for negative values of density */
990 double rhomin{0};
991 #pragma omp parallel for reduction(min:rhomin)
992 for (int ir = 0; ir < num_points; ir++) {
993 vector3d<double> m;
994 for (int j = 0; j < ctx.num_mag_dims(); j++) {
995 m[j] = density__.magnetization(j).f_rg(ir) * (1 + add_delta_mag_xc__);
996 }
997
998 double rho = density__.rho().f_rg(ir);
999 if (add_pseudo_core__) {
1000 rho += density__.rho_pseudo_core().f_rg(ir);
1001 }
