36/// Function in spherical harmonics or spherical coordinates representation.
37/** This class works in conjugation with SHT class which provides the transformation between spherical
38 harmonics and spherical coordinates and also a conversion between real and complex spherical harmonics.
39 */
40template <function_domain_t domain_t, typename T = double_complex>
41class Spheric_function: public mdarray<T, 2>
42{
43 private:
44
45 /// Radial grid.
46 Radial_grid<double> const* radial_grid_{nullptr};
47
48 /// Size of the angular domain.
49 /** This is either a total number of spherical harmonics or a number of (theta, phi) anles covering the sphere. */
50 int angular_domain_size_;
51
52 /* copy constructir is disabled */
53 Spheric_function(Spheric_function<domain_t, T> const& src__) = delete;
54
55 /* copy assignment operator is disabled */
56 Spheric_function<domain_t, T>& operator=(Spheric_function<domain_t, T> const& src__) = delete;
57
58 public:
59
60 /// Constructor of the empty function.
61 Spheric_function()
62 {
63 }
64
65 /// Constructor.
66 Spheric_function(int angular_domain_size__, Radial_grid<double> const& radial_grid__)
67 : mdarray<T, 2>(angular_domain_size__, radial_grid__.num_points())
68 , radial_grid_(&radial_grid__)
69 , angular_domain_size_(angular_domain_size__)
70 {
71 }
72
73 /// Constructor.
74 Spheric_function(T* ptr__, int angular_domain_size__, Radial_grid<double> const& radial_grid__)
75 : mdarray<T, 2>(ptr__, angular_domain_size__, radial_grid__.num_points())
76 , radial_grid_(&radial_grid__)

161 inline int angular_domain_size() const
162 {
163 return angular_domain_size_;

166 inline Radial_grid<double> const& radial_grid() const
167 {
168 assert(radial_grid_ != nullptr);
169 return *radial_grid_;
170 }
171
172 Spline<T> component(int lm__) const
173 {
174 if (domain_t != function_domain_t::spectral) {
175 TERMINATE("function is not is spectral domain");
176 }
177
178 Spline<T> s(radial_grid());
179 for (int ir = 0; ir < radial_grid_->num_points(); ir++) {
180 s(ir) = (*this)(lm__, ir);
181 }
182 s.interpolate();
183 return s;
184 }
185
186 T value(double theta__, double phi__, int jr__, double dr__) const
187 {
188 assert(domain_t == function_domain_t::spectral);
189
190 int lmax = utils::lmax(angular_domain_size_);
191 std::vector<T> ylm(angular_domain_size_);
192 sf::spherical_harmonics(lmax, theta__, phi__, &ylm[0]);
193 T p = 0.0;
194 for (int lm = 0; lm < angular_domain_size_; lm++) {
195 double deriv = ((*this)(lm, jr__ + 1) - (*this)(lm, jr__)) / radial_grid_->dx(jr__);
196 p += ylm[lm] * ((*this)(lm, jr__) + deriv * dr__);
197 }
198 return p;
199 }

241/// Multiplication of two functions in spatial domain.
242/** The result of the operation is a scalar function in spatial domain */
243template <typename T>
244inline Spheric_function<function_domain_t::spatial, T> operator*(Spheric_function<function_domain_t::spatial, T> const& a__,
245 Spheric_function<function_domain_t::spatial, T> const& b__)
246{
247 if (a__.radial_grid().hash() != b__.radial_grid().hash()) {
248 TERMINATE("wrong radial grids");
249 }
250 if (a__.angular_domain_size() != b__.angular_domain_size()) {
251 TERMINATE("wrong angular domain sizes");
252 }
253
254 Spheric_function<function_domain_t::spatial, T> res(a__.angular_domain_size(), a__.radial_grid());
255
256 #pragma omp parallel for schedule(static)
257 for (int ir = 0; ir < res.radial_grid().num_points(); ir++) {
258 for (int tp = 0; tp < res.angular_domain_size(); tp++) {
259 res(tp, ir) = a__(tp, ir) * b__(tp, ir);
260 }
261 }
262
263 return res;
264}

266/// Dot product of two gradiensts of real functions in spatial domain.
267/** The result of the operation is the real scalar function in spatial domain */
268inline Spheric_function<function_domain_t::spatial, double> operator*(Spheric_vector_function<function_domain_t::spatial, double> const& f,
269 Spheric_vector_function<function_domain_t::spatial, double> const& g)
270{
271 if (f.radial_grid().hash() != g.radial_grid().hash()) {
272 TERMINATE("wrong radial grids");
273 }
274
275 for (int x: {0, 1, 2}) {
276 if (f[x].angular_domain_size() != g[x].angular_domain_size()) {
277 TERMINATE("wrong number of angular points");
278 }
279 }
280
281 Spheric_function<function_domain_t::spatial, double> result(f.angular_domain_size(), f.radial_grid());
282 result.zero();
283
284 for (int x: {0, 1, 2}) {
285 #pragma omp parallel for schedule(static)
286 for (int ir = 0; ir < f.radial_grid().num_points(); ir++) {
287 for (int tp = 0; tp < f.angular_domain_size(); tp++) {
288 result(tp, ir) += f[x](tp, ir) * g[x](tp, ir);
289 }
290 }
291 }
292
293 return result;
294}

296/// Summation of two functions.
297template <function_domain_t domain_t, typename T>
298Spheric_function<domain_t, T> operator+(Spheric_function<domain_t, T> const& a__, Spheric_function<domain_t, T> const& b__)
299{
300 if (a__.radial_grid().hash() != b__.radial_grid().hash()) {
301 TERMINATE("wrong radial grids");
302 }
303 if (a__.angular_domain_size() != b__.angular_domain_size()) {
304 TERMINATE("wrong angular domain sizes");
305 }
306
307 Spheric_function<domain_t, T> result(a__.angular_domain_size(), a__.radial_grid());
308
309 #pragma omp parallel for schedule(static)
310 for (int ir = 0; ir < a__.radial_grid().num_points(); ir++) {
311 for (int i = 0; i < a__.angular_domain_size(); i++) {
312 result(i, ir) = a__(i, ir) + b__(i, ir);

342/// Multiply function by a scalar.
343template <function_domain_t domain_t, typename T>
344Spheric_function<domain_t, T> operator*(T a__, Spheric_function<domain_t, T> const& b__)
345{
346 Spheric_function<domain_t, T> res(b__.angular_domain_size(), b__.radial_grid());
347
348 T const* ptr_rhs = &b__(0, 0);
349 T* ptr_res = &res(0, 0);
350
351 #pragma omp parallel for schedule(static)
352 for (size_t i = 0; i < b__.size(); i++) {
353 ptr_res[i] = a__ * ptr_rhs[i];
354 }
355
356 return res;
357}

365/// Inner product of two spherical functions.
366/** The result of the operation is a scalar value. */
367template <function_domain_t domain_t, typename T>
368T inner(Spheric_function<domain_t, T> const& f1, Spheric_function<domain_t, T> const& f2)
369{
370 Spline<T> s(f1.radial_grid());
371
372 if (domain_t == function_domain_t::spectral) {
373 int lmmax = std::min(f1.angular_domain_size(), f2.angular_domain_size());
374 for (int ir = 0; ir < s.num_points(); ir++) {
375 for (int lm = 0; lm < lmmax; lm++) {
376 s(ir) += utils::conj(f1(lm, ir)) * f2(lm, ir);
377 }
378 s(ir) *= std::pow(f1.radial_grid().x(ir), 2);
379 }
380 } else {
381 throw std::runtime_error("not implemented");
382 }
383 return s.interpolate().integrate(0);
384}

386/// Compute Laplacian of the spheric function.
387/** Laplacian in spherical coordinates has the following expression:
388 \f[
389 \Delta = \frac{1}{r^2}\frac{\partial}{\partial r}\Big( r^2 \frac{\partial}{\partial r} \Big) +
390 \frac{1}{r^2}\Delta_{\theta, \phi}
391 \f]
392 */
393template <typename T>
394Spheric_function<function_domain_t::spectral, T>
395laplacian(Spheric_function<function_domain_t::spectral, T> const& f__)
396{
397 Spheric_function<function_domain_t::spectral, T> g;
398 auto& rgrid = f__.radial_grid();
399 int lmmax = f__.angular_domain_size();
400 int lmax = utils::lmax(lmmax);
401 g = Spheric_function<function_domain_t::spectral, T>(lmmax, rgrid);
402
403 Spline<T> s1(rgrid);
404 for (int l = 0; l <= lmax; l++) {
405 int ll = l * (l + 1);
406 for (int m = -l; m <= l; m++) {
407 int lm = utils::lm(l, m);
408 /* get lm component */
409 auto s = f__.component(lm);
410 /* compute 1st derivative */
411 for (int ir = 0; ir < s.num_points(); ir++) {
412 s1(ir) = s.deriv(1, ir);
413 }
414 s1.interpolate();
415
416 for (int ir = 0; ir < s.num_points(); ir++) {
417 g(lm, ir) = 2.0 * s1(ir) * rgrid.x_inv(ir) + s1.deriv(1, ir) -
418 s(ir) * static_cast<double>(ll) / std::pow(rgrid[ir], 2);
419 }
420 }
421 }
422
423 return g;
424}

426/// Convert from Ylm to Rlm representation.
427inline void convert(Spheric_function<function_domain_t::spectral, double_complex> const& f__,
428 Spheric_function<function_domain_t::spectral, double>& g__)
429{
430 int lmax = utils::lmax(f__.angular_domain_size());
431
432 /* cache transformation arrays */
433 std::vector<double_complex> tpp(f__.angular_domain_size());
434 std::vector<double_complex> tpm(f__.angular_domain_size());
435 for (int l = 0; l <= lmax; l++) {
436 for (int m = -l; m <= l; m++) {
437 int lm = utils::lm(l, m);
438 tpp[lm] = SHT::rlm_dot_ylm(l, m, m);
439 tpm[lm] = SHT::rlm_dot_ylm(l, m, -m);
440 }
441 }
442
443 for (int ir = 0; ir < f__.radial_grid().num_points(); ir++) {
444 int lm = 0;
445 for (int l = 0; l <= lmax; l++) {
446 for (int m = -l; m <= l; m++) {
447 if (m == 0) {
448 g__(lm, ir) = std::real(f__(lm, ir));
449 } else {
450 int lm1 = utils::lm(l, -m);
451 g__(lm, ir) = std::real(tpp[lm] * f__(lm, ir) + tpm[lm] * f__(lm1, ir));
452 }
453 lm++;
454 }
455 }
456 }
457}

467/// Convert from Rlm to Ylm representation.
468inline void convert(Spheric_function<function_domain_t::spectral, double> const& f__,
469 Spheric_function<function_domain_t::spectral, double_complex>& g__)
470{
471 int lmax = utils::lmax(f__.angular_domain_size());
472
473 /* cache transformation arrays */
474 std::vector<double_complex> tpp(f__.angular_domain_size());
475 std::vector<double_complex> tpm(f__.angular_domain_size());
476 for (int l = 0; l <= lmax; l++) {
477 for (int m = -l; m <= l; m++) {
478 int lm = utils::lm(l, m);
479 tpp[lm] = SHT::ylm_dot_rlm(l, m, m);
480 tpm[lm] = SHT::ylm_dot_rlm(l, m, -m);
481 }
482 }
483
484 for (int ir = 0; ir < f__.radial_grid().num_points(); ir++) {
485 int lm = 0;
486 for (int l = 0; l <= lmax; l++) {
487 for (int m = -l; m <= l; m++) {
488 if (m == 0) {
489 g__(lm, ir) = f__(lm, ir);
490 } else {
491 int lm1 = utils::lm(l, -m);
492 g__(lm, ir) = tpp[lm] * f__(lm, ir) + tpm[lm] * f__(lm1, ir);
493 }
494 lm++;
495 }
496 }
497 }
498}

541/// Gradient of the function in complex spherical harmonics.
542inline Spheric_vector_function<function_domain_t::spectral, double_complex> gradient(Spheric_function<function_domain_t::spectral, double_complex> const& f)
543{
544 Spheric_vector_function<function_domain_t::spectral, double_complex> g(f.angular_domain_size(), f.radial_grid());
545 for (int i = 0; i < 3; i++) {
546 g[i].zero();
547 }
548
549 int lmax = utils::lmax(f.angular_domain_size());
550
551 for (int l = 0; l <= lmax; l++) {
552 double d1 = std::sqrt(double(l + 1) / double(2 * l + 3));
553 double d2 = std::sqrt(double(l) / double(2 * l - 1));
554
555 for (int m = -l; m <= l; m++) {
556 int lm = utils::lm(l, m);
557 auto s = f.component(lm);
558
559 for (int mu = -1; mu <= 1; mu++) {
560 int j = (mu + 2) % 3; // map -1,0,1 to 1,2,0 (to y,z,x)
561
562 if ((l + 1) <= lmax && std::abs(m + mu) <= l + 1) {
563 int lm1 = utils::lm(l + 1, m + mu);
564 double d = d1 * SHT::clebsch_gordan(l, 1, l + 1, m, mu, m + mu);
565 for (int ir = 0; ir < f.radial_grid().num_points(); ir++) {
566 g[j](lm1, ir) += (s.deriv(1, ir) - f(lm, ir) * f.radial_grid().x_inv(ir) * double(l)) * d;
567 }
568 }
569 if ((l - 1) >= 0 && std::abs(m + mu) <= l - 1) {
570 int lm1 = utils::lm(l - 1, m + mu);
571 double d = d2 * SHT::clebsch_gordan(l, 1, l - 1, m, mu, m + mu);
572 for (int ir = 0; ir < f.radial_grid().num_points(); ir++) {
573 g[j](lm1, ir) -= (s.deriv(1, ir) + f(lm, ir) * f.radial_grid().x_inv(ir) * double(l + 1)) * d;
574 }
575 }
576 }
577 }
578 }
579
580 double_complex d1(1.0 / std::sqrt(2.0), 0);
581 double_complex d2(0, 1.0 / std::sqrt(2.0));
582
583 for (int ir = 0; ir < f.radial_grid().num_points(); ir++) {
584 for (int lm = 0; lm < f.angular_domain_size(); lm++) {
585 double_complex g_p = g[0](lm, ir);
586 double_complex g_m = g[1](lm, ir);
587 g[0](lm, ir) = d1 * (g_m - g_p);
588 g[1](lm, ir) = d2 * (g_m + g_p);
589 }
590 }
591
592 return g;
593}

595/// Gradient of the function in real spherical harmonics.
596inline Spheric_vector_function<function_domain_t::spectral, double> gradient(Spheric_function<function_domain_t::spectral, double> const& f__)
597{
598 int lmax = utils::lmax(f__.angular_domain_size());
599 SHT sht(device_t::CPU, lmax);
600 auto zf = convert(f__);
601 auto zg = gradient(zf);
602 Spheric_vector_function<function_domain_t::spectral, double> g(f__.angular_domain_size(), f__.radial_grid());
603 for (int x: {0, 1, 2}) {
604 g[x] = convert(zg[x]);
605 }
606 return g;
607}
