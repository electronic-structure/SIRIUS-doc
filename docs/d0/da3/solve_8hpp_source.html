25 inline void Band::solve_with_second_variation(K_point& kp__, Potential& potential__) const
26 {
27  /* solve non-magnetic Hamiltonian (so-called first variation) */
28  auto& itso = ctx_.iterative_solver_input();
29  if (itso.type_ == "exact") {
30  diag_fv_exact(&kp__, potential__);
31  } else if (itso.type_ == "davidson") {
32  diag_fv_davidson(&kp__);
33  }
34  /* generate first-variational states */
35  kp__.generate_fv_states();
36  /* solve magnetic Hamiltonian */
37  diag_sv(&kp__, potential__);
38  /* generate spinor wave-functions */
39  kp__.generate_spinor_wave_functions();
40 }

112 inline void Band::solve_for_kset(K_point_set& kset__, Potential& potential__, bool precompute__) const
113 {
114  PROFILE("sirius::Band::solve_for_kset");
115 
116  if (precompute__ && ctx_.full_potential()) {
117  potential__.generate_pw_coefs();
118  potential__.update_atomic_potential();
119  unit_cell_.generate_radial_functions();
120  unit_cell_.generate_radial_integrals();
121  }
122 
123  if (ctx_.full_potential()) {
124  local_op_->prepare(ctx_.gvec_coarse(), ctx_.num_mag_dims(), potential__, ctx_.step_function());
125  } else {
126  local_op_->prepare(ctx_.gvec_coarse(), ctx_.num_mag_dims(), potential__);
127  }
128 
129  if (ctx_.comm().rank() == 0 && ctx_.control().print_memory_usage_) {
130  MEMORY_USAGE_INFO();
131  }
132 
133  int num_dav_iter{0};
134  /* solve secular equation and generate wave functions */
135  for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
136  int ik = kset__.spl_num_kpoints(ikloc);
137  auto kp = kset__[ik];
138 
139  if (ctx_.full_potential() && use_second_variation) {
140  solve_with_second_variation(*kp, potential__);
141  } else {
142  num_dav_iter += solve_with_single_variation(*kp, potential__);
143  }
144  }
145  kset__.comm().allreduce(&num_dav_iter, 1);
146  if (ctx_.comm().rank() == 0 && !ctx_.full_potential()) {
147  printf("Average number of iterations: %12.6f\n", static_cast<double>(num_dav_iter) / kset__.num_kpoints());
148  }
149 
150  local_op_->dismiss();
151 
152  /* synchronize eigen-values */
153  kset__.sync_band_energies();
154 
155  if (ctx_.control().verbosity_ >= 1 && ctx_.comm().rank() == 0) {
156  printf("Lowest band energies\n");
157  for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
158  printf("ik : %2i, ", ik);
159  if (ctx_.num_mag_dims() != 1) {
160  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_bands()); j++) {
161  printf("%12.6f", kset__[ik]->band_energy(j));
162  }
163  } else {
164  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_fv_states()); j++) {
165  printf("%12.6f", kset__[ik]->band_energy(j));
166  }
167  printf("\n ");
168  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_fv_states()); j++) {
169  printf("%12.6f", kset__[ik]->band_energy(ctx_.num_fv_states() + j));
170  }
171  }
172  printf("\n");
173  }
174  }
175 }
