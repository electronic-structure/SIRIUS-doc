25 inline void Band::solve_full_potential(K_point& kp__, Hamiltonian& hamiltonian__) const
26 {
27  if (use_second_variation) {
28  /* solve non-magnetic Hamiltonian (so-called first variation) */
29  auto& itso = ctx_.iterative_solver_input();
30  if (itso.type_ == "exact") {
31  diag_full_potential_first_variation_exact(kp__, hamiltonian__);
32  } else if (itso.type_ == "davidson") {
33  diag_full_potential_first_variation_davidson(kp__, hamiltonian__);
34  }
35  /* generate first-variational states */
36  kp__.generate_fv_states();
37  /* solve magnetic Hamiltonian */
38  diag_full_potential_second_variation(kp__, hamiltonian__);
39  /* generate spinor wave-functions */
40  kp__.generate_spinor_wave_functions();
41  } else {
42  TERMINATE_NOT_IMPLEMENTED
43  //diag_full_potential_single_variation();
44  }
45 }

47 template <typename T>
48 inline int Band::solve_pseudo_potential(K_point& kp__, Hamiltonian& hamiltonian__) const
49 {
50  hamiltonian__.local_op().prepare(kp__.gkvec_partition());
51  ctx_.fft_coarse().prepare(kp__.gkvec_partition());
52 
53  int niter{0};
54 
55  auto& itso = ctx_.iterative_solver_input();
56  if (itso.type_ == "exact") {
57  if (ctx_.num_mag_dims() != 3) {
58  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
59  diag_pseudo_potential_exact<double_complex>(&kp__, ispn, hamiltonian__);
60  }
61  } else {
62  STOP();
63  }
64  } else if (itso.type_ == "davidson") {
65  niter = diag_pseudo_potential_davidson<T>(&kp__, hamiltonian__);
66  } else if (itso.type_ == "rmm-diis") {
67  if (ctx_.num_mag_dims() != 3) {
68  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
69  diag_pseudo_potential_rmm_diis<T>(&kp__, ispn, hamiltonian__);
70  }
71  } else {
72  STOP();
73  }
74  } else if (itso.type_ == "chebyshev") {
75  P_operator<T> p_op(ctx_, kp__.p_mtrx());
76  if (ctx_.num_mag_dims() != 3) {
77  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
78  diag_pseudo_potential_chebyshev<T>(&kp__, ispn, hamiltonian__, p_op);
79  }
80  } else {
81  STOP();
82  }
83  } else {
84  TERMINATE("unknown iterative solver type");
85  }
86 
87  /* check residuals */
88  if (ctx_.control().verification_ >= 1) {
89  check_residuals<T>(&kp__, hamiltonian__);
90  }
91 
92  ctx_.fft_coarse().dismiss();
93  return niter;
94 }

96 inline void Band::solve(K_point_set& kset__, Hamiltonian& hamiltonian__, bool precompute__) const
97 {
98  PROFILE("sirius::Band::solve");
99 
100  if (precompute__ && ctx_.full_potential()) {
101  hamiltonian__.potential().generate_pw_coefs();
102  hamiltonian__.potential().update_atomic_potential();
103  unit_cell_.generate_radial_functions();
104  unit_cell_.generate_radial_integrals();
105  }
106 
107  /* map local potential to a coarse grid */
108  if (ctx_.full_potential()) {
109  hamiltonian__.local_op().prepare(hamiltonian__.potential(), ctx_.step_function());
110  } else {
111  hamiltonian__.local_op().prepare(hamiltonian__.potential());
112  /* prepare non-local operators */
113  if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
114  hamiltonian__.prepare<double>();
115  } else {
116  hamiltonian__.prepare<double_complex>();
117  }
118  }
119 
120  if (ctx_.comm().rank() == 0 && ctx_.control().print_memory_usage_) {
121  MEMORY_USAGE_INFO();
122  }
123 
124  int num_dav_iter{0};
125  /* solve secular equation and generate wave functions */
126  for (int ikloc = 0; ikloc < kset__.spl_num_kpoints().local_size(); ikloc++) {
127  int ik = kset__.spl_num_kpoints(ikloc);
128  auto kp = kset__[ik];
129 
130  if (ctx_.full_potential()) {
131  solve_full_potential(*kp, hamiltonian__);
132  } else {
133  if (ctx_.gamma_point() && (ctx_.so_correction() == false)) {
134  num_dav_iter += solve_pseudo_potential<double>(*kp, hamiltonian__);
135  } else {
136  num_dav_iter += solve_pseudo_potential<double_complex>(*kp, hamiltonian__);
137  }
138  }
139  }
140  kset__.comm().allreduce(&num_dav_iter, 1);
141  if (ctx_.comm().rank() == 0 && !ctx_.full_potential()) {
142  printf("Average number of iterations: %12.6f\n", static_cast<double>(num_dav_iter) / kset__.num_kpoints());
143  }
144 
145  hamiltonian__.local_op().dismiss();
146  if (!ctx_.full_potential()) {
147  hamiltonian__.dismiss();
148  }
149 
150  /* synchronize eigen-values */
151  kset__.sync_band_energies();
152 
153  if (ctx_.control().verbosity_ >= 1 && ctx_.comm().rank() == 0) {
154  printf("Lowest band energies\n");
155  for (int ik = 0; ik < kset__.num_kpoints(); ik++) {
156  printf("ik : %2i, ", ik);
157  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_bands()); j++) {
158  printf("%12.6f", kset__[ik]->band_energy(j, 0));
159  }
160  if (ctx_.num_mag_dims() == 1) {
161  printf("\n ");
162  for (int j = 0; j < std::min(ctx_.control().num_bands_to_print_, ctx_.num_bands()); j++) {
163  printf("%12.6f", kset__[ik]->band_energy(j, 1));
164  }
165  }
166  printf("\n");
167  }
168  }
169 }
