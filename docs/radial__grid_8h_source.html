70  /// Create array of radial grid points.
71  std::vector<double> create_radial_grid_points(radial_grid_t grid_type, int num_points, double rmin, double rmax)
72  {
73  std::vector<double> grid_points(num_points);
74 
75  switch (grid_type) {
76  case linear_grid: {
77  for (int i = 0; i < num_points; i++) {
78  grid_points[i] = rmin + (rmax - rmin) * double(i) / (num_points - 1);
79  }
80  break;
81  }
82  case exponential_grid: {
83  /* x_i = x_min * (x_max/x_min) ^ (i / (N - 1)) */
84  for (int i = 0; i < num_points; i++) {
85  grid_points[i] = rmin * std::pow(rmax / rmin, double(i) / (num_points - 1));
86  }
87  break;
88  }
89  case lin_exp_grid: {
90  /* x_i = x_min + (x_max - x_min) * A(t)
91  * A(t) ~ b * t + Exp[t^a] - 1 */
92  double alpha{6.0};
93  double beta = 1e-6 * num_points / (rmax - rmin);
94  double A = 1.0 / (std::exp(1) + beta - 1);
95 
96  for (int i = 0; i < num_points; i++) {
97  double t = double(i) / (num_points - 1);
98  grid_points[i] = rmin + (rmax - rmin) * (beta * t + std::exp(std::pow(t, alpha)) - 1) * A;
99  }
100  break;
101  }
102  case pow2_grid: {
103  /* x_i = x_min + (x_max - x_min) * (i / N - 1)^2 */
104  for (int i = 0; i < num_points; i++) {
105  grid_points[i] = rmin + (rmax - rmin) * std::pow(double(i) / (num_points - 1), 2);
106  }
107  break;
108  }
109  case pow3_grid: {
110  for (int i = 0; i < num_points; i++) {
111  grid_points[i] = rmin + (rmax - rmin) * std::pow(double(i) / (num_points - 1), 3);
112  }
113  break;
114  }
115  case scaled_pow_grid: {
116  /* ratio of last and first dx */
117  double S = rmax * 1000;
118  double alpha = std::pow(S, 1.0 / (num_points - 2));
119  double x = rmin;
120  for (int i = 0; i < num_points; i++)
121  {
122  grid_points[i] = x;
123  x += (rmax - rmin) * (alpha - 1) * std::pow(S, double(i) / (num_points - 2)) / (S * alpha - 1);
124  }
125  break;
126  }
127  default: {
128  TERMINATE_NOT_IMPLEMENTED
129  }
130 
131  //== case hyperbolic_grid:
132  //== {
133  //== double x = origin;
134  //== int i = 1;
135  //==
136  //== while (x <= infinity + tol)
137  //== {
138  //== grid_points.push_back(x);
139  //== double t = double(i++) / double(num_mt_points - 1);
140  //== x = origin + 2.0 * (mt_radius - origin) * t / (t + 1);
141  //== }
142  //== break;
143  //== }
144  //== case incremental_grid:
145  //== {
146  //== double D = mt_radius - origin;
147  //== double S = 1000.0;
148  //== double dx0 = 2 * D / (2 * (num_mt_points - 1) + (num_mt_points - 1) * (S - 1));
149  //== double alpha = (S - 1) * dx0 / (num_mt_points - 2);
150 
151  //== int i = 0;
152  //== double x = origin;
153  //== while (x <= infinity + tol)
154  //== {
155  //== grid_points.push_back(x);
156  //== x = origin + (dx0 + dx0 + i * alpha) * (i + 1) / 2.0;
157  //== i++;
158  //== }
159  //== break;
160  //== }
161  }
162 
163  /* trivial check */
164  if (std::abs(rmax - grid_points[num_points - 1]) > 1e-10) {
165  std::stringstream s;
166  s << "Wrong radial grid" << std::endl
167  << " num_points : " << num_points << std::endl
168  << " rmax : " << Utils::double_to_string(rmax, 12) << std::endl
169  << " last grid point : " << Utils::double_to_string(grid_points[num_points - 1], 12);
170  TERMINATE(s);
171  }
172 
173  return grid_points;
174  }

327  /// Set new radial points.
328  void set_radial_points(int num_points__, double const* points__)
329  {
330  assert(num_points__ > 0);
331 
332  /* set points */
333  x_ = mdarray<double, 1>(num_points__);
334  std::memcpy(&x_(0), points__, num_points__ * sizeof(double));
335 
336  for (int i = 0; i < num_points__; i++) {

340  /* set x^{-1} */

342  for (int i = 0; i < num_points__; i++) {
343  x_inv_(i) = (x_(i) == 0) ? 0 : 1.0 / x_(i);
