54  void calc_gradient(int calc_component__)
55  {
56  Gvec const& gkvec = bp_->gk_vectors();
57 
58  matrix<double_complex> const& beta_comps = bp_->beta_gk_a();
59 
60  double_complex Im(0, 1);
61 
62  #pragma omp parallel for
63  for (size_t ibf = 0; ibf < bp_->beta_gk_a().size(1); ibf++) {
64  for (int igk_loc = 0; igk_loc < bp_->num_gkvec_loc(); igk_loc++) {
65  int igk = gkvec.gvec_offset(bp_->comm().rank()) + igk_loc;
66 
67  double gkvec_comp = gkvec.gkvec_cart(igk)[calc_component__];
68 
69  components_gk_a_[calc_component__](igk_loc, ibf) = - Im * gkvec_comp * beta_comps(igk_loc,ibf);
70  }
71  }
72  }
