33 class Beta_projectors_gradient: public Beta_projectors_base<3>
34 {
35  private:
36  void generate_pw_coefs_t(Beta_projectors& beta__)
37  {
38  auto& bchunk = ctx_.beta_projector_chunks();
39 
40  if (!bchunk.num_beta_t()) {
41  return;
42  }
43 
44  auto& comm = gkvec_.comm();
45 
46  for (int x = 0; x < 3; x++) {
47  #pragma omp parallel for
48  for (int i = 0; i < bchunk.num_beta_t(); i++) {
49  for (int igkloc = 0; igkloc < gkvec_.gvec_count(comm.rank()); igkloc++) {
50  int igk = gkvec_.gvec_offset(comm.rank()) + igkloc;
51  auto vgc = gkvec_.gkvec_cart(igk);
52  pw_coeffs_t_[x](igkloc, i) = double_complex(0, -vgc[x]) * beta__.pw_coeffs_t(0)(igkloc, i);
53  }
54  }
55  }
56  if (ctx_.processing_unit() == GPU) {
57  for (int x = 0; x < 3; x++) {
58  pw_coeffs_t_[x].copy<memory_t::host, memory_t::device>();
59  }
60  }
61  }
62 
63  public:
64  Beta_projectors_gradient(Simulation_context& ctx__,
65  Gvec const& gkvec__,
66  Beta_projectors& beta__)
67  : Beta_projectors_base<3>(ctx__, gkvec__)
68  {
69  generate_pw_coefs_t(beta__);
70  }
71 };
