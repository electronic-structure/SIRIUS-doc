35 inline void Potential::generate_D_operator_matrix()
36 {
37  PROFILE("sirius::Potential::generate_D_operator_matrix");
38 
39  /* store effective potential and magnetic field in a vector */
40  std::vector<Periodic_function<double>*> veff_vec(ctx_.num_mag_dims() + 1);
41  veff_vec[0] = effective_potential_.get();
42  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
43  veff_vec[1 + j] = effective_magnetic_field_[j];
44  }
45 
46  #ifdef __GPU
47  mdarray<double_complex, 1> veff_tmp(nullptr, ctx_.gvec().count());
48  if (ctx_.processing_unit() == GPU) {
49  veff_tmp.allocate(memory_t::device);
50  }
51  #endif
52 
53  ctx_.augmentation_op(0).prepare(0);
54 
55  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
56  auto& atom_type = unit_cell_.atom_type(iat);
57  int nbf = atom_type.mt_basis_size();
58 
59  /* start copy of Q(G) for the next atom type */
60  #ifdef __GPU
61  if (ctx_.processing_unit() == GPU) {
62  acc::sync_stream(0);
63  if (iat + 1 != unit_cell_.num_atom_types()) {
64  ctx_.augmentation_op(iat + 1).prepare(0);
65  }
66  }
67  #endif
68 
69  /* trivial case */
70  if (!atom_type.pp_desc().augment) {
71  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
72  for (int i = 0; i < atom_type.num_atoms(); i++) {
73  int ia = atom_type.atom_id(i);
74  auto& atom = unit_cell_.atom(ia);
75 
76  for (int xi2 = 0; xi2 < nbf; xi2++) {
77  for (int xi1 = 0; xi1 < nbf; xi1++) {
78  atom.d_mtrx(xi1, xi2, iv) = 0;
79  }
80  }
81  }
82  }
83  continue;
84  }
85  matrix<double> d_tmp(nbf * (nbf + 1) / 2, atom_type.num_atoms());
86  for (int iv = 0; iv < ctx_.num_mag_dims() + 1; iv++) {
87  switch (ctx_.processing_unit()) {
88  case CPU: {
89  matrix<double> veff_a(2 * ctx_.gvec().count(), atom_type.num_atoms());
90 
91  #pragma omp parallel for schedule(static)
92  for (int i = 0; i < atom_type.num_atoms(); i++) {
93  int ia = atom_type.atom_id(i);
94 
95  for (int igloc = 0; igloc < ctx_.gvec().count(); igloc++) {
96  int ig = ctx_.gvec().offset() + igloc;
97  /* conjugate V(G) * exp(i * G * r_{alpha}) */
98  //auto z = std::conj(veff_vec[iv]->f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia));
99  auto z = veff_vec[iv]->f_pw_local(igloc) * ctx_.gvec_phase_factor(ig, ia);
100  veff_a(2 * igloc, i) = z.real();
101  veff_a(2 * igloc + 1, i) = z.imag();
102  }
103  }
104 
105  linalg<CPU>::gemm(0, 0, nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
106  ctx_.augmentation_op(iat).q_pw(), veff_a, d_tmp);
107  break;
108  }
109  case GPU: {
110  #ifdef __GPU
111  /* copy plane wave coefficients of effective potential to GPU */
112  mdarray<double_complex, 1> veff(&veff_vec[iv]->f_pw_local(0), veff_tmp.at<GPU>(),
113  ctx_.gvec().count());
114  veff.copy<memory_t::host, memory_t::device>();
115 
116  matrix<double> veff_a(2 * ctx_.gvec().count(), atom_type.num_atoms(), memory_t::device);
117 
118  d_tmp.allocate(memory_t::device);
119 
120  mul_veff_with_phase_factors_gpu(atom_type.num_atoms(),
121  ctx_.gvec().count(),
122  veff.at<GPU>(),
123  ctx_.gvec_coord().at<GPU>(),
124  ctx_.atom_coord(iat).at<GPU>(),
125  veff_a.at<GPU>(), 1);
126 
127  linalg<GPU>::gemm(0, 0, nbf * (nbf + 1) / 2, atom_type.num_atoms(), 2 * ctx_.gvec().count(),
128  ctx_.augmentation_op(iat).q_pw(), veff_a, d_tmp, 1);
129 
130  d_tmp.copy<memory_t::device, memory_t::host>();
131  #endif
132  break;
133  }
134  }
135 
136  if (ctx_.gvec().reduced()) {
137  if (comm_.rank() == 0) {
138  for (int i = 0; i < atom_type.num_atoms(); i++) {
139  for (int j = 0; j < nbf * (nbf + 1) / 2; j++) {
140  d_tmp(j, i) = 2 * d_tmp(j, i) - veff_vec[iv]->f_pw_local(0).real() * ctx_.augmentation_op(iat).q_pw(j, 0);
141  }
142  }
143  } else {
144  for (int i = 0; i < atom_type.num_atoms(); i++) {
145  for (int j = 0; j < nbf * (nbf + 1) / 2; j++) {
146  d_tmp(j, i) *= 2;
147  }
148  }
149  }
150  }
151 
152  comm_.allreduce(d_tmp.at<CPU>(), static_cast<int>(d_tmp.size()));
153 
154  #ifdef __PRINT_OBJECT_CHECKSUM
155  {
156  auto cs = d_tmp.checksum();
157  DUMP("checksum(d_mtrx): %18.10f", cs);
158  }
159  #endif
160 
161  #pragma omp parallel for schedule(static)
162  for (int i = 0; i < atom_type.num_atoms(); i++) {
163  int ia = atom_type.atom_id(i);
164  auto& atom = unit_cell_.atom(ia);
165 
166  for (int xi2 = 0; xi2 < nbf; xi2++) {
167  for (int xi1 = 0; xi1 <= xi2; xi1++) {
168  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
169  /* D-matix is symmetric */
170  atom.d_mtrx(xi1, xi2, iv) = atom.d_mtrx(xi2, xi1, iv) = d_tmp(idx12, i) * unit_cell_.omega();
171  }
172  }
173  }
174  }
175  ctx_.augmentation_op(iat).dismiss();
176  }
177 
178  /* add d_ion to the effective potential component of D-operator */
179  #pragma omp parallel for schedule(static)
180  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
181  auto& atom_type = unit_cell_.atom(ia).type();
182  int nbf = unit_cell_.atom(ia).mt_basis_size();
183 
184  for (int xi2 = 0; xi2 < nbf; xi2++) {
185  int lm2 = atom_type.indexb(xi2).lm;
186  int idxrf2 = atom_type.indexb(xi2).idxrf;
187  for (int xi1 = 0; xi1 < nbf; xi1++) {
188  int lm1 = atom_type.indexb(xi1).lm;
189  int idxrf1 = atom_type.indexb(xi1).idxrf;
190 
191  if (lm1 == lm2) {
192  unit_cell_.atom(ia).d_mtrx(xi1, xi2, 0) += atom_type.pp_desc().d_mtrx_ion(idxrf1, idxrf2);
193  }
194  }
195  }
196  }
197 }
