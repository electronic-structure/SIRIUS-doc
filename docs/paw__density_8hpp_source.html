8 inline void Density::init_paw()

12  if (!unit_cell_.num_paw_atoms()) {

17  int ia_paw = unit_cell_.spl_num_paw_atoms(i);
18  int ia = unit_cell_.paw_atom_index(ia_paw);

20  auto& atom_type = atom.type();
21 
22  int num_mt_points = atom_type.num_mt_points();
23 
24  int l_max = 2 * atom_type.indexr().lmax_lo();
25  int lm_max_rho = Utils::lmmax(l_max);
26

29  pdd.atom_ = &atom;
30 
31  pdd.ia = ia;
32 
33  // allocate density arrays
34  pdd.ae_density_ = mdarray<double, 2>(lm_max_rho, num_mt_points);
35  pdd.ps_density_ = mdarray<double, 2>(lm_max_rho, num_mt_points);
36 
37  pdd.ae_density_.zero();
38  pdd.ps_density_.zero();
39 
40  // magnetization arrays
41  pdd.ae_magnetization_ = mdarray<double, 3>(lm_max_rho, num_mt_points, 3);
42  pdd.ps_magnetization_ = mdarray<double, 3>(lm_max_rho, num_mt_points, 3);
43 
44  pdd.ae_magnetization_.zero();
45  pdd.ps_magnetization_.zero();
46 
47  paw_density_data_.push_back(std::move(pdd));

51 inline void Density::init_density_matrix_for_paw()
52 {
53  density_matrix_.zero();
54 
55  for (int ipaw = 0; ipaw < unit_cell_.num_paw_atoms(); ipaw++ ) {
56  int ia = unit_cell_.paw_atom_index(ipaw);
57 
58  auto& atom = unit_cell_.atom(ia);
59  auto& atom_type = atom.type();
60 
61  int nbf = atom_type.mt_basis_size();
62 
63  auto& occupations = atom_type.pp_desc().occupations;
64 
65  // magnetization vector
66  vector3d<double> magn = atom.vector_field();
67 
68  double norm = magn.length();
69 
70  for (int xi = 0; xi < nbf; xi++)
71  {
72  basis_function_index_descriptor const& basis_func_index_dsc = atom_type.indexb()[xi];
73 
74  int rad_func_index = basis_func_index_dsc.idxrf;
75 
76  double occ = occupations[rad_func_index];
77 
78  int l = basis_func_index_dsc.l;
79 
80  switch (ctx_.num_mag_dims())
81  {
82  case 0:
83  {
84  density_matrix_(xi,xi,0,ia) = occ / (double)( 2 * l + 1 );
85  break;
86  }
87 
88  case 1:
89  {
90  double nm = (norm < 1. ) ? magn[0] : 1.;
91 
92  density_matrix_(xi,xi,0,ia) = 0.5 * (1.0 + nm ) * occ / (double)( 2 * l + 1 );
93  density_matrix_(xi,xi,1,ia) = 0.5 * (1.0 - nm ) * occ / (double)( 2 * l + 1 );
94  break;
95  }
96  }
97 
98  }
99  }

102 inline void Density::generate_paw_atom_density(paw_density_data_t &pdd)
103 {
104  int ia = pdd.ia;
105 
106  auto& atom_type = pdd.atom_->type();
107 
108  auto& pp_desc = atom_type.pp_desc();
109 
110  auto l_by_lm = Utils::l_by_lm(2 * atom_type.indexr().lmax_lo());
111 
112  //TODO calculate not for every atom but for every atom type
113  Gaunt_coefficients<double> GC(atom_type.indexr().lmax_lo(),
114  2 * atom_type.indexr().lmax_lo(),
115  atom_type.indexr().lmax_lo(),
116  SHT::gaunt_rlm);
117 
118  // get density for current atom
119  pdd.ae_density_.zero(); //ae_atom_density
120  pdd.ps_density_.zero(); //ps_atom_density
121 
122  // and magnetization
123  pdd.ae_magnetization_.zero(); //ae_atom_magnetization
124  pdd.ps_magnetization_.zero(); //ps_atom_magnetization
125 
126  // get radial grid to divide density over r^2
127  auto &grid = atom_type.radial_grid();
128 
129  // iterate over local basis functions (or over lm1 and lm2)
130  for (int ib2 = 0; ib2 < atom_type.indexb().size(); ib2++)
131  {
132  for(int ib1 = 0; ib1 <= ib2; ib1++)
133  {
134  // get lm quantum numbers (lm index) of the basis functions
135  int lm2 = atom_type.indexb(ib2).lm;
136  int lm1 = atom_type.indexb(ib1).lm;
137 
138  //get radial basis functions indices
139  int irb2 = atom_type.indexb(ib2).idxrf;
140  int irb1 = atom_type.indexb(ib1).idxrf;
141 
142  // index to iterate Qij,
143  int iqij = irb2 * (irb2 + 1) / 2 + irb1;
144 
145  // get num of non-zero GC
146  int num_non_zero_gk = GC.num_gaunt(lm1,lm2);
147 
148  double diag_coef = ib1 == ib2 ? 1. : 2. ;
149 
150  // add nonzero coefficients
151  for(int inz = 0; inz < num_non_zero_gk; inz++)
152  {
153  auto& lm3coef = GC.gaunt(lm1,lm2,inz);
154 
155  // iterate over radial points
156  // this part in fortran looks better, is there the same for c++?
157  for(int irad = 0; irad < (int)grid.num_points(); irad++)
158  {
159  // we need to divide density over r^2 since wave functions are stored multiplied by r
160  double inv_r2 = diag_coef /(grid[irad] * grid[irad]);
161 
162  // TODO for 3 spin dimensions 3th density spin component must be complex
163  // replace order of indices for density from {irad,lm} to {lm,irad}
164  // to be in according with ELK and other SIRIUS code
165  double ae_part = inv_r2 * lm3coef.coef * pp_desc.all_elec_wfc(irad,irb1) * pp_desc.all_elec_wfc(irad,irb2);
166  double ps_part = inv_r2 * lm3coef.coef *
167  (pp_desc.pseudo_wfc(irad,irb1) * pp_desc.pseudo_wfc(irad,irb2) + pp_desc.q_radial_functions_l(irad,iqij,l_by_lm[lm3coef.lm3]));
168 
169  // calculate UP density (or total in case of nonmagnetic)
170  double ae_dens_u = density_matrix_(ib1, ib2, 0, ia).real() * ae_part;
171  double ps_dens_u = density_matrix_(ib1, ib2, 0, ia).real() * ps_part;
172 
173  // add density UP to the total density
174  pdd.ae_density_(lm3coef.lm3, irad) += ae_dens_u;
175  pdd.ps_density_(lm3coef.lm3, irad) += ps_dens_u;
176 
177  switch(ctx_.num_spins()) {
178  case 2: {
179  double ae_dens_d = density_matrix_(ib1, ib2, 1, ia).real() * ae_part;
180  double ps_dens_d = density_matrix_(ib1, ib2, 1, ia).real() * ps_part;
181 
182  // add density DOWN to the total density
183  pdd.ae_density_(lm3coef.lm3, irad) += ae_dens_d;
184  pdd.ps_density_(lm3coef.lm3, irad) += ps_dens_d;
185 
186  // add magnetization to 2nd components (0th and 1st are always zero )
187  pdd.ae_magnetization_(lm3coef.lm3, irad, 0) = ae_dens_u - ae_dens_d;
188  pdd.ps_magnetization_(lm3coef.lm3, irad, 0) = ps_dens_u - ps_dens_d;
189  break;
190  }
191  case 3: {
192  TERMINATE("PAW: non collinear is not implemented");
193  }
194  default: {
195  break;
196  }
197  }
198  }
199  }
200  }
201  }
