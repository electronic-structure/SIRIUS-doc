25 inline void Potential::xc_mt_nonmagnetic(Radial_grid<double> const& rgrid,
26  std::vector<XC_functional>& xc_func,
27  Spheric_function<spectral, double> const& rho_lm,
28  Spheric_function<spatial, double>& rho_tp,
29  Spheric_function<spatial, double>& vxc_tp,
30  Spheric_function<spatial, double>& exc_tp)
31 {
32  PROFILE("sirius::Potential::xc_mt_nonmagnetic");
33 
34  bool is_gga = is_gradient_correction();
35 
36  Spheric_function_gradient<spatial, double> grad_rho_tp(sht_->num_points(), rgrid);
37  Spheric_function<spatial, double> lapl_rho_tp;
38  Spheric_function<spatial, double> grad_rho_grad_rho_tp;
39 
40  if (is_gga) {
41  /* compute gradient in Rlm spherical harmonics */
42  auto grad_rho_lm = gradient(rho_lm);
43 
44  /* backward transform gradient from Rlm to (theta, phi) */
45  for (int x = 0; x < 3; x++) {
46  grad_rho_tp[x] = transform(sht_.get(), grad_rho_lm[x]);
47  }
48 
49  /* compute density gradient product */
50  grad_rho_grad_rho_tp = grad_rho_tp * grad_rho_tp;
51 
52  /* compute Laplacian in Rlm spherical harmonics */
53  auto lapl_rho_lm = laplacian(rho_lm);
54 
55  /* backward transform Laplacian from Rlm to (theta, phi) */
56  lapl_rho_tp = transform(sht_.get(), lapl_rho_lm);
57  }
58 
59  exc_tp.zero();
60  vxc_tp.zero();
61 
62  Spheric_function<spatial, double> vsigma_tp;
63  if (is_gga) {
64  vsigma_tp = Spheric_function<spatial, double>(sht_->num_points(), rgrid);
65  vsigma_tp.zero();
66  }
67 
68  /* loop over XC functionals */
69  for (auto& ixc: xc_func) {
70  /* if this is an LDA functional */
71  if (ixc.is_lda()) {
72  #pragma omp parallel
73  {
74  std::vector<double> exc_t(sht_->num_points());
75  std::vector<double> vxc_t(sht_->num_points());
76  #pragma omp for
77  for (int ir = 0; ir < rgrid.num_points(); ir++) {
78  ixc.get_lda(sht_->num_points(), &rho_tp(0, ir), &vxc_t[0], &exc_t[0]);
79  for (int itp = 0; itp < sht_->num_points(); itp++) {
80  /* add Exc contribution */
81  exc_tp(itp, ir) += exc_t[itp];
82 
83  /* directly add to Vxc */
84  vxc_tp(itp, ir) += vxc_t[itp];
85  }
86  }
87  }
88  }
89  if (ixc.is_gga()) {
90  #pragma omp parallel
91  {
92  std::vector<double> exc_t(sht_->num_points());
93  std::vector<double> vrho_t(sht_->num_points());
94  std::vector<double> vsigma_t(sht_->num_points());
95  #pragma omp for
96  for (int ir = 0; ir < rgrid.num_points(); ir++)
97  {
98  ixc.get_gga(sht_->num_points(), &rho_tp(0, ir), &grad_rho_grad_rho_tp(0, ir), &vrho_t[0], &vsigma_t[0], &exc_t[0]);
99  for (int itp = 0; itp < sht_->num_points(); itp++)
100  {
101  /* add Exc contribution */
102  exc_tp(itp, ir) += exc_t[itp];
103 
104  /* directly add to Vxc available contributions */
105  vxc_tp(itp, ir) += (vrho_t[itp] - 2 * vsigma_t[itp] * lapl_rho_tp(itp, ir));
106 
107  /* save the sigma derivative */
108  vsigma_tp(itp, ir) += vsigma_t[itp];
109  }
110  }
111  }
112  }
113  }
114 
115  if (is_gga)
116  {
117  /* forward transform vsigma to Rlm */
118  auto vsigma_lm = transform(sht_.get(), vsigma_tp);
119 
120  /* compute gradient of vsgima in spherical harmonics */
121  auto grad_vsigma_lm = gradient(vsigma_lm);
122 
123  /* backward transform gradient from Rlm to (theta, phi) */
124  Spheric_function_gradient<spatial, double> grad_vsigma_tp(sht_->num_points(), rgrid);
125  for (int x = 0; x < 3; x++) {
126  grad_vsigma_tp[x] = transform(sht_.get(), grad_vsigma_lm[x]);
127  }
128 
129  /* compute scalar product of two gradients */
130  auto grad_vsigma_grad_rho_tp = grad_vsigma_tp * grad_rho_tp;
131 
132  /* add remaining term to Vxc */
133  for (int ir = 0; ir < rgrid.num_points(); ir++)
134  {
135  for (int itp = 0; itp < sht_->num_points(); itp++)
136  {
137  vxc_tp(itp, ir) -= 2 * grad_vsigma_grad_rho_tp(itp, ir);
138  }
139  }
140  }
141 }

143 inline void Potential::xc_mt_magnetic(Radial_grid<double> const& rgrid,
144  std::vector<XC_functional>& xc_func,
145  Spheric_function<spectral, double>& rho_up_lm,
146  Spheric_function<spatial, double>& rho_up_tp,
147  Spheric_function<spectral, double>& rho_dn_lm,
148  Spheric_function<spatial, double>& rho_dn_tp,
149  Spheric_function<spatial, double>& vxc_up_tp,
150  Spheric_function<spatial, double>& vxc_dn_tp,
151  Spheric_function<spatial, double>& exc_tp)
152 {
153  PROFILE("sirius::Potential::xc_mt_magnetic");
154 
155  bool is_gga = is_gradient_correction();
156 
157  Spheric_function_gradient<spatial, double> grad_rho_up_tp(sht_->num_points(), rgrid);
158  Spheric_function_gradient<spatial, double> grad_rho_dn_tp(sht_->num_points(), rgrid);
159 
160  Spheric_function<spatial, double> lapl_rho_up_tp(sht_->num_points(), rgrid);
161  Spheric_function<spatial, double> lapl_rho_dn_tp(sht_->num_points(), rgrid);
162 
163  Spheric_function<spatial, double> grad_rho_up_grad_rho_up_tp;
164  Spheric_function<spatial, double> grad_rho_dn_grad_rho_dn_tp;
165  Spheric_function<spatial, double> grad_rho_up_grad_rho_dn_tp;
166 
167  //assert(rho_up_lm.radial_grid().hash() == rho_dn_lm.radial_grid().hash());
168 
169  vxc_up_tp.zero();
170  vxc_dn_tp.zero();
171  exc_tp.zero();
172 
173  if (is_gga)
174  {
175  /* compute gradient in Rlm spherical harmonics */
176  auto grad_rho_up_lm = gradient(rho_up_lm);
177  auto grad_rho_dn_lm = gradient(rho_dn_lm);
178 
179  /* backward transform gradient from Rlm to (theta, phi) */
180  for (int x = 0; x < 3; x++)
181  {
182  grad_rho_up_tp[x] = transform(sht_.get(), grad_rho_up_lm[x]);
183  grad_rho_dn_tp[x] = transform(sht_.get(), grad_rho_dn_lm[x]);
184  }
185 
186  /* compute density gradient products */
187  grad_rho_up_grad_rho_up_tp = grad_rho_up_tp * grad_rho_up_tp;
188  grad_rho_up_grad_rho_dn_tp = grad_rho_up_tp * grad_rho_dn_tp;
189  grad_rho_dn_grad_rho_dn_tp = grad_rho_dn_tp * grad_rho_dn_tp;
190 
191  /* compute Laplacians in Rlm spherical harmonics */
192  auto lapl_rho_up_lm = laplacian(rho_up_lm);
193  auto lapl_rho_dn_lm = laplacian(rho_dn_lm);
194 
195  /* backward transform Laplacians from Rlm to (theta, phi) */
196  lapl_rho_up_tp = transform(sht_.get(), lapl_rho_up_lm);
197  lapl_rho_dn_tp = transform(sht_.get(), lapl_rho_dn_lm);
198  }
199 
200  Spheric_function<spatial, double> vsigma_uu_tp;
201  Spheric_function<spatial, double> vsigma_ud_tp;
202  Spheric_function<spatial, double> vsigma_dd_tp;
203  if (is_gga)
204  {
205  vsigma_uu_tp = Spheric_function<spatial, double>(sht_->num_points(), rgrid);
206  vsigma_uu_tp.zero();
207 
208  vsigma_ud_tp = Spheric_function<spatial, double>(sht_->num_points(), rgrid);
209  vsigma_ud_tp.zero();
210 
211  vsigma_dd_tp = Spheric_function<spatial, double>(sht_->num_points(), rgrid);
212  vsigma_dd_tp.zero();
213  }
214 
215  /* loop over XC functionals */
216  for (auto& ixc: xc_func)
217  {
218  /* if this is an LDA functional */
219  if (ixc.is_lda()) {
220  #pragma omp parallel
221  {
222  std::vector<double> exc_t(sht_->num_points());
223  std::vector<double> vxc_up_t(sht_->num_points());
224  std::vector<double> vxc_dn_t(sht_->num_points());
225  #pragma omp for
226  for (int ir = 0; ir < rgrid.num_points(); ir++) {
227  ixc.get_lda(sht_->num_points(), &rho_up_tp(0, ir), &rho_dn_tp(0, ir), &vxc_up_t[0], &vxc_dn_t[0], &exc_t[0]);
228  for (int itp = 0; itp < sht_->num_points(); itp++) {
229  /* add Exc contribution */
230  exc_tp(itp, ir) += exc_t[itp];
231 
232  /* directly add to Vxc */
233  vxc_up_tp(itp, ir) += vxc_up_t[itp];
234  vxc_dn_tp(itp, ir) += vxc_dn_t[itp];
235  }
236  }
237  }
238  }
239  if (ixc.is_gga()) {
240  #pragma omp parallel
241  {
242  std::vector<double> exc_t(sht_->num_points());
243  std::vector<double> vrho_up_t(sht_->num_points());
244  std::vector<double> vrho_dn_t(sht_->num_points());
245  std::vector<double> vsigma_uu_t(sht_->num_points());
246  std::vector<double> vsigma_ud_t(sht_->num_points());
247  std::vector<double> vsigma_dd_t(sht_->num_points());
248  #pragma omp for
249  for (int ir = 0; ir < rgrid.num_points(); ir++) {
250  ixc.get_gga(sht_->num_points(),
251  &rho_up_tp(0, ir),
252  &rho_dn_tp(0, ir),
253  &grad_rho_up_grad_rho_up_tp(0, ir),
254  &grad_rho_up_grad_rho_dn_tp(0, ir),
255  &grad_rho_dn_grad_rho_dn_tp(0, ir),
256  &vrho_up_t[0],
257  &vrho_dn_t[0],
258  &vsigma_uu_t[0],
259  &vsigma_ud_t[0],
260  &vsigma_dd_t[0],
261  &exc_t[0]);
262 
263  for (int itp = 0; itp < sht_->num_points(); itp++) {
264  /* add Exc contribution */
265  exc_tp(itp, ir) += exc_t[itp];
266 
267  /* directly add to Vxc available contributions */
268  vxc_up_tp(itp, ir) += (vrho_up_t[itp] - 2 * vsigma_uu_t[itp] * lapl_rho_up_tp(itp, ir) - vsigma_ud_t[itp] * lapl_rho_dn_tp(itp, ir));
269  vxc_dn_tp(itp, ir) += (vrho_dn_t[itp] - 2 * vsigma_dd_t[itp] * lapl_rho_dn_tp(itp, ir) - vsigma_ud_t[itp] * lapl_rho_up_tp(itp, ir));
270 
271  /* save the sigma derivatives */
272  vsigma_uu_tp(itp, ir) += vsigma_uu_t[itp];
273  vsigma_ud_tp(itp, ir) += vsigma_ud_t[itp];
274  vsigma_dd_tp(itp, ir) += vsigma_dd_t[itp];
275  }
276  }
277  }
278  }
279  }
280 
281  if (is_gga) {
282  /* forward transform vsigma to Rlm */
283  auto vsigma_uu_lm = transform(sht_.get(), vsigma_uu_tp);
284  auto vsigma_ud_lm = transform(sht_.get(), vsigma_ud_tp);
285  auto vsigma_dd_lm = transform(sht_.get(), vsigma_dd_tp);
286 
287  /* compute gradient of vsgima in spherical harmonics */
288  auto grad_vsigma_uu_lm = gradient(vsigma_uu_lm);
289  auto grad_vsigma_ud_lm = gradient(vsigma_ud_lm);
290  auto grad_vsigma_dd_lm = gradient(vsigma_dd_lm);
291 
292  /* backward transform gradient from Rlm to (theta, phi) */
293  Spheric_function_gradient<spatial, double> grad_vsigma_uu_tp(sht_->num_points(), rgrid);
294  Spheric_function_gradient<spatial, double> grad_vsigma_ud_tp(sht_->num_points(), rgrid);
295  Spheric_function_gradient<spatial, double> grad_vsigma_dd_tp(sht_->num_points(), rgrid);
296  for (int x = 0; x < 3; x++)
297  {
298  grad_vsigma_uu_tp[x] = transform(sht_.get(), grad_vsigma_uu_lm[x]);
299  grad_vsigma_ud_tp[x] = transform(sht_.get(), grad_vsigma_ud_lm[x]);
300  grad_vsigma_dd_tp[x] = transform(sht_.get(), grad_vsigma_dd_lm[x]);
301  }
302 
303  /* compute scalar product of two gradients */
304  auto grad_vsigma_uu_grad_rho_up_tp = grad_vsigma_uu_tp * grad_rho_up_tp;
305  auto grad_vsigma_dd_grad_rho_dn_tp = grad_vsigma_dd_tp * grad_rho_dn_tp;
306  auto grad_vsigma_ud_grad_rho_up_tp = grad_vsigma_ud_tp * grad_rho_up_tp;
307  auto grad_vsigma_ud_grad_rho_dn_tp = grad_vsigma_ud_tp * grad_rho_dn_tp;
308 
309  /* add remaining terms to Vxc */
310  for (int ir = 0; ir < rgrid.num_points(); ir++)
311  {
312  for (int itp = 0; itp < sht_->num_points(); itp++)
313  {
314  vxc_up_tp(itp, ir) -= (2 * grad_vsigma_uu_grad_rho_up_tp(itp, ir) + grad_vsigma_ud_grad_rho_dn_tp(itp, ir));
315  vxc_dn_tp(itp, ir) -= (2 * grad_vsigma_dd_grad_rho_dn_tp(itp, ir) + grad_vsigma_ud_grad_rho_up_tp(itp, ir));
316  }
317  }
318  }

321 inline void Potential::xc_mt(Density const& density__)
322 {
323  PROFILE("sirius::Potential::xc_mt");
324 
325  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
326  int ia = unit_cell_.spl_num_atoms(ialoc);
327  auto& rgrid = unit_cell_.atom(ia).radial_grid();
328  int nmtp = unit_cell_.atom(ia).num_mt_points();
329 
330  /* backward transform density from Rlm to (theta, phi) */
331  auto rho_tp = transform(sht_.get(), density__.rho().f_mt(ialoc));
332 
333  /* backward transform magnetization from Rlm to (theta, phi) */
334  std::vector< Spheric_function<spatial, double> > vecmagtp(ctx_.num_mag_dims());
335  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
336  vecmagtp[j] = transform(sht_.get(), density__.magnetization(j).f_mt(ialoc));
337  }
338 
339  /* "up" component of the density */
340  Spheric_function<spectral, double> rho_up_lm;
341  Spheric_function<spatial, double> rho_up_tp(sht_->num_points(), rgrid);
342 
343  /* "dn" component of the density */
344  Spheric_function<spectral, double> rho_dn_lm;
345  Spheric_function<spatial, double> rho_dn_tp(sht_->num_points(), rgrid);
346 
347  /* check if density has negative values */
348  double rhomin = 0.0;
349  for (int ir = 0; ir < nmtp; ir++) {
350  for (int itp = 0; itp < sht_->num_points(); itp++) {
351  rhomin = std::min(rhomin, rho_tp(itp, ir));
352  }
353  }
354 
355  if (rhomin < 0.0) {
356  std::stringstream s;
357  s << "Charge density for atom " << ia << " has negative values" << std::endl
358  << "most negatve value : " << rhomin << std::endl
359  << "current Rlm expansion of the charge density may be not sufficient, try to increase lmax_rho";
360  WARNING(s);
361  }
362 
363  if (ctx_.num_spins() == 1) {
364  for (int ir = 0; ir < nmtp; ir++) {
365  /* fix negative density */
366  for (int itp = 0; itp < sht_->num_points(); itp++) {
367  if (rho_tp(itp, ir) < 0.0) {
368  rho_tp(itp, ir) = 0.0;
369  }
370  }
371  }
372  } else {
373  for (int ir = 0; ir < nmtp; ir++) {
374  for (int itp = 0; itp < sht_->num_points(); itp++) {
375  /* compute magnitude of the magnetization vector */
376  double mag = 0.0;
377  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
378  mag += pow(vecmagtp[j](itp, ir), 2);
379  }
380  mag = std::sqrt(mag);
381 
382  /* in magnetic case fix both density and magnetization */
383  for (int itp = 0; itp < sht_->num_points(); itp++) {
384  if (rho_tp(itp, ir) < 0.0) {
385  rho_tp(itp, ir) = 0.0;
386  mag = 0.0;
387  }
388  /* fix numerical noise at high values of magnetization */
389  mag = std::min(mag, rho_tp(itp, ir));
390 
391  /* compute "up" and "dn" components */
392  rho_up_tp(itp, ir) = 0.5 * (rho_tp(itp, ir) + mag);
393  rho_dn_tp(itp, ir) = 0.5 * (rho_tp(itp, ir) - mag);
394  }
395  }
396  }
397 
398  /* transform from (theta, phi) to Rlm */
399  rho_up_lm = transform(sht_.get(), rho_up_tp);
400  rho_dn_lm = transform(sht_.get(), rho_dn_tp);
401  }
402 
403  Spheric_function<spatial, double> exc_tp(sht_->num_points(), rgrid);
404  Spheric_function<spatial, double> vxc_tp(sht_->num_points(), rgrid);
405 
406  if (ctx_.num_spins() == 1) {
407  xc_mt_nonmagnetic(rgrid, xc_func_, density__.rho().f_mt(ialoc), rho_tp, vxc_tp, exc_tp);
408  } else {
409  Spheric_function<spatial, double> vxc_up_tp(sht_->num_points(), rgrid);
410  Spheric_function<spatial, double> vxc_dn_tp(sht_->num_points(), rgrid);
411 
412  xc_mt_magnetic(rgrid, xc_func_, rho_up_lm, rho_up_tp, rho_dn_lm, rho_dn_tp, vxc_up_tp, vxc_dn_tp, exc_tp);
413 
414  for (int ir = 0; ir < nmtp; ir++) {
415  for (int itp = 0; itp < sht_->num_points(); itp++) {
416  /* align magnetic filed parallel to magnetization */
417  /* use vecmagtp as temporary vector */
418  double mag = rho_up_tp(itp, ir) - rho_dn_tp(itp, ir);
419  if (mag > 1e-8) {
420  /* |Bxc| = 0.5 * (V_up - V_dn) */
421  double b = 0.5 * (vxc_up_tp(itp, ir) - vxc_dn_tp(itp, ir));
422  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
423  vecmagtp[j](itp, ir) = b * vecmagtp[j](itp, ir) / mag;
424  }
425  } else {
426  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
427  vecmagtp[j](itp, ir) = 0.0;
428  }
429  }
430  /* Vxc = 0.5 * (V_up + V_dn) */
431  vxc_tp(itp, ir) = 0.5 * (vxc_up_tp(itp, ir) + vxc_dn_tp(itp, ir));
432  }
433  }
434  /* z, x, y order */
435  std::array<int, 3> comp_map = {2, 0, 1};
436  /* convert magnetic field back to Rlm */
437  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
438  auto bxcrlm = transform(sht_.get(), vecmagtp[j]);
439  for (int ir = 0; ir < nmtp; ir++) {
440  /* add auxiliary magnetic field antiparallel to starting magnetization */
441  bxcrlm(0, ir) -= aux_bf_(j, ia) * ctx_.unit_cell().atom(ia).vector_field()[comp_map[j]];
442  for (int lm = 0; lm < ctx_.lmmax_pot(); lm++) {
443  effective_magnetic_field(j).f_mt<index_domain_t::local>(lm, ir, ialoc) = bxcrlm(lm, ir);
444  }
445  }
446  }
447  }
448 
449  /* forward transform from (theta, phi) to Rlm */
450  auto vxcrlm = transform(sht_.get(), vxc_tp);
451  auto excrlm = transform(sht_.get(), exc_tp);
452  for (int ir = 0; ir < nmtp; ir++) {
453  for (int lm = 0; lm < ctx_.lmmax_pot(); lm++) {
454  xc_potential_->f_mt<index_domain_t::local>(lm, ir, ialoc) = vxcrlm(lm, ir);
455  xc_energy_density_->f_mt<index_domain_t::local>(lm, ir, ialoc) = excrlm(lm, ir);
456  }
457  }
458  }

461 template <bool add_pseudo_core__>
462 inline void Potential::xc_rg_nonmagnetic(Density const& density__)
463 {
464  PROFILE("sirius::Potential::xc_rg_nonmagnetic");
465 
466  bool is_gga = is_gradient_correction();
467 
468  int num_points = ctx_.fft().local_size();
469 
470  /* we can use this comm for parallelization */
471  //auto& comm = ctx_.gvec().comm_ortho_fft();
472 
473  Smooth_periodic_function<double> rho(ctx_.fft(), ctx_.gvec_partition());
474 
475  /* split real-space points between available ranks */
476  //splindex<block> spl_np(num_points, comm.size(), comm.rank());
477 
478  /* check for negative values */
479  double rhomin{0};
480  for (int ir = 0; ir < num_points; ir++) {
481 
482  //int ir = spl_np[irloc];
483  double d = density__.rho().f_rg(ir);
484  if (add_pseudo_core__) {
485  d += density__.rho_pseudo_core().f_rg(ir);
486  }
487 
488  rhomin = std::min(rhomin, d);
489  rho.f_rg(ir) = std::max(d, 0.0);
490  }
491  if (rhomin < 0.0) {
492  std::stringstream s;
493  s << "Interstitial charge density has negative values" << std::endl
494  << "most negatve value : " << rhomin;
495  WARNING(s);
496  }
497 
498  if (ctx_.control().print_hash_) {
499  auto h = rho.hash_f_rg();
