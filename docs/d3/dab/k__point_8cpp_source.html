34 {
35  PROFILE("sirius::K_point::initialize");
36 
37  zil_.resize(ctx_.lmax_apw() + 1);
38  for (int l = 0; l <= ctx_.lmax_apw(); l++) {
39  zil_[l] = std::pow(double_complex(0, 1), l);
40  }
41 
42  l_by_lm_ = utils::l_by_lm(ctx_.lmax_apw());
43 
44  int bs = ctx_.cyclic_block_size();
45 
46  if (ctx_.control().use_second_variation_ && ctx_.full_potential()) {
47  assert(ctx_.num_fv_states() > 0);
48  fv_eigen_values_.resize(ctx_.num_fv_states());
49  }
50 
51  /* In case of collinear magnetism we store only non-zero spinor components.
52  *
53  * non magnetic case:
54  * .---.
55  * | |
56  * .---.
57  *
58  * collinear case:
59  * .---. .---.
60  * |uu | 0 |uu |
61  * .---.---. -> .---.
62  * 0 |dd | |dd |
63  * .---. .---.
64  *
65  * non collinear case:
66  * .-------.
67  * | |
68  * .-------.
69  * | |
70  * .-------.
71  */
72  int nst = ctx_.num_bands();
73 
74  auto mem_type_evp = ctx_.std_evp_solver().host_memory_t();
75  auto mem_type_gevp = ctx_.gen_evp_solver().host_memory_t();
76 
77  /* build a full list of G+k vectors for all MPI ranks */
78  generate_gkvec(ctx_.gk_cutoff());
79  /* build a list of basis functions */
80  generate_gklo_basis();
81 
82  if (ctx_.full_potential()) {
83  if (ctx_.control().use_second_variation_) {
84  if (ctx_.need_sv()) {
85  /* in case of collinear magnetism store pure up and pure dn components, otherwise store the full matrix
86  */
87  for (int is = 0; is < ctx_.num_spin_dims(); is++) {
88  sv_eigen_vectors_[is] = dmatrix<double_complex>(nst, nst, ctx_.blacs_grid(), bs, bs, mem_type_evp);
89  }
90  }
91  /* allocate fv eien vectors */
92  fv_eigen_vectors_slab_ = std::unique_ptr<Wave_functions>(
93  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(),
94  [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); },
95  ctx_.num_fv_states(), ctx_.preferred_memory_t()));
96 
97  fv_eigen_vectors_slab_->pw_coeffs(0).prime().zero();
98  fv_eigen_vectors_slab_->mt_coeffs(0).prime().zero();
99  /* starting guess for wave-functions */
100  for (int i = 0; i < ctx_.num_fv_states(); i++) {
101  for (int igloc = 0; igloc < gkvec().gvec_count(comm().rank()); igloc++) {
102  int ig = igloc + gkvec().gvec_offset(comm().rank());
103  if (ig == i) {
104  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 1.0;
105  }
106  if (ig == i + 1) {
107  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.5;
108  }
109  if (ig == i + 2) {
110  fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.125;
111  }
112  }
113  }
114  if (ctx_.iterative_solver_input().type_ == "exact") {
115  /* ELPA needs a full matrix of eigen-vectors as it uses it as a work space */
116  if (ctx_.gen_evp_solver().type() == ev_solver_t::elpa) {
117  fv_eigen_vectors_ = dmatrix<double_complex>(gklo_basis_size(), gklo_basis_size(),
118  ctx_.blacs_grid(), bs, bs, mem_type_gevp);
119  } else{
120  fv_eigen_vectors_ = dmatrix<double_complex>(gklo_basis_size(), ctx_.num_fv_states(),

122  }
123  } else {

125  if (ncomp < 0) {
126  ncomp = ctx_.num_fv_states() / 2;
127  }
128

130  new Wave_functions(gkvec_partition(), ncomp, ctx_.preferred_memory_t()));

132  /* starting guess for wave-functions */
133  for (int i = 0; i < ncomp; i++) {
134  for (int igloc = 0; igloc < gkvec().count(); igloc++) {
135  int ig = igloc + gkvec().offset();
136  if (ig == i) {
137  singular_components_->pw_coeffs(0).prime(igloc, i) = 1.0;
138  }
139  if (ig == i + 1) {
140  singular_components_->pw_coeffs(0).prime(igloc, i) = 0.5;
141  }
142  if (ig == i + 2) {
143  singular_components_->pw_coeffs(0).prime(igloc, i) = 0.125;
144  }
145  }
146  }

149  if (comm().rank() == 0) {
150  utils::print_checksum("singular_components", cs);
151  }
152  }
153  }
154 
155  fv_states_ = std::unique_ptr<Wave_functions>(
156  new Wave_functions(gkvec_partition(), unit_cell_.num_atoms(),
157  [this](int ia) { return unit_cell_.atom(ia).mt_basis_size(); }, ctx_.num_fv_states(),
158  ctx_.preferred_memory_t()));
159 
160  spinor_wave_functions_ = std::make_shared<Wave_functions>(
161  gkvec_partition(), unit_cell_.num_atoms(),
162  [this](int ia) { return unit_cell_.atom(ia).mt_basis_size(); }, nst, ctx_.preferred_memory_t(),
163  ctx_.num_spins());
164  } else {
165  throw std::runtime_error("not implemented");
166  }
167  } else {

171  auto r = unit_cell_.num_wf_with_U();
172  const int num_sc = ctx_.num_mag_dims() == 3 ? 2 : 1;
173  hubbard_wave_functions_ = std::unique_ptr<Wave_functions>(

177  new Wave_functions(gkvec_partition(), r1.first, ctx_.preferred_memory_t(), 1));
178 
179  std::cout << "[K_point::initialize] number of Hubbard WFS (old and new): " << r.first << " " << r1.first <<"\n";
180  }

186 void
187 K_point::orthogonalize_hubbard_orbitals(Wave_functions& phi__)
188 {
189  // do we orthogonalize the all thing
190 
191  const int num_sc = (ctx_.num_mag_dims() == 3) ? 2 : 1;
192 
193  int nwfu = unit_cell_.num_wf_with_U().first;
194 
195  if (ctx_.hubbard_input().orthogonalize_hubbard_orbitals_ || ctx_.hubbard_input().normalize_hubbard_orbitals_) {
196 
197  dmatrix<double_complex> S(nwfu, nwfu);
198  S.zero();
199 
200  if (ctx_.processing_unit() == device_t::GPU) {
201  S.allocate(memory_t::device);
202  }
203 
204  memory_t mem{memory_t::host};
205  linalg_t la{linalg_t::blas};
206  if (ctx_.processing_unit() == device_t::GPU) {
207  mem = memory_t::device;
208  la = linalg_t::gpublas;
209  }
210 
211  /* we do not need to treat both up and down spins for the
212  colinear case because the up and down components are identical */
213  inner<double_complex>(mem, la, (ctx_.num_mag_dims() == 3) ? 2 : 0, phi__, 0, nwfu, this->hubbard_wave_functions(), 0, nwfu, S, 0, 0);
214 
215  if (ctx_.processing_unit() == device_t::GPU) {
216  S.copy_to(memory_t::host);
217  }
218 
219  /* diagonalize the all stuff */
220 
221  if (ctx_.hubbard_input().orthogonalize_hubbard_orbitals_ ) {
222  dmatrix<double_complex> Z(nwfu, nwfu);
223 
224  auto ev_solver = Eigensolver_factory("lapack", nullptr);
225 
226  std::vector<double> eigenvalues(nwfu, 0.0);
227 
228  ev_solver->solve(nwfu, S, &eigenvalues[0], Z);
229 
230  // build the O^{-1/2} operator
231  for (int i = 0; i < static_cast<int>(eigenvalues.size()); i++) {
232  eigenvalues[i] = 1.0 / std::sqrt(eigenvalues[i]);
233  }
234 
235  // // First compute S_{nm} = E_m Z_{nm}
236  S.zero();
237  for (int l = 0; l < nwfu; l++) {
238  for (int m = 0; m < nwfu; m++) {
239  for (int n = 0; n < nwfu; n++) {
240  S(n, m) += eigenvalues[l] * Z(n, l) * std::conj(Z(m, l));
241  }
242  }
243  }
244  } else {
245  for (int l = 0; l < nwfu; l++) {
246  for (int m = 0; m < nwfu; m++) {
247  if (l == m) {
248  S(l, m) = 1.0 / std::sqrt(S(l, l).real());
249  } else {
250  S(l, m) = 0.0;
251  }
252  }
253  }
254  }
255 
256  if (ctx_.processing_unit() == device_t::GPU) {
257  S.copy_to(memory_t::device);
258  }
259 
260  // only need to do that when in the ultra soft case
261  if (unit_cell_.augment()) {
262  for (int s = 0; s < num_sc; s++) {
263  phi__.copy_from(ctx_.processing_unit(), nwfu, hubbard_wave_functions(), s, 0, s, 0);
264  }
265  }
266 
267  // now apply the overlap matrix
268  // Apply the transform on the wave functions
269  transform<double_complex>(mem, la, (ctx_.num_mag_dims() == 3) ? 2 : 0, phi__, 0, nwfu,
270  S, 0, 0, hubbard_wave_functions(), 0, nwfu);
271  }

274 void K_point::generate_gkvec(double gk_cutoff__)
275 {
276  PROFILE("sirius::K_point::generate_gkvec");
277 
278  if (ctx_.full_potential() && (gk_cutoff__ * unit_cell_.max_mt_radius() > ctx_.lmax_apw()) &&
279  comm_.rank() == 0 && ctx_.control().verbosity_ >= 0) {
280  std::stringstream s;
281  s << "G+k cutoff (" << gk_cutoff__ << ") is too large for a given lmax ("
282  << ctx_.lmax_apw() << ") and a maximum MT radius (" << unit_cell_.max_mt_radius() << ")" << std::endl
283  << "suggested minimum value for lmax : " << int(gk_cutoff__ * unit_cell_.max_mt_radius()) + 1;
284  WARNING(s);
285  }
286 
287  if (gk_cutoff__ * 2 > ctx_.pw_cutoff()) {
288  std::stringstream s;
289  s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
290  << " pw cutoff : " << ctx_.pw_cutoff() << std::endl
291  << " doubled G+k cutoff : " << gk_cutoff__ * 2;
292  TERMINATE(s);
293  }
294 
295  /* create G+k vectors; communicator of the coarse FFT grid is used because wave-functions will be transformed
296  * only on the coarse grid; G+k-vectors will be distributed between MPI ranks assigned to the k-point */
297  gkvec_ = std::unique_ptr<Gvec>(new Gvec(vk_, ctx_.unit_cell().reciprocal_lattice_vectors(), gk_cutoff__, comm(),
298  ctx_.gamma_point()));
299 
300  gkvec_partition_ = std::unique_ptr<Gvec_partition>(new Gvec_partition(*gkvec_, ctx_.comm_fft_coarse(),
301  ctx_.comm_band_ortho_fft_coarse()));
302 
303  gkvec_offset_ = gkvec().gvec_offset(comm().rank());
304 
305  const auto fft_type = gkvec_->reduced() ? SPFFT_TRANS_R2C : SPFFT_TRANS_C2C;
306  const auto spfft_pu = ctx_.processing_unit() == device_t::CPU ? SPFFT_PU_HOST : SPFFT_PU_GPU;
307  auto gv = gkvec_partition_->get_gvec();
308  /* create transformation */
309  spfft_transform_.reset(new spfft::Transform(ctx_.spfft_grid_coarse().create_transform(
310  spfft_pu, fft_type, ctx_.fft_coarse_grid()[0], ctx_.fft_coarse_grid()[1], ctx_.fft_coarse_grid()[2],
311  ctx_.spfft_coarse().local_z_length(), gkvec_partition_->gvec_count_fft(), SPFFT_INDEX_TRIPLETS,
312  gv.at(memory_t::host))));

412  fv_evec__(j, ist) = tmp(jloc);

906 {
907  /* total number of Hubbard orbitals */
908  auto r = unit_cell_.num_wf_with_U();
909  const int num_sc = ctx_.num_mag_dims() == 3 ? 2 : 1;
910 
911  Wave_functions phi(gkvec_partition(), r.first, ctx_.preferred_memory_t(), num_sc);
912 
913  for (int ispn = 0; ispn < num_sc; ispn++) {
914  phi.pw_coeffs(ispn).prime().zero();
915  }
916 
917  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
918  auto& atom_type = unit_cell_.atom(ia).type();
919  if (atom_type.hubbard_correction()) {
920  generate_atomic_wave_functions(atom_type.hubbard_indexb_wfc(), ia, r.second[ia], true, phi);
921  }
922  }
923 
924  /* check if we have a norm conserving pseudo potential only */
925  auto q_op = (unit_cell_.augment()) ? std::unique_ptr<Q_operator>(new Q_operator(ctx_)) : nullptr;
926 
927  if (ctx_.processing_unit() == device_t::GPU) {
928  for (int ispn = 0; ispn < num_sc; ispn++) {
929  /* allocate GPU memory */
930  phi.pw_coeffs(ispn).prime().allocate(memory_t::device);
931  // can do async copy
932  phi.pw_coeffs(ispn).copy_to(memory_t::device, 0, r.first);
933  hubbard_wave_functions_->pw_coeffs(ispn).prime().allocate(memory_t::device);
934  }
935  }
936  sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(num_sc), 0, r.first, beta_projectors(),
937  phi, q_op.get(), *hubbard_wave_functions_);
938 
939  orthogonalize_hubbard_orbitals(phi);
940

942  if (ctx_.processing_unit() == device_t::GPU) {
943  for (int ispn = 0; ispn < num_sc; ispn++) {
944  /* copy the hubbard wave functions on the host and then deallocate on GPU */
945  hubbard_wave_functions().pw_coeffs(ispn).copy_to(memory_t::host, 0, r.first);
946  hubbard_wave_functions().pw_coeffs(ispn).deallocate(memory_t::device);
947  }
948  }
