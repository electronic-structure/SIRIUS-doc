32template <typename T>
33void
34K_point<T>::initialize()
35{
36 PROFILE("sirius::K_point::initialize");
37
38 zil_.resize(ctx_.lmax_apw() + 1);
39 for (int l = 0; l <= ctx_.lmax_apw(); l++) {
40 zil_[l] = std::pow(std::complex<T>(0, 1), l);
41 }
42
43 l_by_lm_ = utils::l_by_lm(ctx_.lmax_apw());
44
45 int bs = ctx_.cyclic_block_size();
46
47 /* In case of collinear magnetism we store only non-zero spinor components.
48 *
49 * non magnetic case:
50 * .---.
51 * | |
52 * .---.
53 *
54 * collinear case:
55 * .---. .---.
56 * |uu | 0 |uu |
57 * .---.---. -> .---.
58 * 0 |dd | |dd |
59 * .---. .---.
60 *
61 * non collinear case:
62 * .-------.
63 * | |
64 * .-------.
65 * | |
66 * .-------.
67 */
68 int nst = ctx_.num_bands();
69
70 auto mem_type_evp = ctx_.std_evp_solver().host_memory_t();
71 auto mem_type_gevp = ctx_.gen_evp_solver().host_memory_t();
72
73 /* build a full list of G+k vectors for all MPI ranks */
74 generate_gkvec(ctx_.gk_cutoff());
75 /* build a list of basis functions */
76 generate_gklo_basis();
77
78 if (ctx_.full_potential()) {
79 if (ctx_.cfg().control().use_second_variation()) {
80
81 assert(ctx_.num_fv_states() > 0);
82 fv_eigen_values_ = sddk::mdarray<double, 1>(ctx_.num_fv_states(), memory_t::host, "fv_eigen_values");
83
84 if (ctx_.need_sv()) {
85 /* in case of collinear magnetism store pure up and pure dn components, otherwise store the full matrix
86 */
87 for (int is = 0; is < ctx_.num_spinors(); is++) {
88 sv_eigen_vectors_[is] = dmatrix<std::complex<T>>(nst, nst, ctx_.blacs_grid(), bs, bs, mem_type_evp);
89 }
90 }
91 /* allocate fv eien vectors */
92 fv_eigen_vectors_slab_ = std::unique_ptr<Wave_functions<T>>(
93 new Wave_functions<T>(gkvec_partition(), unit_cell_.num_atoms(),
94 [this](int ia) { return unit_cell_.atom(ia).mt_lo_basis_size(); },
95 ctx_.num_fv_states(), ctx_.preferred_memory_t()));
96
97 fv_eigen_vectors_slab_->pw_coeffs(0).prime().zero();
98 fv_eigen_vectors_slab_->mt_coeffs(0).prime().zero();
99 /* starting guess for wave-functions */
100 for (int i = 0; i < ctx_.num_fv_states(); i++) {
101 for (int igloc = 0; igloc < gkvec().gvec_count(comm().rank()); igloc++) {
102 int ig = igloc + gkvec().gvec_offset(comm().rank());
103 if (ig == i) {
104 fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 1.0;
105 }
106 if (ig == i + 1) {
107 fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.5;
108 }
109 if (ig == i + 2) {
110 fv_eigen_vectors_slab_->pw_coeffs(0).prime(igloc, i) = 0.125;
111 }
112 }
113 }
114 if (ctx_.cfg().iterative_solver().type() == "exact") {
115 /* ELPA needs a full matrix of eigen-vectors as it uses it as a work space */
116 if (ctx_.gen_evp_solver().type() == ev_solver_t::elpa) {
117 fv_eigen_vectors_ = dmatrix<std::complex<T>>(gklo_basis_size(), gklo_basis_size(),
118 ctx_.blacs_grid(), bs, bs, mem_type_gevp);
119 } else{
120 fv_eigen_vectors_ = dmatrix<std::complex<T>>(gklo_basis_size(), ctx_.num_fv_states(),
121 ctx_.blacs_grid(), bs, bs, mem_type_gevp);
122 }
123 } else {
124 int ncomp = ctx_.cfg().iterative_solver().num_singular();
125 if (ncomp < 0) {
126 ncomp = ctx_.num_fv_states() / 2;
127 }
128
129 singular_components_ = std::unique_ptr<Wave_functions<T>>(
130 new Wave_functions<T>(gkvec_partition(), ncomp, ctx_.preferred_memory_t()));
131 singular_components_->pw_coeffs(0).prime().zero();
132 /* starting guess for wave-functions */
133 for (int i = 0; i < ncomp; i++) {
134 for (int igloc = 0; igloc < gkvec().count(); igloc++) {
135 int ig = igloc + gkvec().offset();
136 if (ig == i) {
137 singular_components_->pw_coeffs(0).prime(igloc, i) = 1.0;
138 }
139 if (ig == i + 1) {
140 singular_components_->pw_coeffs(0).prime(igloc, i) = 0.5;
141 }
142 if (ig == i + 2) {
143 singular_components_->pw_coeffs(0).prime(igloc, i) = 0.125;
144 }
145 }
146 }
147 if (ctx_.cfg().control().print_checksum()) {
148 singular_components_->print_checksum(device_t::CPU, "singular_components", 0, ncomp);
149 }
150 }
151
152 fv_states_ = std::unique_ptr<Wave_functions<T>>(
153 new Wave_functions<T>(gkvec_partition(), unit_cell_.num_atoms(),
154 [this](int ia) { return unit_cell_.atom(ia).mt_basis_size(); }, ctx_.num_fv_states(),
155 ctx_.preferred_memory_t()));
156
157 spinor_wave_functions_ = std::make_shared<Wave_functions<T>>(
158 gkvec_partition(), unit_cell_.num_atoms(),
159 [this](int ia) { return unit_cell_.atom(ia).mt_basis_size(); }, nst, ctx_.preferred_memory_t(),
160 ctx_.num_spins());
161 } else {
162 throw std::runtime_error("not implemented");
163 }
164 } else {
165 spinor_wave_functions_ =
166 std::make_shared<Wave_functions<T>>(gkvec_partition(), nst, ctx_.preferred_memory_t(), ctx_.num_spins());
167 if (ctx_.hubbard_correction()) {
168 /* allocate Hubbard wave-functions */
169 auto r = unit_cell_.num_hubbard_wf();
170 wave_functions_S_hub_ = std::unique_ptr<Wave_functions<T>>(
171 new Wave_functions<T>(gkvec_partition(), r.first * ctx_.num_spinor_comp(),
172 ctx_.preferred_memory_t(), ctx_.num_spins()));
173 wave_functions_hub_ = std::unique_ptr<Wave_functions<T>>(
174 new Wave_functions<T>(gkvec_partition(), r.first * ctx_.num_spinor_comp(),
175 ctx_.preferred_memory_t(), ctx_.num_spins()));
176 atomic_wave_functions_hub_ = std::unique_ptr<Wave_functions<T>>(
177 new Wave_functions<T>(gkvec_partition(), r.first * ctx_.num_spinor_comp(),
178 ctx_.preferred_memory_t(), ctx_.num_spins()));
179 atomic_wave_functions_S_hub_ = std::unique_ptr<Wave_functions<T>>(
180 new Wave_functions<T>(gkvec_partition(), r.first * ctx_.num_spinor_comp(),
181 ctx_.preferred_memory_t(), ctx_.num_spins()));
182 }
183 }
184
185 update();
186}

188template <typename T>
189void
190K_point<T>::generate_hubbard_orbitals()
191{
192 PROFILE("sirius::K_point::generate_hubbard_orbitals");
193
194 /* phi and s_phi are aliases for the atomic wave functions. They are *not*
195 * the hubbard orbitals */
196
197 auto &phi = atomic_wave_functions_hub();
198 auto &s_phi = atomic_wave_functions_S_hub();
199
200 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
201 wave_functions_S_hub_->pw_coeffs(ispn).prime().zero();
202 wave_functions_hub_->pw_coeffs(ispn).prime().zero();
203 s_phi.pw_coeffs(ispn).prime().zero();
204 phi.pw_coeffs(ispn).prime().zero();
205 }
206 /* total number of Hubbard orbitals */
207 auto r = unit_cell_.num_hubbard_wf();
208
209 std::vector<int> atoms;
210 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
211 auto& atom_type = unit_cell_.atom(ia).type();
212 if (atom_type.hubbard_correction()) {
213 atoms.push_back(ia);
214 }
215 }
216
217 /* generate the initial atomic wavefunctions */
218 this->generate_atomic_wave_functions(atoms, [&](int iat){return &ctx_.unit_cell().atom_type(iat).indexb_hub();},
219 ctx_.hubbard_wf_ri(), phi);
220
221 if (ctx_.cfg().control().print_checksum()) {
222 phi.print_checksum(device_t::CPU, "phi_hub_init", 0, phi.num_wf());
223 }
224
225 if (ctx_.num_spins() == 2) {
226 /* copy up component to dn component in collinear case
227 * +-------------------------------+
228 * | phi1_{lm}, phi2_{lm}, ... |
229 * +-------------------------------+
230 * | phi1_{lm}, phi2_{lm}, ... |
231 * +-------------------------------+
232 *
233 * or with offset in non-collinear case
234 *
235 * +-------------------------------+---------------------------------+
236 * | phi1_{lm}, phi2_{lm}, ... | 0 |
237 * +-------------------------------+---------------------------------+
238 * | 0 | phi1_{lm}, phi2_{lm}, ... |
239 * +-------------------------------+---------------------------------+
240 */
241 phi.copy_from(device_t::CPU, r.first, phi, 0, 0, 1, (ctx_.num_mag_dims() == 3) ? r.first : 0);
242 }
243
244 /* check if we have a norm conserving pseudo potential only */
245 auto q_op = (unit_cell_.augment()) ? std::unique_ptr<Q_operator<T>>(new Q_operator<T>(ctx_)) : nullptr;
246
247 auto sr = spin_range(ctx_.num_spins() == 2 ? 2 : 0);
248 phi.prepare(sr, true);
249 wave_functions_S_hub_->prepare(sr, false);
250 s_phi.prepare(sr, false);
251 wave_functions_hub_->prepare(sr, false);
252
253 /* compute S|phi> */
254 beta_projectors().prepare();
255 for (int is = 0; is < ctx_.num_spinors(); is++ ) {
256 /* spin range to apply S-operator.
257 * if WFs are non-magnetic, sping range is [0] or [1] - apply to single component
258 * if WFs have two components, spin range is [0,1] and S will be aplpied to both components */
259 auto sr = ctx_.num_mag_dims() == 3 ? spin_range(2) : spin_range(is);
260
261 sirius::apply_S_operator<std::complex<T>>(ctx_.processing_unit(), sr, 0, phi.num_wf(),
262 beta_projectors(), phi, q_op.get(), s_phi);
263 }
264
265 if (ctx_.cfg().control().print_checksum()) {
266 s_phi.print_checksum(device_t::CPU, "sphi_hub_init", 0,
267 s_phi.num_wf());
268 }
269
270 /* now compute the hubbard wfc from the atomic orbitals */
271 orthogonalize_hubbard_orbitals(phi, s_phi, *wave_functions_hub_, *wave_functions_S_hub_);
272
273 for (int is = 0; is < ctx_.num_spinors(); is++ ) {
274 /* spin range to apply S-operator.
275 * if WFs are non-magnetic, sping range is [0] or [1] - apply to single component
276 * if WFs have two components, spin range is [0,1] and S will be aplpied to both components */
277 auto sr = ctx_.num_mag_dims() == 3 ? spin_range(2) : spin_range(is);
278
279 sirius::apply_S_operator<std::complex<T>>(ctx_.processing_unit(), sr, 0, phi.num_wf(),
280 beta_projectors(), *wave_functions_hub_, q_op.get(),
281 *wave_functions_S_hub_);
282 }
283
284 beta_projectors().dismiss();
285
286 /* all calculations on GPU then we need to copy the final result back to the CPUs */
287 wave_functions_S_hub_->dismiss(sr, true);
288 phi.dismiss(sr, true);
289 s_phi.dismiss(sr, true);
290 wave_functions_hub_->dismiss(sr, true);
291
292
293
294 if (ctx_.cfg().control().print_checksum()) {
295 wave_functions_S_hub_->print_checksum(device_t::CPU, "phi_hub", 0, wave_functions_S_hub_->num_wf());
296 }
297}

299template <typename T>
300void
301K_point<T>::orthogonalize_hubbard_orbitals(Wave_functions<T>& phi__, Wave_functions<T>& sphi__,
302 Wave_functions<T>& phi_hub__, Wave_functions<T>& sphi_hub__)
303{
304 int nwfu = phi__.num_wf();
305
306 if (!(ctx_.cfg().hubbard().orthogonalize() || ctx_.cfg().hubbard().normalize())) {
307 for (int s = 0; s < ctx_.num_spins(); s++) {
308 sphi_hub__.copy_from(ctx_.processing_unit(), nwfu, sphi__, s, 0, s, 0);
309 }
310
311 // needed to compute the forces
312 for (int s = 0; s < ctx_.num_spins(); s++) {
313 phi_hub__.copy_from(ctx_.processing_unit(), nwfu, phi__, s, 0, s, 0);
314 }
315 return;
316 }
317
318
319 dmatrix<std::complex<T>> S(nwfu, nwfu);
320 if (ctx_.processing_unit() == device_t::GPU) {
321 S.allocate(memory_t::device);
322 }
323
324 for (int istep = 0; istep < ctx_.num_spinors(); istep++) {
325
326 auto sr = spin_range(ctx_.num_mag_dims() == 3 ? 2 : istep);
327
328 /* compute inner product between full spinors or between indpendent components */
329 inner<std::complex<T>>(ctx_.spla_context(), sr, phi__, 0, nwfu, sphi__, 0, nwfu, S, 0, 0);
330
331 // SPLA should return on CPU as well
332 // if (ctx_.processing_unit() == device_t::GPU) {
333 // S.copy_to(memory_t::host);
334 //}
335
336
337 /* create transformation matrix */
338 if (ctx_.cfg().hubbard().orthogonalize()) {
339 dmatrix<std::complex<T>> Z(nwfu, nwfu);
340
341 auto ev_solver = Eigensolver_factory("lapack", nullptr);
342
343 std::vector<T> eigenvalues(nwfu, 0.0);
344
345 ev_solver->solve(nwfu, S, eigenvalues.data(), Z);
346
347 /* build the O^{-1/2} operator */
348 for (int i = 0; i < static_cast<int>(eigenvalues.size()); i++) {
349 eigenvalues[i] = 1.0 / std::sqrt(eigenvalues[i]);
350 }
351
352 /* first compute S_{nm} = E_m Z_{nm} */
353 S.zero();
354 for (int l = 0; l < nwfu; l++) {
355 for (int m = 0; m < nwfu; m++) {
356 for (int n = 0; n < nwfu; n++) {
357 S(n, m) += eigenvalues[l] * Z(n, l) * std::conj(Z(m, l));
358 }
359 }
360 }
361 } else {
362 for (int l = 0; l < nwfu; l++) {
363 for (int m = 0; m < nwfu; m++) {
364 if (l == m) {
365 S(l, m) = 1.0 / std::sqrt(S(l, l).real());
366 } else {
367 S(l, m) = 0.0;
368 }
369 }
370 }
371 }
372
373 if (ctx_.processing_unit() == device_t::GPU) {
374 S.copy_to(memory_t::device);
375 }
376
377 /* transform on the wave functions */
378 transform<std::complex<T>>(ctx_.spla_context(), sr(), phi__, 0, nwfu, S, 0, 0, phi_hub__, 0, nwfu);
379 }

382template <typename T>
383void K_point<T>::generate_gkvec(double gk_cutoff__)
384{
385 PROFILE("sirius::K_point::generate_gkvec");
386
387 if (ctx_.full_potential() && (gk_cutoff__ * unit_cell_.max_mt_radius() > ctx_.lmax_apw()) &&
388 ctx_.comm().rank() == 0 && ctx_.verbosity() >= 0) {
389 std::stringstream s;
390 s << "G+k cutoff (" << gk_cutoff__ << ") is too large for a given lmax ("
391 << ctx_.lmax_apw() << ") and a maximum MT radius (" << unit_cell_.max_mt_radius() << ")" << std::endl
392 << "suggested minimum value for lmax : " << int(gk_cutoff__ * unit_cell_.max_mt_radius()) + 1;
393 WARNING(s);
394 }
395
396 if (gk_cutoff__ * 2 > ctx_.pw_cutoff()) {
397 std::stringstream s;
398 s << "G+k cutoff is too large for a given plane-wave cutoff" << std::endl
399 << " pw cutoff : " << ctx_.pw_cutoff() << std::endl
400 << " doubled G+k cutoff : " << gk_cutoff__ * 2;
401 TERMINATE(s);
402 }
403
404 gkvec_partition_ = std::unique_ptr<Gvec_partition>(new Gvec_partition(this->gkvec(), ctx_.comm_fft_coarse(),
405 ctx_.comm_band_ortho_fft_coarse()));
406
407 gkvec_offset_ = gkvec().gvec_offset(comm().rank());
408
409 const auto fft_type = gkvec_->reduced() ? SPFFT_TRANS_R2C : SPFFT_TRANS_C2C;
410 const auto spfft_pu = ctx_.processing_unit() == device_t::CPU ? SPFFT_PU_HOST : SPFFT_PU_GPU;
411 auto gv = gkvec_partition_->get_gvec();
412 /* create transformation */
413 spfft_transform_.reset(new spfft_transform_type<T>(ctx_.spfft_grid_coarse<T>().create_transform(
414 spfft_pu, fft_type, ctx_.fft_coarse_grid()[0], ctx_.fft_coarse_grid()[1], ctx_.fft_coarse_grid()[2],
415 ctx_.spfft_coarse<double>().local_z_length(), gkvec_partition_->gvec_count_fft(), SPFFT_INDEX_TRIPLETS,
416 gv.at(memory_t::host))));
417}

497template <typename T>
498void K_point<T>::get_fv_eigen_vectors(mdarray<std::complex<T>, 2> &fv_evec__) const
499{
500 assert((int) fv_evec__.size(0) >= gklo_basis_size());
501 assert((int) fv_evec__.size(1) == ctx_.num_fv_states());
502 assert(gklo_basis_size_row() == fv_eigen_vectors_.num_rows_local());
503
504 mdarray<std::complex<T>, 1> tmp(gklo_basis_size_row());
505
506 fv_evec__.zero();
507
508 for (int ist = 0; ist < ctx_.num_fv_states(); ist++) {
509 auto loc = fv_eigen_vectors_.spl_col().location(ist);
510 if (loc.rank == fv_eigen_vectors_.rank_col()) {
511 std::copy(&fv_eigen_vectors_(0, loc.local_index),
512 &fv_eigen_vectors_(0, loc.local_index) + gklo_basis_size_row(),
513 &tmp(0));
514 }
515 fv_eigen_vectors_.blacs_grid().comm_col().bcast(&tmp(0), gklo_basis_size_row(), loc.rank);
516 for (int jloc = 0; jloc < gklo_basis_size_row(); jloc++) {
517 int j = fv_eigen_vectors_.irow(jloc);
518 fv_evec__(j, ist) = tmp(jloc);
519 }
520 fv_eigen_vectors_.blacs_grid().comm_row().allreduce(&fv_evec__(0, ist), gklo_basis_size());
521 }

878 \endverbatim
879*/
880template <typename T>
881void K_point<T>::save(std::string const& name__, int id__) const
882{
883 /* rank 0 creates placeholders in the HDF5 file */
884 if (comm().rank() == 0) {
885 /* open file with write access */
886 HDF5_tree fout(name__, hdf5_access_t::read_write);
887 /* create /K_point_set/ik */
888 fout["K_point_set"].create_node(id__);
889 fout["K_point_set"][id__].write("vk", &vk_[0], 3);
890 fout["K_point_set"][id__].write("band_energies", band_energies_);
891 fout["K_point_set"][id__].write("band_occupancies", band_occupancies_);
892
893 /* save the entire G+k object */
894 //TODO: only the list of z-columns is probably needed to recreate the G+k vectors
895 //serializer s;
896 //gkvec().pack(s);
897 //fout["K_point_set"][id__].write("gkvec", s.stream());
898
899 /* save the order of G-vectors */
900 mdarray<int, 2> gv(3, num_gkvec());
901 for (int i = 0; i < num_gkvec(); i++) {
902 auto v = gkvec().gvec(i);
903 for (int x: {0, 1, 2}) {
904 gv(x, i) = v[x];
905 }
906 }
907 fout["K_point_set"][id__].write("gvec", gv);
908 fout["K_point_set"][id__].create_node("bands");
909 for (int i = 0; i < ctx_.num_bands(); i++) {
910 fout["K_point_set"][id__]["bands"].create_node(i);
911 fout["K_point_set"][id__]["bands"][i].create_node("spinor_wave_function");
912 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
913 fout["K_point_set"][id__]["bands"][i]["spinor_wave_function"].create_node(ispn);
914 }
915 }
916 }
917 /* wait for rank 0 */
918 comm().barrier();

921 std::vector<std::complex<T>> wf_tmp(num_gkvec());
