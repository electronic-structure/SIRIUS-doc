204  inline void set(const int irow_glob, const int icol_glob, T val)
205  {
206  auto r = spl_row_.location(irow_glob);
207  if (blacs_grid_->rank_row() == r.rank) {
208  auto c = spl_col_.location(icol_glob);
209  if (blacs_grid_->rank_col() == c.rank) {
210  (*this)(r.local_index, c.local_index) = val;
211  }

226  inline void make_real_diag(int n__)
227  {
228  for (int i = 0; i < n__; i++) {
229  auto r = spl_row_.location(i);
230  if (blacs_grid_->rank_row() == r.rank) {
231  auto c = spl_col_.location(i);
232  if (blacs_grid_->rank_col() == c.rank) {
233  T v = (*this)(r.local_index, c.local_index);
234  (*this)(r.local_index, c.local_index) = sddk_type_wrapper<T>::real(v);
235  }
236  }
