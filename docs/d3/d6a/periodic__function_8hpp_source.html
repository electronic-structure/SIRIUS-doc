34 /// Representation of the periodical function on the muffin-tin geometry.
35 /** Inside each muffin-tin the spherical expansion is used:
36  * \f[
37  * f({\bf r}) = \sum_{\ell m} f_{\ell m}(r) Y_{\ell m}(\hat {\bf r})
38  * \f]
39  * or
40  * \f[
41  * f({\bf r}) = \sum_{\ell m} f_{\ell m}(r) R_{\ell m}(\hat {\bf r})
42  * \f]
43  * In the interstitial region function is stored on the real-space grid or as a Fourier series:
44  * \f[
45  * f({\bf r}) = \sum_{{\bf G}} f({\bf G}) e^{i{\bf G}{\bf r}}
46  * \f]
47  */

169  Smooth_periodic_function<T>::add(g__);

171  if (ctx_.full_potential()) {

173  f_mt_local_(ialoc) += g__.f_mt(ialoc);

177  T integrate(std::vector<T>& mt_val, T& it_val) const
178  {
179  PROFILE("sirius::Periodic_function::integrate");
180 
181  it_val = 0;
182 
183  if (!ctx_.full_potential()) {
184  #pragma omp parallel for schedule(static) reduction(+:it_val)
185  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
186  it_val += this->f_rg_(irloc);
187  }
188  } else {
189  #pragma omp parallel for schedule(static) reduction(+:it_val)
190  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
191  it_val += this->f_rg_(irloc) * ctx_.theta(irloc);
192  }
193  }
194  it_val *= (unit_cell_.omega() / this->fft_->size());
195  this->fft_->comm().allreduce(&it_val, 1);
196  T total = it_val;
197 
198  if (ctx_.full_potential()) {
199  mt_val = std::vector<T>(unit_cell_.num_atoms(), 0);
200 
201  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
202  int ia = unit_cell_.spl_num_atoms(ialoc);
203  mt_val[ia] = f_mt_local_(ialoc).component(0).integrate(2) * fourpi * y00;
204  }
205 
206  comm_.allreduce(&mt_val[0], unit_cell_.num_atoms());
207  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
208  total += mt_val[ia];
209  }
210  }
211 
212  return total;
213  }

297  double value_rg(vector3d<double> const& vc)
298  {
299  double p{0};
300  for (int igloc = 0; igloc < gvec_.count(); igloc++) {
301  vector3d<double> vgc = gvec_.gvec_cart<index_domain_t::local>(igloc);
302  p += std::real(this->f_pw_local_(igloc) * std::exp(double_complex(0.0, dot(vc, vgc))));
303  }
304  gvec_.comm().allreduce(&p, 1);
305  return p;
306  }
307 
308  double value(vector3d<double> const& vc)
309  {
310  int ja{-1}, jr{-1};
311  double dr{0}, tp[2];
312 
313  if (unit_cell_.is_point_in_mt(vc, ja, jr, dr, tp)) {
314  int lmax = utils::lmax(angular_domain_size_);
315  std::vector<double> rlm(angular_domain_size_);
316  SHT::spherical_harmonics(lmax, tp[0], tp[1], &rlm[0]);
317  double p{0};
318  for (int lm = 0; lm < angular_domain_size_; lm++) {
319  double d = (f_mt_(lm, jr + 1, ja) - f_mt_(lm, jr, ja)) / unit_cell_.atom(ja).type().radial_grid().dx(jr);
320 
321  p += rlm[lm] * (f_mt_(lm, jr, ja) + d * dr);
322  }
323  return p;
324  } else {
325  return value_rg(vc);
326  }
327  }

334  /// Compute inner product <f|g>
335  T inner(Periodic_function<T> const& g__) const
336  {
337  PROFILE("sirius::Periodic_function::inner");
338 
339  assert(this->fft_ == g__.fft_);
340  assert(&unit_cell_ == &g__.unit_cell_);
341  assert(&comm_ == &g__.comm_);
342 
343  T result_rg{0};
344 
345  if (!ctx_.full_potential()) {
346  result_rg = sirius::inner(static_cast<Smooth_periodic_function<T> const&>(*this),
347  static_cast<Smooth_periodic_function<T> const&>(g__));
348  } else {
349  for (int irloc = 0; irloc < this->fft_->local_size(); irloc++) {
350  //result_rg += type_wrapper<T>::bypass(std::conj(this->f_rg(irloc))) * g__.f_rg(irloc) *
351  // this->ctx_.theta(irloc);
352  result_rg += utils::conj(this->f_rg(irloc)) * g__.f_rg(irloc) *
353  this->ctx_.theta(irloc);
354  }
355  result_rg *= (unit_cell_.omega() / this->fft_->size());
356  this->fft_->comm().allreduce(&result_rg, 1);
357  }
358 
359  T result_mt{0};
360  if (ctx_.full_potential()) {
361  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
362  auto r = sirius::inner(f_mt(ialoc), g__.f_mt(ialoc));
363  result_mt += r;
364  }
365  comm_.allreduce(&result_mt, 1);
366  }
367 
368  return result_mt + result_rg;
369  }
370 };
