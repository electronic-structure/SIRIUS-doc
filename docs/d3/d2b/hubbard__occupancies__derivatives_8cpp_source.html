38 /* compute this |dphi> = dS | phi> + |dphi>, where the derivative is taken
39  * compared to atom_id displacement. we can also use lambda */
40 void
41 Hubbard::apply_dS(K_point& kp, Q_operator& q_op, Beta_projectors_gradient& bp_grad, const int atom_id, const int dir,
42  Wave_functions& phi, Wave_functions& dphi)
43 {
44  // compute d S/ dr^I_a |phi> and add to dphi
45  if (!ctx_.full_potential() && ctx_.unit_cell().augment()) {
46  // it is equal to
47  // \sum Q^I_ij <d \beta^I_i|phi> |\beta^I_j> + < \beta^I_i|phi> |d\beta^I_j>
48  for (int chunk__ = 0; chunk__ < kp.beta_projectors().num_chunks(); chunk__++) {
49  for (int i = 0; i < kp.beta_projectors().chunk(chunk__).num_atoms_; i++) {
50  // need to find the right atom in the chunks.
51  if (kp.beta_projectors().chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::ia), i) == atom_id) {
52  kp.beta_projectors().generate(chunk__);
53  bp_grad.generate(chunk__, dir);
54 
55  // compute Q_ij <\beta_i|\phi> |d \beta_j> and add it to d\phi
56  {
57  /* <beta | phi> for this chunk */
58  auto beta_phi = kp.beta_projectors().inner<double_complex>(chunk__, phi, 0, 0,
59  this->number_of_hubbard_orbitals());
60  q_op.apply(chunk__, i, 0, dphi, 0, this->number_of_hubbard_orbitals(), bp_grad, beta_phi);
61  }
62 
63  // compute Q_ij <d \beta_i|\phi> |\beta_j> and add it to d\phi
64  {
65  /* <dbeta | phi> for this chunk */
66  auto dbeta_phi =
67  bp_grad.inner<double_complex>(chunk__, phi, 0, 0, this->number_of_hubbard_orbitals());
68 
69  /* apply Q operator (diagonal in spin) */
70  /* Effectively compute Q_ij <d beta_i| phi> |beta_j> and add it dphi */
71  q_op.apply(chunk__, i, 0, dphi, 0, this->number_of_hubbard_orbitals(), kp.beta_projectors(),
72  dbeta_phi);
73  }
74  }
75  }
76  }
77  }
78 }

80 void
81 Hubbard::compute_occupancies_derivatives(K_point& kp, Q_operator& q_op, sddk::mdarray<double_complex, 6>& dn__)
82 {
83  PROFILE("sirius::Hubbard::compute_occupancies_derivatives");
84 
85 
86  auto la = linalg_t::none;
87  auto mt = memory_t::none;
88  switch (ctx_.processing_unit()) {
89  case device_t::CPU: {
90  la = linalg_t::blas;
91  mt = memory_t::host;
92  break;
93  }
94  case device_t::GPU: {
95  la = linalg_t::gpublas;
96  mt = memory_t::device;
97  break;
98  }
99  default:
100  break;
101  }
102 
103  dn__.zero(memory_t::host);
104  // check if we have a norm conserving pseudo potential only. Only
105  // derivatives of the atomic wave functions are needed.
106  auto& phi = kp.atomic_wave_functions_hub();
107 
108  Beta_projectors_gradient bp_grad(ctx_, kp.gkvec(), kp.igk_loc(), kp.beta_projectors());
109  bp_grad.prepare();
110 
111  /*
112  Compute the derivatives of the occupancies in two cases.
113 
114  - the atom is pp norm conserving or
115 
116  - the atom is ppus (in that case the derivative of the S operator gives a
117  non zero contribution)
118  */
119 
120  /* temporary wave functions */
121  Wave_functions dphi(kp.gkvec_partition(), this->number_of_hubbard_orbitals(), ctx_.preferred_memory_t(), 1);
122  /* temporary wave functions */
123  Wave_functions phitmp(kp.gkvec_partition(), this->number_of_hubbard_orbitals(), ctx_.preferred_memory_t(), 1);
124 
125  int HowManyBands = kp.num_occupied_bands(0);
126  if (ctx_.num_spins() == 2) {
127  HowManyBands = std::max(kp.num_occupied_bands(1), kp.num_occupied_bands(0));
128  }
129 
130  dmatrix<double_complex> phi_s_psi(HowManyBands, this->number_of_hubbard_orbitals() * ctx_.num_spins());
131 
132  /* <atomic_orbitals | S | atomic_orbitals> */
133  dmatrix<double_complex> overlap__(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
134 
135  std::vector<double> eigenvalues(this->number_of_hubbard_orbitals());
136 
137  /* transformation matrix going from overlap to diagonal form */
138  dmatrix<double_complex> U__(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
139 
140  /* Matrix orthogonalizing the wave functions set */
141  dmatrix<double_complex> O__(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
142 
143  /* derivative of the Matrix orthogonalizing the wave functions set */
144  dmatrix<double_complex> d_O_(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
145 
146  mdarray<double_complex, 5> dn_tmp(max_number_of_orbitals_per_atom(), max_number_of_orbitals_per_atom(),
147  ctx_.num_spins(), ctx_.unit_cell().num_atoms(), 3);
148 
149  if (ctx_.processing_unit() == device_t::GPU) {
150  dn_tmp.allocate(memory_t::device);
151  if (ctx_.cfg().hubbard().orthogonalize()) {
152  overlap__.allocate(memory_t::device);
153  d_O_.allocate(memory_t::device);
154  O__.allocate(memory_t::device);
155  U__.allocate(memory_t::device);
156  }
157  /* allocation of the overlap matrices on GPU */
158  phi_s_psi.allocate(memory_t::device);
159  // dphi_s_psi.allocate(memory_t::device);
160 
161  /* wave functions */
162  phitmp.allocate(spin_range(0), memory_t::device);
163  phi.allocate(spin_range(ctx_.num_spins()), memory_t::device);
164  dphi.allocate(spin_range(0), memory_t::device);
165 
166  kp.hubbard_wave_functions().allocate(spin_range(ctx_.num_spins()), memory_t::device);
167  kp.hubbard_wave_functions().copy_to(spin_range(ctx_.num_spins()), memory_t::device, 0, this->number_of_hubbard_orbitals());
168 
169  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
170  phi.copy_to(spin_range(ispn), memory_t::device, 0, this->number_of_hubbard_orbitals());
171  }
172 
173  kp.spinor_wave_functions().allocate(spin_range(ctx_.num_spins()), memory_t::device);
174  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
175  kp.spinor_wave_functions().copy_to(spin_range(ispn), memory_t::device, 0, kp.num_occupied_bands(ispn));
176  }
177  }
178 
179  /* compute the overlap matrix and diagonalize it */
180  if (ctx_.cfg().hubbard().orthogonalize()) {
181  overlap__.zero(memory_t::host);
182  overlap__.zero(memory_t::device);
183  dphi.pw_coeffs(0).prime().zero(memory_t::host);
184  dphi.pw_coeffs(0).prime().zero(memory_t::device);
185  /* compute S | phi> where |phi> are the atomic orbitals */
186  sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(0), 0,
187  this->number_of_hubbard_orbitals(), kp.beta_projectors(), phi, &q_op,
188  dphi);
189  inner(ctx_.spla_context(), sddk::spin_range(0), phi, 0, this->number_of_hubbard_orbitals(), dphi, 0,
190  this->number_of_hubbard_orbitals(), overlap__, 0, 0);
191 
192  if (ctx_.processing_unit() == device_t::GPU) {
193  overlap__.copy_to(memory_t::host);
194  }
195  auto ev_solver = Eigensolver_factory("lapack", nullptr);
196 
197  ev_solver->solve(this->number_of_hubbard_orbitals(), overlap__, eigenvalues.data(), U__);
198 
199  if (ctx_.processing_unit() == device_t::GPU) {
200  U__.copy_to(memory_t::device);
201  }
202 
203  }
204 
205  phi_s_psi.zero(memory_t::host);
206  phi_s_psi.zero(memory_t::device);
207 
208  /* compute <phi (O)^(-1/2)| S | psi_{nk}> where |phi (O)^(-1/2)> are the hubbard wavefunctions */
209  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
210  inner(ctx_.spla_context(), spin_range(ispn), kp.spinor_wave_functions(), 0, kp.num_occupied_bands(ispn),
211  kp.hubbard_wave_functions(), 0, this->number_of_hubbard_orbitals(), phi_s_psi, 0,
212  ispn * this->number_of_hubbard_orbitals());
213  }
214 
215  for (int atom_id = 0; atom_id < ctx_.unit_cell().num_atoms(); atom_id++) { // loop over the atom displacement.
216  dn_tmp.zero(memory_t::host);
217  dn_tmp.zero(memory_t::device);
218  for (int dir = 0; dir < 3; dir++) {
219  // reset dphi
220  dphi.pw_coeffs(0).prime().zero(memory_t::host);
221  dphi.pw_coeffs(0).prime().zero(memory_t::device);
222 
223  phitmp.pw_coeffs(0).prime().zero(memory_t::host);
224  phitmp.pw_coeffs(0).prime().zero(memory_t::device);
225 
226  // compute S|d\phi>. Will be zero if the atom has no hubbard
227  // correction
228  if (ctx_.unit_cell().atom(atom_id).type().hubbard_correction()) {
229 
230  // atom atom_id has hubbard correction so we need to compute the
231  // derivatives of the hubbard orbitals associated to the atom
232  // atom_id, the derivatives of the others hubbard orbitals been
233  // zero compared to the displacement of atom atom_id
234 
235  // compute the derivative of |phi> corresponding to the
236  // atom atom_id
237  const int lmax_at = 2 * ctx_.unit_cell().atom(atom_id).type().lo_descriptor_hub(0).l + 1;
238 
239  // compute the derivatives of the atomic wave functions
240  // |phi_m^J> (J = atom_id) compared to a displacement of atom J.
241 
242  kp.compute_gradient_wave_functions(phi, this->offset_[atom_id], lmax_at, phitmp, this->offset_[atom_id],
243  dir);
244 
245  if (ctx_.processing_unit() == device_t::GPU) {
246  phitmp.copy_to(spin_range(0), memory_t::device, 0, this->number_of_hubbard_orbitals());
247  }
248 
249  /* now apply S to |d\phi> */
250  sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(0), 0,
251  this->number_of_hubbard_orbitals(), kp.beta_projectors(),
252  phitmp, &q_op, dphi);
253  }
254 
255 
256  if (ctx_.cfg().hubbard().orthogonalize()) {
257  // when we orthogonalize the wave functions the derivatives vs atomic displement will contain three
258  // terms
259 
260  // d(S O^(-1/2) |\phi>) = dS O^(-1/2) | phi> + S (d O^(-1/2)) |
261  // \phi> + S O^(-1/2) |d \phi>.
262  //
263  // The results of the previous step are used to compute d O /dr
264  // which is used afterwards to compute the operator (d O^(-1/2)).
265  //
266  // |phitmp> contains the |d\phi>, dphi at the end contains the
267  // result of the expression above.
268 
269  /* needed to compute DO/dr see Eq.33 of Ref.2 */
270  overlap__.zero(memory_t::host);
271  overlap__.zero(memory_t::device);
272  // compute <dphi^atom | S | phi^all>. Eq.33 of Ref[2]
273  inner(ctx_.spla_context(), sddk::spin_range(0), dphi, 0, this->number_of_hubbard_orbitals(), phi, 0,
274  this->number_of_hubbard_orbitals(), overlap__, 0, 0);
275  }
276 
277  // compute (d S/ d R_K) |phi> and add to dphi. It is Eq.18 of Ref PRB 102, 235159 (2020)
278  apply_dS(kp, q_op, bp_grad, atom_id, dir, phi, dphi);
279 
280  // dphi comtains S |dphi^atom> + dS^atom |phi>
281  if (ctx_.cfg().hubbard().orthogonalize()) {
282 
283  // we first compute the dO/dr
284 
285  // compute <phi | S | dphi> + <phi | dS / dr| phi>. See Eq.33 of Ref[2]
286  inner(ctx_.spla_context(), sddk::spin_range(0), phi, 0, this->number_of_hubbard_orbitals(), dphi, 0,
287  this->number_of_hubbard_orbitals(), overlap__, 0, 0);
288 
289  // we have the derivative of the transformation matrix diagonalizing the overlap matrix. It is Eq.33
290 
291  // Now compute the derivative of the operator O^(-1/2)
292  // first compute the (U^dagger dO/dr U) from Eq.32
293 
294  linalg(la).gemm('N', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
295  this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(),
296  overlap__.at(mt), overlap__.ld(), U__.at(mt), U__.ld(),
297  &linalg_const<double_complex>::zero(), d_O_.at(mt), d_O_.ld());
298 
299  linalg(la).gemm('C', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
300  this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(), U__.at(mt),
301  U__.ld(), d_O_.at(mt), d_O_.ld(), &linalg_const<double_complex>::zero(),
302  overlap__.at(mt), overlap__.ld());
303 
304  if (ctx_.processing_unit() == device_t::GPU) {
305  overlap__.copy_to(memory_t::host);
306  U__.copy_to(memory_t::host);
307  }
308 
309  /* Eq.32 is a double dgemm product although not written explicitly that way */
310  for (int m1 = 0; m1 < this->number_of_hubbard_orbitals(); m1++) {
311  for (int m2 = 0; m2 < this->number_of_hubbard_orbitals(); m2++) {
312  overlap__(m1, m2) *=
313  1.0 / (eigenvalues[m1] / sqrt(eigenvalues[m2]) + eigenvalues[m2] / sqrt(eigenvalues[m1]));
314  }
315  }
316 
317  if (ctx_.processing_unit() == device_t::GPU) {
318  overlap__.copy_to(memory_t::device);
319  }
320 
321  // (d O / dr) * U^dagger
322  linalg(la).gemm('N', 'C', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
323  this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(),
324  overlap__.at(mt), overlap__.ld(), U__.at(mt), U__.ld(),
325  &linalg_const<double_complex>::zero(), d_O_.at(mt), d_O_.ld());
326 
327  // U (d O / dr * U^dagger)
328  linalg(la).gemm('N', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
329  this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(), U__.at(mt),
330  U__.ld(), d_O_.at(mt), d_O_.ld(), &linalg_const<double_complex>::zero(),
331  overlap__.at(mt), overlap__.ld());
332 
333  // apply d O^(-1/2) on the original phi
334  dphi.pw_coeffs(0).prime().zero(memory_t::host);
335  dphi.pw_coeffs(0).prime().zero(memory_t::device);
336 
337  transform<double_complex>(ctx_.spla_context(), 0, phi, 0, this->number_of_hubbard_orbitals(), overlap__,
338  0, 0, dphi, 0, this->number_of_hubbard_orbitals());
339 
340  // dphi = d O^(-1/2) | phi>
341 
342  O__.zero(memory_t::host);
343 
344  /* compute the orthogonalization matrix. It is (Eq.21)
345 
346  U (1/\sqrt(eigenvalues)) U^\dagger
347 
348  */
349  for (int l = 0; l < this->number_of_hubbard_orbitals(); l++) {
350  for (int m = 0; m < this->number_of_hubbard_orbitals(); m++) {
351  for (int n = 0; n < this->number_of_hubbard_orbitals(); n++) {
352  O__(n, m) += 1.0 / sqrt(eigenvalues[l]) * U__(n, l) * std::conj(U__(m, l));
353  }
354  }
355  }
356 
357  if (ctx_.processing_unit() == device_t::GPU) {
358  O__.copy_to(memory_t::device);
359  }
360 
361  transform<double_complex>(ctx_.spla_context(), 0, phitmp, // <- contains the derivatives of |phi>
362  0, this->number_of_hubbard_orbitals(), O__, 0, 0, dphi, 0,
363  this->number_of_hubbard_orbitals());
364 
365  // dphi now contains O^(-1/2) |d\phi> + d O^(-1/2) | \phi>
366 
367  phitmp.pw_coeffs(0).prime().zero(memory_t::host);
368  phitmp.pw_coeffs(0).prime().zero(memory_t::device);
369 
370  // apply S on (d O^(-1/2)) |phi> + O^(-1/2) | d phi> and store it to phitmp
371  sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(0), 0,
372  this->number_of_hubbard_orbitals(), kp.beta_projectors(), dphi,
373  &q_op, phitmp);
374 
375  // final step : compute the (dS) | (O^{-1/2} | phi>) = (dS) |hubbard>.
376 
377  /* transform on the wave functions. compute the second right term of Eq.16 */
378  dphi.pw_coeffs(0).prime().zero(memory_t::host);
379  dphi.pw_coeffs(0).prime().zero(memory_t::device);
380 
381  transform<double_complex>(ctx_.spla_context(), 0, phi, 0, this->number_of_hubbard_orbitals(), O__, 0, 0,
382  dphi, 0, this->number_of_hubbard_orbitals());
383 
384  apply_dS(kp, q_op, bp_grad, atom_id, dir, dphi, phitmp);
385 
386  // copy phitmp to dphi
387  dphi.copy_from(ctx_.processing_unit(), this->number_of_hubbard_orbitals(), phitmp, 0, 0, 0, 0);
388  }
389 
390  // d\phi now contains (\partial_{R^K} S) |\phi^hubbard> + S d|\phi^hubbard>.
391  compute_occupancies(kp, phi_s_psi, dphi, dn_tmp, dir);
392  } // direction x, y, z
393 
394  /* use a memcpy here */
395  std::memcpy(dn__.at(memory_t::host, 0, 0, 0, 0, 0, atom_id), dn_tmp.at(memory_t::host),
396  sizeof(double_complex) * dn_tmp.size());
397  } // atom_id
398 
399  if (ctx_.processing_unit() == device_t::GPU) {
400  phi.deallocate(spin_range(0), memory_t::device);
401  phitmp.deallocate(spin_range(0), memory_t::device);
402  dphi.deallocate(spin_range(0), memory_t::device);
403  dn_tmp.deallocate(memory_t::device);
404  if (ctx_.cfg().hubbard().orthogonalize()) {
405  overlap__.deallocate(memory_t::device);
406  d_O_.deallocate(memory_t::device);
407  O__.deallocate(memory_t::device);
408  U__.deallocate(memory_t::device);
409  }
410  kp.spinor_wave_functions().deallocate(spin_range(ctx_.num_spins()), memory_t::device);
411  }
412 }

414 void
415 Hubbard::compute_occupancies_stress_derivatives(K_point& kp__, Q_operator& q_op__, mdarray<double_complex, 5>& dn__)
416 {
417  PROFILE("sirius::Hubbard::compute_occupancies_stress_derivatives");
418 
419  /* this is the original atomic wave functions without the operator S applied */
420  auto& phi = kp__.atomic_wave_functions_hub();
421 
422  /*
423  dphi contains this
424 
425  \f[
426  \left| d \phi\right> = \partial_{\mu\nu} (S \left| \phi \right>)
427  ]
428 
429  */
430  Wave_functions dphi(kp__.gkvec_partition(), phi.num_wf(), ctx_.preferred_memory_t(), 1);
431 
432  Wave_functions phitmp(kp__.gkvec_partition(), phi.num_wf(), ctx_.preferred_memory_t(), 1);
433 
434  Beta_projectors_strain_deriv bp_strain_deriv(ctx_, kp__.gkvec(), kp__.igk_loc());
435 
436  /* maximum number of occupied bands */
437  int nbnd = (ctx_.num_mag_dims() == 1) ? std::max(kp__.num_occupied_bands(0), kp__.num_occupied_bands(1))
438  : kp__.num_occupied_bands(0);
439 
440  bool augment = ctx_.unit_cell().augment();
441 
442  const int lmax = ctx_.unit_cell().lmax();
443  const int lmmax = utils::lmmax(lmax);
444 
445  sddk::mdarray<double, 2> rlm_g(lmmax, kp__.num_gkvec_loc());
446  sddk::mdarray<double, 3> rlm_dg(lmmax, 3, kp__.num_gkvec_loc());
447 
448  /* overlap between psi_nk and phi_ik <psi|S|phi> */
449  dmatrix<double_complex> psi_s_phi(nbnd, phi.num_wf() * ctx_.num_spinors());
450 
451  /* initialize the beta projectors and derivatives */
452  bp_strain_deriv.prepare();
453 
454  auto sr = spin_range(ctx_.num_spins() == 2 ? 2 : 0);
455  kp__.spinor_wave_functions().prepare(sr, true);
456  kp__.atomic_wave_functions_S_hub().prepare(sr, true);
457  kp__.atomic_wave_functions_hub().prepare(sr, true);
458 
459  if (ctx_.processing_unit() == device_t::GPU) {
460  psi_s_phi.allocate(memory_t::device);
461 
462  dphi.allocate(spin_range(0), memory_t::device);
463 
464  phitmp.allocate(spin_range(0), memory_t::device);
465  }
466 
467  psi_s_phi.zero(memory_t::device);
468  psi_s_phi.zero(memory_t::host);
469  /* compute <psi_nk | S | phi_mk>
470  * treat non-magnetic, collinear and non-collinear cases;
471  * in collinear case psi_s_phi contains two blocks for up- and dn- spin channels */
472  for (int is = 0; is < ctx_.num_spinors(); is++) {
473  auto sr = ctx_.num_mag_dims() == 3 ? spin_range(2) : spin_range(is);
474 
475  inner(ctx_.spla_context(), sr, kp__.spinor_wave_functions(), 0, kp__.num_occupied_bands(is),
476  kp__.atomic_wave_functions_S_hub(), 0, kp__.atomic_wave_functions_S_hub().num_wf(), psi_s_phi, 0,
477  is * kp__.atomic_wave_functions_S_hub().num_wf());
478  }
479 
480  /* array of real spherical harmonics and derivatives for each G-vector */
481  #pragma omp parallel for schedule(static)
482  for (int igkloc = 0; igkloc < kp__.num_gkvec_loc(); igkloc++) {
483  /* gvs = {r, theta, phi} */
484  auto gvc = kp__.gkvec().gkvec_cart<index_domain_t::local>(igkloc);
485  auto rtp = SHT::spherical_coordinates(gvc);
486 
487  sf::spherical_harmonics(lmax, rtp[1], rtp[2], &rlm_g(0, igkloc));
488  sddk::mdarray<double, 2> rlm_dg_tmp(&rlm_dg(0, 0, igkloc), lmmax, 3);
489  sf::dRlm_dr(lmax, gvc, rlm_dg_tmp);
490  }
491 
492  /* the expressions below assume that the hubbard wave-functions are actually non-orthogonalised,
493  spin-independent atomic orbiatals */
494 
495  for (int nu = 0; nu < 3; nu++) {
496  for (int mu = 0; mu < 3; mu++) {
497 
498  // compute the derivatives of all hubbard wave functions
499  // |phi_m^J> compared to the strain
500 
501  /* Compute derivatives of Hubbard atomic functions w.r.t. lattice strain:
502  * - the functions indexing is compatible with K_point::generate_atomic_wave_functions();
503  * - only non-orthogonal atomic orbitals are handled
504  * - S operator is not applied
505  */
506  wavefunctions_strain_deriv(kp__, phitmp, rlm_g, rlm_dg, nu, mu);
507 
508  ///* ths phitmp functions are non-magnetic at the moment, so copy spin-up channel to spin dn;
509  // * this is done in order to have a general case of spin-dependent Hubbard orbitals */
510  // if (ctx_.num_spins() == 2) {
511  // phitmp.copy_from(device_t::CPU, phitmp.num_wf(), phitmp, 0, 0, 1, (ctx_.num_mag_dims() == 3) ?
512  // phitmp.num_wf() : 0);
513  //}
514 
515  /* Just to keep in mind:
516  *
517  * - kp.hubbard_wave_functions() are the hubbard functions with S applied
518  * - kp.atomic_wave_functions_S_hub() are the non-orthogonal (initial) atomic functions with S applied
519  * - kp.atomic_wave_functions_hub() are the non-orthogonal (initial) atomic functions
520  * - both sets are spin-dependent, even if the orbitals have no spin label
521  * - atom_type.indexr_hub() is the new index of Hubbard radial functions
522  * - atom_type.indexb_hub() is the new index of Hubbard orbitals (radial function times a spherical
523  * harmonic)
524  * - wavefunctions_strain_deriv() and K_point::generate_atomic_wave_functions() work with the new indices
525  */
526 
527  if (ctx_.processing_unit() == device_t::GPU) {
528  phitmp.copy_to(spin_range(0), memory_t::device, 0, this->number_of_hubbard_orbitals());
529  }
530 
531  /* dphi is the strain derivative of the hubbard orbitals (with S applied). Derivation imply this
532 
533  d(S phi) = (dS) phi + S (d\phi) = (d\phi) - \sum_{ij} Q_{ij} |beta_i><beta_j| d(phi) -
534  \sum_{ij} Q_{ij} |d beta_i><beta_j|phi> - \sum_{ij} Q_{ij} |beta_i><d beta_j|phi>
535 
536  dphi contains the full expression phitmp contains (d phi).
537  */
538 
539  dphi.copy_from(ctx_.processing_unit(), this->number_of_hubbard_orbitals(), phitmp, 0, 0, 0, 0);
540 
541  // dphi = |phitmp> - \sum_{ij} Q_{ij} | beta_i><beta_j|phitmp> + |d beta_i><beta_j|phi> + |beta_i><d
542  // beta_j|phi> dphi = (1 - \sum_{ij} Q_{ij} | beta_i><beta_j|) |phitmp> - (sum_{ij} Q_{ij} |d
543  // beta_i><beta_j| + |beta_i><d beta_j|) | phi>
544 
545  if (!ctx_.full_potential() && augment) {
546  for (int i = 0; i < kp__.beta_projectors().num_chunks(); i++) {
547  /* generate beta-projectors for a block of atoms */
548  kp__.beta_projectors().generate(i);
549  /* generate derived beta-projectors for a block of atoms */
550  bp_strain_deriv.generate(i, 3 * nu + mu);
551 
552  /* basically apply the second term of the S projector to
553  * phitmp = (strain derivatives of the original atomic
554  * orbitals )*/
555  {
556  /* < beta | dphi > */
557  auto beta_dphi = kp__.beta_projectors().inner<double_complex>(
558  i,
559  phitmp, // contains the strain derivative of the hubbard orbitals.
560  0, 0, this->number_of_hubbard_orbitals());
561  /* apply Q operator (diagonal in spin) */
562 
563  // compute Q_{ij}<beta_i | dphi> |beta_j> and add it to dphi
564  q_op__.apply(i, 0, dphi, 0, this->number_of_hubbard_orbitals(), kp__.beta_projectors(),
565  beta_dphi);
566  }
567 
568  {
569  /* <d(beta) | phi> */
570  auto dbeta_phi =
571  bp_strain_deriv.inner<double_complex>(i, phi, 0, 0, this->number_of_hubbard_orbitals());
572  /* apply Q operator (diagonal in spin) */
573 
574  // compute <d (beta) | phi> |beta> and add it to dphi
575  q_op__.apply(i, 0, dphi, 0, this->number_of_hubbard_orbitals(), kp__.beta_projectors(),
576  dbeta_phi);
577  }
578 
579  {
580  // <beta|phi> |d beta>
581  auto beta_phi = kp__.beta_projectors().inner<double_complex>(
582  i, phi, 0, 0, this->number_of_hubbard_orbitals());
583  /* apply Q operator (diagonal in spin) */
584  q_op__.apply(i, 0, dphi, 0, this->number_of_hubbard_orbitals(), bp_strain_deriv, beta_phi);
585  }
586  }
587  }
588  compute_occupancies(kp__, psi_s_phi, dphi, dn__, 3 * nu + mu);
589  }
590  }
591 
592  kp__.spinor_wave_functions().dismiss(sr, false);
593  kp__.atomic_wave_functions_S_hub().dismiss(sr, false);
594  kp__.atomic_wave_functions_hub().dismiss(sr, false);
595 }

597 void
598 Hubbard::wavefunctions_strain_deriv(K_point& kp__, Wave_functions& dphi__, sddk::mdarray<double, 2> const& rlm_g__,
599  sddk::mdarray<double, 3> const& rlm_dg__, int nu__, int mu__)
600 {
601  PROFILE("sirius::Hubbard::wavefunctions_strain_deriv");
602  #pragma omp parallel for schedule(static)
603  for (int igkloc = 0; igkloc < kp__.num_gkvec_loc(); igkloc++) {
604  /* global index of G+k vector */
605  const int igk = kp__.idxgk(igkloc);
606  /* Cartesian coordinats of G-vector */
607  auto gvc = kp__.gkvec().gkvec_cart<index_domain_t::local>(igkloc);
608  /* vs = {r, theta, phi} */
609  auto gvs = SHT::spherical_coordinates(gvc);
610 
611  std::vector<sddk::mdarray<double, 1>> ri_values(unit_cell_.num_atom_types());
612  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
613  ri_values[iat] = ctx_.hubbard_wf_ri().values(iat, gvs[0]);
614  }
615 
616  std::vector<sddk::mdarray<double, 1>> ridjl_values(unit_cell_.num_atom_types());
617  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
618  ridjl_values[iat] = ctx_.hubbard_wf_djl().values(iat, gvs[0]);
619  }
620 
621  const double p = (mu__ == nu__) ? 0.5 : 0.0;
622 
623  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
624  auto& atom_type = ctx_.unit_cell().atom(ia).type();
625  if (atom_type.hubbard_correction()) {
626  // TODO: this can be optimized, check k_point::generate_atomic_wavefunctions()
627  auto phase = twopi * dot(kp__.gkvec().gkvec(igk), unit_cell_.atom(ia).position());
628  auto phase_factor = std::exp(double_complex(0.0, phase));
629  int offset = this->offset_[ia];
630  for (int xi = 0; xi < atom_type.indexb_hub().size(); xi++) {
631  /* orbital quantum number of this atomic orbital */
632  int l = atom_type.indexb_hub().l(xi);
633  /* composite l,m index */
634  int lm = atom_type.indexb_hub().lm(xi);
635  /* index of the radial function */
636  int idxrf = atom_type.indexb_hub().idxrf(xi);
637 
638  auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
639 
640  /* case |g+k| = 0 */
641  if (gvs[0] < 1e-10) {
642  if (l == 0) {
643  auto d1 = ri_values[atom_type.id()][idxrf] * p * y00;
644 
645  dphi__.pw_coeffs(0).prime(igkloc, offset + xi) = -z * d1 * phase_factor;
646  } else {
647  dphi__.pw_coeffs(0).prime(igkloc, offset + xi) = 0.0;
648  }
649  } else {
650  auto d1 = ri_values[atom_type.id()][idxrf] *
651  (gvc[mu__] * rlm_dg__(lm, nu__, igkloc) + p * rlm_g__(lm, igkloc));
652  auto d2 =
653  ridjl_values[atom_type.id()][idxrf] * rlm_g__(lm, igkloc) * gvc[mu__] * gvc[nu__] / gvs[0];
654 
655  dphi__.pw_coeffs(0).prime(igkloc, offset + xi) = -z * (d1 + d2) * std::conj(phase_factor);
656  }
657  } // xi
658  }
659  }
660  }

663 void
664 Hubbard::compute_occupancies(K_point& kp__, dmatrix<double_complex>& psi_s_phi__, Wave_functions& dphi__,
665  sddk::mdarray<double_complex, 5>& dn__, const int index__)
666 {
667  PROFILE("sirius::Hubbard::compute_occupancies");
668 
669  /* maximum number of occupied bands */
670  int nbnd = (ctx_.num_mag_dims() == 1) ? std::max(kp__.num_occupied_bands(0), kp__.num_occupied_bands(1))
671  : kp__.num_occupied_bands(0);
672 
673  /* overlap between psi_{nk} and dphi */
674  dmatrix<double_complex> psi_s_dphi(nbnd, this->number_of_hubbard_orbitals(), ctx_.mem_pool(memory_t::host));
675 
676  dmatrix<double_complex> dm(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
677  ctx_.mem_pool(memory_t::host));
678 
679  auto la = linalg_t::none;
680  auto mt = memory_t::none;
681  switch (ctx_.processing_unit()) {
682  case device_t::CPU: {
683  la = linalg_t::blas;
684  mt = memory_t::host;
685  break;
686  }
687  case device_t::GPU: {
688  la = linalg_t::gpublas;
689  mt = memory_t::device;
690  dm.allocate(ctx_.mem_pool(memory_t::device));
691  psi_s_dphi.allocate(ctx_.mem_pool(memory_t::device));
692  break;
693  }
694  }
695 
696  psi_s_dphi.zero(memory_t::host);
697  psi_s_dphi.zero(memory_t::device);
698 
699  auto alpha = double_complex(kp__.weight(), 0.0);
700  for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
701  /* compute <psi_{ik}^{sigma}|S|dphi> */
702  /* dphi don't have a spin index; they are derived from scalar atomic orbitals */
703  inner(ctx_.spla_context(), spin_range(ispn), kp__.spinor_wave_functions(), 0, kp__.num_occupied_bands(ispn),
704  dphi__, 0, this->number_of_hubbard_orbitals(), psi_s_dphi, 0, 0);
705 
706  if (ctx_.processing_unit() == device_t::GPU) {
707  psi_s_dphi.copy_to(memory_t::host);
708  }
709 
710  for (int i = 0; i < this->number_of_hubbard_orbitals(); i++) {
711  for (int ibnd = 0; ibnd < kp__.num_occupied_bands(ispn); ibnd++) {
712  psi_s_dphi(ibnd, i) *= kp__.band_occupancy(ibnd, ispn);
713  }
714  }
715 
716  if (ctx_.processing_unit() == device_t::GPU) {
717  psi_s_dphi.copy_to(memory_t::device);
718  }
719 
720  linalg(la).gemm('C', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
721  kp__.num_occupied_bands(ispn), &alpha, psi_s_dphi.at(mt), psi_s_dphi.ld(),
722  psi_s_phi__.at(mt, 0, ispn * this->number_of_hubbard_orbitals()), psi_s_phi__.ld(),
723  &linalg_const<double_complex>::zero(), dm.at(mt), dm.ld());
724 
725  linalg(la).gemm('C', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
726  kp__.num_occupied_bands(ispn), &alpha,
727  psi_s_phi__.at(mt, 0, ispn * this->number_of_hubbard_orbitals()), psi_s_phi__.ld(),
728  psi_s_dphi.at(mt), psi_s_dphi.ld(), &linalg_const<double_complex>::one(), dm.at(mt), dm.ld());
729 
730  if (ctx_.processing_unit() == device_t::GPU) {
731  dm.copy_to(memory_t::host);
732  }
733 
734  #pragma omp parallel for schedule(static)
735  for (int ia1 = 0; ia1 < ctx_.unit_cell().num_atoms(); ++ia1) {
736  const auto& atom = ctx_.unit_cell().atom(ia1);
737  if (atom.type().hubbard_correction()) {
738  const int lmax_at = 2 * atom.type().lo_descriptor_hub(0).l + 1;
739  for (int m2 = 0; m2 < lmax_at; m2++) {
740  for (int m1 = 0; m1 < lmax_at; m1++) {
741  dn__(m1, m2, ispn, ia1, index__) = dm(this->offset_[ia1] + m1, this->offset_[ia1] + m2);
742  }
743  }
744  }
745  }
746  }
747 }
