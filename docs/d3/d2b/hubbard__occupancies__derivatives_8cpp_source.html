26// rotationally invariant one. It can not be used for LDA+U+SO either

28// this code is based on two papers :

30// - PRB 84, 161102(R) (2011) : https://journals.aps.org/prb/abstract/10.1103/PhysRevB.84.161102

40/* compute this |dphi> = dS | phi> + |dphi>, where the derivative is taken
41 * compared to atom_id displacement. we can also use lambda */
42void
43Hubbard::apply_dS(K_point<double>& kp, Q_operator<double>& q_op, Beta_projectors_gradient<double>& bp_grad, const int atom_id,
44 const int dir, Wave_functions<double>& phi, Wave_functions<double>& dphi)
45{
46 // compute d S/ dr^I_a |phi> and add to dphi
47 if (!ctx_.full_potential() && ctx_.unit_cell().augment()) {
48 // it is equal to
49 // \sum Q^I_ij <d \beta^I_i|phi> |\beta^I_j> + < \beta^I_i|phi> |d\beta^I_j>
50 for (int chunk__ = 0; chunk__ < kp.beta_projectors().num_chunks(); chunk__++) {
51 for (int i = 0; i < kp.beta_projectors().chunk(chunk__).num_atoms_; i++) {
52 // need to find the right atom in the chunks.
53 if (kp.beta_projectors().chunk(chunk__).desc_(static_cast<int>(beta_desc_idx::ia), i) == atom_id) {
54 kp.beta_projectors().generate(chunk__);
55 bp_grad.generate(chunk__, dir);
56
57 // compute Q_ij <\beta_i|\phi> |d \beta_j> and add it to d\phi
58 {
59 /* <beta | phi> for this chunk */
60 auto beta_phi = kp.beta_projectors().inner<double_complex>(chunk__, phi, 0, 0,
61 this->number_of_hubbard_orbitals());
62 q_op.apply(chunk__, i, 0, dphi, 0, this->number_of_hubbard_orbitals(), bp_grad, beta_phi);
63 }
64
65 // compute Q_ij <d \beta_i|\phi> |\beta_j> and add it to d\phi
66 {
67 /* <dbeta | phi> for this chunk */
68 auto dbeta_phi =
69 bp_grad.inner<double_complex>(chunk__, phi, 0, 0, this->number_of_hubbard_orbitals());
70
71 /* apply Q operator (diagonal in spin) */
72 /* Effectively compute Q_ij <d beta_i| phi> |beta_j> and add it dphi */
73 q_op.apply(chunk__, i, 0, dphi, 0, this->number_of_hubbard_orbitals(), kp.beta_projectors(),
74 dbeta_phi);
75 }
76 }
77 }
78 }
79 }
80}

82void
83Hubbard::compute_occupancies_derivatives(K_point<double>& kp, Q_operator<double>& q_op, sddk::mdarray<double_complex, 6>& dn__)
84{
85 PROFILE("sirius::Hubbard::compute_occupancies_derivatives");
86
87
88 auto la = linalg_t::none;
89 auto mt = memory_t::none;
90 switch (ctx_.processing_unit()) {
91 case device_t::CPU: {
92 la = linalg_t::blas;
93 mt = memory_t::host;
94 break;
95 }
96 case device_t::GPU: {
97 la = linalg_t::gpublas;
98 mt = memory_t::device;
99 break;
100 }
101 default:
102 break;
103 }
104
105 dn__.zero(memory_t::host);
106 // check if we have a norm conserving pseudo potential only. Only
107 // derivatives of the atomic wave functions are needed.
108 auto& phi = kp.atomic_wave_functions_hub();
109
110 Beta_projectors_gradient<double> bp_grad(ctx_, kp.gkvec(), kp.igk_loc(), kp.beta_projectors());
111 bp_grad.prepare();
112
113 /*
114 Compute the derivatives of the occupancies in two cases.
115
116 - the atom is pp norm conserving or
117
118 - the atom is ppus (in that case the derivative of the S operator gives a
119 non zero contribution)
120 */
121
122 /* temporary wave functions */
123 Wave_functions<double> dphi(kp.gkvec_partition(), this->number_of_hubbard_orbitals(), ctx_.preferred_memory_t(), 1);
124 /* temporary wave functions */
125 Wave_functions<double> phitmp(kp.gkvec_partition(), this->number_of_hubbard_orbitals(), ctx_.preferred_memory_t(), 1);
126
127 int HowManyBands = kp.num_occupied_bands(0);
128 if (ctx_.num_spins() == 2) {
129 HowManyBands = std::max(kp.num_occupied_bands(1), kp.num_occupied_bands(0));
130 }
131
132 dmatrix<double_complex> phi_s_psi(HowManyBands, this->number_of_hubbard_orbitals() * ctx_.num_spins());
133
134 /* <atomic_orbitals | S | atomic_orbitals> */
135 dmatrix<double_complex> overlap__(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
136
137 std::vector<double> eigenvalues(this->number_of_hubbard_orbitals());
138
139 /* transformation matrix going from overlap to diagonal form */
140 dmatrix<double_complex> U__(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
141
142 /* Matrix orthogonalizing the wave functions set */
143 dmatrix<double_complex> O__(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
144
145 /* derivative of the Matrix orthogonalizing the wave functions set */
146 dmatrix<double_complex> d_O_(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals());
147
148 mdarray<double_complex, 5> dn_tmp(max_number_of_orbitals_per_atom(), max_number_of_orbitals_per_atom(),
149 ctx_.num_spins(), ctx_.unit_cell().num_atoms(), 3);
150
151 if (ctx_.processing_unit() == device_t::GPU) {
152 dn_tmp.allocate(memory_t::device);
153 if (ctx_.cfg().hubbard().orthogonalize()) {
154 overlap__.allocate(memory_t::device);
155 d_O_.allocate(memory_t::device);
156 O__.allocate(memory_t::device);
157 U__.allocate(memory_t::device);
158 }
159 /* allocation of the overlap matrices on GPU */
160 phi_s_psi.allocate(memory_t::device);
161 // dphi_s_psi.allocate(memory_t::device);
162
163 /* wave functions */
164 phitmp.allocate(spin_range(0), memory_t::device);
165 phi.allocate(spin_range(ctx_.num_spins()), memory_t::device);
166 dphi.allocate(spin_range(0), memory_t::device);
167
168 kp.wave_functions_S_hub().allocate(spin_range(ctx_.num_spins()), memory_t::device);
169 kp.wave_functions_S_hub().copy_to(spin_range(ctx_.num_spins()), memory_t::device, 0, this->number_of_hubbard_orbitals());
170
171 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
172 phi.copy_to(spin_range(ispn), memory_t::device, 0, this->number_of_hubbard_orbitals());
173 }
174
175 kp.spinor_wave_functions().allocate(spin_range(ctx_.num_spins()), memory_t::device);
176 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
177 kp.spinor_wave_functions().copy_to(spin_range(ispn), memory_t::device, 0, kp.num_occupied_bands(ispn));
178 }
179 }
180
181 /* compute the overlap matrix and diagonalize it */
182 if (ctx_.cfg().hubbard().orthogonalize()) {
183 overlap__.zero(memory_t::host);
184 overlap__.zero(memory_t::device);
185 dphi.pw_coeffs(0).prime().zero(memory_t::host);
186 dphi.pw_coeffs(0).prime().zero(memory_t::device);
187 /* compute S | phi> where |phi> are the atomic orbitals */
188 sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(0), 0,
189 this->number_of_hubbard_orbitals(), kp.beta_projectors(), phi, &q_op,
190 dphi);
191 inner(ctx_.spla_context(), sddk::spin_range(0), phi, 0, this->number_of_hubbard_orbitals(), dphi, 0,
192 this->number_of_hubbard_orbitals(), overlap__, 0, 0);
193
194 if (ctx_.processing_unit() == device_t::GPU) {
195 overlap__.copy_to(memory_t::host);
196 }
197 auto ev_solver = Eigensolver_factory("lapack", nullptr);
198
199 ev_solver->solve(this->number_of_hubbard_orbitals(), overlap__, eigenvalues.data(), U__);
200
201 if (ctx_.processing_unit() == device_t::GPU) {
202 U__.copy_to(memory_t::device);
203 }
204
205 }
206
207 phi_s_psi.zero(memory_t::host);
208 phi_s_psi.zero(memory_t::device);
209
210 /* compute <phi (O)^(-1/2)| S | psi_{nk}> where |phi (O)^(-1/2)> are the hubbard wavefunctions */
211 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
212 inner(ctx_.spla_context(), spin_range(ispn), kp.spinor_wave_functions(), 0, kp.num_occupied_bands(ispn),
213 kp.wave_functions_S_hub(), 0, this->number_of_hubbard_orbitals(), phi_s_psi, 0,
214 ispn * this->number_of_hubbard_orbitals());
215 }
216
217 for (int atom_id = 0; atom_id < ctx_.unit_cell().num_atoms(); atom_id++) { // loop over the atom displacement.
218 dn_tmp.zero(memory_t::host);
219 dn_tmp.zero(memory_t::device);
220 for (int dir = 0; dir < 3; dir++) {
221 // reset dphi
222 dphi.pw_coeffs(0).prime().zero(memory_t::host);
223 dphi.pw_coeffs(0).prime().zero(memory_t::device);
224
225 phitmp.pw_coeffs(0).prime().zero(memory_t::host);
226 phitmp.pw_coeffs(0).prime().zero(memory_t::device);
227
228 // compute S|d\phi>. Will be zero if the atom has no hubbard
229 // correction
230 if (ctx_.unit_cell().atom(atom_id).type().hubbard_correction()) {
231
232 // atom atom_id has hubbard correction so we need to compute the
233 // derivatives of the hubbard orbitals associated to the atom
234 // atom_id, the derivatives of the others hubbard orbitals been
235 // zero compared to the displacement of atom atom_id
236
237 // compute the derivative of |phi> corresponding to the
238 // atom atom_id
239 const int lmax_at = 2 * ctx_.unit_cell().atom(atom_id).type().lo_descriptor_hub(0).l + 1;
240
241 // compute the derivatives of the atomic wave functions
242 // |phi_m^J> (J = atom_id) compared to a displacement of atom J.
243
244 kp.compute_gradient_wave_functions(phi, this->offset_[atom_id], lmax_at, phitmp, this->offset_[atom_id],
245 dir);
246
247 if (ctx_.processing_unit() == device_t::GPU) {
248 phitmp.copy_to(spin_range(0), memory_t::device, 0, this->number_of_hubbard_orbitals());
249 }
250
251 /* now apply S to |d\phi> */
252 sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(0), 0,
253 this->number_of_hubbard_orbitals(), kp.beta_projectors(),
254 phitmp, &q_op, dphi);
255 }
256
257
258 if (ctx_.cfg().hubbard().orthogonalize()) {
259 // when we orthogonalize the wave functions the derivatives vs atomic displement will contain three
260 // terms
261
262 // d(S O^(-1/2) |\phi>) = dS O^(-1/2) | phi> + S (d O^(-1/2)) |
263 // \phi> + S O^(-1/2) |d \phi>.
264 //
265 // The results of the previous step are used to compute d O /dr
266 // which is used afterwards to compute the operator (d O^(-1/2)).
267 //
268 // |phitmp> contains the |d\phi>, dphi at the end contains the
269 // result of the expression above.
270
271 /* needed to compute DO/dr see Eq.33 of Ref.2 */
272 overlap__.zero(memory_t::host);
273 overlap__.zero(memory_t::device);
274 // compute <dphi^atom | S | phi^all>. Eq.33 of Ref[2]
275 inner(ctx_.spla_context(), sddk::spin_range(0), dphi, 0, this->number_of_hubbard_orbitals(), phi, 0,
276 this->number_of_hubbard_orbitals(), overlap__, 0, 0);
277 }
278
279 // compute (d S/ d R_K) |phi> and add to dphi. It is Eq.18 of Ref PRB 102, 235159 (2020)
280 apply_dS(kp, q_op, bp_grad, atom_id, dir, phi, dphi);
281
282 // dphi comtains S |dphi^atom> + dS^atom |phi>
283 if (ctx_.cfg().hubbard().orthogonalize()) {
284
285 // we first compute the dO/dr
286
287 // compute <phi | S | dphi> + <phi | dS / dr| phi>. See Eq.33 of Ref[2]
288 inner(ctx_.spla_context(), sddk::spin_range(0), phi, 0, this->number_of_hubbard_orbitals(), dphi, 0,
289 this->number_of_hubbard_orbitals(), overlap__, 0, 0);
290
291 // we have the derivative of the transformation matrix diagonalizing the overlap matrix. It is Eq.33
292
293 // Now compute the derivative of the operator O^(-1/2)
294 // first compute the (U^dagger dO/dr U) from Eq.32
295
296 linalg(la).gemm('N', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
297 this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(),
298 overlap__.at(mt), overlap__.ld(), U__.at(mt), U__.ld(),
299 &linalg_const<double_complex>::zero(), d_O_.at(mt), d_O_.ld());
300
301 linalg(la).gemm('C', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
302 this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(), U__.at(mt),
303 U__.ld(), d_O_.at(mt), d_O_.ld(), &linalg_const<double_complex>::zero(),
304 overlap__.at(mt), overlap__.ld());
305
306 if (ctx_.processing_unit() == device_t::GPU) {
307 overlap__.copy_to(memory_t::host);
308 U__.copy_to(memory_t::host);
309 }
310
311 /* Eq.32 is a double dgemm product although not written explicitly that way */
312 for (int m1 = 0; m1 < this->number_of_hubbard_orbitals(); m1++) {
313 for (int m2 = 0; m2 < this->number_of_hubbard_orbitals(); m2++) {
314 overlap__(m1, m2) *=
315 1.0 / (eigenvalues[m1] / sqrt(eigenvalues[m2]) + eigenvalues[m2] / sqrt(eigenvalues[m1]));
316 }
317 }
318
319 if (ctx_.processing_unit() == device_t::GPU) {
320 overlap__.copy_to(memory_t::device);
321 }
322
323 // (d O / dr) * U^dagger
324 linalg(la).gemm('N', 'C', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
325 this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(),
326 overlap__.at(mt), overlap__.ld(), U__.at(mt), U__.ld(),
327 &linalg_const<double_complex>::zero(), d_O_.at(mt), d_O_.ld());
328
329 // U (d O / dr * U^dagger)
330 linalg(la).gemm('N', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
331 this->number_of_hubbard_orbitals(), &linalg_const<double_complex>::one(), U__.at(mt),
332 U__.ld(), d_O_.at(mt), d_O_.ld(), &linalg_const<double_complex>::zero(),
333 overlap__.at(mt), overlap__.ld());
334
335 // apply d O^(-1/2) on the original phi
336 dphi.pw_coeffs(0).prime().zero(memory_t::host);
337 dphi.pw_coeffs(0).prime().zero(memory_t::device);
338
339 transform<double_complex>(ctx_.spla_context(), 0, phi, 0, this->number_of_hubbard_orbitals(), overlap__,
340 0, 0, dphi, 0, this->number_of_hubbard_orbitals());
341
342 // dphi = d O^(-1/2) | phi>
343
344 O__.zero(memory_t::host);
345
346 /* compute the orthogonalization matrix. It is (Eq.21)
347
348 U (1/\sqrt(eigenvalues)) U^\dagger
349
350 */
351 for (int l = 0; l < this->number_of_hubbard_orbitals(); l++) {
352 for (int m = 0; m < this->number_of_hubbard_orbitals(); m++) {
353 for (int n = 0; n < this->number_of_hubbard_orbitals(); n++) {
354 O__(n, m) += 1.0 / sqrt(eigenvalues[l]) * U__(n, l) * std::conj(U__(m, l));
355 }
356 }
357 }
358
359 if (ctx_.processing_unit() == device_t::GPU) {
360 O__.copy_to(memory_t::device);
361 }
362
363 transform<double_complex>(ctx_.spla_context(), 0, phitmp, // <- contains the derivatives of |phi>
364 0, this->number_of_hubbard_orbitals(), O__, 0, 0, dphi, 0,
365 this->number_of_hubbard_orbitals());
366
367 // dphi now contains O^(-1/2) |d\phi> + d O^(-1/2) | \phi>
368
369 phitmp.pw_coeffs(0).prime().zero(memory_t::host);
370 phitmp.pw_coeffs(0).prime().zero(memory_t::device);
371
372 // apply S on (d O^(-1/2)) |phi> + O^(-1/2) | d phi> and store it to phitmp
373 sirius::apply_S_operator<double_complex>(ctx_.processing_unit(), spin_range(0), 0,
374 this->number_of_hubbard_orbitals(), kp.beta_projectors(), dphi,
375 &q_op, phitmp);
376
377 // final step : compute the (dS) | (O^{-1/2} | phi>) = (dS) |hubbard>.
378
379 /* transform on the wave functions. compute the second right term of Eq.16 */
380 dphi.pw_coeffs(0).prime().zero(memory_t::host);
381 dphi.pw_coeffs(0).prime().zero(memory_t::device);
382
383 transform<double_complex>(ctx_.spla_context(), 0, phi, 0, this->number_of_hubbard_orbitals(), O__, 0, 0,
384 dphi, 0, this->number_of_hubbard_orbitals());
385
386 apply_dS(kp, q_op, bp_grad, atom_id, dir, dphi, phitmp);
387
388 // copy phitmp to dphi
389 dphi.copy_from(ctx_.processing_unit(), this->number_of_hubbard_orbitals(), phitmp, 0, 0, 0, 0);
390 }
391
392 // d\phi now contains (\partial_{R^K} S) |\phi^hubbard> + S d|\phi^hubbard>.
393 compute_occupancies(kp, phi_s_psi, dphi, dn_tmp, dir);
394 } // direction x, y, z
395
396 /* use a memcpy here */
397 std::memcpy(dn__.at(memory_t::host, 0, 0, 0, 0, 0, atom_id), dn_tmp.at(memory_t::host),
398 sizeof(double_complex) * dn_tmp.size());
399 } // atom_id
400
401 if (ctx_.processing_unit() == device_t::GPU) {
402 phi.deallocate(spin_range(0), memory_t::device);
403 phitmp.deallocate(spin_range(0), memory_t::device);
404 dphi.deallocate(spin_range(0), memory_t::device);
405 dn_tmp.deallocate(memory_t::device);
406 if (ctx_.cfg().hubbard().orthogonalize()) {
407 overlap__.deallocate(memory_t::device);
408 d_O_.deallocate(memory_t::device);
409 O__.deallocate(memory_t::device);
410 U__.deallocate(memory_t::device);
411 }
412 kp.spinor_wave_functions().deallocate(spin_range(ctx_.num_spins()), memory_t::device);
413 }

416void
417Hubbard::compute_occupancies_stress_derivatives(K_point<double>& kp__, Q_operator<double>& q_op__, mdarray<double_complex, 5>& dn__)
418{
419 PROFILE("sirius::Hubbard::compute_occupancies_stress_derivatives");
420
421 /* this is the original atomic wave functions without the operator S applied */
422 auto& phi = kp__.atomic_wave_functions_hub();
423
424 /*
425 dphi contains this
426
427 \f[
428 \left| d \phi\right> = \partial_{\mu\nu} (S \left| \phi \right>)
429 ]
430
431 */
432 Wave_functions<double> dphi(kp__.gkvec_partition(), phi.num_wf(), ctx_.preferred_memory_t(), 1);
433
434 Wave_functions<double> phitmp(kp__.gkvec_partition(), phi.num_wf(), ctx_.preferred_memory_t(), 1);
435
436 Beta_projectors_strain_deriv<double> bp_strain_deriv(ctx_, kp__.gkvec(), kp__.igk_loc());
437
438 /* maximum number of occupied bands */
439 int nbnd = (ctx_.num_mag_dims() == 1) ? std::max(kp__.num_occupied_bands(0), kp__.num_occupied_bands(1))
440 : kp__.num_occupied_bands(0);
441
442 bool augment = ctx_.unit_cell().augment();
443
444 const int lmax = ctx_.unit_cell().lmax();
445 const int lmmax = utils::lmmax(lmax);
446
447 sddk::mdarray<double, 2> rlm_g(lmmax, kp__.num_gkvec_loc());
448 sddk::mdarray<double, 3> rlm_dg(lmmax, 3, kp__.num_gkvec_loc());
449
450 /* overlap between psi_nk and phi_ik <psi|S|phi> */
451 dmatrix<double_complex> psi_s_phi(nbnd, phi.num_wf() * ctx_.num_spinors());
452
453 /* initialize the beta projectors and derivatives */
454 bp_strain_deriv.prepare();
455
456 auto sr = spin_range(ctx_.num_spins() == 2 ? 2 : 0);
457 kp__.spinor_wave_functions().prepare(sr, true);
458 kp__.atomic_wave_functions_S_hub().prepare(sr, true);
459 kp__.atomic_wave_functions_hub().prepare(sr, true);
460
461 if (ctx_.processing_unit() == device_t::GPU) {
462 psi_s_phi.allocate(memory_t::device);
463
464 dphi.allocate(spin_range(0), memory_t::device);
465
466 phitmp.allocate(spin_range(0), memory_t::device);
467 }
468
469 psi_s_phi.zero(memory_t::device);
470 psi_s_phi.zero(memory_t::host);
471 /* compute <psi_nk | S | phi_mk>
472 * treat non-magnetic, collinear and non-collinear cases;
473 * in collinear case psi_s_phi contains two blocks for up- and dn- spin channels */
474 for (int is = 0; is < ctx_.num_spinors(); is++) {
475 auto sr = ctx_.num_mag_dims() == 3 ? spin_range(2) : spin_range(is);
476
477 inner(ctx_.spla_context(), sr, kp__.spinor_wave_functions(), 0, kp__.num_occupied_bands(is),
478 kp__.atomic_wave_functions_S_hub(), 0, kp__.atomic_wave_functions_S_hub().num_wf(), psi_s_phi, 0,
479 is * kp__.atomic_wave_functions_S_hub().num_wf());
480 }
481
482 /* array of real spherical harmonics and derivatives for each G-vector */
483 #pragma omp parallel for schedule(static)
484 for (int igkloc = 0; igkloc < kp__.num_gkvec_loc(); igkloc++) {
485 /* gvs = {r, theta, phi} */
486 auto gvc = kp__.gkvec().gkvec_cart<index_domain_t::local>(igkloc);
487 auto rtp = SHT::spherical_coordinates(gvc);
488
489 sf::spherical_harmonics(lmax, rtp[1], rtp[2], &rlm_g(0, igkloc));
490 sddk::mdarray<double, 2> rlm_dg_tmp(&rlm_dg(0, 0, igkloc), lmmax, 3);
491 sf::dRlm_dr(lmax, gvc, rlm_dg_tmp);
492 }
493
494 /* the expressions below assume that the hubbard wave-functions are actually non-orthogonalised,
495 spin-independent atomic orbiatals */
496
497 for (int nu = 0; nu < 3; nu++) {
498 for (int mu = 0; mu < 3; mu++) {
499
500 // compute the derivatives of all hubbard wave functions
501 // |phi_m^J> compared to the strain
502
503 /* Compute derivatives of Hubbard atomic functions w.r.t. lattice strain:
504 * - the functions indexing is compatible with K_point::generate_atomic_wave_functions();
505 * - only non-orthogonal atomic orbitals are handled
506 * - S operator is not applied
507 */
508 wavefunctions_strain_deriv(kp__, phitmp, rlm_g, rlm_dg, nu, mu);
509
510 ///* ths phitmp functions are non-magnetic at the moment, so copy spin-up channel to spin dn;
511 // * this is done in order to have a general case of spin-dependent Hubbard orbitals */
512 // if (ctx_.num_spins() == 2) {
513 // phitmp.copy_from(device_t::CPU, phitmp.num_wf(), phitmp, 0, 0, 1, (ctx_.num_mag_dims() == 3) ?
514 // phitmp.num_wf() : 0);
515 //}
516
517 /* Just to keep in mind:
518 *
519 * - kp.hubbard_wave_functions() are the hubbard functions with S applied
520 * - kp.atomic_wave_functions_S_hub() are the non-orthogonal (initial) atomic functions with S applied
521 * - kp.atomic_wave_functions_hub() are the non-orthogonal (initial) atomic functions
522 * - both sets are spin-dependent, even if the orbitals have no spin label
523 * - atom_type.indexr_hub() is the new index of Hubbard radial functions
524 * - atom_type.indexb_hub() is the new index of Hubbard orbitals (radial function times a spherical
525 * harmonic)
526 * - wavefunctions_strain_deriv() and K_point::generate_atomic_wave_functions() work with the new indices
527 */
528
529 if (ctx_.processing_unit() == device_t::GPU) {
530 phitmp.copy_to(spin_range(0), memory_t::device, 0, this->number_of_hubbard_orbitals());
531 }
532
533 /* dphi is the strain derivative of the hubbard orbitals (with S applied). Derivation imply this
534
535 d(S phi) = (dS) phi + S (d\phi) = (d\phi) - \sum_{ij} Q_{ij} |beta_i><beta_j| d(phi) -
536 \sum_{ij} Q_{ij} |d beta_i><beta_j|phi> - \sum_{ij} Q_{ij} |beta_i><d beta_j|phi>
537
538 dphi contains the full expression phitmp contains (d phi).
539 */
540
541 dphi.copy_from(ctx_.processing_unit(), this->number_of_hubbard_orbitals(), phitmp, 0, 0, 0, 0);
542
543 // dphi = |phitmp> - \sum_{ij} Q_{ij} | beta_i><beta_j|phitmp> + |d beta_i><beta_j|phi> + |beta_i><d
544 // beta_j|phi> dphi = (1 - \sum_{ij} Q_{ij} | beta_i><beta_j|) |phitmp> - (sum_{ij} Q_{ij} |d
545 // beta_i><beta_j| + |beta_i><d beta_j|) | phi>
546
547 if (!ctx_.full_potential() && augment) {
548 for (int i = 0; i < kp__.beta_projectors().num_chunks(); i++) {
549 /* generate beta-projectors for a block of atoms */
550 kp__.beta_projectors().generate(i);
551 /* generate derived beta-projectors for a block of atoms */
552 bp_strain_deriv.generate(i, 3 * nu + mu);
553
554 /* basically apply the second term of the S projector to
555 * phitmp = (strain derivatives of the original atomic
556 * orbitals )*/
557 {
558 /* < beta | dphi > */
559 auto beta_dphi = kp__.beta_projectors().inner<double_complex>(
560 i,
561 phitmp, // contains the strain derivative of the hubbard orbitals.
562 0, 0, this->number_of_hubbard_orbitals());
563 /* apply Q operator (diagonal in spin) */
564
565 // compute Q_{ij}<beta_i | dphi> |beta_j> and add it to dphi
566 q_op__.apply(i, 0, dphi, 0, this->number_of_hubbard_orbitals(), kp__.beta_projectors(),
567 beta_dphi);
568 }
569
570 {
571 /* <d(beta) | phi> */
572 auto dbeta_phi =
573 bp_strain_deriv.inner<double_complex>(i, phi, 0, 0, this->number_of_hubbard_orbitals());
574 /* apply Q operator (diagonal in spin) */
575
576 // compute <d (beta) | phi> |beta> and add it to dphi
577 q_op__.apply(i, 0, dphi, 0, this->number_of_hubbard_orbitals(), kp__.beta_projectors(),
578 dbeta_phi);
579 }
580
581 {
582 // <beta|phi> |d beta>
583 auto beta_phi = kp__.beta_projectors().inner<double_complex>(
584 i, phi, 0, 0, this->number_of_hubbard_orbitals());
585 /* apply Q operator (diagonal in spin) */
586 q_op__.apply(i, 0, dphi, 0, this->number_of_hubbard_orbitals(), bp_strain_deriv, beta_phi);
587 }
588 }
589 }
590 compute_occupancies(kp__, psi_s_phi, dphi, dn__, 3 * nu + mu);
591 }
592 }
593
594 kp__.spinor_wave_functions().dismiss(sr, false);
595 kp__.atomic_wave_functions_S_hub().dismiss(sr, false);
596 kp__.atomic_wave_functions_hub().dismiss(sr, false);
597}

599void
600Hubbard::wavefunctions_strain_deriv(K_point<double>& kp__, Wave_functions<double>& dphi__, sddk::mdarray<double, 2> const& rlm_g__,
601 sddk::mdarray<double, 3> const& rlm_dg__, int nu__, int mu__)
602{
603 PROFILE("sirius::Hubbard::wavefunctions_strain_deriv");
604 #pragma omp parallel for schedule(static)
605 for (int igkloc = 0; igkloc < kp__.num_gkvec_loc(); igkloc++) {
606 /* global index of G+k vector */
607 const int igk = kp__.idxgk(igkloc);
608 /* Cartesian coordinats of G-vector */
609 auto gvc = kp__.gkvec().gkvec_cart<index_domain_t::local>(igkloc);
610 /* vs = {r, theta, phi} */
611 auto gvs = SHT::spherical_coordinates(gvc);
612
613 std::vector<sddk::mdarray<double, 1>> ri_values(unit_cell_.num_atom_types());
614 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
615 ri_values[iat] = ctx_.hubbard_wf_ri().values(iat, gvs[0]);
616 }
617
618 std::vector<sddk::mdarray<double, 1>> ridjl_values(unit_cell_.num_atom_types());
619 for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
620 ridjl_values[iat] = ctx_.hubbard_wf_djl().values(iat, gvs[0]);
621 }
622
623 const double p = (mu__ == nu__) ? 0.5 : 0.0;
624
625 for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
626 auto& atom_type = ctx_.unit_cell().atom(ia).type();
627 if (atom_type.hubbard_correction()) {
628 // TODO: this can be optimized, check k_point::generate_atomic_wavefunctions()
629 auto phase = twopi * dot(kp__.gkvec().gkvec(igk), unit_cell_.atom(ia).position());
630 auto phase_factor = std::exp(double_complex(0.0, phase));
631 int offset = this->offset_[ia];
632 for (int xi = 0; xi < atom_type.indexb_hub().size(); xi++) {
633 /* orbital quantum number of this atomic orbital */
634 int l = atom_type.indexb_hub().l(xi);
635 /* composite l,m index */
636 int lm = atom_type.indexb_hub().lm(xi);
637 /* index of the radial function */
638 int idxrf = atom_type.indexb_hub().idxrf(xi);
639
640 auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
641
642 /* case |g+k| = 0 */
643 if (gvs[0] < 1e-10) {
644 if (l == 0) {
645 auto d1 = ri_values[atom_type.id()][idxrf] * p * y00;
646
647 dphi__.pw_coeffs(0).prime(igkloc, offset + xi) = -z * d1 * phase_factor;
648 } else {
649 dphi__.pw_coeffs(0).prime(igkloc, offset + xi) = 0.0;
650 }
651 } else {
652 auto d1 = ri_values[atom_type.id()][idxrf] *
653 (gvc[mu__] * rlm_dg__(lm, nu__, igkloc) + p * rlm_g__(lm, igkloc));
654 auto d2 =
655 ridjl_values[atom_type.id()][idxrf] * rlm_g__(lm, igkloc) * gvc[mu__] * gvc[nu__] / gvs[0];
656
657 dphi__.pw_coeffs(0).prime(igkloc, offset + xi) = -z * (d1 + d2) * std::conj(phase_factor);
658 }
659 } // xi
660 }
661 }
662 }

665void
666Hubbard::compute_occupancies(K_point<double>& kp__, dmatrix<double_complex>& psi_s_phi__, Wave_functions<double>& dphi__,
667 sddk::mdarray<double_complex, 5>& dn__, const int index__)
668{
669 PROFILE("sirius::Hubbard::compute_occupancies");
670
671 /* maximum number of occupied bands */
672 int nbnd = (ctx_.num_mag_dims() == 1) ? std::max(kp__.num_occupied_bands(0), kp__.num_occupied_bands(1))
673 : kp__.num_occupied_bands(0);
674
675 /* overlap between psi_{nk} and dphi */
676 dmatrix<double_complex> psi_s_dphi(nbnd, this->number_of_hubbard_orbitals(), ctx_.mem_pool(memory_t::host));
677
678 dmatrix<double_complex> dm(this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
679 ctx_.mem_pool(memory_t::host));
680
681 auto la = linalg_t::none;
682 auto mt = memory_t::none;
683 switch (ctx_.processing_unit()) {
684 case device_t::CPU: {
685 la = linalg_t::blas;
686 mt = memory_t::host;
687 break;
688 }
689 case device_t::GPU: {
690 la = linalg_t::gpublas;
691 mt = memory_t::device;
692 dm.allocate(ctx_.mem_pool(memory_t::device));
693 psi_s_dphi.allocate(ctx_.mem_pool(memory_t::device));
694 break;
695 }
696 }
697
698 psi_s_dphi.zero(memory_t::host);
699 psi_s_dphi.zero(memory_t::device);
700
701 auto alpha = double_complex(kp__.weight(), 0.0);
702 for (int ispn = 0; ispn < ctx_.num_spins(); ispn++) {
703 /* compute <psi_{ik}^{sigma}|S|dphi> */
704 /* dphi don't have a spin index; they are derived from scalar atomic orbitals */
705 inner(ctx_.spla_context(), spin_range(ispn), kp__.spinor_wave_functions(), 0, kp__.num_occupied_bands(ispn),
706 dphi__, 0, this->number_of_hubbard_orbitals(), psi_s_dphi, 0, 0);
707
708 if (ctx_.processing_unit() == device_t::GPU) {
709 psi_s_dphi.copy_to(memory_t::host);
710 }
711
712 for (int i = 0; i < this->number_of_hubbard_orbitals(); i++) {
713 for (int ibnd = 0; ibnd < kp__.num_occupied_bands(ispn); ibnd++) {
714 psi_s_dphi(ibnd, i) *= kp__.band_occupancy(ibnd, ispn);
715 }
716 }
717
718 if (ctx_.processing_unit() == device_t::GPU) {
719 psi_s_dphi.copy_to(memory_t::device);
720 }
721
722 linalg(la).gemm('C', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
723 kp__.num_occupied_bands(ispn), &alpha, psi_s_dphi.at(mt), psi_s_dphi.ld(),
724 psi_s_phi__.at(mt, 0, ispn * this->number_of_hubbard_orbitals()), psi_s_phi__.ld(),
725 &linalg_const<double_complex>::zero(), dm.at(mt), dm.ld());
726
727 linalg(la).gemm('C', 'N', this->number_of_hubbard_orbitals(), this->number_of_hubbard_orbitals(),
728 kp__.num_occupied_bands(ispn), &alpha,
729 psi_s_phi__.at(mt, 0, ispn * this->number_of_hubbard_orbitals()), psi_s_phi__.ld(),
730 psi_s_dphi.at(mt), psi_s_dphi.ld(), &linalg_const<double_complex>::one(), dm.at(mt), dm.ld());
731
732 if (ctx_.processing_unit() == device_t::GPU) {
733 dm.copy_to(memory_t::host);
734 }
735
736 #pragma omp parallel for schedule(static)
737 for (int ia1 = 0; ia1 < ctx_.unit_cell().num_atoms(); ++ia1) {
738 const auto& atom = ctx_.unit_cell().atom(ia1);
739 if (atom.type().hubbard_correction()) {
740 const int lmax_at = 2 * atom.type().lo_descriptor_hub(0).l + 1;
741 for (int m2 = 0; m2 < lmax_at; m2++) {
742 for (int m1 = 0; m1 < lmax_at; m1++) {
743 dn__(m1, m2, ispn, ia1, index__) = dm(this->offset_[ia1] + m1, this->offset_[ia1] + m2);
744 }
745 }
746 }
747 }
748 }
749}
