129  /// Return sum of valence eigen-values.
130  double valence_eval_sum() const
131  {
132  double eval_sum{0};
133 
134  for (int ik = 0; ik < num_kpoints(); ik++) {
135  auto const& kp = kpoints_[ik];
136  double wk = kp->weight();
137  for (int j = 0; j < ctx_.num_bands(); j++) {
138  for (int ispn = 0; ispn < ctx_.num_spin_dims(); ispn++) {
139  eval_sum += wk * kp->band_energy(j, ispn) * kp->band_occupancy(j, ispn);
140  }
141  }
142  }
143 
144  return eval_sum;
145  }

147  /// Return maximum number of G+k vectors among all k-points.
148  int max_num_gkvec() const
149  {
150  int max_num_gkvec{0};
151  for (int ikloc = 0; ikloc < spl_num_kpoints_.local_size(); ikloc++) {
152  auto ik = spl_num_kpoints_[ikloc];
153  max_num_gkvec = std::max(max_num_gkvec, kpoints_[ik]->num_gkvec());
154  }
155  comm().allreduce<int, mpi_op_t::max>(&max_num_gkvec, 1);

239  /// Send G+k vectors of k-point jk to a given rank.
240  /** Other ranks receive an empty Gvec placeholder */
241  inline Gvec send_recv_gkvec(int jk__, int rank__)
242  {
243  /* rank in the k-point communicator */
244  int my_rank = comm().rank();
245 
246  /* rank that stores jk */
247  int jrank = spl_num_kpoints().local_rank(jk__);
248 
249  /* placeholder for G+k vectors of kpoint jk */
250  Gvec gkvec(ctx_.comm_band());
251 
252  /* if this rank stores the k-point, then send it */
253  if (jrank == my_rank) {
254  auto kp = kpoints_[jk__].get();
255  kp->gkvec().send_recv(comm(), jrank, rank__, gkvec);
256  }
257  /* this rank receives the k-point */
258  if (rank__ == my_rank) {
259  gkvec.send_recv(comm(), jrank, rank__, gkvec);
260  }
261  return gkvec;
262  }
