32 /// Unit step function is defined to be 1 in the interstitial and 0 inside muffin-tins.
33 /** Unit step function is constructed from it's plane-wave expansion coefficients which are computed
34  * analytically:
35  * \f[
36  * \Theta({\bf r}) = \sum_{\bf G} \Theta({\bf G}) e^{i{\bf Gr}},
37  * \f]
38  * where
39  * \f[
40  * \Theta({\bf G}) = \frac{1}{\Omega} \int \Theta({\bf r}) e^{-i{\bf Gr}} d{\bf r} =
41  * \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} d{\bf r} - \frac{1}{\Omega} \int_{MT} e^{-i{\bf Gr}} d{\bf r} =
42  * \delta_{\bf G, 0} - \sum_{\alpha} \frac{1}{\Omega} \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r}
43  * \f]
44  * Integral of a plane-wave over the muffin-tin volume is taken using the spherical expansion of the plane-wave
45  * around central point \f$ \tau_{\alpha} \f$:
46  * \f[
47  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} =
48  * e^{-i{\bf G\tau_{\alpha}}} \int_{MT_{\alpha}} 4\pi \sum_{\ell m} (-i)^{\ell} j_{\ell}(Gr)
49  * Y_{\ell m}(\hat {\bf G}) Y_{\ell m}^{*}(\hat {\bf r}) r^2 \sin \theta dr d\phi d\theta
50  * \f]
51  * In the above integral only \f$ \ell=m=0 \f$ term survives. So we have:
52  * \f[
53  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = 4\pi e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
54  * \f]
55  * where
56  * \f[
57  * \Theta(\alpha, G) = \int_{0}^{R_{\alpha}} \frac{\sin(Gr)}{Gr} r^2 dr =
58  * \left\{ \begin{array}{ll} \displaystyle R_{\alpha}^3 / 3 & G=0 \\
59  * \Big( \sin(GR_{\alpha}) - GR_{\alpha}\cos(GR_{\alpha}) \Big) / G^3 & G \ne 0 \end{array} \right.
60  * \f]
61  * are the so-called step function form factors. With this we have a final expression for the plane-wave coefficients
62  * of the unit step function:
63  * \f[
64  * \Theta({\bf G}) = \delta_{\bf G, 0} - \sum_{\alpha} \frac{4\pi}{\Omega} e^{-i{\bf G\tau_{\alpha}}}
65  * \Theta(\alpha, G)
66  * \f]
67  */
68 class Step_function
69 {
70  private:
71  /// Plane wave expansion coefficients of the step function.
72  std::vector<double_complex> step_function_pw_;
73 
74  /// Step function on the real-space grid.
75  std::vector<double> step_function_;
76 
77  public:
78  /// Constructor
79  Step_function(Simulation_context_base& ctx__)
80  {
81  PROFILE("sirius::Step_function::Step_function");
82 
83  if (ctx__.unit_cell().num_atoms() == 0) {
84  return;
85  }
86 
87  step_function_pw_.resize(ctx__.gvec().num_gvec());
88  step_function_.resize(ctx__.fft().local_size());
89 
90  auto ri = [&](int iat, double g) {
91  auto R = ctx__.unit_cell().atom_type(iat).mt_radius();
92  if (g < 1e-12) {
93  return std::pow(R, 3) / 3.0;
94  } else {
95  return (std::sin(g * R) - g * R * std::cos(g * R)) / std::pow(g, 3);
96  }
97  };
98 
99  auto f_pw = ctx__.make_periodic_function<index_domain_t::global>(ri);
100 
101  for (int ig = 0; ig < ctx__.gvec().num_gvec(); ig++) {
102  step_function_pw_[ig] = -f_pw[ig];
103  }
104  step_function_pw_[0] += 1.0;
105 
106  std::vector<double_complex> ftmp(ctx__.gvec_partition().gvec_count_fft());
107  for (int i = 0; i < ctx__.gvec_partition().gvec_count_fft(); i++) {
108  ftmp[i] = step_function_pw_[ctx__.gvec_partition().idx_gvec(i)];
109  }
110  ctx__.fft().transform<1>(ftmp.data());
111  ctx__.fft().output(&step_function_[0]);
112 
113  double vit{0};
114  for (int i = 0; i < ctx__.fft().local_size(); i++) {
115  vit += step_function_[i];
116  }
117  vit *= (ctx__.unit_cell().omega() / ctx__.fft().size());
118  ctx__.fft().comm().allreduce(&vit, 1);
119 
120  if (std::abs(vit - ctx__.unit_cell().volume_it()) > 1e-10) {
121  std::stringstream s;
122  s << "step function gives a wrong volume for IT region" << std::endl
123  << " difference with exact value : " << std::abs(vit - ctx__.unit_cell().volume_it());
124  WARNING(s);
125  }
126  if (ctx__.control().print_checksum_) {
127  double_complex z1 = mdarray<double_complex, 1>(&step_function_pw_[0], ctx__.gvec().num_gvec()).checksum();
128  double d1 = mdarray<double, 1>(&step_function_[0], ctx__.fft().local_size()).checksum();
129  ctx__.fft().comm().allreduce(&d1, 1);
130  if (ctx__.comm().rank() == 0) {
131  print_checksum("step_function", d1);
132  print_checksum("step_function_pw", z1);
133  }
134  }
135  }
136 
137  /// Return plane-wave coefficient of the step function.
138  inline double_complex const& theta_pw(int ig__) const
139  {
140  assert(ig__ >= 0 && ig__ < (int)step_function_pw_.size());
141  return step_function_pw_[ig__];
142  }
143 
144  /// Return the value of the step function for the grid point ir.
145  inline double theta_r(int ir__) const
146  {
147  assert(ir__ >= 0 && ir__ < (int)step_function_.size());
148  return step_function_[ir__];
149  }
150 };
