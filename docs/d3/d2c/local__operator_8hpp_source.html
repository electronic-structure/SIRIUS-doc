66 /// Representation of the local operator.
67 /** The following functionality is implementated:
68  * - application of the local part of Hamiltonian (kinetic + potential) to the wave-functions in the PP-PW case
69  * - application of the interstitial part of H and O in the case of FP-LAPW
70  * - application of the interstitial part of effective magnetic field to the first-variational functios
71  * - remapping of potential and unit-step functions from fine to coarse mesh of G-vectors
72  */
73 class Local_operator
74 {
75  private:
76  /// Common parameters.
77  Simulation_context const& ctx_;
78 
79  /// Coarse-grid FFT driver for this operator.
80  spfft::Transform& fft_coarse_;
81 
82  /// Distribution of the G-vectors for the FFT transformation.
83  sddk::Gvec_partition const& gvec_coarse_p_;
84 
85  /// Kinetic energy of G+k plane-waves.
86  sddk::mdarray<double, 1> pw_ekin_;
87 
88  /// Effective potential components and unit step function on a coarse FFT grid.
89  /** The following elements are stored in the array:
90  - V(r) + B_z(r) (in PP-PW case) or V(r) (in FP-LAPW case)
91  - V(r) - B_z(r) (in PP-PW case) or B_z(r) (in FP-LAPW case)
92  - B_x(r)
93  - B_y(r)
94  - Theta(r) (in FP-LAPW case)
95  - inverse of 1 + relative mass (needed for ZORA LAPW)
96  */
97  std::array<std::unique_ptr<Smooth_periodic_function<double>>, 6> veff_vec_;
98 
99  /// Temporary array to store [V*phi](G)
100  sddk::mdarray<double_complex, 1> vphi_;
101 
102  /// Temporary array to store psi_{up}(r).
103  /** The size of the array is equal to the size of FFT buffer. */
104  sddk::mdarray<double_complex, 1> buf_rg_;
105 
106  /// V(G=0) matrix elements.
107  double v0_[2];
108 
109  public:
110  /// Constructor.
111  /** Prepares k-point independent part of the local potential. If potential is provided, it is mapped to the
112  * coarse FFT grid, otherwise the constant potential is assumed for the debug and benchmarking purposes.
113  * In the case of GPU-enabled FFT driver all effective fields on the coarse grid are copied to the device
114  * and remain there until the local operator is destroyed.
115 
116  * \param [in] ctx Simulation context.
117  * \param [in] fft_coarse Explicit FFT driver for the coarse mesh.
118  * \param [in] gvec_coarse_p FFT-friendly G-vector distribution for the coarse mesh.
119  * \param [in] potential Effective potential and magnetic fields \f$ V_{eff}({\bf r}) \f$ and
120  * \f$ {\bf B}_{eff}({\bf r}) \f$ on the fine FFT grid.
121  */
122  Local_operator(Simulation_context const& ctx__,
123  spfft::Transform& fft_coarse__,
124  sddk::Gvec_partition const& gvec_coarse_p__,
125  Potential* potential__ = nullptr);

131  /// Apply local part of Hamiltonian to pseudopotential wave-functions.
132  /** \param [in] spfftk SpFFT transform object for G+k vectors.
133  * \param [in] gkvec_p FFT-friendly G+k vector partitioning.
134  * \param [in] spins Range of wave-function spins to which Hloc is applied.
135  * \param [in] phi Input wave-functions.
136  * \param [out] hphi Local hamiltonian applied to wave-function.
137  * \param [in] idx0 Starting index of wave-functions.
138  * \param [in] n Number of wave-functions to which H is applied.
139  *
140  * Spin range can take the following values:
141  * - [0, 0]: apply H_{uu} to the up- component of wave-functions
142  * - [1, 1]: apply H_{dd} to the dn- component of wave-functions
143  * - [0, 1]: apply full Hamiltonian to the spinor wave-functions
144  *
145  * Local Hamiltonian includes kinetic term and local part of potential.
146  */
147  void apply_h(spfft::Transform& spfftk__, sddk::Gvec_partition const& gkvec_p__, sddk::spin_range spins__,
148  sddk::Wave_functions& phi__, sddk::Wave_functions& hphi__, int idx0__, int n__);
149 
150  /// Apply local part of LAPW Hamiltonian and overlap operators.
151  /** \param [in] spfftk SpFFT transform object for G+k vectors.
152  * \param [in] gkvec_p FFT-friendly G+k vector partitioning.
153  * \param [in] N Starting index of wave-functions.
154  * \param [in] n Number of wave-functions to which H and O are applied.
155  * \param [in] phi Input wave-functions [always on CPU].
156  * \param [out] hphi LAPW Hamiltonian applied to wave-function [CPU || GPU].
157  * \param [out] ophi LAPW overlap matrix applied to wave-function [CPU || GPU].
158  *
159  * Only plane-wave part of output wave-functions is changed.
160  */
161  void apply_h_o(spfft::Transform& spfftik__, sddk::Gvec_partition const& gkvec_p__, int N__, int n__,
162  sddk::Wave_functions& phi__, sddk::Wave_functions* hphi__, sddk::Wave_functions* ophi__);
163 
164  /// Apply magnetic field to the full-potential wave-functions.
165  /** In case of collinear magnetism only Bz is applied to <tt>phi</tt> and stored in the first component of
166  * <tt>bphi</tt>. In case of non-collinear magnetims Bx-iBy is also applied and stored in the third
167  * component of <tt>bphi</tt>. The second component of <tt>bphi</tt> is used to store -Bz|phi>.
168  *
169  * \param [in] spfftk SpFFT transform object for G+k vectors.
170  * \param [in] N Starting index of wave-functions.
171  * \param [in] n Number of wave-functions to which H and O are applied.
172  * \param [in] phi Input wave-functions.
173  * \param [out] bphi Output vector of magentic field components, applied to the wave-functions.
174  */
175  void apply_b(spfft::Transform& spfftk__, int N__, int n__, sddk::Wave_functions& phi__,
176  std::vector<sddk::Wave_functions>& bphi__); // TODO: align argument order with apply_h()
