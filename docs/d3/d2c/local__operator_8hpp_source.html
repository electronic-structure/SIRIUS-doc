80/// Representation of the local operator.
81/** The following functionality is implementated:
82 * - application of the local part of Hamiltonian (kinetic + potential) to the wave-functions in the PP-PW case
83 * - application of the interstitial part of H and O in the case of FP-LAPW
84 * - application of the interstitial part of effective magnetic field to the first-variational functios
85 * - remapping of potential and unit-step functions from fine to coarse mesh of G-vectors
86 */
87template <typename T>
88class Local_operator
89{
90 private:
91 /// Common parameters.
92 Simulation_context const& ctx_;
93
94 /// Coarse-grid FFT driver for this operator.
95 spfft_transform_type<T>& fft_coarse_;
96
97 /// Distribution of the G-vectors for the FFT transformation.
98 sddk::Gvec_partition const& gvec_coarse_p_;
99
100 /// Kinetic energy of G+k plane-waves.
101 sddk::mdarray<T, 1> pw_ekin_;
102
103 /// Effective potential components and unit step function on a coarse FFT grid.
104 /** The following elements are stored in the array:
105 - V(r) + B_z(r) (in PP-PW case) or V(r) (in FP-LAPW case)
106 - V(r) - B_z(r) (in PP-PW case) or B_z(r) (in FP-LAPW case)
107 - B_x(r)
108 - B_y(r)
109 - Theta(r) (in FP-LAPW case)
110 - inverse of 1 + relative mass (needed for ZORA LAPW)
111 */
112 std::array<std::unique_ptr<Smooth_periodic_function<T>>, 6> veff_vec_;
113
114 /// Temporary array to store [V*phi](G)
115 sddk::mdarray<std::complex<T>, 1> vphi_;
116
117 /// Temporary array to store psi_{up}(r).
118 /** The size of the array is equal to the size of FFT buffer. */
119 sddk::mdarray<std::complex<T>, 1> buf_rg_;
120
121 /// V(G=0) matrix elements.
122 T v0_[2];
123
124 public:
125 /// Constructor.
126 /** Prepares k-point independent part of the local potential. If potential is provided, it is mapped to the
127 * coarse FFT grid, otherwise the constant potential is assumed for the debug and benchmarking purposes.
128 * In the case of GPU-enabled FFT driver all effective fields on the coarse grid are copied to the device
129 * and remain there until the local operator is destroyed.
130
131 * \param [in] ctx Simulation context.
132 * \param [in] fft_coarse Explicit FFT driver for the coarse mesh.
133 * \param [in] gvec_coarse_p FFT-friendly G-vector distribution for the coarse mesh.
134 * \param [in] potential Effective potential and magnetic fields \f$ V_{eff}({\bf r}) \f$ and
135 * \f$ {\bf B}_{eff}({\bf r}) \f$ on the fine FFT grid.
136 */
137 Local_operator(Simulation_context const& ctx__,
138 spfft_transform_type<T>& fft_coarse__,
139 sddk::Gvec_partition const& gvec_coarse_p__,
140 Potential* potential__ = nullptr);
141
142 /// Prepare the k-point dependent arrays.
143 /** \param [in] gkvec_p FFT-friendly G+k vector partitioning. */
144 void prepare_k(sddk::Gvec_partition const& gkvec_p__);
145
146 /// Apply local part of Hamiltonian to pseudopotential wave-functions.
147 /** \param [in] spfftk SpFFT transform object for G+k vectors.
148 * \param [in] gkvec_p FFT-friendly G+k vector partitioning.
149 * \param [in] spins Range of wave-function spins to which Hloc is applied.
150 * \param [in] phi Input wave-functions.
151 * \param [out] hphi Local hamiltonian applied to wave-function.
152 * \param [in] idx0 Starting index of wave-functions.
153 * \param [in] n Number of wave-functions to which H is applied.
154 *
155 * Spin range can take the following values:
156 * - [0, 0]: apply H_{uu} to the up- component of wave-functions
157 * - [1, 1]: apply H_{dd} to the dn- component of wave-functions
158 * - [0, 1]: apply full Hamiltonian to the spinor wave-functions
159 *
160 * Local Hamiltonian includes kinetic term and local part of potential.
161 */
162 void apply_h(spfft_transform_type<T>& spfftk__, sddk::Gvec_partition const& gkvec_p__, sddk::spin_range spins__,
163 sddk::Wave_functions<T>& phi__, sddk::Wave_functions<T>& hphi__, int idx0__, int n__);
164
165 /// Apply local part of LAPW Hamiltonian and overlap operators.
166 /** \param [in] spfftk SpFFT transform object for G+k vectors.
167 * \param [in] gkvec_p FFT-friendly G+k vector partitioning.
168 * \param [in] N Starting index of wave-functions.
169 * \param [in] n Number of wave-functions to which H and O are applied.
170 * \param [in] phi Input wave-functions [always on CPU].
171 * \param [out] hphi LAPW Hamiltonian applied to wave-function [CPU || GPU].
172 * \param [out] ophi LAPW overlap matrix applied to wave-function [CPU || GPU].
173 *
174 * Only plane-wave part of output wave-functions is changed.
175 */
176 void apply_h_o(spfft_transform_type<T>& spfftik__, sddk::Gvec_partition const& gkvec_p__, int N__, int n__,
177 sddk::Wave_functions<T>& phi__, sddk::Wave_functions<T>* hphi__, sddk::Wave_functions<T>* ophi__);
178
179 /// Apply magnetic field to the full-potential wave-functions.
180 /** In case of collinear magnetism only Bz is applied to <tt>phi</tt> and stored in the first component of
181 * <tt>bphi</tt>. In case of non-collinear magnetims Bx-iBy is also applied and stored in the third
182 * component of <tt>bphi</tt>. The second component of <tt>bphi</tt> is used to store -Bz|phi>.
183 *
184 * \param [in] spfftk SpFFT transform object for G+k vectors.
185 * \param [in] N Starting index of wave-functions.
186 * \param [in] n Number of wave-functions to which H and O are applied.
187 * \param [in] phi Input wave-functions.
188 * \param [out] bphi Output vector of magentic field components, applied to the wave-functions.
189 */
190 void apply_b(spfft_transform_type<T>& spfftk__, int N__, int n__, sddk::Wave_functions<T>& phi__,
191 std::vector<sddk::Wave_functions<T>>& bphi__); // TODO: align argument order with apply_h()
192
193 inline T v0(int ispn__) const
194 {
195 return v0_[ispn__];
196 }
197};
