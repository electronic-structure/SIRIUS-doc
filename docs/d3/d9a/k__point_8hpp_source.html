258  /// Generate first-variational states from eigen-vectors.
259  /** APW+lo basis \f$ \varphi_{\mu {\bf k}}({\bf r}) = \{ \varphi_{\bf G+k}({\bf r}),
260  \varphi_{j{\bf k}}({\bf r}) \} \f$ is used to expand first-variational wave-functions:
261 
262  \f[
263  \psi_{i{\bf k}}({\bf r}) = \sum_{\mu} c_{\mu i}^{\bf k} \varphi_{\mu \bf k}({\bf r}) =
264  \sum_{{\bf G}}c_{{\bf G} i}^{\bf k} \varphi_{\bf G+k}({\bf r}) +
265  \sum_{j}c_{j i}^{\bf k}\varphi_{j{\bf k}}({\bf r})
266  \f]
267 
268  Inside muffin-tins the expansion is converted into the following form:
269  \f[
270  \psi_{i {\bf k}}({\bf r})= \begin{array}{ll}
271  \displaystyle \sum_{L} \sum_{\lambda=1}^{N_{\ell}^{\alpha}}
272  F_{L \lambda}^{i {\bf k},\alpha}f_{\ell \lambda}^{\alpha}(r)
273  Y_{\ell m}(\hat {\bf r}) & {\bf r} \in MT_{\alpha} \end{array}
274  \f]
275 
276  Thus, the total number of coefficients representing a wave-funstion is equal
277  to the number of muffin-tin basis functions of the form \f$ f_{\ell \lambda}^{\alpha}(r)
278  Y_{\ell m}(\hat {\bf r}) \f$ plust the number of G+k plane waves.
279  First-variational states are obtained from the first-variational eigen-vectors and
280  LAPW matching coefficients.
281 
282  APW part:
283  \f[
284  \psi_{\xi j}^{\bf k} = \sum_{{\bf G}} Z_{{\bf G} j}^{\bf k} * A_{\xi}({\bf G+k})
285  \f]
286  */
287  void generate_fv_states();

297  /// Generate plane-wave coefficients of the atomic wave-functions.
298  /** Plane-wave coefficients of the atom-centered wave-functions
299  \f$ \varphi^{\alpha}_{\ell m}({\bf r}) = \varphi^{\alpha}_{\ell}(r)R_{\ell m}(\theta, \phi) \f$
300  are computed in the following way:
301  \f[
302  \varphi^{\alpha}_{\ell m}({\bf q}) = \frac{1}{\sqrt{\Omega}}
303  \int e^{-i{\bf q}{\bf r}} \varphi^{\alpha}_{\ell m}({\bf r} - {\bf r}_{\alpha}) d{\bf r} =
304  \frac{e^{-i{\bf q}{\bf r}_{\alpha}}}{\sqrt{\Omega}} \int e^{-i{\bf q}{\bf r}}
305  \varphi^{\alpha}_{\ell}(r)R_{\ell m}(\theta, \phi) r^2 \sin\theta dr d\theta d\phi
306  \f]
307  where \f$ {\bf q} = {\bf G+k} \f$. Using the expansion of the plane wave in terms of spherical Bessel
308  functions and real spherical harmonics:
309  \f[
310  e^{-i{\bf q}{\bf r}}=4\pi \sum_{\ell m} (-i)^\ell j_{\ell}(q r)R_{\ell m}({\bf \hat q})R_{\ell m}({\bf \hat r})
311  \f]
312  we arrive to the following expression:
313  \f[
314  \varphi^{\alpha}_{\ell m}({\bf q}) = e^{-i{\bf q}{\bf r}_{\alpha}} \frac{4\pi}{\sqrt{\Omega}} (-i)^\ell
315  R_{\ell m}({\bf q}) \int \varphi^{\alpha}_{\ell}(r) j_{\ell}(q r) r^2 dr
316  \f]
317 
318  \note In the current implementation wave-functions are generated as scalars (without spin index). Spinor atomic
319  wave-functions might be necessary in future for the more advanced LDA+U implementation.
320 
321  \param [in] atoms List of atoms, for which the wave-functions are generated.
322  \param [in] indexb Lambda function that returns index of the basis functions for each atom type.
323  \param [in] ri Radial integrals of the product of sperical Bessel functions and atomic functions.
324  \param [out] wf Resulting wave-functions for the list of atoms. Output wave-functions must have
325  sufficient storage space.
326  */
327  void generate_atomic_wave_functions(std::vector<int> atoms__,
328  std::function<sirius::experimental::basis_functions_index const*(int)> indexb__,
329  Radial_integrals_atomic_wf<false> const& ri__, sddk::Wave_functions& wf__);

371  /// Collect distributed second-variational vectors into a global array.
372  void get_sv_eigen_vectors(mdarray<double_complex, 2>& sv_evec__) const
373  {
374  assert((int)sv_evec__.size(0) == ctx_.num_spins() * ctx_.num_fv_states());
375  assert((int)sv_evec__.size(1) == ctx_.num_spins() * ctx_.num_fv_states());
376 
377  sv_evec__.zero();
378 
379  if (!ctx_.need_sv()) {
380  for (int i = 0; i < ctx_.num_fv_states(); i++) {
381  sv_evec__(i, i) = 1;
382  }
383  return;
384  }
385 
386  int nsp = (ctx_.num_mag_dims() == 3) ? 1 : ctx_.num_spins();
387 
388  for (int ispn = 0; ispn < nsp; ispn++) {
389  int offs = ctx_.num_fv_states() * ispn;
390  for (int jloc = 0; jloc < sv_eigen_vectors_[ispn].num_cols_local(); jloc++) {
391  int j = sv_eigen_vectors_[ispn].icol(jloc);
392  for (int iloc = 0; iloc < sv_eigen_vectors_[ispn].num_rows_local(); iloc++) {
393  int i = sv_eigen_vectors_[ispn].irow(iloc);
394  sv_evec__(i + offs, j + offs) = sv_eigen_vectors_[ispn](iloc, jloc);
395  }
396  }
397  }
398 
399  comm().allreduce(sv_evec__.at(memory_t::host), (int)sv_evec__.size());
400  }

827 /** \page basis Basis functions for Kohn-Sham wave-functions expansion
828  *
829  * \section basis1 LAPW+lo basis
830  *
831  * LAPW+lo basis consists of two different sets of functions: LAPW functions \f$ \varphi_{{\bf G+k}} \f$ defined over
832  * entire unit cell:
833  * \f[
834  * \varphi_{{\bf G+k}}({\bf r}) = \left\{ \begin{array}{ll}
835  * \displaystyle \sum_{L} \sum_{\nu=1}^{O_{\ell}^{\alpha}} a_{L\nu}^{\alpha}({\bf G+k})u_{\ell \nu}^{\alpha}(r)
836  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in {\rm MT} \alpha \\
837  * \displaystyle \frac{1}{\sqrt \Omega} e^{i({\bf G+k}){\bf r}} & {\bf r} \in {\rm I} \end{array} \right.
838  * \f]
839  * and Bloch sums of local orbitals defined inside muffin-tin spheres only:
840  * \f[
841  * \begin{array}{ll} \displaystyle \varphi_{j{\bf k}}({\bf r})=\sum_{{\bf T}} e^{i{\bf kT}}
842  * \varphi_{j}({\bf r - T}) & {\rm {\bf r} \in MT} \end{array}
843  * \f]
844  * Each local orbital is composed of radial and angular parts:
845  * \f[
846  * \varphi_{j}({\bf r}) = \phi_{\ell_j}^{\zeta_j,\alpha_j}(r) Y_{\ell_j m_j}(\hat {\bf r})
847  * \f]
848  * Radial part of local orbital is defined as a linear combination of radial functions (minimum two radial functions
849  * are required) such that local orbital vanishes at the sphere boundary:
850  * \f[
851  * \phi_{\ell}^{\zeta, \alpha}(r) = \sum_{p}\gamma_{p}^{\zeta,\alpha} u_{\ell \nu_p}^{\alpha}(r)
852  * \f]
853  *
854  * Arbitrary number of local orbitals can be introduced for each angular quantum number (this is highlighted by
855  * the index \f$ \zeta \f$).
856  *
857  * Radial functions are m-th order (with zero-order being a function itself) energy derivatives of the radial
858  * Schrdinger equation:
859  * \f[
860  * u_{\ell \nu}^{\alpha}(r) = \frac{\partial^{m_{\nu}}}{\partial^{m_{\nu}}E}u_{\ell}^{\alpha}(r,E)\Big|_{E=E_{\nu}}
861  * \f]
862  */
863

869  * into the memory of a single node. For some operations we need a "panel" distribution of data, where each
870  * MPI rank gets a local panel of block-cyclic distributed matrix. This way of storing data is necessary for the
