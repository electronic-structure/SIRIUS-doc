314 /// Generate first-variational states from eigen-vectors.
315 /** APW+lo basis \f$ \varphi_{\mu {\bf k}}({\bf r}) = \{ \varphi_{\bf G+k}({\bf r}),
316 \varphi_{j{\bf k}}({\bf r}) \} \f$ is used to expand first-variational wave-functions:
317
318 \f[
319 \psi_{i{\bf k}}({\bf r}) = \sum_{\mu} c_{\mu i}^{\bf k} \varphi_{\mu \bf k}({\bf r}) =
320 \sum_{{\bf G}}c_{{\bf G} i}^{\bf k} \varphi_{\bf G+k}({\bf r}) +
321 \sum_{j}c_{j i}^{\bf k}\varphi_{j{\bf k}}({\bf r})
322 \f]
323
324 Inside muffin-tins the expansion is converted into the following form:
325 \f[
326 \psi_{i {\bf k}}({\bf r})= \begin{array}{ll}
327 \displaystyle \sum_{L} \sum_{\lambda=1}^{N_{\ell}^{\alpha}}
328 F_{L \lambda}^{i {\bf k},\alpha}f_{\ell \lambda}^{\alpha}(r)
329 Y_{\ell m}(\hat {\bf r}) & {\bf r} \in MT_{\alpha} \end{array}
330 \f]
331
332 Thus, the total number of coefficients representing a wave-funstion is equal
333 to the number of muffin-tin basis functions of the form \f$ f_{\ell \lambda}^{\alpha}(r)
334 Y_{\ell m}(\hat {\bf r}) \f$ plust the number of G+k plane waves.
335 First-variational states are obtained from the first-variational eigen-vectors and
336 LAPW matching coefficients.
337
338 APW part:
339 \f[
340 \psi_{\xi j}^{\bf k} = \sum_{{\bf G}} Z_{{\bf G} j}^{\bf k} * A_{\xi}({\bf G+k})
341 \f]
342 */
343 void generate_fv_states();

347 states and second-variational eigen-vectors. */

350 /// Generate plane-wave coefficients of the atomic wave-functions.
351 /** Plane-wave coefficients of the atom-centered wave-functions
352 \f$ \varphi^{\alpha}_{\ell m}({\bf r}) = \varphi^{\alpha}_{\ell}(r)R_{\ell m}(\theta, \phi) \f$
353 are computed in the following way:
354 \f[
355 \varphi^{\alpha}_{\ell m}({\bf q}) = \frac{1}{\sqrt{\Omega}}
356 \int e^{-i{\bf q}{\bf r}} \varphi^{\alpha}_{\ell m}({\bf r} - {\bf r}_{\alpha}) d{\bf r} =
357 \frac{e^{-i{\bf q}{\bf r}_{\alpha}}}{\sqrt{\Omega}} \int e^{-i{\bf q}{\bf r}}
358 \varphi^{\alpha}_{\ell}(r)R_{\ell m}(\theta, \phi) r^2 \sin\theta dr d\theta d\phi
359 \f]
360 where \f$ {\bf q} = {\bf G+k} \f$. Using the expansion of the plane wave in terms of spherical Bessel
361 functions and real spherical harmonics:
362 \f[
363 e^{-i{\bf q}{\bf r}}=4\pi \sum_{\ell m} (-i)^\ell j_{\ell}(q r)R_{\ell m}({\bf \hat q})R_{\ell m}({\bf \hat r})
364 \f]
365 we arrive to the following expression:
366 \f[
367 \varphi^{\alpha}_{\ell m}({\bf q}) = e^{-i{\bf q}{\bf r}_{\alpha}} \frac{4\pi}{\sqrt{\Omega}} (-i)^\ell
368 R_{\ell m}({\bf q}) \int \varphi^{\alpha}_{\ell}(r) j_{\ell}(q r) r^2 dr
369 \f]
370
371 \note In the current implementation wave-functions are generated as scalars (without spin index). Spinor atomic
372 wave-functions might be necessary in future for the more advanced LDA+U implementation.
373
374 \param [in] atoms List of atoms, for which the wave-functions are generated.
375 \param [in] indexb Lambda function that returns index of the basis functions for each atom type.
376 \param [in] ri Radial integrals of the product of sperical Bessel functions and atomic functions.
377 \param [out] wf Resulting wave-functions for the list of atoms. Output wave-functions must have
378 sufficient storage space.
379 */
380 void generate_atomic_wave_functions(std::vector<int> atoms__,
381 std::function<sirius::experimental::basis_functions_index const*(int)> indexb__,
382 Radial_integrals_atomic_wf<false> const& ri__, sddk::Wave_functions<T>& wf__);

425 /// Collect distributed second-variational vectors into a global array.
426 void get_sv_eigen_vectors(mdarray<std::complex<T>, 2>& sv_evec__) const
427 {
428 assert((int)sv_evec__.size(0) == ctx_.num_spins() * ctx_.num_fv_states());
429 assert((int)sv_evec__.size(1) == ctx_.num_spins() * ctx_.num_fv_states());
430
431 sv_evec__.zero();
432
433 if (!ctx_.need_sv()) {
434 for (int i = 0; i < ctx_.num_fv_states(); i++) {
435 sv_evec__(i, i) = 1;
436 }
437 return;
438 }
439
440 int nsp = (ctx_.num_mag_dims() == 3) ? 1 : ctx_.num_spins();
441
442 for (int ispn = 0; ispn < nsp; ispn++) {
443 int offs = ctx_.num_fv_states() * ispn;
444 for (int jloc = 0; jloc < sv_eigen_vectors_[ispn].num_cols_local(); jloc++) {
445 int j = sv_eigen_vectors_[ispn].icol(jloc);
446 for (int iloc = 0; iloc < sv_eigen_vectors_[ispn].num_rows_local(); iloc++) {
447 int i = sv_eigen_vectors_[ispn].irow(iloc);
448 sv_evec__(i + offs, j + offs) = sv_eigen_vectors_[ispn](iloc, jloc);
449 }
450 }
451 }
452
453 comm().allreduce(sv_evec__.at(memory_t::host), (int)sv_evec__.size());
454 }

459 /// Get the number of occupied bands for each spin channel.
460 inline int num_occupied_bands(int ispn__ = -1) const
461 {
462 if (ctx_.num_mag_dims() == 3) {
463 ispn__ = 0;
464 }
465 for (int j = ctx_.num_bands() - 1; j >= 0; j--) {
466 if (std::abs(band_occupancy(j, ispn__)) > ctx_.min_occupancy() * ctx_.max_occupancy()) {
467 return j + 1;
468 }
469 }
470 return 0;
471 }

861/** \page basis Basis functions for Kohn-Sham wave-functions expansion
862 *
863 * \section basis1 LAPW+lo basis
864 *
865 * LAPW+lo basis consists of two different sets of functions: LAPW functions \f$ \varphi_{{\bf G+k}} \f$ defined over
866 * entire unit cell:
867 * \f[
868 * \varphi_{{\bf G+k}}({\bf r}) = \left\{ \begin{array}{ll}
869 * \displaystyle \sum_{L} \sum_{\nu=1}^{O_{\ell}^{\alpha}} a_{L\nu}^{\alpha}({\bf G+k})u_{\ell \nu}^{\alpha}(r)
870 * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in {\rm MT} \alpha \\
871 * \displaystyle \frac{1}{\sqrt \Omega} e^{i({\bf G+k}){\bf r}} & {\bf r} \in {\rm I} \end{array} \right.
872 * \f]
873 * and Bloch sums of local orbitals defined inside muffin-tin spheres only:
874 * \f[
875 * \begin{array}{ll} \displaystyle \varphi_{j{\bf k}}({\bf r})=\sum_{{\bf T}} e^{i{\bf kT}}
876 * \varphi_{j}({\bf r - T}) & {\rm {\bf r} \in MT} \end{array}
877 * \f]
878 * Each local orbital is composed of radial and angular parts:
879 * \f[
880 * \varphi_{j}({\bf r}) = \phi_{\ell_j}^{\zeta_j,\alpha_j}(r) Y_{\ell_j m_j}(\hat {\bf r})
881 * \f]
882 * Radial part of local orbital is defined as a linear combination of radial functions (minimum two radial functions
883 * are required) such that local orbital vanishes at the sphere boundary:
884 * \f[
885 * \phi_{\ell}^{\zeta, \alpha}(r) = \sum_{p}\gamma_{p}^{\zeta,\alpha} u_{\ell \nu_p}^{\alpha}(r)
886 * \f]
887 *
888 * Arbitrary number of local orbitals can be introduced for each angular quantum number (this is highlighted by
889 * the index \f$ \zeta \f$).
890 *
891 * Radial functions are m-th order (with zero-order being a function itself) energy derivatives of the radial
892 * Schrdinger equation:
893 * \f[
894 * u_{\ell \nu}^{\alpha}(r) = \frac{\partial^{m_{\nu}}}{\partial^{m_{\nu}}E}u_{\ell}^{\alpha}(r,E)\Big|_{E=E_{\nu}}
895 * \f]
896 */
897
898/** \page data_dist K-point data distribution
899 *
900 * \section data_dist1 "Panel" and "full" data storage
901 *
902 * We have to deal with big arrays (matching coefficients, eigen vectors, wave functions, etc.) which may not fit
903 * into the memory of a single node. For some operations we need a "panel" distribution of data, where each
904 * MPI rank gets a local panel of block-cyclic distributed matrix. This way of storing data is necessary for the
905 * distributed PBLAS and ScaLAPACK operations.
906 *
907 */
