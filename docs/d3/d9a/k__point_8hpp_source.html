227  /// Generate first-variational states from eigen-vectors.
228  /** APW+lo basis \f$ \varphi_{\mu {\bf k}}({\bf r}) = \{ \varphi_{\bf G+k}({\bf r}),
229  * \varphi_{j{\bf k}}({\bf r}) \} \f$ is used to expand first-variational wave-functions:
230  *
231  * \f[
232  * \psi_{i{\bf k}}({\bf r}) = \sum_{\mu} c_{\mu i}^{\bf k} \varphi_{\mu \bf k}({\bf r}) =
233  * \sum_{{\bf G}}c_{{\bf G} i}^{\bf k} \varphi_{\bf G+k}({\bf r}) +
234  * \sum_{j}c_{j i}^{\bf k}\varphi_{j{\bf k}}({\bf r})
235  * \f]
236  *
237  * Inside muffin-tins the expansion is converted into the following form:
238  * \f[
239  * \psi_{i {\bf k}}({\bf r})= \begin{array}{ll}
240  * \displaystyle \sum_{L} \sum_{\lambda=1}^{N_{\ell}^{\alpha}}
241  * F_{L \lambda}^{i {\bf k},\alpha}f_{\ell \lambda}^{\alpha}(r)
242  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in MT_{\alpha} \end{array}
243  * \f]
244  *
245  * Thus, the total number of coefficients representing a wave-funstion is equal
246  * to the number of muffin-tin basis functions of the form \f$ f_{\ell \lambda}^{\alpha}(r)
247  * Y_{\ell m}(\hat {\bf r}) \f$ plust the number of G+k plane waves.
248  * First-variational states are obtained from the first-variational eigen-vectors and
249  * LAPW matching coefficients.
250  *
251  * APW part:
252  * \f[
253  * \psi_{\xi j}^{\bf k} = \sum_{{\bf G}} Z_{{\bf G} j}^{\bf k} * A_{\xi}({\bf G+k})
254  * \f]
255  */
256  void generate_fv_states();

304  /// Collect distributed second-variational vectors into a global array.
305  void get_sv_eigen_vectors(mdarray<double_complex, 2>& sv_evec__) const
306  {
307  assert((int)sv_evec__.size(0) == ctx_.num_spins() * ctx_.num_fv_states());
308  assert((int)sv_evec__.size(1) == ctx_.num_spins() * ctx_.num_fv_states());
309 
310  sv_evec__.zero();
311 
312  if (!ctx_.need_sv()) {
313  for (int i = 0; i < ctx_.num_fv_states(); i++) {
314  sv_evec__(i, i) = 1;
315  }
316  return;
317  }
318 
319  int nsp = (ctx_.num_mag_dims() == 3) ? 1 : ctx_.num_spins();
320 
321  for (int ispn = 0; ispn < nsp; ispn++) {
322  int offs = ctx_.num_fv_states() * ispn;
323  for (int jloc = 0; jloc < sv_eigen_vectors_[ispn].num_cols_local(); jloc++) {
324  int j = sv_eigen_vectors_[ispn].icol(jloc);
325  for (int iloc = 0; iloc < sv_eigen_vectors_[ispn].num_rows_local(); iloc++) {
326  int i = sv_eigen_vectors_[ispn].irow(iloc);
327  sv_evec__(i + offs, j + offs) = sv_eigen_vectors_[ispn](iloc, jloc);
328  }
329  }
330  }
331 
332  comm().allreduce(sv_evec__.at(memory_t::host), (int)sv_evec__.size());
333  }

339  inline int num_occupied_bands(int ispn__ = -1) const
340  {
341  for (int j = ctx_.num_bands() - 1; j >= 0; j--) {
342  if (std::abs(band_occupancy(j, ispn__) * weight()) > 1e-14) {

738 /** \page basis Basis functions for Kohn-Sham wave-functions expansion
739  *
740  * \section basis1 LAPW+lo basis
741  *
742  * LAPW+lo basis consists of two different sets of functions: LAPW functions \f$ \varphi_{{\bf G+k}} \f$ defined over
743  * entire unit cell:
744  * \f[
745  * \varphi_{{\bf G+k}}({\bf r}) = \left\{ \begin{array}{ll}
746  * \displaystyle \sum_{L} \sum_{\nu=1}^{O_{\ell}^{\alpha}} a_{L\nu}^{\alpha}({\bf G+k})u_{\ell \nu}^{\alpha}(r)
747  * Y_{\ell m}(\hat {\bf r}) & {\bf r} \in {\rm MT} \alpha \\
748  * \displaystyle \frac{1}{\sqrt \Omega} e^{i({\bf G+k}){\bf r}} & {\bf r} \in {\rm I} \end{array} \right.
749  * \f]
750  * and Bloch sums of local orbitals defined inside muffin-tin spheres only:
751  * \f[
752  * \begin{array}{ll} \displaystyle \varphi_{j{\bf k}}({\bf r})=\sum_{{\bf T}} e^{i{\bf kT}}
753  * \varphi_{j}({\bf r - T}) & {\rm {\bf r} \in MT} \end{array}
754  * \f]
755  * Each local orbital is composed of radial and angular parts:
756  * \f[
757  * \varphi_{j}({\bf r}) = \phi_{\ell_j}^{\zeta_j,\alpha_j}(r) Y_{\ell_j m_j}(\hat {\bf r})
758  * \f]
759  * Radial part of local orbital is defined as a linear combination of radial functions (minimum two radial functions
760  * are required) such that local orbital vanishes at the sphere boundary:
761  * \f[
762  * \phi_{\ell}^{\zeta, \alpha}(r) = \sum_{p}\gamma_{p}^{\zeta,\alpha} u_{\ell \nu_p}^{\alpha}(r)
763  * \f]
764  *
765  * Arbitrary number of local orbitals can be introduced for each angular quantum number (this is highlighted by
766  * the index \f$ \zeta \f$).
767  *
768  * Radial functions are m-th order (with zero-order being a function itself) energy derivatives of the radial
769  * Schrödinger equation:
770  * \f[
771  * u_{\ell \nu}^{\alpha}(r) = \frac{\partial^{m_{\nu}}}{\partial^{m_{\nu}}E}u_{\ell}^{\alpha}(r,E)\Big|_{E=E_{\nu}}
772  * \f]
773  */
774

780  * into the memory of a single node. For some operations we need a "panel" distribution of data, where each
781  * MPI rank gets a local panel of block-cyclic distributed matrix. This way of storing data is necessary for the
782  * distributed PBLAS and ScaLAPACK operations.
783  *
784  */
