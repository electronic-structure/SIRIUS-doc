60 template <typename T, int what>
61 std::pair<sddk::mdarray<double, 2>, sddk::mdarray<double, 2>>
62 Hamiltonian_k::get_h_o_diag_pw() const
63 {
64  PROFILE("sirius::Hamiltonian_k::get_h_o_diag");
65 
66  auto const& uc = H0_.ctx().unit_cell();
67 
68  sddk::mdarray<double, 2> h_diag(kp_.num_gkvec_loc(), H0_.ctx().num_spins());
69  sddk::mdarray<double, 2> o_diag(kp_.num_gkvec_loc(), H0_.ctx().num_spins());
70 
71  h_diag.zero();
72  o_diag.zero();
73 
74  for (int ispn = 0; ispn < H0_.ctx().num_spins(); ispn++) {
75 
76  /* local H contribution */
77  #pragma omp parallel for schedule(static)
78  for (int ig_loc = 0; ig_loc < kp_.num_gkvec_loc(); ig_loc++) {
79  if (what & 1) {
80  auto ekin = 0.5 * kp_.gkvec().gkvec_cart<index_domain_t::local>(ig_loc).length2();
81  h_diag(ig_loc, ispn) = ekin + H0_.local_op().v0(ispn);
82  }
83  if (what & 2) {
84  o_diag(ig_loc, ispn) = 1;
85  }
86  }
87 
88  PROFILE_START("sirius::Hamiltonian_k::get_h_o_diag|1");
89  /* non-local H contribution */
90  auto beta_gk_t = kp_.beta_projectors().pw_coeffs_t(0);
91  matrix<double_complex> beta_gk_tmp(kp_.num_gkvec_loc(), uc.max_mt_basis_size());
92 
93  for (int iat = 0; iat < uc.num_atom_types(); iat++) {
94  auto& atom_type = uc.atom_type(iat);
95  int nbf = atom_type.mt_basis_size();
96 
97  matrix<double_complex> d_sum;
98  if (what & 1) {
99  d_sum = matrix<double_complex>(nbf, nbf);
100  d_sum.zero();
101  }
102 
103  matrix<double_complex> q_sum;
104  if (what & 2) {
105  q_sum = matrix<double_complex>(nbf, nbf);
106  q_sum.zero();
107  }
108 
109  for (int i = 0; i < atom_type.num_atoms(); i++) {
110  int ia = atom_type.atom_id(i);
111 
112  for (int xi2 = 0; xi2 < nbf; xi2++) {
113  for (int xi1 = 0; xi1 < nbf; xi1++) {
114  if (what & 1) {
115  d_sum(xi1, xi2) += H0_.D().value<T>(xi1, xi2, ispn, ia);
116  }
117  if (what & 2) {
118  q_sum(xi1, xi2) += H0_.Q().value<T>(xi1, xi2, ispn, ia);
119  }
120  }
121  }
122  }
123  PROFILE_STOP("sirius::Hamiltonian_k::get_h_o_diag|1");
124 
125  int offs = uc.atom_type(iat).offset_lo();
126 
127  PROFILE_START("sirius::Hamiltonian_k::get_h_o_diag|3");
128  if (what & 1) {
129  sddk::linalg(linalg_t::blas).gemm('N', 'N', kp_.num_gkvec_loc(), nbf, nbf,
130  &sddk::linalg_const<double_complex>::one(), &beta_gk_t(0, offs), beta_gk_t.ld(),
131  &d_sum(0, 0), d_sum.ld(), &sddk::linalg_const<double_complex>::zero(),
132  &beta_gk_tmp(0, 0), beta_gk_tmp.ld());
133  #pragma omp parallel
134  for (int xi = 0; xi < nbf; xi++) {
135  #pragma omp for schedule(static) nowait
136  for (int ig_loc = 0; ig_loc < kp_.num_gkvec_loc(); ig_loc++) {
137  /* compute <G+k|beta_xi1> D_{xi1, xi2} <beta_xi2|G+k> contribution from all atoms */
138  h_diag(ig_loc, ispn) +=
139  std::real(beta_gk_tmp(ig_loc, xi) * std::conj(beta_gk_t(ig_loc, offs + xi)));
140  }
141  }
142  }
143 
144  if (what & 2) {
145  sddk::linalg(linalg_t::blas).gemm('N', 'N', kp_.num_gkvec_loc(), nbf, nbf,
146  &sddk::linalg_const<double_complex>::one(), &beta_gk_t(0, offs), beta_gk_t.ld(),
147  &q_sum(0, 0), q_sum.ld(), &sddk::linalg_const<double_complex>::zero(),
148  &beta_gk_tmp(0, 0), beta_gk_tmp.ld());
149  #pragma omp parallel
150  for (int xi = 0; xi < nbf; xi++) {
151  #pragma omp for schedule(static) nowait
152  for (int ig_loc = 0; ig_loc < kp_.num_gkvec_loc(); ig_loc++) {
153  /* compute <G+k|beta_xi1> Q_{xi1, xi2} <beta_xi2|G+k> contribution from all atoms */
154  o_diag(ig_loc, ispn) +=
155  std::real(beta_gk_tmp(ig_loc, xi) * std::conj(beta_gk_t(ig_loc, offs + xi)));
156  }
157  }
158  }
159  PROFILE_STOP("sirius::Hamiltonian_k::get_h_o_diag|3");
160  }
161  }
162  if (H0_.ctx().processing_unit() == device_t::GPU) {
163  if (what & 1) {
164  h_diag.allocate(memory_t::device).copy_to(memory_t::device);
165  }
166  if (what & 2) {
167  o_diag.allocate(memory_t::device).copy_to(memory_t::device);
168  }
169  }
170  return std::make_pair(std::move(h_diag), std::move(o_diag));
171 }

173 template <int what>
174 std::pair<mdarray<double, 2>, mdarray<double, 2>>
175 Hamiltonian_k::get_h_o_diag_lapw() const
176 {
177  PROFILE("sirius::Hamiltonian::get_h_o_diag");
178 
179  auto const& uc = H0_.ctx().unit_cell();
180 
181  splindex<splindex_t::block> spl_num_atoms(uc.num_atoms(), kp_.comm().size(), kp_.comm().rank());
182  int nlo{0};
183  for (int ialoc = 0; ialoc < spl_num_atoms.local_size(); ialoc++) {
184  int ia = spl_num_atoms[ialoc];
185  nlo += uc.atom(ia).mt_lo_basis_size();
186  }
187 
188  mdarray<double, 2> h_diag = (what & 1) ? mdarray<double, 2>(kp_.num_gkvec_loc() + nlo, 1) : mdarray<double, 2>();
189  mdarray<double, 2> o_diag = (what & 2) ? mdarray<double, 2>(kp_.num_gkvec_loc() + nlo, 1) : mdarray<double, 2>();
190 
191  #pragma omp parallel for schedule(static)
192  for (int igloc = 0; igloc < kp_.num_gkvec_loc(); igloc++) {
193  if (what & 1) {
194  auto gvc = kp_.gkvec().gkvec_cart<index_domain_t::local>(igloc);
195  double ekin = 0.5 * dot(gvc, gvc);
196  h_diag[igloc] = H0_.local_op().v0(0) + ekin * H0_.ctx().theta_pw(0).real();
197  }
198  if (what & 2) {
199  o_diag[igloc] = H0_.ctx().theta_pw(0).real();
200  }
201  }
202 
203  #pragma omp parallel
204  {
205  matrix<double_complex> alm(kp_.num_gkvec_loc(), uc.max_mt_aw_basis_size());
206 
207  matrix<double_complex> halm = (what & 1) ?
208  matrix<double_complex>(kp_.num_gkvec_loc(), uc.max_mt_aw_basis_size()) : matrix<double_complex>();
209 
210  auto h_diag_omp = (what & 1) ? mdarray<double, 1>(kp_.num_gkvec_loc()) : mdarray<double, 1>();
211  if (what & 1) {
212  h_diag_omp.zero();
213  }
214 
215  auto o_diag_omp = (what & 2) ? mdarray<double, 1>(kp_.num_gkvec_loc()) : mdarray<double, 1>();
216  if (what & 2) {
217  o_diag_omp.zero();
218  }
219 
220  #pragma omp for
221  for (int ia = 0; ia < uc.num_atoms(); ia++) {
222  auto& atom = uc.atom(ia);
223  int nmt = atom.mt_aw_basis_size();
224 
225  kp_.alm_coeffs_loc().generate<false>(atom, alm);
226  if (what & 1) {
227  H0_.apply_hmt_to_apw<spin_block_t::nm>(atom, kp_.num_gkvec_loc(), alm, halm);
228  }
229 
230  for (int xi = 0; xi < nmt; xi++) {
231  for (int igloc = 0; igloc < kp_.num_gkvec_loc(); igloc++) {
232  if (what & 1) {
233  h_diag_omp[igloc] += std::real(std::conj(alm(igloc, xi)) * halm(igloc, xi));
234  }
235  if (what & 2) {
236  o_diag_omp[igloc] += std::real(std::conj(alm(igloc, xi)) * alm(igloc, xi));
237  }
238  }
239  }
240  }
241 
242  #pragma omp critical
243  for (int igloc = 0; igloc < kp_.num_gkvec_loc(); igloc++) {
244  if (what & 1) {
245  h_diag[igloc] += h_diag_omp[igloc];
246  }
247  if (what & 2) {
248  o_diag[igloc] += o_diag_omp[igloc];
249  }
250  }
251  }
252 
253  nlo = 0;
254  for (int ialoc = 0; ialoc < spl_num_atoms.local_size(); ialoc++) {
255  int ia = spl_num_atoms[ialoc];
256  auto& atom = uc.atom(ia);
257  auto& type = atom.type();
258  #pragma omp parallel for
259  for (int ilo = 0; ilo < type.mt_lo_basis_size(); ilo++) {
260  int xi_lo = type.mt_aw_basis_size() + ilo;
261  /* local orbital indices */
262  int lm_lo = type.indexb(xi_lo).lm;
263  int idxrf_lo = type.indexb(xi_lo).idxrf;
264 
265  if (what & 1) {
266  h_diag[kp_.num_gkvec_loc() + nlo + ilo] =
267  atom.radial_integrals_sum_L3<spin_block_t::nm>(idxrf_lo, idxrf_lo, H0_.gaunt_coefs().gaunt_vector(
268  lm_lo, lm_lo)).real();
269  }
270  if (what & 2) {
271  o_diag[kp_.num_gkvec_loc() + nlo + ilo] = 1;
272  }
273  }
274  nlo += atom.mt_lo_basis_size();
275  }
276 
277  if (H0_.ctx().processing_unit() == device_t::GPU) {
278  if (what & 1) {
279  h_diag.allocate(memory_t::device).copy_to(memory_t::device);
280  }
281  if (what & 2) {
282  o_diag.allocate(memory_t::device).copy_to(memory_t::device);
283  }
284  }
285  return std::make_pair(std::move(h_diag), std::move(o_diag));
286 }
