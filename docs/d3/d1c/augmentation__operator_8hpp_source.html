60  void generate_pw_coeffs(Radial_integrals_aug<false> const& radial_integrals__, memory_pool& mp__)
61  {
62  if (!atom_type_.augment()) {
63  return;
64  }
65  PROFILE("sirius::Augmentation_operator::generate_pw_coeffs");
66 
67  double fourpi_omega = fourpi / gvec_.omega();
68 
69  /* maximum l of beta-projectors */
70  int lmax_beta = atom_type_.indexr().lmax();
71  int lmmax = utils::lmmax(2 * lmax_beta);
72 
73  auto l_by_lm = utils::l_by_lm(2 * lmax_beta);
74 
75  std::vector<double_complex> zilm(lmmax);
76  for (int l = 0, lm = 0; l <= 2 * lmax_beta; l++) {
77  for (int m = -l; m <= l; m++, lm++) {
78  zilm[lm] = std::pow(double_complex(0, 1), l);
79  }
80  }
81 
82  /* Gaunt coefficients of three real spherical harmonics */
83  Gaunt_coefficients<double> gaunt_coefs(lmax_beta, 2 * lmax_beta, lmax_beta, SHT::gaunt_rlm);
84 
85  /* split G-vectors between ranks */
86  int gvec_count = gvec_.count();
87  int gvec_offset = gvec_.offset();
88 
89  /* array of real spherical harmonics for each G-vector */
90  mdarray<double, 2> gvec_rlm(utils::lmmax(2 * lmax_beta), gvec_count);
91  #pragma omp parallel for schedule(static)
92  for (int igloc = 0; igloc < gvec_count; igloc++) {
93  auto rtp = SHT::spherical_coordinates(gvec_.gvec_cart<index_domain_t::local>(igloc));
94  SHT::spherical_harmonics(2 * lmax_beta, rtp[1], rtp[2], &gvec_rlm(0, igloc));
95  }
96 
97  /* number of beta-projectors */
98  int nbf = atom_type_.mt_basis_size();
99 
100  /* array of plane-wave coefficients */
101  q_pw_ = mdarray<double, 2>(mp__, nbf * (nbf + 1) / 2, 2 * gvec_count, "q_pw_");
102  #pragma omp parallel for schedule(static)
103  for (int igloc = 0; igloc < gvec_count; igloc++) {
104  int ig = gvec_offset + igloc;
105  double g = gvec_.gvec_len(ig);
106 
107  std::vector<double_complex> v(lmmax);
108 
109  auto ri = radial_integrals__.values(atom_type_.id(), g);
110 
111  for (int xi2 = 0; xi2 < nbf; xi2++) {
112  int lm2 = atom_type_.indexb(xi2).lm;
113  int idxrf2 = atom_type_.indexb(xi2).idxrf;
114 
115  for (int xi1 = 0; xi1 <= xi2; xi1++) {
116  int lm1 = atom_type_.indexb(xi1).lm;
117  int idxrf1 = atom_type_.indexb(xi1).idxrf;
118 
119  /* packed orbital index */
120  int idx12 = utils::packed_index(xi1, xi2);
121  /* packed radial-function index */
122  int idxrf12 = utils::packed_index(idxrf1, idxrf2);
123 
124  for (int lm3 = 0; lm3 < lmmax; lm3++) {
125  v[lm3] = std::conj(zilm[lm3]) * gvec_rlm(lm3, igloc) * ri(idxrf12, l_by_lm[lm3]);
126  }
127 
128  double_complex z = fourpi_omega * gaunt_coefs.sum_L3_gaunt(lm2, lm1, &v[0]);
129 
130  q_pw_(idx12, 2 * igloc) = z.real();
131  q_pw_(idx12, 2 * igloc + 1) = z.imag();
132  }
133  }
134  }
135 
136  memory_t mem{memory_t::host};
137  if (atom_type_.parameters().processing_unit() == device_t::GPU) {
138  mem = memory_t::host_pinned;
139  }
140  sym_weight_ = mdarray<double, 1>(nbf * (nbf + 1) / 2, mem, "sym_weight_");
141  for (int xi2 = 0; xi2 < nbf; xi2++) {
142  for (int xi1 = 0; xi1 <= xi2; xi1++) {
143  /* packed orbital index */
144  int idx12 = utils::packed_index(xi1, xi2);
145  sym_weight_(idx12) = (xi1 == xi2) ? 1 : 2;
146  }
147  }
148 
149  q_mtrx_ = mdarray<double, 2>(nbf, nbf);
150  q_mtrx_.zero();
151 
152  if (comm_.rank() == 0) {
153  for (int xi2 = 0; xi2 < nbf; xi2++) {
154  for (int xi1 = 0; xi1 <= xi2; xi1++) {
155  /* packed orbital index */
156  int idx12 = utils::packed_index(xi1, xi2);
157  q_mtrx_(xi1, xi2) = q_mtrx_(xi2, xi1) = gvec_.omega() * q_pw_(idx12, 0);
158  }
159  }
160  }
161  /* broadcast from rank#0 */
162  comm_.bcast(&q_mtrx_(0, 0), nbf * nbf, 0);
163 
164  if (atom_type_.parameters().control().print_checksum_) {
165  auto cs = q_pw_.checksum();
166  auto cs1 = q_mtrx_.checksum();
167  comm_.allreduce(&cs, 1);
168  if (comm_.rank() == 0) {
169  utils::print_checksum("q_pw", cs);
170  utils::print_checksum("q_mtrx", cs1);
171  }
172  }
173  }

232 /// Derivative of augmentation operator PW coefficients with respect to the Cartesian component of G-vector.
233 class Augmentation_operator_gvec_deriv
234 {
235  private:
236  Gvec const& gvec_;
237 
238  Communicator const& comm_;
239 
240  mdarray<double, 2> q_pw_;
241 
242  mdarray<double, 1> sym_weight_;
243 
244  mdarray<double, 2> rlm_g_;
245 
246  mdarray<double, 3> rlm_dg_;
247 
248  std::unique_ptr<Gaunt_coefficients<double>> gaunt_coefs_;
249 
250  public:
251  Augmentation_operator_gvec_deriv(int lmax__,
252  Gvec const& gvec__,
253  Communicator const& comm__)
254  : gvec_(gvec__)
255  , comm_(comm__)
256  {
257  PROFILE("sirius::Augmentation_operator_gvec_deriv|constructor");
258 
259  int lmax = lmax__;
260  int lmmax = utils::lmmax(2 * lmax);
261 
262  /* Gaunt coefficients of three real spherical harmonics */
263  gaunt_coefs_ = std::unique_ptr<Gaunt_coefficients<double>>(new Gaunt_coefficients<double>(lmax, 2 * lmax, lmax, SHT::gaunt_rlm));
264 
265  /* split G-vectors between ranks */
266  int gvec_count = gvec__.count();
267 
268  rlm_g_ = mdarray<double, 2>(lmmax, gvec_count);
269  rlm_dg_ = mdarray<double, 3>(lmmax, 3, gvec_count);
270 
271  /* array of real spherical harmonics and derivatives for each G-vector */
272  #pragma omp parallel for schedule(static)
273  for (int igloc = 0; igloc < gvec_count; igloc++) {
274  auto rtp = SHT::spherical_coordinates(gvec__.gvec_cart<index_domain_t::local>(igloc));
275 
276  double theta = rtp[1];
277  double phi = rtp[2];
278  vector3d<double> dtheta_dq({std::cos(phi) * std::cos(theta), std::cos(theta) * std::sin(phi), -std::sin(theta)});
279  vector3d<double> dphi_dq({-std::sin(phi), std::cos(phi), 0.0});
280 
281  SHT::spherical_harmonics(2 * lmax, theta, phi, &rlm_g_(0, igloc));
282 
283  mdarray<double, 1> dRlm_dtheta(lmmax);
284  mdarray<double, 1> dRlm_dphi_sin_theta(lmmax);
285 
286  SHT::dRlm_dtheta(2 * lmax, theta, phi, dRlm_dtheta);
287  SHT::dRlm_dphi_sin_theta(2 * lmax, theta, phi, dRlm_dphi_sin_theta);
288  for (int nu = 0; nu < 3; nu++) {
289  for (int lm = 0; lm < lmmax; lm++) {
290  rlm_dg_(lm, nu, igloc) = dRlm_dtheta[lm] * dtheta_dq[nu] + dRlm_dphi_sin_theta[lm] * dphi_dq[nu];
291  }
292  }
293  }
294  }

296  void generate_pw_coeffs(Atom_type const& atom_type__,
297  Radial_integrals_aug<false> const& ri__,
298  Radial_integrals_aug<true> const& ri_dq__,
299  int nu__,
300  memory_pool& mp__)
301  {
302  PROFILE("sirius::Augmentation_operator_gvec_deriv::generate_pw_coeffs");
303 
304  /* maximum l of beta-projectors */
305  int lmax_beta = atom_type__.indexr().lmax();
306  int lmmax = utils::lmmax(2 * lmax_beta);
307 
308  auto l_by_lm = utils::l_by_lm(2 * lmax_beta);
309 
310  std::vector<double_complex> zilm(lmmax);
311  for (int l = 0, lm = 0; l <= 2 * lmax_beta; l++) {
312  for (int m = -l; m <= l; m++, lm++) {
313  zilm[lm] = std::pow(double_complex(0, 1), l);
314  }
315  }
316 
317  /* split G-vectors between ranks */
318  int gvec_count = gvec_.count();
319  int gvec_offset = gvec_.offset();
320 
321  /* number of beta-projectors */
322  int nbf = atom_type__.mt_basis_size();
323 
324  /* array of plane-wave coefficients */
325  q_pw_ = mdarray<double, 2>(mp__, nbf * (nbf + 1) / 2, 2 * gvec_count, "q_pw_dg_");
326  utils::timer t2("sirius::Augmentation_operator_gvec_deriv::generate_pw_coeffs|qpw");
327  #pragma omp parallel for schedule(static)
328  for (int igloc = 0; igloc < gvec_count; igloc++) {
329  int ig = gvec_offset + igloc;
330  double g = gvec_.gvec_len(ig);
331  auto gvc = gvec_.gvec_cart<index_domain_t::local>(igloc);
332 
333  std::vector<double_complex> v(lmmax);
334 
335  auto ri = ri__.values(atom_type__.id(), g);
336  auto ri_dg = ri_dq__.values(atom_type__.id(), g);
337 
338  for (int xi2 = 0; xi2 < nbf; xi2++) {
339  int lm2 = atom_type__.indexb(xi2).lm;
340  int idxrf2 = atom_type__.indexb(xi2).idxrf;
341 
342  for (int xi1 = 0; xi1 <= xi2; xi1++) {
343  int lm1 = atom_type__.indexb(xi1).lm;
344  int idxrf1 = atom_type__.indexb(xi1).idxrf;
345 
346  /* packed orbital index */
347  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
348  /* packed radial-function index */
349  int idxrf12 = idxrf2 * (idxrf2 + 1) / 2 + idxrf1;
350 
351  for (int lm3 = 0; lm3 < lmmax; lm3++) {
352  v[lm3] = std::conj(zilm[lm3]) * (rlm_dg_(lm3, nu__, igloc) * ri(idxrf12, l_by_lm[lm3]) +
353  rlm_g_(lm3, igloc) * ri_dg(idxrf12, l_by_lm[lm3]) * gvc[nu__]);
354  }
355 
356  double_complex z = fourpi * gaunt_coefs_->sum_L3_gaunt(lm2, lm1, &v[0]);
357 
358  q_pw_(idx12, 2 * igloc) = z.real();
359  q_pw_(idx12, 2 * igloc + 1) = z.imag();
360  }
361  }
362  }
363  t2.stop();
364 
365  memory_t mem{memory_t::host};
366  if (atom_type__.parameters().processing_unit() == device_t::GPU) {
367  mem = memory_t::host_pinned;
368  }
369  sym_weight_ = mdarray<double, 1>(nbf * (nbf + 1) / 2, mem, "sym_weight_");
370  for (int xi2 = 0; xi2 < nbf; xi2++) {
371  for (int xi1 = 0; xi1 <= xi2; xi1++) {
372  /* packed orbital index */
373  int idx12 = xi2 * (xi2 + 1) / 2 + xi1;
374  sym_weight_(idx12) = (xi1 == xi2) ? 1 : 2;
375  }
376  }
377  }
