59  void prepare(stream_id sid, sddk::memory_pool* mp__) const
60  {
61  if (atom_type_.parameters().processing_unit() == device_t::GPU && atom_type_.augment()) {
62  if (mp__) {
63  sym_weight_.allocate(*mp__);
64  q_pw_.allocate(*mp__);
65  } else {
66  sym_weight_.allocate(memory_t::device);
67  q_pw_.allocate(memory_t::device);
68  }
69  sym_weight_.copy_to(memory_t::device, sid);
70  q_pw_.copy_to(memory_t::device, sid);
71  }
72  }
73 
74  void dismiss() const
75  {
76  if (atom_type_.parameters().processing_unit() == device_t::GPU && atom_type_.augment()) {
77  q_pw_.deallocate(memory_t::device);
78  sym_weight_.deallocate(memory_t::device);
79  }
80  }

92  /// Set Q-matrix.
93  void q_mtrx(sddk::mdarray<double, 2> const& q_mtrx__)
94  {
95  int nbf = atom_type_.mt_basis_size();
96  for (int i = 0; i < nbf; i++) {
97  for (int j = 0; j < nbf; j++) {
98  q_mtrx_(j, i) = q_mtrx__(j, i);
99  }
100  }
101  }

114  /// Weight of Q_{\xi,\xi'}.
115  /** 2 if off-diagonal (xi != xi'), 1 if diagonal (xi=xi') */
116  inline double sym_weight(int idx__) const
117  {
118  return sym_weight_(idx__);
119  }
120
