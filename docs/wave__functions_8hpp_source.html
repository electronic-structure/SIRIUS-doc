198  inline void copy_from(wave_functions const& src__,
199  int i0__,
200  int n__,
201  int j0__,
202  device_t pu__)
203  {
204  switch (pu__) {
205  case CPU: {
206  /* copy PW part */
207  std::memcpy(pw_coeffs().prime().at<CPU>(0, j0__),
208  src__.pw_coeffs().prime().at<CPU>(0, i0__),
209  pw_coeffs().num_rows_loc() * n__ * sizeof(double_complex));
210  /* copy MT part */
211  if (has_mt_ && mt_coeffs().num_rows_loc()) {
212  std::memcpy(mt_coeffs().prime().at<CPU>(0, j0__),
213  src__.mt_coeffs().prime().at<CPU>(0, i0__),
214  mt_coeffs().num_rows_loc() * n__ * sizeof(double_complex));
215  }
216  break;
217  }
218  case GPU: {
219  #ifdef __GPU
220  /* copy PW part */
221  acc::copy(pw_coeffs().prime().at<GPU>(0, j0__),
222  src__.pw_coeffs().prime().at<GPU>(0, i0__),
223  pw_coeffs().num_rows_loc() * n__);
224  /* copy MT part */
225  if (has_mt_ && mt_coeffs().num_rows_loc()) {
226  acc::copy(mt_coeffs().prime().at<GPU>(0, j0__),

231  break;
232  }
233  }

241  /// Compute L2 norm of first n wave-functions.
242  inline mdarray<double,1> l2norm(int n__) const
243  {
244  assert(n__ != 0);
245 
246  mdarray<double, 1> norm(n__, memory_t::host, "l2norm");
247  norm.zero();
248  if (pu_ == GPU) {
249  norm.allocate(memory_t::device);
250  norm.zero<memory_t::device>();
251  }
252 
253  if (pu_ == CPU) {
254  #pragma omp parallel for
255  for (int i = 0; i < n__; i++) {
256  for (int ig = 0; ig < pw_coeffs().num_rows_loc(); ig++) {
257  norm[i] += (std::pow(pw_coeffs().prime(ig, i).real(), 2) + std::pow(pw_coeffs().prime(ig, i).imag(), 2));
258  }
259  if (gkvec_.reduced()) {
260  if (comm_.rank() == 0) {
261  norm[i] = 2 * norm[i] - std::pow(pw_coeffs().prime(0, i).real(), 2);
262  } else {
263  norm[i] *= 2;
264  }
265  }
266  if (has_mt_ && mt_coeffs().num_rows_loc()) {
267  for (int j = 0; j < mt_coeffs().num_rows_loc(); j++) {
268  norm[i] += (std::pow(mt_coeffs().prime(j, i).real(), 2) + std::pow(mt_coeffs().prime(j, i).imag(), 2));
269  }
270  }
271  }
272  }
273  #ifdef __GPU
274  if (pu_ == GPU) {
275  add_square_sum_gpu(pw_coeffs().prime().at<GPU>(), pw_coeffs().num_rows_loc(), n__,
276  gkvec_.reduced(), comm_.rank(), norm.at<GPU>());
277  if (has_mt_ && mt_coeffs().num_rows_loc()) {
278  add_square_sum_gpu(mt_coeffs().prime().at<GPU>(), mt_coeffs().num_rows_loc(), n__,
279  0, comm_.rank(), norm.at<GPU>());
280  }
281  norm.copy<memory_t::device, memory_t::host>();
282  }
283  #endif
284 
285  comm_.allreduce(norm.at<CPU>(), n__);
286  for (int i = 0; i < n__; i++) {
287  norm[i] = std::sqrt(norm[i]);
288  }
289 
290  //if (pu_ == GPU) {
291  // norm.copy<memory_t::host, memory_t::device>();
292  //}
293 
294  return std::move(norm);
295  }

312  inline double_complex checksum_pw(int i0__, int n__, device_t pu__)
313  {
314  assert(n__ != 0);
315  double_complex cs(0, 0);
316  switch (pu__) {
317  case CPU: {
318  for (int i = 0; i < n__; i++) {
319  for (int j = 0; j < pw_coeffs().num_rows_loc(); j++) {
320  cs += pw_coeffs().prime(j, i0__ + i);
321  }
322  }
323  break;
324  }
325  case GPU: {
326  mdarray<double_complex, 1> cs1(n__, memory_t::host | memory_t::device, "checksum");
327  cs1.zero<memory_t::device>();
328  #ifdef __GPU
329  add_checksum_gpu(pw_coeffs().prime().at<GPU>(0, i0__), pw_coeffs().num_rows_loc(), n__, cs1.at<GPU>());
330  cs1.copy_to_host();
331  cs = cs1.checksum();
332  #endif
333  break;
334  }
335  }
336  comm_.allreduce(&cs, 1);
337  return cs;
338  }

340  /// Compute the checksum for n wave-functions starting from i0.
341  inline double_complex checksum(int i0__, int n__)
342  {
343  assert(n__ != 0);
344  double_complex cs(0, 0);
345  #ifdef __GPU
346  if (pu_ == GPU) {
347  mdarray<double_complex, 1> cs1(n__, memory_t::host | memory_t::device, "checksum");
348  cs1.zero<memory_t::device>();
349  add_checksum_gpu(pw_coeffs().prime().at<GPU>(0, i0__), pw_coeffs().num_rows_loc(), n__, cs1.at<GPU>());
350  if (has_mt_ && mt_coeffs().num_rows_loc()) {
351  add_checksum_gpu(mt_coeffs().prime().at<GPU>(0, i0__), mt_coeffs().num_rows_loc(), n__, cs1.at<GPU>());
352  }
353  cs1.copy_to_host();
354  cs = cs1.checksum();
355  }
356  #endif
357  if (pu_ == CPU) {
358  for (int i = 0; i < n__; i++) {
359  for (int j = 0; j < pw_coeffs().num_rows_loc(); j++) {
360  cs += pw_coeffs().prime(j, i0__ + i);
361  }
362  }
363  if (has_mt_ && mt_coeffs().num_rows_loc()) {
364  for (int i = 0; i < n__; i++) {
365  for (int j = 0; j < mt_coeffs().num_rows_loc(); j++) {
366  cs += mt_coeffs().prime(j, i0__ + i);
367  }
368  }
369  }
370  }
371  comm_.allreduce(&cs, 1);
372  return cs;
373  }
