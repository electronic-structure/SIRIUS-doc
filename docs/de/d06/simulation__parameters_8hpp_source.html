574  /// Return a reference to a memory pool.
575  /** A memory pool is created when this function called for the first time. */
576  memory_pool& mem_pool(memory_t M__) const
577  {
578  if (memory_pool_.count(M__) == 0) {
579  memory_pool_.emplace(M__, memory_pool(M__));
580  }
581  return memory_pool_.at(M__);
582  }
583 
584  /// Get a default memory pool for a given device.
585  memory_pool& mem_pool(device_t dev__)
586  {
587  switch (dev__) {
588  case device_t::CPU: {
589  return mem_pool(memory_t::host);
590  break;
591  }
592  case device_t::GPU: {
593  return mem_pool(memory_t::device);
594  break;
595  }
596  }
597  return mem_pool(memory_t::host); // make compiler happy
598  }
