290 /// Number of components in the complex density matrix.
291 /** In case of non-collinear magnetism only one out of two non-diagonal components is stored. */
292 inline int num_mag_comp() const // TODO: rename; current name does not reflect the meaning
293 {
294 return (num_mag_dims() == 3) ? 3 : num_spins(); // std::max(mag_dims, spins)
295 }
296
297 /// Number of non-zero spinor components.
298 /** In non magnetic case this is equal to 1, in collinear magnetic case it is also equal to 1 (pure spinors),
299 * in non-collinear case the number of components is 2 (general spinor case). */
300 inline int num_spinor_comp() const
301 {
302 if (num_mag_dims() != 3) {
303 return 1;
304 } else {
305 return 2;
306 }
307 }
308
309 /// Number of spinor wave-functions labeled by a sinlge band index.
310 /** In magnetic collinear case the wave-functions have two spin components, but they describe different
311 * states (pure spin-up, pure spin-dn), thus the number of spinors packed in a single band index is 2.
312 * In non-collinear case we have full two-component spinors for each band index. */
313 inline int num_spinors() const
314 {
315 if (num_mag_dims() == 1) {
316 return 2;
317 } else {
318 return 1;
319 }
320 }

526 /// Return a reference to a memory pool.

528 memory_pool& mem_pool(memory_t M__) const
529 {
530 if (memory_pool_.count(M__) == 0) {
531 memory_pool_.emplace(M__, memory_pool(M__));
532 }
533 return memory_pool_.at(M__);
534 }
535
536 /// Get a default memory pool for a given device.
537 memory_pool& mem_pool(device_t dev__)
538 {
539 switch (dev__) {
540 case device_t::CPU: {
541 return mem_pool(memory_t::host);
542 break;
543 }
544 case device_t::GPU: {
545 return mem_pool(memory_t::device);
546 break;
547 }
548 }
549 return mem_pool(memory_t::host); // make compiler happy
