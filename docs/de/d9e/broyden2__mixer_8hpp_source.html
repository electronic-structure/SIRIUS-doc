44 /// Broyden mixer.
45 /** Second version of the Broyden mixer, which doesn't require inversion of the Jacobian matrix.
46  * Reference paper: "Robust acceleration of self consistent field calculations for
47  * density functional theory", Baarman K, Eirola T, Havu V., J Chem Phys. 134, 134109 (2011)
48  */
49 template <typename... FUNCS>
50 class Broyden2 : public Mixer<FUNCS...>
51 {
52  public:
53  Broyden2(std::size_t max_history, double beta, double beta0, double beta_scaling_factor, double linear_mix_rmse_tol)
54  : Mixer<FUNCS...>(max_history)
55  , beta_(beta)
56  , beta0_(beta0)
57  , beta_scaling_factor_(beta_scaling_factor)
58  , linear_mix_rmse_tol_(linear_mix_rmse_tol)
59  {
60  }
61 
62  void mix_impl() override
63  {
64  const auto idx_step = this->idx_hist(this->step_);
65  const auto idx_next_step = this->idx_hist(this->step_ + 1);
66 
67  const auto history_size = std::min(this->step_, this->max_history_);
68 
69  // beta scaling
70  if (this->step_ > this->max_history_) {
71  const double rmse_avg = std::accumulate(this->rmse_history_.begin(), this->rmse_history_.end(), 0.0) /
72  this->rmse_history_.size();
73  if (this->rmse_history_[idx_step] > rmse_avg) {
74  this->beta_ = std::max(beta0_, this->beta_ * beta_scaling_factor_);
75  }
76  }
77 
78  const double rmse = this->rmse_history_[idx_step];
79 
80  const bool normalize = false;
81 
82  if ((history_size > 1 && rmse < this->linear_mix_rmse_tol_ && this->linear_mix_rmse_tol_ > 0) ||
83  (this->linear_mix_rmse_tol_ <= 0 && this->step_ > this->max_history_)) {
84  sddk::mdarray<double, 2> S(history_size, history_size);
85  S.zero();
86  for (int j1 = 0; j1 < static_cast<int>(history_size); j1++) {
87  int i1 = this->idx_hist(this->step_ - history_size + j1);
88  for (int j2 = 0; j2 <= j1; j2++) {
89  int i2 = this->idx_hist(this->step_ - history_size + j2);
90 
91  S(j2, j1) = S(j1, j2) =
92  this->template inner_product<normalize>(this->residual_history_[i1], this->residual_history_[i2]);
93  }
94  }
95 
96  sddk::mdarray<long double, 2> gamma_k(2 * history_size, history_size);
97  gamma_k.zero();
98  /* initial gamma_0 */
99  for (int i = 0; i < static_cast<int>(history_size); i++) {
100  gamma_k(i, i) = 0.25;
101  }
102 
103  std::vector<long double> v1(history_size);
104  std::vector<long double> v2(2 * history_size);
105 
106  /* update gamma_k by recursion */
107  for (int k = 0; k < static_cast<int>(history_size) - 1; k++) {
108  /* denominator df_k^{T} S df_k */
109  long double d = S(k, k) + S(k + 1, k + 1) - S(k, k + 1) - S(k + 1, k);
110  /* nominator */
111  std::memset(&v1[0], 0, history_size * sizeof(long double));
112  for (int j = 0; j < static_cast<int>(history_size); j++) {
113  v1[j] = S(k + 1, j) - S(k, j);
114  }
115 
116  std::memset(&v2[0], 0, 2 * history_size * sizeof(long double));
117  for (int j = 0; j < static_cast<int>(2 * history_size); j++) {
118  v2[j] = -(gamma_k(j, k + 1) - gamma_k(j, k));
119  }
120  v2[history_size + k] -= 1;
121  v2[history_size + k + 1] += 1;
122 
123  for (int j1 = 0; j1 < static_cast<int>(history_size); j1++) {
124  for (int j2 = 0; j2 < static_cast<int>(2 * history_size); j2++) {
125  gamma_k(j2, j1) += v2[j2] * v1[j1] / d;
126  }
127  }
128  }
129 
130  std::memset(&v2[0], 0, 2 * history_size * sizeof(long double));
131  for (int j = 0; j < static_cast<int>(2 * history_size); j++) {
132  v2[j] = -gamma_k(j, static_cast<int>(history_size - 1));
133  }
134  v2[2 * history_size - 1] += 1;
135 
136  // set input to 0, to use as buffer
137  this->scale(0.0, this->input_);
138 
139  /* make linear combination of vectors and residuals; this is the update vector \tilda x */
140  for (int j = 0; j < static_cast<int>(history_size); j++) {
141  int i1 = this->idx_hist(this->step_ - history_size + j);
142  this->axpy(v2[j], this->residual_history_[i1], this->input_);
143  this->axpy(v2[j + history_size], this->output_history_[i1], this->input_);
144  }
145  }
146  this->copy(this->input_, this->output_history_[idx_next_step]);
147  this->scale(beta_, this->output_history_[idx_next_step]);
148  this->axpy(1.0 - beta_, this->output_history_[idx_step], this->output_history_[idx_next_step]);
149  }
150 
151  private:
152  double beta_;
153  double beta0_;
154  double beta_scaling_factor_;
155  double linear_mix_rmse_tol_;
156 };
