92template <typename T>
93Wave_functions<T>::Wave_functions(const Gvec_partition& gkvecp__, int num_atoms__, std::function<int(int)> mt_size__,
94 int num_wf__, memory_t preferred_memory_t__, int num_sc__)
95 : comm_(gkvecp__.gvec().comm())
96 , gkvecp_(gkvecp__)
97 , num_wf_(num_wf__)
98 , num_sc_(num_sc__)
99 , has_mt_(true)
100 , preferred_memory_t_(preferred_memory_t__)
101{
102 if (!(num_sc__ == 1 || num_sc__ == 2)) {
103 TERMINATE("wrong number of spin components");
104 }
105
106 for (int ispn = 0; ispn < num_sc_; ispn++) {
107 pw_coeffs_[ispn] = std::unique_ptr<matrix_storage<std::complex<T>, matrix_storage_t::slab>>(
108 new matrix_storage<std::complex<T>, matrix_storage_t::slab>(gkvecp_, num_wf_));
109 }
110
111 spl_num_atoms_ = splindex<splindex_t::block>(num_atoms__, comm_.size(), comm_.rank());
112 mt_coeffs_distr_ = block_data_descriptor(comm_.size());
113
114 for (int ia = 0; ia < num_atoms__; ia++) {
115 int rank = spl_num_atoms_.local_rank(ia);
116 if (rank == comm_.rank()) {
117 offset_mt_coeffs_.push_back(mt_coeffs_distr_.counts[rank]);
118 }
119 mt_coeffs_distr_.counts[rank] += mt_size__(ia);
120 }
121 mt_coeffs_distr_.calc_offsets();
122
123 num_mt_coeffs_ = mt_coeffs_distr_.offsets.back() + mt_coeffs_distr_.counts.back();
124
125 for (int ispn = 0; ispn < num_sc_; ispn++) {
126 mt_coeffs_[ispn] = std::unique_ptr<matrix_storage<std::complex<T>, matrix_storage_t::slab>>(
127 new matrix_storage<std::complex<T>, matrix_storage_t::slab>(mt_coeffs_distr_.counts[comm_.rank()], num_wf_));
128 }

131template <typename T>
132void Wave_functions<T>::copy_from(device_t pu__, int n__, const Wave_functions<T>& src__, int ispn__, int i0__,
133 int jspn__, int j0__)
134{
135 assert(ispn__ == 0 || ispn__ == 1);
136 assert(jspn__ == 0 || jspn__ == 1);
137
138 int ngv = pw_coeffs(jspn__).num_rows_loc();
139 int nmt = has_mt() ? mt_coeffs(jspn__).num_rows_loc() : 0;
140
141 switch (pu__) {
142 case device_t::CPU: {
143 /* copy PW part */
144 std::copy(src__.pw_coeffs(ispn__).prime().at(memory_t::host, 0, i0__),
145 src__.pw_coeffs(ispn__).prime().at(memory_t::host, 0, i0__) + ngv * n__,
146 pw_coeffs(jspn__).prime().at(memory_t::host, 0, j0__));
147 /* copy MT part */
148 if (has_mt()) {
149 std::copy(src__.mt_coeffs(ispn__).prime().at(memory_t::host, 0, i0__),
150 src__.mt_coeffs(ispn__).prime().at(memory_t::host, 0, i0__) + nmt * n__,
151 mt_coeffs(jspn__).prime().at(memory_t::host, 0, j0__));
152 }
153 break;
154 }
155 case device_t::GPU: {
156 /* copy PW part */
157 acc::copy(pw_coeffs(jspn__).prime().at(memory_t::device, 0, j0__),
158 src__.pw_coeffs(ispn__).prime().at(memory_t::device, 0, i0__), ngv * n__);
159 /* copy MT part */
160 if (has_mt()) {
161 acc::copy(mt_coeffs(jspn__).prime().at(memory_t::device, 0, j0__),
162 src__.mt_coeffs(ispn__).prime().at(memory_t::device, 0, i0__), nmt * n__);
163 }
164 break;
165 }
166 }

186template <typename T>
187std::complex<T> Wave_functions<T>::checksum_pw(device_t pu__, int ispn__, int i0__, int n__) const
188{
189 assert(n__ != 0);
190 std::complex<T> cs(0, 0);
191 for (int s = s0(ispn__); s <= s1(ispn__); s++) {
192 cs += pw_coeffs(s).checksum(pu__, i0__, n__);
193 }
194 comm_.allreduce(&cs, 1);
195 return cs;
196}

198template <typename T>
199std::complex<T> Wave_functions<T>::checksum_mt(device_t pu__, int ispn__, int i0__, int n__) const
200{
201 assert(n__ != 0);
202 std::complex<T> cs(0, 0);
203 if (!this->has_mt_) {
204 return cs;
205 }
206 for (int s = s0(ispn__); s <= s1(ispn__); s++) {
207 if (mt_coeffs_distr_.counts[comm_.rank()]) {
208 cs += mt_coeffs(s).checksum(pu__, i0__, n__);
209 }
210 }
211 comm_.allreduce(&cs, 1);
212 return cs;
213}

281template <typename T>
282mdarray<T, 1> Wave_functions<T>::l2norm(device_t pu__, spin_range spins__, int n__) const
283{
284 assert(n__ != 0);
285
286 auto norm = sumsqr(pu__, spins__, n__);
287 for (int i = 0; i < n__; i++) {
288 norm[i] = std::sqrt(norm[i]);
289 }
290
291 return norm;

294template <typename T>
295void Wave_functions<T>::normalize(device_t pu__, spin_range spins__, int n__)
296{
297 auto norm = this->l2norm(pu__, spins__, n__);
298 for (int i = 0; i < n__; i++) {
299 norm[i] = 1.0 / norm[i];
300 }
301 if (pu__ == device_t::GPU) {
302 norm.copy_to(memory_t::device);
303 }
304 for (int ispn : spins__) {
305 switch (pu__) {
306 case device_t::CPU: {
307 #pragma omp parallel for schedule(static)
308 for (int i = 0; i < n__; i++) {
309 for (int ig = 0; ig < this->pw_coeffs(ispn).num_rows_loc(); ig++) {
310 this->pw_coeffs(ispn).prime(ig, i) *= norm[i];
311 }
312 if (this->has_mt()) {
313 for (int j = 0; j < this->mt_coeffs(ispn).num_rows_loc(); j++) {
314 this->mt_coeffs(ispn).prime(j, i) *= norm[i];
315 }
316 }
317 }
318 break;
319 }
320 case device_t::GPU: {
321#if defined(SIRIUS_GPU)
322 scale_matrix_columns_gpu(
323 this->pw_coeffs(ispn).num_rows_loc(), n__,
324 this->pw_coeffs(ispn).prime().at(memory_t::device),
325 norm.at(memory_t::device));
326
327 if (this->has_mt()) {
328 scale_matrix_columns_gpu(
329 this->mt_coeffs(ispn).num_rows_loc(), n__,
330 this->mt_coeffs(ispn).prime().at(memory_t::device),
331 norm.at(memory_t::device));
332 }
333#endif
334 } break;
335 }
336 }

383template <typename T>
384mdarray<T, 1> Wave_functions<T>::sumsqr(device_t pu__, spin_range spins__, int n__) const
385{
386 mdarray<T, 1> s(n__, memory_t::host, "sumsqr");
387 s.zero();
388 if (pu__ == device_t::GPU) {
389 s.allocate(memory_t::device).zero(memory_t::device);
390 }
391
392 for (int is : spins__) {
393 switch (pu__) {
394 case device_t::CPU: {
395 #pragma omp parallel for
396 for (int i = 0; i < n__; i++) {
397 for (int ig = 0; ig < pw_coeffs(is).num_rows_loc(); ig++) {
398 s[i] += (std::pow(pw_coeffs(is).prime(ig, i).real(), 2) +
399 std::pow(pw_coeffs(is).prime(ig, i).imag(), 2));
400 }
401 if (gkvecp_.gvec().reduced()) {
402 if (comm_.rank() == 0) {
403 s[i] = 2 * s[i] - std::pow(pw_coeffs(is).prime(0, i).real(), 2);
404 } else {
405 s[i] *= 2;
406 }
407 }
408 if (has_mt()) {
409 for (int j = 0; j < mt_coeffs(is).num_rows_loc(); j++) {
410 s[i] += (std::pow(mt_coeffs(is).prime(j, i).real(), 2) +
411 std::pow(mt_coeffs(is).prime(j, i).imag(), 2));
412 }
413 }
414 }
415 break;
416 }
417 case device_t::GPU: {
418#if defined(SIRIUS_GPU)
419 add_square_sum_gpu(
420 pw_coeffs(is).prime().at(memory_t::device),
421 pw_coeffs(is).num_rows_loc(), n__, gkvecp_.gvec().reduced(), comm_.rank(),
422 s.at(memory_t::device));
423 if (has_mt()) {
424 add_square_sum_gpu(
425 mt_coeffs(is).prime().at(memory_t::device),
426 mt_coeffs(is).num_rows_loc(), n__, 0, comm_.rank(),
427 s.at(memory_t::device));
428 }
429#endif
430 break;
431 }
432 }
433 }
434 if (pu__ == device_t::GPU) {
435 s.copy_to(memory_t::host);
436 }
437 comm_.allreduce(s.at(memory_t::host), n__);
438 return s;
439}
