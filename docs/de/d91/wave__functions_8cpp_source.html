106 void Wave_functions::copy_from(device_t pu__, int n__, const Wave_functions& src__, int ispn__, int i0__,
107  int jspn__, int j0__)
108 {
109  assert(ispn__ == 0 || ispn__ == 1);
110  assert(jspn__ == 0 || jspn__ == 1);
111 
112  int ngv = pw_coeffs(jspn__).num_rows_loc();
113  int nmt = has_mt() ? mt_coeffs(jspn__).num_rows_loc() : 0;
114 
115  switch (pu__) {
116  case device_t::CPU: {
117  /* copy PW part */
118  std::copy(src__.pw_coeffs(ispn__).prime().at(memory_t::host, 0, i0__),
119  src__.pw_coeffs(ispn__).prime().at(memory_t::host, 0, i0__) + ngv * n__,
120  pw_coeffs(jspn__).prime().at(memory_t::host, 0, j0__));
121  /* copy MT part */
122  if (has_mt()) {
123  std::copy(src__.mt_coeffs(ispn__).prime().at(memory_t::host, 0, i0__),
124  src__.mt_coeffs(ispn__).prime().at(memory_t::host, 0, i0__) + nmt * n__,
125  mt_coeffs(jspn__).prime().at(memory_t::host, 0, j0__));
126  }
127  break;
128  }
129  case device_t::GPU: {
130  /* copy PW part */
131  acc::copy(pw_coeffs(jspn__).prime().at(memory_t::device, 0, j0__),
132  src__.pw_coeffs(ispn__).prime().at(memory_t::device, 0, i0__), ngv * n__);
133  /* copy MT part */
134  if (has_mt()) {
135  acc::copy(mt_coeffs(jspn__).prime().at(memory_t::device, 0, j0__),
136  src__.mt_coeffs(ispn__).prime().at(memory_t::device, 0, i0__), nmt * n__);
137  }
138  break;
139  }
140  }

261 void Wave_functions::normalize(device_t pu__, spin_range spins__, int n__)
262 {
263  auto norm = this->l2norm(pu__, spins__, n__);
264  for (int i = 0; i < n__; i++) {
265  norm[i] = 1.0 / norm[i];
266  }
267  if (pu__ == device_t::GPU) {
268  norm.copy_to(memory_t::device);
269  }
270  for (int ispn : spins__) {
271  switch (pu__) {
272  case device_t::CPU: {
273  #pragma omp parallel for schedule(static)
274  for (int i = 0; i < n__; i++) {
275  for (int ig = 0; ig < this->pw_coeffs(ispn).num_rows_loc(); ig++) {
276  this->pw_coeffs(ispn).prime(ig, i) *= norm[i];
277  }
278  if (this->has_mt()) {
279  for (int j = 0; j < this->mt_coeffs(ispn).num_rows_loc(); j++) {
280  this->mt_coeffs(ispn).prime(j, i) *= norm[i];
281  }
282  }
283  }
284  break;
285  }
286  case device_t::GPU: {
287 #if defined(SIRIUS_GPU)
288  scale_matrix_columns_gpu(this->pw_coeffs(ispn).num_rows_loc(), n__,
289  (acc_complex_double_t*)this->pw_coeffs(ispn).prime().at(memory_t::device),
290  norm.at(memory_t::device));
291 
292  if (this->has_mt()) {
293  scale_matrix_columns_gpu(this->mt_coeffs(ispn).num_rows_loc(), n__,
294  (acc_complex_double_t*)this->mt_coeffs(ispn).prime().at(memory_t::device),
295  norm.at(memory_t::device));
296  }
297 #endif
298  } break;
299  }
300  }

343 mdarray<double, 1>
344 Wave_functions::sumsqr(device_t pu__, spin_range spins__, int n__) const
345 {
346  mdarray<double, 1> s(n__, memory_t::host, "sumsqr");
347  s.zero();
348  if (pu__ == device_t::GPU) {
349  s.allocate(memory_t::device).zero(memory_t::device);
350  }
351 
352  for (int is : spins__) {
353  switch (pu__) {
354  case device_t::CPU: {
355  #pragma omp parallel for
356  for (int i = 0; i < n__; i++) {
357  for (int ig = 0; ig < pw_coeffs(is).num_rows_loc(); ig++) {
358  s[i] += (std::pow(pw_coeffs(is).prime(ig, i).real(), 2) +
359  std::pow(pw_coeffs(is).prime(ig, i).imag(), 2));
360  }
361  if (gkvecp_.gvec().reduced()) {
362  if (comm_.rank() == 0) {
363  s[i] = 2 * s[i] - std::pow(pw_coeffs(is).prime(0, i).real(), 2);
364  } else {
365  s[i] *= 2;
366  }
367  }
368  if (has_mt()) {
369  for (int j = 0; j < mt_coeffs(is).num_rows_loc(); j++) {
370  s[i] += (std::pow(mt_coeffs(is).prime(j, i).real(), 2) +
371  std::pow(mt_coeffs(is).prime(j, i).imag(), 2));
372  }
373  }
374  }
375  break;
376  }
377  case device_t::GPU: {
378 #if defined(SIRIUS_GPU)
379  add_square_sum_gpu(pw_coeffs(is).prime().at(memory_t::device), pw_coeffs(is).num_rows_loc(), n__,
380  gkvecp_.gvec().reduced(), comm_.rank(), s.at(memory_t::device));
381  if (has_mt()) {
382  add_square_sum_gpu(mt_coeffs(is).prime().at(memory_t::device), mt_coeffs(is).num_rows_loc(), n__, 0,
383  comm_.rank(), s.at(memory_t::device));
384  }
385 #endif
386  break;
387  }
388  }
389  }
390  if (pu__ == device_t::GPU) {
391  s.copy_to(memory_t::host);
392  }
393  comm_.allreduce(s.at(memory_t::host), n__);
394  return s;
395 }
