212  /// Apply the muffin-tin part of the Hamiltonian to the apw basis functions of an atom.
213  /** The following matrix is computed:
214  * \f[
215  * b_{L_2 \nu_2}^{\alpha}({\bf G'}) = \sum_{L_1 \nu_1} \sum_{L_3}
216  * a_{L_1\nu_1}^{\alpha}({\bf G'})
217  * \langle u_{\ell_1\nu_1}^{\alpha} | h_{L3}^{\alpha} | u_{\ell_2\nu_2}^{\alpha}
218  * \rangle \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
219  * \f]
220  */
221  template <spin_block_t sblock>
222  void apply_hmt_to_apw(Atom const& atom__,
223  int num_gkvec__,
224  mdarray<double_complex, 2>& alm__,
225  mdarray<double_complex, 2>& halm__) const
226  {
227  auto& type = atom__.type();
228 
229  // TODO: this is k-independent and can in principle be precomputed together with radial integrals if memory is
230  // available
231  // TODO: for spin-collinear case hmt is Hermitian; compute upper triangular part and use zhemm
232  mdarray<double_complex, 2> hmt(type.mt_aw_basis_size(), type.mt_aw_basis_size());
233  /* compute the muffin-tin Hamiltonian */
234  for (int j2 = 0; j2 < type.mt_aw_basis_size(); j2++) {
235  int lm2 = type.indexb(j2).lm;
236  int idxrf2 = type.indexb(j2).idxrf;
237  for (int j1 = 0; j1 < type.mt_aw_basis_size(); j1++) {
238  int lm1 = type.indexb(j1).lm;
239  int idxrf1 = type.indexb(j1).idxrf;
240  hmt(j1, j2) =
241  atom__.radial_integrals_sum_L3<sblock>(idxrf1, idxrf2, gaunt_coefs_->gaunt_vector(lm1, lm2));
242  }
243  }
244  linalg<CPU>::gemm(0, 1, num_gkvec__, type.mt_aw_basis_size(), type.mt_aw_basis_size(), alm__, hmt, halm__);
245  }

247  void apply_o1mt_to_apw(Atom const& atom__,
248  int num_gkvec__,
249  mdarray<double_complex, 2>& alm__,
250  mdarray<double_complex, 2>& oalm__) const
251  {
252  auto& type = atom__.type();
253 
254  for (int j = 0; j < type.mt_aw_basis_size(); j++) {
255  int l = type.indexb(j).l;
256  int lm = type.indexb(j).lm;
257  int idxrf = type.indexb(j).idxrf;
258  for (int order = 0; order < type.aw_order(l); order++) {
259  int j1 = type.indexb().index_by_lm_order(lm, order);
260  int idxrf1 = type.indexr().index_by_l_order(l, order);
261  for (int ig = 0; ig < num_gkvec__; ig++) {
262  oalm__(ig, j) += atom__.symmetry_class().o1_radial_integral(idxrf, idxrf1) * alm__(ig, j1);
263  }
264  }
265  }
266  }

294  /// Setup the Hamiltonian and overlap matrices in APW+lo basis
295  /** The Hamiltonian matrix has the following expression:
296  * \f[
297  * H_{\mu' \mu}=\langle \varphi_{\mu' } | \hat H | \varphi_{\mu } \rangle =
298  * \left( \begin{array}{cc}
299  * H_{\bf G'G} & H_{{\bf G'}j} \\
300  * H_{j'{\bf G}} & H_{j'j}
301  * \end{array} \right)
302  * \f]
303  * APW-APW block:
304  * \f{eqnarray*}{
305  * H_{{\bf G'} {\bf G}}^{\bf k} &=& \sum_{\alpha} \sum_{L'\nu', L\nu} a_{L'\nu'}^{\alpha *}({\bf G'+k})
306  * \langle u_{\ell' \nu'}^{\alpha}Y_{\ell' m'}|\hat h^{\alpha} | u_{\ell \nu}^{\alpha}Y_{\ell m} \rangle
307  * a_{L\nu}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde V_{eff}({\bf
308  * G - G'}) \\
309  * &=& \sum_{\alpha} \sum_{\xi' } a_{\xi'}^{\alpha *}({\bf G'+k})
310  * b_{\xi'}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde
311  * V_{eff}({\bf G - G'})
312  * \f}
313  * APW-lo block:
314  * \f[
315  * H_{{\bf G'} j}^{\bf k} = \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'+k})
316  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j}
317  * Y_{\ell_j m_j} \rangle
318  * \f]
319  * lo-APW block:
320  * \f[
321  * H_{j' {\bf G}}^{\bf k} = \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
322  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G+k})
323  * \f]
324  * lo-lo block:
325  * \f[
326  * H_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
327  * |\hat h^{\alpha_{j}} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle \delta_{\alpha_j
328  * \alpha_{j'}}
329  * \f]
330  *
331  * The overlap matrix has the following expression:
332  * \f[
333  * O_{\mu' \mu} = \langle \varphi_{\mu'} | \varphi_{\mu} \rangle
334  * \f]
335  * APW-APW block:
336  * \f[
337  * O_{{\bf G'} {\bf G}}^{\bf k} = \sum_{\alpha} \sum_{L\nu} a_{L\nu}^{\alpha *}({\bf G'+k})
338  * a_{L\nu}^{\alpha}({\bf G+k}) + \Theta({\bf G-G'})
339  * \f]
340  *
341  * APW-lo block:
342  * \f[
343  * O_{{\bf G'} j}^{\bf k} = \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'+k})
344  * \langle u_{\ell_j \nu'}^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} \rangle
345  * \f]
346  *
347  * lo-APW block:
348  * \f[
349  * O_{j' {\bf G}}^{\bf k} =
350  * \sum_{\nu'} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} | u_{\ell_{j'} \nu'}^{\alpha_{j'}} \rangle
351  * a_{\ell_{j'} m_{j'} \nu'}^{\alpha_{j'}}({\bf G+k})
352  * \f]
353  *
354  * lo-lo block:
355  * \f[
356  * O_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} |
357  * \phi_{\ell_{j}}^{\zeta_{j} \alpha_{j}} \rangle \delta_{\alpha_{j'} \alpha_j}
358  * \delta_{\ell_{j'} \ell_j} \delta_{m_{j'} m_j}
359  * \f]
360  */
361  template <device_t pu, electronic_structure_method_t basis>
362  inline void set_fv_h_o(K_point* kp, dmatrix<double_complex>& h, dmatrix<double_complex>& o) const;

364  /// Apply LAPW Hamiltonain and overlap to the trial wave-functions.
365  /** Check the documentation of Band::set_fv_h_o() for the expressions of Hamiltonian and overlap
366  * matrices and \ref basis for the definition of the LAPW+lo basis.
367  *
368  * For the set of wave-functions expanded in LAPW+lo basis (k-point index is dropped for simplicity)
369  * \f[
370  * \psi_{i} = \sum_{\mu} \phi_{\mu} C_{\mu i}
371  * \f]
372  * where \f$ \mu = \{ {\bf G}, j \} \f$ is a combined index of LAPW and local orbitals we want to contrusct
373  * a subspace Hamiltonian and overlap matrices:
374  * \f[
375  * H_{i' i} = \langle \psi_{i'} | \hat H | \psi_i \rangle =
376  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i} =
377  * \sum_{\mu'} C_{\mu' i'}^{*} h_{\mu' i}(\psi)
378  * \f]
379  * \f[
380  * O_{i' i} = \langle \psi_{i'} | \psi_i \rangle =
381  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i} =
382  * \sum_{\mu'} C_{\mu' i'}^{*} o_{\mu' i}(\psi)
383  * \f]
384  * where
385  * \f[
386  * h_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i}
387  * \f]
388  * and
389  * \f[
390  * o_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i}
391  * \f]
392  * For the APW block of \f$ h_{\mu' i}(\psi) \f$ and \f$ o_{\mu' i}(\psi) \f$ we have:
393  * \f[
394  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
395  * \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i}
396  * \f]
397  * \f[
398  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
399  * \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i}
400  * \f]
401  * and for the lo block:
402  * \f[
403  * h_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
404  * \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i}
405  * \f]
406  * \f[
407  * o_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
408  * \sum_{j} \langle \phi_{j'} | \phi_{j} \rangle C_{j i}
409  * \f]
410  *
411  * APW-APW contribution, muffin-tin part:
412  * \f[
413  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
414  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) b_{\xi'}^{\alpha}({\bf G})
415  * C_{{\bf G} i}
416  * \f]
417  * \f[
418  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
419  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) a_{\xi'}^{\alpha}({\bf G})
420  * C_{{\bf G} i}
421  * \f]
422  * APW-APW contribution, interstitial effective potential part:
423  * \f[
424  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} V({\bf r}) \psi_{i}({\bf r}) d{\bf r}
425  * \f]
426  * This is done by transforming \f$ \psi_i({\bf G}) \f$ to the real space, multiplying by effectvive potential
427  * and step function and transforming the result back to the \f$ {\bf G} \f$ domain.
428  *
429  * APW-APW contribution, interstitial kinetic energy part:
430  * \f[
431  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \Big( -\frac{1}{2} \nabla \Big)
432  * \Big( \nabla \psi_{i}({\bf r}) \Big) d{\bf r}
433  * \f]
434  * and the gradient of the wave-function is computed with FFT as:
435  * \f[
436  * \Big( \nabla \psi_{i}({\bf r}) \Big) = \sum_{\bf G} i{\bf G}e^{i{\bf G}{\bf r}}\psi_i({\bf G})
437  * \f]
438  *
439  * APW-APW contribution, interstitial overlap:
440  * \f[
441  * o_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \psi_{i}({\bf r}) d{\bf r}
442  * \f]
443  *
444  * APW-lo contribution:
445  * \f[
446  * h_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i} =
447  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
448  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j}
449  * Y_{\ell_j m_j} \rangle =
450  * \sum_{j} C_{j i} \sum_{\xi'} a_{\xi'}^{\alpha_j *}({\bf G'}) h_{\xi' \xi_j}^{\alpha_j}
451  * \f]
452  * \f[
453  * o_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i} =
454  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
455  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}| \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle
456  * =
457  * \sum_{j} C_{j i} \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'}) o_{\nu' \zeta_j \ell_j}^{\alpha_j}
458  * \f]
459  * lo-APW contribution:
460  * \f[
461  * h_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
462  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'}
463  * m_{j'}}
464  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
465  * \sum_{\bf G} C_{{\bf G} i} \sum_{\xi} h_{\xi_{j'} \xi}^{\alpha_{j'}} a_{\xi}^{\alpha_{j'}}({\bf G})
466  * \f]
467  * \f[
468  * o_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
469  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'}
470  * m_{j'}}
471  * | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
472  * \sum_{\bf G} C_{{\bf G} i} \sum_{\nu} o_{\zeta_{j'} \nu \ell_{j'}}^{\alpha_{j'}} a_{\ell_{j'} m_{j'}
473  * \nu}^{\alpha_{j'}}({\bf G})
474  * \f]
475  * lo-lo contribution:
476  * \f[
477  * h_{j' i}(\psi) = \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i} = \sum_{j} C_{j i}
478  * h_{\xi_{j'} \xi_j}^{\alpha_j}
479  * \delta_{\alpha_j \alpha_{j'}}
480  * \f]
481  * \f[
