211  /// Apply first-variational LAPW Hamiltonian and overlap matrices.
212  /** Check the documentation of Hamiltonain::set_fv_h_o() for the expressions of Hamiltonian and overlap
213  * matrices and \ref basis for the definition of the LAPW+lo basis.
214  *
215  * For the set of wave-functions expanded in LAPW+lo basis (k-point index is dropped for simplicity)
216  * \f[
217  * \psi_{i} = \sum_{\mu} \phi_{\mu} C_{\mu i}
218  * \f]
219  * where \f$ \mu = \{ {\bf G}, j \} \f$ is a combined index of LAPW and local orbitals we want to contrusct
220  * a subspace Hamiltonian and overlap matrices:
221  * \f[
222  * H_{i' i} = \langle \psi_{i'} | \hat H | \psi_i \rangle =
223  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i} =
224  * \sum_{\mu'} C_{\mu' i'}^{*} h_{\mu' i}(\psi)
225  * \f]
226  * \f[
227  * O_{i' i} = \langle \psi_{i'} | \psi_i \rangle =
228  * \sum_{\mu' \mu} C_{\mu' i'}^{*} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i} =
229  * \sum_{\mu'} C_{\mu' i'}^{*} o_{\mu' i}(\psi)
230  * \f]
231  * where
232  * \f[
233  * h_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \hat H | \phi_{\mu} \rangle C_{\mu i}
234  * \f]
235  * and
236  * \f[
237  * o_{\mu' i}(\psi) = \sum_{\mu} \langle \phi_{\mu'} | \phi_{\mu} \rangle C_{\mu i}
238  * \f]
239  * For the APW block of \f$ h_{\mu' i}(\psi) \f$ and \f$ o_{\mu' i}(\psi) \f$ we have:
240  * \f[
241  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
242  * \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i}
243  * \f]
244  * \f[
245  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
246  * \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i}
247  * \f]
248  * and for the lo block:
249  * \f[
250  * h_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} +
251  * \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i}
252  * \f]
253  * \f[
254  * o_{j' i}(\psi) = \sum_{{\bf G}} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} +
255  * \sum_{j} \langle \phi_{j'} | \phi_{j} \rangle C_{j i}
256  * \f]
257  *
258  * APW-APW contribution, muffin-tin part:
259  * \f[
260  * h_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
261  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) b_{\xi'}^{\alpha}({\bf G})
262  * C_{{\bf G} i}
263  * \f]
264  * \f[
265  * o_{{\bf G'} i}(\psi) = \sum_{{\bf G}} \langle \phi_{\bf G'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
266  * \sum_{{\bf G}} \sum_{\alpha} \sum_{\xi'} a_{\xi'}^{\alpha *}({\bf G'}) a_{\xi'}^{\alpha}({\bf G})
267  * C_{{\bf G} i}
268  * \f]
269  * APW-APW contribution, interstitial effective potential part:
270  * \f[
271  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} V({\bf r}) \psi_{i}({\bf r}) d{\bf r}
272  * \f]
273  * This is done by transforming \f$ \psi_i({\bf G}) \f$ to the real space, multiplying by effectvive potential
274  * and step function and transforming the result back to the \f$ {\bf G} \f$ domain.
275  *
276  * APW-APW contribution, interstitial kinetic energy part:
277  * \f[
278  * h_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \Big( -\frac{1}{2} \nabla \Big)
279  * \Big( \nabla \psi_{i}({\bf r}) \Big) d{\bf r}
280  * \f]
281  * and the gradient of the wave-function is computed with FFT as:
282  * \f[
283  * \Big( \nabla \psi_{i}({\bf r}) \Big) = \sum_{\bf G} i{\bf G}e^{i{\bf G}{\bf r}}\psi_i({\bf G})
284  * \f]
285  *
286  * APW-APW contribution, interstitial overlap:
287  * \f[
288  * o_{{\bf G'} i}(\psi) = \int \Theta({\bf r}) e^{-i{\bf G'}{\bf r}} \psi_{i}({\bf r}) d{\bf r}
289  * \f]
290  *
291  * APW-lo contribution:
292  * \f[
293  * h_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \hat H | \phi_{j} \rangle C_{j i} =
294  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
295  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j}
296  * Y_{\ell_j m_j} \rangle =
297  * \sum_{j} C_{j i} \sum_{\xi'} a_{\xi'}^{\alpha_j *}({\bf G'}) h_{\xi' \xi_j}^{\alpha_j}
298  * \f]
299  * \f[
300  * o_{{\bf G'} i}(\psi) = \sum_{j} \langle \phi_{\bf G'} | \phi_{j} \rangle C_{j i} =
301  * \sum_{j} C_{j i} \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'})
302  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}| \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle
303  * =
304  * \sum_{j} C_{j i} \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'}) o_{\nu' \zeta_j \ell_j}^{\alpha_j}
305  * \f]
306  * lo-APW contribution:
307  * \f[
308  * h_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \hat H | \phi_{\bf G} \rangle C_{{\bf G} i} =
309  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'}
310  * m_{j'}}
311  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
312  * \sum_{\bf G} C_{{\bf G} i} \sum_{\xi} h_{\xi_{j'} \xi}^{\alpha_{j'}} a_{\xi}^{\alpha_{j'}}({\bf G})
313  * \f]
314  * \f[
315  * o_{j' i}(\psi) = \sum_{\bf G} \langle \phi_{j'} | \phi_{\bf G} \rangle C_{{\bf G} i} =
316  * \sum_{\bf G} C_{{\bf G} i} \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'}
317  * m_{j'}}
318  * | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G}) =
319  * \sum_{\bf G} C_{{\bf G} i} \sum_{\nu} o_{\zeta_{j'} \nu \ell_{j'}}^{\alpha_{j'}} a_{\ell_{j'} m_{j'}
320  * \nu}^{\alpha_{j'}}({\bf G})
321  * \f]
322  * lo-lo contribution:
323  * \f[
324  * h_{j' i}(\psi) = \sum_{j} \langle \phi_{j'} | \hat H | \phi_{j} \rangle C_{j i} = \sum_{j} C_{j i}
325  * h_{\xi_{j'} \xi_j}^{\alpha_j}
326  * \delta_{\alpha_j \alpha_{j'}}
327  * \f]
328  * \f[
329  * o_{j' i}(\psi) = \sum_{j} \langle \phi_{j'} | \phi_{j} \rangle C_{j i} = \sum_{j} C_{j i}
330  * o_{\zeta_{j'} \zeta_{j} \ell_j}^{\alpha_j}
331  * \delta_{\alpha_j \alpha_{j'}} \delta_{\ell_j \ell_{j'}} \delta_{m_j m_{j'}}
332  * \f]
333  *
334  * \param [in] apw_only True if only APW-APW block of H and O are applied.
335  * \param [in] phi_is_lo True if input wave-functions are pure local orbitals.
336  * \param [in] N Starting index of wave-functions.
337  * \param [in] n Number of wave-functions to which H and S are applied.
338  * \param [in] phi Input wave-functions.
339  * \param [out] hphi Result of Hamiltonian, applied to wave-functions.
340  * \param [out] ophi Result of overlap operator, applied to wave-functions.
341  */
342  void apply_fv_h_o(bool apw_only__, bool phi_is_lo__, int N__, int n__, sddk::Wave_functions& phi__,
343  sddk::Wave_functions* hphi__, sddk::Wave_functions* ophi__);

345  /// Setup the Hamiltonian and overlap matrices in APW+lo basis
346  /** The Hamiltonian matrix has the following expression:
347  * \f[
348  * H_{\mu' \mu}=\langle \varphi_{\mu' } | \hat H | \varphi_{\mu } \rangle =
349  * \left( \begin{array}{cc}
350  * H_{\bf G'G} & H_{{\bf G'}j} \\
351  * H_{j'{\bf G}} & H_{j'j}
352  * \end{array} \right)
353  * \f]
354  * APW-APW block:
355  * \f{eqnarray*}{
356  * H_{{\bf G'} {\bf G}}^{\bf k} &=& \sum_{\alpha} \sum_{L'\nu', L\nu} a_{L'\nu'}^{\alpha *}({\bf G'+k})
357  * \langle u_{\ell' \nu'}^{\alpha}Y_{\ell' m'}|\hat h^{\alpha} | u_{\ell \nu}^{\alpha}Y_{\ell m} \rangle
358  * a_{L\nu}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde V_{eff}({\bf
359  * G - G'}) \\
360  * &=& \sum_{\alpha} \sum_{\xi' } a_{\xi'}^{\alpha *}({\bf G'+k})
361  * b_{\xi'}^{\alpha}({\bf G+k}) + \frac{1}{2}{\bf G'} {\bf G} \cdot \Theta({\bf G - G'}) + \tilde
362  * V_{eff}({\bf G - G'})
363  * \f}
364  * APW-lo block:
365  * \f[
366  * H_{{\bf G'} j}^{\bf k} = \sum_{L'\nu'} a_{L'\nu'}^{\alpha_j *}({\bf G'+k})
367  * \langle u_{\ell' \nu'}^{\alpha_j}Y_{\ell' m'}|\hat h^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j}
368  * Y_{\ell_j m_j} \rangle
369  * \f]
370  * lo-APW block:
371  * \f[
372  * H_{j' {\bf G}}^{\bf k} = \sum_{L\nu} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
373  * |\hat h^{\alpha_{j'}} | u_{\ell \nu}^{\alpha_{j'}}Y_{\ell m} \rangle a_{L\nu}^{\alpha_{j'}}({\bf G+k})
374  * \f]
375  * lo-lo block:
376  * \f[
377  * H_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} Y_{\ell_{j'} m_{j'}}
378  * |\hat h^{\alpha_{j}} | \phi_{\ell_j}^{\zeta_j \alpha_j} Y_{\ell_j m_j} \rangle \delta_{\alpha_j
379  * \alpha_{j'}}
380  * \f]
381  *
382  * The overlap matrix has the following expression:
383  * \f[
384  * O_{\mu' \mu} = \langle \varphi_{\mu'} | \varphi_{\mu} \rangle
385  * \f]
386  * APW-APW block:
387  * \f[
388  * O_{{\bf G'} {\bf G}}^{\bf k} = \sum_{\alpha} \sum_{L\nu} a_{L\nu}^{\alpha *}({\bf G'+k})
389  * a_{L\nu}^{\alpha}({\bf G+k}) + \Theta({\bf G-G'})
390  * \f]
391  *
392  * APW-lo block:
393  * \f[
394  * O_{{\bf G'} j}^{\bf k} = \sum_{\nu'} a_{\ell_j m_j \nu'}^{\alpha_j *}({\bf G'+k})
395  * \langle u_{\ell_j \nu'}^{\alpha_j} | \phi_{\ell_j}^{\zeta_j \alpha_j} \rangle
396  * \f]
397  *
398  * lo-APW block:
399  * \f[
400  * O_{j' {\bf G}}^{\bf k} =
401  * \sum_{\nu'} \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} | u_{\ell_{j'} \nu'}^{\alpha_{j'}} \rangle
402  * a_{\ell_{j'} m_{j'} \nu'}^{\alpha_{j'}}({\bf G+k})
403  * \f]
404  *
405  * lo-lo block:
406  * \f[
407  * O_{j' j}^{\bf k} = \langle \phi_{\ell_{j'}}^{\zeta_{j'} \alpha_{j'}} |
408  * \phi_{\ell_{j}}^{\zeta_{j} \alpha_{j}} \rangle \delta_{\alpha_{j'} \alpha_j}
409  * \delta_{\ell_{j'} \ell_j} \delta_{m_{j'} m_j}
410  * \f]
411  */
