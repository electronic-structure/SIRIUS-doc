29 void K_point::generate_atomic_wave_functions(const basis_functions_index& index, const int atom, const int offset,
30  const bool hubbard, Wave_functions& phi)
31 {
32  if (index.size() == 0) {
33  return;
34  }
35 
36  int lmax{0};
37  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
38  auto& atom_type = unit_cell_.atom_type(iat);
39  lmax = std::max(lmax, atom_type.lmax_ps_atomic_wf());
40  }
41  lmax = std::max(lmax, unit_cell_.lmax());
42 
43  auto& atom_type = unit_cell_.atom(atom).type();
44  #pragma omp parallel for schedule(static)
45  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
46  /* global index of G+k vector */
47  int igk = this->idxgk(igk_loc);
48  /* vs = {r, theta, phi} */
49  auto vs = SHT::spherical_coordinates(this->gkvec().gkvec_cart<index_domain_t::local>(igk_loc));
50 
51  /* compute real spherical harmonics for G+k vector */
52  std::vector<double> rlm(utils::lmmax(lmax));
53  SHT::spherical_harmonics(lmax, vs[1], vs[2], &rlm[0]);
54 
55  /* need to find the index of the atom type get values of radial
56  integrals for a given G+k vector length */
57  mdarray<double, 1> ri_values;
58  ri_values = ctx_.atomic_wf_ri().values(atom_type.id(), vs[0]);
59 
60  int n{0};
61  for (int xi = 0; xi < index.size();) {
62  int l = index[xi].l;
63  /* index of the radial function */
64  int index_radial_function = index[xi].idxrf;
65 
66  const auto phase = twopi * dot(gkvec().gkvec(igk), unit_cell_.atom(atom).position());
67  const auto phase_factor = std::exp(double_complex(0.0, phase));
68  const auto& atom_type = unit_cell_.atom(atom).type();
69  const auto z = std::pow(double_complex(0, -1), l) * fourpi / std::sqrt(unit_cell_.omega());
70 
71  if (!hubbard) {
72  /* for spin orbit coupling the initial wave functions of a given
73  * shell are full spinors with a different radial part. the
74  * orbitals j = l - 1/2 and j = l + 1/2 are next to each other
75  * in the index structure, so I explicitely compute the
76  * average */
77 
78  for (int m = -l; m <= l; m++) {
79  const int lm = utils::lm(l, m);
80  phi.pw_coeffs(0).prime(igk_loc, n + offset) =
81  z * std::conj(phase_factor) * rlm[lm] * ri_values[index_radial_function];
82  n++;
83  }
84  xi += 2 * l + 1;
85  } else {
86  /* hubbard atomic orbitals */
87 
88  if (atom_type.spin_orbit_coupling()) {
89  /* in that case each atomic orbital has a distinct j and are
90  considered as independent orbitals. */
91  const auto average = (ri_values[atom_type.hubbard_orbital(index_radial_function).rindex()] +
92  ri_values[atom_type.hubbard_orbital(index_radial_function + 1).rindex()]);
93 
94  for (int m = -l; m <= l; m++) {
95  const int lm = utils::lm(l, m);
96  phi.pw_coeffs(1).prime(igk_loc, n + offset + 2 * l + 1) =
97  0.5 * z * std::conj(phase_factor) * rlm[lm] * average;
98  phi.pw_coeffs(0).prime(igk_loc, n + offset) =
99  0.5 * z * std::conj(phase_factor) * rlm[lm] * average;
100  n += 1;
101  }
102  xi += 2 * (2 * l + 1);
103  n += 2 * l + 1;
104  } else {
105 
106  /* it is a one orbital but with degeneracy 2. we should
107  consider the magnetic and non magnetic case */
108  for (int m = -l; m <= l; m++) {
109  const int lm = utils::lm(l, m);
110  phi.pw_coeffs(0).prime(igk_loc, n + offset) =
111  z * std::conj(phase_factor) * rlm[lm] *
112  ri_values[atom_type.hubbard_orbital(index_radial_function).rindex()];
113  /* copy from spin up */
114  if (ctx_.num_mag_dims() == 3) {
115  phi.pw_coeffs(1).prime(igk_loc, n + offset + 2 * l + 1) =
116  phi.pw_coeffs(0).prime(igk_loc, n + offset);
117  }
118  n++;
119  }
120  xi += 2 * l + 1;
121  }
122  }
123  }
124  }
125 }

127 void K_point::compute_gradient_wave_functions(Wave_functions& phi, const int starting_position_i, const int num_wf,
128  Wave_functions& dphi, const int starting_position_j, const int direction)
129 {
130  std::vector<double_complex> qalpha(this->num_gkvec_loc());
131 
132  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
133  auto G = this->gkvec().gkvec_cart<index_domain_t::local>(igk_loc);
134 
135  qalpha[igk_loc] = double_complex(0.0, -G[direction]);
136  }
137 
138  #pragma omp parallel for schedule(static)
139  for (int nphi = 0; nphi < num_wf; nphi++) {
140  for (int ispn = 0; ispn < phi.num_sc(); ispn++) {
141  for (int igk_loc = 0; igk_loc < this->num_gkvec_loc(); igk_loc++) {
142  dphi.pw_coeffs(ispn).prime(igk_loc, nphi + starting_position_j) =
143  qalpha[igk_loc] * phi.pw_coeffs(ispn).prime(igk_loc, nphi + starting_position_i);
144  }
145  }
146  }
147 }
