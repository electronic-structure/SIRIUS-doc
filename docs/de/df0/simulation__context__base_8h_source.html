127  void init_atoms_to_grid_idx()
128  {
129  PROFILE("sirius::Simulation_context_base::init_atoms_to_grid_idx");
130 
131  atoms_to_grid_idx_.resize(unit_cell_.num_atoms());
132 
133  vector3d<double> delta(1.0 / (fft_->grid().size(0) ), 1.0 / (fft_->grid().size(1) ), 1.0 / (fft_->grid().size(2) ));
134 
135  int z_off = fft_->offset_z();
136  vector3d<int> grid_beg(0, 0, z_off);
137  vector3d<int> grid_end(fft_->grid().size(0), fft_->grid().size(1), z_off + fft_->local_size_z());
138 
139  double R = av_atom_radius_; // appRoximate atom radius in bohr
140  std::vector<vector3d<double>> verts_cart{{-R,-R,-R},{R,-R,-R},{-R,R,-R},{R,R,-R},{-R,-R,R},{R,-R,R},{-R,R,R},{R,R,R}};
141 
142  auto bounds_box = [&](vector3d<double> pos)
143  {
144  std::vector<vector3d<double>> verts;
145 
146  for (auto v : verts_cart) {
147  verts.push_back( pos + unit_cell_.get_fractional_coordinates(v) );
148  }
149 
150  std::pair<vector3d<int>,vector3d<int>> bounds_ind;
151 
152  size_t size = verts.size();
153  for (int i : {0,1,2}) {
154  std::sort(verts.begin(), verts.end(), [i](vector3d<double>& a, vector3d<double>& b) { return a[i] < b[i]; });
155  bounds_ind.first[i] = std::max((int)(verts[0][i] / delta[i])-1, grid_beg[i]);
156  bounds_ind.second[i] = std::min((int)(verts[size-1][i] / delta[i])+1, grid_end[i]);
157  }
158 
159  return bounds_ind;
160  };
161 
162  #pragma omp parallel for
163  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
164 
165  std::vector<std::pair<int,double>> atom_to_inds_map;
166 
167  for (int t0 = -1; t0 <= 1; t0++) {
168  for (int t1 = -1; t1 <= 1; t1++) {
169  for (int t2 = -1; t2 <= 1; t2++) {
170  auto position = unit_cell_.atom(ia).position() + vector3d<double>(t0, t1, t2);
171 
172  auto box = bounds_box(position);
173 
174  for (int j0 = box.first[0]; j0 < box.second[0]; j0++) {
175  for (int j1 = box.first[1]; j1 < box.second[1]; j1++) {
176  for (int j2 = box.first[2]; j2 < box.second[2]; j2++) {
177  auto dist = position - vector3d<double>(double(j0)* delta[0], double(j1) * delta[1], double(j2) * delta[2]);
178  auto r = unit_cell_.get_cartesian_coordinates(dist).length();
179  auto ir = fft_->grid().index_by_coord(j0, j1, j2 - z_off);
180 
181  if (r <= R) {
182  atom_to_inds_map.push_back({ir, r});
183  }
184  }
185  }
186  }
187  }
188  }
189  }
190 
191  atoms_to_grid_idx_[ia] = std::move(atom_to_inds_map);
192  }
193  }

387  /// Generate phase factors \f$ e^{i {\bf G} {\bf r}_{\alpha}} \f$ for all atoms of a given type.
388  inline void generate_phase_factors(int iat__, mdarray<double_complex, 2>& phase_factors__) const
389  {
390  PROFILE("sirius::Simulation_context_base::generate_phase_factors");
391 
392  int na = unit_cell_.atom_type(iat__).num_atoms();
393  switch (processing_unit_) {
394  case CPU: {
395  #pragma omp parallel for
396  for (int igloc = 0; igloc < gvec_.count(); igloc++) {
397  int ig = gvec_.offset() + igloc;
398  for (int i = 0; i < na; i++) {
399  int ia = unit_cell_.atom_type(iat__).atom_id(i);
400  phase_factors__(igloc, i) = gvec_phase_factor(ig, ia);
401  }
402  }
403  break;
404  }
405  case GPU: {
406  #ifdef __GPU
407  generate_phase_factors_gpu(gvec_.count(), na, gvec_coord().at<GPU>(), atom_coord(iat__).at<GPU>(),
408  phase_factors__.at<GPU>());
409  #else
410  TERMINATE_NO_GPU
411  #endif
412  break;
413  }
414  }
415  }

417  /// Make periodic function out of form factors.
418  /** Return vector of plane-wave coefficients */
419  template <index_domain_t index_domain>
420  inline std::vector<double_complex> make_periodic_function(std::function<double(int, double)> form_factors__) const
421  {
422  PROFILE("sirius::Simulation_context_base::make_periodic_function");
423 
424  double fourpi_omega = fourpi / unit_cell_.omega();
425 
426  int ngv = (index_domain == index_domain_t::local) ? gvec_.count() : gvec().num_gvec();
427  std::vector<double_complex> f_pw(ngv, double_complex(0, 0));
428 
429  #pragma omp parallel for schedule(static)
430  for (int igloc = 0; igloc < gvec_.count(); igloc++) {
431  /* global index of G-vector */
432  int ig = gvec_.offset() + igloc;
433  double g = gvec().gvec_len(ig);
434 
435  int j = (index_domain == index_domain_t::local) ? igloc : ig;
436  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
437  int iat = unit_cell_.atom(ia).type_id();
438  f_pw[j] += fourpi_omega * std::conj(gvec_phase_factor(ig, ia)) * form_factors__(iat, g);
439  }
440  }
441 
442  if (index_domain == index_domain_t::global) {
443  comm_.allgather(&f_pw[0], gvec_.offset(), gvec_.count());
444  }
445 
446  return std::move(f_pw);
447  }

475  /// Find the lambda parameter used in the Ewald summation.
476  /** lambda parameter scales the erfc function argument:
477  * \f[
478  * {\rm erf}(\sqrt{\lambda}x)
479  * \f]
480  */
481  double ewald_lambda()
482  {
483  /* alpha = 1 / (2*sigma^2), selecting alpha here for better convergence */
484  double lambda{1};
485  double gmax = pw_cutoff();
486  double upper_bound{0};
487  double charge = unit_cell_.num_electrons();
488 
489  /* iterate to find lambda */
490  do {
491  lambda += 0.1;
492  upper_bound = charge * charge * std::sqrt(2.0 * lambda / twopi) * gsl_sf_erfc(gmax * std::sqrt(1.0 / (4.0 * lambda)));
493  } while (upper_bound < 1.0e-8);
494 
495  if (lambda < 1.5) {
496  std::stringstream s;
497  s << "Ewald forces error: probably, pw_cutoff is too small.";
498  WARNING(s);
499  }
500  return lambda;
501  }

509 inline void Simulation_context_base::init_fft()
510 {
511  auto rlv = unit_cell_.reciprocal_lattice_vectors();
512 
513  if (!(control().fft_mode_ == "serial" || control().fft_mode_ == "parallel")) {
514  TERMINATE("wrong FFT mode");
515  }
516 
517  /* create FFT driver for dense mesh (density and potential) */
518  fft_ = std::unique_ptr<FFT3D>(new FFT3D(find_translations(pw_cutoff(), rlv), comm_fft(), processing_unit()));
519 
520  /* create a list of G-vectors for dense FFT grid; G-vectors are divided between all available MPI ranks.*/
521  gvec_ = Gvec(rlv, pw_cutoff(), comm(), comm_fft(), control().reduce_gvec_);
522 
523  remap_gvec_ = std::unique_ptr<remap_gvec_to_shells>(new remap_gvec_to_shells(comm(), gvec()));
524 
525  /* prepare fine-grained FFT driver for the entire simulation */
526  fft_->prepare(gvec_.partition());
527 
528  /* create FFT driver for coarse mesh */
529  auto fft_coarse_grid = FFT3D_grid(find_translations(2 * gk_cutoff(), rlv));
530  //auto pu = (fft_coarse_grid.size() < std::pow(64, 3)) ? CPU : processing_unit();
531  //if (full_potential()) {
532  // pu = processing_unit();
533  //}
534  auto pu = processing_unit();
535  fft_coarse_ = std::unique_ptr<FFT3D>(new FFT3D(fft_coarse_grid, comm_fft_coarse(), pu));
536 
537  /* create a list of G-vectors for corase FFT grid */
538  gvec_coarse_ = Gvec(rlv, gk_cutoff() * 2, comm(), comm_fft_coarse(), control().reduce_gvec_);
539 }

541 inline void Simulation_context_base::initialize()
542 {
543  PROFILE("sirius::Simulation_context_base::initialize");
544 
545  /* can't initialize twice */
546  if (initialized_) {
547  TERMINATE("Simulation parameters are already initialized.");
548  }
549  /* Gamma-point calculation and non-collinear magnetism are not compatible */
550  if (num_mag_dims() == 3) {
551  set_gamma_point(false);
552  }
553 
554  set_esm_type(parameters_input().esm_);
555  set_core_relativity(parameters_input().core_relativity_);
556  set_valence_relativity(parameters_input().valence_relativity_);
557 
558  /* get processing unit */
559  std::string pu = control().processing_unit_;
560  if (pu == "") {
561  #ifdef __GPU
562  pu = "gpu";
563  #else
564  pu = "cpu";
565  #endif
566  }
567  set_processing_unit(pu);
568 
569  /* check if we can use a GPU device */
570  if (processing_unit() == GPU) {
571  #ifndef __GPU
572  TERMINATE_NO_GPU
573  #endif
574  }
575 
576  /* check MPI grid dimensions and set a default grid if needed */
577  if (!control().mpi_grid_dims_.size()) {
578  set_mpi_grid_dims({1, 1});
579  }
580  if (control().mpi_grid_dims_.size() != 2) {
581  TERMINATE("wrong MPI grid");
582  }
583 
584  int npr = control_input_.mpi_grid_dims_[0];
585  int npc = control_input_.mpi_grid_dims_[1];
586  int npb = npr * npc;
587  int npk = comm_.size() / npb;
588  if (npk * npb != comm_.size()) {
589  std::stringstream s;
590  s << "Can't divide " << comm_.size() << " ranks into groups of size " << npb;
591  TERMINATE(s);
592  }
593 
594  /* setup MPI grid */
595  mpi_grid_ = std::unique_ptr<MPI_grid>(new MPI_grid({npr, npc, npk}, comm_));
596 
597  /* setup BLACS grid */
598  blacs_grid_ = std::unique_ptr<BLACS_grid>(new BLACS_grid(comm_band(), npr, npc));
599 
600  /* can't use reduced G-vectors in LAPW code */
601  if (full_potential()) {
602  control_input_.reduce_gvec_ = false;
603  }
604 
605  if (!iterative_solver_input_.type_.size()) {
606  if (full_potential()) {
607  iterative_solver_input_.type_ = "exact";
608  } else {
609  iterative_solver_input_.type_ = "davidson";
610  }
611  }
612 
613  /* initialize variables, related to the unit cell */
614  unit_cell_.initialize();
615 
616  /* find the cutoff for G+k vectors (derived from rgkmax (aw_cutoff here) and minimum MT radius) */
617  if (full_potential()) {
618  set_gk_cutoff(aw_cutoff() / unit_cell_.min_mt_radius());
619  }
620 
621  if (!full_potential()) {
622  set_lmax_rho(unit_cell_.lmax() * 2);
623  set_lmax_pot(unit_cell_.lmax() * 2);
624  set_lmax_apw(-1);
625  }
626 
627  /* initialize FFT interface */
628  init_fft();
629 
630  init_atoms_to_grid_idx();
631 
632  if (comm_.rank() == 0 && control().print_memory_usage_) {
633  MEMORY_USAGE_INFO();
634  }
635 
636  if (unit_cell_.num_atoms() != 0 && use_symmetry()) {
637  unit_cell_.symmetry().check_gvec_symmetry(gvec_, comm_);
638  if (!full_potential()) {
639  unit_cell_.symmetry().check_gvec_symmetry(gvec_coarse_, comm_);
640  }
641  }
642 
643  auto& fft_grid = fft().grid();
644  std::pair<int, int> limits(0, 0);
645  for (int x: {0, 1, 2}) {
646  limits.first = std::min(limits.first, fft_grid.limits(x).first);
647  limits.second = std::max(limits.second, fft_grid.limits(x).second);
648  }
649 
650  phase_factors_ = mdarray<double_complex, 3>(3, limits, unit_cell().num_atoms(), memory_t::host, "phase_factors_");
651 
652  #pragma omp parallel for
653  for (int i = limits.first; i <= limits.second; i++) {
654  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
655  auto pos = unit_cell_.atom(ia).position();
656  for (int x: {0, 1, 2}) {
