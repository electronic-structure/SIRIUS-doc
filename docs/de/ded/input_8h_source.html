35 /// Parse unit cell input section.
36 /** The following part of the input file is parsed:
37  * \code{.json}
38  * "unit_cell" : {
39  * "lattice_vectors" : [
40  * [a1_x, a1_y, a1_z],
41  * [a2_x, a2_y, a2_z],
42  * [a3_x, a3_y, a3_z]
43  * ],
44  *
45  * "lattice_vectors_scale" : scale,
46  *
47  * "atom_types" : [label_A, label_B, ...],
48  *
49  * "atom_files" : {
50  * label_A : file_A,
51  * label_B : file_B,
52  * ...
53  * },
54  *
55  * "atom_coordinate_units" : units,
56  *
57  * "atoms" : {
58  * label_A: [
59  * coordinates_A_1,
60  * coordinates_A_2,
61  * ...
62  * ],
63  * label_B : [
64  * coordinates_B_1,
65  * coordinates_B_2,
66  * ...
67  * ]
68  * }
69  * }
70  * \endcode
71  *
72  * The "atom_coordinate_units" string is optional. By default it is assumed to be "lattice" which means that the
73  * atomic coordinates are provided in lattice (fractional) units. It can also be specified in "A" or "au" which
74  * means that the input atomic coordinates are Cartesian and provided in Angstroms or atomic units of length.
75  * This is useful in setting up the molecule calculation.
76  */
77 struct Unit_cell_input
78 {
79  /// First vector of the unit cell.
80  vector3d<double> a0_;
81 
82  /// Second vector of the unit cell.
83  vector3d<double> a1_;
84 
85  /// Third vector of the unit cell.
86  vector3d<double> a2_;
87 
88  /// Labels of the atom types.
89  std::vector<std::string> labels_;
90 
91  /// Mapping between a label of atom type and corresponding atomic species file.
92  std::map<std::string, std::string> atom_files_;
93 
94  /// Atomic coordinates.
95  /** Outer vector size is equal to the number of atom types. */
96  std::vector<std::vector<std::vector<double>>> coordinates_;
97 
98  /// True if this section exists in the input file.
99  bool exist_{false};
100 
101  /// Read the \b unit_cell input section.
102  void read(json const& parser)
103  {
104  if (parser.count("unit_cell")) {
105  exist_ = true;
106 
107  auto section = parser["unit_cell"];
108  auto a0 = section["lattice_vectors"][0].get<std::vector<double>>();
109  auto a1 = section["lattice_vectors"][1].get<std::vector<double>>();
110  auto a2 = section["lattice_vectors"][2].get<std::vector<double>>();
111 
112  if (a0.size() != 3 || a1.size() != 3 || a2.size() != 3) {
113  TERMINATE("wrong lattice vectors");
114  }
115 
116  double scale = section.value("lattice_vectors_scale", 1.0);
117 
118  for (int x : {0, 1, 2}) {
119  a0_[x] = a0[x] * scale;
120  a1_[x] = a1[x] * scale;
121  a2_[x] = a2[x] * scale;
122  }
123 
124  matrix3d<double> lv;
125  for (int x : {0, 1, 2}) {
126  lv(x, 0) = a0_[x];
127  lv(x, 1) = a1_[x];
128  lv(x, 2) = a2_[x];
129  }
130  auto ilv = inverse(lv);
131 
132  labels_.clear();
133  coordinates_.clear();
134 
135  std::string units = section.value("atom_coordinate_units", "lattice");
136 
137  for (auto& label : section["atom_types"]) {
138  if (std::find(std::begin(labels_), std::end(labels_), label) != std::end(labels_)) {
139  TERMINATE("duplicate atom type label");
140  }
141  labels_.push_back(label);
142  }
143 
144  if (section.count("atom_files")) {
145  for (auto& label : labels_) {
146  atom_files_[label] = section["atom_files"].value(label, "");
147  }
148  }
149 
150  for (int iat = 0; iat < (int)labels_.size(); iat++) {
151  coordinates_.push_back(std::vector<std::vector<double>>());
152  for (size_t ia = 0; ia < section["atoms"][labels_[iat]].size(); ia++) {
153  auto v = section["atoms"][labels_[iat]][ia].get<std::vector<double>>();
154 
155  if (!(v.size() == 3 || v.size() == 6)) {
156  TERMINATE("wrong coordinates size");
157  }
158  if (v.size() == 3) {
159  v.resize(6, 0.0);
160  }
161 
162  vector3d<double> v1(v[0], v[1], v[2]);
163  if (units == "A") {
164  for (int x : {0, 1, 2}) {
165  v1[x] /= bohr_radius;
166  }
167  }
168  if (units == "au" || units == "A") {
169  v1 = ilv * v1;
170  auto rv1 = reduce_coordinates(v1);
171  for (int x : {0, 1, 2}) {
172  v[x] = rv1.first[x];
173  }
174  }
175 
176  coordinates_[iat].push_back(v);
177  }
178  }
179  }
180  }
181 };

224 /// Parse the parameters of iterative solver.
225 struct Iterative_solver_input
226 {
227  /// Type of the iterative solver.
228  std::string type_{""};
229 
230  /// Number of steps (iterations) of the solver.
231  int num_steps_{20};
232 
233  /// Size of the variational subspace is this number times the number of bands.
234  int subspace_size_{4};
235 
236  /// Tolerance for the eigen-energy difference \f$ |\epsilon_i^{old} - \epsilon_i^{new} | \f$.
237  /** This parameter is reduced during the SCF cycle to reach the high accuracy of the wave-functions. */
238  double energy_tolerance_{1e-2};
239 
240  /// Tolerance for the residual L2 norm.
241  double residual_tolerance_{1e-6};
242 
243  /// Additional tolerance for empty states.
244  /** Setting this variable to 0 will treat empty states with the same tolerance as occupied states. */
245  double empty_states_tolerance_{1e-5};
246 
247  /// Defines the flavour of the iterative solver.
248  /** If converge_by_energy is set to 0, then the residuals are estimated by their norm. If converge_by_energy
249  * is set to 1 then the residuals are estimated by the eigen-energy difference. This allows to estimate the
250  * unconverged residuals and only then compute only the unconverged. */
251  int converge_by_energy_{1}; // TODO: rename, this is meaningless
252 
253  /// Minimum number of residuals to continue iterative diagonalization process.
254  int min_num_res_{0};
255 
256  /// Number of singular components for the LAPW Davidson solver.
257  int num_singular_{-1};
258 
259  /// Control the subspace expansion.
260  /** If true, keep basis orthogonal and solve standard eigen-value problem. If false, add preconditioned residuals
261  * as they are and solve generalized eigen-value problem. */
262  bool orthogonalize_{true};
263 
264  /// Initialize eigen-values with previous (old) values.
265  bool init_eval_old_{true};
266 
267  /// Tell how to initialize the subspace.
268  /** It can be either "lcao", i.e. start from the linear combination of atomic orbitals or "random" –- start from
269  * the randomized wave functions. */
270  std::string init_subspace_{"lcao"};
271 
272  void read(json const& parser)
273  {
274  if (parser.count("iterative_solver")) {
275  auto section = parser["iterative_solver"];
276  type_ = section.value("type", type_);
277  num_steps_ = section.value("num_steps", num_steps_);
278  subspace_size_ = section.value("subspace_size", subspace_size_);
279  energy_tolerance_ = section.value("energy_tolerance", energy_tolerance_);
280  residual_tolerance_ = section.value("residual_tolerance", residual_tolerance_);
281  empty_states_tolerance_ = section.value("empty_states_tolerance", empty_states_tolerance_);
282  converge_by_energy_ = section.value("converge_by_energy", converge_by_energy_);
283  min_num_res_ = section.value("min_num_res", min_num_res_);
284  num_singular_ = section.value("num_singular", num_singular_);
285  orthogonalize_ = section.value("orthogonalize", orthogonalize_);
286  init_eval_old_ = section.value("init_eval_old", init_eval_old_);
287  init_subspace_ = section.value("init_subspace", init_subspace_);
288  std::transform(init_subspace_.begin(), init_subspace_.end(), init_subspace_.begin(), ::tolower);
289  }
290  }

293 /// Parse control input section.
294 /** The following part of the input file is parsed:
295  * \code{.json}
296  * "control" : {
297  * "mpi_grid_dims" : (1- 2- or 3-dimensional vector<int>) MPI grid layout
298  * "cyclic_block_size" : (int) PBLAS / ScaLAPACK block size
299  * "reduce_gvec" : (bool) use reduced G-vector set (reduce_gvec = true) or full set (reduce_gvec = false)
300  * "std_evp_solver_type" : (string) type of eigen-solver for the standard eigen-problem
301  * "gen_evp_solver_type" : (string) type of eigen-solver for the generalized eigen-problem
302  * "electronic_structure_method" : (string) electronic structure method
303  * "processing_unit" : (string) primary processing unit
304  * "fft_mode" : (string) serial or parallel FFT
305  * }
306  * \endcode
307  * Parameters of the control input sections do not in general change the numerics, but instead control how the
308  * results are obtained. Changing paremeters in control section should not change the significant digits in final
309  * results.
310  */
311 struct Control_input
312 {
313  /// Dimensions of the MPI grid (if used).
314  std::vector<int> mpi_grid_dims_;
315 
316  /// Block size for ScaLAPACK and ELPA.
317  int cyclic_block_size_{-1};
318 
319  /// Reduce G-vectors by inversion symmetry.
320  /** For real-valued functions like density and potential it is sufficient to store only half of the G-vectors
321  * and use the relation f(G) = f^{*}(-G) to recover second half of the plane-wave expansion coefficients. */
322  bool reduce_gvec_{true};
323 
324  /// Standard eigen-value solver to use.
325  std::string std_evp_solver_name_{""};
326 
327  /// Generalized eigen-value solver to use.
328  std::string gen_evp_solver_name_{""};
329 
330  /// Coarse grid FFT mode ("serial" or "parallel").
331  std::string fft_mode_{"serial"};
332 
333  /// Main processing unit to run on.
334  std::string processing_unit_{""};
335 
336  /// Maximum allowed muffin-tin radius in case of LAPW.
337  double rmt_max_{2.2};
338 
339  /// Tolerance of the spglib in finding crystal symmetries.
340  double spglib_tolerance_{1e-4};
341 
342  /// Level of verbosity.
343  /** The following convention in proposed:
344  * - 0: silent mode (no output is printed) \n
345  * - 1: basic output (low level of output) \n
346  * - 2: extended output (medium level of output) \n
347  * - 3: extensive output (hi level of output) */
348 #if defined(__VERBOSITY)
349  int verbosity_{__VERBOSITY};
350 #else
351  int verbosity_{0};
352 #endif
353 
354  /// Level of internal verification.
355  int verification_{0};
356  int num_bands_to_print_{10};
357  bool print_performance_{false};
358  bool print_memory_usage_{false};
359  bool print_checksum_{false};
360  bool print_hash_{false};
361  bool print_stress_{false};
362  bool print_forces_{false};
363  bool print_timers_{true};
364  bool print_neighbors_{false};
365
