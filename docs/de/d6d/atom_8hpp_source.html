141 /// Generate radial Hamiltonian and effective magnetic field integrals
142 /** Hamiltonian operator has the following representation inside muffin-tins:
143 * \f[
144 * \hat H = -\frac{1}{2}\nabla^2 + \sum_{\ell m} V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) =
145 * \underbrace{-\frac{1}{2} \nabla^2+V_{00}(r)R_{00}}_{H_{s}(r)} +\sum_{\ell=1} \sum_{m=-\ell}^{\ell}
146 * V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) = \sum_{\ell m} \widetilde V_{\ell m}(r) R_{\ell m}(\hat {\bf r})
147 * \f]
148 * where
149 * \f[
150 * \widetilde V_{\ell m}(r) = \left\{ \begin{array}{ll}
151 * \frac{H_{s}(r)}{R_{00}} & \ell = 0 \\
152 * V_{\ell m}(r) & \ell > 0 \end{array} \right.
153 * \f]
154 */
155 inline void generate_radial_integrals(device_t pu__, Communicator const& comm__)
156 {
157 PROFILE("sirius::Atom::generate_radial_integrals");
158
159 int lmmax = utils::lmmax(lmax_pot_);
160 int nmtp = type().num_mt_points();
161 int nrf = type().indexr().size();
162 int num_mag_dims = type().parameters().num_mag_dims();
163
164 if (comm__.size() != 1) {
165 TERMINATE("not yet mpi parallel");
166 }
167
168 splindex<splindex_t::block> spl_lm(lmmax, comm__.size(), comm__.rank());
169
170 auto l_by_lm = utils::l_by_lm(lmax_pot_);
171
172 h_radial_integrals_.zero();
173 if (num_mag_dims) {
174 b_radial_integrals_.zero();
175 }
176
177 /* copy radial functions to spline objects */
178 std::vector<Spline<double>> rf_spline(nrf);
179 #pragma omp parallel for
180 for (int i = 0; i < nrf; i++) {
181 rf_spline[i] = Spline<double>(type().radial_grid());
182 for (int ir = 0; ir < nmtp; ir++) {
183 rf_spline[i](ir) = symmetry_class().radial_function(ir, i);
184 }
185 }
186
187 /* copy effective potential components to spline objects */
188 std::vector<Spline<double>> v_spline(lmmax * (1 + num_mag_dims));
189 #pragma omp parallel for
190 for (int lm = 0; lm < lmmax; lm++) {
191 v_spline[lm] = Spline<double>(type().radial_grid());
192 for (int ir = 0; ir < nmtp; ir++) {
193 v_spline[lm](ir) = veff_(lm, ir);
194 }
195
196 for (int j = 0; j < num_mag_dims; j++) {
197 v_spline[lm + (j + 1) * lmmax] = Spline<double>(type().radial_grid());
198 for (int ir = 0; ir < nmtp; ir++) {
199 v_spline[lm + (j + 1) * lmmax](ir) = beff_[j](lm, ir);
200 }
201 }
202 }
203
204 /* interpolate potential multiplied by a radial function */
205 std::vector<Spline<double>> vrf_spline(lmmax * nrf * (1 + num_mag_dims));
206
207 auto& idx_ri = type().idx_radial_integrals();
208
209 mdarray<double, 1> result(idx_ri.size(1));
210
211 if (pu__ == device_t::GPU) {
212#ifdef SIRIUS_GPU
213 auto& rgrid = type().radial_grid();
214 auto& rf_coef = type().rf_coef();
215 auto& vrf_coef = type().vrf_coef();
216
217 PROFILE_START("sirius::Atom::generate_radial_integrals|interp");
218 #pragma omp parallel
219 {
220 #pragma omp for
221 for (int i = 0; i < nrf; i++) {
222 rf_spline[i].interpolate();
223 std::copy(rf_spline[i].coeffs().at(memory_t::host),
224 rf_spline[i].coeffs().at(memory_t::host) + nmtp * 4,
225 rf_coef.at(memory_t::host, 0, 0, i));
226 // cuda_async_copy_to_device(rf_coef.at<GPU>(0, 0, i), rf_coef.at<CPU>(0, 0, i), nmtp * 4 *
227 // sizeof(double), tid);
228 }
229 #pragma omp for
230 for (int i = 0; i < lmmax * (1 + num_mag_dims); i++) {
231 v_spline[i].interpolate();
232 }
233 }
234 rf_coef.copy_to(memory_t::device, stream_id(-1));
235
236 #pragma omp parallel for
237 for (int lm = 0; lm < lmmax; lm++) {
238 for (int i = 0; i < nrf; i++) {
239 for (int j = 0; j < num_mag_dims + 1; j++) {
240 int idx = lm + lmmax * i + lmmax * nrf * j;
241 vrf_spline[idx] = rf_spline[i] * v_spline[lm + j * lmmax];
242 std::memcpy(vrf_coef.at(memory_t::host, 0, 0, idx), vrf_spline[idx].coeffs().at(memory_t::host),
243 nmtp * 4 * sizeof(double));
244 // cuda_async_copy_to_device(vrf_coef.at<GPU>(0, 0, idx), vrf_coef.at<CPU>(0, 0, idx), nmtp * 4
245 // *sizeof(double), tid);
246 }
247 }
248 }
249 vrf_coef.copy_to(memory_t::device);
250 PROFILE_STOP("sirius::Atom::generate_radial_integrals|interp");
251
252 result.allocate(memory_t::device);
253 spline_inner_product_gpu_v3(idx_ri.at(memory_t::device), (int)idx_ri.size(1), nmtp,
254 rgrid.x().at(memory_t::device), rgrid.dx().at(memory_t::device),
255 rf_coef.at(memory_t::device), vrf_coef.at(memory_t::device),
256 result.at(memory_t::device));
257 acc::sync();
258 //if (type().parameters().control().print_performance_) {
259 // double tval = t2.stop();
260 // DUMP("spline GPU integration performance: %12.6f GFlops",
261 // 1e-9 * double(idx_ri.size(1)) * nmtp * 85 / tval);
262 //}
263 result.copy_to(memory_t::host);
264 result.deallocate(memory_t::device);
265#endif
266 }
267 if (pu__ == device_t::CPU) {
268 PROFILE_START("sirius::Atom::generate_radial_integrals|interp");
269 #pragma omp parallel
270 {
271 #pragma omp for
272 for (int i = 0; i < nrf; i++) {
273 rf_spline[i].interpolate();
274 }
275 #pragma omp for
276 for (int i = 0; i < lmmax * (1 + num_mag_dims); i++) {
277 v_spline[i].interpolate();
278 }
279
280 #pragma omp for
281 for (int lm = 0; lm < lmmax; lm++) {
282 for (int i = 0; i < nrf; i++) {
283 for (int j = 0; j < num_mag_dims + 1; j++) {
284 vrf_spline[lm + lmmax * i + lmmax * nrf * j] = rf_spline[i] * v_spline[lm + j * lmmax];
285 }
286 }
287 }
288 }
289 PROFILE_STOP("sirius::Atom::generate_radial_integrals|interp");
290
291 PROFILE("sirius::Atom::generate_radial_integrals|inner");
292 #pragma omp parallel for
293 for (int j = 0; j < (int)idx_ri.size(1); j++) {
294 result(j) = inner(rf_spline[idx_ri(0, j)], vrf_spline[idx_ri(1, j)], 2);
295 }
296 //if (type().parameters().control().print_performance_) {
297 // double tval = t2.stop();
298 // DUMP("spline CPU integration performance: %12.6f GFlops",
299 // 1e-9 * double(idx_ri.size(1)) * nmtp * 85 / tval);
300 //}
301 }
302
303 int n{0};
304 for (int lm = 0; lm < lmmax; lm++) {
305 int l = l_by_lm[lm];
306
307 for (int i2 = 0; i2 < type().indexr().size(); i2++) {
308 int l2 = type().indexr(i2).l;
309 for (int i1 = 0; i1 <= i2; i1++) {
310 int l1 = type().indexr(i1).l;
311 if ((l + l1 + l2) % 2 == 0) {
312 if (lm) {
313 h_radial_integrals_(lm, i1, i2) = h_radial_integrals_(lm, i2, i1) = result(n++);
314 } else {
315 h_radial_integrals_(0, i1, i2) = symmetry_class().h_spherical_integral(i1, i2);
316 h_radial_integrals_(0, i2, i1) = symmetry_class().h_spherical_integral(i2, i1);
317 }
318 for (int j = 0; j < num_mag_dims; j++) {
319 b_radial_integrals_(lm, i1, i2, j) = b_radial_integrals_(lm, i2, i1, j) = result(n++);
320 }
321 }
322 }
323 }
324 }
325
326 //if (type().parameters().control().print_checksum_) {
327 // DUMP("checksum(h_radial_integrals): %18.10f", h_radial_integrals_.checksum());
328 //}
329 }

431 /** Compute the following kinds of sums for different spin-blocks of the Hamiltonian:
432 * \f[
433 * \sum_{L_3} \langle Y_{L_1} u_{\ell_1 \nu_1} | R_{L_3} h_{L_3} | Y_{L_2} u_{\ell_2 \nu_2} \rangle =
434 * \sum_{L_3} \langle u_{\ell_1 \nu_1} | h_{L_3} | u_{\ell_2 \nu_2} \rangle
435 * \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
436 * \f]
437 */
438 template <spin_block_t sblock>
439 inline double_complex
440 radial_integrals_sum_L3(int idxrf1__, int idxrf2__, std::vector<gaunt_L3<double_complex>> const& gnt__) const
441 {
442 double_complex zsum(0, 0);
443
444 for (size_t i = 0; i < gnt__.size(); i++) {
445 switch (sblock) {
446 case spin_block_t::nm: {
447 /* just the Hamiltonian */
448 zsum += gnt__[i].coef * h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__);
449 break;
450 }
451 case spin_block_t::uu: {
452 /* h + Bz */
453 zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) +
454 b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
455 break;
456 }
457 case spin_block_t::dd: {
458 /* h - Bz */
459 zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) -
460 b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
461 break;
462 }
463 case spin_block_t::ud: {
464 /* Bx - i By */
465 zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
466 -b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
467 break;
468 }
469 case spin_block_t::du: {
470 /* Bx + i By */
471 zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
472 b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
473 break;
474 }
475 }
476 }
477 return zsum;
478 }
