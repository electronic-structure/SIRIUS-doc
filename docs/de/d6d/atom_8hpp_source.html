139  /// Generate radial Hamiltonian and effective magnetic field integrals
140  /** Hamiltonian operator has the following representation inside muffin-tins:
141  * \f[
142  * \hat H = -\frac{1}{2}\nabla^2 + \sum_{\ell m} V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) =
143  * \underbrace{-\frac{1}{2} \nabla^2+V_{00}(r)R_{00}}_{H_{s}(r)} +\sum_{\ell=1} \sum_{m=-\ell}^{\ell}
144  * V_{\ell m}(r) R_{\ell m}(\hat {\bf r}) = \sum_{\ell m} \widetilde V_{\ell m}(r) R_{\ell m}(\hat {\bf r})
145  * \f]
146  * where
147  * \f[
148  * \widetilde V_{\ell m}(r) = \left\{ \begin{array}{ll}
149  * \frac{H_{s}(r)}{R_{00}} & \ell = 0 \\
150  * V_{\ell m}(r) & \ell > 0 \end{array} \right.
151  * \f]
152  */
153  inline void generate_radial_integrals(device_t pu__, Communicator const& comm__)
154  {
155  PROFILE("sirius::Atom::generate_radial_integrals");
156 
157  int lmmax = utils::lmmax(lmax_pot_);
158  int nmtp = type().num_mt_points();
159  int nrf = type().indexr().size();
160  int num_mag_dims = type().parameters().num_mag_dims();
161 
162  if (comm__.size() != 1) {
163  TERMINATE("not yet mpi parallel");
164  }
165 
166  splindex<splindex_t::block> spl_lm(lmmax, comm__.size(), comm__.rank());
167 
168  auto l_by_lm = utils::l_by_lm(lmax_pot_);
169 
170  h_radial_integrals_.zero();
171  if (num_mag_dims) {
172  b_radial_integrals_.zero();
173  }
174 
175  /* copy radial functions to spline objects */
176  std::vector<Spline<double>> rf_spline(nrf);
177  #pragma omp parallel for
178  for (int i = 0; i < nrf; i++) {
179  rf_spline[i] = Spline<double>(type().radial_grid());
180  for (int ir = 0; ir < nmtp; ir++) {
181  rf_spline[i](ir) = symmetry_class().radial_function(ir, i);
182  }
183  }
184 
185  /* copy effective potential components to spline objects */
186  std::vector<Spline<double>> v_spline(lmmax * (1 + num_mag_dims));
187  #pragma omp parallel for
188  for (int lm = 0; lm < lmmax; lm++) {
189  v_spline[lm] = Spline<double>(type().radial_grid());
190  for (int ir = 0; ir < nmtp; ir++) {
191  v_spline[lm](ir) = veff_(lm, ir);
192  }
193 
194  for (int j = 0; j < num_mag_dims; j++) {
195  v_spline[lm + (j + 1) * lmmax] = Spline<double>(type().radial_grid());
196  for (int ir = 0; ir < nmtp; ir++) {
197  v_spline[lm + (j + 1) * lmmax](ir) = beff_[j](lm, ir);
198  }
199  }
200  }
201 
202  /* interpolate potential multiplied by a radial function */
203  std::vector<Spline<double>> vrf_spline(lmmax * nrf * (1 + num_mag_dims));
204 
205  auto& idx_ri = type().idx_radial_integrals();
206 
207  mdarray<double, 1> result(idx_ri.size(1));
208 
209  if (pu__ == device_t::GPU) {
210 #ifdef SIRIUS_GPU
211  auto& rgrid = type().radial_grid();
212  auto& rf_coef = type().rf_coef();
213  auto& vrf_coef = type().vrf_coef();
214 
215  PROFILE_START("sirius::Atom::generate_radial_integrals|interp");
216  #pragma omp parallel
217  {
218  #pragma omp for
219  for (int i = 0; i < nrf; i++) {
220  rf_spline[i].interpolate();
221  std::memcpy(rf_coef.at(memory_t::host, 0, 0, i), rf_spline[i].coeffs().at(memory_t::host), nmtp * 4 * sizeof(double));
222  // cuda_async_copy_to_device(rf_coef.at<GPU>(0, 0, i), rf_coef.at<CPU>(0, 0, i), nmtp * 4 *
223  // sizeof(double), tid);
224  }
225  #pragma omp for
226  for (int i = 0; i < lmmax * (1 + num_mag_dims); i++) {
227  v_spline[i].interpolate();
228  }
229  }
230  rf_coef.copy_to(memory_t::device, stream_id(-1));
231 
232  #pragma omp parallel for
233  for (int lm = 0; lm < lmmax; lm++) {
234  for (int i = 0; i < nrf; i++) {
235  for (int j = 0; j < num_mag_dims + 1; j++) {
236  int idx = lm + lmmax * i + lmmax * nrf * j;
237  vrf_spline[idx] = rf_spline[i] * v_spline[lm + j * lmmax];
238  std::memcpy(vrf_coef.at(memory_t::host, 0, 0, idx), vrf_spline[idx].coeffs().at(memory_t::host),
239  nmtp * 4 * sizeof(double));
240  // cuda_async_copy_to_device(vrf_coef.at<GPU>(0, 0, idx), vrf_coef.at<CPU>(0, 0, idx), nmtp * 4
241  // *sizeof(double), tid);
242  }
243  }
244  }
245  vrf_coef.copy_to(memory_t::device);
246  PROFILE_STOP("sirius::Atom::generate_radial_integrals|interp");
247 
248  result.allocate(memory_t::device);
249  spline_inner_product_gpu_v3(idx_ri.at(memory_t::device), (int)idx_ri.size(1), nmtp, rgrid.x().at(memory_t::device),
250  rgrid.dx().at(memory_t::device), rf_coef.at(memory_t::device), vrf_coef.at(memory_t::device), result.at(memory_t::device));
251  acc::sync();
252  //if (type().parameters().control().print_performance_) {
253  // double tval = t2.stop();
254  // DUMP("spline GPU integration performance: %12.6f GFlops",
255  // 1e-9 * double(idx_ri.size(1)) * nmtp * 85 / tval);
256  //}
257  result.copy_to(memory_t::host);
258  result.deallocate(memory_t::device);
259 #endif
260  }
261  if (pu__ == device_t::CPU) {
262  PROFILE_START("sirius::Atom::generate_radial_integrals|interp");
263  #pragma omp parallel
264  {
265  #pragma omp for
266  for (int i = 0; i < nrf; i++) {
267  rf_spline[i].interpolate();
268  }
269  #pragma omp for
270  for (int i = 0; i < lmmax * (1 + num_mag_dims); i++) {
271  v_spline[i].interpolate();
272  }
273 
274  #pragma omp for
275  for (int lm = 0; lm < lmmax; lm++) {
276  for (int i = 0; i < nrf; i++) {
277  for (int j = 0; j < num_mag_dims + 1; j++) {
278  vrf_spline[lm + lmmax * i + lmmax * nrf * j] = rf_spline[i] * v_spline[lm + j * lmmax];
279  }
280  }
281  }
282  }
283  PROFILE_STOP("sirius::Atom::generate_radial_integrals|interp");
284 
285  PROFILE("sirius::Atom::generate_radial_integrals|inner");
286  #pragma omp parallel for
287  for (int j = 0; j < (int)idx_ri.size(1); j++) {
288  result(j) = inner(rf_spline[idx_ri(0, j)], vrf_spline[idx_ri(1, j)], 2);
289  }
290  //if (type().parameters().control().print_performance_) {
291  // double tval = t2.stop();
292  // DUMP("spline CPU integration performance: %12.6f GFlops",
293  // 1e-9 * double(idx_ri.size(1)) * nmtp * 85 / tval);
294  //}
295  }
296 
297  int n{0};
298  for (int lm = 0; lm < lmmax; lm++) {
299  int l = l_by_lm[lm];
300 
301  for (int i2 = 0; i2 < type().indexr().size(); i2++) {
302  int l2 = type().indexr(i2).l;
303  for (int i1 = 0; i1 <= i2; i1++) {
304  int l1 = type().indexr(i1).l;
305  if ((l + l1 + l2) % 2 == 0) {
306  if (lm) {
307  h_radial_integrals_(lm, i1, i2) = h_radial_integrals_(lm, i2, i1) = result(n++);
308  } else {
309  h_radial_integrals_(0, i1, i2) = symmetry_class().h_spherical_integral(i1, i2);
310  h_radial_integrals_(0, i2, i1) = symmetry_class().h_spherical_integral(i2, i1);
311  }
312  for (int j = 0; j < num_mag_dims; j++) {
313  b_radial_integrals_(lm, i1, i2, j) = b_radial_integrals_(lm, i2, i1, j) = result(n++);
314  }
315  }
316  }
317  }
318  }
319

321  // DUMP("checksum(h_radial_integrals): %18.10f", h_radial_integrals_.checksum());
322  //}

425  /** Compute the following kinds of sums for different spin-blocks of the Hamiltonian:
426  * \f[
427  * \sum_{L_3} \langle Y_{L_1} u_{\ell_1 \nu_1} | R_{L_3} h_{L_3} | Y_{L_2} u_{\ell_2 \nu_2} \rangle =
428  * \sum_{L_3} \langle u_{\ell_1 \nu_1} | h_{L_3} | u_{\ell_2 \nu_2} \rangle
429  * \langle Y_{L_1} | R_{L_3} | Y_{L_2} \rangle
430  * \f]
431  */
432  template <spin_block_t sblock>
433  inline double_complex
434  radial_integrals_sum_L3(int idxrf1__, int idxrf2__, std::vector<gaunt_L3<double_complex>> const& gnt__) const
435  {
436  double_complex zsum(0, 0);
437 
438  for (size_t i = 0; i < gnt__.size(); i++) {
439  switch (sblock) {
440  case spin_block_t::nm: {
441  /* just the Hamiltonian */
442  zsum += gnt__[i].coef * h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__);
443  break;
444  }
445  case spin_block_t::uu: {
446  /* h + Bz */
447  zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) +
448  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
449  break;
450  }
451  case spin_block_t::dd: {
452  /* h - Bz */
453  zsum += gnt__[i].coef * (h_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__) -
454  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 0));
455  break;
456  }
457  case spin_block_t::ud: {
458  /* Bx - i By */
459  zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
460  -b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
461  break;
462  }
463  case spin_block_t::du: {
464  /* Bx + i By */
465  zsum += gnt__[i].coef * double_complex(b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 1),
466  b_radial_integrals_(gnt__[i].lm3, idxrf1__, idxrf2__, 2));
467  break;
468  }
469  }
470  }
471  return zsum;
472  }
