44 inline void Density::initial_density_pseudo()
45 {
46  auto v = ctx_.make_periodic_function<index_domain_t::local>([&](int iat, double g)
47  {
48  return ctx_.ps_rho_ri().value<int>(iat, g);
49  });
50 
51  if (ctx_.control().print_checksum_) {
52  auto z1 = mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).checksum();
53  ctx_.comm().allreduce(&z1, 1);
54  if (ctx_.comm().rank() == 0) {
55  utils::print_checksum("rho_pw_init", z1);
56  }
57  }
58  std::copy(v.begin(), v.end(), &rho().f_pw_local(0));
59 
60  double charge = rho().f_0().real() * unit_cell_.omega();
61 
62  if (std::abs(charge - unit_cell_.num_valence_electrons()) > 1e-6) {
63  std::stringstream s;
64  s << "wrong initial charge density" << std::endl
65  << " integral of the density : " << std::setprecision(12) << charge << std::endl
66  << " target number of electrons : " << std::setprecision(12) << unit_cell_.num_valence_electrons();
67  if (ctx_.comm().rank() == 0) {
68  WARNING(s);
69  }
70  }
71  rho().fft_transform(1);
72 
73  /* remove possible negative noise */
74  for (int ir = 0; ir < ctx_.fft().local_size(); ir++) {
75  rho().f_rg(ir) = std::max(rho().f_rg(ir), 0.0);
76  }
77  /* renormalize charge */
78  normalize();
79 
80  if (ctx_.control().print_checksum_) {
81  auto cs = rho().checksum_rg();
82  if (ctx_.comm().rank() == 0) {
83  utils::print_checksum("rho_rg", cs);
84  }
85  }
86 
87  /* initialize the magnetization */
88  if (ctx_.num_mag_dims()) {
89  double R = ctx_.av_atom_radius();
90 
91  auto w = [R](double x)
92  {
93  /* the constants are picked in such a way that the volume integral of the
94  weight function is equal to the volume of the atomic sphere;
95  in this case the starting magnetiation in the atomic spehre
96  integrates to the starting magnetization vector */
97 
98  /* volume of the sphere */
99  const double norm = fourpi * std::pow(R, 3) / 3.0;
100  return (35.0 / 8) * std::pow(1 - std::pow(x / R, 2), 2) / norm;
101  //return 10 * std::pow(1 - x / R, 2) / norm;
102  //const double b = 1.1016992073677703;
103  //return b * 1.0 / (std::exp(10 * (a - R)) + 1) / norm;
104  //const double norm = pi * std::pow(R, 3) / 3.0;
105  //return 1.0 / (std::exp(10 * (x - R)) + 1) / norm;
106  };
107 
108  #pragma omp parallel for
109  for (int ia = 0; ia < unit_cell_.num_atoms(); ia++) {
110  auto& atom_to_grid_map = ctx_.atoms_to_grid_idx_map()[ia];
111  vector3d<double> v = unit_cell_.atom(ia).vector_field();
112 
113  for (auto coord: atom_to_grid_map) {
114  int ir = coord.first;
115  double a = coord.second;
116  magnetization(0).f_rg(ir) += v[2] * w(a);
117  if (ctx_.num_mag_dims() == 3) {
118  magnetization(1).f_rg(ir) += v[0] * w(a);
119  magnetization(2).f_rg(ir) += v[1] * w(a);
120  }
121  }
122  }
123  }
124 
125  if (ctx_.control().print_checksum_) {
126  for (int i = 0; i < ctx_.num_mag_dims() + 1; i++) {
127  auto cs = component(i).checksum_rg();
128  if (ctx_.comm().rank() == 0) {
129  std::stringstream s;
130  s << "component[" << i << "]";
131  utils::print_checksum(s.str(), cs);
132  }
133  }
134  }
135 
136  rho().fft_transform(-1);
137  for (int j = 0; j < ctx_.num_mag_dims(); j++) {
138  magnetization(j).fft_transform(-1);
139  }
140 
141  //if (ctx_.control().print_checksum_ && ctx_.comm().rank() == 0) {
142  // double_complex cs = mdarray<double_complex, 1>(&rho_->f_pw(0), ctx_.gvec().num_gvec()).checksum();
143  // DUMP("checksum(rho_pw): %20.14f %20.14f", std::real(cs), std::imag(cs));
144  //}

147 inline void Density::initial_density_full_pot()
148 {
149  /* initialize smooth density of free atoms */
150  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
151  unit_cell_.atom_type(iat).init_free_atom_density(true);
152  }
153 
154  /* compute radial integrals */
155  Radial_integrals_rho_free_atom ri(ctx_.unit_cell(), ctx_.pw_cutoff(), 40);
156 
157  /* compute contribution from free atoms to the interstitial density */
158  auto v = ctx_.make_periodic_function<index_domain_t::local>([&ri](int iat, double g)
159  {
160  return ri.value(iat, g);
161  });
162 
163  /* initialize density of free atoms (not smoothed) */
164  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
165  unit_cell_.atom_type(iat).init_free_atom_density(false);
166  }
167 
168  if (ctx_.control().print_checksum_) {
169  auto z = mdarray<double_complex, 1>(&v[0], ctx_.gvec().count()).checksum();
170  ctx_.comm().allreduce(&z, 1);
171  if (ctx_.comm().rank() == 0) {
172  utils::print_checksum("rho_pw", z);
173  }
174  }
175 
176  /* set plane-wave coefficients of the charge density */
177  std::copy(v.begin(), v.end(), &rho().f_pw_local(0));
178  /* convert charge density to real space mesh */
179  rho().fft_transform(1);
180 
181  if (ctx_.control().print_checksum_) {
182  auto cs = rho().checksum_rg();
183  if (ctx_.comm().rank() == 0) {
184  utils::print_checksum("rho_rg", cs);
185  }
186  }
187 
188  /* remove possible negative noise */
189  for (int ir = 0; ir < ctx_.fft().local_size(); ir++) {
190  rho().f_rg(ir) = std::max(0.0, rho().f_rg(ir));
191  }
192 
193  /* set Y00 component of charge density */
194  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
195  int nmtp = ctx_.unit_cell().atom(ia).num_mt_points();
196 
197  for (int ir = 0; ir < nmtp; ir++) {
198  double x = ctx_.unit_cell().atom(ia).radial_grid(ir);
199  rho().f_mt<index_domain_t::global>(0, ir, ia) = unit_cell_.atom(ia).type().free_atom_density(x) / y00;
200  }
201  }
202 
203  int lmax = ctx_.lmax_rho();
204  int lmmax = utils::lmmax(lmax);
205 
206  auto l_by_lm = utils::l_by_lm(lmax);
207 
208  std::vector<double_complex> zil(lmax + 1);
209  for (int l = 0; l <= lmax; l++) {
210  zil[l] = std::pow(double_complex(0, 1), l);
211  }
212 
213  /* compute boundary value at MT sphere from the plane-wave exapansion */
214  auto gvec_ylm = ctx_.generate_gvec_ylm(lmax);
215 
216  auto sbessel_mt = ctx_.generate_sbessel_mt(lmax);
217 
218  auto flm = ctx_.sum_fg_fl_yg(lmax, v.data(), sbessel_mt, gvec_ylm);
219 
220  /* this is the difference between the value of periodic charge density at MT boundary and
221  a value of the atom's free density at the boundary */
222  for (int ia = 0; ia < ctx_.unit_cell().num_atoms(); ia++) {
223  double R = ctx_.unit_cell().atom(ia).mt_radius();
224  double c = unit_cell_.atom(ia).type().free_atom_density(R) / y00;
225  flm(0, ia) -= c;
226  }
227 
228  /* match density at MT */
229  for (int iat = 0; iat < ctx_.unit_cell().num_atom_types(); iat++) {
230  mdarray<double, 2> rRl(ctx_.unit_cell().max_num_mt_points(), lmax + 1);
231  double R = ctx_.unit_cell().atom_type(iat).mt_radius();
232  int nmtp = ctx_.unit_cell().atom_type(iat).num_mt_points();
233 
234  #pragma omp parallel for default(shared)
235  for (int l = 0; l <= lmax; l++) {
236  for (int ir = 0; ir < nmtp; ir++) {
237  rRl(ir, l) = std::pow(ctx_.unit_cell().atom_type(iat).radial_grid(ir) / R, 2);
238  }
239  }
240  #pragma omp parallel for default(shared)
241  for (int i = 0; i < unit_cell_.atom_type(iat).num_atoms(); i++) {
242  int ia = unit_cell_.atom_type(iat).atom_id(i);
243  std::vector<double> glm(lmmax);
244  SHT::convert(lmax, &flm(0, ia), &glm[0]);
245  for (int lm = 0; lm < lmmax; lm++) {
246  int l = l_by_lm[lm];
247  for (int ir = 0; ir < nmtp; ir++) {
248  rho().f_mt<index_domain_t::global>(lm, ir, ia) += glm[lm] * rRl(ir, l);
249  }
250  }
251  }
252  }
253 
254  /* normalize charge density */
255  normalize();
256 
257  check_num_electrons();

270  // double val = rho().value_rg(v);
271  // fprintf(fout2, "%18.12f %18.12f\n", v.length(), val);
272  //}
273  //fclose(fout2);
274 
275  /* initialize the magnetization */
276  if (ctx_.num_mag_dims()) {
277  for (int ialoc = 0; ialoc < unit_cell_.spl_num_atoms().local_size(); ialoc++) {
278  int ia = unit_cell_.spl_num_atoms(ialoc);
279  vector3d<double> v = unit_cell_.atom(ia).vector_field();
280  double len = v.length();
281 
282  int nmtp = unit_cell_.atom(ia).num_mt_points();
283  Spline<double> rho_s(unit_cell_.atom(ia).type().radial_grid());
284  double R = unit_cell_.atom(ia).mt_radius();
285  for (int ir = 0; ir < nmtp; ir++) {
286  double x = unit_cell_.atom(ia).type().radial_grid(ir);
287  rho_s(ir) = this->rho().f_mt<index_domain_t::local>(0, ir, ialoc) * y00 * (1 - 3 * std::pow(x / R, 2) + 2 * std::pow(x / R, 3));
288  }
289 
290  /* maximum magnetization which can be achieved if we smooth density towards MT boundary */
291  double q = fourpi * rho_s.interpolate().integrate(2);
292 
293  /* if very strong initial magnetization is given */
294  if (q < len) {
295  /* renormalize starting magnetization */
296  for (int x: {0, 1, 2}) {
297  v[x] *= (q / len);
298  }
299  len = q;
300  }
301 
302  if (len > 1e-8) {
303  for (int ir = 0; ir < nmtp; ir++) {
304  magnetization(0).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[2] / q / y00;
305  }
306  if (ctx_.num_mag_dims() == 3) {
307  for (int ir = 0; ir < nmtp; ir++) {
308  magnetization(1).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[0] / q / y00;
309  magnetization(2).f_mt<index_domain_t::local>(0, ir, ialoc) = rho_s(ir) * v[1] / q / y00;
310  }
311  }
312  }
313  }
314  }
