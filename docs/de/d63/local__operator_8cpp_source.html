33 Local_operator::Local_operator(Simulation_context const& ctx__, spfft::Transform& fft_coarse__,
34  Gvec_partition const& gvec_coarse_p__, Potential* potential__)
35  : ctx_(ctx__)
36  , fft_coarse_(fft_coarse__)
37  , gvec_coarse_p_(gvec_coarse_p__)
38 
39 {
40  PROFILE("sirius::Local_operator");
41 
42  /* allocate functions */
43  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
44  veff_vec_[j] = std::unique_ptr<Smooth_periodic_function<double>>(
45  new Smooth_periodic_function<double>(fft_coarse__, gvec_coarse_p__, &ctx_.mem_pool(memory_t::host)));
46  #pragma omp parallel for schedule(static)
47  for (int ir = 0; ir < fft_coarse__.local_slice_size(); ir++) {
48  veff_vec_[j]->f_rg(ir) = 2.71828;
49  }
50  }
51  /* map Theta(r) to the coarse mesh */
52  if (ctx_.full_potential()) {
53  auto& gvec_dense_p = ctx_.gvec_partition();
54  veff_vec_[4] = std::unique_ptr<Smooth_periodic_function<double>>(
55  new Smooth_periodic_function<double>(fft_coarse__, gvec_coarse_p__, &ctx_.mem_pool(memory_t::host)));
56  /* map unit-step function */
57  #pragma omp parallel for schedule(static)
58  for (int igloc = 0; igloc < gvec_coarse_p_.gvec().count(); igloc++) {
59  /* map from fine to coarse set of G-vectors */
60  veff_vec_[4]->f_pw_local(igloc) =
61  ctx_.theta_pw(gvec_dense_p.gvec().gvec_base_mapping(igloc) + gvec_dense_p.gvec().offset());
62  }
63  veff_vec_[4]->fft_transform(1);
64  if (fft_coarse_.processing_unit() == SPFFT_PU_GPU) {
65  veff_vec_[4]->f_rg().allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
66  }
67  if (ctx_.control().print_checksum_) {
68  auto cs1 = veff_vec_[4]->checksum_pw();
69  auto cs2 = veff_vec_[4]->checksum_rg();
70  if (ctx_.comm().rank() == 0) {
71  utils::print_checksum("theta_pw", cs1);
72  utils::print_checksum("theta_rg", cs2);
73  }
74  }
75  }
76 
77  /* map potential */
78  if (potential__) {
79 
80  if (ctx_.full_potential()) {
81 
82  auto& fft_dense = ctx_.spfft();
83  auto& gvec_dense_p = ctx_.gvec_partition();
84 
85  Smooth_periodic_function<double> ftmp(const_cast<Simulation_context&>(ctx_).spfft(), gvec_dense_p,
86  &ctx_.mem_pool(memory_t::host));
87 
88  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
89  /* multiply potential by step function theta(r) */
90  for (int ir = 0; ir < fft_dense.local_slice_size(); ir++) {
91  ftmp.f_rg(ir) = potential__->component(j).f_rg(ir) * ctx_.theta(ir);
92  }
93  /* transform to plane-wave domain */
94  ftmp.fft_transform(-1);
95  if (j == 0) {
96  v0_[0] = ftmp.f_0().real();
97  }
98  /* loop over local set of coarse G-vectors */
99  #pragma omp parallel for schedule(static)
100  for (int igloc = 0; igloc < gvec_coarse_p_.gvec().count(); igloc++) {
101  /* map from fine to coarse set of G-vectors */
102  veff_vec_[j]->f_pw_local(igloc) = ftmp.f_pw_local(gvec_dense_p.gvec().gvec_base_mapping(igloc));
103  }
104  /* transform to real space */
105  veff_vec_[j]->fft_transform(1);
106  }
107  if (ctx_.valence_relativity() == relativity_t::zora) {
108  veff_vec_[5] = std::unique_ptr<Smooth_periodic_function<double>>(
109  new Smooth_periodic_function<double>(fft_coarse__, gvec_coarse_p__, &ctx_.mem_pool(memory_t::host)));
110  /* loop over local set of coarse G-vectors */
111  #pragma omp parallel for schedule(static)
112  for (int igloc = 0; igloc < gvec_coarse_p_.gvec().count(); igloc++) {
113  /* map from fine to coarse set of G-vectors */
114  veff_vec_[5]->f_pw_local(igloc) =
115  potential__->rm_inv_pw(gvec_dense_p.gvec().offset() + gvec_dense_p.gvec().gvec_base_mapping(igloc));
116  }
117  /* transform to real space */
118  veff_vec_[5]->fft_transform(1);
119  }
120 
121  } else {
122 
123  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
124  /* loop over local set of coarse G-vectors */
125  #pragma omp parallel for schedule(static)
126  for (int igloc = 0; igloc < gvec_coarse_p_.gvec().count(); igloc++) {
127  /* map from fine to coarse set of G-vectors */
128  veff_vec_[j]->f_pw_local(igloc) =
129  potential__->component(j).f_pw_local(potential__->component(j).gvec().gvec_base_mapping(igloc));
130  }
131  /* transform to real space */
132  veff_vec_[j]->fft_transform(1);
133  }
134 
135  /* change to canonical form */
136  if (ctx_.num_mag_dims()) {
137  #pragma omp parallel for schedule(static)
138  for (int ir = 0; ir < fft_coarse_.local_slice_size(); ir++) {
139  double v0 = veff_vec_[0]->f_rg(ir);
140  double v1 = veff_vec_[1]->f_rg(ir);
141  veff_vec_[0]->f_rg(ir) = v0 + v1; // v + Bz
142  veff_vec_[1]->f_rg(ir) = v0 - v1; // v - Bz
143  }
144  }
145 
146  if (ctx_.num_mag_dims() == 0) {
147  v0_[0] = potential__->component(0).f_0().real();
148  } else {
149  v0_[0] = potential__->component(0).f_0().real() + potential__->component(1).f_0().real();
150  v0_[1] = potential__->component(0).f_0().real() - potential__->component(1).f_0().real();
151  }
152  }
153 
154  if (ctx_.control().print_checksum_) {
155  for (int j = 0; j < ctx_.num_mag_dims() + 1; j++) {
156  auto cs1 = veff_vec_[j]->checksum_pw();
157  auto cs2 = veff_vec_[j]->checksum_rg();
158  if (ctx_.comm().rank() == 0) {
159  utils::print_checksum("veff_pw", cs1);
160  utils::print_checksum("veff_rg", cs2);
161  }
162  }
163  }
164  }
165 
166  buf_rg_ = mdarray<double_complex, 1>(fft_coarse_.local_slice_size(), ctx_.mem_pool(memory_t::host),
167  "Local_operator::buf_rg_");
168  /* move functions to GPU */
169  if (fft_coarse_.processing_unit() == SPFFT_PU_GPU) {
170  for (int j = 0; j < 6; j++) {
171  if (veff_vec_[j]) {
172  veff_vec_[j]->f_rg().allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
173  }
174  }
175  buf_rg_.allocate(ctx_.mem_pool(memory_t::device));
176  }

179 void Local_operator::prepare_k(Gvec_partition const& gkvec_p__)
180 {
181  PROFILE("sirius::Local_operator::prepare_k");
182 
183  int ngv_fft = gkvec_p__.gvec_count_fft();
184 
185  /* cache kinteic energy of plane-waves */
186  if (static_cast<int>(pw_ekin_.size()) < ngv_fft) {
187  pw_ekin_ = mdarray<double, 1>(ngv_fft, ctx_.mem_pool(memory_t::host), "Local_operator::pw_ekin");
188  }
189  #pragma omp parallel for schedule(static)
190  for (int ig_loc = 0; ig_loc < ngv_fft; ig_loc++) {
191  /* global index of G-vector */
192  int ig = gkvec_p__.idx_gvec(ig_loc);
193  /* get G+k in Cartesian coordinates */
194  auto gv = gkvec_p__.gvec().gkvec_cart<index_domain_t::global>(ig);
195  pw_ekin_[ig_loc] = 0.5 * dot(gv, gv);
196  }
197 
198  if (static_cast<int>(vphi_.size(0)) < ngv_fft) {
199  vphi_ = mdarray<double_complex, 1>(ngv_fft, ctx_.mem_pool(memory_t::host), "Local_operator::vphi");
200  }
201 
202  if (fft_coarse_.processing_unit() == SPFFT_PU_GPU) {
203  pw_ekin_.allocate(ctx_.mem_pool(memory_t::device)).copy_to(memory_t::device);
204  vphi_.allocate(ctx_.mem_pool(memory_t::device));
205  }

208 static inline void mul_by_veff(spfft::Transform& spfftk__, double* buff__,
209  std::array<std::unique_ptr<Smooth_periodic_function<double>>, 6>& veff_vec__,
210  int idx_veff__)
211 {
212  int nr = spfftk__.local_slice_size();
213 
214  switch (spfftk__.processing_unit()) {
215  case SPFFT_PU_HOST: {
216  if (idx_veff__ <= 1 || idx_veff__ >= 4) { /* up-up or dn-dn block or Theta(r) */
217  switch (spfftk__.type()) {
218  case SPFFT_TRANS_R2C: {
219  #pragma omp parallel for schedule(static)
220  for (int ir = 0; ir < nr; ir++) {
221  /* multiply by V+Bz or V-Bz (in PP-PW case) or by V(r), B_z(r) or Theta(r) (in LAPW case) */
222  buff__[ir] *= veff_vec__[idx_veff__]->f_rg(ir);
223  }
224  break;
225  }
226  case SPFFT_TRANS_C2C: {
227  auto wf = reinterpret_cast<double_complex*>(buff__);
228  #pragma omp parallel for schedule(static)
229  for (int ir = 0; ir < nr; ir++) {
230  /* multiply by V+Bz or V-Bz (in PP-PW case) or by V(r), B_z(r) or Theta(r) (in LAPW case) */
231  wf[ir] *= veff_vec__[idx_veff__]->f_rg(ir);
232  }
233  break;
234  }
235  }
236  } else {
237  double pref = (idx_veff__ == 2) ? -1 : 1;
238  auto wf = reinterpret_cast<double_complex*>(buff__);
239  #pragma omp parallel for schedule(static)
240  for (int ir = 0; ir < nr; ir++) {
241  /* multiply by Bx +/- i*By */
242  wf[ir] *= double_complex(veff_vec__[2]->f_rg(ir), pref * veff_vec__[3]->f_rg(ir));
243  }
244  }
245  break;
246  }
247  case SPFFT_PU_GPU: {
248 #if defined(__GPU)
249  if (idx_veff__ <= 1 || idx_veff__ >= 4) { /* up-up or dn-dn block or Theta(r) */
250  switch (spfftk__.type()) {
251  case SPFFT_TRANS_R2C: {
252  /* multiply by V+Bz or V-Bz (in PP-PW case) or by V(r), B_z(r) or Theta(r) (in LAPW case) */
253  mul_by_veff_real_real_gpu(nr, buff__, veff_vec__[idx_veff__]->f_rg().at(memory_t::device));
254  break;
255  }
256  case SPFFT_TRANS_C2C: {
257  auto wf = reinterpret_cast<double_complex*>(buff__);
258  /* multiply by V+Bz or V-Bz (in PP-PW case) or by V(r), B_z(r) or Theta(r) (in LAPW case) */
259  mul_by_veff_complex_real_gpu(nr, wf, veff_vec__[idx_veff__]->f_rg().at(memory_t::device));
260  break;
261  }
262  }
263  } else {
264  /* multiply by Bx +/- i*By */
265  double pref = (idx_veff__ == 2) ? -1 : 1;
266  auto wf = reinterpret_cast<double_complex*>(buff__);
267  mul_by_veff_complex_complex_gpu(nr, wf, pref, veff_vec__[2]->f_rg().at(memory_t::device),
268  veff_vec__[3]->f_rg().at(memory_t::device));
269  }
270  break;
271 #endif
272  }
273  break;
274  }

277 void Local_operator::apply_h(spfft::Transform& spfftk__, Gvec_partition const& gkvec_p__, spin_range spins__,
278  Wave_functions& phi__, Wave_functions& hphi__, int idx0__, int n__)
279 {
280  PROFILE("sirius::Local_operator::apply_h");
281 
282  if ((spfftk__.dim_x() != fft_coarse_.dim_x()) ||
283  (spfftk__.dim_y() != fft_coarse_.dim_y()) ||
284  (spfftk__.dim_z() != fft_coarse_.dim_z())) {
285  TERMINATE("wrong FFT dimensions");
286  }
287 
288  /* increment the counter by the number of wave-functions */
289  num_applied(n__);
290 
291  /* this memory pool will be used to allocate extra storage in the host memory */
292  auto& mp = const_cast<Simulation_context&>(ctx_).mem_pool(ctx_.host_memory_t());
293  /* this memory pool will be used to allocate extra storage in the device memory */
294 #if defined(__GPU)
295  memory_pool* mpd = &const_cast<Simulation_context&>(ctx_).mem_pool(memory_t::device);
296 #else
297  memory_pool* mpd{nullptr};
298 #endif
299  /* alias array for all wave functions in FFT-friendly storage */
300  std::array<mdarray<double_complex, 2>, 2> phi;
301  /* alias array for all hphi in FFT-friendly storage */
302  std::array<mdarray<double_complex, 2>, 2> hphi;
303 
304  /* alias for wave-functions that are currently computed */
305  std::array<mdarray<double_complex, 1>, 2> phi1;
306  std::array<mdarray<double_complex, 1>, 2> hphi1;
307 
308  /* local number of G-vectors for the FFT transformation */
309  int ngv_fft = gkvec_p__.gvec_count_fft();
310 
311  if (ngv_fft != spfftk__.num_local_elements()) {
312  TERMINATE("wrong number of G-vectors");
313  }
314 
315  memory_t mem_phi{memory_t::none};
316  memory_t mem_hphi{memory_t::none};
317 
318  /* remap wave-functions to FFT friendly distribution */
319  for (int ispn : spins__) {
320  /* if we store wave-functions in the device memory and if the wave functions are remapped
321  we need to copy the wave functions to host memory */
322  if (is_device_memory(phi__.preferred_memory_t()) && phi__.pw_coeffs(ispn).is_remapped()) {
323  phi__.pw_coeffs(ispn).copy_to(memory_t::host, idx0__, n__);
324  }
325  /* set FFT friendly distribution */
326  phi__.pw_coeffs(ispn).remap_forward(n__, idx0__, &mp);
327  /* memory location of phi in extra storage */
328  mem_phi = (phi__.pw_coeffs(ispn).is_remapped()) ? memory_t::host : phi__.preferred_memory_t();
329  /* set FFT friednly distribution */
330  hphi__.pw_coeffs(ispn).set_num_extra(n__, idx0__, &mp);
331  /* memory location of hphi in extra storage */
332  mem_hphi = (hphi__.pw_coeffs(ispn).is_remapped()) ? memory_t::host : hphi__.preferred_memory_t();
333 
334  /* local number of wave-functions in extra-storage distribution */
335  int num_wf_loc = phi__.pw_coeffs(ispn).spl_num_col().local_size();
336 
337  /* set alias for phi extra storage */
338  double_complex* ptr_d{nullptr};
339  if (phi__.pw_coeffs(ispn).extra().on_device()) {
340  ptr_d = phi__.pw_coeffs(ispn).extra().at(memory_t::device);
341  }
342  phi[ispn] =
343  mdarray<double_complex, 2>(phi__.pw_coeffs(ispn).extra().at(memory_t::host), ptr_d, ngv_fft, num_wf_loc);
344 
345  /* set alias for hphi extra storage */
346  ptr_d = nullptr;
347  if (hphi__.pw_coeffs(ispn).extra().on_device()) {
348  ptr_d = hphi__.pw_coeffs(ispn).extra().at(memory_t::device);
349  }
350  hphi[ispn] =
351  mdarray<double_complex, 2>(hphi__.pw_coeffs(ispn).extra().at(memory_t::host), ptr_d, ngv_fft, num_wf_loc);
352  }
353 
354  /* assume the location of data on the current processing unit */
355  auto spfft_mem = spfftk__.processing_unit();
356 
357  /* number of real-space points in the local part of FFT buffer */
358  int nr = spfftk__.local_slice_size();
359 
360  /* pointer to FFT buffer */
361  auto spfft_buf = spfftk__.space_domain_data(spfft_mem);
362 
363  auto prepare_phi_hphi = [&](int i) {
364  for (int ispn : spins__) {
365  switch (spfft_mem) {
366  case SPFFT_PU_HOST: { /* FFT is done on CPU */
367  if (is_host_memory(mem_phi)) { /* wave-functions are also on host memory */
368  phi1[ispn] = mdarray<double_complex, 1>(phi[ispn].at(memory_t::host, 0, i), ngv_fft);
369  } else { /* wave-functions are on the device memory */
370  phi1[ispn] = mdarray<double_complex, 1>(ngv_fft, mp);
371  /* copy wave-functions to host memory */
372  acc::copyout(phi1[ispn].at(memory_t::host), phi[ispn].at(memory_t::device, 0, i), ngv_fft);
373  }
374  if (is_host_memory(mem_hphi)) {
375  hphi1[ispn] = mdarray<double_complex, 1>(hphi[ispn].at(memory_t::host, 0, i), ngv_fft);
376  } else {
377  hphi1[ispn] = mdarray<double_complex, 1>(ngv_fft, mp);
378  }
379  hphi1[ispn].zero(memory_t::host);
380  break;
381  }
382  case SPFFT_PU_GPU: { /* FFT is done on GPU */
383  if (is_host_memory(mem_phi)) {
384  phi1[ispn] = mdarray<double_complex, 1>(ngv_fft, *mpd);
385  /* copy wave-functions to device */
386  acc::copyin(phi1[ispn].at(memory_t::device), phi[ispn].at(memory_t::host, 0, i), ngv_fft);
387  } else {
388  phi1[ispn] = mdarray<double_complex, 1>(nullptr, phi[ispn].at(memory_t::device, 0, i), ngv_fft);
389  }
390  if (is_host_memory(mem_hphi)) {
391  /* small buffers in the device memory */
392  hphi1[ispn] = mdarray<double_complex, 1>(ngv_fft, *mpd);
393  } else {
394  hphi1[ispn] =
395  mdarray<double_complex, 1>(nullptr, hphi[ispn].at(memory_t::device, 0, i), ngv_fft);
396  }
397  hphi1[ispn].zero(memory_t::device);
398  break;
399  }
400  }
401  }
402  };
403 
404  auto store_hphi = [&](int i) {
405  for (int ispn : spins__) {
406  switch (spfft_mem) {
407  case SPFFT_PU_HOST: { /* FFT is done on CPU */
408  if (is_device_memory(mem_hphi)) {
409  /* copy to device */
410  acc::copyin(hphi[ispn].at(memory_t::device, 0, i), hphi1[ispn].at(memory_t::host), ngv_fft);
411  }
412  break;
413  }
414  case SPFFT_PU_GPU: { /* FFT is done on GPU */
415  if (is_host_memory(mem_hphi)) {
416  /* copy back to host */
417  acc::copyout(hphi[ispn].at(memory_t::host, 0, i), hphi1[ispn].at(memory_t::device), ngv_fft);
418  }
419  break;
420  }
421  }
422  }
423  };
424 
425  /* transform wave-function to real space; the result of the transformation is stored in the FFT buffer */
426  auto phi_to_r = [&](int ispn) {
427  spfftk__.backward(reinterpret_cast<double const*>(phi1[ispn].at(spfft_memory_t.at(spfft_mem))), spfft_mem);
428  };
429 
430  /* transform function to PW domain */
431  auto vphi_to_G = [&]() {
432  spfftk__.forward(spfft_mem, reinterpret_cast<double*>(vphi_.at(spfft_memory_t.at(spfft_mem))),
433  SPFFT_FULL_SCALING);
434  };
435 
436  /* store the resulting hphi
437  spin block (ispn_block) is used as a bit mask:
438  - first bit: spin component which is updated
439  - second bit: add or not kinetic energy term */
440  auto add_to_hphi = [&](int ispn_block) {
441  /* index of spin component */
442  int ispn = ispn_block & 1;
443  /* add kinetic energy if this is a diagonal block */
444  int ekin = (ispn_block & 2) ? 0 : 1;
445 
446  switch (spfft_mem) {
447  case SPFFT_PU_HOST: {
448  /* CPU case */
449  if (ekin) {
450  #pragma omp parallel for schedule(static)
451  for (int ig = 0; ig < gkvec_p__.gvec_count_fft(); ig++) {
452  hphi1[ispn][ig] += (phi1[ispn][ig] * pw_ekin_[ig] + vphi_[ig]);
453  }
454  } else {
455  #pragma omp parallel for schedule(static)
456  for (int ig = 0; ig < gkvec_p__.gvec_count_fft(); ig++) {
457  hphi1[ispn][ig] += vphi_[ig];
458  }
459  }
460  break;
461  }
462  case SPFFT_PU_GPU: {
463 #if defined(__GPU)
464  double alpha = static_cast<double>(ekin);
465  add_pw_ekin_gpu(gkvec_p__.gvec_count_fft(), alpha, pw_ekin_.at(memory_t::device),
466  phi1[ispn].at(memory_t::device), vphi_.at(memory_t::device),
467  hphi1[ispn].at(memory_t::device));
468 #endif
469  break;
470  }
471  }
472  };
473 
474  /* local number of wave-functions in extra-storage distribution */
475  int num_wf_loc = phi__.pw_coeffs(0).spl_num_col().local_size();
476 
477  /* if we don't have G-vector reductions, first = 0 and we start a normal loop */
478  for (int i = 0; i < num_wf_loc; i++) {
479 
480  /* non-collinear case */
481  /* 2x2 Hamiltonian in applied to spinor wave-functions
482  .--------.--------. .-----. .------.
483  | | | | | | |
484  | H_{uu} | H_{ud} | |phi_u| |hphi_u|
485  | | | | | | |
486  .--------.--------. x .-----. = .------.
487  | | | | | | |
488  | H_{du} | H_{dd} | |phi_d| |hphi_d|
489  | | | | | | |
490  .--------.--------. .-----. .------.
491 
492  hphi_u = H_{uu} phi_u + H_{ud} phi_d
493  hphi_d = H_{du} phi_u + H_{dd} phi_d
494 
495  The following indexing scheme will be used for spin-blocks
496  .---.---.
497  | 0 | 2 |
498  .---.---.
499  | 3 | 1 |
500  .---.---.
501  */
502  if (spins__() == 2) {
503  prepare_phi_hphi(i);
504  /* phi_u(G) -> phi_u(r) */
505  phi_to_r(0);
506  /* save phi_u(r) in temporary buf_rg array */
507  switch (spfft_mem) {
508  /* this is a non-collinear case, so the wave-functions and FFT buffer are complex and
509  we can copy memory */
510  case SPFFT_PU_HOST: {
511  auto inp = reinterpret_cast<double_complex*>(spfft_buf);
512  std::copy(inp, inp + nr, buf_rg_.at(memory_t::host));
513  break;
514  }
515  case SPFFT_PU_GPU: {
516  acc::copy(buf_rg_.at(memory_t::device), reinterpret_cast<double_complex*>(spfft_buf), nr);
517  break;
518  }
519  }
520  /* multiply phi_u(r) by effective potential */
521  mul_by_veff(spfftk__, spfft_buf, veff_vec_, 0);
522 
523  /* V_{uu}(r)phi_{u}(r) -> [V*phi]_{u}(G) */
524  vphi_to_G();
525  /* add kinetic energy */
526  add_to_hphi(0);
527  /* multiply phi_{u} by V_{du} and copy to FFT buffer */
528  switch (spfft_mem) {
529  case SPFFT_PU_HOST: {
530  mul_by_veff(spfftk__, reinterpret_cast<double*>(buf_rg_.at(memory_t::host)), veff_vec_, 3);
531  std::copy(buf_rg_.at(memory_t::host), buf_rg_.at(memory_t::host) + nr,
532  reinterpret_cast<double_complex*>(spfft_buf));
533  break;
534  }
535  case SPFFT_PU_GPU: {
536  mul_by_veff(spfftk__, reinterpret_cast<double*>(buf_rg_.at(memory_t::device)), veff_vec_, 3);
537  acc::copy(reinterpret_cast<double_complex*>(spfft_buf), buf_rg_.at(memory_t::device), nr);
538  break;
539  }
540  }
541  /* V_{du}(r)phi_{u}(r) -> [V*phi]_{d}(G) */
542  vphi_to_G();
543  /* add to hphi_{d} */
544  add_to_hphi(3);
545 
546  /* for the second spin component */
547 
548  /* phi_d(G) -> phi_d(r) */
549  phi_to_r(1);
550  /* save phi_d(r) */
551  switch (spfft_mem) {
552  case SPFFT_PU_HOST: {
553  auto inp = reinterpret_cast<double_complex*>(spfft_buf);
554  std::copy(inp, inp + nr, buf_rg_.at(memory_t::host));
555  break;
556  }
557  case SPFFT_PU_GPU: {
558  acc::copy(buf_rg_.at(memory_t::device), reinterpret_cast<double_complex*>(spfft_buf), nr);
559  break;
560  }
561  }
562  /* multiply phi_d(r) by effective potential */
563  mul_by_veff(spfftk__, spfft_buf, veff_vec_, 1);
564  /* V_{dd}(r)phi_{d}(r) -> [V*phi]_{d}(G) */
565  vphi_to_G();
566  /* add kinetic energy */
567  add_to_hphi(1);
568  /* multiply phi_{d} by V_{ud} and copy to FFT buffer */
569  switch (spfft_mem) {
570  case SPFFT_PU_HOST: {
571  mul_by_veff(spfftk__, reinterpret_cast<double*>(buf_rg_.at(memory_t::host)), veff_vec_, 2);
572  std::copy(buf_rg_.at(memory_t::host), buf_rg_.at(memory_t::host) + nr,
573  reinterpret_cast<double_complex*>(spfft_buf));
574  break;
575  }
576  case SPFFT_PU_GPU: {
577  mul_by_veff(spfftk__, reinterpret_cast<double*>(buf_rg_.at(memory_t::device)), veff_vec_, 2);
578  acc::copy(reinterpret_cast<double_complex*>(spfft_buf), buf_rg_.at(memory_t::device), nr);
579  break;
580  }
581  }
582  /* V_{ud}(r)phi_{d}(r) -> [V*phi]_{u}(G) */
583  vphi_to_G();
584  /* add to hphi_{u} */
585  add_to_hphi(2);
586  /* copy to main hphi array */
587  store_hphi(i);
588  } else { /* spin-collinear or non-magnetic case */
589  prepare_phi_hphi(i);
590  /* phi(G) -> phi(r) */
591  phi_to_r(spins__());
592  /* multiply by effective potential */
593  mul_by_veff(spfftk__, spfft_buf, veff_vec_, spins__());
594  /* V(r)phi(r) -> [V*phi](G) */
595  vphi_to_G();
596  /* add kinetic energy */
597  add_to_hphi(spins__());
598  store_hphi(i);
599  }
600  }
601 
602  /* remap hphi backward */
603  for (int ispn : spins__) {
604  hphi__.pw_coeffs(ispn).remap_backward(n__, idx0__);
605  if (is_device_memory(hphi__.preferred_memory_t()) && hphi__.pw_coeffs(ispn).is_remapped()) {
606  hphi__.pw_coeffs(ispn).copy_to(memory_t::device, idx0__, n__);
607  }
608  }
609  /* at this point hphi in prime storage is both on CPU and GPU memory; however if the memory pool
610  was used for the device memory allocation, device storage is destroyed */
611 }

614  Wave_functions& phi__, Wave_functions* hphi__, Wave_functions* ophi__)
615 {

619 
620  mdarray<double_complex, 1> buf_pw(gkvec_p__.gvec_count_fft(), ctx_.mem_pool(memory_t::host));
621

623  phi__.pw_coeffs(0).copy_to(memory_t::host, N__, n__);
624  }

626  // auto cs = phi__.checksum_pw(N__, n__, ctx_->processing_unit());

635

638  }
639 
640  if (ophi__ != nullptr) {
641  ophi__->pw_coeffs(0).set_num_extra(n__, N__, &mp);
642  }
643 
644  /* assume the location of data on the current processing unit */
645  auto spfft_mem = spfftk__.processing_unit();
646 
647  /* number of real-space points in the local part of FFT buffer */
648  int nr = spfftk__.local_slice_size();
649 
650  /* pointer to FFT buffer */
651  auto spfft_buf = spfftk__.space_domain_data(spfft_mem);
652 
653  for (int j = 0; j < phi__.pw_coeffs(0).spl_num_col().local_size(); j++) {
654  utils::timer t1("sirius::Local_operator::apply_h_o|pot");
655  /* phi(G) -> phi(r) */
656  spfftk__.backward(reinterpret_cast<double const*>(phi__.pw_coeffs(0).extra().at(memory_t::host, 0, j)),
657  spfft_mem);
658  if (ophi__ != nullptr) {
659  /* save phi(r) */
660  if (hphi__ != nullptr) {
661  switch (spfft_mem) {
662  case SPFFT_PU_HOST: {
663  auto inp = reinterpret_cast<double_complex*>(spfft_buf);
664  std::copy(inp, inp + nr, buf_rg_.at(memory_t::host));
665  break;
666  }
667  case SPFFT_PU_GPU: {
668  acc::copy(buf_rg_.at(memory_t::device), reinterpret_cast<double_complex*>(spfft_buf), nr);
669  break;
670  }
671  }
672  }
673 
674  /* multiply phi(r) by step function */
675  mul_by_veff(spfftk__, spfft_buf, veff_vec_, 4);
676 
677  /* phi(r) * Theta(r) -> ophi(G) */
678  spfftk__.forward(spfft_mem,
679  reinterpret_cast<double*>(ophi__->pw_coeffs(0).extra().at(memory_t::host, 0, j)),
680  SPFFT_FULL_SCALING);
681  /* load phi(r) back */
682  if (hphi__ != nullptr) {
683  switch (spfft_mem) {
684  case SPFFT_PU_HOST: {
685  std::copy(buf_rg_.at(memory_t::host), buf_rg_.at(memory_t::host) + nr,
686  reinterpret_cast<double_complex*>(spfft_buf));
687  break;
688  }
689  case SPFFT_PU_GPU: {
690  acc::copy(reinterpret_cast<double_complex*>(spfft_buf), buf_rg_.at(memory_t::device), nr);
691  break;
692  }
693  }
694  }
695  }
696  if (hphi__ != nullptr) {
697  /* multiply be effective potential, which itself was multiplied by the step function */
698  mul_by_veff(spfftk__, spfft_buf, veff_vec_, 0);
699  /* phi(r) * Theta(r) * V(r) -> hphi(G) */
700  spfftk__.forward(spfft_mem,
701  reinterpret_cast<double*>(hphi__->pw_coeffs(0).extra().at(memory_t::host, 0, j)),
702  SPFFT_FULL_SCALING);
703  }
704  t1.stop();
705 
706  if (hphi__ != nullptr) {
707  utils::timer t2("sirius::Local_operator::apply_h_o|kin");
708  /* add kinetic energy */
709  for (int x : {0, 1, 2}) {
710  #pragma omp parallel for schedule(static)
711  for (int igloc = 0; igloc < gkvec_p__.gvec_count_fft(); igloc++) {
712  /* global index of G-vector */
713  int ig = gkvec_p__.idx_gvec(igloc);
714  /* \hat P phi = phi(G+k) * (G+k), \hat P is momentum operator */
715  buf_pw[igloc] = phi__.pw_coeffs(0).extra()(igloc, j) *
716  gkvec_p__.gvec().gkvec_cart<index_domain_t::global>(ig)[x];
717  }
718  /* transform Cartesian component of wave-function gradient to real space */
719  spfftk__.backward(reinterpret_cast<double const*>(&buf_pw[0]), spfft_mem);
720  /* multiply by real-space function */
721  switch (ctx_.valence_relativity()) {
722  case relativity_t::iora:
723  case relativity_t::zora: {
724  /* multiply be inverse relative mass */
725  mul_by_veff(spfftk__, spfft_buf, veff_vec_, 5);
726  break;
727  }
728  case relativity_t::none: {
729  /* multiply be step function */
730  mul_by_veff(spfftk__, spfft_buf, veff_vec_, 4);
731  break;
732  }
733  default: {
734  break;
735  }
736  }
737  /* transform back to PW domain */
738  spfftk__.forward(spfft_mem, reinterpret_cast<double*>(&buf_pw[0]), SPFFT_FULL_SCALING);
739  #pragma omp parallel for schedule(static)
740  for (int igloc = 0; igloc < gkvec_p__.gvec_count_fft(); igloc++) {
741  int ig = gkvec_p__.idx_gvec(igloc);
742  hphi__->pw_coeffs(0).extra()(igloc, j) +=
743  0.5 * buf_pw[igloc] * gkvec_p__.gvec().gkvec_cart<index_domain_t::global>(ig)[x];
744  }
745  }
746  }
747  }
748 
749  if (hphi__ != nullptr) {
750  hphi__->pw_coeffs(0).remap_backward(n__, N__);
751  }
752  if (ophi__ != nullptr) {
753  ophi__->pw_coeffs(0).remap_backward(n__, N__);
754  }
755 
756  if (ctx_.processing_unit() == device_t::GPU) {
757  if (hphi__ != nullptr) {
758  hphi__->pw_coeffs(0).copy_to(memory_t::device, N__, n__);
759  }
760  if (ophi__ != nullptr) {
761  ophi__->pw_coeffs(0).copy_to(memory_t::device, N__, n__);
762  }
763  }
764  // if (ctx_->control().print_checksum_) {
765  // auto cs1 = hphi__.checksum_pw(N__, n__, ctx_->processing_unit());
766  // auto cs2 = ophi__.checksum_pw(N__, n__, ctx_->processing_unit());
767  // if (phi__.comm().rank() == 0) {
768  // DUMP("checksum(hphi_pw): %18.10f %18.10f", cs1.real(), cs1.imag());
769  // DUMP("checksum(ophi_pw): %18.10f %18.10f", cs2.real(), cs2.imag());
770  // }
771  //}
