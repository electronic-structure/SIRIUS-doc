91inline relativity_t get_relativity_t(std::string name__)
92{
93 std::transform(name__.begin(), name__.end(), name__.begin(), ::tolower);
94 std::map<std::string, relativity_t> const m = {
95 {"none", relativity_t::none},
96 {"koelling_harmon", relativity_t::koelling_harmon},
97 {"zora", relativity_t::zora},
98 {"iora", relativity_t::iora},
99 {"dirac", relativity_t::dirac}
100 };
101
102 if (m.count(name__) == 0) {
103 std::stringstream s;
104 s << "get_relativity_t(): wrong label of the relativity_t enumerator: " << name__;
105 throw std::runtime_error(s.str());
106 }
107 return m.at(name__);
108}
