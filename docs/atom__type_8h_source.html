73  void init(std::vector<radial_solution_descriptor_set> const& aw_descriptors,
74  std::vector<local_orbital_descriptor> const& lo_descriptors)
75  {
76  lmax_aw_ = static_cast<int>(aw_descriptors.size()) - 1;
77  lmax_lo_ = -1;
78  for (size_t idxlo = 0; idxlo < lo_descriptors.size(); idxlo++) {
79  int l = lo_descriptors[idxlo].l;
80  lmax_lo_ = std::max(lmax_lo_, l);
81  }
82 
83  lmax_ = std::max(lmax_aw_, lmax_lo_);
84 
85  num_rf_ = std::vector<int>(lmax_ + 1, 0);
86  num_lo_ = std::vector<int>(lmax_ + 1, 0);
87 
88  max_num_rf_ = 0;
89 
90  radial_function_index_descriptors_.clear();
91 
92  for (int l = 0; l <= lmax_aw_; l++) {
93  assert(aw_descriptors[l].size() <= 3);
94 
95  for (size_t order = 0; order < aw_descriptors[l].size(); order++) {
96  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l]));
97  num_rf_[l]++;
98  }
99  }
100 
101  for (int idxlo = 0; idxlo < static_cast<int>(lo_descriptors.size()); idxlo++) {
102  int l = lo_descriptors[idxlo].l;
103  radial_function_index_descriptors_.push_back(radial_function_index_descriptor(l, num_rf_[l], idxlo));
104  num_rf_[l]++;
105  num_lo_[l]++;
106  }
107 
108  for (int l = 0; l <= lmax_; l++) {
109  max_num_rf_ = std::max(max_num_rf_, num_rf_[l]);
110  }
111 
112  index_by_l_order_ = mdarray<int, 2>(lmax_ + 1, max_num_rf_);
113 
114  if (lo_descriptors.size()) {
115  index_by_idxlo_ = mdarray<int, 1>(lo_descriptors.size());
116  }
117 
118  for (int i = 0; i < (int)radial_function_index_descriptors_.size(); i++) {
119  int l = radial_function_index_descriptors_[i].l;
120  int order = radial_function_index_descriptors_[i].order;
121  int idxlo = radial_function_index_descriptors_[i].idxlo;
122  index_by_l_order_(l, order) = i;
123  if (idxlo >= 0) index_by_idxlo_(idxlo) = i;
124  }
125  }

205  index_by_idxrf_ = mdarray<int, 1>(indexr.size());
206 
207  for (int idxrf = 0; idxrf < indexr.size(); idxrf++)
208  {
209  int l = indexr[idxrf].l;
210  int order = indexr[idxrf].order;
211  int idxlo = indexr[idxrf].idxlo;

214

217  }

219  index_by_lm_order_ = mdarray<int, 2>(Utils::lmmax(indexr.lmax()), indexr.max_num_rf());
220 
221  for (int i = 0; i < (int)basis_function_index_descriptors_.size(); i++)
222  {
223  int lm = basis_function_index_descriptors_[i].lm;
224  int order = basis_function_index_descriptors_[i].order;
225  index_by_lm_order_(lm, order) = i;
226 
227  // get number of aw basis functions
228  if (basis_function_index_descriptors_[i].idxlo < 0) size_aw_ = i + 1;

230 
231  size_lo_ = (int)basis_function_index_descriptors_.size() - size_aw_;

233  assert(size_aw_ >= 0);
234  assert(size_lo_ >= 0);
235  }

469  /// Add augmented-wave descriptor.
470  inline void add_aw_descriptor(int n, int l, double enu, int dme, int auto_enu)
471  {
472  if ((int)aw_descriptors_.size() < (l + 1)) {
473  aw_descriptors_.resize(l + 1, radial_solution_descriptor_set());
474  }
475 
476  radial_solution_descriptor rsd;
477 
478  rsd.n = n;
479  if (n == -1) {
480  /* default principal quantum number value for any l */
481  rsd.n = l + 1;
482  for (int ist = 0; ist < num_atomic_levels(); ist++) {
483  /* take next level after the core */
484  if (atomic_level(ist).core && atomic_level(ist).l == l) {
485  rsd.n = atomic_level(ist).n + 1;
486  }
487  }
488  }
489 
490  rsd.l = l;
491  rsd.dme = dme;
492  rsd.enu = enu;
493  rsd.auto_enu = auto_enu;
494  aw_descriptors_[l].push_back(rsd);
495  }

497  /// Add local orbital descriptor

499  {
500  if ((int)lo_descriptors_.size() == ilo) {

508  << "idxlo: " << ilo << std::endl
509  << "n: " << l << std::endl
510  << "l: " << n << std::endl
511  << "expected l: " << lo_descriptors_[ilo].l << std::endl;
512  TERMINATE(s);
513  }
514  }
515 
516  radial_solution_descriptor rsd;
517 
518  rsd.n = n;
519  if (n == -1) {
520  /* default value for any l */

523  if (atomic_level(ist).core && atomic_level(ist).l == l) {

879 inline void Atom_type::init(int offset_lo__)
880 {
881  PROFILE("sirius::Atom_type::init");
882 
883  /* check if the class instance was already initialized */
884  if (initialized_) {
885  TERMINATE("can't initialize twice");
886  }
887 
888  offset_lo_ = offset_lo__;
889 
890  /* read data from file if it exists */
891  if (file_name_.length() > 0) {
892  if (!Utils::file_exists(file_name_)) {
893  std::stringstream s;
894  s << "file " + file_name_ + " doesn't exist";
895  TERMINATE(s);
896  } else {
897  read_input(file_name_);
898  }
899  }
900 
901  /* check the nuclear charge */
902  if (zn_ == 0) {
903  TERMINATE("zero atom charge");
904  }
905 
906  /* add valence levels to the list of core levels */
907  if (parameters_.full_potential()) {
908  atomic_level_descriptor level;
909  for (int ist = 0; ist < 28; ist++) {
910  bool found = false;
911  level.n = atomic_conf[zn_ - 1][ist][0];
912  level.l = atomic_conf[zn_ - 1][ist][1];
913  level.k = atomic_conf[zn_ - 1][ist][2];
914  level.occupancy = double(atomic_conf[zn_ - 1][ist][3]);
915  level.core = false;
916 
917  if (level.n != -1) {
918  for (size_t jst = 0; jst < atomic_levels_.size(); jst++) {
919  if (atomic_levels_[jst].n == level.n &&
920  atomic_levels_[jst].l == level.l &&
921  atomic_levels_[jst].k == level.k) {
922  found = true;
923  }
924  }
925  if (!found) {
926  atomic_levels_.push_back(level);
927  }
928  }
929  }
930  /* get the number of core electrons */
931  for (auto& e: atomic_levels_) {
932  if (e.core) {
933  num_core_electrons_ += e.occupancy;
934  }
935  }
936  }
937 
938  /* set default radial grid if it was not done by user */
939  if (radial_grid_.num_points() == 0) {
940  set_radial_grid();
941  }
942 
943  if (parameters_.full_potential()) {
944  /* initialize aw descriptors if they were not set manually */
945  if (aw_descriptors_.size() == 0) {
946  init_aw_descriptors(parameters_.lmax_apw());
947  }
948 
949  if (static_cast<int>(aw_descriptors_.size()) != (parameters_.lmax_apw() + 1)) {
950  TERMINATE("wrong size of augmented wave descriptors");
951  }
952 
953  max_aw_order_ = 0;
954  for (int l = 0; l <= parameters_.lmax_apw(); l++) {
955  max_aw_order_ = std::max(max_aw_order_, (int)aw_descriptors_[l].size());
956  }
957 
958  if (max_aw_order_ > 3) {
959  TERMINATE("maximum aw order > 3");
960  }
961  }
962 
963  if (!parameters_.full_potential()) {
964  local_orbital_descriptor lod;
965  for (int i = 0; i < pp_desc_.num_beta_radial_functions; i++) {
966  /* think of |beta> functions as of local orbitals */
967  lod.l = pp_desc_.beta_l[i];
968  lo_descriptors_.push_back(lod);
969  }
970  }
971 
972  /* initialize index of radial functions */
973  indexr_.init(aw_descriptors_, lo_descriptors_);
974 
975  /* initialize index of muffin-tin basis functions */
976  indexb_.init(indexr_);
977 
978  if (!parameters_.full_potential()) {
979  /* number of radial beta-functions */
980  int nbrf = mt_radial_basis_size();
981  /* maximum l of beta-projectors */
982  int lmax_beta = indexr().lmax();
983  /* interpolate beta radial functions */
984  beta_rf_ = mdarray<Spline<double>, 1>(mt_radial_basis_size());
985  for (int idxrf = 0; idxrf < nbrf; idxrf++) {
986  beta_rf_[idxrf] = Spline<double>(radial_grid());
987  int nr = pp_desc().num_beta_radial_points[idxrf];
988  for (int ir = 0; ir < nr; ir++) {
989  beta_rf_[idxrf][ir] = pp_desc().beta_radial_functions(ir, idxrf);
990  }
991  beta_rf_[idxrf].interpolate();
992  }
993  /* interpolate Q-operator radial functions */
994  if (pp_desc().augment) {
995  q_rf_ = mdarray<Spline<double>, 2>(nbrf * (nbrf + 1) / 2, 2 * lmax_beta + 1);
996  #pragma omp parallel for
997  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
998  for (int l = 0; l <= 2 * lmax_beta; l++) {
999  q_rf_(idx, l) = Spline<double>(radial_grid());
1000  for (int ir = 0; ir < num_mt_points(); ir++) {
1001  q_rf_(idx, l)[ir] = pp_desc().q_radial_functions_l(ir, idx, l);
1002  }
1003  q_rf_(idx, l).interpolate();
1004  }
1005  }
1006  }
1007  }
1008 
1009  /* get number of valence electrons */
1010  num_valence_electrons_ = zn_ - num_core_electrons_;
1011 
1012  int lmmax_pot = Utils::lmmax(parameters_.lmax_pot());
1013 
1014  if (parameters_.full_potential()) {
1015  auto l_by_lm = Utils::l_by_lm(parameters_.lmax_pot());
1016 
1017  /* index the non-zero radial integrals */
1018  std::vector<std::pair<int, int>> non_zero_elements;
1019 
1020  for (int lm = 0; lm < lmmax_pot; lm++) {
1021  int l = l_by_lm[lm];
1022 
1023  for (int i2 = 0; i2 < indexr().size(); i2++) {
1024  int l2 = indexr(i2).l;
1025  for (int i1 = 0; i1 <= i2; i1++) {
1026  int l1 = indexr(i1).l;
1027  if ((l + l1 + l2) % 2 == 0) {
1028  if (lm) {
1029  non_zero_elements.push_back(std::pair<int, int>(i2, lm + lmmax_pot * i1));
1030  }
1031  for (int j = 0; j < parameters_.num_mag_dims(); j++) {
1032  int offs = (j + 1) * lmmax_pot * indexr().size();
1033  non_zero_elements.push_back(std::pair<int, int>(i2, lm + lmmax_pot * i1 + offs));
1034  }
1035  }
1036  }
1037  }
1038  }
1039  idx_radial_integrals_ = mdarray<int, 2>(2, non_zero_elements.size());
1040  for (size_t j = 0; j < non_zero_elements.size(); j++) {
1041  idx_radial_integrals_(0, j) = non_zero_elements[j].first;
1042  idx_radial_integrals_(1, j) = non_zero_elements[j].second;
1043  }
1044  }
1045 
1046  if (parameters_.processing_unit() == GPU && parameters_.full_potential()) {
1047  #ifdef __GPU
1048  idx_radial_integrals_.allocate(memory_t::device);
1049  idx_radial_integrals_.copy_to_device();
1050  rf_coef_ = mdarray<double, 3>(num_mt_points_, 4, indexr().size(), memory_t::host_pinned | memory_t::device);
1051  vrf_coef_ = mdarray<double, 3>(num_mt_points_, 4, lmmax_pot * indexr().size() * (parameters_.num_mag_dims() + 1), memory_t::host_pinned | memory_t::device);
1052  #else
1053  TERMINATE_NO_GPU
1054  #endif
1055  }
1056 
1057  initialized_ = true;
1058 }
